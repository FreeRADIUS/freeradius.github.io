31 static struct msgb *mtp_create_sltm(struct mtp_link *link)
32 {
33  const uint8_t test_ptrn[14] = { 'G', 'S', 'M', 'M', 'M', 'S', };
34  struct mtp_level_3_hdr *hdr;
35  struct mtp_level_3_mng *mng;
36  struct msgb *msg = mtp_msg_alloc(link->set);
37  uint8_t *data;
38  if (!msg)
39  return NULL;
40 
41  hdr = (struct mtp_level_3_hdr *) msg->l2h;
42  hdr->ser_ind = MTP_SI_MNT_REG_MSG;
43  hdr->addr = MTP_ADDR(link->nr % 16, link->set->dpc, link->set->opc);
44 
45  mng = (struct mtp_level_3_mng *) msgb_put(msg, sizeof(*mng));
46  mng->cmn.h0 = MTP_TST_MSG_GRP;
47  mng->cmn.h1 = MTP_TST_MSG_SLTM;
48  mng->length = ARRAY_SIZE(test_ptrn);
49 
50  data = msgb_put(msg, ARRAY_SIZE(test_ptrn));
51  memcpy(data, test_ptrn, ARRAY_SIZE(test_ptrn));
52 
53  /* remember the last tst ptrn... once we have some */
54  memcpy(link->test_ptrn, test_ptrn, ARRAY_SIZE(test_ptrn));
55 
56  return msg;
57 }

59 static void mtp_send_sltm(struct mtp_link *link)
60 {
61  struct msgb *msg;
62 
63  link->sltm_pending = 1;
64  msg = mtp_create_sltm(link);
65  if (!msg) {
66  LOGP(DINP, LOGL_ERROR, "Failed to allocate SLTM on link %d/%s of %d/%s.\n",
67  link->nr, link->name, link->set->nr, link->set->name);
68  return;
69  }
70 
71  mtp_link_submit(link, msg);
72 }

74 static void mtp_sltm_t1_timeout(void *_link)
75 {
76  struct mtp_link *link = (struct mtp_link *) _link;
77 
78  rate_ctr_inc(&link->ctrg->ctr[MTP_LNK_SLTM_TOUT]);
79 
80  if (link->slta_misses == 0) {
81  LOGP(DINP, LOGL_ERROR,
82  "No SLTM response on link %d/%s of %d/%s.\n",
83  link->nr, link->name, link->set->nr, link->set->name);
84  ++link->slta_misses;
85  mtp_send_sltm(link);
86  osmo_timer_schedule(&link->t1_timer, MTP_T1);
87  } else {
88  LOGP(DINP, LOGL_ERROR,
89  "Two missing SLTAs on link %d/%s of %d/%s.\n",
90  link->nr, link->name, link->set->nr, link->set->name);
91  osmo_timer_del(&link->t2_timer);
92  mtp_link_failure(link);
93  }
94 }

96 static void mtp_sltm_t2_timeout(void *_link)
97 {
98  struct mtp_link *link = (struct mtp_link *) _link;
99 
100  if (!link->set->running) {
101  LOGP(DINP, LOGL_INFO,
102  "The linkset is not active. Stopping link test on %d/%s of %d/%s.\n",
103  link->nr, link->name, link->set->nr, link->set->name);
104  return;
105  }
106 
107  link->slta_misses = 0;
108  mtp_send_sltm(link);
109 
110  osmo_timer_schedule(&link->t1_timer, MTP_T1);
111 
112  if (link->set->sltm_once && link->was_up)
113  LOGP(DINP, LOGL_INFO, "Not sending SLTM again on link %d/%s of %d/%s.\n",
114  link->nr, link->name, link->set->nr, link->set->name);
115  else
116  osmo_timer_schedule(&link->t2_timer, MTP_T2);
117 }

147 int mtp_link_slta(struct mtp_link *link, uint16_t l3_len,
148  struct mtp_level_3_mng *mng)
149 {
150  if (mng->length != 14) {
151  LOGP(DINP, LOGL_ERROR, "Wrongly sized SLTA: %u\n", mng->length);
152  return -1;
153  }
154 
155  if (l3_len != 16) {
156  LOGP(DINP, LOGL_ERROR, "Wrongly sized SLTA: %u\n", mng->length);
157  return -1;
158  }
159 
160  if (memcmp(mng->data, link->test_ptrn, sizeof(link->test_ptrn)) != 0) {
161  LOGP(DINP, LOGL_ERROR, "Wrong test pattern SLTA\n");
162  return -1;
163  }
164 
165  /* we had a matching slta */
166  osmo_timer_del(&link->t1_timer);
167  link->sltm_pending = 0;
168  link->was_up = 1;
169 
170  return 0;
171 }

216 struct mtp_link *mtp_link_alloc(struct mtp_link_set *set)
217 {
218  struct mtp_link *link;
219 
220  link = talloc_zero(set, struct mtp_link);
221  if (!link) {
222  LOGP(DINP, LOGL_ERROR, "Failed to allocate the link.\n");
223  return NULL;
224  }
225 
226  link->nr = set->nr_links++;
227  link->ctrg = rate_ctr_group_alloc(link,
228  mtp_link_rate_ctr_desc(), link->nr);
229  if (!link->ctrg) {
230  LOGP(DINP, LOGL_ERROR, "Failed to allocate rate_ctr.\n");
231  talloc_free(link);
232  return NULL;
233  }
234 
235  /* make sure a unconfigured link does not crash */
236  link->write = dummy_arg2;
237  link->shutdown = dummy_arg1;
238  link->reset = dummy_arg1;
239  link->clear_queue = dummy_arg1;
240 
241  link->pcap_fd = -1;
242 
243  link->t1_timer.data = link;
244  link->t1_timer.cb = mtp_sltm_t1_timeout;
245  link->t2_timer.data = link;
246  link->t2_timer.cb = mtp_sltm_t2_timeout;
247 
248  link->set = set;
249 
250  llist_add_tail(&link->entry, &set->links);
251  mtp_link_set_init_slc(set);
252 
253  return link;
254 }
