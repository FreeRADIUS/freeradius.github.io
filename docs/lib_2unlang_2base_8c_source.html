43 /** Register an operation with the interpreter
44  *
45  * The main purpose of this registration API is to avoid intermixing the xlat,
46  * condition, map APIs with the interpreter, i.e. the callbacks needed for that
47  * functionality can be in their own source files, and we don't need to include
48  * supporting types and function declarations in the interpreter.
49  *
50  * Later, this could potentially be used to register custom operations for modules.
51  *
52  * The reason why there's a function instead of accessing the unlang_op array
53  * directly, is because 'type' really needs to go away, as needing to add ops to
54  * the unlang_type_t enum breaks the pluggable module model. If there's no
55  * explicit/consistent type values we need to enumerate the operations ourselves.
56  *
57  * @param[in] type Operation identifier. Used to map compiled unlang code
58  * to operations.
59  * @param[in] op unlang_op to register.
60  */
61 void unlang_register(int type, unlang_op_t *op)
62 {
63  fr_assert(type < UNLANG_TYPE_MAX); /* Unlang max isn't a valid type */
64 
65  memcpy(&unlang_ops[type], op, sizeof(unlang_ops[type]));
66 }

68 /** Initialize the unlang compiler / interpreter.
69  *
70  * For now, just register the magic xlat function.
71  */
72 int unlang_init(void)
73 {
74  /*
75  * Explicitly initialise the xlat tree, and perform dictionary lookups.
76  */
77  if (xlat_init() < 0) return -1;
78 
79  unlang_interpret_init();
80  /* Register operations for the default keywords */
81  unlang_condition_init();
82  unlang_foreach_init();
83  unlang_function_init();
84  unlang_group_init();
85  unlang_load_balance_init();
86  unlang_map_init();
87  unlang_module_init();
88  unlang_parallel_init();
89  unlang_return_init();
90  if (unlang_subrequest_op_init() < 0) return -1;
91  unlang_switch_init();
92  unlang_call_init();
93  unlang_caller_init();
94  unlang_tmpl_init_shallow();
95 
96  return 0;
97 }
