39 /** Convert multiple group names into a DNs
40  *
41  * Given an array of group names, builds a filter matching all names, then retrieves all group objects
42  * and stores the DN associated with each group object.
43  *
44  * @param[out] p_result The result of trying to resolve a group name to a dn.
45  * @param[in] inst rlm_ldap configuration.
46  * @param[in] request Current request.
47  * @param[in,out] pconn to use. May change as this function calls functions which auto re-connect.
48  * @param[in] names to convert to DNs (NULL terminated).
49  * @param[out] out Where to write the DNs. DNs must be freed with
50  * ldap_memfree(). Will be NULL terminated.
51  * @param[in] outlen Size of out.
52  * @return One of the RLM_MODULE_* values.
53  */
54 static unlang_action_t rlm_ldap_group_name2dn(rlm_rcode_t *p_result, rlm_ldap_t const *inst, request_t *request,
55  fr_ldap_connection_t **pconn,
56  char **names, char **out, size_t outlen)
57 {
58  rlm_rcode_t rcode = RLM_MODULE_OK;
59  fr_ldap_rcode_t status;
60  int ldap_errno;
61 
62  unsigned int name_cnt = 0;
63  unsigned int entry_cnt;
64  char const *attrs[] = { NULL };
65 
66  LDAPMessage *result = NULL, *entry;
67 
68  char **name = names;
69  char **dn = out;
70  char const *base_dn = NULL;
71  char base_dn_buff[LDAP_MAX_DN_STR_LEN];
72  char buffer[LDAP_MAX_GROUP_NAME_LEN + 1];
73 
74  char *filter;
75 
76  *dn = NULL;
77 
78  if (!*names) RETURN_MODULE_OK;
79 
80  if (!inst->groupobj_name_attr) {
81  REDEBUG("Told to convert group names to DNs but missing 'group.name_attribute' directive");
82 
83  RETURN_MODULE_INVALID;
84  }
85 
86  RDEBUG2("Converting group name(s) to group DN(s)");
87 
88  /*
89  * It'll probably only save a few ms in network latency, but it means we can send a query
90  * for the entire group list at once.
91  */
92  filter = talloc_typed_asprintf(request, "%s%s%s",
93  inst->groupobj_filter ? "(&" : "",
94  inst->groupobj_filter ? inst->groupobj_filter : "",
95  names[0] && names[1] ? "(|" : "");
96  while (*name) {
97  fr_ldap_escape_func(request, buffer, sizeof(buffer), *name++, NULL);
98  filter = talloc_asprintf_append_buffer(filter, "(%s=%s)", inst->groupobj_name_attr, buffer);
99 
100  name_cnt++;
101  }
102  filter = talloc_asprintf_append_buffer(filter, "%s%s",
103  inst->groupobj_filter ? ")" : "",
104  names[0] && names[1] ? ")" : "");
105 
106  if (tmpl_expand(&base_dn, base_dn_buff, sizeof(base_dn_buff), request,
107  inst->groupobj_base_dn, fr_ldap_escape_func, NULL) < 0) {
108  REDEBUG("Failed creating base_dn");
109 
110  RETURN_MODULE_INVALID;
111  }
112 
113  status = fr_ldap_search(&result, request, pconn, base_dn, inst->groupobj_scope,
114  filter, attrs, NULL, NULL);
115  switch (status) {
116  case LDAP_PROC_SUCCESS:
117  break;
118 
119  case LDAP_PROC_NO_RESULT:
120  RDEBUG2("Tried to resolve group name(s) to DNs but got no results");
121  goto finish;
122 
123  default:
124  rcode = RLM_MODULE_FAIL;
125  goto finish;
126  }
127 
128  entry_cnt = ldap_count_entries((*pconn)->handle, result);
129  if (entry_cnt > name_cnt) {
130  REDEBUG("Number of DNs exceeds number of names, group and/or dn should be more restrictive");
131  rcode = RLM_MODULE_INVALID;
132 
133  goto finish;
134  }
135 
136  if (entry_cnt > (outlen - 1)) {
137  REDEBUG("Number of DNs exceeds limit (%zu)", outlen - 1);
138  rcode = RLM_MODULE_INVALID;
139 
140  goto finish;
141  }
142 
143  if (entry_cnt < name_cnt) {
144  RWDEBUG("Got partial mapping of group names (%i) to DNs (%i), membership information may be incomplete",
145  name_cnt, entry_cnt);
146  }
147 
148  entry = ldap_first_entry((*pconn)->handle, result);
149  if (!entry) {
150  ldap_get_option((*pconn)->handle, LDAP_OPT_RESULT_CODE, &ldap_errno);
151  REDEBUG("Failed retrieving entry: %s", ldap_err2string(ldap_errno));
152 
153  rcode = RLM_MODULE_FAIL;
154  goto finish;
155  }
156 
157  do {
158  *dn = ldap_get_dn((*pconn)->handle, entry);
159  if (!*dn) {
160  ldap_get_option((*pconn)->handle, LDAP_OPT_RESULT_CODE, &ldap_errno);
161  REDEBUG("Retrieving object DN from entry failed: %s", ldap_err2string(ldap_errno));
162 
163  rcode = RLM_MODULE_FAIL;
164  goto finish;
165  }
166  fr_ldap_util_normalise_dn(*dn, *dn);
167 
168  RDEBUG2("Got group DN \"%s\"", *dn);
169  dn++;
170  } while((entry = ldap_next_entry((*pconn)->handle, entry)));
171 
172  *dn = NULL;
173 
174 finish:
175  talloc_free(filter);
176  if (result) ldap_msgfree(result);
177 
178  /*
179  * Be nice and cleanup the output array if we error out.
180  */
181  if (rcode != RLM_MODULE_OK) {
182  dn = out;
183  while(*dn) ldap_memfree(*dn++);
184  *dn = NULL;
185  }
186 
187  RETURN_MODULE_RCODE(rcode);
188 }

190 /** Convert a single group name into a DN
191  *
192  * Unlike the inverse conversion of a name to a DN, most LDAP directories don't allow filtering by DN,
193  * so we need to search for each DN individually.
194  *
195  * @param[out] p_result The result of trying to resolve a dn to a group name.
196  * @param[in] inst rlm_ldap configuration.
197  * @param[in] request Current request.
198  * @param[in,out] pconn to use. May change as this function calls functions which auto re-connect.
199  * @param[in] dn to resolve.
200  * @param[out] out Where to write group name (must be freed with talloc_free).
201  * @return One of the RLM_MODULE_* values.
202  */
203 static unlang_action_t rlm_ldap_group_dn2name(rlm_rcode_t *p_result, rlm_ldap_t const *inst, request_t *request,
204  fr_ldap_connection_t **pconn, char const *dn, char **out)
205 {
206  rlm_rcode_t rcode = RLM_MODULE_OK;
207  fr_ldap_rcode_t status;
208  int ldap_errno;
209 
210  struct berval **values = NULL;
211  char const *attrs[] = { inst->groupobj_name_attr, NULL };
212  LDAPMessage *result = NULL, *entry;
213 
214  *out = NULL;
215 
216  if (!inst->groupobj_name_attr) {
217  REDEBUG("Told to resolve group DN to name but missing 'group.name_attribute' directive");
218 
219  RETURN_MODULE_INVALID;
220  }
221 
222  RDEBUG2("Resolving group DN \"%s\" to group name", dn);
223 
224  status = fr_ldap_search(&result, request, pconn, dn, LDAP_SCOPE_BASE, NULL, attrs, NULL, NULL);
225  switch (status) {
226  case LDAP_PROC_SUCCESS:
227  break;
228 
229  case LDAP_PROC_NO_RESULT:
230  REDEBUG("Group DN \"%s\" did not resolve to an object", dn);
231  RETURN_MODULE_RCODE(inst->allow_dangling_group_refs ? RLM_MODULE_NOOP : RLM_MODULE_INVALID);
232 
233  default:
234  RETURN_MODULE_FAIL;
235  }
236 
237  entry = ldap_first_entry((*pconn)->handle, result);
238  if (!entry) {
239  ldap_get_option((*pconn)->handle, LDAP_OPT_RESULT_CODE, &ldap_errno);
240  REDEBUG("Failed retrieving entry: %s", ldap_err2string(ldap_errno));
241 
242  rcode = RLM_MODULE_INVALID;
243  goto finish;
244  }
245 
246  values = ldap_get_values_len((*pconn)->handle, entry, inst->groupobj_name_attr);
247  if (!values) {
248  REDEBUG("No %s attributes found in object", inst->groupobj_name_attr);
249 
250  rcode = RLM_MODULE_INVALID;
251 
252  goto finish;
253  }
254 
255  *out = fr_ldap_berval_to_string(request, values[0]);
256  RDEBUG2("Group DN \"%s\" resolves to name \"%s\"", dn, *out);
257 
258 finish:
259  if (result) ldap_msgfree(result);
260  if (values) ldap_value_free_len(values);
261 
262  RETURN_MODULE_RCODE(rcode);
263 }

265 /** Convert group membership information into attributes
266  *
267  * @param[out] p_result The result of trying to resolve a dn to a group name.
268  * @param[in] inst rlm_ldap configuration.
269  * @param[in] request Current request.
270  * @param[in,out] pconn to use. May change as this function calls functions which auto re-connect.
271  * @param[in] entry retrieved by rlm_ldap_find_user or fr_ldap_search.
272  * @param[in] attr membership attribute to look for in the entry.
273  * @return One of the RLM_MODULE_* values.
274  */
275 unlang_action_t rlm_ldap_cacheable_userobj(rlm_rcode_t *p_result, rlm_ldap_t const *inst,
276  request_t *request, fr_ldap_connection_t **pconn,
277  LDAPMessage *entry, char const *attr)
278 {
279  rlm_rcode_t rcode = RLM_MODULE_OK;
280 
281  struct berval **values;
282 
283  char *group_name[LDAP_MAX_CACHEABLE + 1];
284  char **name_p = group_name;
285 
286  char *group_dn[LDAP_MAX_CACHEABLE + 1];
287  char **dn_p;
288 
289  char *name;
290 
291  fr_pair_t *vp, **list, *groups = NULL;
292  TALLOC_CTX *list_ctx, *value_ctx;
293  fr_cursor_t list_cursor, groups_cursor;
294 
295  int is_dn, i, count;
296 
297  fr_assert(entry);
298  fr_assert(attr);
299 
300  /*
301  * Parse the membership information we got in the initial user query.
302  */
303  values = ldap_get_values_len((*pconn)->handle, entry, attr);
304  if (!values) {
305  RDEBUG2("No cacheable group memberships found in user object");
306 
307  RETURN_MODULE_OK;
308  }
309  count = ldap_count_values_len(values);
310 
311  list = radius_list(request, PAIR_LIST_CONTROL);
312  list_ctx = radius_list_ctx(request, PAIR_LIST_CONTROL);
313  fr_assert(list != NULL);
314  fr_assert(list_ctx != NULL);
315 
316  /*
317  * Simplifies freeing temporary values
318  */
319  value_ctx = talloc_new(request);
320 
321  /*
322  * Temporary list to hold new group VPs, will be merged
323  * once all group info has been gathered/resolved
324  * successfully.
325  */
326  fr_cursor_init(&groups_cursor, &groups);
327 
328  for (i = 0; (i < LDAP_MAX_CACHEABLE) && (i < count); i++) {
329  is_dn = fr_ldap_util_is_dn(values[i]->bv_val, values[i]->bv_len);
330 
331  if (inst->cacheable_group_dn) {
332  /*
333  * The easy case, we're caching DNs and we got a DN.
334  */
335  if (is_dn) {
336  MEM(vp = fr_pair_afrom_da(list_ctx, inst->cache_da));
337  fr_pair_value_bstrndup(vp, values[i]->bv_val, values[i]->bv_len, true);
338  fr_cursor_append(&groups_cursor, vp);
339  /*
340  * We were told to cache DNs but we got a name, we now need to resolve
341  * this to a DN. Store all the group names in an array so we can do one query.
342  */
343  } else {
344  *name_p++ = fr_ldap_berval_to_string(value_ctx, values[i]);
345  }
346  }
347 
348  if (inst->cacheable_group_name) {
349  /*
350  * The easy case, we're caching names and we got a name.
351  */
352  if (!is_dn) {
353  MEM(vp = fr_pair_afrom_da(list_ctx, inst->cache_da));
354  fr_pair_value_bstrndup(vp, values[i]->bv_val, values[i]->bv_len, true);
355  fr_cursor_append(&groups_cursor, vp);
356  /*
357  * We were told to cache names but we got a DN, we now need to resolve
358  * this to a name.
359  * Only Active Directory supports filtering on DN, so we have to search
360  * for each individual group.
361  */
362  } else {
363  char *dn;
364 
365  dn = fr_ldap_berval_to_string(value_ctx, values[i]);
366  rlm_ldap_group_dn2name(&rcode, inst, request, pconn, dn, &name);
367  talloc_free(dn);
368 
369  if (rcode == RLM_MODULE_NOOP) continue;
370 
371  if (rcode != RLM_MODULE_OK) {
372  ldap_value_free_len(values);
373  talloc_free(value_ctx);
374  fr_pair_list_free(&groups);
375 
376  RETURN_MODULE_RCODE(rcode);
377  }
378 
379  MEM(vp = fr_pair_afrom_da(list_ctx, inst->cache_da));
380  fr_pair_value_bstrdup_buffer(vp, name, true);
381  fr_cursor_append(&groups_cursor, vp);
382  talloc_free(name);
383  }
384  }
385  }
386  *name_p = NULL;
387 
388  rlm_ldap_group_name2dn(&rcode, inst, request, pconn, group_name, group_dn, sizeof(group_dn));
389 
390  ldap_value_free_len(values);
391  talloc_free(value_ctx);
392 
393  if (rcode != RLM_MODULE_OK) RETURN_MODULE_RCODE(rcode);
394 
395  fr_cursor_init(&list_cursor, list);
396 
397  RDEBUG2("Adding cacheable user object memberships");
398  RINDENT();
399  if (RDEBUG_ENABLED) {
400  for (vp = fr_cursor_head(&groups_cursor);
401  vp;
402  vp = fr_cursor_next(&groups_cursor)) {
403  RDEBUG2("&control.%s += \"%pV\"", inst->cache_da->name, &vp->data);
404  }
405  }
406 
407  fr_cursor_head(&groups_cursor);
408  fr_cursor_merge(&list_cursor, &groups_cursor);
409 
410  for (dn_p = group_dn; *dn_p; dn_p++) {
411  MEM(vp = fr_pair_afrom_da(list_ctx, inst->cache_da));
412  fr_pair_value_strdup(vp, *dn_p);
413  fr_cursor_append(&list_cursor, vp);
414 
415  RDEBUG2("&control.%s += \"%pV\"", inst->cache_da->name, &vp->data);
416  ldap_memfree(*dn_p);
417  }
418  REXDENT();
419 
420  RETURN_MODULE_RCODE(rcode);
421 }

423 /** Convert group membership information into attributes
424  *
425  * @param[out] p_result The result of trying to resolve a dn to a group name.
426  * @param[in] inst rlm_ldap configuration.
427  * @param[in] request Current request.
428  * @param[in,out] pconn to use. May change as this function calls functions which auto re-connect.
429  * @return One of the RLM_MODULE_* values.
430  */
431 unlang_action_t rlm_ldap_cacheable_groupobj(rlm_rcode_t *p_result, rlm_ldap_t const *inst,
432  request_t *request, fr_ldap_connection_t **pconn)
433 {
434  rlm_rcode_t rcode = RLM_MODULE_OK;
435  fr_ldap_rcode_t status;
436  int ldap_errno;
437 
438  LDAPMessage *result = NULL;
439  LDAPMessage *entry;
440 
441  char const *base_dn;
442  char base_dn_buff[LDAP_MAX_DN_STR_LEN];
443 
444  char const *filters[] = { inst->groupobj_filter, inst->groupobj_membership_filter };
445  char filter[LDAP_MAX_FILTER_STR_LEN + 1];
446 
447  char const *attrs[] = { inst->groupobj_name_attr, NULL };
448 
449  fr_pair_t *vp;
450  char *dn;
451 
452  fr_assert(inst->groupobj_base_dn);
453 
454  if (!inst->groupobj_membership_filter) {
455  RDEBUG2("Skipping caching group objects as directive 'group.membership_filter' is not set");
456 
457  RETURN_MODULE_OK;
458  }
459 
460  if (fr_ldap_xlat_filter(request,
461  filters, NUM_ELEMENTS(filters),
462  filter, sizeof(filter)) < 0) {
463  RETURN_MODULE_INVALID;
464  }
465 
466  if (tmpl_expand(&base_dn, base_dn_buff, sizeof(base_dn_buff), request,
467  inst->groupobj_base_dn, fr_ldap_escape_func, NULL) < 0) {
468  REDEBUG("Failed creating base_dn");
469 
470  RETURN_MODULE_INVALID;
471  }
472 
473  status = fr_ldap_search(&result, request, pconn, base_dn,
474  inst->groupobj_scope, filter, attrs, NULL, NULL);
475  switch (status) {
476  case LDAP_PROC_SUCCESS:
477  break;
478 
479  case LDAP_PROC_NO_RESULT:
480  RDEBUG2("No cacheable group memberships found in group objects");
481  goto finish;
482 
483  default:
484  rcode = RLM_MODULE_FAIL;
485  goto finish;
486  }
487 
488  entry = ldap_first_entry((*pconn)->handle, result);
489  if (!entry) {
490  ldap_get_option((*pconn)->handle, LDAP_OPT_RESULT_CODE, &ldap_errno);
491  REDEBUG("Failed retrieving entry: %s", ldap_err2string(ldap_errno));
492 
493  goto finish;
494  }
495 
496  RDEBUG2("Adding cacheable group object memberships");
497  do {
498  if (inst->cacheable_group_dn) {
499  dn = ldap_get_dn((*pconn)->handle, entry);
500  if (!dn) {
501  ldap_get_option((*pconn)->handle, LDAP_OPT_RESULT_CODE, &ldap_errno);
502  REDEBUG("Retrieving object DN from entry failed: %s", ldap_err2string(ldap_errno));
503 
504  goto finish;
505  }
506  fr_ldap_util_normalise_dn(dn, dn);
507 
508  MEM(pair_add_control(&vp, inst->cache_da) == 0);
509  fr_pair_value_strdup(vp, dn);
510 
511  RINDENT();
512  RDEBUG2("&control.%pP", vp);
513  REXDENT();
514  ldap_memfree(dn);
515  }
516 
517  if (inst->cacheable_group_name) {
518  struct berval **values;
519 
520  values = ldap_get_values_len((*pconn)->handle, entry, inst->groupobj_name_attr);
521  if (!values) continue;
522 
523  MEM(pair_add_control(&vp, inst->cache_da) == 0);
524  fr_pair_value_bstrndup(vp, values[0]->bv_val, values[0]->bv_len, true);
525 
526  RINDENT();
527  RDEBUG2("&control.%pP", vp);
528  REXDENT();
529 
530  ldap_value_free_len(values);
531  }
532  } while ((entry = ldap_next_entry((*pconn)->handle, entry)));
533 
534 finish:
535  if (result) ldap_msgfree(result);
536 
537  RETURN_MODULE_RCODE(rcode);
538 }

540 /** Query the LDAP directory to check if a group object includes a user object as a member
541  *
542  * @param[out] p_result Result of calling the module.
543  * @param[in] inst rlm_ldap configuration.
544  * @param[in] request Current request.
545  * @param[in,out] pconn to use. May change as this function calls functions which auto re-connect.
546  * @param[in] check vp containing the group value (name or dn).
547  */
548 unlang_action_t rlm_ldap_check_groupobj_dynamic(rlm_rcode_t *p_result, rlm_ldap_t const *inst, request_t *request,
549  fr_ldap_connection_t **pconn, fr_pair_t *check)
550 {
551  fr_ldap_rcode_t status;
552 
553  char const *base_dn;
554  char base_dn_buff[LDAP_MAX_DN_STR_LEN + 1];
555  char filter[LDAP_MAX_FILTER_STR_LEN + 1];
556  int ret;
557 
558  fr_assert(inst->groupobj_base_dn);
559 
560  switch (check->op) {
561  case T_OP_CMP_EQ:
562  case T_OP_CMP_FALSE:
563  case T_OP_CMP_TRUE:
564  case T_OP_REG_EQ:
565  case T_OP_REG_NE:
566  break;
567 
568  default:
569  REDEBUG("Operator \"%s\" not allowed for LDAP group comparisons",
570  fr_table_str_by_value(fr_tokens_table, check->op, "<INVALID>"));
571  return 1;
572  }
573 
574  RDEBUG2("Checking for user in group objects");
575 
576  if (fr_ldap_util_is_dn(check->vp_strvalue, check->vp_length)) {
577  char const *filters[] = { inst->groupobj_filter, inst->groupobj_membership_filter };
578 
579  RINDENT();
580  ret = fr_ldap_xlat_filter(request,
581  filters, NUM_ELEMENTS(filters),
582  filter, sizeof(filter));
583  REXDENT();
584 
585  if (ret < 0) RETURN_MODULE_INVALID;
586 
587  base_dn = check->vp_strvalue;
588  } else {
589  char name_filter[LDAP_MAX_FILTER_STR_LEN];
590  char const *filters[] = { name_filter, inst->groupobj_filter, inst->groupobj_membership_filter };
591 
592  if (!inst->groupobj_name_attr) {
593  REDEBUG("Told to search for group by name, but missing 'group.name_attribute' "
594  "directive");
595 
596  RETURN_MODULE_INVALID;
597  }
598 
599  snprintf(name_filter, sizeof(name_filter), "(%s=%s)", inst->groupobj_name_attr, check->vp_strvalue);
600  RINDENT();
601  ret = fr_ldap_xlat_filter(request,
602  filters, NUM_ELEMENTS(filters),
603  filter, sizeof(filter));
604  REXDENT();
605  if (ret < 0) RETURN_MODULE_INVALID;
606 
607 
608  /*
609  * rlm_ldap_find_user does this, too. Oh well.
610  */
611  RINDENT();
612  ret = tmpl_expand(&base_dn, base_dn_buff, sizeof(base_dn_buff), request, inst->groupobj_base_dn,
613  fr_ldap_escape_func, NULL);
614  REXDENT();
615  if (ret < 0) {
616  REDEBUG("Failed creating base_dn");
617 
618  RETURN_MODULE_INVALID;
619  }
620  }
621 
622  RINDENT();
623  status = fr_ldap_search(NULL, request, pconn, base_dn, inst->groupobj_scope, filter, NULL, NULL, NULL);
624  REXDENT();
625  switch (status) {
626  case LDAP_PROC_SUCCESS:
627  RDEBUG2("User found in group object \"%s\"", base_dn);
628  break;
629

641  *
642  * @param[out] p_result Result of calling the module.
643  * @param[in] inst rlm_ldap configuration.
644  * @param[in] request Current request.
645  * @param[in,out] pconn to use. May change as this function calls functions which auto re-connect.
646  * @param[in] dn of user object.
647  * @param[in] check vp containing the group value (name or dn).
648  */
649 unlang_action_t rlm_ldap_check_userobj_dynamic(rlm_rcode_t *p_result, rlm_ldap_t const *inst, request_t *request,
650  fr_ldap_connection_t **pconn,
651  char const *dn, fr_pair_t *check)
652 {
653  rlm_rcode_t rcode = RLM_MODULE_NOTFOUND, ret;
654  fr_ldap_rcode_t status;
655  bool name_is_dn = false, value_is_dn = false;
656 
657  LDAPMessage *result = NULL;
658  LDAPMessage *entry = NULL;
659  struct berval **values = NULL;
660 
661  char const *attrs[] = { inst->userobj_membership_attr, NULL };
662  int i, count, ldap_errno;
663 
664  RDEBUG2("Checking user object's %s attributes", inst->userobj_membership_attr);
665  RINDENT();
666  status = fr_ldap_search(&result, request, pconn, dn, LDAP_SCOPE_BASE, NULL, attrs, NULL, NULL);
667  REXDENT();
668  switch (status) {
669  case LDAP_PROC_SUCCESS:
670  break;
671 
672  case LDAP_PROC_NO_RESULT:
673  RDEBUG2("Can't check membership attributes, user object not found");
674 
675  rcode = RLM_MODULE_NOTFOUND;
676 
677  FALL_THROUGH;
678  default:
679  goto finish;
680  }
681 
682  entry = ldap_first_entry((*pconn)->handle, result);
683  if (!entry) {
684  ldap_get_option((*pconn)->handle, LDAP_OPT_RESULT_CODE, &ldap_errno);
685  REDEBUG("Failed retrieving entry: %s", ldap_err2string(ldap_errno));
686 
687  rcode = RLM_MODULE_FAIL;
688 
689  goto finish;
690  }
691 
692  values = ldap_get_values_len((*pconn)->handle, entry, inst->userobj_membership_attr);
693  if (!values) {
694  RDEBUG2("No group membership attribute(s) found in user object");
695 
696  goto finish;
697  }
698 
699  /*
700  * Loop over the list of groups the user is a member of,
701  * looking for a match.
702  */
703  name_is_dn = fr_ldap_util_is_dn(check->vp_strvalue, check->vp_length);
704  count = ldap_count_values_len(values);
705  for (i = 0; i < count; i++) {
706  value_is_dn = fr_ldap_util_is_dn(values[i]->bv_val, values[i]->bv_len);
707 
708  RDEBUG2("Processing %s value \"%pV\" as a %s", inst->userobj_membership_attr,
709  fr_box_strvalue_len(values[i]->bv_val, values[i]->bv_len),
710  value_is_dn ? "DN" : "group name");
711 
712  /*
713  * Both literal group names, do case sensitive comparison
714  */
715  if (!name_is_dn && !value_is_dn) {
716  if ((check->vp_length == values[i]->bv_len) &&
717  (memcmp(values[i]->bv_val, check->vp_strvalue, values[i]->bv_len) == 0)) {
718  RDEBUG2("User found in group \"%s\". Comparison between membership: name, check: name",
719  check->vp_strvalue);
720  rcode = RLM_MODULE_OK;
721 
722  goto finish;
723  }
724 
725  continue;
726  }
727 
728  /*
729  * Both DNs, do case insensitive, binary safe comparison
730  */
731  if (name_is_dn && value_is_dn) {
732  if (check->vp_length == values[i]->bv_len) {
733  int j;
734 
735  for (j = 0; j < (int)values[i]->bv_len; j++) {
736  if (tolower(values[i]->bv_val[j]) != tolower(check->vp_strvalue[j])) break;
737  }
738  if (j == (int)values[i]->bv_len) {
739  RDEBUG2("User found in group DN \"%s\". "
740  "Comparison between membership: dn, check: dn", check->vp_strvalue);
741  rcode = RLM_MODULE_OK;
742 
743  goto finish;
744  }
745  }
746 
747  continue;
748  }
749 
750  /*
751  * If the value is not a DN, and the name we were given is a dn
752  * convert the value to a DN and do a comparison.
753  */
754  if (!value_is_dn && name_is_dn) {
755  char *resolved;
756  bool eq = false;
757 
758  RINDENT();
759  rlm_ldap_group_dn2name(&ret, inst, request, pconn, check->vp_strvalue, &resolved);
760  REXDENT();
761 
762  if (ret == RLM_MODULE_NOOP) continue;
763 
764  if (ret != RLM_MODULE_OK) {
765  rcode = ret;
766  goto finish;
767  }
768 
769  if (((talloc_array_length(resolved) - 1) == values[i]->bv_len) &&
770  (memcmp(values[i]->bv_val, resolved, values[i]->bv_len) == 0)) eq = true;
771  talloc_free(resolved);
772  if (eq) {
773  RDEBUG2("User found in group \"%pV\". Comparison between membership: name, check: name "
774  "(resolved from DN \"%s\")",
775  fr_box_strvalue_len(values[i]->bv_val, values[i]->bv_len), check->vp_strvalue);
776  rcode = RLM_MODULE_OK;
777 
778  goto finish;
779  }
780 
781  continue;
782  }
783 
784  /*
785  * We have a value which is a DN, and a check item which specifies the name of a group,
786  * convert the value to a name so we can do a comparison.
787  */
788  if (value_is_dn && !name_is_dn) {
789  char *resolved;
790  char *value;
791  bool eq = false;
792 
793  value = fr_ldap_berval_to_string(request, values[i]);
794  RINDENT();
795  rlm_ldap_group_dn2name(&ret, inst, request, pconn, value, &resolved);
796  REXDENT();
797  talloc_free(value);
798 
799  if (ret == RLM_MODULE_NOOP) continue;
800 
801  if (ret != RLM_MODULE_OK) {
802  rcode = ret;
803  goto finish;
804  }
805 
806  if (((talloc_array_length(resolved) - 1) == check->vp_length) &&
807  (memcmp(check->vp_strvalue, resolved, check->vp_length) == 0)) eq = true;
808  talloc_free(resolved);
809  if (eq) {
810  RDEBUG2("User found in group \"%pV\". Comparison between membership: name "
811  "(resolved from DN \"%s\"), check: name", &check->data, value);
812  rcode = RLM_MODULE_OK;
813 
814  goto finish;
815  }
816 
817  continue;
818  }
819  fr_assert(0);
820  }
821 
822 finish:
823  if (values) ldap_value_free_len(values);
824  if (result) ldap_msgfree(result);
825 
826  RETURN_MODULE_RCODE(rcode);
827 }
