64 static bool all_digits(char const *string)
65 {
66  char const *p = string;
67 
68  fr_assert(p != NULL);
69 
70  if (*p == '\0') return false;
71 
72  if (*p == '-') p++;
73 
74  while (isdigit((int) *p)) p++;
75 
76  return (*p == '\0');
77 }

124 /** Evaluate a template
125  *
126  * Converts a tmpl_t to a boolean value.
127  *
128  * @param[in] request the request_t
129  * @param[in] depth of the recursion (only used for debugging)
130  * @param[in] vpt the template to evaluate
131  * @return
132  * - -1 on failure.
133  * - 0 for "no match".
134  * - 1 for "match".
135  */
136 int cond_eval_tmpl(request_t *request, UNUSED int depth, tmpl_t const *vpt)
137 {
138  int rcode = -1;
139 
140  switch (vpt->type) {
141  case TMPL_TYPE_UNRESOLVED:
142  /*
143  * Empty string is false, non-empty string
144  * is true.
145  *
146  * @todo: Maybe also check for digits?
147  */
148  rcode = (*vpt->data.unescaped != '\0');
149  break;
150 
151  case TMPL_TYPE_ATTR:
152  case TMPL_TYPE_LIST:
153  if (tmpl_find_vp(NULL, request, vpt) == 0) {
154  rcode = true;
155  } else {
156  rcode = false;
157  }
158  break;
159 
160  case TMPL_TYPE_XLAT:
161  case TMPL_TYPE_EXEC:
162  {
163  char *p;
164  ssize_t slen;
165 
166  slen = tmpl_aexpand(request, &p, request, vpt, NULL, NULL);
167  if (slen < 0) {
168  EVAL_DEBUG("FAIL %d", __LINE__);
169  return -1;
170  }
171  talloc_free(p);
172 
173  if (slen == 0) return false;
174  }
175  return true;
176 
177  /*
178  * Can't have a bare ... (/foo/) ...
179  */
180  case TMPL_TYPE_REGEX:
181  case TMPL_TYPE_REGEX_UNCOMPILED:
182  case TMPL_TYPE_REGEX_XLAT:
183  case TMPL_TYPE_REGEX_XLAT_UNRESOLVED:
184  fr_assert(0 == 1);
185  FALL_THROUGH;
186 
187  default:
188  EVAL_DEBUG("FAIL %d", __LINE__);
189  rcode = -1;
190  break;
191  }
192 
193  return rcode;
194 }

197 /** Perform a regular expressions comparison between two operands
198  *
199  * @return
200  * - -1 on failure.
201  * - 0 for "no match".
202  * - 1 for "match".
203  */
204 static int cond_do_regex(request_t *request, fr_cond_t const *c,
205  fr_value_box_t const *lhs,
206  fr_value_box_t const *rhs)
207 {
208  map_t const *map = c->data.map;
209 
210  ssize_t slen;
211  uint32_t subcaptures;
212  int ret;
213 
214  regex_t *preg, *rreg = NULL;
215  fr_regmatch_t *regmatch;
216 
217  if (!fr_cond_assert(lhs != NULL)) return -1;
218  if (!fr_cond_assert(lhs->type == FR_TYPE_STRING)) return -1;
219 
220  EVAL_DEBUG("CMP WITH REGEX");
221 
222  switch (map->rhs->type) {
223  case TMPL_TYPE_REGEX: /* pre-compiled to a regex */
224  preg = tmpl_regex(map->rhs);
225  break;
226 
227  default:
228  if (!fr_cond_assert(rhs && rhs->type == FR_TYPE_STRING)) return -1;
229  if (!fr_cond_assert(rhs && rhs->vb_strvalue)) return -1;
230  slen = regex_compile(request, &rreg, rhs->vb_strvalue, rhs->vb_length,
231  tmpl_regex_flags(map->rhs), true, true);
232  if (slen <= 0) {
233  REMARKER(rhs->vb_strvalue, -slen, "%s", fr_strerror());
234  EVAL_DEBUG("FAIL %d", __LINE__);
235 
236  return -1;
237  }
238  preg = rreg;
239  break;
240  }
241 
242  subcaptures = regex_subcapture_count(preg);
243  if (!subcaptures) subcaptures = REQUEST_MAX_REGEX + 1; /* +1 for %{0} (whole match) capture group */
244  MEM(regmatch = regex_match_data_alloc(NULL, subcaptures));
245 
246  /*
247  * Evaluate the expression
248  */
249  ret = regex_exec(preg, lhs->vb_strvalue, lhs->vb_length, regmatch);
250  switch (ret) {
251  case 0:
252  EVAL_DEBUG("CLEARING SUBCAPTURES");
253  regex_sub_to_request(request, NULL, NULL); /* clear out old entries */
254  break;
255 
256  case 1:
257  EVAL_DEBUG("SETTING SUBCAPTURES");
258  regex_sub_to_request(request, &preg, &regmatch);
259  break;
260 
261  case -1:
262  EVAL_DEBUG("REGEX ERROR");
263  RPEDEBUG("regex failed");
264  break;
265 
266  default:
267  break;
268  }
269 
270  talloc_free(regmatch); /* free if not consumed */
271  if (preg) talloc_free(rreg);
272 
273  return ret;
274 }
275 #endif

302 /** Call the correct data comparison function for the condition
303  *
304  * Deals with regular expression comparisons, virtual attribute
305  * comparisons, and data comparisons.
306  *
307  * @return
308  * - -1 on failure.
309  * - 0 for "no match".
310  * - 1 for "match".
311  */
312 static int cond_cmp_values(request_t *request, fr_cond_t const *c, fr_value_box_t const *lhs, fr_value_box_t const *rhs)
313 {
314  map_t const *map = c->data.map;
315  int rcode;
316 
317 #ifdef WITH_EVAL_DEBUG
318  EVAL_DEBUG("CMP OPERANDS");
319  cond_print_operands(lhs, rhs);
320 #endif
321 
322 #ifdef HAVE_REGEX
323  /*
324  * Regex comparison
325  */
326  if (map->op == T_OP_REG_EQ) {
327  rcode = cond_do_regex(request, c, lhs, rhs);
328  goto finish;
329  }
330 #endif
331  /*
332  * Virtual attribute comparison.
333  */
334  if (c->pass2_fixup == PASS2_PAIRCOMPARE) {
335  fr_pair_t *vp;
336 
337  EVAL_DEBUG("CMP WITH PAIRCOMPARE");
338  fr_assert(tmpl_is_attr(map->lhs));
339 
340  MEM(vp = fr_pair_afrom_da(request, tmpl_da(map->lhs)));
341  vp->op = c->data.map->op;
342 
343  fr_value_box_copy(vp, &vp->data, rhs);
344 
345  rcode = paircmp(request, request->request_pairs, vp);
346  rcode = (rcode == 0) ? 1 : 0;
347  talloc_free(vp);
348  goto finish;
349  }
350 
351  EVAL_DEBUG("CMP WITH VALUE DATA");
352  rcode = fr_value_box_cmp_op(map->op, lhs, rhs);
353 finish:
354  switch (rcode) {
355  case 0:
356  EVAL_DEBUG("FALSE");
357  break;
358 
359  case 1:
360  EVAL_DEBUG("TRUE");
361  break;
362 
363  default:
364  EVAL_DEBUG("ERROR %i", rcode);
365  break;
366  }
367 
368  return rcode;
369 }

372 static size_t regex_escape(UNUSED request_t *request, char *out, size_t outlen, char const *in, UNUSED void *arg)
373 {
374  char *p = out;
375 
376  while (*in && (outlen >= 2)) {
377  switch (*in) {
378  case '\\':
379  case '.':
380  case '*':
381  case '+':
382  case '?':
383  case '|':
384  case '^':
385  case '$':
386  case '[': /* we don't list close braces */
387  case '{':
388  case '(':
389  if (outlen < 3) goto done;
390 
391  *(p++) = '\\';
392  outlen--;
393  FALL_THROUGH;
394 
395  default:
396  *(p++) = *(in++);
397  outlen--;
398  break;
399  }
400  }
401 
402 done:
403  *(p++) = '\0';
404  return p - out;
405 }

407 /** Convert both operands to the same type
408  *
409  * If casting is successful, we call cond_cmp_values to do the comparison
410  *
411  * @return
412  * - -1 on failure.
413  * - 0 for "no match".
414  * - 1 for "match".
415  */
416 static int cond_normalise_and_cmp(request_t *request, fr_cond_t const *c, fr_value_box_t const *lhs)
417 {
418  map_t const *map = c->data.map;
419 
420  int rcode;
421 
422  fr_value_box_t const *rhs = NULL;
423 
424  fr_dict_attr_t const *cast = NULL;
425  fr_type_t cast_type = FR_TYPE_INVALID;
426 
427  fr_value_box_t lhs_cast = { .type = FR_TYPE_INVALID };
428  fr_value_box_t rhs_cast = { .type = FR_TYPE_INVALID };
429 
430  xlat_escape_legacy_t escape = NULL;
431 
432  /*
433  * Cast operand to correct type.
434  */
435 #define CAST(_s) \
436 do {\
437  if ((cast_type != FR_TYPE_INVALID) && _s && (_s ->type != FR_TYPE_INVALID) && (cast_type != _s->type)) {\
438  EVAL_DEBUG("CASTING " #_s " FROM %s TO %s",\
439  fr_table_str_by_value(fr_value_box_type_table, _s->type, "<INVALID>"),\
440  fr_table_str_by_value(fr_value_box_type_table, cast_type, "<INVALID>"));\
441  if (fr_value_box_cast(request, &_s ## _cast, cast_type, cast, _s) < 0) {\
442  if (request) RPEDEBUG("Failed casting " #_s " operand");\
443  rcode = -1;\
444  goto finish;\
445  }\
446  _s = &_s ## _cast;\
447  }\
448 } while (0)
449 
450 #define CHECK_INT_CAST(_l, _r) \
451 do {\
452  if ((cast_type == FR_TYPE_INVALID) &&\
453  _l && (_l->type == FR_TYPE_STRING) &&\
454  _r && (_r->type == FR_TYPE_STRING) &&\
455  all_digits(lhs->vb_strvalue) && all_digits(rhs->vb_strvalue)) {\
456  cast_type = FR_TYPE_UINT64;\
457  EVAL_DEBUG("OPERANDS ARE NUMBER STRINGS, SETTING CAST TO uint64");\
458  }\
459 } while (0)
460 
461  /*
462  * Regular expressions need both operands to be strings
463  */
464 #ifdef HAVE_REGEX
465  if (map->op == T_OP_REG_EQ) {
466  cast_type = FR_TYPE_STRING;
467 
468  if (tmpl_is_regex_xlat(map->rhs)) escape = regex_escape;
469  }
470  else
471 #endif
472  /*
473  * If it's a pair comparison, data gets cast to the
474  * type of the pair comparison attribute.
475  *
476  * Magic attribute is always the LHS.
477  */
478  if (c->pass2_fixup == PASS2_PAIRCOMPARE) {
479  fr_assert(!c->cast);
480  fr_assert(tmpl_is_attr(map->lhs));
481  fr_assert(!tmpl_is_attr(map->rhs) || !paircmp_find(tmpl_da(map->rhs))); /* expensive assert */
482 
483  cast = tmpl_da(map->lhs);
484 
485  EVAL_DEBUG("NORMALISATION TYPE %s (PAIRCMP TYPE)",
486  fr_table_str_by_value(fr_value_box_type_table, cast->type, "<INVALID>"));
487  /*
488  * Otherwise we use the explicit cast, or implicit
489  * cast (from an attribute reference).
490  * We already have the data for the lhs, so we convert
491  * it here.
492  */
493  } else if (c->cast) {
494  cast = c->cast;
495  EVAL_DEBUG("NORMALISATION TYPE %s (EXPLICIT CAST)",
496  fr_table_str_by_value(fr_value_box_type_table, cast->type, "<INVALID>"));
497  } else if (tmpl_is_attr(map->lhs)) {
498  cast = tmpl_da(map->lhs);
499  EVAL_DEBUG("NORMALISATION TYPE %s (IMPLICIT FROM LHS REF)",
500  fr_table_str_by_value(fr_value_box_type_table, cast->type, "<INVALID>"));
501  } else if (tmpl_is_attr(map->rhs)) {
502  cast = tmpl_da(map->rhs);
503  EVAL_DEBUG("NORMALISATION TYPE %s (IMPLICIT FROM RHS REF)",
504  fr_table_str_by_value(fr_value_box_type_table, cast->type, "<INVALID>"));
505  } else if (tmpl_is_data(map->lhs)) {
506  cast_type = tmpl_value_type(map->lhs);
507  EVAL_DEBUG("NORMALISATION TYPE %s (IMPLICIT FROM LHS DATA)",
508  fr_table_str_by_value(fr_value_box_type_table, cast_type, "<INVALID>"));
509  } else if (tmpl_is_data(map->rhs)) {
510  cast_type = tmpl_value_type(map->rhs);
511  EVAL_DEBUG("NORMALISATION TYPE %s (IMPLICIT FROM RHS DATA)",
512  fr_table_str_by_value(fr_value_box_type_table, cast_type, "<INVALID>"));
513  }
514 
515  if (cast) cast_type = cast->type;
516 
517  switch (map->rhs->type) {
518  case TMPL_TYPE_ATTR:
519  {
520  fr_pair_t *vp;
521  fr_cursor_t cursor;
522  tmpl_cursor_ctx_t cc;
523 
524  for (vp = tmpl_cursor_init(&rcode, request, &cc, &cursor, request, map->rhs);
525  vp;
526  vp = fr_cursor_next(&cursor)) {
527  rhs = &vp->data;
528 
529  CHECK_INT_CAST(lhs, rhs);
530  CAST(lhs);
531  CAST(rhs);
532 
533  rcode = cond_cmp_values(request, c, lhs, rhs);
534  if (rcode != 0) break;
535 
536  fr_value_box_clear(&rhs_cast);
537  }
538  tmpl_cursor_clear(&cc);
539  }
540  break;
541 
542  case TMPL_TYPE_DATA:
543  rhs = tmpl_value(map->rhs);
544 
545  CHECK_INT_CAST(lhs, rhs);
546  CAST(lhs);
547  CAST(rhs);
548 
549  rcode = cond_cmp_values(request, c, lhs, rhs);
550  break;
551 
552  /*
553  * Expanded types start as strings, then get converted
554  * to the type of the attribute or the explicit cast.
555  */
556  case TMPL_TYPE_UNRESOLVED:
557  case TMPL_TYPE_EXEC:
558  case TMPL_TYPE_XLAT:
559  case TMPL_TYPE_REGEX_XLAT:
560  {
561  ssize_t ret;
562  fr_value_box_t data;
563 
564  if (!tmpl_is_unresolved(map->rhs)) {
565  char *p;
566 
567  ret = tmpl_aexpand(request, &p, request, map->rhs, escape, NULL);
568  if (ret < 0) {
569  EVAL_DEBUG("FAIL [%i]", __LINE__);
570  rcode = -1;
571  goto finish;
572  }
573  fr_value_box_bstrndup_shallow(&data, NULL, p, ret, false);
574  } else {
575  fr_value_box_bstrdup_buffer_shallow(NULL, &data, NULL, map->rhs->data.unescaped, false);
576  }
577  rhs = &data;
578 
579  CHECK_INT_CAST(lhs, rhs);
580  CAST(lhs);
581  CAST(rhs);
582 
583  rcode = cond_cmp_values(request, c, lhs, rhs);
584  if (!tmpl_is_unresolved(map->rhs)) talloc_free(data.datum.ptr);
585 
586  break;
587  }
588 
589  /*
590  * RHS is a compiled regex, we don't need to do anything with it.
591  */
592  case TMPL_TYPE_REGEX:
593  CAST(lhs);
594  rcode = cond_cmp_values(request, c, lhs, NULL);
595  break;
596  /*
597  * Unsupported types (should have been parse errors)
598  */
599  case TMPL_TYPE_NULL:
600  case TMPL_TYPE_LIST:
601  case TMPL_TYPE_UNINITIALISED:
602  case TMPL_TYPE_ATTR_UNRESOLVED:
603  case TMPL_TYPE_XLAT_UNRESOLVED: /* should now be a TMPL_TYPE_XLAT */
604  case TMPL_TYPE_EXEC_UNRESOLVED: /* should now be a TMPL_TYPE_EXEC */
605  case TMPL_TYPE_REGEX_UNCOMPILED: /* should now be a TMPL_TYPE_REGEX */
606  case TMPL_TYPE_REGEX_XLAT_UNRESOLVED: /* Should now be a TMPL_TYPE_REGEX_XLAT */
607  case TMPL_TYPE_MAX:
608  fr_assert(0);
609  rcode = -1;
610  break;
611  }
612 
613 finish:
614  fr_value_box_clear(&lhs_cast);
615  fr_value_box_clear(&rhs_cast);
616 
617  return rcode;
618 }

621 /** Evaluate a map
622  *
623  * @param[in] request the request_t
624  * @param[in] depth of the recursion (only used for debugging)
625  * @param[in] c the condition to evaluate
626  * @return
627  * - -1 on failure.
628  * - 0 for "no match".
629  * - 1 for "match".
630  */
631 int cond_eval_map(request_t *request, UNUSED int depth, fr_cond_t const *c)
632 {
633  int rcode = 0;
634 
635  map_t const *map = c->data.map;
636 
637  EVAL_DEBUG(">>> MAP TYPES LHS: %s, RHS: %s",
638  fr_table_str_by_value(tmpl_type_table, map->lhs->type, "???"),
639  fr_table_str_by_value(tmpl_type_table, map->rhs->type, "???"));
640 
641  MAP_VERIFY(map);
642 
643  switch (map->lhs->type) {
644  /*
645  * LHS is an attribute or list
646  */
647  case TMPL_TYPE_LIST:
648  case TMPL_TYPE_ATTR:
649  {
650  fr_pair_t *vp;
651  fr_cursor_t cursor;
652  tmpl_cursor_ctx_t cc;
653  /*
654  * Legacy paircmp call, skip processing the magic attribute
655  * if it's the LHS and cast RHS to the same type.
656  */
657  if ((c->pass2_fixup == PASS2_PAIRCOMPARE) && (map->op != T_OP_REG_EQ)) {
658 #ifndef NDEBUG
659  fr_assert(paircmp_find(tmpl_da(map->lhs))); /* expensive assert */
660 #endif
661  rcode = cond_normalise_and_cmp(request, c, NULL);
662  break;
663  }
664  for (vp = tmpl_cursor_init(&rcode, request, &cc, &cursor, request, map->lhs);
665  vp;
666  vp = fr_cursor_next(&cursor)) {
667  /*

776  break;
777

784  }

787
