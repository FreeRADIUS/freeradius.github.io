48 void request_stats_final(request_t *request)
49 {
50  if (request->master_state == REQUEST_COUNTED) return;
51 
52  if (!request->listener) return;
53  if (!request->client) return;
54  if (!request->packet) return;
55 
56  if ((request->listener->type != RAD_LISTEN_NONE) &&
57  (request->listener->type != RAD_LISTEN_ACCT) &&
58  (request->listener->type != RAD_LISTEN_AUTH)) return;
59 
60  /* don't count statistic requests */
61  if (request->packet->code == FR_CODE_STATUS_SERVER)
62  return;
63 
64 #undef INC_AUTH
65 #define INC_AUTH(_x) radius_auth_stats._x++;request->client->auth._x++;
66 
67 #undef INC_ACCT
68 #define INC_ACCT(_x) radius_acct_stats._x++;request->client->acct._x++
69 
70  /*
71  * Update the statistics.
72  *
73  * Note that we do NOT do this in a child thread.
74  * Instead, we update the stats when a request is
75  * deleted, because only the main server thread calls
76  * this function, which makes it thread-safe.
77  */
78  if (request->reply && (request->packet->code != FR_CODE_STATUS_SERVER)) switch (request->reply->code) {
79  case FR_CODE_ACCESS_ACCEPT:
80  INC_AUTH(total_access_accepts);
81 
82  auth_stats:
83  INC_AUTH(total_responses);
84 
85  /*
86  * FIXME: Do the time calculations once...
87  */
88  fr_stats_bins(&radius_auth_stats,
89  request->packet->timestamp,
90  request->reply->timestamp);
91  fr_stats_bins(&request->client->auth,
92  request->packet->timestamp,
93  request->reply->timestamp);
94  break;
95 
96  case FR_CODE_ACCESS_REJECT:
97  INC_AUTH(total_access_rejects);
98  goto auth_stats;
99 
100  case FR_CODE_ACCESS_CHALLENGE:
101  INC_AUTH(total_access_challenges);
102  goto auth_stats;
103 
104  case FR_CODE_ACCOUNTING_RESPONSE:
105  INC_ACCT(total_responses);
106  fr_stats_bins(&radius_acct_stats,
107  request->packet->timestamp,
108  request->reply->timestamp);
109  break;
110 
111  /*
112  * No response, it must have been a bad
113  * authenticator.
114  */
115  case 0:
116  switch (request->packet->code) {
117  case FR_CODE_ACCESS_REQUEST:
118  if (request->reply->id == -1) {
119  INC_AUTH(total_bad_authenticators);
120  } else {
121  INC_AUTH(total_packets_dropped);
122  }
123  break;
124 
125 
126  case FR_CODE_ACCOUNTING_REQUEST:
127  if (request->reply->id == -1) {
128  INC_ACCT(total_bad_authenticators);
129  } else {
130  INC_ACCT(total_packets_dropped);
131  }
132  break;
133 
134  default:
135  break;
136  }
137  break;
138 
139  default:
140  break;
141  }
142 
143  request->master_state = REQUEST_COUNTED;
144 }

156 void radius_stats_ema(fr_stats_ema_t *ema, fr_time_t start, fr_time_t end)
157 {
158  uint64_t tdiff;
159 #ifdef WITH_STATS_DEBUG
160  static int n = 0;
161 #endif
162  if (ema->window == 0) return;
163 
164  fr_assert(start <= end);
165 
166  /*
167  * Initialize it.
168  */
169  if (ema->f1 == 0) {
170  if (ema->window > 10000) ema->window = 10000;
171 
172  ema->f1 = (2 * F_EMA_SCALE) / (ema->window + 1);
173  ema->f10 = (2 * F_EMA_SCALE) / ((10 * ema->window) + 1);
174  }
175 
176  tdiff = fr_time_delta_to_usec(start);
177  tdiff -= fr_time_delta_to_usec(end);
178  tdiff *= EMA_SCALE;
179 
180  if (ema->ema1 == 0) {
181  ema->ema1 = tdiff;
182  ema->ema10 = tdiff;
183  } else {
184  int diff;
185 
186  diff = ema->f1 * (tdiff - ema->ema1);
187  ema->ema1 += (diff / 1000000);
188 
189  diff = ema->f10 * (tdiff - ema->ema10);
190  ema->ema10 += (diff / 1000000);
191  }
192 
193 
194 #ifdef WITH_STATS_DEBUG
195  DEBUG("time %d %d.%06d\t%d.%06d\t%d.%06d\n",
196  n, tdiff / PREC, (tdiff / EMA_SCALE) % USEC,
197  ema->ema1 / PREC, (ema->ema1 / EMA_SCALE) % USEC,
198  ema->ema10 / PREC, (ema->ema10 / EMA_SCALE) % USEC);
199  n++;
200 #endif
201 }

203 /** Sort latency times into bins
204  *
205  * This solves the problem of attempting to keep min/max/avg latencies, whilst
206  * not knowing what the polling frequency will be.
207  *
208  * @param[out] stats Holding monotonically increasing stats bins.
209  * @param[in] start of the request.
210  * @param[in] end of the request.
211  */
212 void fr_stats_bins(fr_stats_t *stats, fr_time_t start, fr_time_t end)
213 {
214  fr_time_t diff;
215  uint32_t delay;
216 
217  if (end < start) return; /* bad data */
218  diff = end - start;
219 
220  if (diff >= fr_time_delta_from_sec(10)) {
221  stats->elapsed[7]++;
222  } else {
223  int i;
224  uint32_t cmp;
225 
226  delay = fr_time_delta_to_usec(diff);
227 
228  cmp = 10;
229  for (i = 0; i < 7; i++) {
230  if (delay < cmp) {
231  stats->elapsed[i]++;
232  break;
233  }
234  cmp *= 10;
235  }
236  }
237 }
