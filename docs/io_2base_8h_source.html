152 /** Read from a socket.
153  *
154  * The network side guarantees that the read routine can leave partial
155  * data in the buffer. That data will be there on the next call to
156  * read. However, the data MAY have moved, so please do not keep a
157  * pointer to 'buffer' around.
158  *
159  * datagram sockets should always set '*leftover = 0'.
160  *
161  * stream sockets can read one packet, and set '*leftover' to how many
162  * bytes are left in the buffer. The read routine will be called
163  * again, with a (possibly new) buffer, but with 'leftover' bytes left
164  * in the buffer. The value in 'leftover'' will be the same as from
165  * the previous call, so the reader does not need to track it.
166  *
167  * @param[in] li the listener for this socket
168  * @param[out] packet_ctx Where to write a newly allocated packet_ctx struct containing request specific data.
169  * @param[out] recv_time A pointer to a time when the packet was received
170  * @param[in,out] buffer the buffer where the raw packet will be written to (or read from)
171  * @param[in] buffer_len the length of the buffer
172  * @param[out] leftover bytes left in the buffer after reading a full packet.
173  * @param[out] priority priority of this packet (0 = low, 65535 = high)
174  * @param[out] dup is it dup or new
175  * @return
176  * - <0 on error
177  * - >=0 length of the data read or written.
178  */
179 typedef ssize_t (*fr_io_data_read_t)(fr_listen_t *li, void **packet_ctx, fr_time_t *recv_time, uint8_t *buffer, size_t buffer_len, size_t *leftover, uint32_t *priority, bool *dup);

181 /** Write a socket.
182  *
183  * If the socket is a datagram socket, then the function can read or
184  * write directly into the buffer. Stream sockets are a bit more complicated.
185  *
186  * A stream reader can read data into the buffer, and be guaranteed
187  * that the data will not change in between subsequent calls to the
188  * read routine.
189  *
190  * A stream writer MUST be prepared for the caller to delete the data
191  * immediately after calling the write routine. This means that if
192  * the socket is not ready, the writer MUST copy the data to an
193  * internal buffer, usually in instance. It MUST then have a
194  * write callback on the socket, which is called when the socket is
195  * ready for writing. That callback can then write the internal
196  * buffer to the socket.
197  *
198  * i.e. this write() function is a way for the network thread to
199  * write packets to the transport context. The data may or may not
200  * go out to the network right away.
201  *
202  * If the write function does a partial write, it should return a
203  * value smaller than buffer_len to indicate this. The network
204  * functions will then pass that value to a subsequent write call, in
205  * the "written" argument.
206  *
207  * The reason for this odd API is that read/write should be writing
208  * *packets*, not raw streams of bytes. This API allows the "buffer"
209  * parameter to always contain a full packet.
210  *
211  * @param[in] li the listener for this socket
212  * @param[in] packet_ctx Request specific data.
213  * @param[in] request_time when the original request was received
214  * @param[in] buffer the buffer where the raw packet will be written from
215  * @param[in] buffer_len the length of the buffer
216  * @param[in] written total number of bytes written in previous calls for this packet.
217  * @return
218  * - <0 on error
219  * - >=0 length of the data read or written.
220  */
221 typedef ssize_t (*fr_io_data_write_t)(fr_listen_t *li, void *packet_ctx, fr_time_t request_time,
222  uint8_t *buffer, size_t buffer_len, size_t written);

224 /** Inject data into a socket.
225  *
226  * This function allows callers to inject data into a socket, just as if the data
227  * was read from a socket.
228  *
229  * Note that this function is NOT an analog to fr_io_data_read_t.
230  * That is, the called function MUST copy the packet pointer into an
231  * internal list, so that subsequent calls to read() will return this
232  * packet.
233  *
234  * The network side ensures that the packet buffer remains available
235  * to the called function for the duration of an inject() and read()
236  * call. i.e. the packet contents do NOT have to be saved, and the
237  * code can instead save the pointer to the buffer.
238  *
239  * However, this buffer MUST be immediately returned on a subsequent
240  * call to read(). If it is not returned, the memory is still freed,
241  * and the pointer becomes invalid. Subsequent access to the buffer
242  * will result in crashes.
243  *
244  * @param[in] li the listener for this socket
245  * @param[in] buffer the buffer where the raw packet to be injected
246  * @param[in] buffer_len the length of the buffer
247  * @param[in] recv_time when the packet was received
248  * @return
249  * - <0 on error
250  * - 0 on success
251  */
252 typedef int (*fr_io_data_inject_t)(fr_listen_t *li,uint8_t *buffer, size_t buffer_len, fr_time_t recv_time);

274 /** Compare two tracking structures for storing in a duplicate detection tree.
275  *
276  * We presume that the packets are well formed.
277  *
278  * The comparison should be stable. i.e. compare the packets a field
279  * at a time. If the field is different, return the result from that
280  * field.
281  *
282  * The comparison order of the fields should be "very different" to
283  * "much the same". The packets are put into an rbtree, so having a
284  * large fanout at the top is useful.
285  *
286  * Note that this function should not check if the packets are
287  * completely identical. Instead, it checks particular fields in the
288  * packet so that we can distinguish packets without checking the entire packet.
289  *
290  * @param[in] instance the context for this function
291  * @param[in] thread_instance the thread instance for this function
292  * @param[in] client the client associated with this packet
293  * @param[in] one packet tracking structure one
294  * @param[in] two packet tracking structure two
295  * @return
296  * - <0 on packet one "smaller" than packet two
297  * - >0 on packet two "larger" than packet one
298  * - =0 on the two packets being identical
299  */
