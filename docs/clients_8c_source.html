38 /** Iterate over pairs in mapping section recording their values in an array
39  *
40  * This array is the list of attributes we retrieve from LDAP, and is NULL
41  * terminated.
42  *
43  * If we hit a CONF_SECTION we recurse and process its CONF_PAIRS too.
44  *
45  * @param[out] values array of char pointers.
46  * @param[in,out] idx records current array offset.
47  * @param[in] cs to iterate over.
48  * @return
49  * - 0 on success.
50  * - -1 on failure.
51  */
52 static int rlm_ldap_client_get_attrs(char const **values, int *idx, CONF_SECTION const *cs)
53 {
54  CONF_ITEM const *ci;
55 
56  for (ci = cf_item_next(cs, NULL);
57  ci != NULL;
58  ci = cf_item_next(cs, ci)) {
59  char const *value;
60 
61  if (cf_item_is_section(ci)) {
62  if (rlm_ldap_client_get_attrs(values, idx, cf_item_to_section(ci)) < 0) return -1;
63  continue;
64  }
65 
66  value = cf_pair_value(cf_item_to_pair(ci));
67  if (!value) return -1;
68 
69  values[(*idx)++] = value;
70  }
71 
72  values[*idx] = NULL;
73 
74  return 0;
75 }

82 static int _get_client_value(char **out, CONF_PAIR const *cp, void *data)
83 {
84  struct berval **values;
85  ldap_client_data_t *this = data;
86 
87  values = ldap_get_values_len(this->conn->handle, this->entry, cf_pair_value(cp));
88  if (!values) {
89  *out = NULL;
90  return 0;
91  }
92 
93  *out = fr_ldap_berval_to_string(NULL, values[0]);
94  ldap_value_free_len(values);
95 
96  if (!*out) return -1;
97  return 0;
98 }

100 /** Load clients from LDAP on server start
101  *
102  * @param[in] inst rlm_ldap configuration.
103  * @param[in] tmpl to use as the base for the new client.
104  * @param[in] map to load client attribute/LDAP attribute mappings from.
105  * @return
106  * - 0 on success.
107  * - -1 on failure.
108  */
109 int rlm_ldap_client_load(rlm_ldap_t const *inst, CONF_SECTION *tmpl, CONF_SECTION *map)
110 {
111  int ret = 0;
112  fr_ldap_rcode_t status;
113  fr_ldap_connection_t *conn = NULL;
114 
115  char const **attrs = NULL;
116 
117  CONF_PAIR *cp;
118  int count = 0, idx = 0;
119 
120  LDAPMessage *result = NULL;
121  LDAPMessage *entry;
122  char *dn = NULL;
123 
124  RADCLIENT *c;
125 
126  DEBUG("Loading dynamic clients");
127 
128  fr_assert(inst->clientobj_base_dn);
129 
130  count = cf_pair_count(map);
131  count++;
132 
133  /*
134  * Create an array of LDAP attributes to feed to fr_ldap_search.
135  */
136  attrs = talloc_array(inst, char const *, count);
137  if (rlm_ldap_client_get_attrs(attrs, &idx, map) < 0) {
138  talloc_free(attrs);
139  return -1;
140  }
141 
142  conn = mod_conn_get(inst, NULL);
143  if (!conn) {
144  talloc_free(attrs);
145  return -1;
146  }
147 
148  /*
149  * Perform all searches as the admin user.
150  */
151  if (conn->rebound) {
152  status = fr_ldap_bind(NULL, &conn,
153  conn->config->admin_identity, conn->config->admin_password,
154  &(conn->config->admin_sasl),
155  NULL,
156  NULL, NULL);
157  if (status != LDAP_PROC_SUCCESS) {
158  ret = -1;
159  goto finish;
160  }
161 
162  fr_assert(conn);
163 
164  conn->rebound = false;
165  }
166 
167  status = fr_ldap_search(&result, NULL, &conn, inst->clientobj_base_dn, inst->clientobj_scope,
168  inst->clientobj_filter, attrs, NULL, NULL);
169  switch (status) {
170  case LDAP_PROC_SUCCESS:
171  break;
172 
173  case LDAP_PROC_NO_RESULT:
174  INFO("No clients were found in the directory");
175  ret = 0;
176  goto finish;
177 
178  default:
179  ret = -1;
180  goto finish;
181  }
182 
183  fr_assert(conn);
184  entry = ldap_first_entry(conn->handle, result);
185  if (!entry) {
186  int ldap_errno;
187 
188  ldap_get_option(conn->handle, LDAP_OPT_RESULT_CODE, &ldap_errno);
189  ERROR("Failed retrieving entry: %s", ldap_err2string(ldap_errno));
190 
191  ret = -1;
192  goto finish;
193  }
194 
195  do {
196  ldap_client_data_t data;
197 
198  CONF_SECTION *client;
199  char *id;
200 
201  struct berval **values;
202 
203  id = dn = ldap_get_dn(conn->handle, entry);
204  if (!dn) {
205  int ldap_errno;
206 
207  ldap_get_option(conn->handle, LDAP_OPT_RESULT_CODE, &ldap_errno);
208  ERROR("Retrieving object DN from entry failed: %s", ldap_err2string(ldap_errno));
209 
210  goto finish;
211  }
212  fr_ldap_util_normalise_dn(dn, dn);
213 
214  cp = cf_pair_find(map, "identifier");
215  if (cp) {
216  values = ldap_get_values_len(conn->handle, entry, cf_pair_value(cp));
217  if (values) id = fr_ldap_berval_to_string(NULL, values[0]);
218  ldap_value_free_len(values);
219  }
220 
221  /*
222  * Iterate over mapping sections
223  */
224  client = tmpl ? cf_section_dup(NULL, NULL, tmpl, "client", id, true) :
225  cf_section_alloc(NULL, NULL, "client", id);
226 
227  data.conn = conn;
228  data.entry = entry;
229 
230  if (client_map_section(client, map, _get_client_value, &data) < 0) {
231  talloc_free(client);
232  ret = -1;
233  goto finish;
234  }
235 
236  /*
237  *@todo these should be parented from something
238  */
239  c = client_afrom_cs(NULL, client, NULL);
240  if (!c) {
241  talloc_free(client);
242  ret = -1;
243  goto finish;
244  }
245 
246  /*
247  * Client parents the CONF_SECTION which defined it
248  */
249  talloc_steal(c, client);
250 
251  if (!client_add(NULL, c)) {
252  ERROR("Failed to add client \"%s\", possible duplicate?", dn);
253  ret = -1;
254  client_free(c);
255  goto finish;
256  }
257 
258  DEBUG("Client \"%s\" added", dn);
259 
260  ldap_memfree(dn);
261  dn = NULL;
262  } while ((entry = ldap_next_entry(conn->handle, entry)));
263 
264 finish:
265  talloc_free(attrs);
266  if (dn) ldap_memfree(dn);
267  if (result) ldap_msgfree(result);
268 
269  mod_conn_release(inst, NULL, conn);
270 
271  return ret;
272 }
