465 static size_t config_escape_func(UNUSED request_t *request, char *out, size_t outlen, char const *in, UNUSED void *arg)
466 {
467  size_t len = 0;
468  static char const disallowed[] = "%{}\\'\"`";
469 
470  while (in[0]) {
471  /*
472  * Non-printable characters get replaced with their
473  * mime-encoded equivalents.
474  */
475  if ((in[0] < 32)) {
476  if (outlen <= 3) break;
477 
478  snprintf(out, outlen, "=%02X", (unsigned char) in[0]);
479  in++;
480  out += 3;
481  outlen -= 3;
482  len += 3;
483  continue;
484 
485  } else if (strchr(disallowed, *in) != NULL) {
486  if (outlen <= 2) break;
487 
488  out[0] = '\\';
489  out[1] = *in;
490  in++;
491  out += 2;
492  outlen -= 2;
493  len += 2;
494  continue;
495  }
496 
497  /*
498  * Only one byte left.
499  */
500  if (outlen <= 1) {
501  break;
502  }
503 
504  /*
505  * Allowed character.
506  */
507  *out = *in;
508  out++;
509  in++;
510  outlen--;
511  len++;
512  }
513  *out = '\0';
514  return len;
515 }

525 static ssize_t xlat_config(UNUSED TALLOC_CTX *ctx, char **out, size_t outlen,

579 /*
580  * Do chroot, if requested.
581  *
582  * Switch UID and GID to what is specified in the config file
583  */
584 static int switch_users(main_config_t *config, CONF_SECTION *cs)
585 {
586  bool do_suid = false;
587  bool do_sgid = false;
588 
589  /*
590  * Get the current maximum for core files. Do this
591  * before anything else so as to ensure it's properly
592  * initialized.
593  */
594  if (fr_set_dumpable_init() < 0) {
595  fr_perror("%s", config->name);
596  return -1;
597  }
598 
599  if (cf_section_rules_push(cs, switch_users_config) < 0) {
600  fprintf(stderr, "%s: Error: Failed pushing parse rules for user/group information.\n",
601  config->name);
602  return -1;
603  }
604 
605  DEBUG("Parsing security rules to bootstrap UID / GID / chroot / etc.");
606  if (cf_section_parse(config, config, cs) < 0) {
607  fprintf(stderr, "%s: Error: Failed to parse user/group information.\n",
608  config->name);
609  return -1;
610  }
611 
612  /*
613  * Don't do chroot/setuid/setgid if we're in debugging
614  * as non-root.
615  */
616  if (DEBUG_ENABLED && (getuid() != 0)) {
617  WARN("Ignoring configured UID / GID / chroot as we're running in debug mode");
618  return 0;
619  }
620 #ifdef HAVE_GRP_H
621  /*
622  * Get the correct GID for the server.
623  */
624  config->server_gid = getgid();
625  if (config->gid_is_set && (config->server_gid != config->gid)) {
626  config->server_gid = config->gid;
627  do_sgid = true;
628  }
629 #endif
630 
631  /*
632  * Get the correct UID for the server.
633  */
634  config->server_uid = getuid();
635  if (config->uid_is_set && (config->server_uid != config->uid)) {
636  /*
637  * We're not the correct user. Go set that.
638  */
639  config->server_uid = config->uid;
640  do_suid = true;
641 
642 #ifdef HAVE_INITGROUPS
643  {
644  struct passwd *user;
645 
646  if (rad_getpwuid(config, &user, config->uid) < 0) {
647  fprintf(stderr, "%s: Failed resolving UID %i: %s\n",
648  config->name, (int)config->uid, fr_syserror(errno));
649  return -1;
650  }
651 
652  if (initgroups(user->pw_name, config->server_gid) < 0) {
653  fprintf(stderr, "%s: Cannot initialize supplementary group list "
654  "for user %s: %s\n",
655  config->name, user->pw_name, fr_syserror(errno));
656  talloc_free(user);
657  return -1;
658  }
659 
660  talloc_free(user);
661  }
662 #endif
663  }
664 
665  /*
666  * Set the user/group we're going to use
667  * to check read permissions on configuration files.
668  */
669  cf_file_check_user(config->server_uid ? config->server_uid : (uid_t)-1,
670  config->server_gid ? config->server_gid : (gid_t)-1);
671 
672  /*
673  * Do chroot BEFORE changing UIDs.
674  */
675  if (config->chroot_dir) {
676  if (chroot(config->chroot_dir) < 0) {
677  fprintf(stderr, "%s: Failed to perform chroot %s: %s",
678  config->name, config->chroot_dir, fr_syserror(errno));
679  return -1;
680  }
681 
682  /*
683  * Note that we leave chdir alone. It may be
684  * OUTSIDE of the root. This allows us to read
685  * the configuration from "-d ./etc/raddb", with
686  * the chroot as "./chroot/" for example. After
687  * the server has been loaded, it does a "cd
688  * ${logdir}" below, so that core files (if any)
689  * go to a logging directory.
690  *
691  * This also allows the configuration of the
692  * server to be outside of the chroot. If the
693  * server is statically linked, then the only
694  * things needed inside of the chroot are the
695  * logging directories.
696  */
697  }
698 
699 #ifdef HAVE_GRP_H
700  /*
701  * Set the GID. Don't bother checking it.
702  */
703  if (do_sgid) {
704  if (setgid(config->server_gid) < 0) {
705  struct group *group;
706 
707  if (rad_getgrgid(config, &group, config->gid) < 0) {
708  fprintf(stderr, "%s: Failed resolving GID %i: %s\n",
709  config->name, (int)config->gid, fr_syserror(errno));
710  return -1;
711  }
712 
713  fprintf(stderr, "%s: Failed setting group to %s: %s",
714  config->name, group->gr_name, fr_syserror(errno));
715  return -1;
716  }
717  }
718 #endif
719 
720  /*
721  * The directories for PID files and logs must exist. We
722  * need to create them if we're told to write files to
723  * those directories.
724  *
725  * Because this creation is new in 3.0.9, it's a soft
726  * fail.
727  *
728  */
729  if (config->write_pid) {
730  /*
731  * Control sockets may be accessible by users
732  * other than the freeradius user, so we need
733  * to allow 'other' to traverse the run
734  * directory.
735  *
736  * The freeradius user should be the only one
737  * allowed to write to this directory however.
738  */
739  if (fr_mkdir(NULL, config->run_dir, -1, 0755, mkdir_chown, config) < 0) {
740  WARN("Failed creating run_dir %s: %s", config->run_dir, fr_syserror(errno));
741  }
742  }
743 
744  if ((default_log.dst == L_DST_FILES) && config->log_dir) {
745  /*
746  * Every other Linux daemon allows 'other'
747  * to traverse the log directory. That doesn't
748  * mean the actual files should be world
749  * readable.
750  */
751  if (fr_mkdir(NULL, config->log_dir, -1, 0755, mkdir_chown, config) < 0) {
752  WARN("Failed creating log_dir %s: %s", config->log_dir, fr_syserror(errno));
753  }
754  }
755 
756  /*
757  * If we don't already have a log file open, open one
758  * now. We may not have been logging anything yet. The
759  * server normally starts up fairly quietly.
760  */
761  if ((default_log.dst == L_DST_FILES) &&
762  (default_log.fd < 0)) {
763  default_log.fd = open(config->log_file,
764  O_WRONLY | O_APPEND | O_CREAT, 0640);
765  if (default_log.fd < 0) {
766  fprintf(stderr, "%s: Failed to open log file %s: %s\n",
767  config->name, config->log_file, fr_syserror(errno));
768  return -1;
769  }
770  }
771 
772  /*
773  * If we need to change UID, ensure that the log files
774  * have the correct owner && group.
775  *
776  * We have to do this because some log files MAY already
777  * have been written as root. We need to change them to
778  * have the correct ownership before proceeding.
779  */
780  if ((do_suid || do_sgid) &&
781  (default_log.dst == L_DST_FILES)) {
782  if (fchown(default_log.fd, config->server_uid, config->server_gid) < 0) {
783  fprintf(stderr, "%s: Cannot change ownership of log file %s: %s\n",
784  config->name, config->log_file, fr_syserror(errno));
785  return -1;
786  }
787  }
788 
789  /*
790  * Once we're done with all of the privileged work,
791  * permanently change the UID.
792  */
793  if (do_suid) {
794  rad_suid_set_down_uid(config->server_uid);
795  rad_suid_down();
796  }
797 
798  /*
799  * This also clears the dumpable flag if core dumps
800  * aren't allowed.
801  */
802  if (fr_set_dumpable(config->allow_core_dumps) < 0) PERROR("Failed enabling core dumps");
803 
804  if (config->allow_core_dumps) INFO("Core dumps are enabled");
805 
806  return 0;
807 }
808 #endif /* HAVE_SETUID */

894  *
895  * This function can ONLY be called from the main server process.

897 int main_config_init(main_config_t *config)

902  bool can_colourise = false;
903  char buffer[1024];

913  "Refusing to start due to insecure configuration", config->raddb_dir);

920  ERROR("Configuration directory %s is globally readable. "
921  "Refusing to start due to insecure configuration", config->raddb_dir);

927  /*
928  * About sizeof(request_t) + sizeof(fr_radius_packet_t) * 2 + sizeof(fr_pair_t) * 400

930  * Which should be enough for many configurations.

940 do {\

947  break;\
948  default:\
949  break;\
950  }\

952 
953  /*
954  * It's OK if this one doesn't exist.
955  */
956  DICT_READ_OPTIONAL(config->raddb_dir, FR_DICTIONARY_FILE);
957 
958  cs = cf_section_alloc(NULL, NULL, "main", NULL);
959  if (!cs) return -1;

963  * we're debugging, colourise things. This flag also
964  * removes the "Debug : " prefix from the log messages.
965  */
966  p = getenv("TERM");
967  if (p && isatty(default_log.fd) && strstr(p, "xterm") && fr_debug_lvl) {
968  can_colourise = default_log.colourise = true;
969  } else {
970  can_colourise = default_log.colourise = false;
971  }
972  default_log.line_number = config->log_line_number;

976  * We check if it's defined first, as the user may
977  * have defined their own feature flags, or want
978  * to manually override the ones set by modules
979  * or the server.
980  */
981  subcs = cf_section_find(cs, "feature", NULL);
982  if (!subcs) {
983  subcs = cf_section_alloc(cs, cs, "feature", NULL);
984  if (!subcs) {
985  failure:
986  talloc_free(cs);
987  return -1;
988  }
989  }
990  dependency_features_init(subcs);
991 
992  /*
993  * Add a 'version' subsection off the main config
994  * We check if it's defined first, this is for
995  * backwards compatibility.
996  */
997  subcs = cf_section_find(cs, "version", NULL);
998  if (!subcs) {
999  subcs = cf_section_alloc(cs, cs, "version", NULL);
1000  if (!subcs) goto failure;
1001  }
1002  dependency_version_numbers_init(subcs);
1003 
1004  /*
1005  * @todo - not quite done yet... these dictionaries have
1006  * to be loaded from raddb_dir. But the
1007  * fr_dict_autoload_t has a base_dir pointer
1008  * there... it's probably best to pass raddb_dir into
1009  * fr_dict_autoload() and have it use that instead.
1010  *
1011  * Once that's done, the proto_foo dictionaries SHOULD be
1012  * autoloaded, AND loaded before the configuration files
1013  * are read.
1014  *
1015  * And then all of the modules have to be updated to use
1016  * their local dict pointer, instead of NULL.
1017  */
1018  if (cf_section_rules_push(cs, lib_dir_on_read_config) < 0) goto failure;
1019  if (cf_section_rules_push(cs, virtual_servers_on_read_config) < 0) goto failure;
1020 
1021  /* Read the configuration file */
1022  snprintf(buffer, sizeof(buffer), "%.200s/%.50s.conf", config->raddb_dir, config->name);
1023  if (cf_file_read(cs, buffer) < 0) {
1024  ERROR("Error reading or parsing %s", buffer);
1025  goto failure;
1026  }
1027 
1028  /*
1029  * Do any fixups here that might be used in references
1030  */
1031  if (config->name) {
1032  CONF_PAIR *cp;
1033 
1034  cp = cf_pair_find(cs, "name");
1035  if (cp){
1036  if (config->overwrite_config_name && (cf_pair_replace(cs, cp, config->name) < 0)) {
1037  ERROR("Failed adding/replacing \"name\" config item");
1038  goto failure;
1039  }
1040  } else {
1041  MEM(cp = cf_pair_alloc(cs, "name", config->name, T_OP_EQ, T_BARE_WORD, T_DOUBLE_QUOTED_STRING));
1042  cf_pair_add(cs, cp);
1043  }
1044  }
1045 
1046  if (cf_section_pass2(cs) < 0) goto failure;
1047

1049  * Parse environment variables first.
1050  */
1051  subcs = cf_section_find(cs, "ENV", NULL);
1052  if (subcs) {
1053  char const *attr, *value;
1054  CONF_PAIR *cp;
1055  CONF_ITEM *ci;
1056 
1057  for (ci = cf_item_next(subcs, NULL);
1058  ci != NULL;
1059  ci = cf_item_next(subcs, ci)) {
1060  if (cf_item_is_data(ci)) continue;
1061 
1062  if (!cf_item_is_pair(ci)) {
1063  cf_log_err(ci, "Unexpected item in ENV section");
1064  goto failure;
1065  }
1066 
1067  cp = cf_item_to_pair(ci);
1068  if (cf_pair_operator(cp) != T_OP_EQ) {
1069  cf_log_err(ci, "Invalid operator for item in ENV section");
1070  goto failure;
1071  }
1072 
1073  attr = cf_pair_attr(cp);
1074  value = cf_pair_value(cp);
1075  if (!value) {
1076  if (unsetenv(attr) < 0) {
1077  cf_log_err(ci, "Failed deleting environment variable %s: %s",
1078  attr, fr_syserror(errno));
1079  goto failure;
1080  }
1081  } else {
1082  void *handle;
1083  void **handle_p;
1084 
1085  if (setenv(attr, value, 1) < 0) {
1086  cf_log_err(ci, "Failed setting environment variable %s: %s",
1087  attr, fr_syserror(errno));
1088  goto failure;
1089  }
1090 
1091  /*
1092  * Hacks for LD_PRELOAD.
1093  */
1094  if (strcmp(attr, "LD_PRELOAD") != 0) continue;
1095 
1096  handle = dlopen(value, RTLD_NOW | RTLD_GLOBAL);
1097  if (!handle) {
1098  cf_log_err(ci, "Failed loading library %s: %s", value, dlerror());
1099  goto failure;
1100  }
1101 
1102  /*
1103  * Wrap the pointer, so we can set a destructor.
1104  */
1105  MEM(handle_p = talloc(NULL, void *));
1106  *handle_p = handle;
1107  talloc_set_destructor(handle_p, _dlhandle_free);
1108  (void) cf_data_add(subcs, handle_p, value, true);
1109  }
1110  } /* loop over pairs in ENV */
1111  } /* there's an ENV subsection */
1112 
1113  /*
1114  * If there was no log destination set on the command line,
1115  * set it now.
1116  */
1117  if (default_log.dst == L_DST_NULL) {
1118  if (cf_section_rules_push(cs, initial_logging_config) < 0) {
1119  fprintf(stderr, "%s: Error: Failed pushing rules for log {} section.\n",
1120  config->name);
1121  goto failure;
1122  }
1123 
1124  DEBUG("Parsing initial logging configuration.");
1125  if (cf_section_parse(config, config, cs) < 0) {
1126  fprintf(stderr, "%s: Error: Failed to parse log{} section.\n",
1127  config->name);
1128  goto failure;
1129  }
1130

1132  fprintf(stderr, "%s: Error: No log destination specified.\n",
1133  config->name);
1134  goto failure;
1135  }
1136 
1137  default_log.dst = fr_table_value_by_str(log_str2dst, config->log_dest, L_DST_NUM_DEST);
1138 
1139  switch (default_log.dst) {
1140  case L_DST_NUM_DEST:
1141  fprintf(stderr, "%s: Error: Unknown log_destination %s\n",
1142  config->name, config->log_dest);
1143  goto failure;
1144

1146  case L_DST_SYSLOG:
1147  /*
1148  * Call openlog only once, when the
1149  * program starts.
1150  */
1151  openlog(config->name, LOG_PID, config->syslog_facility);
1152  break;
1153 #endif
1154 
1155  case L_DST_FILES:
1156  if (!config->log_file) {
1157  fprintf(stderr, "%s: Error: Specified \"files\" as a log destination, but no log filename was given!\n",
1158  config->name);
1159  goto failure;
1160  }
1161  break;
1162 
1163  default:
1164  break;
1165  }
1166  }
1167 
1168  /*
1169  * Only set timestamp logging from the config file if no value was
1170  * specified on the command line.
1171  */
1172  if (config->log_timestamp_is_set && (default_log.timestamp == L_TIMESTAMP_AUTO)) {
1173  default_log.timestamp = config->log_timestamp ? L_TIMESTAMP_ON : L_TIMESTAMP_OFF;
1174  }
1175 
1176 #ifdef HAVE_SETUID
1177  /*
1178  * Switch users as early as possible.
1179  */
1180  if (switch_users(config, cs) < 0) goto failure;
1181 #endif
1182 
1183  /*
1184  * This allows us to figure out where, relative to
1185  * radiusd.conf, the other configuration files exist.
1186  */
1187  if (cf_section_rules_push(cs, server_config) < 0) goto failure;
1188  if (cf_section_rules_push(cs, virtual_servers_config) < 0) goto failure;
1189 
1190  DEBUG("Parsing main configuration.");
1191  if (cf_section_parse(config, config, cs) < 0) goto failure;
1192 
1193  /*
1194  * Reset the colourisation state. The configuration
1195  * files can disable colourisation if the terminal
1196  * supports it. The configation files *cannot* enable
1197  * colourisation if the terminal window doesn't support
1198  * it.
1199  */
1200  if (can_colourise && !config->do_colourise) {
1201  default_log.colourise = false;
1202  }
1203 
1204  /*
1205  * Starting the server, WITHOUT "-x" on the
1206  * command-line: use whatever is in the config
1207  * file.
1208  */
1209  if (fr_debug_lvl == 0) fr_debug_lvl = config->debug_level;
1210 
1211  INFO("Switching to configured log settings");
1212 
1213  /*
1214  * Free the old configuration items, and replace them
1215  * with the new ones.
1216  *
1217  * Note that where possible, we do atomic switch-overs,
1218  * to ensure that the pointers are always valid.
1219  */
1220  fr_assert(config->root_cs == NULL);
1221 
1222  DEBUG2("%s: #### Loading Clients ####", config->name);
1223  if (!client_list_parse_section(cs, 0, false)) goto failure;
1224 
1225  /*
1226  * Register the %{config:section.subsection} xlat function.
1227  */
1228  xlat_register_legacy(NULL, "config", xlat_config, NULL, NULL, 0, XLAT_DEFAULT_BUF_LEN);
1229 
1230  /*
1231  * Ensure cwd is inside the chroot.
1232  */
