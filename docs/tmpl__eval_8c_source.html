150 /** Resolve a #request_ref_t to a #request_t.
151  *
152  * Sometimes #request_t structs may be chained to each other, as is the case
153  * when internally proxying EAP. This function resolves a #request_ref_t
154  * to a #request_t higher in the chain than the current #request_t.
155  *
156  * @see radius_list
157  * @param[in,out] context #request_t to start resolving from, and where to write
158  * a pointer to the resolved #request_t back to.
159  * @param[in] name (request) to resolve.
160  * @return
161  * - 0 if request is valid in this context.
162  * - -1 if request is not valid in this context.
163  */
164 int radius_request(request_t **context, request_ref_t name)
165 {
166  request_t *request = *context;
167 
168  switch (name) {
169  case REQUEST_CURRENT:
170  return 0;
171 
172  case REQUEST_PARENT: /* Navigate up one level */
173  if (!request->parent) return -1;
174  *context = request->parent;
175  break;
176 
177  case REQUEST_OUTER: /* Navigate to the outermost request */
178  if (!request->parent) return -1;
179  while (request->parent) request = request->parent;
180  *context = request;
181  break;
182 
183  case REQUEST_UNKNOWN:
184  default:
185  fr_assert(0);
186  return -1;
187  }
188 
189  return 0;
190 }

237 /** Expand a #tmpl_t to a string writing the result to a buffer
238  *
239  * The intended use of #tmpl_expand and #tmpl_aexpand is for modules to easily convert a #tmpl_t
240  * provided by the conf parser, into a usable value.
241  * The value returned should be raw and undoctored for #FR_TYPE_STRING and #FR_TYPE_OCTETS types,
242  * and the printable (string) version of the data for all others.
243  *
244  * Depending what arguments are passed, either copies the value to buff, or writes a pointer
245  * to a string buffer to out. This allows the most efficient access to the value resolved by
246  * the #tmpl_t, avoiding unecessary string copies.
247  *
248  * @note This function is used where raw string values are needed, which may mean the string
249  * returned may be binary data or contain unprintable chars. #fr_snprint or #fr_asprint
250  * should be used before using these values in debug statements. #is_printable can be used to
251  * check if the string only contains printable chars.
252  *
253  * @param[out] out Where to write a pointer to the string buffer. On return may
254  * point to buff if buff was used to store the value. Otherwise will
255  * point to a #fr_value_box_t buffer, or the name of the template.
256  * Must not be NULL.
257  * @param[out] buff Expansion buffer, may be NULL except for the following types:
258  * - #TMPL_TYPE_EXEC
259  * - #TMPL_TYPE_XLAT
260  * @param[in] bufflen Length of expansion buffer. Must be >= 2.
261  * @param[in] request Current request.
262  * @param[in] vpt to expand. Must be one of the following types:
263  * - #TMPL_TYPE_UNRESOLVED
264  * - #TMPL_TYPE_EXEC
265  * - #TMPL_TYPE_XLAT
266  * - #TMPL_TYPE_ATTR
267  * - #TMPL_TYPE_DATA
268  * @param[in] escape xlat escape function (only used for xlat types).
269  * @param[in] escape_ctx xlat escape function data.
270  * @param dst_type FR_TYPE_* matching out pointer. @see tmpl_expand.
271  * @return
272  * - -1 on failure.
273  * - The length of data written to buff, or pointed to by out.
274  */
275 ssize_t _tmpl_to_type(void *out,
276  uint8_t *buff, size_t bufflen,
277  request_t *request,
278  tmpl_t const *vpt,
279  xlat_escape_legacy_t escape, void const *escape_ctx,
280  fr_type_t dst_type)
281 {
282  fr_value_box_t value_to_cast;
283  fr_value_box_t value_from_cast = { .type = FR_TYPE_INVALID };
284  fr_value_box_t const *to_cast = &value_to_cast;
285  fr_value_box_t const *from_cast = &value_from_cast;
286 
287  fr_pair_t *vp = NULL;
288 
289  fr_type_t src_type = FR_TYPE_INVALID;
290 
291  ssize_t slen = -1; /* quiet compiler */
292 
293  TMPL_VERIFY(vpt);
294 
295  fr_assert(!tmpl_is_list(vpt));
296  fr_assert(!buff || (bufflen >= 2));
297 
298  switch (vpt->type) {
299  case TMPL_TYPE_UNRESOLVED:
300  RDEBUG4("EXPAND TMPL UNRESOLVED");
301  fr_value_box_bstrndup_shallow(&value_to_cast, NULL, vpt->name, vpt->len, false);
302  src_type = FR_TYPE_STRING;
303  break;
304 
305  case TMPL_TYPE_EXEC:
306  {
307  RDEBUG4("EXPAND TMPL EXEC");
308  if (!buff) {
309  fr_strerror_printf("Missing expansion buffer for EXEC");
310  return -1;
311  }
312 
313  if (radius_exec_program(request, (char *)buff, bufflen, NULL, request, vpt->name, NULL,
314  true, false, fr_time_delta_from_sec(EXEC_TIMEOUT)) != 0) return -1;
315  fr_value_box_strdup_shallow(&value_to_cast, NULL, (char *)buff, true);
316  src_type = FR_TYPE_STRING;
317  }
318  break;
319 
320  case TMPL_TYPE_XLAT:
321  {
322  size_t len;
323 
324  RDEBUG4("EXPAND TMPL XLAT PARSED");
325  RDEBUG2("EXPAND %s", vpt->name); /* xlat_struct doesn't do this */
326  if (!buff) {
327  fr_strerror_printf("Missing expansion buffer for XLAT_STRUCT");
328  return -1;
329  }
330  /* Error in expansion, this is distinct from zero length expansion */
331  slen = xlat_eval_compiled((char *)buff, bufflen, request, tmpl_xlat(vpt), escape, escape_ctx);
332  if (slen < 0) return slen;
333 
334  RDEBUG2(" --> %s", (char *)buff); /* Print pre-unescaping (so it's escaped) */
335 
336  /*
337  * Undo any of the escaping that was done by the
338  * xlat expansion function.
339  *
340  * @fixme We need a way of signalling xlat not to escape things.
341  */
342  len = fr_value_str_unescape(&FR_SBUFF_IN((char *)buff, slen),
343  &FR_SBUFF_IN((char *)buff, slen), SIZE_MAX, '"');
344  fr_assert(buff);
345  fr_value_box_bstrndup_shallow(&value_to_cast, NULL, (char *)buff, len, true);
346  src_type = FR_TYPE_STRING;
347  }
348  break;
349 
350  case TMPL_TYPE_ATTR:
351  {
352  int ret;
353 
354  RDEBUG4("EXPAND TMPL ATTR");
355  ret = tmpl_find_vp(&vp, request, vpt);
356  if (ret < 0) return -2;
357 
358  to_cast = &vp->data;
359  src_type = vp->da->type;
360  }
361  break;
362 
363  case TMPL_TYPE_DATA:
364  RDEBUG4("EXPAND TMPL DATA");
365  to_cast = tmpl_value(vpt);
366  src_type = tmpl_value_type(vpt);
367  break;
368 
369  /*
370  * We should never be expanding these.
371  */
372  case TMPL_TYPE_UNINITIALISED:
373  case TMPL_TYPE_NULL:
374  case TMPL_TYPE_LIST:
375  case TMPL_TYPE_EXEC_UNRESOLVED:
376  case TMPL_TYPE_ATTR_UNRESOLVED:
377  case TMPL_TYPE_XLAT_UNRESOLVED:
378  case TMPL_TYPE_REGEX:
379  case TMPL_TYPE_REGEX_UNCOMPILED:
380  case TMPL_TYPE_REGEX_XLAT:
381  case TMPL_TYPE_REGEX_XLAT_UNRESOLVED:
382  case TMPL_TYPE_MAX:
383  fr_assert(0);
384  return -1;
385  }
386 
387  /*
388  * Deal with casts.
389  */
390  switch (src_type) {
391  case FR_TYPE_STRING:
392  switch (dst_type) {
393  case FR_TYPE_STRING:
394  case FR_TYPE_OCTETS:
395  from_cast = to_cast;
396  break;
397 
398  default:
399  break;
400  }
401  break;
402 
403  case FR_TYPE_OCTETS:
404  switch (dst_type) {
405  /*
406  * Need to use the expansion buffer for this conversion as
407  * we need to add a \0 terminator.
408  */
409  case FR_TYPE_STRING:
410  if (!buff) {
411  fr_strerror_printf("Missing expansion buffer for octet->string cast");
412  return -1;
413  }
414  if (bufflen <= to_cast->vb_length) {
415  fr_strerror_printf("Expansion buffer too small. "
416  "Have %zu bytes, need %zu bytes", bufflen,
417  to_cast->vb_length + 1);
418  return -1;
419  }
420  memcpy(buff, to_cast->vb_octets, to_cast->vb_length);
421  buff[to_cast->vb_length] = '\0';
422 
423  fr_value_box_bstrndup_shallow(&value_from_cast, NULL,
424  (char *)buff, to_cast->vb_length, true);
425  break;
426 
427  /*
428  * Just copy the pointer. Length does not include \0.
429  */
430  case FR_TYPE_OCTETS:
431  from_cast = to_cast;
432  break;
433 
434  default:
435  break;
436  }
437  break;
438 
439  default:
440  {
441  int ret;
442  TALLOC_CTX *ctx;
443 
444  /*
445  * Same type, just set from_cast to to_cast and copy the value.
446  */
447  if (src_type == dst_type) {
448  from_cast = to_cast;
449  break;
450  }
451 
452  MEM(ctx = talloc_new(request));
453 
454  from_cast = &value_from_cast;
455 
456  /*
457  * Data type conversion...
458  */
459  ret = fr_value_box_cast(ctx, &value_from_cast, dst_type, NULL, to_cast);
460  if (ret < 0) goto error;
461 
462 
463  /*
464  * For the dynamic types we need to copy the output
465  * to the buffer. Really we need a version of fr_value_box_cast
466  * that works with buffers, but it's not a high priority...
467  */
468  switch (dst_type) {
469  case FR_TYPE_STRING:
470  if (!buff) {
471  fr_strerror_printf("Missing expansion buffer to store cast output");
472  error:
473  talloc_free(ctx);
474  return -1;
475  }
476  if (from_cast->vb_length >= bufflen) {
477  fr_strerror_printf("Expansion buffer too small. "
478  "Have %zu bytes, need %zu bytes", bufflen,
479  from_cast->vb_length + 1);
480  goto error;
481  }
482  memcpy(buff, from_cast->vb_strvalue, from_cast->vb_length);
483  buff[from_cast->vb_length] = '\0';
484 
485  fr_value_box_bstrndup_shallow(&value_from_cast, NULL,
486  (char *)buff, from_cast->vb_length, from_cast->tainted);
487  break;
488 
489  case FR_TYPE_OCTETS:
490  if (!buff) {
491  fr_strerror_printf("Missing expansion buffer to store cast output");
492  goto error;
493  }
494  if (from_cast->vb_length > bufflen) {
495  fr_strerror_printf("Expansion buffer too small. "
496  "Have %zu bytes, need %zu bytes", bufflen, from_cast->vb_length);
497  goto error;
498  }
499  memcpy(buff, from_cast->vb_octets, from_cast->vb_length);
500  fr_value_box_memdup_shallow(&value_from_cast, NULL,
501  buff, from_cast->vb_length, from_cast->tainted);
502  break;
503 
504  default:
505  break;
506  }
507 
508  talloc_free(ctx); /* Free any dynamically allocated memory from the cast */
509  }
510  }
511 
512  RDEBUG4("Copying %zu bytes to %p from offset %zu",
513  fr_value_box_field_sizes[dst_type], *((void **)out), fr_value_box_offsets[dst_type]);
514 
515  memcpy(out, ((uint8_t const *) from_cast) + fr_value_box_offsets[dst_type], fr_value_box_field_sizes[dst_type]);
516 
517  return from_cast->vb_length;
518 }

521  *

540  * @param request Current request.
541  * @param vpt to expand. Must be one of the following types:
542  * - #TMPL_TYPE_UNRESOLVED

549  * @param dst_type FR_TYPE_* matching out pointer. @see tmpl_aexpand.

566

627  }

635

641  if (slen < 0) goto error;

778 /** Traverse a TLV attribute
779  *
780  * @param[in,out] prev The previous pair in the list.
781  * @param[in] current The pair to evaluate.
782  * @param[in] ns Tracks tree position between cursor calls.
783  * @return the number of attributes matching ar.
784  */
785 static fr_pair_t *_tmpl_cursor_tlv_eval(fr_pair_t **prev, UNUSED fr_pair_t *current, tmpl_cursor_nested_t *ns)
786 {
787  fr_cursor_stack_t *cs;
788  fr_cursor_t *cursor;
789  fr_pair_t *vp;
790 
791  cs = ns->tlv.cursor_stack;
792  cursor = &cs->cursor[cs->depth - 1];
793 
794  /*
795  * Our final stopping condition is when the shallowest
796  * cursor returns NULL. That means we've evaluated
797  * the entire subtree.
798  */
799  while (cs->depth > 0) {
800  for (vp = fr_cursor_current(cursor);
801  vp;
802  vp = fr_cursor_next(cursor)) {
803  *prev = cursor->prev;
804 
805  /*
806  * Exact match, we're done.
807  */
808  if (fr_dict_attr_cmp(vp->da, ns->ar->ar_da) == 0) {
809  fr_cursor_next(cursor); /* Advance to correct position for next call */
810  return vp;
811  }
812 
813  /*
814  * Traverse the intermediary VP
815  */
816  if ((vp->da->depth < ns->ar->ar_da->depth) &&
817  (fr_dict_attr_cmp(ns->tlv.da_stack.da[vp->da->depth], vp->da) == 0)) {
818  cursor = &cs->cursor[cs->depth++];
819  fr_cursor_init(cursor, &vp->vp_group);
820  continue;
821  }
822  }
823 
824  /*
825  * We're done processing children at this level.
826  *
827  * Jump up one in the cursor stack and continue.
828  */
829  cursor = &cs->cursor[--cs->depth];
830  }
831 
832  return NULL;
833 }

1005 /** Evaluates, then, sometimes, pops evaluation contexts from the tmpl stack
1006  *
1007  * To pop or not to pop is determined by whether evaluating the context again
1008  * would/should/could produce another fr_pair_t.
1009  */
1010 static inline CC_HINT(always_inline)
1011 fr_pair_t *_tmpl_cursor_eval(fr_pair_t **prev, fr_pair_t *curr, tmpl_cursor_ctx_t *cc)
1012 {
1013  tmpl_attr_t const *ar;
1014  tmpl_cursor_nested_t *ns;
1015  fr_pair_t *iter = curr, *vp;
1016 
1017  ns = fr_dlist_tail(&cc->nested);
1018  ar = ns->ar;
1019 
1020  if (ar) switch (ar->ar_num) {
1021  /*
1022  * Get the first instance
1023  */
1024  case NUM_ANY:
1025  vp = ns->func(prev, curr, ns);
1026  _tmpl_cursor_eval_pop(cc);
1027  break;
1028 
1029  /*
1030  * Get all instances
1031  */
1032  case NUM_ALL:
1033  case NUM_COUNT:
1034  all_inst:
1035  vp = ns->func(prev, curr, ns);
1036  if (!vp) _tmpl_cursor_eval_pop(cc); /* pop only when we're done */
1037  break;
1038 
1039  /*
1040  * Get the last instance
1041  */
1042  case NUM_LAST:
1043  vp = NULL;
1044  while ((iter = ns->func(prev, iter, ns))) {
1045  vp = iter;
1046 
1047  if (!vp->next) break;
1048 
1049  iter = vp->next;
1050  *prev = vp;
1051  }
1052  _tmpl_cursor_eval_pop(cc);
1053  break;
1054 
1055  /*
1056  * Get the n'th instance
1057  */
1058  default:
1059  {
1060  int16_t i = 0;
1061 
1062  for (;;) {
1063  vp = ns->func(prev, iter, ns);
1064  if (!vp) break; /* Prev and next at the correct points */
1065 
1066  if (++i > ar->num) break;
1067 
1068  iter = vp->next;
1069  *prev = vp;
1070  };
1071  _tmpl_cursor_eval_pop(cc);
1072  }
1073  break;
1074  } else goto all_inst; /* Used for TMPL_TYPE_LIST */
1075 
1076  return vp;
1077 }

1101 {
1102  tmpl_cursor_ctx_t *cc = uctx;
1103  tmpl_t const *vpt = cc->vpt;
1104 
1105  fr_pair_t *vp;
1106  fr_pair_t **list_head;
1107 
1108  switch (vpt->type) {
1109  case TMPL_TYPE_ATTR:
1110  {
1111  tmpl_attr_t const *ar = NULL;
1112  tmpl_cursor_nested_t *ns = NULL;
1113 
1114  /*
1115  * - Continue until there are no evaluation contexts
1116  * - Push a evaluation context if evaluating the head of the
1117  * stack yields a VP and we're not at the deepest attribute
1118  * reference.
1119  * - Return if we have a VP and there are no more attribute
1120  * references to push, i.e. we're at the deepest attribute
1121  * reference.
1122  */
1123  while ((ns = fr_dlist_tail(&cc->nested))) {
1124  ar = ns->ar;
1125  vp = _tmpl_cursor_eval((fr_pair_t **)prev, curr, cc);
1126  if (!vp) continue;

1129  if (ar) {

1131  _tmpl_cursor_init(vp, list_head, ar, cc);

1133  *prev = NULL;
1134  continue;
1135  }
1136 
1137  return vp;
1138  }
1139 
1140  null_result:
1141  *prev = curr;
1142  return NULL;
1143  }
1144 
1145  /*
1146  * Hacks for evaluating lists
1147  * Hopefully this tmpl type goes away soon...
1148  */

1151
