254 /** Send a log message to its destination, possibly including fields from the request
255  *
256  * @param[in] type of log message, #L_ERR, #L_WARN, #L_INFO, #L_DBG.
257  * @param[in] lvl Minimum required server or request level to output this message.
258  * @param[in] request The current request.
259  * @param[in] file src file the log message was generated in.
260  * @param[in] line number the log message was generated on.
261  * @param[in] fmt with printf style substitution tokens.
262  * @param[in] ap Substitution arguments.
263  * @param[in] uctx The #fr_log_t specifying the destination for log messages.
264  */
265 void vlog_request(fr_log_type_t type, fr_log_lvl_t lvl, request_t *request,
266  char const *file, int line,
267  char const *fmt, va_list ap, void *uctx)
268 {
269  char const *filename;
270  FILE *fp = NULL;
271 
272  char *p;
273  char const *extra = "";
274  uint8_t unlang_indent, module_indent;
275  va_list aq;
276 
277  char const *fmt_location = "";
278  char const *fmt_prefix = "";
279  char const *fmt_module = "";
280  char const *fmt_exp = "";
281 
282  fr_log_t *log_dst = uctx;
283  TALLOC_CTX *pool;
284 
285  /*
286  * No output means no output.
287  */
288  if (!log_dst) return;
289  if (!log_rdebug_enabled(lvl, request)) return;
290 
291  /*
292  * Allocate a thread local, 4k pool so we don't
293  * need to keep allocating memory on the heap.
294  */
295  pool = fr_vlog_request_pool;
296  if (!pool) {
297  pool = talloc_pool(NULL, 4096);
298  if (!pool) {
299  fr_perror("Failed allocating memory for vlog_request_pool");
300  return;
301  }
302  fr_thread_local_set_destructor(fr_vlog_request_pool, _fr_vlog_request_pool_free, pool);
303  }
304 
305  filename = log_dst->file;
306 
307  /*
308  * Debug messages get treated specially.
309  */
310  if ((type & L_DBG) != 0) {
311  /*
312  * If we're debugging to a file, then use that.
313  *
314  * @todo: have fr_vlog() take a fr_log_t*, so
315  * that we can cache the opened descriptor, and
316  * we don't need to re-open it on every log
317  * message.
318  */
319  switch (log_dst->dst) {
320  case L_DST_FILES:
321  fp = fopen(log_dst->file, "a");
322  if (!fp) goto finish;
323  break;
324 
325 #if defined(HAVE_FOPENCOOKIE) || defined (HAVE_FUNOPEN)
326  case L_DST_EXTRA:
327  {
328 # ifdef HAVE_FOPENCOOKIE
329  cookie_io_functions_t io;
330 
331  /*
332  * These must be set separately as they have different prototypes.
333  */
334  io.read = NULL;
335  io.seek = NULL;
336  io.close = NULL;
337  io.write = log_dst->cookie_write;
338 
339  fp = fopencookie(log_dst->cookie, "w", io);
340 # else
341  fp = funopen(log_dst->cookie, NULL, log_dst->cookie_write, NULL, NULL);
342 
343 # endif
344  if (!fp) goto finish;
345  }
346  break;
347 #endif
348  default:
349  break;
350  }
351  goto print_fmt;
352  }
353 
354  if (filename) {
355  char *exp;
356  log_dst_t *dst;
357 
358  dst = request->log.dst;
359 
360  /*
361  * Prevent infinitely recursive calls if
362  * xlat_aeval attempts to write to the request log.
363  */
364  request->log.dst = NULL;
365 
366  /*
367  * This is SLOW! Doing it for every log message
368  * in every request is NOT recommended!
369  */
370  if (xlat_aeval(request, &exp, request, filename, rad_filename_escape, NULL) < 0) return;
371 
372  /*
373  * Restore the original logging function
374  */
375  request->log.dst = dst;
376 
377  /*
378  * Ensure the directory structure exists, for
379  * where we're going to write the log file.
380  */
381  p = strrchr(exp, FR_DIR_SEP);
382  if (p) {
383  *p = '\0';
384  if (fr_mkdir(NULL, exp, -1, S_IRWXU, NULL, NULL) < 0) {
385  ERROR("Failed creating %s: %s", exp, fr_syserror(errno));
386  talloc_free(exp);
387  return;
388  }
389  *p = FR_DIR_SEP;
390  }
391 
392  fp = fopen(exp, "a");
393  talloc_free(exp);
394  }
395 
396 print_fmt:
397  /*
398  * Request prefix i.e.
399  *
400  * (0) <fmt>
401  */
402  if (request->name) {
403  if ((request->seq_start == 0) || (request->number == request->seq_start)) {
404  fmt_prefix = talloc_typed_asprintf(pool, "(%s) ", request->name);
405  } else {
406  fmt_prefix = talloc_typed_asprintf(pool, "(%s,%" PRIu64 ") ",
407  request->name, request->seq_start);
408  }
409  }
410 
411  /*
412  * Make sure the indent isn't set to something crazy
413  */
414  unlang_indent = request->log.unlang_indent > sizeof(spaces) - 1 ?
415  sizeof(spaces) - 1 :
416  request->log.unlang_indent;
417 
418  module_indent = request->log.module_indent > sizeof(spaces) - 1 ?
419  sizeof(spaces) - 1 :
420  request->log.module_indent;
421 
422  /*
423  * Module name and indentation i.e.
424  *
425  * test - <fmt>
426  */
427  if (request->module) {
428  fmt_module = talloc_typed_asprintf(pool, "%s - %.*s", request->module, module_indent, spaces);
429  }
430 
431  /*
432  * If we don't copy the original ap we get a segfault from vasprintf. This is apparently
433  * due to ap sometimes being implemented with a stack offset which is invalidated if
434  * ap is passed into another function. See here:
435  * http://julipedia.meroh.net/2011/09/using-vacopy-to-safely-pass-ap.html
436  *
437  * I don't buy that explanation, but doing a va_copy here does prevent SEGVs seen when
438  * running unit tests which generate errors under CI.
439  */
440  va_copy(aq, ap);
441  fmt_exp = fr_vasprintf(pool, fmt, aq);
442  va_end(aq);
443 
444  /*
445  * Logging to a file descriptor
446  */
447  if (fp) {
448  char time_buff[64]; /* The current timestamp */
449 
450  time_t timeval;
451  timeval = time(NULL);
452 
453 #if 0
454  fmt_location = talloc_typed_asprintf(pool, "%s[%i]: ", file, line);
455 #endif
456 
457 #ifdef HAVE_GMTIME_R
458  if (log_dates_utc) {
459  struct tm utc;
460  gmtime_r(&timeval, &utc);
461  ASCTIME_R(&utc, time_buff, sizeof(time_buff));
462  } else
463 #endif
464  {
465  CTIME_R(&timeval, time_buff, sizeof(time_buff));
466  }
467 
468  /*
469  * Strip trailing new lines
470  */
471  p = strrchr(time_buff, '\n');
472  if (p) p[0] = '\0';
473 
474  fprintf(fp,
475  "%s" /* location */
476  "%s" /* prefix */
477  "%s : " /* time */
478  "%s" /* facility */
479  "%.*s" /* indent */
480  "%s" /* module */
481  "%s" /* message */
482  "\n",
483  fmt_location,
484  fmt_prefix,
485  time_buff,
486  fr_table_str_by_value(fr_log_levels, type, ""),
487  unlang_indent, spaces,
488  fmt_module,
489  fmt_exp);
490  fclose(fp);
491  goto finish;
492  }
493 
494  /*
495  * Logging everywhere else
496  */
497  if (!DEBUG_ENABLED3) switch (type) {
498  case L_DBG_WARN:
499  extra = "WARNING: ";
500  type = L_DBG_WARN_REQ;
501  break;
502 
503  case L_DBG_ERR:
504  extra = "ERROR: ";
505  type = L_DBG_ERR_REQ;
506  break;
507  default:
508  break;
509  };
510 
511  log_always(log_dst, type, file, line,
512  "%s" /* prefix */
513  "%.*s" /* indent */
514  "%s" /* module */
515  "%s" /* extra */
516  "%s", /* message */
517  fmt_prefix,
518  unlang_indent, spaces,
519  fmt_module,
520  extra,
521  fmt_exp);
522 
523 finish:
524  talloc_free_children(pool);
525 }

608 /** Martial variadic log arguments into a va_list and pass to error logging functions
609  *
610  * This could all be done in a macro, but it turns out some implementations of the
611  * variadic macros do not work at all well if the va_list being written to is further
612  * up the stack (which is required as you still need a function to convert the elipses
613  * into a va_list).
614  *
615  * So, we use this small wrapper function instead, which will hopefully guarantee
616  * consistent behaviour.
617  *
618  * @param[in] type the log category.
619  * @param[in] lvl of debugging this message should be logged at.
620  * @param[in] request The current request.
621  * @param[in] file src file the log message was generated in.
622  * @param[in] line number the log message was generated on.
623  * @param[in] fmt with printf style substitution tokens.
624  * @param[in] ... Substitution arguments.
625  */
626 void log_request_error(fr_log_type_t type, fr_log_lvl_t lvl, request_t *request,
627  char const *file, int line, char const *fmt, ...)
628 {
629  va_list ap;
630  log_dst_t *dst_p;
631 
632  if (!request->log.dst) return;
633 
634  va_start(ap, fmt);
635  for (dst_p = request->log.dst; dst_p; dst_p = dst_p->next) {
636  dst_p->func(type, lvl, request, file, line, fmt, ap, dst_p->uctx);
637  }
638  if ((type == L_ERR) || (type == L_DBG_ERR) || (type == L_DBG_ERR_REQ)) {
639  vlog_module_failure_msg(request, fmt, ap);

641 
642  va_end(ap);

645 /** Drain any outstanding messages from the fr_strerror buffers
646  *
647  * This function drains any messages from fr_strerror buffer adding a prefix (fmt)
648  * to the first message.
649  *
650  * @param[in] type the log category.
651  * @param[in] lvl of debugging this message should be logged at.
652  * @param[in] request The current request.
653  * @param[in] file src file the log message was generated in.
654  * @param[in] line number the log message was generated on.
655  * @param[in] fmt with printf style substitution tokens.
656  * @param[in] ... Substitution arguments.
657  */
658 void log_request_perror(fr_log_type_t type, fr_log_lvl_t lvl, request_t *request,
659  char const *file, int line, char const *fmt, ...)
660 {
661  char const *strerror;
662 
663  if (!request->log.dst) return;
664 
665  /*
666  * No strerror gets us identical behaviour to log_request_error
667  */
668  strerror = fr_strerror_pop();
669  if (!strerror) {
670  va_list ap;
671  log_dst_t *dst_p;
672 
673  if (!fmt) return; /* NOOP */
674 
675  va_start(ap, fmt);
676  for (dst_p = request->log.dst; dst_p; dst_p = dst_p->next) {
677  dst_p->func(type, lvl, request, file, line, fmt, ap, dst_p->uctx);
678  }
679  va_end(ap);
680 
681  return; /* DONE */
682  }
683 
684  /*
685  * Concatenate fmt with fr_strerror()
686  */
687  if (fmt) {
688  va_list ap;
689  char *tmp;
690 
691  va_start(ap, fmt);
692  tmp = fr_vasprintf(request, fmt, ap);
693  va_end(ap);
694 
695  if (!tmp) return;
696 
697  log_request_error(type, lvl, request, file, line, "%s: %s", tmp, strerror);
698  talloc_free(tmp);
699  } else {
700  log_request_error(type, lvl, request, file, line, "%s", strerror);
701  }
702 
703  /*
704  * Only the first message gets the prefix
705  */
706  while ((strerror = fr_strerror_pop())) {
707  log_request_error(type, lvl, request, file, line, "%s", strerror);
708  }
709 }
