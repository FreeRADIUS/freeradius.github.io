203 /** Insert cdefs into the lua environment
204  *
205  * For LuaJIT using the FFI is significantly faster than the Lua interface.
206  * Help people wishing to use the FFI by inserting cdefs for standard functions.
207  *
208  * @param inst Current instance of the fr_lua module.
209  * @param L Lua interpreter.
210  * @return 0 (no arguments).
211  */
212 int fr_lua_util_jit_log_register(rlm_lua_t const *inst, lua_State *L)
213 {
214  char const *search_path;
215  char *lua_str;
216  int ret;
217 
218  search_path = dl_module_search_path();
219  lua_str = talloc_asprintf(NULL, "\
220  ffi = require(\"ffi\")\
221  ffi.cdef [[\
222  void fr_lua_util_jit_log_debug(char const *msg);\
223  void fr_lua_util_jit_log_info(char const *msg);\
224  void fr_lua_util_jit_log_warn(char const *msg);\
225  void fr_lua_util_jit_log_error(char const *msg);\
226  ]]\
227  fr_lua = ffi.load(\"%s%clibfreeradius-lua%s\")\
228  _fr_log = {}\
229  _fr_log.debug = function(msg)\
230  fr_lua.fr_lua_util_jit_log_debug(msg)\
231  end\
232  _fr_log.info = function(msg)\
233  fr_lua.fr_lua_util_jit_log_info(msg)\
234  end\
235  _fr_log.warn = function(msg)\
236  fr_lua.fr_lua_util_jit_log_warn(msg)\
237  end\
238  _fr_log.error = function(msg)\
239  fr_lua.fr_lua_util_jit_log_error(msg)\
240  end\
241  function _ro_log(table) \
242  return setmetatable({}, { \
243  __index = table,\
244  __newindex = function(table, key, value)\
245  _fr_log.warn(\"fr.log.$func() is read-only\")\
246  end, \
247  __metatable = false \
248  }); \
249  end\
250  fr.log = _ro_log(_fr_log)\
251  ", search_path, FR_DIR_SEP, DL_EXTENSION);
252  ret = luaL_dostring(L, lua_str);
253  talloc_free(lua_str);
254  if (ret != 0) {
255  ERROR("Failed setting up FFI: %s",
256  lua_gettop(L) ? lua_tostring(L, -1) : "Unknown error");
257 
258  return -1;
259  }
260 
261  return 0;
262 }

264 /** Register utililiary functions in the lua environment
265  *
266  * @param inst Current instance of the fr_lua module.
267  * @param L Lua interpreter.
268  * @return 0 (no arguments).
269  */
270 int fr_lua_util_log_register(UNUSED rlm_lua_t const *inst, lua_State *L)
271 {
272  /* fr.{} */
273  lua_getglobal(L, "fr");
274  luaL_checktype(L, -1, LUA_TTABLE);
275 
276  /* fr.log.{} */
277  lua_newtable(L);
278  {
279  lua_newtable(L); //__metatable
280  {
281  lua_pushvalue(L, -1);
282  lua_setfield(L, -2, "__index");
283 
284  lua_pushcfunction(L, _util_log_newindex);
285  lua_setfield(L, -2, "__newindex");
286  }
287 
288  lua_pushcfunction(L, _util_log_debug);
289  lua_setfield(L, -2, "debug");
290 
291  lua_pushcfunction(L, _util_log_info);
292  lua_setfield(L, -2, "info");
293 
294  lua_pushcfunction(L, _util_log_warn);
295  lua_setfield(L, -2, "warn");
296 
297  lua_pushcfunction(L, _util_log_error);
298  lua_setfield(L, -2, "error");
299  }
300 
301  lua_setmetatable(L, -2);
302  lua_setfield(L, -2, "log");
303 
304  return 0;
305 }
