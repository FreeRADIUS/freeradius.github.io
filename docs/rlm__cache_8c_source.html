127 /** Free memory associated with a cache entry
128  *
129  * This does not necessarily remove the entry from the cache, cache_expire
130  * should be used for that.
131  *
132  * This function should be called when an entry that is known to have been
133  * retrieved or inserted into a data store successfully, is no longer needed.
134  *
135  * Some drivers (like rlm_cache_rbtree) don't register a free function.
136  * This means that the cache entry never needs to be explicitly freed.
137  *
138  * @param[in] inst Module instance.
139  * @param[in,out] c Cache entry to free.
140  */
141 static void cache_free(rlm_cache_t const *inst, rlm_cache_entry_t **c)
142 {
143  if (!c || !*c || !inst->driver->free) return;
144 
145  inst->driver->free(*c);
146  *c = NULL;
147 }

149 /** Merge a cached entry into a #request_t
150  *
151  * @return
152  * - #RLM_MODULE_OK if no entries were merged.
153  * - #RLM_MODULE_UPDATED if entries were merged.
154  */
155 static rlm_rcode_t cache_merge(rlm_cache_t const *inst, request_t *request, rlm_cache_entry_t *c) CC_HINT(nonnull);
156 static rlm_rcode_t cache_merge(rlm_cache_t const *inst, request_t *request, rlm_cache_entry_t *c)
157 {
158  fr_pair_t *vp;
159  map_t *map;
160  int merged = 0;
161 
162  RDEBUG2("Merging cache entry into request");
163  RINDENT();
164  for (map = c->maps; map; map = map->next) {
165  /*
166  * The only reason that the application of a map entry
167  * can fail, is if the destination list or request
168  * isn't valid. For now we don't consider this fatal
169  * and continue merging the rest of the maps.
170  */
171  if (map_to_request(request, map, map_to_vp, NULL) < 0) {
172  char buffer[1024];
173 
174  map_print(&FR_SBUFF_OUT(buffer, sizeof(buffer)), map);
175  REXDENT();
176  RDEBUG2("Skipping %s", buffer);
177  RINDENT();
178  continue;
179  }
180  merged++;
181  }
182  REXDENT();
183 
184  if (inst->config.stats) {
185  fr_assert(request->packet != NULL);
186  MEM(pair_update_request(&vp, attr_cache_entry_hits) >= 0);
187  vp->vp_uint32 = c->hits;
188  }
189 
190  return merged > 0 ?
191  RLM_MODULE_UPDATED :
192  RLM_MODULE_OK;
193 }

195 /** Find a cached entry.
196  *
197  * @return
198  * - #RLM_MODULE_OK on cache hit.
199  * - #RLM_MODULE_FAIL on failure.
200  * - #RLM_MODULE_NOTFOUND on cache miss.
201  */
202 static unlang_action_t cache_find(rlm_rcode_t *p_result, rlm_cache_entry_t **out,
203  rlm_cache_t const *inst, request_t *request,
204  rlm_cache_handle_t **handle, uint8_t const *key, size_t key_len)
205 {
206  cache_status_t ret;
207 
208  rlm_cache_entry_t *c;
209 
210  *out = NULL;
211 
212  for (;;) {
213  ret = inst->driver->find(&c, &inst->config, inst->driver_inst->dl_inst->data, request, *handle, key, key_len);
214  switch (ret) {
215  case CACHE_RECONNECT:
216  RDEBUG2("Reconnecting...");
217  if (cache_reconnect(handle, inst, request) == 0) continue;
218  RETURN_MODULE_FAIL;
219 
220  case CACHE_OK:
221  break;
222 
223  case CACHE_MISS:
224  RDEBUG2("No cache entry found for \"%pV\"", fr_box_strvalue_len((char const *)key, key_len));
225  RETURN_MODULE_NOTFOUND;
226 
227  default:
228  RETURN_MODULE_FAIL;
229 
230  }
231 
232  break;
233  }
234 
235  /*
236  * Yes, but it expired, OR the "forget all" epoch has
237  * passed. Delete it, and pretend it doesn't exist.
238  */
239  if ((c->expires < fr_time_to_unix_time(request->packet->timestamp)) ||
240  (c->created < fr_unix_time_from_sec(inst->config.epoch))) {
241  RDEBUG2("Found entry for \"%pV\", but it expired %pV seconds ago. Removing it",
242  fr_box_strvalue_len((char const *)key, key_len),
243  fr_box_date(fr_time_to_unix_time(request->packet->timestamp -
244  fr_time_delta_from_sec(c->expires))));
245 
246  inst->driver->expire(&inst->config, inst->driver_inst->dl_inst->data, request, handle, c->key, c->key_len);
247  cache_free(inst, &c);
248  RETURN_MODULE_NOTFOUND; /* Couldn't find a non-expired entry */
249  }
250 
251  RDEBUG2("Found entry for \"%pV\"", fr_box_strvalue_len((char const *)key, key_len));
252 
253  c->hits++;
254  *out = c;
255 
256  RETURN_MODULE_OK;
257 }

259 /** Expire a cache entry (removing it from the datastore)
260  *
261  * @return
262  * - #RLM_MODULE_OK on success.
263  * - #RLM_MODULE_NOTFOUND if no entry existed.
264  * - #RLM_MODULE_FAIL on failure.
265  */
266 static unlang_action_t cache_expire(rlm_rcode_t *p_result,
267  rlm_cache_t const *inst, request_t *request,
268  rlm_cache_handle_t **handle, uint8_t const *key, size_t key_len)
269 {
270  RDEBUG2("Expiring cache entry");
271  for (;;) switch (inst->driver->expire(&inst->config, inst->driver_inst->dl_inst->data, request,
272  *handle, key, key_len)) {
273  case CACHE_RECONNECT:
274  if (cache_reconnect(handle, inst, request) == 0) continue;
275  FALL_THROUGH;
276 
277  default:
278  RETURN_MODULE_FAIL;
279 
280  case CACHE_OK:
281  RETURN_MODULE_OK;
282 
283  case CACHE_MISS:
284  RETURN_MODULE_NOTFOUND;
285  }
286 }

288 /** Create and insert a cache entry
289  *
290  * @return
291  * - #RLM_MODULE_OK on success.
292  * - #RLM_MODULE_UPDATED if we merged the cache entry.
293  * - #RLM_MODULE_FAIL on failure.
294  */
295 static unlang_action_t cache_insert(rlm_rcode_t *p_result,
296  rlm_cache_t const *inst, request_t *request, rlm_cache_handle_t **handle,
297  uint8_t const *key, size_t key_len, int ttl)
298 {
299  map_t const *map;
300  map_t **last, *c_map;
301 
302  fr_pair_t *vp;
303  bool merge = false;
304  rlm_cache_entry_t *c;
305 
306  TALLOC_CTX *pool;
307 
308  if ((inst->config.max_entries > 0) && inst->driver->count &&
309  (inst->driver->count(&inst->config, inst->driver_inst->dl_inst->data, request, handle) > inst->config.max_entries)) {
310  RWDEBUG("Cache is full: %d entries", inst->config.max_entries);
311  RETURN_MODULE_FAIL;
312  }
313 
314  c = cache_alloc(inst, request);
315  if (!c) {
316  RETURN_MODULE_FAIL;
317  }
318  c->key = talloc_memdup(c, key, key_len);
319  c->key_len = key_len;
320 
321  /*
322  * All in NSEC resolution
323  */
324  c->created = c->expires = fr_time_to_unix_time(request->packet->timestamp);
325  c->expires += fr_time_delta_from_sec(ttl);
326 
327  last = &c->maps;
328 
329  RDEBUG2("Creating new cache entry");
330 
331  /*
332  * Alloc a pool so we don't have excessive allocs when
333  * gathering fr_pair_ts to cache.
334  */
335  pool = talloc_pool(NULL, 2048);
336  for (map = inst->maps; map != NULL; map = map->next) {
337  fr_pair_t *to_cache = NULL;
338  fr_cursor_t cursor;
339 
340  fr_assert(map->lhs && map->rhs);
341 
342  /*
343  * Calling map_to_vp gives us exactly the same result,
344  * as if this were an update section.
345  */
346  if (map_to_vp(pool, &to_cache, request, map, NULL) < 0) {
347  RDEBUG2("Skipping %s", map->rhs->name);
348  continue;
349  }
350 
351  for (vp = fr_cursor_init(&cursor, &to_cache);
352  vp;
353  vp = fr_cursor_next(&cursor)) {
354  /*
355  * Prevent people from accidentally caching
356  * cache control attributes.
357  */
358  if (tmpl_is_list(map->rhs)) switch (vp->da->attr) {
359  case FR_CACHE_TTL:
360  case FR_CACHE_STATUS_ONLY:
361  case FR_CACHE_MERGE_NEW:
362  case FR_CACHE_ENTRY_HITS:
363  RDEBUG2("Skipping %s", vp->da->name);
364  continue;
365 
366  default:
367  break;
368  }
369 
370  RINDENT();
371  if (RDEBUG_ENABLED2) map_debug_log(request, map, vp);
372  REXDENT();
373 
374  MEM(c_map = talloc_zero(c, map_t));
375  c_map->op = map->op;
376 
377  /*
378  * Now we turn the fr_pair_ts into maps.
379  */
380  switch (map->lhs->type) {
381  /*
382  * Attributes are easy, reuse the LHS, and create a new
383  * RHS with the fr_value_box_t from the fr_pair_t.
384  */
385  case TMPL_TYPE_ATTR:
386  {
387  fr_token_t quote;
388  c_map->lhs = map->lhs; /* lhs shouldn't be touched, so this is ok */
389  do_rhs:
390  if (vp->vp_type == FR_TYPE_STRING) {
391  quote = is_printable(vp->vp_strvalue, vp->vp_length) ?
392  T_SINGLE_QUOTED_STRING : T_DOUBLE_QUOTED_STRING;
393  } else {
394  quote = T_BARE_WORD;
395  }
396 
397  MEM(c_map->rhs = tmpl_alloc(c_map,
398  TMPL_TYPE_DATA, quote, map->rhs->name, map->rhs->len));
399  if (fr_value_box_copy(c_map->rhs, tmpl_value(c_map->rhs), &vp->data) < 0) {
400  REDEBUG("Failed copying attribute value");
401  error:
402  talloc_free(pool);
403  talloc_free(c);
404  RETURN_MODULE_FAIL;
405  }
406  }
407  break;
408 
409  /*
410  * Lists are weird... We need to fudge a new LHS template,
411  * which is a combination of the LHS list and the attribute.
412  */
413  case TMPL_TYPE_LIST:
414  if (tmpl_attr_afrom_list(c_map, &c_map->lhs, map->lhs, vp->da) < 0) {
415  RPERROR("Failed attribute -> list copy");
416  goto error;
417  }
418  goto do_rhs;
419 
420  default:
421  fr_assert(0);
422  }
423  MAP_VERIFY(c_map);
424  *last = c_map;
425  last = &(*last)->next;
426  }
427  talloc_free_children(pool); /* reset pool state */
428  }
429  talloc_free(pool);
430 
431  /*
432  * Check to see if we need to merge the entry into the request
433  */
434  vp = fr_pair_find_by_da(&request->control_pairs, attr_cache_merge_new);
435  if (vp && vp->vp_bool) merge = true;
436 
437  if (merge) cache_merge(inst, request, c);
438 
439  for (;;) {
440  cache_status_t ret;
441 
442  ret = inst->driver->insert(&inst->config, inst->driver_inst->dl_inst->data, request, *handle, c);
443  switch (ret) {
444  case CACHE_RECONNECT:
445  if (cache_reconnect(handle, inst, request) == 0) continue;
446  RETURN_MODULE_FAIL;
447 
448  case CACHE_OK:
449  RDEBUG2("Committed entry, TTL %d seconds", ttl);
450  cache_free(inst, &c);
451  RETURN_MODULE_RCODE(merge ? RLM_MODULE_UPDATED : RLM_MODULE_OK);
452 
453  default:
454  talloc_free(c); /* Failed insertion - use talloc_free not the driver free */
455  RETURN_MODULE_FAIL;
456  }
457  }
458 }

460 /** Update the TTL of an entry
461  *
462  * @return
463  * - #RLM_MODULE_OK on success.
464  * - #RLM_MODULE_FAIL on failure.
465  */
466 static unlang_action_t cache_set_ttl(rlm_rcode_t *p_result,
467  rlm_cache_t const *inst, request_t *request,
468  rlm_cache_handle_t **handle, rlm_cache_entry_t *c)
469 {
470  /*
471  * Call the driver's insert method to overwrite the old entry
472  */
473  if (!inst->driver->set_ttl) for (;;) {
474  cache_status_t ret;
475 
476  ret = inst->driver->insert(&inst->config, inst->driver_inst->dl_inst->data, request, *handle, c);
477  switch (ret) {
478  case CACHE_RECONNECT:
479  if (cache_reconnect(handle, inst, request) == 0) continue;
480  RETURN_MODULE_FAIL;
481 
482  case CACHE_OK:
483  RDEBUG2("Updated entry TTL");
484  RETURN_MODULE_OK;
485 
486  default:
487  RETURN_MODULE_FAIL;
488  }
489  }
490 
491  /*
492  * Or call the set ttl method if the driver can do this more
493  * efficiently.
494  */
495  for (;;) {
496  cache_status_t ret;
497 
498  ret = inst->driver->set_ttl(&inst->config, inst->driver_inst->dl_inst->data, request, *handle, c);
499  switch (ret) {
500  case CACHE_RECONNECT:
501  if (cache_reconnect(handle, inst, request) == 0) continue;
502  RETURN_MODULE_FAIL;
503 
504  case CACHE_OK:
505  RDEBUG2("Updated entry TTL");
506  RETURN_MODULE_OK;
507 
508  default:
509  RETURN_MODULE_FAIL;
510  }
511  }

530 /** Do caching checks
531  *
532  * Since we can update ANY VP list, we do exactly the same thing for all sections
533  * (autz / auth / etc.)
534  *
535  * If you want to cache something different in different sections, configure
536  * another cache module.
537  */
538 static unlang_action_t CC_HINT(nonnull) mod_cache_it(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
539 {
540  rlm_cache_entry_t *c = NULL;
541  rlm_cache_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_cache_t);
542 
543  rlm_cache_handle_t *handle;
544 
545  fr_cursor_t cursor;
546  fr_pair_t *vp;
547 
548  bool merge = true, insert = true, expire = false, set_ttl = false;
549  int exists = -1;
550 
551  uint8_t buffer[1024];
552  uint8_t const *key;
553  ssize_t key_len;
554  rlm_rcode_t rcode = RLM_MODULE_NOOP;
555 
556  int ttl = inst->config.ttl;
557 
558  key_len = tmpl_expand((char const **)&key, (char *)buffer, sizeof(buffer),
559  request, inst->config.key, NULL, NULL);
560  if (key_len < 0) {
561  RETURN_MODULE_FAIL;
562  }
563 
564  if (key_len == 0) {
565  REDEBUG("Zero length key string is invalid");
566  RETURN_MODULE_INVALID;
567  }
568 
569  /*
570  * If Cache-Status-Only == yes, only return whether we found a
571  * valid cache entry
572  */
573  vp = fr_pair_find_by_da(&request->control_pairs, attr_cache_status_only);
574  if (vp && vp->vp_bool) {
575  RINDENT();
576  RDEBUG3("status-only: yes");
577  REXDENT();
578 
579  if (cache_acquire(&handle, inst, request) < 0) {
580  RETURN_MODULE_FAIL;
581  }
582 
583  cache_find(&rcode, &c, inst, request, &handle, key, key_len);
584  if (rcode == RLM_MODULE_FAIL) goto finish;
585  fr_assert(!inst->driver->acquire || handle);
586 
587  rcode = c ? RLM_MODULE_OK:
588  RLM_MODULE_NOTFOUND;
589  goto finish;
590  }
591 
592  /*
593  * Figure out what operation we're doing
594  */
595  vp = fr_pair_find_by_da(&request->control_pairs, attr_cache_allow_merge);
596  if (vp) merge = vp->vp_bool;
597 
598  vp = fr_pair_find_by_da(&request->control_pairs, attr_cache_allow_insert);
599  if (vp) insert = vp->vp_bool;
600 
601  vp = fr_pair_find_by_da(&request->control_pairs, attr_cache_ttl);
602  if (vp) {
603  if (vp->vp_int32 == 0) {
604  expire = true;
605  } else if (vp->vp_int32 < 0) {
606  expire = true;
607  ttl = -(vp->vp_int32);
608  /* Updating the TTL */
609  } else {
610  set_ttl = true;
611  ttl = vp->vp_int32;
612  }
613  }
614 
615  RINDENT();
616  RDEBUG3("merge : %s", merge ? "yes" : "no");
617  RDEBUG3("insert : %s", insert ? "yes" : "no");
618  RDEBUG3("expire : %s", expire ? "yes" : "no");
619  RDEBUG3("ttl : %i", ttl);
620  REXDENT();
621  if (cache_acquire(&handle, inst, request) < 0) {
622  RETURN_MODULE_FAIL;
623  }
624 
625  /*
626  * Retrieve the cache entry and merge it with the current request
627  * recording whether the entry existed.
628  */
629  if (merge) {
630  cache_find(&rcode, &c, inst, request, &handle, key, key_len);
631  switch (rcode) {
632  case RLM_MODULE_FAIL:
633  goto finish;
634 
635  case RLM_MODULE_OK:
636  rcode = cache_merge(inst, request, c);
637  exists = 1;
638  break;
639 
640  case RLM_MODULE_NOTFOUND:
641  rcode = RLM_MODULE_NOTFOUND;
642  exists = 0;
643  break;
644 
645  default:
646  fr_assert(0);
647  }
648  fr_assert(!inst->driver->acquire || handle);
649  }
650 
651  /*
652  * Expire the entry if told to, and we either don't know whether
653  * it exists, or we know it does.
654  *
655  * We only expire if we're not inserting, as driver insert methods
656  * should perform upserts.
657  */
658  if (expire && ((exists == -1) || (exists == 1))) {
659  if (!insert) {
660  rlm_rcode_t tmp;
661 
662  fr_assert(!set_ttl);
663  cache_expire(&tmp, inst, request, &handle, key, key_len);
664  switch (tmp) {
665  case RLM_MODULE_FAIL:
666  rcode = RLM_MODULE_FAIL;
667  goto finish;
668 
669  case RLM_MODULE_OK:
670  if (rcode == RLM_MODULE_NOOP) rcode = RLM_MODULE_OK;
671  break;
672 
673  case RLM_MODULE_NOTFOUND:
674  if (rcode == RLM_MODULE_NOOP) rcode = RLM_MODULE_NOTFOUND;
675  break;
676 
677  default:
678  fr_assert(0);
679  break;
680  }
681  /* If it previously existed, it doesn't now */
682  }
683  /* Otherwise use insert to overwrite */
684  exists = 0;
685  }
686 
687  /*
688  * If we still don't know whether it exists or not
689  * and we need to do an insert or set_ttl operation
690  * determine that now.
691  */
692  if ((exists < 0) && (insert || set_ttl)) {
693  rlm_rcode_t tmp;
694 
695  cache_find(&tmp, &c, inst, request, &handle, key, key_len);
696  switch (tmp) {
697  case RLM_MODULE_FAIL:
698  rcode = RLM_MODULE_FAIL;
699  goto finish;
700 
701  case RLM_MODULE_OK:
702  exists = 1;
703  if (rcode != RLM_MODULE_UPDATED) rcode = RLM_MODULE_OK;
704  break;
705 
706  case RLM_MODULE_NOTFOUND:
707  exists = 0;
708  break;
709 
710  default:
711  fr_assert(0);
712  }
713  fr_assert(!inst->driver->acquire || handle);
714  }
715 
716  /*
717  * We can only alter the TTL on an entry if it exists.
718  */
719  if (set_ttl && (exists == 1)) {
720  rlm_rcode_t tmp;
721 
722  fr_assert(c);
723 
724  c->expires = fr_time_to_unix_time(request->packet->timestamp) + fr_unix_time_from_sec(ttl);
725 
726  cache_set_ttl(&tmp, inst, request, &handle, c);
727  switch (tmp) {
728  case RLM_MODULE_FAIL:
729  rcode = RLM_MODULE_FAIL;
730  goto finish;
731 
732  case RLM_MODULE_NOTFOUND:
733  case RLM_MODULE_OK:
734  if (rcode != RLM_MODULE_UPDATED) rcode = RLM_MODULE_OK;
735  goto finish;
736 
737  default:
738  fr_assert(0);
739  }
740  }
741 
742  /*
743  * Inserts are upserts, so we don't care about the
744  * entry state, just that we're not meant to be
745  * setting the TTL, which precludes performing an
746  * insert.
747  */
748  if (insert && (exists == 0)) {
749  rlm_rcode_t tmp;
750 
751  cache_insert(&tmp, inst, request, &handle, key, key_len, ttl);
752  switch (tmp) {
753  case RLM_MODULE_FAIL:
754  rcode = RLM_MODULE_FAIL;
755  goto finish;
756 
757  case RLM_MODULE_OK:
758  if (rcode != RLM_MODULE_UPDATED) rcode = RLM_MODULE_OK;
759  break;
760 
761  case RLM_MODULE_UPDATED:
762  rcode = RLM_MODULE_UPDATED;
763  break;
764 
765  default:
766  fr_assert(0);
767  }
768  fr_assert(!inst->driver->acquire || handle);
769  goto finish;
770  }
771 
772 
773 finish:
774  cache_free(inst, &c);
775  cache_release(inst, request, &handle);
776 
777  /*
778  * Clear control attributes
779  */
780  for (vp = fr_cursor_init(&cursor, &request->control_pairs);
781  vp;
782  vp = fr_cursor_next(&cursor)) {
783  again:
784  if (!fr_dict_attr_is_top_level(vp->da)) continue;
785 
786  switch (vp->da->attr) {
787  case FR_CACHE_TTL:
788  case FR_CACHE_STATUS_ONLY:
789  case FR_CACHE_ALLOW_MERGE:
790  case FR_CACHE_ALLOW_INSERT:
791  case FR_CACHE_MERGE_NEW:
792  RDEBUG2("Removing &control.%s", vp->da->name);
793  vp = fr_cursor_remove(&cursor);
794  talloc_free(vp);
795  vp = fr_cursor_current(&cursor);
796  if (!vp) break;
797  goto again;
798  }
799  }
800 
801  RETURN_MODULE_RCODE(rcode);
802 }

804 /** Allow single attribute values to be retrieved from the cache

808 static ssize_t cache_xlat(TALLOC_CTX *ctx, char **out, UNUSED size_t freespace,
809  void const *mod_inst, UNUSED void const *xlat_inst,
810  request_t *request, char const *fmt) CC_HINT(nonnull);
811 static ssize_t cache_xlat(TALLOC_CTX *ctx, char **out, UNUSED size_t freespace,
812  void const *mod_inst, UNUSED void const *xlat_inst,
813  request_t *request, char const *fmt)
814 {
815  rlm_cache_entry_t *c = NULL;
816  rlm_cache_t const *inst = mod_inst;
817  rlm_cache_handle_t *handle = NULL;
818 
819  ssize_t slen;
820  ssize_t ret = 0;
821 
822  uint8_t buffer[1024];
823  uint8_t const *key;
824  ssize_t key_len;
825 
826  tmpl_t *target = NULL;
827  map_t *map = NULL;
828  rlm_rcode_t rcode = RLM_MODULE_NOOP;
829 
830  key_len = tmpl_expand((char const **)&key, (char *)buffer, sizeof(buffer),
831  request, inst->config.key, NULL, NULL);
832  if (key_len < 0) return -1;
833 
834  slen = tmpl_afrom_attr_substr(ctx, NULL, &target,
835  &FR_SBUFF_IN(fmt, strlen(fmt)),
836  NULL,
837  &(tmpl_rules_t){
838  .dict_def = request->dict,
839  .prefix = TMPL_ATTR_REF_PREFIX_AUTO
840  });
841  if (slen <= 0) {
842  RPEDEBUG("Invalid key");
843  return -1;
844  }
845 
846  if (cache_acquire(&handle, mod_inst, request) < 0) {
847  talloc_free(target);
848  return -1;
849  }
850 
851  cache_find(&rcode, &c, mod_inst, request, &handle, key, key_len);
852  switch (rcode) {
853  case RLM_MODULE_OK: /* found */
854  break;
855 
856  case RLM_MODULE_NOTFOUND: /* not found */
857  return 0;
858 
859  default:
860  talloc_free(target);
861  return -1;
862  }
863 
864  for (map = c->maps; map; map = map->next) {
865  if ((tmpl_da(map->lhs) != tmpl_da(target)) ||
866  (tmpl_list(map->lhs) != tmpl_list(target))) continue;
867 
868  fr_value_box_aprint(request, out, tmpl_value(map->rhs), NULL);
869  ret = talloc_array_length(*out) - 1;
870  break;
871  }
872 
873  talloc_free(target);
874 
875  /*
876  * Check if we found a matching map
877  */
878  if (!map) return 0;
879 
880  cache_free(mod_inst, &c);
881  cache_release(mod_inst, request, &handle);
882 
883  return ret;
884 }
