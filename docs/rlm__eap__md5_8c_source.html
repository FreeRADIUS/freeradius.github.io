47 /*
48  * Authenticate a previously sent challenge.
49  */
50 static unlang_action_t mod_process(rlm_rcode_t *p_result, UNUSED module_ctx_t const *mctx, request_t *request)
51 {
52  eap_session_t *eap_session = eap_session_get(request->parent);
53  MD5_PACKET *packet;
54  MD5_PACKET *reply;
55  fr_pair_t *known_good;
56  fr_dict_attr_t const *allowed_passwords[] = { attr_cleartext_password };
57  bool ephemeral;
58 
59  /*
60  * Get the Cleartext-Password for this user.
61  */
62  fr_assert(eap_session->request != NULL);
63 
64  known_good = password_find(&ephemeral, request, request->parent,
65  allowed_passwords, NUM_ELEMENTS(allowed_passwords),
66  false);
67  if (!known_good) {
68  REDEBUG("No \"known good\" password found for user");
69  RETURN_MODULE_FAIL;
70  }
71 
72  /*
73  * Extract the EAP-MD5 packet.
74  */
75  packet = eap_md5_extract(eap_session->this_round);
76  if (!packet) {
77  if (ephemeral) talloc_list_free(&known_good);
78  RETURN_MODULE_INVALID;
79  }
80 
81  /*
82  * Create a reply, and initialize it.
83  */
84  MEM(reply = talloc(packet, MD5_PACKET));
85  reply->id = eap_session->this_round->request->id;
86  reply->length = 0;
87 
88  /*
89  * Verify the received packet against the previous packet
90  * (i.e. challenge) which we sent out.
91  */
92  if (eap_md5_verify(packet, known_good, eap_session->opaque)) {
93  reply->code = FR_MD5_SUCCESS;
94  } else {
95  reply->code = FR_MD5_FAILURE;
96  }
97 
98  /*
99  * Compose the EAP-MD5 packet out of the data structure,
100  * and free it.
101  */
102  eap_md5_compose(eap_session->this_round, reply);
103  talloc_free(packet);
104 
105  if (ephemeral) talloc_list_free(&known_good);
106 
107  RETURN_MODULE_OK;
108 }

110 /*
111  * Initiate the EAP-MD5 session by sending a challenge to the peer.
112  */
113 static unlang_action_t mod_session_init(rlm_rcode_t *p_result, UNUSED module_ctx_t const *mctx, request_t *request)
114 {
115  eap_session_t *eap_session = eap_session_get(request->parent);
116  MD5_PACKET *reply;
117  int i;
118 
119  fr_assert(eap_session != NULL);
120 
121  /*
122  * Allocate an EAP-MD5 packet.
123  */
124  MEM(reply = talloc(eap_session, MD5_PACKET));
125 
126  /*
127  * Fill it with data.
128  */
129  reply->code = FR_MD5_CHALLENGE;
130  reply->length = 1 + MD5_CHALLENGE_LEN; /* one byte of value size */
131  reply->value_size = MD5_CHALLENGE_LEN;
132 
133  /*
134  * Allocate user data.
135  */
136  MEM(reply->value = talloc_array(reply, uint8_t, reply->value_size));
137  /*
138  * Get a random challenge.
139  */
140  for (i = 0; i < reply->value_size; i++) reply->value[i] = fr_rand();
141  RDEBUG2("Issuing MD5 Challenge");
142 
143  /*
144  * Keep track of the challenge.
145  */
146  MEM(eap_session->opaque = talloc_array(eap_session, uint8_t, reply->value_size));
147  memcpy(eap_session->opaque, reply->value, reply->value_size);
148 
149  /*
150  * Compose the EAP-MD5 packet out of the data structure,
151  * and free it.
152  */
153  eap_md5_compose(eap_session->this_round, reply);
154 
155  /*
156  * We don't need to authorize the user at this point.
157  *
158  * We also don't need to keep the challenge, as it's
159  * stored in 'eap_session->this_round', which will be given back
160  * to us...
161  */
162  eap_session->process = mod_process;
163 
164  RETURN_MODULE_HANDLED;
165 }
