226 /** Do Ascend-Send / Recv-Secret calculation.
227  *
228  * The secret is hidden by xoring with a MD5 digest created from
229  * the RADIUS shared secret and the authentication vector.
230  * We put them into MD5 in the reverse order from that used when
231  * encrypting passwords to RADIUS.
232  */
233 ssize_t fr_radius_ascend_secret(fr_dbuff_t *dbuff, uint8_t const *in, size_t inlen,
234  char const *secret, uint8_t const vector[static RADIUS_AUTH_VECTOR_LENGTH])
235 {
236  fr_md5_ctx_t *md5_ctx;
237  size_t i;
238  uint8_t digest[MD5_DIGEST_LENGTH];
239  fr_dbuff_t work_dbuff = FR_DBUFF_NO_ADVANCE(dbuff);
240 
241  FR_DBUFF_EXTEND_LOWAT_OR_RETURN(&work_dbuff, sizeof(digest));
242 
243  md5_ctx = fr_md5_ctx_alloc(true);
244  fr_md5_update(md5_ctx, vector, RADIUS_AUTH_VECTOR_LENGTH);
245  fr_md5_update(md5_ctx, (uint8_t const *) secret, talloc_array_length(secret) - 1);
246  fr_md5_final(digest, md5_ctx);
247  fr_md5_ctx_free(&md5_ctx);
248 
249  if (inlen > sizeof(digest)) inlen = sizeof(digest);
250  for (i = 0; i < inlen; i++) digest[i] ^= in[i];
251 
252  fr_dbuff_in_memcpy(&work_dbuff, digest, sizeof(digest));
253 
254  return fr_dbuff_set(dbuff, &work_dbuff);
255 }

257 /** Basic validation of RADIUS packet header
258  *
259  * @note fr_strerror errors are only available if fr_debug_lvl > 0. This is to reduce CPU time
260  * consumed when discarding malformed packet.
261  *
262  * @param[in] sockfd we're reading from.
263  * @param[out] src_ipaddr of the packet.
264  * @param[out] src_port of the packet.
265  * @param[out] code Pointer to where to write the packet code.
266  * @return
267  * - -1 on failure.
268  * - 1 on decode error.
269  * - >= RADIUS_HEADER_LENGTH on success. This is the packet length as specified in the header.
270  */
271 ssize_t fr_radius_recv_header(int sockfd, fr_ipaddr_t *src_ipaddr, uint16_t *src_port, unsigned int *code)
272 {
273  ssize_t data_len, packet_len;
274  uint8_t header[4];
275 
276  data_len = udp_recv_peek(sockfd, header, sizeof(header), UDP_FLAGS_PEEK, src_ipaddr, src_port);
277  if (data_len < 0) {
278  if ((errno == EAGAIN) || (errno == EINTR)) return 0;
279  return -1;
280  }
281 
282  /*
283  * Too little data is available, discard the packet.
284  */
285  if (data_len < 4) {
286  char buffer[INET6_ADDRSTRLEN];
287 
288  FR_DEBUG_STRERROR_PRINTF("Expected at least 4 bytes of header data, got %zu bytes", data_len);
289 invalid:
290  FR_DEBUG_STRERROR_PRINTF("Invalid data from %s",
291  inet_ntop(src_ipaddr->af, &src_ipaddr->addr, buffer, sizeof(buffer)));
292  (void) udp_recv_discard(sockfd);
293 
294  return 0;
295  }
296 
297  /*
298  * See how long the packet says it is.
299  */
300  packet_len = (header[2] * 256) + header[3];
301 
302  /*
303  * The length in the packet says it's less than
304  * a RADIUS header length: discard it.
305  */
306  if (packet_len < RADIUS_HEADER_LENGTH) {
307  FR_DEBUG_STRERROR_PRINTF("Expected at least " STRINGIFY(RADIUS_HEADER_LENGTH) " bytes of packet "
308  "data, got %zu bytes", packet_len);
309  goto invalid;
310  }
311 
312  /*
313  * Enforce RFC requirements, for sanity.
314  * Anything after 4k will be discarded.
315  */
316  if (packet_len > MAX_PACKET_LEN) {
317  FR_DEBUG_STRERROR_PRINTF("Length field value too large, expected maximum of "
318  STRINGIFY(MAX_PACKET_LEN) " bytes, got %zu bytes", packet_len);
319  goto invalid;
320  }
321 
322  *code = header[0];
323 
324  /*
325  * The packet says it's this long, but the actual UDP
326  * size could still be smaller.
327  */
328  return packet_len;
329 }

331 /** Sign a previously encoded packet
332  *
333  * Calculates the request/response authenticator for packets which need it, and fills
334  * in the message-authenticator value if the attribute is present in the encoded packet.
335  *
336  * @param[in,out] packet (request or response).
337  * @param[in] original request (only if this is a response).
338  * @param[in] secret to sign the packet with.
339  * @param[in] secret_len The length of the secret.
340  * @return
341  * - <0 on error
342  * - 0 on success
343  */
344 int fr_radius_sign(uint8_t *packet, uint8_t const *original,
345  uint8_t const *secret, size_t secret_len)
346 {
347  uint8_t *msg, *end;
348  size_t packet_len = (packet[2] << 8) | packet[3];
349 
350  /*
351  * No real limit on secret length, this is just
352  * to catch uninitialised fields.
353  */
354  if (!fr_cond_assert(secret_len <= UINT16_MAX)) {
355  fr_strerror_printf("Secret is too long. Expected <= %u, got %zu", UINT16_MAX, secret_len);
356  return -1;
357  }
358 
359  if (packet_len < RADIUS_HEADER_LENGTH) {
360  fr_strerror_printf("Packet must be encoded before calling fr_radius_sign()");
361  return -1;
362  }
363 
364  /*
365  * Find Message-Authenticator. Its value has to be
366  * calculated before we calculate the Request
367  * Authenticator or the Response Authenticator.
368  */
369  msg = packet + RADIUS_HEADER_LENGTH;
370  end = packet + packet_len;
371 
372  while (msg < end) {
373  if ((end - msg) < 2) goto invalid_attribute;
374 
375  if (msg[0] != FR_MESSAGE_AUTHENTICATOR) {
376  if (msg[1] < 2) goto invalid_attribute;
377 
378  if ((msg + msg[1]) > end) {
379  invalid_attribute:
380  fr_strerror_printf("Invalid attribute at offset %zd", msg - packet);
381  return -1;
382  }
383  msg += msg[1];
384  continue;
385  }
386 
387  if (msg[1] < 18) {
388  fr_strerror_printf("Message-Authenticator is too small");
389  return -1;
390  }
391 
392  switch (packet[0]) {
393  case FR_CODE_ACCOUNTING_RESPONSE:
394  case FR_CODE_DISCONNECT_ACK:
395  case FR_CODE_DISCONNECT_NAK:
396  case FR_CODE_COA_ACK:
397  case FR_CODE_COA_NAK:
398  if (!original) goto need_original;
399  if (original[0] == FR_CODE_STATUS_SERVER) goto do_ack;
400  FALL_THROUGH;
401 
402  case FR_CODE_ACCOUNTING_REQUEST:
403  case FR_CODE_DISCONNECT_REQUEST:
404  case FR_CODE_COA_REQUEST:
405  memset(packet + 4, 0, RADIUS_AUTH_VECTOR_LENGTH);
406  break;
407 
408  case FR_CODE_ACCESS_ACCEPT:
409  case FR_CODE_ACCESS_REJECT:
410  case FR_CODE_ACCESS_CHALLENGE:
411  do_ack:
412  if (!original) goto need_original;
413  memcpy(packet + 4, original + 4, RADIUS_AUTH_VECTOR_LENGTH);
414  break;
415 
416  case FR_CODE_ACCESS_REQUEST:
417  case FR_CODE_STATUS_SERVER:
418  /* packet + 4 MUST be the Request Authenticator filled with random data */
419  break;
420 
421  default:
422  goto bad_packet;
423  }
424 
425  /*
426  * Force Message-Authenticator to be zero,
427  * calculate the HMAC, and put it into the
428  * Message-Authenticator attribute.
429  */
430  memset(msg + 2, 0, RADIUS_AUTH_VECTOR_LENGTH);
431  fr_hmac_md5(msg + 2, packet, packet_len, secret, secret_len);
432  break;
433  }
434 
435  /*
436  * Initialize the request authenticator.
437  */
438  switch (packet[0]) {
439  case FR_CODE_ACCOUNTING_REQUEST:
440  case FR_CODE_DISCONNECT_REQUEST:
441  case FR_CODE_COA_REQUEST:
442  memset(packet + 4, 0, RADIUS_AUTH_VECTOR_LENGTH);
443  break;
444 
445  case FR_CODE_ACCESS_ACCEPT:
446  case FR_CODE_ACCESS_REJECT:
447  case FR_CODE_ACCESS_CHALLENGE:
448  case FR_CODE_ACCOUNTING_RESPONSE:
449  case FR_CODE_DISCONNECT_ACK:
450  case FR_CODE_DISCONNECT_NAK:
451  case FR_CODE_COA_ACK:
452  case FR_CODE_COA_NAK:
453  case FR_CODE_PROTOCOL_ERROR:
454  if (!original) {
455  need_original:
456  fr_strerror_printf("Cannot sign response packet without a request packet");
457  return -1;
458  }
459  memcpy(packet + 4, original + 4, RADIUS_AUTH_VECTOR_LENGTH);
460  break;
461 
462  /*
463  * The Request Authenticator is random numbers.
464  * We don't need to sign anything else, so
465  * return.
466  */
467  case FR_CODE_ACCESS_REQUEST:
468  case FR_CODE_STATUS_SERVER:
469  return 0;
470 
471  default:
472  bad_packet:
473  fr_strerror_printf("Cannot sign unknown packet code %u", packet[0]);
474  return -1;
475  }
476 
477  /*
478  * Request / Response Authenticator = MD5(packet + secret)
479  */
480  {
481  fr_md5_ctx_t *md5_ctx;
482 
483  md5_ctx = fr_md5_ctx_alloc(true);
484  fr_md5_update(md5_ctx, packet, packet_len);
485  fr_md5_update(md5_ctx, secret, secret_len);
486  fr_md5_final(packet + 4, md5_ctx);
487  fr_md5_ctx_free(&md5_ctx);
488  }
489 
490  return 0;
491 }

494 /** See if the data pointed to by PTR is a valid RADIUS packet.
495  *
496  * @param[in] packet to check.
497  * @param[in,out] packet_len_p The size of the packet data.
498  * @param[in] max_attributes to allow in the packet.
499  * @param[in] require_ma whether we require Message-Authenticator.
500  * @param[in] reason if not NULL, will have the failure reason written to where it points.
501  * @return
502  * - True on success.
503  * - False on failure.
504  */
505 bool fr_radius_ok(uint8_t const *packet, size_t *packet_len_p,
506  uint32_t max_attributes, bool require_ma, decode_fail_t *reason)
507 {
508  uint8_t const *attr, *end;
509  size_t totallen;
510  bool seen_ma = false;
511  uint32_t num_attributes;
512  decode_fail_t failure = DECODE_FAIL_NONE;
513  size_t packet_len = *packet_len_p;
514 
515  /*
516  * Check for packets smaller than the packet header.
517  *
518  * RFC 2865, Section 3., subsection 'length' says:
519  *
520  * "The minimum length is 20 ..."
521  */
522  if (packet_len < RADIUS_HEADER_LENGTH) {
523  FR_DEBUG_STRERROR_PRINTF("packet is too short (received %zu < minimum 20)",
524  packet_len);
525  failure = DECODE_FAIL_MIN_LENGTH_PACKET;
526  goto finish;
527  }
528 
529 
530  /*
531  * Check for packets with mismatched size.
532  * i.e. We've received 128 bytes, and the packet header
533  * says it's 256 bytes long.
534  */
535  totallen = (packet[2] << 8) | packet[3];
536 
537  /*
538  * Code of 0 is not understood.
539  * Code of 16 or greate is not understood.
540  */
541  if ((packet[0] == 0) ||
542  (packet[0] >= FR_RADIUS_MAX_PACKET_CODE)) {
543  FR_DEBUG_STRERROR_PRINTF("unknown packet code %d", packet[0]);
544  failure = DECODE_FAIL_UNKNOWN_PACKET_CODE;
545  goto finish;
546  }
547 
548  /*
549  * Message-Authenticator is required in Status-Server
550  * packets, otherwise they can be trivially forged.
551  */
552  if (packet[0] == FR_CODE_STATUS_SERVER) require_ma = true;
553 
554  /*
555  * Repeat the length checks. This time, instead of
556  * looking at the data we received, look at the value
557  * of the 'length' field inside of the packet.
558  *
559  * Check for packets smaller than the packet header.
560  *
561  * RFC 2865, Section 3., subsection 'length' says:
562  *
563  * "The minimum length is 20 ..."
564  */
565  if (totallen < RADIUS_HEADER_LENGTH) {
566  FR_DEBUG_STRERROR_PRINTF("length in header is too small (length %zu < minimum 20)",
567  totallen);
568  failure = DECODE_FAIL_MIN_LENGTH_FIELD;
569  goto finish;
570  }
571 
572  /*
573  * And again, for the value of the 'length' field.
574  *
575  * RFC 2865, Section 3., subsection 'length' says:
576  *
577  * " ... and maximum length is 4096."
578  *
579  * HOWEVER. This requirement is for the network layer.
580  * If the code gets here, we assume that a well-formed
581  * packet is an OK packet.
582  *
583  * We allow both the UDP data length, and the RADIUS
584  * "length" field to contain up to 64K of data.
585  */
586 
587  /*
588  * RFC 2865, Section 3., subsection 'length' says:
589  *
590  * "If the packet is shorter than the Length field
591  * indicates, it MUST be silently discarded."
592  *
593  * i.e. No response to the NAS.
594  */
595  if (packet_len < totallen) {
596  FR_DEBUG_STRERROR_PRINTF("packet is truncated (received %zu < packet header length of %zu)",
597  packet_len, totallen);
598  failure = DECODE_FAIL_MIN_LENGTH_MISMATCH;
599  goto finish;
600  }
601 
602  /*
603  * RFC 2865, Section 3., subsection 'length' says:
604  *
605  * "Octets outside the range of the Length field MUST be
606  * treated as padding and ignored on reception."
607  */
608  if (packet_len > totallen) {
609  *packet_len_p = packet_len = totallen;
610  }
611 
612  /*
613  * Walk through the packet's attributes, ensuring that
614  * they add up EXACTLY to the size of the packet.
615  *
616  * If they don't, then the attributes either under-fill
617  * or over-fill the packet. Any parsing of the packet
618  * is impossible, and will result in unknown side effects.
619  *
620  * This would ONLY happen with buggy RADIUS implementations,
621  * or with an intentional attack. Either way, we do NOT want
622  * to be vulnerable to this problem.
623  */
624  attr = packet + RADIUS_HEADER_LENGTH;
625  end = packet + packet_len;
626  num_attributes = 0;
627 
628  while (attr < end) {
629  /*
630  * We need at least 2 bytes to check the
631  * attribute header.
632  */
633  if ((end - attr) < 2) {
634  FR_DEBUG_STRERROR_PRINTF("attribute header overflows the packet");
635  failure = DECODE_FAIL_HEADER_OVERFLOW;
636  goto finish;
637  }
638 
639  /*
640  * Attribute number zero is NOT defined.
641  */
642  if (attr[0] == 0) {
643  FR_DEBUG_STRERROR_PRINTF("invalid attribute 0 at offset %zd", attr - packet);
644  failure = DECODE_FAIL_INVALID_ATTRIBUTE;
645  goto finish;
646  }
647

649  * Attributes are at LEAST as long as the ID & length
650  * fields. Anything shorter is an invalid attribute.
651  */
652  if (attr[1] < 2) {
653  FR_DEBUG_STRERROR_PRINTF("attribute %u is too short at offset %zd",
654  attr[0], attr - packet);
655  failure = DECODE_FAIL_ATTRIBUTE_TOO_SHORT;
656  goto finish;
657  }
658

660  * If there are fewer bytes in the packet than in the
661  * attribute, it's a bad packet.
662  */
663  if ((attr + attr[1]) > end) {
664  FR_DEBUG_STRERROR_PRINTF("attribute %u data overflows the packet starting at offset %zd",
665  attr[0], attr - packet);
666  failure = DECODE_FAIL_ATTRIBUTE_OVERFLOW;
667  goto finish;
668  }
669

671  * Sanity check the attributes for length.
672  */
673  switch (attr[0]) {
674  default: /* don't do anything by default */
675  break;
676 
677 #if 0
678  /*

680  */
681  case FR_STATE:
682  if (attr[1] > 2) packet->rounds = attr[2];

686  /*
687  * If there's an EAP-Message, we require
688  * a Message-Authenticator.
689  */
690  case FR_EAP_MESSAGE:
691  require_ma = true;
692  break;
693

695  if (attr[1] != 2 + RADIUS_AUTH_VECTOR_LENGTH) {
696  FR_DEBUG_STRERROR_PRINTF("Message-Authenticator has invalid length (%d != 18) at offset %zd",
697  attr[1] - 2, attr - packet);
698  failure = DECODE_FAIL_MA_INVALID_LENGTH;
699  goto finish;

701  seen_ma = true;
702  break;
703  }
704 
705  attr += attr[1];
706  num_attributes++; /* seen one more attribute */
707  }
708 
709  /*
710  * If the attributes add up to a packet, it's allowed.
711  *
712  * If not, we complain, and throw the packet away.
713  */
714  if (attr != end) {
715  FR_DEBUG_STRERROR_PRINTF("attributes do NOT exactly fill the packet");
716  failure = DECODE_FAIL_ATTRIBUTE_UNDERFLOW;
717  goto finish;
718  }
719

721  * If we're configured to look for a maximum number of
722  * attributes, and we've seen more than that maximum,
723  * then throw the packet away, as a possible DoS.
724  */

726  (num_attributes > max_attributes)) {

728  num_attributes, max_attributes);

736  * A packet with an EAP-Message attribute MUST also have
737  * a Message-Authenticator attribute.
738  *
739  * A Message-Authenticator all by itself is OK, though.
740  *
741  * Similarly, Status-Server packets MUST contain
742  * Message-Authenticator attributes.
743  */
744  if (require_ma && !seen_ma) {

746  failure = DECODE_FAIL_MA_MISSING;
747  goto finish;
748  }

752  if (reason) {
753  *reason = failure;
754  }
755  return (failure == DECODE_FAIL_NONE);

759 /** Verify a request / response packet
760  *
761  * This function does its work by calling fr_radius_sign(), and then
762  * comparing the signature in the packet with the one we calculated.
763  * If they differ, there's a problem.
764  *
765  * @param packet the raw RADIUS packet (request or response)
766  * @param original the raw original request (if this is a response)
767  * @param secret the shared secret
768  * @param secret_len the length of the secret
769  * @return
770  * - <0 on error
771  * - 0 on success
772  */
773 int fr_radius_verify(uint8_t *packet, uint8_t const *original,
774  uint8_t const *secret, size_t secret_len)
775 {
776  int rcode;
777  uint8_t *msg, *end;
778  size_t packet_len = (packet[2] << 8) | packet[3];
779  uint8_t request_authenticator[RADIUS_AUTH_VECTOR_LENGTH];
780  uint8_t message_authenticator[RADIUS_AUTH_VECTOR_LENGTH];
781 
782  if (packet_len < RADIUS_HEADER_LENGTH) {
783  fr_strerror_printf("invalid packet length %zd", packet_len);
784  return -1;
785  }
786 
787  memcpy(request_authenticator, packet + 4, sizeof(request_authenticator));
788 
789  /*
790  * Find Message-Authenticator. Its value has to be
791  * calculated before we calculate the Request
792  * Authenticator or the Response Authenticator.
793  */
794  msg = packet + RADIUS_HEADER_LENGTH;
795  end = packet + packet_len;
796 
797  while (msg < end) {
798  if ((end - msg) < 2) goto invalid_attribute;
799 
800  if (msg[0] != FR_MESSAGE_AUTHENTICATOR) {
801  if (msg[1] < 2) goto invalid_attribute;
802 
803  if ((msg + msg[1]) > end) {
804  invalid_attribute:
805  fr_strerror_printf("invalid attribute at offset %zd", msg - packet);
806  return -1;
807  }
808  msg += msg[1];
809  continue;
810  }
811 
812  if (msg[1] < 18) {
813  fr_strerror_printf("too small Message-Authenticator");
814  return -1;
815  }
816 
817  /*
818  * Found it, save a copy.
819  */
820  memcpy(message_authenticator, msg + 2, sizeof(message_authenticator));
821  break;
822  }
823 
824  /*
825  * Implement verification as a signature, followed by
826  * checking our signature against the sent one. This is
827  * slightly more CPU work than having verify-specific
828  * functions, but it ends up being cleaner in the code.
829  */
830  rcode = fr_radius_sign(packet, original, secret, secret_len);
831  if (rcode < 0) {
832  fr_strerror_printf_push("Failed calculating correct authenticator");
833  return -1;
834  }
835 
836  /*
837  * Check the Message-Authenticator first.
838  *
839  * If it's invalid, restore the original
840  * Message-Authenticator and Request Authenticator
841  * fields.
842  */
843  if ((msg < end) &&
844  (fr_digest_cmp(message_authenticator, msg + 2, sizeof(message_authenticator)) != 0)) {
845  memcpy(msg + 2, message_authenticator, sizeof(message_authenticator));
846  memcpy(packet + 4, request_authenticator, sizeof(request_authenticator));
847 
848  fr_strerror_printf("invalid Message-Authenticator (shared secret is incorrect)");
849  return -1;
850  }
851 
852  /*
853  * These are random numbers, so there's no point in
854  * comparing them.
855  */
856  if ((packet[0] == FR_CODE_ACCESS_REQUEST) || (packet[0] == FR_CODE_STATUS_SERVER)) {
857  return 0;
858  }
859 
860  /*
861  * Check the Request Authenticator.
862  */
863  if (fr_digest_cmp(request_authenticator, packet + 4, sizeof(request_authenticator)) != 0) {
864  memcpy(packet + 4, request_authenticator, sizeof(request_authenticator));
865  if (original) {
866  fr_strerror_printf("invalid Response Authenticator (shared secret is incorrect)");
867  } else {
868  fr_strerror_printf("invalid Request Authenticator (shared secret is incorrect)");
869  }
870  return -1;
871  }
872 
873  return 0;
874 }

908 ssize_t fr_radius_encode_dbuff(fr_dbuff_t *dbuff, uint8_t const *original,
909  char const *secret, UNUSED size_t secret_len, int code, int id, fr_pair_t *vps)
910 {
911  ssize_t slen;
912  fr_pair_t const *vp;
913  fr_cursor_t cursor;
914  fr_radius_ctx_t packet_ctx;
915  fr_dbuff_t work_dbuff, length_dbuff;
916 
917  packet_ctx.secret = secret;
918  packet_ctx.rand_ctx.a = fr_rand();
919  packet_ctx.rand_ctx.b = fr_rand();
920 
921  /*
922  * The RADIUS header can't do more than 64K of data.
923  */
924  work_dbuff = FR_DBUFF_MAX_NO_ADVANCE(dbuff, 65535);
925 
926  FR_DBUFF_IN_BYTES_RETURN(&work_dbuff, code, id);
927  length_dbuff = FR_DBUFF_NO_ADVANCE(&work_dbuff);
928  FR_DBUFF_IN_RETURN(&work_dbuff, (uint16_t) RADIUS_HEADER_LENGTH);
929 
930  switch (code) {
931  case FR_CODE_ACCESS_REQUEST:
932  case FR_CODE_STATUS_SERVER:
933  /*
934  * Callers in these cases have preloaded the buffer with the authentication vector.
935  */
936  FR_DBUFF_OUT_MEMCPY_RETURN(packet_ctx.vector, &work_dbuff, sizeof(packet_ctx.vector));
937  break;
938 
939  case FR_CODE_ACCESS_ACCEPT:
940  case FR_CODE_ACCESS_REJECT:
941  case FR_CODE_ACCESS_CHALLENGE:
942  case FR_CODE_ACCOUNTING_RESPONSE:
943  case FR_CODE_COA_ACK:
944  case FR_CODE_COA_NAK:
945  case FR_CODE_DISCONNECT_ACK:
946  case FR_CODE_DISCONNECT_NAK:
947  case FR_CODE_PROTOCOL_ERROR:
948  if (!original) {
949  fr_strerror_printf("Cannot encode response without request");
950  return -1;
951  }
952  memcpy(packet_ctx.vector, original + 4, sizeof(packet_ctx.vector));
953  FR_DBUFF_IN_MEMCPY_RETURN(&work_dbuff, packet_ctx.vector, RADIUS_AUTH_VECTOR_LENGTH);
954  break;
955 
956  case FR_CODE_ACCOUNTING_REQUEST:
957  case FR_CODE_COA_REQUEST:
958  case FR_CODE_DISCONNECT_REQUEST:
959  memset(packet_ctx.vector, 0, sizeof(packet_ctx.vector));
960  FR_DBUFF_MEMSET_RETURN(&work_dbuff, 0, RADIUS_AUTH_VECTOR_LENGTH);
961  break;
962 
963  default:
964  fr_strerror_printf("Cannot encode unknown packet code %d", code);
965  return -1;
966  }
967 
968  /*
969  * If we're sending Protocol-Error, add in
970  * Original-Packet-Code manually. If the user adds it
971  * later themselves, well, too bad.
972  */
973  if (code == FR_CODE_PROTOCOL_ERROR) {
974  FR_DBUFF_IN_BYTES_RETURN(&work_dbuff, FR_EXTENDED_ATTRIBUTE_1, 0x07, 0x04 /* Original-Packet-Code */,
975  0x00, 0x00, 0x00, original[0]);
976  }
977 
978  /*
979  * Loop over the reply attributes for the packet.
980  */
981  fr_cursor_talloc_iter_init(&cursor, &vps, fr_radius_next_encodable, dict_radius, fr_pair_t);
982  while ((vp = fr_cursor_current(&cursor))) {
983  VP_VERIFY(vp);
984 
985  /*
986  * Ignore non-wire attributes, but allow extended
987  * attributes.
988  */
989  if (vp->da->flags.internal) {
990 #ifndef NDEBUG
991  /*
992  * Permit the admin to send BADLY formatted
993  * attributes with a debug build.
994  */
995  if (vp->da == attr_raw_attribute) {
996  CHECK_FREESPACE(vp->vp_length, vp->vp_length);
997 
998  /*
999  * Skip really badly formatted attributes
1000  */
1001  if (vp->vp_length > (RADIUS_MAX_STRING_LENGTH + 2)) {
1002  fr_cursor_next(&cursor);
1003  continue;
1004  }
1005 
1006  FR_DBUFF_IN_MEMCPY_RETURN(&work_dbuff, vp->vp_octets, vp->vp_length);
1007  fr_cursor_next(&cursor);
1008  continue;
1009  }
1010 #endif
1011  /*
1012  * Skip internal attributes...
1013  */
1014  fr_cursor_next(&cursor);
1015  continue;
1016  }
1017 
1018  /*
1019  * Encode an individual VP
1020  */
1021  slen = fr_radius_encode_pair(&work_dbuff, &cursor, &packet_ctx);
1022  if (slen < 0) {
1023  if (slen == PAIR_ENCODE_SKIPPED) continue;
1024  return slen;
1025  }
1026  } /* done looping over all attributes */
1027 
1028  /*
1029  * Fill in the length field we zeroed out earlier.
1030  *
1031  */
1032  fr_dbuff_in(&length_dbuff, (uint16_t) (fr_dbuff_used(&work_dbuff)));
1033 
1034  FR_PROTO_HEX_DUMP(fr_dbuff_start(&work_dbuff), fr_dbuff_used(&work_dbuff), "%s encoded packet", __FUNCTION__);
1035 
1036  return fr_dbuff_set(dbuff, &work_dbuff);
1037 }
