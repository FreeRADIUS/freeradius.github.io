114 /** Call the callback registered for a timeout event
115  *
116  * @param[in] el the event timer was inserted into.
117  * @param[in] now The current time, as held by the event_list.
118  * @param[in] ctx unlang_module_event_t structure holding callbacks.
119  *
120  */
121 static void unlang_xlat_event_timeout_handler(UNUSED fr_event_list_t *el, fr_time_t now, void *ctx)
122 {
123  unlang_xlat_event_t *ev = talloc_get_type_abort(ctx, unlang_xlat_event_t);
124 
125  void *mutable_ctx;
126  void *mutable_inst;
127 
128  memcpy(&mutable_ctx, &ev->ctx, sizeof(mutable_ctx));
129  memcpy(&mutable_inst, &ev->inst, sizeof(mutable_inst));
130 
131  ev->timeout(ev->request, mutable_inst, ev->thread, mutable_ctx, now);
132 
133  /* Remove old references from the request */
134  talloc_free(ev);
135 }

137 /** Add a timeout for an xlat handler
138  *
139  * @note The timeout is automatically removed when the xlat is cancelled or resumed.
140  *
141  * @param[in] request the request
142  * @param[in] callback to run when the timeout hits
143  * @param[in] ctx passed to the callback
144  * @param[in] when when the timeout fires
145  * @return
146  * - <0 on error
147  * - 0 on success
148  */
149 int unlang_xlat_event_timeout_add(request_t *request, fr_unlang_xlat_timeout_t callback,
150  void const *ctx, fr_time_t when)
151 {
152  unlang_stack_t *stack = request->stack;
153  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
154  unlang_xlat_event_t *ev;
155  unlang_frame_state_xlat_t *state = talloc_get_type_abort(frame->state, unlang_frame_state_xlat_t);
156 
157  fr_assert(stack->depth > 0);
158  fr_assert(frame->instruction->type == UNLANG_TYPE_XLAT);
159 
160  if (!state->event_ctx) MEM(state->event_ctx = talloc_zero(state, bool));
161 
162  ev = talloc_zero(state->event_ctx, unlang_xlat_event_t);
163  if (!ev) return -1;
164 
165  ev->request = request;
166  ev->fd = -1;
167  ev->timeout = callback;
168  ev->inst = state->exp->call.inst;
169  ev->thread = xlat_thread_instance_find(state->exp);
170  ev->ctx = ctx;
171 
172  if (fr_event_timer_at(request, request->el, &ev->ev, when, unlang_xlat_event_timeout_handler, ev) < 0) {
173  RPEDEBUG("Failed inserting event");
174  talloc_free(ev);
175  return -1;
176  }
177 
178  talloc_set_destructor(ev, _unlang_xlat_event_free);
179 
180  return 0;
181 }

183 /** Push a pre-compiled xlat onto the stack for evaluation
184  *
185  * @param[in] ctx To allocate value boxes and values in.
186  * @param[out] out Where to write the result of the expansion.
187  * @param[in] request to push xlat onto.
188  * @param[in] exp node to evaluate.
189  * @param[in] top_frame Set to UNLANG_TOP_FRAME if the interpreter should return.
190  * Set to UNLANG_SUB_FRAME if the interprer should continue.
191  * @return
192  * - 0 on success.
193  * - -1 on failure.
194  */
195 int unlang_xlat_push(TALLOC_CTX *ctx, fr_value_box_t **out,
196  request_t *request, xlat_exp_t const *exp, bool top_frame)
197 {
198 
199  unlang_frame_state_xlat_t *state;
200  unlang_stack_t *stack = request->stack;
201  unlang_stack_frame_t *frame;
202 
203  /*
204  * Push a new xlat eval frame onto the stack
205  */
206  if (unlang_interpret_push(request, &xlat_instruction, RLM_MODULE_UNKNOWN, UNLANG_NEXT_STOP, top_frame) < 0) {
207  return -1;
208  }
209  frame = &stack->frame[stack->depth];
210 
211  /*
212  * Allocate its state, and setup a cursor for the xlat nodes
213  */
214  MEM(frame->state = state = talloc_zero(stack, unlang_frame_state_xlat_t));
215  state->exp = talloc_get_type_abort_const(exp, xlat_exp_t); /* Ensure the node is valid */
216 
217  fr_cursor_talloc_init(&state->values, out, fr_value_box_t);
218 
219  state->ctx = ctx;
220 
221  return 0;
222 }

224 /** Stub function for calling the xlat interpreter
225  *
226  * Calls the xlat interpreter and translates its wants and needs into
227  * unlang_action_t codes.
228  */
229 static unlang_action_t unlang_xlat(rlm_rcode_t *p_result, request_t *request)
230 {
231  unlang_stack_t *stack = request->stack;
232  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
233  unlang_frame_state_xlat_t *state = talloc_get_type_abort(frame->state, unlang_frame_state_xlat_t);
234  xlat_exp_t const *child = NULL;
235  xlat_action_t xa;
236 
237  if (is_repeatable(frame)) {
238  xa = xlat_frame_eval_repeat(state->ctx, &state->values, &child,
239  &state->alternate, request, &state->exp, &state->rhead);
240  } else {
241  xa = xlat_frame_eval(state->ctx, &state->values, &child, request, &state->exp);
242  }
243 
244  switch (xa) {
245  case XLAT_ACTION_PUSH_CHILD:
246  fr_assert(child);
247 
248  repeatable_set(frame);
249 
250  /*
251  * Clear out the results of any previous expansions
252  * at this level. A frame may be used to evaluate
253  * multiple sibling nodes.
254  */
255  talloc_list_free(&state->rhead);
256  if (unlang_xlat_push(state->ctx, &state->rhead, request, child, false) < 0) {
257  *p_result = RLM_MODULE_FAIL;
258  return UNLANG_ACTION_STOP_PROCESSING;
259  }
260  return UNLANG_ACTION_PUSHED_CHILD;
261 
262  case XLAT_ACTION_YIELD:
263  if (!state->resume) {
264  RWDEBUG("Missing call to unlang_xlat_yield()");
265  goto fail;
266  }
267  return UNLANG_ACTION_YIELD;
268 
269  case XLAT_ACTION_DONE:
270  *p_result = RLM_MODULE_OK;
271  return UNLANG_ACTION_CALCULATE_RESULT;
272 
273  case XLAT_ACTION_FAIL:
274  fail:
275  *p_result = RLM_MODULE_FAIL;
276  return UNLANG_ACTION_CALCULATE_RESULT;
277  }
278 
279  fr_assert(0);
280  *p_result = RLM_MODULE_FAIL;
281  return UNLANG_ACTION_CALCULATE_RESULT;
282 }

284 /** Send a signal (usually stop) to a request that's running an xlat expansions
285  *
286  * This is typically called via an "async" action, i.e. an action
287  * outside of the normal processing of the request.
288  *
289  * If there is no #xlat_func_signal_t callback defined, the action is ignored.
290  *
291  * @param[in] request The current request.
292  * @param[in] action What the request should do (the type of signal).
293  */
294 static void unlang_xlat_signal(request_t *request, fr_state_signal_t action)
295 {
296  unlang_stack_t *stack = request->stack;
297  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
298  unlang_frame_state_xlat_t *state = talloc_get_type_abort(frame->state, unlang_frame_state_xlat_t);
299 
300  /*
301  * Delete timers, etc. when the xlat is cancelled.
302  */
303  if (action == FR_SIGNAL_CANCEL) {
304  TALLOC_FREE(state->event_ctx);
305  }
306 
307  if (!state->signal) return;
308 
309  xlat_signal(state->signal, state->exp, request, state->rctx, action);
310 }

312 /** Called when we're ready to resume processing the request
313  *
314  * @param[in] p_result the result of the xlat function.
315  * - RLM_MODULE_OK on success.
316  * - RLM_MODULE_FAIL on failure.
317  * - RLM_MODULE_YIELD if additional asynchronous
318  * operations need to be performed.
319  * @param[in] request to resume processing.
320  * @return
321  * - UNLANG_ACTION_YIELD if yielding.
322  * - UNLANG_ACTION_CALCULATE_RESULT if done.
323  */
324 static unlang_action_t unlang_xlat_resume(rlm_rcode_t *p_result, request_t *request)
325 {
326  unlang_stack_t *stack = request->stack;
327  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
328  unlang_frame_state_xlat_t *state = talloc_get_type_abort(frame->state, unlang_frame_state_xlat_t);
329  xlat_action_t xa;
330 
331  fr_assert(state->resume != NULL);
332 
333  /*
334  * Delete timers, etc. when the xlat is resumed.
335  */
336  TALLOC_FREE(state->event_ctx);
337 
338  xa = xlat_frame_eval_resume(state->ctx, &state->values,
339  state->resume, state->exp,
340  request, &state->rhead, state->rctx);
341  switch (xa) {
342  case XLAT_ACTION_YIELD:
343  return UNLANG_ACTION_YIELD;
344 
345  case XLAT_ACTION_DONE:
346  *p_result = RLM_MODULE_OK;
347  return UNLANG_ACTION_CALCULATE_RESULT;
348 
349  case XLAT_ACTION_PUSH_CHILD:
350  fr_assert(0);
351  FALL_THROUGH;
352 
353  case XLAT_ACTION_FAIL:
354  *p_result = RLM_MODULE_FAIL;
355  return UNLANG_ACTION_CALCULATE_RESULT;
356  /* DON'T SET DEFAULT */
357  }
358 
359  fr_assert(0); /* Garbage xlat action */
360 
361  *p_result = RLM_MODULE_FAIL;
362  return UNLANG_ACTION_CALCULATE_RESULT;
363 }

365 /** Yield a request back to the interpreter from within a module
366  *
367  * This passes control of the request back to the unlang interpreter, setting
368  * callbacks to execute when the request is 'signalled' asynchronously, or whatever
369  * timer or I/O event the module was waiting for occurs.
370  *
371  * @note The module function which calls #unlang_module_yield should return control
372  * of the C stack to the unlang interpreter immediately after calling #unlang_xlat_yield.
373  * A common pattern is to use ``return unlang_xlat_yield(...)``.
374  *
375  * @param[in] request The current request.
376  * @param[in] resume Called on unlang_interpret_resumable().
377  * @param[in] signal Called on unlang_action().
378  * @param[in] rctx to pass to the callbacks.
379  * @return always returns RLM_MODULE_YIELD.
380  */
381 xlat_action_t unlang_xlat_yield(request_t *request,
382  xlat_func_resume_t resume, xlat_func_signal_t signal,
383  void *rctx)
384 {
385  unlang_stack_t *stack = request->stack;
386  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
387  unlang_frame_state_xlat_t *state = talloc_get_type_abort(frame->state, unlang_frame_state_xlat_t);
388 
389  frame->process = unlang_xlat_resume;
390 
391  /*
392  * Over-ride whatever functions were there before.
393  */
394  state->resume = resume;
395  state->signal = signal;
396  state->rctx = rctx;
397 
398  return XLAT_ACTION_YIELD;
399 }
