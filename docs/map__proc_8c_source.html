35 /** Compare two map_proc_t structs, based ONLY on the name
36  *
37  * @param[in] one First map struct.
38  * @param[in] two Second map struct.
39  * @return Integer specifying order of map func instances.
40  */
41 static int map_proc_cmp(void const *one, void const *two)
42 {
43  map_proc_t const *a = one, *b = two;
44 
45  if (a->length != b->length) return a->length - b->length;
46 
47  return memcmp(a->name, b->name, a->length);
48 }

96 /** Register a map processor
97  *
98  * This should be called by every module that provides a map processing function.
99  *
100  * @param[in] mod_inst of module registering the map_proc.
101  * @param[in] name of map processor. If processor already exists, it is replaced.
102  * @param[in] evaluate Module's map processor function.
103  * @param[in] instantiate function (optional).
104  * @param[in] inst_size of talloc chunk to allocate for instance data (optional).
105  * @return
106  * - 0 on success.
107  * - -1 on failure.
108  */
109 int map_proc_register(void *mod_inst, char const *name,
110  map_proc_func_t evaluate,
111  map_proc_instantiate_t instantiate, size_t inst_size)
112 {
113  map_proc_t *proc;
114 
115  fr_assert(name && name[0]);
116 
117  if (!map_proc_root) {
118  map_proc_root = rbtree_talloc_alloc(NULL, map_proc_cmp, map_proc_t,
119  _map_proc_tree_free, RBTREE_FLAG_REPLACE);
120  if (!map_proc_root) {
121  DEBUG("map_proc: Failed to create tree");
122  return -1;
123  }
124  }
125 
126  /*
127  * If it already exists, replace it.
128  */
129  proc = map_proc_find(name);
130  if (!proc) {
131  rbnode_t *node;
132 
133  proc = talloc_zero(mod_inst, map_proc_t);
134  strlcpy(proc->name, name, sizeof(proc->name));
135  proc->length = strlen(proc->name);
136 
137  node = rbtree_insert_node(map_proc_root, proc);
138  if (!node) {
139  talloc_free(proc);
140  return -1;
141  }
142 
143  talloc_set_destructor(proc, _map_proc_talloc_free);
144  }
145 
146  DEBUG3("map_proc_register: %s", proc->name);
147 
148  proc->mod_inst = mod_inst;
149  proc->evaluate = evaluate;
150  proc->instantiate = instantiate;
151  proc->inst_size = inst_size;
152 
153  return 0;
154 }

156 /** Create a new map proc instance
157  *
158  * This should be called for every map {} section in the configuration.
159  *
160  * @param[in] ctx to allocate proc instance in.
161  * @param[in] proc resolved with #map_proc_find.
162  * @param[in] cs #CONF_SECTION representing this instance of a map processor.
163  * @param[in] src template.
164  * @param[in] maps Head of the list of maps.
165  * @return
166  * - New #map_proc_inst_t on success.
167  * - NULL on error.
168  */
169 map_proc_inst_t *map_proc_instantiate(TALLOC_CTX *ctx, map_proc_t const *proc,
170  CONF_SECTION *cs, tmpl_t const *src, map_t const *maps)
171 {
172  map_proc_inst_t *inst;
173 
174  inst = talloc_zero(ctx, map_proc_inst_t);
175  inst->proc = proc;
176  inst->src = src;
177  inst->maps = maps;
178 
179  if (proc->instantiate) {
180  if (proc->inst_size > 0) {
181  inst->data = talloc_zero_array(inst, uint8_t, proc->inst_size);
182  if (!inst->data) return NULL;
183  }
184 
185  if (proc->instantiate(cs, proc->mod_inst, inst->data, src, maps) < 0) {
186  talloc_free(inst);
187  return NULL;
188  }
189  }
190 
191  return inst;
192 }
