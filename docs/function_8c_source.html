59 /** Call a generic function
60  *
61  * @param[out] p_result The frame result.
62  * @param[in] request The current request.
63  */
64 static unlang_action_t unlang_function_call(rlm_rcode_t *p_result, request_t *request)
65 {
66  unlang_stack_t *stack = request->stack;
67  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
68  unlang_frame_state_func_t *state = talloc_get_type_abort(frame->state, unlang_frame_state_func_t);
69  unlang_action_t ua;
70  char const *caller;
71 
72  /*
73  * Don't let the callback mess with the current
74  * module permanently.
75  */
76  caller = request->module;
77  request->module = NULL;
78  if (!is_repeatable(frame)) {
79  ua = state->func(p_result, NULL, request, state->uctx);
80  } else {
81  ua = state->repeat(p_result, NULL, request, state->uctx);
82  }
83  request->module = caller;
84 
85  return ua;
86 }

88 /** Push a generic function onto the unlang stack
89  *
90  * These can be pushed by any other type of unlang op to allow a submodule or function
91  * deeper in the C call stack to establish a new resumption point.
92  *
93  * @param[in] request The current request.
94  * @param[in] func to call going up the stack.
95  * @param[in] repeat function to call going back down the stack (may be NULL).
96  * This may be the same as func.
97  * @param[in] uctx to pass to func.
98  * @return
99  * - 0 on success.
100  * - -1 on failure.
101  */
102 int unlang_interpret_push_function(request_t *request, unlang_function_t func, unlang_function_t repeat, void *uctx)
103 {
104  unlang_stack_t *stack = request->stack;
105  unlang_stack_frame_t *frame;
106  unlang_frame_state_func_t *state;
107 
108  /*
109  * Push module's function
110  */
111  if (unlang_interpret_push(request, &function_instruction,
112  RLM_MODULE_UNKNOWN, UNLANG_NEXT_STOP, UNLANG_SUB_FRAME) < 0) return -1;
113 
114  frame = &stack->frame[stack->depth];
115 
116  /*
117  * Tell the interpreter to call unlang_function_call
118  * again when going back up the stack.
119  */
120  if (repeat) repeatable_set(frame);
121 
122  /*
123  * Initialize state
124  */
125  state = frame->state;
126  state->func = func;
127  state->repeat = repeat;
128  state->uctx = uctx;
129 
130  return 0;
131 }
