81 int fb_error(rlm_sql_firebird_conn_t *conn)
82 {
83  ISC_SCHAR error[2048]; /* Only 1024 bytes should be written to this, but were playing it extra safe */
84  ISC_STATUS *pstatus;
85 
86  conn->sql_code = 0;
87 
88  /*
89  * Free any previous errors.
90  */
91  TALLOC_FREE(conn->error);
92 
93  /*
94  * Check if the status array contains an error
95  */
96  if (IS_ISC_ERROR(conn->status)) {
97  conn->sql_code = isc_sqlcode(conn->status);
98 
99  /*
100  * pstatus is a pointer into the status array which is
101  * advanced by isc_interprete. It's initialised to the
102  * first element of the status array.
103  */
104  pstatus = &conn->status[0];
105 
106  /*
107  * It's deprecated because the size of the buffer isn't
108  * passed and this isn't safe. But as were passing a very
109  * large buffer it's unlikely this will be an issue, and
110  * allows us to maintain compatibility with the interbase
111  * API.
112  */
113  isc_interprete(&error[0], &pstatus);
114  conn->error = talloc_typed_asprintf(conn, "%s. ", &error[0]);
115 
116  while (isc_interprete(&error[0], &pstatus)) {
117  conn->error = talloc_asprintf_append(conn->error, "%s. ", &error[0]);
118  }
119 
120  memset(&conn->status, 0, sizeof(conn->status));
121  }
122 
123  return conn->sql_code;
124 }

149 void fb_store_row(rlm_sql_firebird_conn_t *conn)
150 {
151  int dtype;
152  struct tm times;
153  ISC_QUAD bid;
154  int i;
155  XSQLVAR *var;
156  VARY * vary;
157 
158  /* assumed: id, username, attribute, value, op */
159  if (conn->row_fcount<conn->sqlda_out->sqld) {
160  i = conn->row_fcount;
161  conn->row_fcount = conn->sqlda_out->sqld;
162  conn->row = (char **) realloc(conn->row, conn->row_fcount * sizeof(char *));
163  conn->row_sizes = (int *) realloc(conn->row_sizes, conn->row_fcount * sizeof(int));
164 
165  while( i <conn->row_fcount) {
166  conn->row[i] = 0;
167  conn->row_sizes[i++] = 0;
168  }
169  }
170 
171  for (i = 0, var = conn->sqlda_out->sqlvar; i < conn->sqlda_out->sqld; var++, i++) {
172  /*
173  * Initial buffer size to store field's data is 256 bytes
174  */
175  if (conn->row_sizes[i]<256) {
176  conn->row[i] = (char *) realloc(conn->row[i], 256);
177  conn->row_sizes[i] = 256;
178  }
179 
180  if (IS_NULL(var)) {
181  strcpy(conn->row[i], "NULL");
182  continue;
183  }
184 
185  dtype = var->sqltype & ~1;
186 
187  switch (dtype) {
188  case SQL_TEXT:
189  if (conn->row_sizes[i]<= var->sqllen) {
190  conn->row_sizes[i] = var->sqllen + 1;
191  conn->row[i] = realloc(conn->row[i],
192  conn->row_sizes[i]);
193  }
194 
195  memmove(conn->row[i], var->sqldata, var->sqllen);
196  conn->row[i][var->sqllen] = 0;
197  break;
198 
199  case SQL_VARYING:
200  vary = (VARY*) var->sqldata;
201  if (conn->row_sizes[i] <= vary->vary_length) {
202  conn->row_sizes[i] = vary->vary_length + 1;
203  conn->row[i] = realloc(conn->row[i],
204  conn->row_sizes[i]);
205  }
206  memmove(conn->row[i], vary->vary_string, vary->vary_length);
207  conn->row[i][vary->vary_length] = 0;
208  break;
209 
210  case SQL_FLOAT:
211  snprintf(conn->row[i], conn->row_sizes[i], "%15g",
212  *(double ISC_FAR *) (var->sqldata));
213  break;
214 
215  case SQL_SHORT:
216  case SQL_LONG:
217  case SQL_INT64:
218  {
219  ISC_INT64 value = 0;
220  short field_width = 0;
221  short dscale = 0;
222  char *p;
223  p = conn->row[i];
224 
225  switch (dtype) {
226  case SQL_SHORT:
227  value = (ISC_INT64) *(short *)var->sqldata;
228  field_width = 6;
229  break;
230 
231  case SQL_LONG:
232  value = (ISC_INT64) *(int *)var->sqldata;
233  field_width = 11;
234  break;
235 
236  case SQL_INT64:
237  value = (ISC_INT64) *(ISC_INT64 *)var->sqldata;
238  field_width = 21;
239  break;
240  }
241  dscale = var->sqlscale;
242 
243  if (dscale < 0) {
244  ISC_INT64 tens;
245  short j;
246 
247  tens = 1;
248  for (j = 0; j > dscale; j--) {
249  tens *= 10;
250  }
251 
252  if (value >= 0) {
253  sprintf(p, "%*lld.%0*lld",
254  field_width - 1 + dscale,
255  (ISC_INT64) value / tens,
256  -dscale,
257  (ISC_INT64) value % tens);
258  } else if ((value / tens) != 0) {
259  sprintf (p, "%*lld.%0*lld",
260  field_width - 1 + dscale,
261  (ISC_INT64) (value / tens),
262  -dscale,
263  (ISC_INT64) -(value % tens));
264  } else {
265  sprintf(p, "%*s.%0*lld", field_width - 1 + dscale,
266  "-0", -dscale, (ISC_INT64) - (value % tens));
267  }
268  } else if (dscale) {
269  sprintf(p, "%*lld%0*d", field_width,
270  (ISC_INT64) value, dscale, 0);
271  } else {
272  sprintf(p, "%*lld", field_width,
273  (ISC_INT64) value);
274  }
275  }
276  break;
277 
278  case SQL_D_FLOAT:
279  case SQL_DOUBLE:
280  snprintf(conn->row[i], conn->row_sizes[i], "%24f",
281  *(double ISC_FAR *) (var->sqldata));
282  break;
283 
284  case SQL_TIMESTAMP:
285  isc_decode_timestamp((ISC_TIMESTAMP ISC_FAR *)var->sqldata, &times);
286  snprintf(conn->row[i], conn->row_sizes[i], "%04d-%02d-%02d %02d:%02d:%02d.%04d",
287  times.tm_year + 1900,
288  times.tm_mon + 1,
289  times.tm_mday,
290  times.tm_hour,
291  times.tm_min,
292  times.tm_sec,
293  ((ISC_TIMESTAMP *)var->sqldata)->timestamp_time % 10000);
294  break;
295 
296  case SQL_TYPE_DATE:
297  isc_decode_sql_date((ISC_DATE ISC_FAR *)var->sqldata, &times);
298  snprintf(conn->row[i], conn->row_sizes[i], "%04d-%02d-%02d",
299  times.tm_year + 1900,
300  times.tm_mon + 1,
301  times.tm_mday);
302  break;
303 
304  case SQL_TYPE_TIME:
305  isc_decode_sql_time((ISC_TIME ISC_FAR *)var->sqldata, &times);
306  snprintf(conn->row[i], conn->row_sizes[i], "%02d:%02d:%02d.%04d",
307  times.tm_hour,
308  times.tm_min,
309  times.tm_sec,
310  (*((ISC_TIME *)var->sqldata)) % 10000);
311  break;
312 
313  case SQL_BLOB:
314  case SQL_ARRAY:
315  /* Print the blob id on blobs or arrays */
316  bid = *(ISC_QUAD ISC_FAR *) var->sqldata;
317  snprintf(conn->row[i], conn->row_sizes[i], "%08" ISC_LONG_FMT "x:%08" ISC_LONG_FMT "x",
318  bid.gds_quad_high, bid.gds_quad_low);
319  break;
320  }
321  }

347 int fb_connect(rlm_sql_firebird_conn_t *conn, rlm_sql_config_t *config)
348 {
349  char *p;
350  char *database;
351 
352  conn->dpb_len = 4;
353  if (config->sql_login) {
354  conn->dpb_len+= strlen(config->sql_login) + 2;
355  }
356 
357  if (config->sql_password) {
358  conn->dpb_len += strlen(config->sql_password) + 2;
359  }
360 
361  MEM(conn->dpb = (char *) malloc(conn->dpb_len));
362  p = conn->dpb;
363 
364  *conn->dpb++= isc_dpb_version1;
365  *conn->dpb++= isc_dpb_num_buffers;
366  *conn->dpb++= 1;
367  *conn->dpb++= 90;
368 
369  fb_dpb_add_str(&conn->dpb, isc_dpb_user_name, config->sql_login);
370  fb_dpb_add_str(&conn->dpb, isc_dpb_password, config->sql_password);
371 
372  conn->dpb = p;
373 
374  /*
375  * Check if database and server in the form of server:database.
376  * If config->sql_server contains ':', then config->sql_db
377  * parameter ignored.
378  */
379  if (strchr(config->sql_server, ':')) {
380  database = strdup(config->sql_server);
381  } else {
382  /*
383  * Make database and server to be in the form
384  * of server:database
385  */
386  int ls = strlen(config->sql_server);
387  int ld = strlen(config->sql_db);
388  MEM(database = (char *) calloc(ls + ld + 2, 1));
389  strcpy(database, config->sql_server);
390  database[ls] = ':';
391  memmove(database + ls + 1, config->sql_db, ld);
392  }
393  isc_attach_database(conn->status, 0, database, &conn->dbh,
394  conn->dpb_len, conn->dpb);
395  free(database);
396 
397  return fb_error(conn);
398 }

421 static int fb_prepare(rlm_sql_firebird_conn_t *conn, char const *query)
422 {
423  static char stmt_info[] = { isc_info_sql_stmt_type };
424  char info_buffer[128];
425  short l;
426 
427  if (!conn->trh) {
428  isc_start_transaction(conn->status, &conn->trh, 1, &conn->dbh,
429  conn->tpb_len, conn->tpb);
430  if (!conn->trh) {
431  return -4;
432  }
433  }
434 
435  fb_free_statement(conn);
436  if (!conn->stmt) {
437  isc_dsql_allocate_statement(conn->status, &conn->dbh,
438  &conn->stmt);
439  if (!conn->stmt) {
440  return -1;
441  }
442  }
443 
444  fb_free_sqlda(conn->sqlda_out);
445  isc_dsql_prepare(conn->status, &conn->trh, &conn->stmt, 0, query,
446  conn->sql_dialect, conn->sqlda_out);
447  if (IS_ISC_ERROR(conn->status)) {
448  return -2;
449  }
450 
451  if (conn->sqlda_out->sqln<conn->sqlda_out->sqld) {
452  conn->sqlda_out->sqln = conn->sqlda_out->sqld;
453  conn->sqlda_out = (XSQLDA ISC_FAR *) realloc(conn->sqlda_out,
454  XSQLDA_LENGTH(conn->sqlda_out->sqld));
455  isc_dsql_describe(conn->status, &conn->stmt, SQL_DIALECT_V6,
456  conn->sqlda_out);
457 
458  if (IS_ISC_ERROR(conn->status)) {
459  return -3;
460  }
461  }
462  /*
463  * Get statement type
464  */
465  isc_dsql_sql_info(conn->status, &conn->stmt, sizeof(stmt_info),
466  stmt_info, sizeof(info_buffer), info_buffer);
467  if (IS_ISC_ERROR(conn->status)) return -4;
468 
469  l = (short) isc_vax_integer((char ISC_FAR *) info_buffer + 1, 2);
470  conn->statement_type = isc_vax_integer((char ISC_FAR *) info_buffer + 3,
471  l);
472 
473  if (conn->sqlda_out->sqld) {
474  fb_set_sqlda(conn->sqlda_out); //set out sqlda
475  }
476 
477  return 0;
478 }
