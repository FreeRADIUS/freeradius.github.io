45 int LLVMFuzzerInitialize(int *argc, char ***argv)
46 {
47  dl_t *dl;
48  dl_loader_t *dl_loader;
49  char const *lib_dir = getenv("FR_LIBRARY_PATH");
50  char const *proto = getenv("FR_LIBRARY_FUZZ_PROTOCOL");
51  char const *dict_dir = getenv("FR_DICTIONARY_DIR");
52  char buffer[1024];
53 
54  if (!argc || !argv || !*argv) return -1; /* shut up clang scan */
55 
56  /*
57  * Get the name from the binary name of fuzzer_foo
58  */
59  if (!proto) {
60  proto = strrchr((*argv)[0], '_');
61  if (proto) proto++;
62  }
63 
64  /*
65  * Look for -D dir
66  *
67  * If found, nuke it from the argument list.
68  */
69  if (!dict_dir) {
70  int i, j;
71 
72  for (i = 0; i < *argc - 1; i++) {
73  char *p = (*argv)[i];
74 
75  if ((p[0] == '-') && (p[1] == 'D')) {
76  dict_dir = (*argv)[i + 1];
77 
78  for (j = i + 2; j < *argc; j++) {
79  (*argv)[i] = (*argv)[j];
80  }
81 
82  *argc -= 2;
83  break;
84  }
85  }
86  }
87 
88  if (!dict_dir) dict_dir = DICTDIR;
89 
90  if (!fr_dict_global_ctx_init(NULL, dict_dir)) {
91  fr_perror("dict_global");
92  return 0;
93  }
94 
95  if (fr_dict_internal_afrom_file(&dict, FR_DICTIONARY_INTERNAL_DIR) < 0) {
96  fprintf(stderr, "fuzzer: Failed initializing internal dictionary: %s\n",
97  fr_strerror());
98  fr_exit_now(1);
99  }
100 
101  if (!proto) {
102  fprintf(stderr, "Failed to find protocol for fuzzer\n");
103  fr_exit_now(1);
104  }
105 
106  dl_loader = dl_loader_init(NULL, NULL, 0, false);
107  if (!dl_loader) {
108  fprintf(stderr, "fuzzer: Failed initializing library loader: %s\n",
109  fr_strerror());
110  fr_exit_now(1);
111  }
112  dl_search_path_prepend(dl_loader, lib_dir);
113 
114  snprintf(buffer, sizeof(buffer), "libfreeradius-%s", proto);
115  dl = dl_by_name(dl_loader, buffer, NULL, false);
116  if (!dl) {
117  fprintf(stderr, "fuzzer: Failed loading library %s: %s\n",
118  buffer, fr_strerror());
119  fr_exit_now(1);
120  }
121 
122  snprintf(buffer, sizeof(buffer), "%s_tp_decode_proto", proto);
123 
124  tp = dlsym(dl->handle, buffer);
125  if (!tp) {
126  fprintf(stderr, "fuzzer: Failed finding test point %s: %s\n",
127  buffer, fr_strerror());
128  fr_exit_now(1);
129  }
130 
131  if (tp->test_ctx(&decode_ctx, NULL) < 0) {
132  fprintf(stderr, "fuzzer: Failed finding test point %s: %s\n",
133  buffer, fr_strerror());
134  fr_exit_now(1);
135  }
136 
137  init = true;
138 
139  return 1;
140 }
