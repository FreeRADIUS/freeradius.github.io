73 static int process_file(char const *filename)
74 {
75  int rcode;
76  char const *name1, *name2;
77  CONF_SECTION *cs;
78  map_t *head, *map;
79  char buffer[8192];
80 
81  main_config_t *config;
82 
83  tmpl_rules_t parse_rules = {
84  .dict_def = dict_radius,
85  .allow_foreign = true /* Because we don't know what protocol we're operating with */
86  };
87 
88  /*
89  * Must be called first, so the handler is called last
90  */
91  fr_thread_local_atexit_setup();
92 
93  config = main_config_alloc(NULL);
94  if (!config) {
95  fprintf(stderr, "Failed allocating main config");
96  return EXIT_FAILURE;
97  }
98  config->root_cs = cf_section_alloc(config, NULL, "main", NULL);
99  if ((cf_file_read(config->root_cs, filename) < 0) || (cf_section_pass2(config->root_cs) < 0)) {
100  fprintf(stderr, "unit_test_map: Failed parsing %s\n", filename);
101  return EXIT_FAILURE;
102  }
103 
104  main_config_name_set_default(config, "unit_test_map", false);
105 
106  /*
107  * Always has to be an "update" section.
108  */
109  cs = cf_section_find(config->root_cs, "update", CF_IDENT_ANY);
110  if (!cs) {
111  talloc_free(config->root_cs);
112  return EXIT_FAILURE;
113  }
114 
115  /*
116  * Convert the update section to a list of maps.
117  */
118  rcode = map_afrom_cs(cs, &head, cs, &parse_rules, &parse_rules, unlang_fixup_update, NULL, 128);
119  if (rcode < 0) {
120  cf_log_perr(cs, "map_afrom_cs failed");
121  return EXIT_FAILURE; /* message already printed */
122  }
123  if (!head) {
124  cf_log_err(cs, "'update' sections cannot be empty");
125  return EXIT_FAILURE;
126  }
127 
128  buffer[0] = '\t';
129 
130  name1 = cf_section_name1(cs);
131  name2 = cf_section_name2(cs);
132 
133  /*
134  * And print it all out.
135  */
136  if (!name2) {
137  printf("%s {\n", name1);
138  } else {
139  printf("%s %s {\n", name1, name2);
140  }
141 
142  for (map = head; map != NULL; map = map->next) {
143  map_print(&FR_SBUFF_OUT(buffer + 1, sizeof(buffer) - 1), map);
144  puts(buffer);
145  }
146  printf("}\n");
147 
148  talloc_free(config->root_cs);
149  talloc_free(config);
150 
151  return EXIT_SUCCESS;
152 }

154 int main(int argc, char *argv[])
155 {
156  int c, ret = EXIT_SUCCESS;
157  char const *raddb_dir = RADDBDIR;
158  char const *dict_dir = DICTDIR;
159  fr_dict_t *dict = NULL;
160  char const *receipt_file = NULL;
161 
162  TALLOC_CTX *autofree;
163 
164  /*
165  * Must be called first, so the handler is called last
166  */
167  fr_thread_local_atexit_setup();
168 
169  autofree = talloc_autofree_context();
170 
171 #ifndef NDEBUG
172  if (fr_fault_setup(autofree, getenv("PANIC_ACTION"), argv[0]) < 0) {
173  fr_perror("unit_test_map");
174  fr_exit(EXIT_FAILURE);
175  }
176 #else
177  fr_disable_null_tracking_on_free(autofree);
178 #endif
179 
180  while ((c = getopt(argc, argv, "d:D:xMhr:")) != -1) switch (c) {
181  case 'd':
182  raddb_dir = optarg;
183  break;
184 
185  case 'D':
186  dict_dir = optarg;
187  break;
188 
189  case 'x':
190  fr_debug_lvl++;
191  break;
192 
193  case 'M':
194  talloc_enable_leak_report();
195  break;
196 
197  case 'r':
198  receipt_file = optarg;
199  break;
200 
201  case 'h':
202  default:
203  usage(argv);
204  }
205  argc -= (optind - 1);
206  argv += (optind - 1);
207 
208  if (receipt_file && (fr_unlink(receipt_file) < 0)) {
209  fr_perror("unit_test_map");
210  EXIT_WITH_FAILURE;
211  }
212 
213  /*
214  * Mismatch between the binary and the libraries it depends on
215  */
216  if (fr_check_lib_magic(RADIUSD_MAGIC_NUMBER) < 0) {
217  fr_perror("unit_test_map");
218  EXIT_WITH_FAILURE;
219  }
220 
221  if (!fr_dict_global_ctx_init(autofree, dict_dir)) {
222  fr_perror("unit_test_map");
223  EXIT_WITH_FAILURE;
224  }
225 
226  if (fr_dict_internal_afrom_file(&dict, FR_DICTIONARY_INTERNAL_DIR) < 0) {
227  fr_perror("unit_test_map");
228  EXIT_WITH_FAILURE;
229  }
230 
231  /*
232  * Load the custom dictionary
233  */
234  if (fr_dict_read(dict, raddb_dir, FR_DICTIONARY_FILE) == -1) {
235  fr_strerror_printf_push("Failed to initialize the dictionaries");
236  fr_perror("unit_test_map");
237  EXIT_WITH_FAILURE;
238  }
239 
240  if (fr_dict_autoload(unit_test_module_dict) < 0) {
241  fr_perror("unit_test_map");
242  EXIT_WITH_FAILURE;
243  }
244 
245  if (argc < 2) {
246  ret = process_file("-");
247 
248  } else {
249  ret = process_file(argv[1]);
250  }
251 
252  if (ret < 0) ret = 1; /* internal to Unix process return code */
253 
254 cleanup:
255  /*
256  * Try really hard to free any allocated
257  * memory, so we get clean talloc reports.
258  */
259  xlat_free();
260 
261  /*
262  * Free any autoload dictionaries
263  */
264  fr_dict_autofree(unit_test_module_dict);
265 
266  fr_dict_free(&dict);
267 
268  if (receipt_file && (ret == EXIT_SUCCESS) && (fr_touch(NULL, receipt_file, 0644, true, 0755) <= 0)) {
269  fr_perror("unit_test_map");
270  ret = EXIT_FAILURE;
271  }
272 
273  return ret;
274 }
