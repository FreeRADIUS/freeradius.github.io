36 /** Callback for implementing custom iterators
37  *
38  * @param[in,out] prev the attribute to the one passed as to_eval.
39  * Must be updated to the previous attribute to the one
40  * returned.
41  * @param[in] to_eval the next item in the list. Iterator should check to
42  * see if it matches the iterator's filter, and if it doesn't

44  * @param[in] uctx passed to #fr_cursor_init.

204 /** Allocate a stack of cursors for traversing trees
205  *
206  * @param[in] ctx to allocate the cursor stack in.
207  * @param[in] depth Maximum depth of the cursor stack.
208  * @return
209  * - A new cursor stack.
210  * - NULL on error.
211  */
212 static inline fr_cursor_stack_t *fr_cursor_stack_alloc(TALLOC_CTX *ctx, uint8_t depth)
213 {
214  fr_cursor_stack_t *stack;
215 
216  stack = talloc_array_size(ctx, sizeof(fr_cursor_stack_t) + (sizeof(fr_cursor_t) * depth), 1);
217  if (unlikely(!stack)) return NULL;
218 
219  talloc_set_name_const(stack, "fr_cursor_stack_t");
220 
221  return stack;
222 }
