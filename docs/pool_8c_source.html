76 /** A connection pool
77  *
78  * Defines the configuration of the connection pool, all the counters and
79  * timestamps related to the connection pool, the mutex that stops multiple
80  * threads leaving the pool in an inconsistent state, and the callbacks
81  * required to open, close and check the status of connections within the pool.
82  *
83  * @see fr_connection
84  */
85 struct fr_pool_s {
86  int ref; //!< Reference counter to prevent connection
87  //!< pool being freed multiple times.
88  uint32_t start; //!< Number of initial connections.
89  uint32_t min; //!< Minimum number of concurrent connections to keep open.
90  uint32_t max; //!< Maximum number of concurrent connections to allow.
91  uint32_t max_pending; //!< Max number of pending connections to allow.
92  uint32_t spare; //!< Number of spare connections to try.
93  uint64_t max_uses; //!< Maximum number of times a connection can be used
94  //!< before being closed.
95  uint32_t pending_window; //!< Sliding window of pending connections.
96 
97  fr_time_delta_t retry_delay; //!< seconds to delay re-open after a failed open.
98  fr_time_delta_t cleanup_interval; //!< Initial timer for how often we sweep the pool
99  //!< for free connections. (0 is infinite).
100  fr_time_delta_t delay_interval; //!< When we next do a cleanup. Initialized to
101  //!< cleanup_interval, and increase from there based
102  //!< on the delay.
103  fr_time_delta_t lifetime; //!< How long a connection can be open before being
104  //!< closed (irrespective of whether it's idle or not).
105  fr_time_delta_t idle_timeout; //!< How long a connection can be idle before
106  //!< being closed.
107  fr_time_delta_t connect_timeout; //!< New connection timeout, enforced by the create
108  //!< callback.
109 
110  bool spread; //!< If true we spread requests over the connections,
111  //!< using the connection released longest ago, first.
112 
113  fr_heap_t *heap; //!< For the next connection heap
114 
115  fr_pool_connection_t *head; //!< Start of the connection list.
116  fr_pool_connection_t *tail; //!< End of the connection list.
117 
118  pthread_mutex_t mutex; //!< Mutex used to keep consistent state when making
119  //!< modifications in threaded mode.
120  pthread_cond_t done_spawn; //!< Threads that need to ensure no spawning is in progress,

125  CONF_SECTION const *cs; //!< Configuration section holding the section of parsed

127  void *opaque; //!< Pointer to context data that will be passed to callbacks.

129  char const *log_prefix; //!< Log prefix to prepend to all log messages created

137 
138  fr_time_delta_t held_trigger_min; //!< If a connection is held for less than the specified

143  fr_pool_connection_create_t create; //!< Function used to create new connections.

187 /** Removes a connection from the connection list
188  *
189  * @note Must be called with the mutex held.
190  *
191  * @param[in] pool to modify.
192  * @param[in] this Connection to delete.
193  */
194 static void connection_unlink(fr_pool_t *pool, fr_pool_connection_t *this)
195 {
196  if (this->prev) {
197  fr_assert(pool->head != this);
198  this->prev->next = this->next;
199  } else {
200  fr_assert(pool->head == this);
201  pool->head = this->next;
202  }
203  if (this->next) {
204  fr_assert(pool->tail != this);
205  this->next->prev = this->prev;
206  } else {
207  fr_assert(pool->tail == this);
208  pool->tail = this->prev;
209  }
210 
211  this->prev = this->next = NULL;
212 }

262 /** Find a connection handle in the connection list
263  *
264  * Walks over the list of connections searching for a specified connection
265  * handle and returns the first connection that contains that pointer.
266  *
267  * @note Will lock mutex and only release mutex if connection handle
268  * is not found, so will usually return will mutex held.
269  * @note Must be called with the mutex free.
270  *
271  * @param[in] pool to search in.
272  * @param[in] conn handle to search for.
273  * @return
274  * - Connection containing the specified handle.
275  * - NULL if non if connection was found.
276  */
277 static fr_pool_connection_t *connection_find(fr_pool_t *pool, void *conn)
278 {
279  fr_pool_connection_t *this;
280 
281  if (!pool || !conn) return NULL;
282 
283  pthread_mutex_lock(&pool->mutex);
284 
285  /*
286  * FIXME: This loop could be avoided if we passed a 'void
287  * **connection' instead. We could use "offsetof" in
288  * order to find top of the parent structure.
289  */
290  for (this = pool->head; this != NULL; this = this->next) {
291  if (this->connection == conn) {
292 #ifdef PTHREAD_DEBUG
293  pthread_t pthread_id;
294 
295  pthread_id = pthread_self();
296  fr_assert(pthread_equal(this->pthread_id, pthread_id) != 0);
297 #endif
298 
299  fr_assert(this->in_use == true);
300  /* coverity[missing_unlock] */
301  return this;
302  }
303  }
304 
305  pthread_mutex_unlock(&pool->mutex);
306  return NULL;
307 }

309 /** Spawns a new connection
310  *
311  * Spawns a new connection using the create callback, and returns it for
312  * adding to the connection list.
313  *
314  * @note Will call the 'open' trigger.
315  * @note Must be called with the mutex free.
316  *
317  * @param[in] pool to modify.
318  * @param[in] request The current request.
319  * @param[in] now Current time.
320  * @param[in] in_use whether the new connection should be "in_use" or not
321  * @param[in] unlock whether we should unlock the mutex before returning
322  * @return
323  * - New connection struct.
324  * - NULL on error.
325  */
326 static fr_pool_connection_t *connection_spawn(fr_pool_t *pool, request_t *request, fr_time_t now, bool in_use, bool unlock)
327 {
328  uint64_t number;
329  uint32_t pending_window;
330  TALLOC_CTX *ctx;
331 
332  fr_pool_connection_t *this;
333  void *conn;
334 
335  fr_assert(pool != NULL);
336 
337  /*
338  * If we have NO connections, and we've previously failed
339  * opening connections, don't open multiple connections until
340  * we successfully open at least one.
341  */
342  if ((pool->state.num == 0) && pool->state.pending && pool->state.last_failed) return NULL;
343 
344  pthread_mutex_lock(&pool->mutex);
345  fr_assert(pool->state.num <= pool->max);
346 
347  /*
348  * Don't spawn too many connections at the same time.
349  */
350  if ((pool->state.num + pool->state.pending) >= pool->max) {
351  pthread_mutex_unlock(&pool->mutex);
352 
353  ROPTIONAL(RERROR, ERROR, "Cannot open new connection, already at max");
354  return NULL;
355  }
356 
357  /*
358  * If the last attempt failed, wait a bit before
359  * retrying.
360  */
361  if (pool->state.last_failed && ((pool->state.last_failed + pool->retry_delay) > now)) {
362  bool complain = false;
363 
364  if ((now - pool->state.last_throttled) >= NSEC) {
365  complain = true;
366 
367  pool->state.last_throttled = now;
368  }
369 
370  pthread_mutex_unlock(&pool->mutex);
371 
372  if (!fr_rate_limit_enabled() || complain) {
373  ROPTIONAL(RERROR, ERROR, "Last connection attempt failed, waiting %pV seconds before retrying",
374  fr_box_time_delta(pool->state.last_failed + pool->retry_delay - now));
375  }
376 
377  return NULL;
378  }
379 
380  /*
381  * We limit the rate of new connections after a failed attempt.
382  */
383  if (pool->state.pending > pool->pending_window) {
384  pthread_mutex_unlock(&pool->mutex);
385 
386  RATE_LIMIT_GLOBAL_ROPTIONAL(RWARN, WARN,
387  "Cannot open a new connection due to rate limit after failure");
388 
389  return NULL;
390  }
391 
392  pool->state.pending++;
393  number = pool->state.count++;
394 
395  /*
396  * Don't starve out the thread trying to reconnect
397  * the pool, by continuously opening new connections.
398  */
399  while (pool->state.reconnecting) pthread_cond_wait(&pool->done_reconnecting, &pool->mutex);
400 
401  /*
402  * Unlock the mutex while we try to open a new
403  * connection. If there are issues with the back-end,
404  * opening a new connection may take a LONG time. In
405  * that case, we want the other connections to continue
406  * to be used.
407  */
408  pthread_mutex_unlock(&pool->mutex);
409 
410  /*
411  * The true value for pending_window is the smaller of
412  * free connection slots, or pool->pending_window.
413  */
414  pending_window = (pool->max - pool->state.num);
415  if (pool->pending_window < pending_window) pending_window = pool->pending_window;
416  ROPTIONAL(RDEBUG2, DEBUG2, "Opening additional connection (%" PRIu64 "), %u of %u pending slots used",
417  number, pool->state.pending, pending_window);
418 
419  /*
420  * Allocate a new top level ctx for the create callback
421  * to hang its memory off of.
422  */
423  ctx = talloc_init("fr_connection_ctx");
424  if (!ctx) return NULL;
425 
426  /*
427  * This may take a long time, which prevents other
428  * threads from releasing connections. We don't care
429  * about other threads opening new connections, as we
430  * already have no free connections.
431  */
432  conn = pool->create(ctx, pool->opaque, pool->connect_timeout);
433  if (!conn) {
434  ROPTIONAL(RERROR, ERROR, "Opening connection failed (%" PRIu64 ")", number);
435 
436  pool->state.last_failed = now;
437  pthread_mutex_lock(&pool->mutex);
438  pool->pending_window = 1;
439  pool->state.pending--;
440 
441  /*
442  * Must be done inside the mutex, reconnect callback
443  * may modify args.
444  */
445  fr_pool_trigger_exec(pool, request, "fail");
446  pthread_cond_broadcast(&pool->done_spawn);
447  pthread_mutex_unlock(&pool->mutex);
448 
449  talloc_free(ctx);
450 
451  return NULL;
452  }
453 
454  /*
455  * And lock the mutex again while we link the new
456  * connection back into the pool.
457  */
458  pthread_mutex_lock(&pool->mutex);
459 
460  this = talloc_zero(pool, fr_pool_connection_t);
461  if (!this) {
462  pthread_cond_broadcast(&pool->done_spawn);
463  pthread_mutex_unlock(&pool->mutex);
464 
465  talloc_free(ctx);
466 
467  return NULL;
468  }
469  talloc_link_ctx(this, ctx);
470 
471  this->created = now;
472  this->connection = conn;
473  this->in_use = in_use;
474 
475  this->number = number;
476  this->last_reserved = fr_time();
477  this->last_released = this->last_reserved;
478 
479  /*
480  * The connection pool is starting up. Insert the
481  * connection into the heap.
482  */
483  if (!in_use) fr_heap_insert(pool->heap, this);
484 
485  connection_link_head(pool, this);
486 
487  /*
488  * Do NOT insert the connection into the heap. That's
489  * done when the connection is released.
490  */
491 
492  pool->state.num++;
493 
494  fr_assert(pool->state.pending > 0);
495  pool->state.pending--;
496 
497  /*
498  * We've successfully opened one more connection. Allow
499  * more connections to open in parallel.
500  */
501  if ((pool->pending_window < pool->max) &&
502  ((pool->max_pending == 0) || (pool->pending_window < pool->max_pending))) {
503  pool->pending_window++;
504  }
505 
506  pool->state.last_spawned = fr_time();
507  pool->delay_interval = pool->cleanup_interval;
508  pool->state.next_delay = pool->cleanup_interval;
509  pool->state.last_failed = 0;
510 
511  /*
512  * Must be done inside the mutex, reconnect callback
513  * may modify args.
514  */
515  fr_pool_trigger_exec(pool, request, "open");
516 
517  pthread_cond_broadcast(&pool->done_spawn);
518  if (unlock) pthread_mutex_unlock(&pool->mutex);
519 
520  /* coverity[missing_unlock] */
521  return this;
522 }

524 /** Close an existing connection.
525  *
526  * Removes the connection from the list, calls the delete callback to close
527  * the connection, then frees memory allocated to the connection.
528  *
529  * @note Will call the 'close' trigger.
530  * @note Must be called with the mutex held.
531  *
532  * @param[in] pool to modify.
533  * @param[in] request The current request.
534  * @param[in] this Connection to delete.
535  */
536 static void connection_close_internal(fr_pool_t *pool, request_t *request, fr_pool_connection_t *this)
537 {
538  /*
539  * If it's in use, release it.
540  */
541  if (this->in_use) {
542 #ifdef PTHREAD_DEBUG
543  pthread_t pthread_id = pthread_self();
544  fr_assert(pthread_equal(this->pthread_id, pthread_id) != 0);
545 #endif
546 
547  this->in_use = false;
548 
549  fr_assert(pool->state.active != 0);
550  pool->state.active--;
551 
552  } else {
553  /*
554  * Connection isn't used, remove it from the heap.
555  */
556  fr_heap_extract(pool->heap, this);
557  }
558 
559  fr_pool_trigger_exec(pool, request, "close");
560 
561  connection_unlink(pool, this);
562 
563  fr_assert(pool->state.num > 0);
564  pool->state.num--;
565  talloc_free(this);
566 }

568 /** Check whether a connection needs to be removed from the pool
569  *
570  * Will verify that the connection is within idle_timeout, max_uses, and
571  * lifetime values. If it is not, the connection will be closed.
572  *
573  * @note Will only close connections not in use.
574  * @note Must be called with the mutex held.
575  *
576  * @param[in] pool to modify.
577  * @param[in] request The current request.
578  * @param[in] this Connection to manage.
579  * @param[in] now Current time.
580  * @return
581  * - 0 if connection was closed.
582  * - 1 if connection handle was left open.
583  */
584 static int connection_manage(fr_pool_t *pool, request_t *request, fr_pool_connection_t *this, time_t now)
585 {
586  fr_assert(pool != NULL);
587  fr_assert(this != NULL);
588 
589  /*
590  * Don't terminated in-use connections
591  */
592  if (this->in_use) return 1;
593 
594  if (this->needs_reconnecting) {
595  ROPTIONAL(RDEBUG2, DEBUG2, "Closing expired connection (%" PRIu64 "): Needs reconnecting",
596  this->number);
597  do_delete:
598  if (pool->state.num <= pool->min) {
599  ROPTIONAL(RDEBUG2, DEBUG2, "You probably need to lower \"min\"");
600  }
601  connection_close_internal(pool, request, this);
602  return 0;
603  }
604 
605  if ((pool->max_uses > 0) &&
606  (this->num_uses >= pool->max_uses)) {
607  ROPTIONAL(RDEBUG2, DEBUG2, "Closing expired connection (%" PRIu64 "): Hit max_uses limit",
608  this->number);
609  goto do_delete;
610  }
611 
612  if ((pool->lifetime > 0) &&
613  ((this->created + pool->lifetime) < now)) {
614  ROPTIONAL(RDEBUG2, DEBUG2, "Closing expired connection (%" PRIu64 "): Hit lifetime limit",
615  this->number);
616  goto do_delete;
617  }
618 
619  if ((pool->idle_timeout > 0) &&
620  ((this->last_released + pool->idle_timeout) < now)) {
621  ROPTIONAL(RINFO, INFO, "Closing connection (%" PRIu64 "): Hit idle_timeout, was idle for %pVs",
622  this->number, fr_box_time_delta(now - this->last_released));
623  goto do_delete;
624  }
625 
626  return 1;
627 }

630 /** Check whether any connections need to be removed from the pool
631  *
632  * Maintains the number of connections in the pool as per the configuration
633  * parameters for the connection pool.
634  *
635  * @note Will only run checks the first time it's called in a given second,
636  * to throttle connection spawning/closing.
637  * @note Will only close connections not in use.
638  * @note Must be called with the mutex held, will release mutex before returning.
639  *
640  * @param[in] pool to manage.
641  * @param[in] request The current request.
642  * @return 1
643  */
644 static int connection_check(fr_pool_t *pool, request_t *request)
645 {
646  uint32_t spawn, idle, extra;
647  fr_time_t now = fr_time();
648  fr_pool_connection_t *this, *next;
649 
650  if ((now - pool->state.last_checked) >= NSEC) {
651  pthread_mutex_unlock(&pool->mutex);
652  return 1;
653  }
654 
655  /*
656  * Some idle connections are OK, if they're within the
657  * configured "spare" range. Any extra connections
658  * outside of that range can be closed.
659  */
660  idle = pool->state.num - pool->state.active;
661  if (idle <= pool->spare) {
662  extra = 0;
663  } else {
664  extra = idle - pool->spare;
665  }
666 
667  /*
668  * The other end can close connections. If so, we'll
669  * have fewer than "min". When that happens, open more
670  * connections to enforce "min".
671  */
672  if ((pool->state.num + pool->state.pending) < pool->min) {
673  spawn = pool->min - (pool->state.num + pool->state.pending);
674  extra = 0;
675 
676  ROPTIONAL(RINFO, INFO, "Need %i more connections to reach min connections (%i)", spawn, pool->min);
677 
678  /*
679  * If we're about to create more than "max",
680  * don't create more.
681  */
682  } else if ((pool->state.num + pool->state.pending) >= pool->max) {
683  /*
684  * Ensure we don't spawn more connections. If
685  * there are extra idle connections, we can
686  * delete all of them.
687  */
688  spawn = 0;
689  /* leave extra alone from above */
690 
691  /*
692  * min < num < max
693  *
694  * AND we don't have enough idle connections.
695  * Open some more.
696  */
697  } else if (idle < pool->spare) {
698  /*
699  * Not enough spare connections. Spawn a few.
700  * But cap the pool size at "max"
701  */
702  spawn = pool->spare - idle;
703  extra = 0;
704 
705  if ((pool->state.num + pool->state.pending + spawn) > pool->max) {
706  spawn = pool->max - (pool->state.num + pool->state.pending);
707  }
708 
709  ROPTIONAL(RINFO, INFO, "Need %i more connections to reach %i spares", spawn, pool->spare);
710 
711  /*
712  * min < num < max
713  *
714  * We have more than enough idle connections, AND
715  * some are pending. Don't open or close any.
716  */
717  } else if (pool->state.pending) {
718  spawn = 0;
719  extra = 0;
720 
721  /*
722  * We have too many idle connections, but closing
723  * some would take us below "min", so we only
724  * close enough to take us to "min".
725  */
726  } else if ((pool->min + extra) >= pool->state.num) {
727  spawn = 0;
728  extra = pool->state.num - pool->min;
729 
730  } else {
731  /*
732  * Closing the "extra" connections won't take us
733  * below "min". It's therefore safe to close
734  * them all.
735  */
736  spawn = 0;
737  /* leave extra alone from above */
738  }
739 
740  /*
741  * Only try to open spares if we're not already attempting to open
742  * a connection. Avoids spurious log messages.
743  */
744  if (spawn) {
745  pthread_mutex_unlock(&pool->mutex);
746  (void) connection_spawn(pool, request, now, false, true);
747  pthread_mutex_lock(&pool->mutex);
748  }
749 
750  /*
751  * We haven't spawned connections in a while, and there
752  * are too many spare ones. Close the one which has been
753  * unused for the longest.
754  */
755  if (extra && (now >= (pool->state.last_spawned + pool->delay_interval))) {
756  fr_pool_connection_t *found = NULL;
757 
758  for (this = pool->tail; this != NULL; this = this->prev) {
759  if (this->in_use) continue;
760 
761  if (!found || (this->last_reserved < found->last_reserved)) found = this;
762  }
763 
764  if (!fr_cond_assert(found)) goto done;
765 
766  ROPTIONAL(RDEBUG2, DEBUG2, "Closing connection (%" PRIu64 "), from %d unused connections",
767  found->number, extra);
768  connection_close_internal(pool, request, found);
769 
770  /*
771  * Decrease the delay for the next time we clean up.
772  */
773  pool->state.next_delay >>= 1;
774  if (pool->state.next_delay == 0) pool->state.next_delay = 1;
775  pool->delay_interval += pool->state.next_delay;
776  }
777 
778  /*
779  * Pass over all of the connections in the pool, limiting
780  * lifetime, idle time, max requests, etc.
781  */
782  for (this = pool->head; this != NULL; this = next) {
783  next = this->next;
784  connection_manage(pool, request, this, now);
785  }
786 
787  pool->state.last_checked = now;
788 done:
789  pthread_mutex_unlock(&pool->mutex);
790 
791  return 1;
792 }

794 /** Get a connection from the connection pool
795  *
796  * @note Must be called with the mutex free.
797  *
798  * @param[in] pool to reserve the connection from.
799  * @param[in] request The current request.
800  * @param[in] spawn whether to spawn a new connection
801  * @return
802  * - A pointer to the connection handle.
803  * - NULL on error.
804  */
805 static void *connection_get_internal(fr_pool_t *pool, request_t *request, bool spawn)
806 {
807  fr_time_t now;
808  fr_pool_connection_t *this;
809 
810  if (!pool) return NULL;
811 
812  pthread_mutex_lock(&pool->mutex);
813 
814  now = fr_time();
815 
816  /*
817  * Grab the link with the lowest latency, and check it
818  * for limits. If "connection manage" says the link is
819  * no longer usable, go grab another one.
820  */
821  do {
822  this = fr_heap_peek(pool->heap);
823  if (!this) break;
824  } while (!connection_manage(pool, request, this, now));
825 
826  /*
827  * We have a working connection. Extract it from the
828  * heap and use it.
829  */
830  if (this) {
831  fr_heap_extract(pool->heap, this);
832  goto do_return;
833  }
834 
835  if (pool->state.num == pool->max) {
836  bool complain = false;
837 
838  /*
839  * Rate-limit complaints.
840  */
841  if ((now - pool->state.last_at_max) > NSEC) {
842  complain = true;
843  pool->state.last_at_max = now;
844  }
845 
846  pthread_mutex_unlock(&pool->mutex);
847  if (!fr_rate_limit_enabled() || complain) {
848  ROPTIONAL(RERROR, ERROR, "No connections available and at max connection limit");
849  /*
850  * Must be done inside the mutex, reconnect callback
851  * may modify args.
852  */
853  fr_pool_trigger_exec(pool, request, "none");
854  }
855 
856  return NULL;
857  }
858 
859  pthread_mutex_unlock(&pool->mutex);
860 
861  if (!spawn) return NULL;
862 
863  ROPTIONAL(RDEBUG2, DEBUG2, "%i of %u connections in use. You may need to increase \"spare\"",
864  pool->state.active, pool->state.num);
865 
866  /*
867  * Returns unlocked on failure, or locked on success
868  */
869  this = connection_spawn(pool, request, now, true, false);
870  if (!this) return NULL;
871 
872 do_return:
873  pool->state.active++;
874  this->num_uses++;
875  this->last_reserved = fr_time();
876  this->in_use = true;
877 
878 #ifdef PTHREAD_DEBUG
879  this->pthread_id = pthread_self();
880 #endif
881  pthread_mutex_unlock(&pool->mutex);
882 
883  ROPTIONAL(RDEBUG2, DEBUG2, "Reserved connection (%" PRIu64 ")", this->number);
884 
885  return this->connection;
886 }

916 /** Create a new connection pool
917  *
918  * Allocates structures used by the connection pool, initialises the various
919  * configuration options and counters, and sets the callback functions.
920  *
921  * Will also spawn the number of connections specified by the 'start' configuration
922  * option.
923  *
924  * @note Will call the 'start' trigger.
925  *
926  * @param[in] ctx Context to link pool's destruction to.
927  * @param[in] cs pool section.
928  * @param[in] opaque data pointer to pass to callbacks.
929  * @param[in] c Callback to create new connections.
930  * @param[in] a Callback to check the status of connections.
931  * @param[in] log_prefix prefix to prepend to all log messages.
932  * @return
933  * - New connection pool.
934  * - NULL on error.
935  */
936 fr_pool_t *fr_pool_init(TALLOC_CTX *ctx,
937  CONF_SECTION const *cs,
938  void *opaque,
939  fr_pool_connection_create_t c, fr_pool_connection_alive_t a,
940  char const *log_prefix)
941 {
942  fr_pool_t *pool = NULL;
943 
944  if (!cs || !opaque || !c) return NULL;
945 
946  /*
947  * Pool is allocated in the NULL context as
948  * threads are likely to allocate memory
949  * beneath the pool.
950  */
951  MEM(pool = talloc_zero(NULL, fr_pool_t));
952 
953  /*
954  * Ensure the pool is freed at the same time
955  * as its parent.
956  */
957  if (ctx && (talloc_link_ctx(ctx, pool) < 0)) {
958  PERROR("%s: Failed linking pool ctx", __FUNCTION__);
959  talloc_free(pool);
960 
961  return NULL;
962  }
963 
964  pool->cs = cs;
965  pool->opaque = opaque;
966  pool->create = c;
967  pool->alive = a;
968 
969  pool->head = pool->tail = NULL;
970 
971  /*
972  * We keep a heap of connections, sorted by the last time
973  * we STARTED using them. Newly opened connections
974  * aren't in the heap. They're only inserted in the list
975  * once they're released.
976  *
977  * We do "most recently started" instead of "most
978  * recently used", because MRU is done as most recently
979  * *released*. We want to order connections by
980  * responsiveness, and MRU prioritizes high latency
981  * connections.
982  *
983  * We want most recently *started*, which gives
984  * preference to low latency links, and pushes high
985  * latency links down in the priority heap.
986  *
987  * https://code.facebook.com/posts/1499322996995183/solving-the-mystery-of-link-imbalance-a-metastable-failure-state-at-scale/
988  */
989  if (!pool->spread) {
990  pool->heap = fr_heap_talloc_alloc(pool, last_reserved_cmp, fr_pool_connection_t, heap_id);
991  /*
992  * For some types of connections we need to used a different
993  * algorithm, because load balancing benefits are secondary
994  * to maintaining a cache of open connections.
995  *
996  * With libcurl's multihandle, connections can only be reused
997  * if all handles that make up the multhandle are done processing
998  * their requests.
999  *
1000  * We can't tell when that's happened using libcurl, and even
1001  * if we could, blocking until all servers had responded
1002  * would have huge cost.
1003  *
1004  * The solution is to order the heap so that the connection that
1005  * was released longest ago is at the top.
1006  *
1007  * That way we maximise time between connection use.
1008  */
1009  } else {
1010  pool->heap = fr_heap_talloc_alloc(pool, last_released_cmp, fr_pool_connection_t, heap_id);
1011  }
1012  if (!pool->heap) {
1013  ERROR("%s: Failed creating connection heap", __FUNCTION__);
1014  error:
1015  fr_pool_free(pool);
1016  return NULL;
1017  }
1018 
1019  pool->log_prefix = log_prefix ? talloc_typed_strdup(pool, log_prefix) : "core";
1020  pthread_mutex_init(&pool->mutex, NULL);
1021  pthread_cond_init(&pool->done_spawn, NULL);
1022  pthread_cond_init(&pool->done_reconnecting, NULL);
1023 
1024  DEBUG2("Initialising connection pool");
1025 
1026  {
1027  CONF_SECTION *mutable;
1028 
1029  memcpy(&mutable, &cs, sizeof(mutable));
1030 
1031  if (cf_section_rules_push(mutable, pool_config) < 0) goto error;
1032  if (cf_section_parse(pool, pool, mutable) < 0) {
1033  PERROR("Configuration parsing failed");
1034  goto error;
1035  }
1036  }
1037 
1038  /*
1039  * Some simple limits
1040  */
1041  if (pool->max == 0) {
1042  cf_log_err(cs, "Cannot set 'max' to zero");
1043  goto error;
1044  }
1045 
1046  /* coverity[missing_unlock] */
1047  pool->pending_window = (pool->max_pending > 0) ? pool->max_pending : pool->max;
1048 
1049  if (pool->min > pool->max) {
1050  cf_log_err(cs, "Cannot set 'min' to more than 'max'");
1051  goto error;
1052  }
1053 
1054  FR_INTEGER_BOUND_CHECK("max", pool->max, <=, 1024);
1055  FR_INTEGER_BOUND_CHECK("start", pool->start, <=, pool->max);
1056  FR_INTEGER_BOUND_CHECK("spare", pool->spare, <=, (pool->max - pool->min));
1057 
1058  if (pool->lifetime > 0) {
1059  FR_TIME_DELTA_COND_CHECK("idle_timeout", pool->idle_timeout,
1060  (pool->idle_timeout <= pool->lifetime), 0);
1061  }
1062 
1063  if (pool->idle_timeout > 0) {
1064  FR_TIME_DELTA_BOUND_CHECK("cleanup_interval", pool->cleanup_interval, <=, pool->idle_timeout);
1065  }
1066 
1067  /*
1068  * Some libraries treat 0.0 as infinite timeout, others treat it
1069  * as instantaneous timeout. Solve the inconsistency by making
1070  * the smallest allowable timeout 100ms.
1071  */
1072  FR_TIME_DELTA_BOUND_CHECK("connect_timeout", pool->connect_timeout, >=, fr_time_delta_from_msec(100));
1073 
1074  /*
1075  * Don't open any connections. Instead, force the limits
1076  * to only 1 connection.
1077  *
1078  */
1079  if (check_config) {
1080  pool->start = pool->min = pool->max = 1;
1081  return pool;
1082  }
1083 
1084  return pool;
1085 }

1200 /** Mark connections for reconnection, and spawn at least 'start' connections
1201  *
1202  * @note This call may block whilst waiting for pending connection attempts to complete.
1203  *
1204  * This intended to be called on a connection pool that's in use, to have it reflect
1205  * a configuration change, or because the administrator knows that all connections
1206  * in the pool are inviable and need to be reconnected.
1207  *
1208  * @param[in] pool to reconnect.
1209  * @param[in] request The current request.
1210  * @return
1211  * - 0 On success.
1212  * - -1 If we couldn't create start connections, this may be ignored
1213  * depending on the context in which this function is being called.
1214  */
1215 int fr_pool_reconnect(fr_pool_t *pool, request_t *request)
1216 {
1217  uint32_t i;
1218  fr_pool_connection_t *this;
1219  time_t now;
1220 
1221  pthread_mutex_lock(&pool->mutex);
1222 
1223  /*
1224  * Pause new spawn attempts (we release the mutex
1225  * during our cond wait).
1226  */
1227  pool->state.reconnecting = true;
1228 
1229  /*
1230  * When the loop exits, we'll hold the lock for the pool,
1231  * and we're guaranteed the connection create callback
1232  * will not be using the opaque data.
1233  */
1234  while (pool->state.pending) pthread_cond_wait(&pool->done_spawn, &pool->mutex);
1235 
1236  /*
1237  * We want to ensure at least 'start' connections
1238  * have been reconnected. We can't call reconnect
1239  * because, we might get the same connection each
1240  * time we reserve one, so we close 'start'
1241  * connections, and then attempt to spawn them again.
1242  */
1243  for (i = 0; i < pool->start; i++) {
1244  this = fr_heap_peek(pool->heap);
1245  if (!this) break; /* There wasn't 'start' connections available */
1246 
1247  connection_close_internal(pool, request, this);
1248  }
1249 
1250  /*
1251  * Mark all remaining connections in the pool as
1252  * requiring reconnection.
1253  */
1254  for (this = pool->head; this; this = this->next) this->needs_reconnecting = true;
1255 
1256  /*
1257  * Call the reconnect callback (if one's set)
1258  * This may modify the opaque data associated
1259  * with the pool.
1260  */
1261  if (pool->reconnect) pool->reconnect(pool, pool->opaque);
1262 
1263  /*
1264  * Must be done inside the mutex, reconnect callback
1265  * may modify args.
1266  */
1267  fr_pool_trigger_exec(pool, request, "reconnect");
1268 
1269  /*
1270  * Allow new spawn attempts, and wakeup any threads
1271  * waiting to spawn new connections.
1272  */
1273  pool->state.reconnecting = false;
1274  pthread_cond_broadcast(&pool->done_reconnecting);
1275  pthread_mutex_unlock(&pool->mutex);
1276 
1277  now = time(NULL);
1278 
1279  /*
1280  * Now attempt to spawn 'start' connections.
1281  */
1282  for (i = 0; i < pool->start; i++) {
1283  this = connection_spawn(pool, request, now, false, true);
1284  if (!this) return -1;
1285  }
1286 
1287  return 0;
1288 }

1342 /** Reserve a connection in the connection pool
1343  *
1344  * Will attempt to find an unused connection in the connection pool, if one is
1345  * found, will mark it as in in use increment the number of active connections
1346  * and return the connection handle.
1347  *
1348  * If no free connections are found will attempt to spawn a new one, conditional

1370  * connections.

1372  * @see fr_pool_connection_get
1373  * @param[in] pool to release the connection in.
1374  * @param[in] request The current request.
1375  * @param[in] conn to release.
1376  */
1377 void fr_pool_connection_release(fr_pool_t *pool, request_t *request, void *conn)
1378 {
1379  fr_pool_connection_t *this;
1380  fr_time_delta_t held;
1381  bool trigger_min = false, trigger_max = false;
1382 
1383  this = connection_find(pool, conn);
1384  if (!this) return;
1385 
1386  this->in_use = false;
1387 
1388  /*
1389  * Record when the connection was last released
1390  */
1391  this->last_reserved = fr_time();
1392  pool->state.last_released = this->last_released;
1393 
1394  /*
1395  * This is done inside the mutex to ensure
1396  * updates are atomic.
1397  */
1398  held = this->last_released - this->last_reserved;
1399 
1400  /*
1401  * Check we've not exceeded out trigger limits
1402  *
1403  * These should only fire once per second.
1404  */
1405  if (pool->held_trigger_min &&
1406  (held < pool->held_trigger_min) &&
1407  ((this->last_released - pool->state.last_held_min) >= NSEC)) {
1408  trigger_min = true;
1409  pool->state.last_held_min = this->last_released;
1410  }
1411 
1412  if (pool->held_trigger_min &&
1413  (held > pool->held_trigger_max) &&
1414  ((this->last_released - pool->state.last_held_max) >= NSEC)) {
1415  trigger_max = true;
1416  pool->state.last_held_max = this->last_released;
1417  }
1418 
1419  /*
1420  * Insert the connection in the heap.
1421  *
1422  * This will either be based on when we *started* using it
1423  * (allowing fast links to be re-used, and slow links to be
1424  * gradually expired), or when we released it (allowing
1425  * the maximum amount of time between connection use).
1426  */
1427  fr_heap_insert(pool->heap, this);
1428 
1429  fr_assert(pool->state.active != 0);
1430  pool->state.active--;
1431 
1432  ROPTIONAL(RDEBUG2, DEBUG2, "Released connection (%" PRIu64 ")", this->number);
1433 
1434  /*
1435  * We mirror the "spawn on get" functionality by having
1436  * "delete on release". If there are too many spare
1437  * connections, go manage the pool && clean some up.
1438  */
1439  connection_check(pool, request);
1440 
1441  if (trigger_min) fr_pool_trigger_exec(pool, request, "min");
1442  if (trigger_max) fr_pool_trigger_exec(pool, request, "max");
1443 }
