47 /** Calculate HMAC using OpenSSL's MD5 implementation
48  *
49  * @param digest Caller digest to be filled in.
50  * @param in Pointer to data stream.
51  * @param inlen length of data stream.
52  * @param key Pointer to authentication key.
53  * @param key_len Length of authentication key.
54  *
55  */
56 void fr_hmac_md5(uint8_t digest[MD5_DIGEST_LENGTH], uint8_t const *in, size_t inlen,
57  uint8_t const *key, size_t key_len)
58 {
59  HMAC_CTX *ctx;
60 
61  if (unlikely(!md5_hmac_ctx)) {
62  ctx = HMAC_CTX_new();
63  if (unlikely(!ctx)) return;
64  fr_thread_local_set_destructor(md5_hmac_ctx, _hmac_md5_ctx_free_on_exit, ctx);
65  } else {
66  ctx = md5_hmac_ctx;
67  }
68 
69 #ifdef EVP_MD_CTX_FLAG_NON_FIPS_ALLOW
70  /* Since MD5 is not allowed by FIPS, explicitly allow it. */
71  HMAC_CTX_set_flags(ctx, EVP_MD_CTX_FLAG_NON_FIPS_ALLOW);
72 #endif /* EVP_MD_CTX_FLAG_NON_FIPS_ALLOW */
73 
74  HMAC_Init_ex(ctx, key, key_len, EVP_md5(), NULL);
75  HMAC_Update(ctx, in, inlen);
76  HMAC_Final(ctx, digest, NULL);
77  HMAC_CTX_reset(ctx);
78 }

80 /** Calculate HMAC using internal MD5 implementation
81  *
82  * @param digest Caller digest to be filled in.
83  * @param in Pointer to data stream.
84  * @param inlen length of data stream.
85  * @param key Pointer to authentication key.
86  * @param key_len Length of authentication key.
87  *
88  */
89 void fr_hmac_md5(uint8_t digest[MD5_DIGEST_LENGTH], uint8_t const *in, size_t inlen,
90  uint8_t const *key, size_t key_len)
91 {
92  fr_md5_ctx_t *ctx;
93  uint8_t k_ipad[65]; /* inner padding - key XORd with ipad */
94  uint8_t k_opad[65]; /* outer padding - key XORd with opad */
95  uint8_t tk[16];
96  int i;
97 
98  ctx = fr_md5_ctx_alloc(true);
99 
100  /* if key is longer than 64 bytes reset it to key=MD5(key) */
101  if (key_len > 64) {
102  fr_md5_update(ctx, key, key_len);
103  fr_md5_final(tk, ctx);
104  fr_md5_ctx_reset(ctx);
105 
106  key = tk;
107  key_len = 16;
108  }
109 
110  /*
111  * the HMAC_MD5 transform looks like:
112  *
113  * MD5(K XOR opad, MD5(K XOR ipad, in))
114  *
115  * where K is an n byte key
116  * ipad is the byte 0x36 repeated 64 times
117 
118  * opad is the byte 0x5c repeated 64 times
119  * and in is the data being protected
120  */
121 
122  /* start out by storing key in pads */
123  memset(k_ipad, 0, sizeof(k_ipad));
124  memset(k_opad, 0, sizeof(k_opad));
125  memcpy(k_ipad, key, key_len);
126  memcpy(k_opad, key, key_len);
127 
128  /* XOR key with ipad and opad values */
129  for (i = 0; i < 64; i++) {
130  k_ipad[i] ^= 0x36;
131  k_opad[i] ^= 0x5c;
132  }
133  /*
134  * perform inner MD5
135  */
136  fr_md5_update(ctx, k_ipad, 64); /* start with inner pad */
137  fr_md5_update(ctx, in, inlen); /* then in of datagram */
138  fr_md5_final(digest, ctx); /* finish up 1st pass */
139 
140 
141  /*
142  * perform outer MD5
143  */
144  fr_md5_ctx_reset(ctx);
145  fr_md5_update(ctx, k_opad, 64); /* start with outer pad */
146  fr_md5_update(ctx, digest, 16); /* then results of 1st hash */
147  fr_md5_final(digest, ctx); /* finish up 2nd pass */
148 
149  fr_md5_ctx_free(&ctx);
150 }
