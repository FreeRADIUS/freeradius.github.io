72  struct osmo_fd *bfd)
73 {
74  uint8_t msg_type = *(msg->l2h);

76 
77  switch (msg_type) {

91  * read one ipa message from the socket
92  * return NULL in case of error
93  */
94 struct msgb *ipaccess_read_msg(struct osmo_fd *bfd, int *error)
95 {
96  struct msgb *msg = msgb_alloc(TS1_ALLOC_SIZE, "Abis/IP");
97  struct ipaccess_head *hh;
98  int len, ret = 0;
99 
100  if (!msg) {
101  *error = -ENOMEM;
102  return NULL;
103  }
104 
105  /* first read our 3-byte header */
106  hh = (struct ipaccess_head *) msg->data;
107  ret = recv(bfd->fd, msg->data, 3, 0);
108  if (ret < 0) {
109  msgb_free(msg);
110  *error = ret;
111  return NULL;
112  } else if (ret == 0) {
113  msgb_free(msg);
114  *error = ret;
115  return NULL;
116  }
117 
118  msgb_put(msg, ret);
119 
120  /* then read te length as specified in header */
121  msg->l2h = msg->data + sizeof(*hh);
122  len = ntohs(hh->len);
123  ret = recv(bfd->fd, msg->l2h, len, 0);
124  if (ret < len) {
125  msgb_free(msg);
126  *error = -EIO;
127  return NULL;
128  }
129  msgb_put(msg, ret);
130 
131  return msg;
132 }
