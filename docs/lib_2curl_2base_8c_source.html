74 /** Initialise global curl options
75  *
76  * libcurl is meant to performa reference counting, but still seems to
77  * leak lots of memory if we call curl_global_init many times.
78  */
79 int fr_curl_init(void)
80 {
81  CURLcode ret;
82  curl_version_info_data *curlversion;
83 
84  if (instance_count > 0) {
85  instance_count++;
86  return 0;
87  }
88 
89 #ifdef WITH_TLS
90  /*
91  * Use our OpenSSL init with the hope that
92  * the free function will also free the
93  * memory allocated during SSL init.
94  */
95  if (fr_openssl_init() < 0) return -1;
96 #endif
97 
98  if (fr_dict_autoload(curl_dict) < 0) {
99  PERROR("Failed loading dictionaries for curl");
100  return -1;
101  }
102 
103  ret = curl_global_init(CURL_GLOBAL_ALL);
104  if (ret != CURLE_OK) {
105  ERROR("CURL init returned error: %i - %s", ret, curl_easy_strerror(ret));
106  fr_dict_autofree(curl_dict);
107  return -1;
108  }
109 
110  curlversion = curl_version_info(CURLVERSION_NOW);
111  if (strcmp(LIBCURL_VERSION, curlversion->version) != 0) {
112  WARN("libcurl version changed since the server was built");
113  WARN("linked: %s built: %s", curlversion->version, LIBCURL_VERSION);
114  }
115 
116  INFO("libcurl version: %s", curl_version());
117 
118  instance_count++;
119 
120  return 0;
121 }

135 int fr_curl_easy_tls_init(fr_curl_io_request_t *randle, fr_curl_tls_t const *conf)
136 {
137  request_t *request = randle->request;
138 
139  if (conf->certificate_file) FR_CURL_ROPTIONAL_SET_OPTION(CURLOPT_SSLCERT, conf->certificate_file);
140  if (conf->private_key_file) FR_CURL_ROPTIONAL_SET_OPTION(CURLOPT_SSLKEY, conf->private_key_file);
141  if (conf->private_key_password) FR_CURL_ROPTIONAL_SET_OPTION(CURLOPT_KEYPASSWD, conf->private_key_password);
142  if (conf->ca_file) FR_CURL_ROPTIONAL_SET_OPTION(CURLOPT_CAINFO, conf->ca_file);
143  if (conf->ca_issuer_file) FR_CURL_ROPTIONAL_SET_OPTION(CURLOPT_ISSUERCERT, conf->ca_issuer_file);
144  if (conf->ca_path) FR_CURL_ROPTIONAL_SET_OPTION(CURLOPT_CAPATH, conf->ca_path);
145  if (conf->random_file) FR_CURL_ROPTIONAL_SET_OPTION(CURLOPT_RANDOM_FILE, conf->random_file);
146  FR_CURL_ROPTIONAL_SET_OPTION(CURLOPT_USE_SSL, conf->require_cert);
147 
148  FR_CURL_ROPTIONAL_SET_OPTION(CURLOPT_SSL_VERIFYPEER, (conf->check_cert == true) ? 1L : 0L);
149  FR_CURL_ROPTIONAL_SET_OPTION(CURLOPT_SSL_VERIFYHOST, (conf->check_cert_cn == true) ? 2L : 0L);
150  if (conf->extract_cert_attrs) FR_CURL_ROPTIONAL_SET_OPTION(CURLOPT_CERTINFO, 1L);
151 
152  return 0;
153 error:
154  return -1;
155 }

157 int fr_curl_response_certinfo(request_t *request, fr_curl_io_request_t *randle)
158 {
159  CURL *candle = randle->candle;
160  CURLcode ret;
161  int i;
162  char buffer[265];
163  char *p , *q, *attr = buffer;
164  fr_cursor_t cursor, list;
165  fr_pair_t *cert_vps = NULL;
166  /*
167  * Examples and documentation show cert_info being
168  * a struct curl_certinfo *, but CPP checks require
169  * it to be a struct curl_slist *.
170  *
171  * https://curl.haxx.se/libcurl/c/certinfo.html
172  */
173  union {
174  struct curl_slist *to_info;
175  struct curl_certinfo *to_certinfo;
176  } ptr;
177  ptr.to_info = NULL;
178 
179  fr_cursor_init(&list, &request->request_pairs);
180 
181  ret = curl_easy_getinfo(candle, CURLINFO_CERTINFO, &ptr.to_info);
182  if (ret != CURLE_OK) {
183  REDEBUG("Getting certificate info failed: %i - %s", ret, curl_easy_strerror(ret));
184 
185  return -1;
186  }
187 
188  attr += strlcpy(attr, "TLS-Cert-", sizeof(buffer));
189 
190  RDEBUG2("Chain has %i certificate(s)", ptr.to_certinfo->num_of_certs);
191  for (i = 0; i < ptr.to_certinfo->num_of_certs; i++) {
192  struct curl_slist *cert_attrs;
193 
194  RDEBUG2("Processing certificate %i",i);
195  fr_cursor_init(&cursor, &cert_vps);
196 
197  for (cert_attrs = ptr.to_certinfo->certinfo[i];
198  cert_attrs;
199  cert_attrs = cert_attrs->next) {
200  fr_pair_t *vp;
201  fr_dict_attr_t const *da;
202 
203  q = strchr(cert_attrs->data, ':');
204  if (!q) {
205  RWDEBUG("Malformed certinfo from libcurl: %s", cert_attrs->data);
206  continue;
207  }
208 
209  strlcpy(attr, cert_attrs->data, (q - cert_attrs->data) + 1);
210  for (p = attr; *p != '\0'; p++) if (*p == ' ') *p = '-';
211 
212  da = fr_dict_attr_by_name(dict_freeradius, buffer);
213  if (!da) {
214  RDEBUG3("Skipping %s += '%s'", buffer, q + 1);
215  RDEBUG3("If this value is required, define attribute \"%s\"", buffer);
216  continue;
217  }
218  MEM(vp = fr_pair_afrom_da(request->packet, da));
219  fr_pair_value_from_str(vp, q + 1, -1, '\0', true);
220 
221  fr_cursor_append(&cursor, vp);
222  }
223  /*
224  * Add a copy of the cert_vps to the request list.
225  *
226  * Both PVS studio and Coverity detect the condition
227  * below as logically dead code unless we explicitly
228  * set cert_vps. This is because they're too dumb
229  * to realise that the cursor argument passed to
230  * tls_session_pairs_from_x509_cert contains a
231  * reference to cert_vps.
232  */
233  cert_vps = fr_cursor_head(&cursor);
234  if (cert_vps) {
235  /*
236  * Print out all the pairs we have so far
237  */
238  log_request_pair_list(L_DBG_LVL_2, request, cert_vps, NULL);
239  fr_cursor_merge(&list, &cursor);
240  cert_vps = NULL;
241  }
242  }
243  return 0;
244 }
