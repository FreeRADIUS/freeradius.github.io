112 static int dl_symbol_init_cmp(void const *one, void const *two)
113 {
114  dl_symbol_init_t const *a = one, *b = two;
115  int ret;
116 
117  fr_assert(a && b);
118 
119  ret = ((void *)a->func > (void *)b->func) - ((void *)a->func < (void *)b->func);
120  if (ret != 0) return ret;
121 
122  ret = (a->symbol && !b->symbol) - (!a->symbol && b->symbol);
123  if (ret != 0) return ret;
124 
125  if (!a->symbol && !b->symbol) return 0;
126 
127 #ifdef __clang_analyzer__
128  if (!fr_cond_assert(a->symbol && b->symbol)) return 0; /* Bug in clang scan ? */
129 #endif
130 
131  return strcmp(a->symbol, b->symbol);
132 }

164 /** Utility function to dlopen the library containing a particular symbol
165  *
166  * @note Not really part of our 'dl' API, just a convenience function.
167  *
168  * @param[in] sym_name to resolve.
169  * @param[in] flags to pass to dlopen.
170  * @return
171  * - NULL on error.
172  * - A new handle on success.
173  */
174 void *dl_open_by_sym(char const *sym_name, int flags)
175 {
176  Dl_info info;
177  void *sym;
178  void *handle;
179 
180  /*
181  * Resolve the test symbol in our own symbol space by
182  * iterating through all the libraries.
183  * This might be slow. Don't do this at runtime!
184  */
185  sym = dlsym(RTLD_DEFAULT, sym_name);
186  if (!sym) {
187  fr_strerror_printf("Can't resolve symbol %s", sym_name);
188  return NULL;
189  }
190 
191  /*
192  * Lookup the library the symbol belongs to
193  */
194  if (dladdr(sym, &info) == 0) {
195  fr_strerror_printf("Failed retrieving info for \"%s\" (%p)", sym_name, sym);
196  return NULL;
197  }
198 
199  handle = dlopen(info.dli_fname, flags);
200  if (!handle) {
201  fr_strerror_printf("Failed loading \"%s\": %s", info.dli_fname, dlerror());
202  return NULL;
203  }
204 
205  return handle;
206 }

208 /** Walk over the registered init callbacks, searching for the symbols they depend on
209  *
210  * Allows code outside of the dl API to register initialisation functions that get
211  * executed depending on whether the dl exports a particular symbol.
212  *
213  * This cuts down the amount of boilerplate code in 'mod_load' functions.
214  *
215  * @param[in] dl_loader Tree of dynamically loaded libraries, and callbacks.
216  * @param[in] dl to search for symbols in.
217  * @return
218  * - 0 continue walking.
219  * - -1 error.
220  */
221 int dl_symbol_init(dl_loader_t *dl_loader, dl_t const *dl)
222 {
223  dl_symbol_init_t *init;
224  fr_cursor_t cursor;
225  void *sym = NULL;
226  char buffer[256];
227 
228  for (init = fr_cursor_init(&cursor, &dl_loader->sym_init);
229  init;
230  init = fr_cursor_next(&cursor)) {
231  if (init->symbol) {
232  char *p;
233 
234  snprintf(buffer, sizeof(buffer), "%s_%s", dl->name, init->symbol);
235 
236  /*
237  * '-' is not a valid symbol character in
238  * C. But "libfreeradius-radius" is a
239  * valid library name. So we hash things together.
240  */
241  for (p = buffer; *p != '\0'; p++) {
242  if (*p == '-') *p = '_';
243  }
244 
245  sym = dlsym(dl->handle, buffer);
246  if (!sym) {
247  continue;
248  }
249  }
250 
251  if (init->func(dl, sym, init->ctx) < 0) return -1;
252  }
253 
254  return 0;
255 }

257 /** Walk over the registered init callbacks, searching for the symbols they depend on
258  *
259  * Allows code outside of the dl API to register free functions that get
260  * executed depending on whether the dl exports a particular symbol.
261  *
262  * This cuts down the amount of boilerplate code in 'mod_unload' functions.
263  *
264  * @param[in] dl_loader Tree of dynamically loaded libraries, and callbacks.
265  * @param[in] dl to search for symbols in.
266  */
267 static int dl_symbol_free(dl_loader_t *dl_loader, dl_t const *dl)
268 {
269  dl_symbol_free_t *free;
270  fr_cursor_t cursor;
271  void *sym = NULL;
272 
273  for (free = fr_cursor_init(&cursor, &dl_loader->sym_free);
274  free;
275  free = fr_cursor_next(&cursor)) {
276  if (free->symbol) {
277  char *sym_name = NULL;
278 
279  sym_name = talloc_typed_asprintf(NULL, "%s_%s", dl->name, free->symbol);
280  if (!sym_name) return -1;
281 
282  sym = dlsym(dl->handle, sym_name);
283  talloc_free(sym_name);
284 
285  if (!sym) continue;
286  }
287 
288  free->func(dl, sym, free->ctx);
289  }
290 
291  return 0;
292 }

294 /** Register a callback to execute when a dl with a particular symbol is first loaded
295  *
296  * @note Will replace ctx data for callbacks with the same symbol/func.
297  *
298  * @param[in] dl_loader Tree of dynamically loaded libraries, and callbacks.
299  * @param[in] priority Execution priority. Callbacks with a higher priority get
300  * called first.
301  * @param[in] symbol that determines whether func should be called. "<modname>_" is
302  * added as a prefix to the symbol. The prefix is added because
303  * some dls are loaded with RTLD_GLOBAL into the global symbol
304  * space, so the symbols they export must be unique.
305  * May be NULL to always call the function.
306  * @param[in] func to register. Called when dl is loaded.
307  * @param[in] ctx to pass to func.
308  * @return
309  * - 0 on success (or already registered).
310  * - -1 on failure.
311  */
312 int dl_symbol_init_cb_register(dl_loader_t *dl_loader, unsigned int priority,
313  char const *symbol, dl_onload_t func, void *ctx)
314 {
315  dl_symbol_init_t *n, *p;
316  fr_cursor_t cursor;
317 
318  dl_symbol_init_cb_unregister(dl_loader, symbol, func);
319 
320  n = talloc(dl_loader, dl_symbol_init_t);
321  if (!n) return -1;
322 
323  n->priority = priority;
324  n->symbol = symbol;
325  n->func = func;
326  n->ctx = ctx;
327 
328  for (p = fr_cursor_init(&cursor, &dl_loader->sym_init);
329  p && (p->priority >= priority);
330  fr_cursor_next(&cursor));
331  fr_cursor_insert(&cursor, n);
332 
333  return 0;
334 }

438 /** Search for a dl's shared object in various locations
439  *
440  * @note You must call dl_symbol_init when ready to call autoloader callbacks.
441  *
442  * @param[in] dl_loader Tree of dynamically loaded libraries, and callbacks.
443  * @param[in] name of library to load. May be a relative path.
444  * @param[in] uctx Data to store within the dl_t.
445  * @param[in] uctx_free talloc_free the passed in uctx data if this
446  * dl_t is freed.
447  * @return
448  * - A new dl_t on success, or a pointer to an existing
449  * one with the reference count increased.
450  * - NULL on error.
451  */
452 dl_t *dl_by_name(dl_loader_t *dl_loader, char const *name, void *uctx, bool uctx_free)
453 {
454  int flags = RTLD_NOW;
455  void *handle = NULL;
456  char const *search_path;
457  dl_t *dl;
458 
459  /*
460  * There's already something in the tree,
461  * just return that instead.
462  */
463  dl = rbtree_finddata(dl_loader->tree, &(dl_t){ .name = name });
464  if (dl) {
465  talloc_increase_ref_count(dl);
466  return dl;
467  }
468 
469  flags |= RTLD_LOCAL;
470 
471  /*
472  * Forces dlopened libraries to resolve symbols within
473  * their local symbol tables instead of the global symbol
474  * table.
475  *
476  * May help resolve issues with symbol conflicts.
477  */
478 #if defined(RTLD_DEEPBIND) && !defined(__SANITIZE_ADDRESS__)
479  flags |= RTLD_DEEPBIND;
480  fr_strerror(); /* clear error buffer */
481 #endif
482 
483  /*
484  * Bind all the symbols *NOW* so we don't hit errors later
485  */
486  flags |= RTLD_NOW;
487 
488  search_path = dl_search_path(dl_loader);
489 
490  /*
491  * Prefer loading our libraries by absolute path.
492  */
493  if (search_path) {
494  char *ctx, *paths, *path;
495  char *p;
496 
497  fr_strerror();
498 
499  ctx = paths = talloc_typed_strdup(NULL, search_path);
500  while ((path = strsep(&paths, ":")) != NULL) {
501  int access_mode = R_OK | X_OK;
502 
503  /*
504  * Trim the trailing slash
505  */
506  p = strrchr(path, '/');
507  if (p && ((p[1] == '\0') || (p[1] == ':'))) *p = '\0';
508 
509  path = talloc_typed_asprintf(ctx, "%s/%s%s", path, name, DL_EXTENSION);
510  handle = dlopen(path, flags);
511  if (handle) {
512  talloc_free(path);
513  break;
514  }
515 
516 #ifdef AT_ACCESS
517  access_mode |= AT_ACCESS;
518 #endif
519 
520  /*
521  * Check if the dlopen() failed
522  * because of access permissions.
523  */
524  if (access(path, access_mode) < 0) {
525  /*
526  * It doesn't exist,
527  * continue with the next
528  * element of "path".
529  */
530  if (errno == ENOENT) continue;
531 
532  /*
533  * Stop looking for more
534  * libraries, and instead
535  * complain about access
536  * permissions.
537  */
538  fr_strerror_printf("Access check failed for %s - %s", path, fr_syserror(errno));
539  talloc_free(path);
540  break;
541  }
542 
543  talloc_free(path);
544  }
545 
546  /*
547  * No element of "path" had the library. Return
548  * the error from the last dlopen().
549  */
550  if (!handle) {
551  talloc_free(ctx);
552  fr_strerror_printf("%s", dlerror());
553  return NULL;
554  }
555 
556  talloc_free(ctx);
557  } else {
558  char buffer[2048];
559 
560  strlcpy(buffer, name, sizeof(buffer));
561  /*
562  * FIXME: Make this configurable...
563  */
564  strlcat(buffer, DL_EXTENSION, sizeof(buffer));
565 
566  handle = dlopen(buffer, flags);
567  if (!handle) {
568  char *error = dlerror();
569 
570  /*
571  * Append the error
572  */
573  fr_strerror_printf("%s", error);
574  return NULL;
575  }
576  }
577 
578  dl = talloc_zero(dl_loader, dl_t);
579  if (!dl) {
580  dlclose(handle);
581  return NULL;
582  }
583 
584  dl->name = talloc_typed_strdup(dl, name);
585  dl->handle = handle;
586  dl->loader = dl_loader;
587  dl->uctx = uctx;
588  dl->uctx_free = uctx_free;
589  talloc_set_destructor(dl, _dl_free);
590 
591  dl->in_tree = rbtree_insert(dl_loader->tree, dl);
592  if (!dl->in_tree) {
593  talloc_free(dl);
594  return NULL;
595  }
596 
597  if (!dl_loader->defer_symbol_init) dl_symbol_init(dl_loader, dl);
598 
599  return dl;
600 }

631 static int _dl_loader_free(dl_loader_t *dl_loader)
632 {
633  int ret = 0;
634 
635  if (dl_loader->uctx_free) {
636  ret = talloc_free(dl_loader->uctx);
637  if (ret != 0) goto finish;
638  }
639 
640  /*
641  * Prevent freeing if we still have dls loaded
642  * We do reference counting, we know exactly what
643  * should still be active.
644  */
645  if (rbtree_num_elements(dl_loader->tree) > 0) {
646  ret = -1;
647 #ifndef NDEBUG
648  /*
649  * Yes, this is the correct call order
650  */
651  rbtree_walk(dl_loader->tree, RBTREE_IN_ORDER, _dl_walk_print, NULL);
652  fr_strerror_printf_push("Refusing to cleanup dl loader, the following dynamically loaded "
653  "libraries are still in use:");
654 #endif
655  goto finish;
656  }
657 
658 finish:
659  return ret;
660 }

776  * @param[in] uctx API client opaque data to store in dl_loader_t.

780  * #dl_symbol_init after calling #dl_by_name.

782  * from executing until #dl_symbol_init is
783  * called explicitly.
784  */
785 dl_loader_t *dl_loader_init(TALLOC_CTX *ctx, void *uctx, bool uctx_free, bool defer_symbol_init)

792  return NULL;
793  }
794

796  if (!dl_loader->tree) {
797  fr_strerror_printf("Failed initialising dl->tree");
798  error:
799  TALLOC_FREE(dl_loader);
800  return NULL;
801  }
802 
803  talloc_link_ctx(ctx, dl_loader);
804 
805  dl_loader->lib_dir = talloc_strdup(dl_loader, fr_path_default_lib_dir());
806  if (!dl_loader->lib_dir) {
