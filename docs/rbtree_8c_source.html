83 static void free_walker(rbtree_t *tree, rbnode_t *x)
84 {
85  (void) talloc_get_type_abort(x, rbnode_t);
86 
87  if (x->left != NIL) free_walker(tree, x->left);

103 /** Free the rbtree cleaning up any nodes
104  *
105  * Walk the tree deleting nodes, then free any children of the tree.
106  *
107  * @note If the destructor of a talloc descendent needs to lookup any
108  * information in the tree, it will be unavailable at the point
109  * of freeing. We could fix this by introducing a pre-free callback
110  * which gets called before any of the nodes are deleted.
111  *
112  * @param[in] tree to tree.
113  * @return
114  * - 0 if tree was freed.
115  * - -1 if tree is already being freed.
116  */
117 static int _tree_free(rbtree_t *tree)
118 {
119  /*
120  * Prevent duplicate frees
121  */
122  if (unlikely(tree->being_freed)) return -1;
123  tree->being_freed = true;
124 
125  /*
126  * walk the tree, deleting the nodes...
127  */
128  if (tree->root != NIL) free_walker(tree, tree->root);
129 
130 #ifndef NDEBUG
131  tree->magic = 0;
132 #endif
133  tree->root = NIL;
134  tree->num_elements = 0;
135 
136  /*
137  * Ensure all dependents on the tree run their
138  * destructors. The tree at this point should
139  * and any tree operations should be empty.
140  */
141  talloc_free_children(tree);
142 
143  /*
144  * Clear up locks.
145  */
146  if (tree->lock) pthread_mutex_destroy(&tree->mutex);
147 
148  return 0;
149 }

151 /** Create a new RED-BLACK tree
152  *
153  * @note Due to the node memory being allocated from a different pool to the main
154  */
155 rbtree_t *_rbtree_alloc(TALLOC_CTX *ctx, rb_comparator_t compare,
156  char const *type, rb_free_t node_free, int flags)
157 {
158  rbtree_t *tree;
159 
160  if (!compare) return NULL;
161 
162  tree = talloc_zero(ctx, rbtree_t);
163  if (!tree) return NULL;
164 
165 #ifndef NDEBUG
166  tree->magic = RBTREE_MAGIC;
167 #endif
168  tree->root = NIL;
169  tree->compare = compare;
170  tree->replace = (flags & RBTREE_FLAG_REPLACE) != 0 ? true : false;
171  tree->lock = (flags & RBTREE_FLAG_LOCK) != 0 ? true : false;
172  tree->node_ctx = talloc_new(tree);
173  if (tree->lock) pthread_mutex_init(&tree->mutex, NULL);
174 
175  talloc_set_destructor(tree, _tree_free);
176  tree->free = node_free;
177  tree->type = type;
178 
179  return tree;

182 /** Rotate Node x to left
183  *
184  */
185 static void rotate_left(rbtree_t *tree, rbnode_t *x)
186 {
187 
188  rbnode_t *y = x->right;
189 
190  /* establish x->right link */
191  x->right = y->left;
192  if (y->left != NIL) y->left->parent = x;
193 
194  /* establish y->parent link */
195  if (y != NIL) y->parent = x->parent;
196  if (x->parent) {
197  if (x == x->parent->left) {
198  x->parent->left = y;
199  } else {
200  x->parent->right = y;
201  }
202  } else {
203  tree->root = y;
204  }
205 
206  /* link x and y */
207  y->left = x;
208  if (x != NIL) x->parent = y;
209 }

211 /** Rotate Node x to right
212  *
213  */
214 static void rotate_right(rbtree_t *tree, rbnode_t *x)
215 {
216  rbnode_t *y = x->left;
217 
218  /* establish x->left link */
219  x->left = y->right;
220  if (y->right != NIL) y->right->parent = x;
221 
222  /* establish y->parent link */
223  if (y != NIL) y->parent = x->parent;
224  if (x->parent) {
225  if (x == x->parent->right) {
226  x->parent->right = y;
227  } else {
228  x->parent->left = y;
229  }
230  } else {
231  tree->root = y;
232  }
233 
234  /* link x and y */
235  y->right = x;
236  if (x != NIL) x->parent = y;

239 /** Maintain red-black tree balance after inserting node x
240  *
241  */
242 static void insert_fixup(rbtree_t *tree, rbnode_t *x)
243 {
244  /* check RED-BLACK properties */
245  while ((x != tree->root) && (x->parent->colour == RED)) {
246  /* we have a violation */
247  if (x->parent == x->parent->parent->left) {
248  rbnode_t *y = x->parent->parent->right;
249  if (y->colour == RED) {
250 
251  /* uncle is RED */
252  x->parent->colour = BLACK;
253  y->colour = BLACK;
254  x->parent->parent->colour = RED;
255  x = x->parent->parent;
256  } else {
257 
258  /* uncle is BLACK */
259  if (x == x->parent->right) {
260  /* make x a left child */
261  x = x->parent;
262  rotate_left(tree, x);
263  }
264 
265  /* recolour and rotate */
266  x->parent->colour = BLACK;
267  x->parent->parent->colour = RED;
268  rotate_right(tree, x->parent->parent);
269  }
270  } else {
271 
272  /* mirror image of above code */
273  rbnode_t *y = x->parent->parent->left;
274  if (y->colour == RED) {
275 
276  /* uncle is RED */
277  x->parent->colour = BLACK;
278  y->colour = BLACK;
279  x->parent->parent->colour = RED;
280  x = x->parent->parent;
281  } else {
282 
283  /* uncle is BLACK */
284  if (x == x->parent->left) {
285  x = x->parent;
286  rotate_right(tree, x);
287  }
288  x->parent->colour = BLACK;
289  x->parent->parent->colour = RED;
290  rotate_left(tree, x->parent->parent);
291  }
292  }
293  }
294 
295  tree->root->colour = BLACK;
296 }

299 /** Insert an element into the tree
300  *
301  */
302 rbnode_t *rbtree_insert_node(rbtree_t *tree, void *data)
303 {
304  rbnode_t *current, *parent, *x;
305 
306  if (unlikely(tree->being_freed)) return NULL;
307 
308 #ifndef TALLOC_GET_TYPE_ABORT_NOOP
309  if (tree->type) (void)_talloc_get_type_abort(data, tree->type, __location__);
310 #endif
311 
312  if (tree->lock) pthread_mutex_lock(&tree->mutex);
313 
314  /* find where node belongs */
315  current = tree->root;
316  parent = NULL;
317  while (current != NIL) {
318  int result;
319 
320  /*
321  * See if two entries are identical.
322  */
323  result = tree->compare(data, current->data);
324  if (result == 0) {
325  /*
326  * Don't replace the entry.
327  */
328  if (!tree->replace) {
329  if (tree->lock) pthread_mutex_unlock(&tree->mutex);
330  return NULL;
331  }
332 
333  /*
334  * Do replace the entry.
335  */
336  rbtree_free_data(tree, current->data);
337  current->data = data;
338  if (tree->lock) pthread_mutex_unlock(&tree->mutex);
339  return current;
340  }
341 
342  parent = current;
343  current = (result < 0) ? current->left : current->right;
344  }
345 
346  /* setup new node */
347  x = talloc_zero(tree->node_ctx, rbnode_t);
348  if (!x) {
349  fr_strerror_printf("No memory for new rbtree node");
350  if (tree->lock) pthread_mutex_unlock(&tree->mutex);
351  return NULL;
352  }
353 
354  x->data = data;
355  x->parent = parent;
356  x->left = NIL;
357  x->right = NIL;
358  x->colour = RED;
359 
360  /* insert node in tree */
361  if (parent) {
362  if (tree->compare(data, parent->data) <= 0) {
363  parent->left = x;
364  } else {
365  parent->right = x;
366  }
367  } else {
368  tree->root = x;
369  }
370 
371  insert_fixup(tree, x);
372 
373  tree->num_elements++;
374 
375  if (tree->lock) pthread_mutex_unlock(&tree->mutex);
376  return x;
377 }

391 /** Maintain RED-BLACK tree balance after deleting node x
392  *
393  */
394 static void delete_fixup(rbtree_t *tree, rbnode_t *x, rbnode_t *parent)
395 {
396 
397  while (x != tree->root && x->colour == BLACK) {
398  if (x == parent->left) {
399  rbnode_t *w = parent->right;
400  if (w->colour == RED) {
401  w->colour = BLACK;
402  parent->colour = RED; /* parent != NIL? */
403  rotate_left(tree, parent);
404  w = parent->right;
405  }
406  if ((w->left->colour == BLACK) && (w->right->colour == BLACK)) {
407  if (w != NIL) w->colour = RED;
408  x = parent;
409  parent = x->parent;
410  } else {
411  if (w->right->colour == BLACK) {
412  if (w->left != NIL) w->left->colour = BLACK;
413  w->colour = RED;
414  rotate_right(tree, w);
415  w = parent->right;
416  }
417  w->colour = parent->colour;
418  if (parent != NIL) parent->colour = BLACK;
419  if (w->right->colour != BLACK) {
420  w->right->colour = BLACK;
421  }
422  rotate_left(tree, parent);
423  x = tree->root;
424  }
425  } else {
426  rbnode_t *w = parent->left;
427  if (w->colour == RED) {
428  w->colour = BLACK;
429  parent->colour = RED; /* parent != NIL? */
430  rotate_right(tree, parent);
431  w = parent->left;
432  }
433  if ((w->right->colour == BLACK) && (w->left->colour == BLACK)) {
434  if (w != NIL) w->colour = RED;
435  x = parent;
436  parent = x->parent;
437  } else {
438  if (w->left->colour == BLACK) {
439  if (w->right != NIL) w->right->colour = BLACK;
440  w->colour = RED;
441  rotate_left(tree, w);
442  w = parent->left;
443  }
444  w->colour = parent->colour;
445  if (parent != NIL) parent->colour = BLACK;
446  if (w->left->colour != BLACK) {
447  w->left->colour = BLACK;
448  }
449  rotate_right(tree, parent);
450  x = tree->root;
451  }
452  }
453  }
454  if (x != NIL) x->colour = BLACK; /* Avoid cache-dirty on NIL */
455 }

457 /** Delete an element (z) from the tree
458  *
459  */
460 static void rbtree_delete_internal(rbtree_t *tree, rbnode_t *z, bool skiplock)
461 {
462  rbnode_t *x, *y;
463  rbnode_t *parent;
464 
465  if (!z || z == NIL) return;
466 
467  if (!skiplock) {
468  if (tree->lock) pthread_mutex_lock(&tree->mutex);
469  }
470 
471  if (z->left == NIL || z->right == NIL) {
472  /* y has a NIL node as a child */
473  y = z;
474  } else {
475  /* find tree successor with a NIL node as a child */
476  y = z->right;
477  while (y->left != NIL) y = y->left;
478  }
479 
480  /* x is y's only child */
481  if (y->left != NIL) {
482  x = y->left;
483  } else {
484  x = y->right; /* may be NIL! */
485  }
486 
487  /* remove y from the parent chain */
488  parent = y->parent;
489  if (x != NIL) x->parent = parent;
490 
491  if (parent) {
492  if (y == parent->left) {
493  parent->left = x;
494  } else {
495  parent->right = x;
496  }
497  } else {
498  tree->root = x;
499  }
500 
501  if (y != z) {
502  rbtree_free_data(tree, z);
503  z->data = y->data;
504  y->data = NULL;
505 
506  if ((y->colour == BLACK) && parent) {
507  delete_fixup(tree, x, parent);
508  }
509 
510  /*
511  * The user structure in y->data MAy include a
512  * pointer to y. In that case, we CANNOT delete
513  * y. Instead, we copy z (which is now in the
514  * tree) to y, and fix up the parent/child
515  * pointers.
516  */
517  memcpy(y, z, sizeof(*y));
518 
519  if (!y->parent) {
520  tree->root = y;
521  } else {
522  if (y->parent->left == z) y->parent->left = y;
523  if (y->parent->right == z) y->parent->right = y;
524  }
525  if (y->left->parent == z) y->left->parent = y;
526  if (y->right->parent == z) y->right->parent = y;
527 
528  talloc_free(z);
529 
530  } else {
531  rbtree_free_data(tree, y);
532 
533  if (y->colour == BLACK)
534  delete_fixup(tree, x, parent);
535 
536  talloc_free(y);
537  }
538 
539  tree->num_elements--;
540  if (!skiplock) {
541  if (tree->lock) pthread_mutex_unlock(&tree->mutex);
542  }
543 }

615 /** Walk the tree, Pre-order
616  *
617  * We call ourselves recursively for each function, but that's OK,
618  * as the stack is only log(N) deep, which is ~12 entries deep.
619  */
620 static int walk_node_pre_order(rbnode_t *x, rb_walker_t compare, void *uctx)
621 {
622  int ret;
623  rbnode_t *left, *right;
624 
625  left = x->left;
626  right = x->right;
627 
628  ret = compare(x->data, uctx);
629  if (ret != 0) return ret;
630 
631  if (left != NIL) {
632  ret = walk_node_pre_order(left, compare, uctx);
633  if (ret != 0) return ret;
634  }
635 
636  if (right != NIL) {
637  ret = walk_node_pre_order(right, compare, uctx);
638  if (ret != 0) return ret;
639  }
640 
641  return 0; /* we know everything returned zero */
642 }

647 static int walk_node_in_order(rbnode_t *x, rb_walker_t compare, void *uctx)
648 {

651

695 /** rbtree_delete_order
696  *
697  * This executes an rbtree_in_order-like walk that adapts to changes in the
698  * tree above it, which may occur because we allow the compare to
699  * tell us to delete the current node.
700  *
701  * The compare should return:
702  *
703  * < 0 - on error
704  * 0 - continue walking, don't delete the node
705  * 1 - delete the node and stop walking
706  * 2 - delete the node and continue walking
707  */
708 static int walk_delete_order(rbtree_t *tree, rb_walker_t compare, void *uctx)
709 {
710  rbnode_t *solid, *x;
711  int ret = 0;
712 
713  /* Keep track of last node that refused deletion. */
714  solid = NIL;
715  while (solid == NIL) {
716  x = tree->root;
717  if (x == NIL) break;
718  descend:
719  while (x->left != NIL) {
720  x = x->left;
721  }
722  visit:
723  ret = compare(x->data, uctx);
724  if (ret < 0) {
725  return ret;
726  }
727  if (ret) {
728  rbtree_delete_internal(tree, x, true);
729  if (ret != 2) {
730  return ret;
731  }
732  } else {
733  solid = x;
734  }
735  }
736  if (solid != NIL) {
737  x = solid;
738  if (x->right != NIL) {
739  x = x->right;
740  goto descend;
741  }
742  while (x->parent) {
743  if (x->parent->left == x) {
744  x = x->parent;
745  goto visit;
746  }
747  x = x->parent;
748  }
749  }
750  return ret;
751 }

754 /*
755  * walk the entire tree. The compare function CANNOT modify
756  * the tree.
757  *
758  * The compare function should return 0 to continue walking.
759  * Any other value stops the walk, and is returned.
760  */
761 int rbtree_walk(rbtree_t *tree, rb_order_t order, rb_walker_t compare, void *uctx)
762 {
763  int ret;
764 
765  if (tree->root == NIL) return 0;
766 
767  if (tree->lock) pthread_mutex_lock(&tree->mutex);
768 
769  switch (order) {
770  case RBTREE_PRE_ORDER:
771  ret = walk_node_pre_order(tree->root, compare, uctx);
772  break;
