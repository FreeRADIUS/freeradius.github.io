26 /** Print the value of an attribute to a string
27  *
28  * @param[in] out Where to write the string.
29  * @param[in] vp to print.
30  * @param[in] quote Char to add before and after printed value,
31  * if 0 no char will be added, if < 0 raw string
32  * will be added.
33  * @return
34  * - >= 0 length of data written to out.
35  * - <0 the number of bytes we would have needed to write
36  * the complete string to out.
37  */
38 ssize_t fr_pair_print_value_quoted(fr_sbuff_t *out, fr_pair_t const *vp, fr_token_t quote)
39 {
40  fr_sbuff_t our_out;
41  fr_pair_t *child, *head;
42  fr_cursor_t cursor;
43 
44  VP_VERIFY(vp);
45 
46  /*
47  * Legacy crap that needs to be removed
48  */
49  if (vp->type == VT_XLAT) {
50  char const *quote_str = fr_table_str_by_value(fr_token_quotes_table, quote, "");
51 
52  return fr_sbuff_in_sprintf(out, "%s%s%s", quote_str, vp->xlat, quote_str);
53  }
54 
55  /*
56  * For simple types just print the box
57  */
58  if (vp->da->type != FR_TYPE_GROUP) {
59  return fr_value_box_print_quoted(out, &vp->data, quote);
60  }
61 
62  /*
63  * Serialize all child VPs as full quoted
64  * <pair> = ["]<child>["]
65  */
66  our_out = FR_SBUFF_NO_ADVANCE(out);
67 
68  head = vp->vp_ptr;
69  if (!fr_cond_assert(head != NULL)) return 0;
70 
71  FR_SBUFF_IN_CHAR_RETURN(&our_out, '{', ' ');
72  for (child = fr_cursor_init(&cursor, &head);
73  child != NULL;
74  child = fr_cursor_next(&cursor)) {
75  VP_VERIFY(child);
76 
77  FR_SBUFF_RETURN(fr_pair_print, &our_out, child);
78  FR_SBUFF_IN_CHAR_RETURN(&our_out, ',', ' ');
79  }
80 
81  if (fr_sbuff_used(&our_out)) {
82  fr_sbuff_set(&our_out, fr_sbuff_current(&our_out) - 2);
83  *fr_sbuff_current(&our_out) = '\0';
84  }
85 
86  FR_SBUFF_IN_CHAR_RETURN(&our_out, ' ', '}');
87 
88  return fr_sbuff_set(out, &our_out);
89 }

91 /** Print one attribute and value to a string
92  *
93  * Print a fr_pair_t in the format:
94 @verbatim
95  <attribute_name>[:tag] <op> [q]<value>[q]
96 @endverbatim
97  * to a string.
98  *
99  * @param[in] out Where to write the string.
100  * @param[in] vp to print.
101  * @return
102  * - Length of data written to out.
103  * - value >= outlen on truncation.
104  */
105 ssize_t fr_pair_print(fr_sbuff_t *out, fr_pair_t const *vp)
106 {
107  char const *token = NULL;
108  fr_sbuff_t our_out = FR_SBUFF_NO_ADVANCE(out);
109 
110  if (!out) return 0;
111 
112  if (!vp || !vp->da) return 0;
113 
114  VP_VERIFY(vp);
115 
116  if ((vp->op > T_INVALID) && (vp->op < T_TOKEN_LAST)) {
117  token = fr_tokens[vp->op];
118  } else {
119  token = "<INVALID-TOKEN>";
120  }
121 
122  FR_SBUFF_IN_SPRINTF_RETURN(&our_out, "%s %s ", vp->da->name, token);
123  FR_SBUFF_RETURN(fr_pair_print_value_quoted, &our_out, vp, T_DOUBLE_QUOTED_STRING);
124 
125  return fr_sbuff_set(out, &our_out);
126 }

128 /** Print one attribute and value to FP
129  *
130  * Complete string with '\\t' and '\\n' is written to buffer before printing to
131  * avoid issues when running with multiple threads.
132  *
133  * @param fp to output to.
134  * @param vp to print.
135  */
136 void fr_pair_fprint(FILE *fp, fr_pair_t const *vp)
137 {
138  char buff[1024];
139  fr_sbuff_t sbuff = FR_SBUFF_OUT(buff, sizeof(buff));
140 
141  if (!fp) return;
142  VP_VERIFY(vp);
143 
144  fr_sbuff_in_char(&sbuff, '\t');
145  fr_pair_print(&sbuff, vp);
146  fr_sbuff_in_char(&sbuff, '\n');
147 
148  fputs(buff, fp);
149 }
