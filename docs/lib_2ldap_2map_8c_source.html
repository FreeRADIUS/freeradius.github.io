33 /** Callback for map_to_request
34  *
35  * Performs exactly the same job as map_to_vp, but pulls attribute values from LDAP entries
36  *
37  * @see map_to_vp
38  */
39 int fr_ldap_map_getvalue(TALLOC_CTX *ctx, fr_pair_t **out, request_t *request, map_t const *map, void *uctx)
40 {
41  fr_ldap_result_t *self = uctx;
42  fr_pair_t *head = NULL, *vp;
43  fr_cursor_t cursor, to_append;
44  int i;
45 
46  fr_cursor_init(&cursor, &head);
47 
48  switch (map->lhs->type) {
49  /*
50  * This is a mapping in the form of:
51  * <list>: += <ldap attr>
52  *
53  * Where <ldap attr> is:
54  * <list>:<attr> <op> <value>
55  *
56  * It is to allow for legacy installations which stored
57  * RADIUS control and reply attributes in separate LDAP
58  * attributes.
59  */
60  case TMPL_TYPE_LIST:
61  for (i = 0; i < self->count; i++) {
62  map_t *attr = NULL;
63  char *attr_str;
64 
65  tmpl_rules_t lhs_rules = {
66  .dict_def = request->dict,
67  .request_def = tmpl_request(map->lhs),
68  .list_def = tmpl_list(map->lhs),
69  .prefix = TMPL_ATTR_REF_PREFIX_AUTO,
70  };
71 
72  tmpl_rules_t rhs_rules = {
73  .dict_def = request->dict
74  };
75 
76  RDEBUG3("Parsing valuepair string \"%pV\"",
77  fr_box_strvalue_len(self->values[i]->bv_val, self->values[i]->bv_len));
78 
79  /*
80  * bv_val is NOT \0 terminated, so we need to make it
81  * safe (\0 terminate it) before passing it to any
82  * functions which take C strings and no lengths.
83  */
84  attr_str = talloc_bstrndup(NULL, self->values[i]->bv_val, self->values[i]->bv_len);
85  if (!attr_str) {
86  RWDEBUG("Failed making attribute string safe");
87  continue;
88  }
89 
90  if (map_afrom_attr_str(ctx, &attr,
91  attr_str,
92  &lhs_rules, &rhs_rules) < 0) {
93  RPWDEBUG("Failed parsing \"%pV\" as valuepair, skipping...",
94  fr_box_strvalue_len(self->values[i]->bv_val, self->values[i]->bv_len));
95  talloc_free(attr_str);
96  continue;
97  }
98 
99  talloc_free(attr_str);
100 
101  if (tmpl_is_unresolved(attr->lhs)) {
102  RWDEBUG("Failed parsing left side of \"%pV\", skipping...",
103  fr_box_strvalue_len(self->values[i]->bv_val, self->values[i]->bv_len));
104  talloc_free(attr);
105  continue;
106  }
107 
108  if (tmpl_request(attr->lhs) != tmpl_request(map->lhs)) {
109  RWDEBUG("valuepair \"%pV\" has conflicting request qualifier (%s vs %s), skipping...",
110  fr_box_strvalue_len(self->values[i]->bv_val, self->values[i]->bv_len),
111  fr_table_str_by_value(request_ref_table, tmpl_request(attr->lhs), "<INVALID>"),
112  fr_table_str_by_value(request_ref_table, tmpl_request(map->lhs), "<INVALID>"));
113  next_pair:
114  talloc_free(attr);
115  continue;
116  }
117 
118  if ((tmpl_list(attr->lhs) != tmpl_list(map->lhs))) {
119  RWDEBUG("valuepair \"%pV\" has conflicting list qualifier (%s vs %s), skipping...",
120  fr_box_strvalue_len(self->values[i]->bv_val, self->values[i]->bv_len),
121  fr_table_str_by_value(pair_list_table, tmpl_list(attr->lhs), "<INVALID>"),
122  fr_table_str_by_value(pair_list_table, tmpl_list(map->lhs), "<INVALID>"));
123  goto next_pair;
124  }
125 
126  if (map_to_vp(ctx, &vp, request, attr, NULL) < 0) {
127  RWDEBUG("Failed creating attribute for valuepair \"%pV\", skipping...",
128  fr_box_strvalue_len(self->values[i]->bv_val, self->values[i]->bv_len));
129  goto next_pair;
130  }
131 
132  fr_cursor_init(&to_append, &vp);
133  fr_cursor_merge(&cursor, &to_append);
134  talloc_free(attr);
135 
136  /*
137  * Only process the first value, unless the operator is +=
138  */
139  if (map->op != T_OP_ADD) break;
140  }
141  break;
142 
143  /*
144  * Iterate over all the retrieved values,
145  * don't try and be clever about changing operators
146  * just use whatever was set in the attribute map.
147  */
148  case TMPL_TYPE_ATTR:
149  for (i = 0; i < self->count; i++) {
150  if (!self->values[i]->bv_len) continue;
151 
152  MEM(vp = fr_pair_afrom_da(ctx, tmpl_da(map->lhs)));
153 
154  if (fr_pair_value_from_str(vp, self->values[i]->bv_val,
155  self->values[i]->bv_len, '\0', true) < 0) {
156  RPWDEBUG("Failed parsing value \"%pV\" for attribute %s",
157  fr_box_strvalue_len(self->values[i]->bv_val, self->values[i]->bv_len),
158  tmpl_da(map->lhs)->name);
159 
160  talloc_free(vp); /* also frees escaped */
161  continue;
162  }
163 
164  vp->op = map->op;
165  fr_cursor_append(&cursor, vp);
166 
167  /*
168  * Only process the first value, unless the operator is +=
169  */
170  if (map->op != T_OP_ADD) break;
171  }
172  break;
173 
174  default:
175  fr_assert(0);
176  }
177 
178  *out = head;
179 
180  return 0;
181 }

183 int fr_ldap_map_verify(map_t *map, UNUSED void *instance)
184 {
185  /*
186  * Destinations where we can put the fr_pair_ts we
187  * create using LDAP values.
188  */
189  switch (map->lhs->type) {
190  case TMPL_TYPE_LIST:
191  case TMPL_TYPE_ATTR:
192  break;
193 
194  case TMPL_TYPE_ATTR_UNRESOLVED:
195  cf_log_err(map->ci, "Unknown attribute %s", tmpl_attr_unresolved(map->lhs));
196  return -1;
197 
198  default:
199  cf_log_err(map->ci, "Left hand side of map must be an attribute or list, not a %s",
200  fr_table_str_by_value(tmpl_type_table, map->lhs->type, "<INVALID>"));
201  return -1;
202  }
203 
204  /*
205  * Sources we can use to get the name of the attribute
206  * we're retrieving from LDAP.
207  */
208  switch (map->rhs->type) {
209  case TMPL_TYPE_XLAT_UNRESOLVED:
210  case TMPL_TYPE_ATTR:
211  case TMPL_TYPE_EXEC:
212  case TMPL_TYPE_UNRESOLVED:
213  break;
214 
215  case TMPL_TYPE_ATTR_UNRESOLVED:
216  cf_log_err(map->ci, "Unknown attribute %s", tmpl_attr_unresolved(map->rhs));
217  return -1;
218 
219  default:
220  cf_log_err(map->ci, "Right hand side of map must be an xlat, attribute, exec, or literal, not a %s",
221  fr_table_str_by_value(tmpl_type_table, map->rhs->type, "<INVALID>"));
222  return -1;
223  }
224 
225  /*
226  * Only =, :=, += and -= operators are supported for LDAP mappings.
227  */
228  switch (map->op) {
229  case T_OP_SET:
230  case T_OP_EQ:
231  case T_OP_SUB:
232  case T_OP_ADD:
233  break;
234 
235  default:
236  cf_log_err(map->ci, "Operator \"%s\" not allowed for LDAP mappings",
237  fr_table_str_by_value(fr_tokens_table, map->op, "<INVALID>"));
238  return -1;
239  }
240 
241  return 0;
242 }

244 /** Expand values in an attribute map where needed
245  *
246  * @param[out] expanded array of attributes. Need not be initialised (we'll initialise).
247  * @param[in] request The current request.
248  * @param[in] maps to expand.
249  * @return
250  * - 0 on success.
251  * - -1 on failure.
252  */
253 int fr_ldap_map_expand(fr_ldap_map_exp_t *expanded, request_t *request, map_t const *maps)
254 {
255  map_t const *map;
256  unsigned int total = 0;
257 
258  TALLOC_CTX *ctx = NULL;
259  char const *attr;
260  char attr_buff[1024 + 1]; /* X.501 says we need to support at least 1024 chars for attr names */
261 
262  for (map = maps; map != NULL; map = map->next) {
263  if (tmpl_expand(&attr, attr_buff, sizeof(attr_buff), request, map->rhs, NULL, NULL) < 0) {
264  REDEBUG("Expansion of LDAP attribute \"%s\" failed", map->rhs->name);
265  TALLOC_FREE(ctx);
266  return -1;
267  }
268 
269  /*
270  * Dynamic value
271  */
272  if (attr == attr_buff) {
273  if (!ctx) ctx = talloc_new(NULL);
274  expanded->attrs[total++] = talloc_strdup(ctx, attr_buff);
275  continue;
276  }
277  expanded->attrs[total++] = attr;
278  }
279  expanded->attrs[total] = NULL;
280  expanded->ctx = ctx; /* Freeing this frees any dynamic values */
281  expanded->count = total;
282  expanded->maps = maps;
283 
284  return 0;
285 }

288 /** Convert attribute map into valuepairs
289  *
290  * Use the attribute map built earlier to convert LDAP values into valuepairs and insert them into whichever
291  * list they need to go into.
292  *
293  * This is *NOT* atomic, but there's no condition for which we should error out...
294  *
295  * @param[in] request Current request.
296  * @param[in] conn associated with entry.
297  * @param[in] valuepair_attr Treat attribute with this name as holding complete AVP definitions.
298  * @param[in] expanded attributes (rhs of map).
299  * @param[in] entry to retrieve attributes from.
300  * @return
301  * - Number of maps successfully applied.
302  * - -1 on failure.
303  */
304 int fr_ldap_map_do(request_t *request, fr_ldap_connection_t *conn,
305  char const *valuepair_attr, fr_ldap_map_exp_t const *expanded, LDAPMessage *entry)
306 {
307  map_t const *map;
308  unsigned int total = 0;
309  int applied = 0; /* How many maps have been applied to the current request */
310 
311  fr_ldap_result_t result;
312  char const *name;
313 
314  for (map = expanded->maps; map != NULL; map = map->next) {
315  int ret;
316 
317  name = expanded->attrs[total++];
318 
319  /*
320  * Binary safe
321  */
322  result.values = ldap_get_values_len(conn->handle, entry, name);
323  if (!result.values) {
324  RDEBUG3("Attribute \"%s\" not found in LDAP object", name);
325 
326  goto next;
327  }
328 
329  /*
330  * Find out how many values there are for the
331  * attribute and extract all of them.
332  */
333  result.count = ldap_count_values_len(result.values);
334 
335  /*
336  * If something bad happened, just skip, this is probably
337  * a case of the dst being incorrect for the current
338  * request context
339  */
340  ret = map_to_request(request, map, fr_ldap_map_getvalue, &result);
341  if (ret == -1) return -1; /* Fail */
342 
343  /*
344  * How many maps we've processed
345  */
346  applied++;
347 
348  next:
349  ldap_value_free_len(result.values);
350  }
351 
352 
353  /*
354  * Retrieve any valuepair attributes from the result, these are generic values specifying
355  * a radius list, operator and value.
356  */
357  if (valuepair_attr) {
358  struct berval **values;
359  int count, i;
360 
361  values = ldap_get_values_len(conn->handle, entry, valuepair_attr);
362  count = ldap_count_values_len(values);
363 
364  for (i = 0; i < count; i++) {
365  map_t *attr;
366  char *value;
367 
368  tmpl_rules_t parse_rules = {
369  .dict_def = request->dict,
370  .prefix = TMPL_ATTR_REF_PREFIX_AUTO,
371  };
372 
373  value = fr_ldap_berval_to_string(request, values[i]);
374  RDEBUG3("Parsing attribute string '%s'", value);
375  if (map_afrom_attr_str(request, &attr, value,
376  &parse_rules, &parse_rules) < 0) {
377  RPWDEBUG("Failed parsing '%s' value \"%s\" as valuepair, skipping...",
378  valuepair_attr, value);
379  talloc_free(value);
380  continue;
381  }
382  if (map_to_request(request, attr, map_to_vp, NULL) < 0) {
383  RWDEBUG("Failed adding \"%s\" to request, skipping...", value);
384  } else {
385  applied++;
386  }
387  talloc_free(attr);
388  talloc_free(value);
389  }
390  ldap_value_free_len(values);
391  }
392 
393  return applied;
394 }
