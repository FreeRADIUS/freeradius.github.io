236 /** Write out an unsigned 64bit integer in wire format using the fewest bytes possible
237  *
238  * @param[out] out Where to write big endian encoding of num.
239  * Should be at least 8 bytes.
240  * @param[in] num Number to encode.
241  * @return the number of bytes written to out.
242  */
243 static inline size_t fr_net_from_uint64v(uint8_t out[static sizeof(uint64_t)], uint64_t num)
244 {
245  size_t ret;
246 
247  ret = ROUND_UP_DIV((size_t)fr_high_bit_pos(num | 0x80), 8);
248  num = ntohll(num);
249  memcpy(out, ((uint8_t *)&num) + (sizeof(uint64_t) - ret), ret); /* aligned */
250 
251  return ret;
252 }

302 /** Read an unsigned 64bit integer from wire format (big endian) with a variable length encoding
303  *
304  * @param[in] data Buffer containing the number.
305  * @param[in] data_len Length of number.
306  * @return a 64 bit unsigned integer of native endianness.
307  */
308 static inline uint64_t fr_net_to_uint64v(uint8_t const *data, size_t data_len)
309 {
310  uint64_t num = 0;
311 
312  if (unlikely(data_len > sizeof(uint64_t))) return 0;
313 
314  /*
315  * Copy at an offset into memory
316  * allocated for the uin64_t
317  */
318  memcpy(((uint8_t *)&num) + (sizeof(uint64_t) - data_len), data, data_len); /* aligned */
319  return ntohll(num);
320 }
