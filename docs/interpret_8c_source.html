157 /** Push a new frame onto the stack
158  *
159  * @param[in] request to push the frame onto.
160  * @param[in] instruction One or more unlang_t nodes describing the operations to execute.
161  * @param[in] default_rcode The default result.
162  * @param[in] do_next_sibling Whether to only execute the first node in the #unlang_t program
163  * or to execute subsequent nodes.
164  * @param[in] top_frame Return out of the unlang interpreter when popping this frame.
165  * Hands execution back to whatever called the interpreter.
166  * @return
167  * - 0 on success.
168  * - -1 on call stack too deep.
169  */
170 int unlang_interpret_push(request_t *request, unlang_t *instruction,
171  rlm_rcode_t default_rcode, bool do_next_sibling, bool top_frame)
172 {
173  unlang_stack_t *stack = request->stack;
174  unlang_stack_frame_t *frame;
175 
176  fr_assert(instruction || top_frame);
177 
178 #ifndef NDEBUG
179  if (DEBUG_ENABLED5) RDEBUG3("unlang_interpret_push called with instruction %s - args %s %s",
180  instruction ? instruction->debug_name : "<none>",
181  do_next_sibling ? "UNLANG_NEXT_SIBLING" : "UNLANG_NEXT_STOP",
182  top_frame ? "UNLANG_TOP_FRAME" : "UNLANG_SUB_FRAME");
183 #endif
184 
185  /*
186  * This is not a cancellation point.
187  *
188  * If we cancel here bad things happen inside the interpreter.
189  */
190  if (stack->depth >= (UNLANG_STACK_MAX - 1)) {
191  RERROR("Call stack is too deep");
192  return - 1;
193  }
194 
195  stack->depth++;
196 
197  /*
198  * Initialize the next stack frame.
199  */
200  frame = &stack->frame[stack->depth];
201  memset(frame, 0, sizeof(*frame));
202 
203  frame->instruction = instruction;
204 
205  if (do_next_sibling) {
206  fr_assert(instruction != NULL);
207  frame->next = instruction->next;
208  }
209  /* else frame->next MUST be NULL */
210 
211  frame->uflags = UNWIND_FLAG_NONE;
212  if (top_frame) top_frame_set(frame);
213 
214  frame->result = default_rcode;
215  frame->priority = -1;
216 
217  if (!instruction) return 0;
218 
219  frame_state_init(stack, frame);
220 
221  return 0;
222 }

224 /** Update the current result after each instruction, and after popping each stack frame
225  *
226  * @param[in] request The current request.
227  * @param[in] frame The curren stack frame.
228  * @param[in,out] result The current section result.
229  * @param[in,out] priority The current section priority.
230  * @return
231  * - UNLANG_FRAME_ACTION_NEXT evaluate more instructions.
232  * - UNLANG_FRAME_ACTION_POP the final result has been calculated for this frame.
233  */
234 static inline unlang_frame_action_t result_calculate(request_t *request, unlang_stack_frame_t *frame,
235  rlm_rcode_t *result, int *priority)
236 {
237  unlang_t const *instruction = frame->instruction;
238  unlang_stack_t *stack = request->stack;
239 
240  RDEBUG4("** [%i] %s - have (%s %d) module returned (%s %d)",
241  stack->depth, __FUNCTION__,
242  fr_table_str_by_value(mod_rcode_table, frame->result, "<invalid>"),
243  frame->priority,
244  fr_table_str_by_value(mod_rcode_table, *result, "<invalid>"),
245  *priority);
246 
247  /*
248  * Don't set action or priority if we don't have one.
249  */
250  if (*result == RLM_MODULE_UNKNOWN) return UNLANG_FRAME_ACTION_NEXT;
251 
252  /*
253  * The child's action says return. Do so.
254  */
255  if (instruction->actions[*result] == MOD_ACTION_RETURN) {
256  if (*priority < 0) *priority = 0;
257 
258  RDEBUG4("** [%i] %s - action says to return with (%s %d)",
259  stack->depth, __FUNCTION__,
260  fr_table_str_by_value(mod_rcode_table, *result, "<invalid>"),
261  *priority);
262  frame->result = *result;
263  frame->priority = *priority;
264  return UNLANG_FRAME_ACTION_POP;
265  }
266 
267  /*
268  * If "reject", break out of the loop and return
269  * reject.
270  */
271  if (instruction->actions[*result] == MOD_ACTION_REJECT) {
272  if (*priority < 0) *priority = 0;
273 
274  RDEBUG4("** [%i] %s - action says to return with (%s %d)",
275  stack->depth, __FUNCTION__,
276  fr_table_str_by_value(mod_rcode_table, RLM_MODULE_REJECT, "<invalid>"),
277  *priority);
278  frame->result = RLM_MODULE_REJECT;
279  frame->priority = *priority;
280  return UNLANG_FRAME_ACTION_POP;
281  }
282 
283  /*
284  * The array holds a default priority for this return
285  * code. Grab it in preference to any unset priority.
286  */
287  if (*priority < 0) {
288  *priority = instruction->actions[*result];
289 
290  RDEBUG4("** [%i] %s - setting priority to (%s %d)",
291  stack->depth, __FUNCTION__,
292  fr_table_str_by_value(mod_rcode_table, *result, "<invalid>"),
293  *priority);
294  }
295 
296  /*
297  * We're higher than any previous priority, remember this
298  * return code and priority.
299  */
300  if (*priority > frame->priority) {
301  frame->result = *result;
302  frame->priority = *priority;
303 
304  RDEBUG4("** [%i] %s - over-riding result from higher priority to (%s %d)",
305  stack->depth, __FUNCTION__,
306  fr_table_str_by_value(mod_rcode_table, *result, "<invalid>"),
307  *priority);
308  }
309 
310  /*
311  * Not allowed in frame uflags...
312  */
313  fr_assert(!(frame->uflags & UNWIND_FLAG_NO_CLEAR));
314 
315  /*
316  * If we are unwinding the stack due to a break / return,
317  * then handle it now.
318  */
319  if (stack->unwind) {
320  /*
321  * Continue unwinding...
322  */
323  if (!(stack->unwind & frame->uflags) || (stack->unwind & UNWIND_FLAG_NO_CLEAR)) {
324  RDEBUG4("** [%i] %s - unwinding current frame with (%s %d) - flags - stack (%i), frame (%i)",
325  stack->depth, __FUNCTION__,
326  fr_table_str_by_value(mod_rcode_table, frame->result, "<invalid>"),
327  frame->priority, stack->unwind, frame->uflags);
328 
329  return UNLANG_FRAME_ACTION_POP;
330  }
331 
332  /*
333  * If we've been told to unwind, and we've hit
334  * the frame we should be unwinding to,
335  * and the "NO_CLEAR" flag hasn't been set, then
336  * clear the unwind field so we stop unwinding.
337  */
338  stack->unwind = UNWIND_FLAG_NONE;
339 
340  RDEBUG4("** [%i] %s - unwind stop (%s %d) - flags - stack (%i), frame (%i)",
341  stack->depth, __FUNCTION__,
342  fr_table_str_by_value(mod_rcode_table, frame->result, "<invalid>"),
343  frame->priority, stack->unwind, frame->uflags);
344  }
345 
346  return frame->next ? UNLANG_FRAME_ACTION_NEXT : UNLANG_FRAME_ACTION_POP;
347 }

382 /** Pop a stack frame, removing any associated dynamically allocated state
383  *
384  * @param[in] stack frame to pop.
385  */
386 static inline void frame_pop(unlang_stack_t *stack)
387 {
388  unlang_stack_frame_t *frame;
389 
390  fr_assert(stack->depth > 1);
391 
392  frame = &stack->frame[stack->depth];
393 
394  frame_cleanup(frame);
395 
396  frame = &stack->frame[--stack->depth];
397 
398  /*
399  * The child was break / return, AND the current frame is
400  * a break / return point. Stop unwinding the stack.
401  *
402  * NOTE: This is the frame further UP the stack, not the
403  * one we just popped. So we *MUST* clear repeat,
404  * otherwise the interpreter will just keep repeating
405  * this frame instead of continuing to unwind the stack.
406  *
407  * An alternative way of doing this would be to rearrange
408  * the code so repeat unwinding had precedence, but it's
409  * probably more robust to have the interpreter not be
410  * dependent on flag evaluation order.
411  */
412  if (stack->unwind && is_repeatable(frame)) repeatable_clear(frame);
413 }

415 /** Evaluates all the unlang nodes in a section
416  *
417  * @param[in] request The current request.
418  * @param[in] frame The curren stack frame.
419  * @param[in,out] result The current section result.
420  * @param[in,out] priority The current section priority.
421  * @return
422  * - UNLANG_FRAME_ACTION_NEXT evaluate more instructions in the current stack frame
423  * which may not be the same frame as when this function
424  * was called.
425  * - UNLANG_FRAME_ACTION_POP the final result has been calculated for this frame.
426  */
427 static inline unlang_frame_action_t frame_eval(request_t *request, unlang_stack_frame_t *frame,
428  rlm_rcode_t *result, int *priority)
429 {
430  unlang_stack_t *stack = request->stack;
431 
432  /*
433  * Loop over all the instructions in this list.
434  */
435  while (frame->instruction) {
436  unlang_t const *instruction = frame->instruction;
437  unlang_action_t action = UNLANG_ACTION_UNWIND;
438 
439  DUMP_STACK;
440 
441  fr_assert(instruction->debug_name != NULL); /* if this happens, all bets are off. */
442 
443  REQUEST_VERIFY(request);
444 
445  /*
446  * We're running this frame, so it can't possibly be yielded.
447  */
448  if (is_yielded(frame)) {
449  RDEBUG("%s - Resuming execution", instruction->debug_name);
450  yielded_clear(frame);
451  }
452 
453  /*
454  * unlang_interpret_signal() takes care of
455  * marking the requests as STOP on a CANCEL
456  * signal.
457  */
458  if (request->master_state == REQUEST_STOP_PROCESSING) {
459  do_stop:
460  frame->result = RLM_MODULE_FAIL;
461  frame->priority = 9999;
462 
463  RDEBUG4("** [%i] %s - STOP current subsection with (%s %d)",
464  stack->depth, __FUNCTION__,
465  fr_table_str_by_value(mod_rcode_table, frame->result, "<invalid>"),
466  frame->priority);
467 
468  unwind_all(stack);
469  return UNLANG_FRAME_ACTION_POP;
470  }
471 
472  if (!is_repeatable(frame) && (unlang_ops[instruction->type].debug_braces)) {
473  RDEBUG2("%s {", instruction->debug_name);
474  RINDENT();
475  }
476 
477  /*
478  * Execute an operation
479  */
480  RDEBUG4("** [%i] %s >> %s", stack->depth, __FUNCTION__,
481  unlang_ops[instruction->type].name);
482 
483  fr_assert(frame->process != NULL);
484  action = frame->process(result, request);
485 
486  RDEBUG4("** [%i] %s << %s (%d)", stack->depth, __FUNCTION__,
487  fr_table_str_by_value(unlang_action_table, action, "<INVALID>"), *priority);
488 
489  fr_assert(*priority >= -1);
490  fr_assert(*priority <= MOD_PRIORITY_MAX);
491 
492  switch (action) {
493  /*
494  * The request is now defunct, and we should not
495  * continue processing it.
496  */
497  case UNLANG_ACTION_STOP_PROCESSING:
498  goto do_stop;
499 
500  /*
501  * The operation resulted in additional frames
502  * being pushed onto the stack, execution should
503  * now continue at the deepest frame.
504  */
505  case UNLANG_ACTION_PUSHED_CHILD:
506  fr_assert(&stack->frame[stack->depth] > frame);
507  *result = frame->result;
508  return UNLANG_FRAME_ACTION_NEXT;
509 
510  /*
511  * We're in a looping construct and need to stop
512  * execution of the current section.
513  */
514  case UNLANG_ACTION_UNWIND:
515  if (*priority < 0) *priority = 0;
516  frame->result = *result;
517  frame->priority = *priority;
518  frame->next = NULL;
519  fr_assert(stack->unwind != UNWIND_FLAG_NONE);
520  return UNLANG_FRAME_ACTION_POP;
521 
522  /*
523  * Yield control back to the scheduler, or whatever
524  * called the interpreter.
525  */
526  case UNLANG_ACTION_YIELD:
527  *result = RLM_MODULE_YIELD; /* Fixup rcode */
528  yield:
529  /*
530  * Detach is magic. The parent "subrequest" function
531  * takes care of bumping the instruction
532  * pointer...
533  */
534  switch (frame->instruction->type) {
535  /*
536  * For sanity, we ensure that
537  * only known keywords can yield.
538  */
539  case UNLANG_TYPE_MODULE:
540  case UNLANG_TYPE_PARALLEL:
541  case UNLANG_TYPE_SUBREQUEST:
542  case UNLANG_TYPE_XLAT:
543  case UNLANG_TYPE_CALL:
544  case UNLANG_TYPE_DETACH:
545  case UNLANG_TYPE_FUNCTION:
546  case UNLANG_TYPE_TMPL:
547  repeatable_set(frame);
548  yielded_set(frame);
549  RDEBUG4("** [%i] %s - yielding with current (%s %d)", stack->depth, __FUNCTION__,
550  fr_table_str_by_value(mod_rcode_table, frame->result, "<invalid>"),
551  frame->priority);
552  DUMP_STACK;
553  return UNLANG_FRAME_ACTION_YIELD;
554 
555  default:
556  fr_assert(0);
557  return UNLANG_FRAME_ACTION_YIELD;
558  }
559 
560  /*
561  * Instruction finished execution,
562  * check to see what we need to do next, and update
563  * the section rcode and priority.
564  */
565  case UNLANG_ACTION_CALCULATE_RESULT:
566  /*
567  * If a module returns yield, then do
568  * yield. Only keywords will return
569  * UNLANG_ACTION_YIELD.
570  */
571  if (*result == RLM_MODULE_YIELD) goto yield;
572 
573  fr_assert(*result != RLM_MODULE_UNKNOWN);
574 
575  repeatable_clear(frame);
576 
577  if (unlang_ops[instruction->type].debug_braces) {
578  REXDENT();
579 
580  /*
581  * If we're at debug level 1, don't emit the closing
582  * brace as the opening brace wasn't emitted.
583  */
584  if (RDEBUG_ENABLED && !RDEBUG_ENABLED2) {
585  RDEBUG("# %s (%s)", instruction->debug_name,
586  fr_table_str_by_value(mod_rcode_table, *result, "<invalid>"));
587  } else {
588  RDEBUG2("} # %s (%s)", instruction->debug_name,
589  fr_table_str_by_value(mod_rcode_table, *result, "<invalid>"));
590  }
591  }
592 
593  *priority = instruction->actions[*result];
594 
595  if (result_calculate(request, frame, result, priority) == UNLANG_FRAME_ACTION_POP) {
596  return UNLANG_FRAME_ACTION_POP;
597  }
598  FALL_THROUGH;
599 
600  /*
601  * Execute the next instruction in this frame
602  */
603  case UNLANG_ACTION_EXECUTE_NEXT:
604  if ((action == UNLANG_ACTION_EXECUTE_NEXT) && unlang_ops[instruction->type].debug_braces) {
605  REXDENT();
606  RDEBUG2("}");
607  }
608  break;
609  } /* switch over return code from the interpreter function */
610 
611  frame_next(stack, frame);
612  }
613 
614  RDEBUG4("** [%i] %s - done current subsection with (%s %d)",
615  stack->depth, __FUNCTION__,
616  fr_table_str_by_value(mod_rcode_table, frame->result, "<invalid>"),
617  frame->priority);
618 
619  return UNLANG_FRAME_ACTION_POP;
620 }

622 /*
623  * Interpret the various types of blocks.
624  */
625 rlm_rcode_t unlang_interpret(request_t *request)
626 {
627  int priority = -1;
628  unlang_frame_action_t fa = UNLANG_FRAME_ACTION_NEXT;
629 
630  /*
631  * We don't have a return code yet.
632  */
633  unlang_stack_t *stack = request->stack;
634  unlang_stack_frame_t *frame = &stack->frame[stack->depth]; /* Quiet static analysis */
635 
636 #ifndef NDEBUG
637  if (DEBUG_ENABLED5) DEBUG("###### unlang_interpret is starting");
638  DUMP_STACK;
639 #endif
640 
641  fr_assert(!unlang_request_is_scheduled(request)); /* if we're running it, it can't be scheduled */
642 
643  RDEBUG4("** [%i] %s - interpreter entered", stack->depth, __FUNCTION__);
644 
645  for (;;) {
646  RDEBUG4("** [%i] %s - frame action %s", stack->depth, __FUNCTION__,
647  fr_table_str_by_value(unlang_frame_action_table, fa, "<INVALID>"));
648  switch (fa) {
649  case UNLANG_FRAME_ACTION_NEXT: /* Evaluate the current frame */
650  fr_assert(stack->depth > 0);
651  fr_assert(stack->depth < UNLANG_STACK_MAX);
652 
653  frame = &stack->frame[stack->depth];
654  fa = frame_eval(request, frame, &stack->result, &priority);
655 
656  /*
657  * We were executing a frame, frame_eval()
658  * indicated we should pop it, but we're now at
659  * a top_frame, so we need to break out of the loop
660  * and calculate the final result for this substack.
661  *
662  * Note that we only stop on a top frame.
663  * If there's a return point such as in a
664  * policy, then the "return" causes a
665  * "pop" until the return point. BUT we
666  * then continue execution with the next
667  * instruction. And we don't return all
668  * of the way up the stack.
669  */
670  if ((fa == UNLANG_FRAME_ACTION_POP) && is_top_frame(frame)) break; /* stop */
671  continue;
672 
673  case UNLANG_FRAME_ACTION_POP: /* Pop this frame and check the one beneath it */
674  /*
675  * The result / priority is returned from the sub-section,
676  * and made into our current result / priority, as
677  * if we had performed a module call.
678  */
679  stack->result = frame->result;
680  priority = frame->priority;
681 
682  /*
683  * Head on back up the stack
684  */
685  frame_pop(stack);
686  frame = &stack->frame[stack->depth];
687  DUMP_STACK;
688 
689  /*
690  * Resume a "foreach" loop, or a "load-balance" section
691  * or anything else that needs to be checked on the way
692  * back on up the stack.
693  */
694  if (is_repeatable(frame)) {
695  fa = UNLANG_FRAME_ACTION_NEXT;
696  continue;
697  }
698 
699  /*
700  * If we're done, merge the last stack->result / priority in.
701  */
702  if (is_top_frame(frame)) break; /* stop */
703 
704  /*
705  * Close out the section we entered earlier
706  */
707  if (unlang_ops[frame->instruction->type].debug_braces) {
708  REXDENT();
709 
710  /*
711  * If we're at debug level 1, don't emit the closing
712  * brace as the opening brace wasn't emitted.
713  */
714  if (RDEBUG_ENABLED && !RDEBUG_ENABLED2) {
715  RDEBUG("# %s (%s)", frame->instruction->debug_name,
716  fr_table_str_by_value(mod_rcode_table, stack->result, "<invalid>"));
717  } else {
718  RDEBUG2("} # %s (%s)", frame->instruction->debug_name,
719  fr_table_str_by_value(mod_rcode_table, stack->result, "<invalid>"));
720  }
721  }
722 
723  fa = result_calculate(request, frame, &stack->result, &priority);
724 
725  /*
726  * If we're continuing after popping a frame
727  * then we advance the instruction else we
728  * end up executing the same code over and over...
729  */
730  if (fa == UNLANG_FRAME_ACTION_NEXT) {
731  RDEBUG4("** [%i] %s - continuing after subsection with (%s %d)",
732  stack->depth, __FUNCTION__,
733  fr_table_str_by_value(mod_rcode_table, stack->result, "<invalid>"),
734  priority);
735  frame_next(stack, frame);
736  /*
737  * Else if we're really done with this frame
738  * print some helpful debug...
739  */
740  } else {
741  RDEBUG4("** [%i] %s - done current subsection with (%s %d)",
742  stack->depth, __FUNCTION__,
743  fr_table_str_by_value(mod_rcode_table, frame->result, "<invalid>"),
744  frame->priority);
745  }
746  continue;
747 
748  case UNLANG_FRAME_ACTION_YIELD:
749  fr_assert(stack->result == RLM_MODULE_YIELD);
750  RDEBUG4("** [%i] %s - interpreter yielding", stack->depth, __FUNCTION__);
751  return stack->result;
752  }
753  break;
754  }
755 
756  /*
757  * Nothing in this section, use the top frame stack->result.
758  */
759  if ((priority < 0) || (stack->result == RLM_MODULE_UNKNOWN)) {
760  RDEBUG4("** [%i] %s - empty section, using stack result (%s %d)", stack->depth, __FUNCTION__,
761  fr_table_str_by_value(mod_rcode_table, stack->result, "<invalid>"), priority);
762  stack->result = frame->result;
763  priority = frame->priority;
764  }
765 
766  if (priority > frame->priority) {
767  frame->result = stack->result;
768  frame->priority = priority;
769 
770  RDEBUG4("** [%i] %s - over-riding stack->result from higher priority to (%s %d)",
771  stack->depth, __FUNCTION__,
772  fr_table_str_by_value(mod_rcode_table, stack->result, "<invalid>"),
773  priority);
774  }
775 
776  /*
777  * We're at the top frame, return the result from the
778  * stack, and get rid of the top frame.
779  */
780  RDEBUG4("** [%i] %s - interpreter exiting, returning %s", stack->depth, __FUNCTION__,
781  fr_table_str_by_value(mod_rcode_table, frame->result, "<invalid>"));
782  stack->result = frame->result;
783  stack->depth--;
784  DUMP_STACK;
785 
786  return stack->result;
787 }

880 /** Execute an unlang section synchronously
881  *
882  * Create a temporary event loop and swap it out for the one in the request.
883  * Execute unlang operations until we receive a non-yield return code then return.
884  *
885  * @note The use cases for this are very limited. If you need to use it, chances
886  * are what you're doing could be done better using one of the thread
887  * event loops.
888  *
889  * @param[in] request The current request.
890  * @param[in] cs Section with compiled unlang associated with it.
891  * @param[in] action The default return code to use.
892  * @param[in] child_el Use a child event list
893  * @return One of the RLM_MODULE_* macros.
894  */
895 rlm_rcode_t unlang_interpret_synchronous(request_t *request, CONF_SECTION *cs, rlm_rcode_t action, bool child_el)
896 {
897  fr_event_list_t *el, *old_el;
898  fr_heap_t *backlog, *old_backlog;
899  rlm_rcode_t rcode;
900  char const *caller;
901  request_t *sub_request = NULL;
902  bool wait_for_event;
903  int iterations = 0;
904 
905 
906  old_el = request->el;
907  if (child_el) {
908  /*
909  * Don't talloc from the request
910  * as we'll almost certainly leave holes in the memory pool.
911  */
912  el = fr_event_list_alloc(NULL, NULL, NULL);
913  if (!el) {
914  RPERROR("Failed creating temporary event loop");
915  fr_assert(0); /* Cause debug builds to die */
916  return RLM_MODULE_FAIL;
917  }
918  }
919 
920  MEM(backlog = fr_heap_talloc_alloc(request, fr_pointer_cmp, request_t, runnable_id));
921  old_backlog = request->backlog;
922  caller = request->module;
923 
924  request->backlog = backlog;
925 
926  rcode = unlang_interpret_section(request, cs, action);
927  wait_for_event = (rcode == RLM_MODULE_YIELD);
928 
929  while (true) {
930  rlm_rcode_t sub_rcode;
931  int num_events;
932 
933  /*
934  * Wait for a timer / IO event. If there's a
935  * failure, all kinds of bad things happen. Oh
936  * well.
937  */
938  DEBUG3("Gathering events - %s", wait_for_event ? "will wait" : "Will not wait");
939  num_events = fr_event_corral(request->el, fr_time(), wait_for_event);
940  if (num_events < 0) {
941  RPERROR("Failed retrieving events");
942  return RLM_MODULE_FAIL;
943  }
944 
945  DEBUG3("%u event(s) pending%s",
946  num_events == -1 ? 0 : num_events, num_events == -1 ? " - event loop exiting" : "");
947 
948  /*
949  * We were NOT waiting, AND there are no more
950  * events to run, AND there are no more requests
951  * to run. We can exit the loop.
952  */
953  if (!wait_for_event && (num_events == 0) &&
954  (fr_heap_num_elements(backlog) == 0)) {
955  break;
956  }
957 
958  /*
959  * This function ends up pushing a
960  * runnable request into the backlog, OR
961  * setting new timers.
962  */
963  if (num_events > 0) {
964  DEBUG4("Servicing event(s)");
965  fr_event_service(request->el);
966  }
967 
968  /*
969  * If there are no runnable requests, then go
970  * back to check the timers again. Note that we
971  * only wait if there are timer events left to
972  * service.
973  *
974  * If there WAS a timer event, but servicing that
975  * timer event did not result in a runnable
976  * request, THEN we're guaranteed that there is
977  * still a timer event left.
978  */
979  sub_request = fr_heap_pop(backlog);
980  if (!sub_request) {
981  wait_for_event = (num_events > 0);
982  continue;
983  }
984 
985  /*
986  * Continue interpretation until there's nothing
987  * in the backlog. If this request YIELDs, then
988  * do another loop around.
989  */
990  RDEBUG4(">>> interpreter (iteration %i)", ++iterations);
991  sub_rcode = unlang_interpret(sub_request);
992  RDEBUG4("<<< interpreter (iteration %i) - %s", iterations,
993  fr_table_str_by_value(mod_rcode_table, sub_rcode, "<INVALID>"));
994  if (sub_rcode == RLM_MODULE_YIELD) {
995  wait_for_event = true;
996  continue;
997  }
998 
999  /*
1000  * This request is done. Clean up, and do a
1001  * non-blocking check for more events in the next
1002  * loop.
1003  */
1004  wait_for_event = false;
1005  if (sub_request == request) {
1006  rcode = sub_rcode;
1007  } else {
1008  RDEBUG4("Cleaning up subrequest");
1009  talloc_free(sub_request);
1010  }
1011  }
1012 
1013  if (child_el) talloc_free(el);
1014  talloc_free(backlog);
1015  request->el = old_el;
1016  request->backlog = old_backlog;
1017  request->module = caller;
1018 
1019  return rcode;
1020 }

1052 /** Send a signal (usually stop) to a request
1053  *
1054  * This is typically called via an "async" action, i.e. an action
1055  * outside of the normal processing of the request.
1056  *
1057  * If there is no #unlang_module_signal_t callback defined, the action is ignored.
1058  *
1059  * The signaling stops at the "limit" frame. This is so that keywords
1060  * such as "timeout" and "limit" can signal frames *lower* than theirs
1061  * to stop, but then continue with their own work.
1062  *
1063  * @param[in] request The current request.
1064  * @param[in] action to signal.
1065  * @param[in] limit the frame at which to stop signaling.
1066  */
1067 static void frame_signal(request_t *request, fr_state_signal_t action, int limit)
1068 {
1069  unlang_stack_frame_t *frame;
1070  unlang_stack_t *stack = request->stack;
1071  int i, depth = stack->depth;
1072 
1073  (void)talloc_get_type_abort(request, request_t); /* Check the request hasn't already been freed */
1074 
1075  fr_assert(stack->depth > 0);
1076 
1077  /*
1078  * Walk back up the stack, calling signal handlers
1079  * to cancel any pending operations and free/release
1080  * any resources.
1081  *
1082  * There may be multiple resumption points in the
1083  * stack, as modules can push xlats and function
1084  * calls.
1085  */
1086  for (i = depth; i > limit; i--) {
1087  stack->depth = i; /* We could also pass in the frame to the signal function */
1088  frame = &stack->frame[stack->depth];
1089 
1090  if (is_top_frame(frame)) continue; /* Skip top frames as they have no instruction */
1091 
1092  /*
1093  * Be gracious in errors.
1094  */
1095  if (!is_yielded(frame)) continue;
1096 
1097  if (!frame->signal) continue;
1098 
1099  frame->signal(request, action);
1100  }
1101  stack->depth = depth; /* Reset */
1102 }

1162 /** Mark a request as resumable.
1163  *
1164  * It's not called "unlang_interpret", because it doesn't actually
1165  * resume the request, it just schedules it for resumption.
1166  *
1167  * @note that this schedules the request for resumption. It does not immediately
1168  * start running the request.
1169  *
1170  * @param[in] request The current request.
1171  */
1172 void unlang_interpret_resumable(request_t *request)
1173 {
1174  unlang_stack_t *stack = request->stack;
1175  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
1176 
1177  /*
1178  * Child requests aren't runnable, only parent requests
1179  * are.
1180  */
1181  while (request->parent) {
1182  fr_assert(!unlang_request_is_scheduled(request));
1183  request->runnable_id = -2; /* hack for now, see unlang_parallel_process() */
1184  request = request->parent;
1185  }
1186 
1187  /*
1188  * The request hasn't yielded, OR it's already been
1189  * marked as runnable. Don't do anything.
1190  *
1191  * The IO code, or children have no idea where they're
1192  * being called from. They just ask to mark the parent
1193  * resumable when they're done. So we have to check here
1194  * if this request is resumable.
1195  *
1196  * If the parent called the child directly, then the
1197  * parent hasn't yielded, so it isn't resumable. When
1198  * the child is done, the parent will automatically
1199  * continue running. We therefore don't need to insert
1200  * the parent into the backlog.
1201  *
1202  * Multiple child request may also mark a parent request
1203  * runnable, before the parent request starts running.
1204  */
1205  if (!is_yielded(frame) || unlang_request_is_scheduled(request)) {
1206  RDEBUG3("Not marking resumable due to %s %s",
1207  is_yielded(frame) ?
1208  "it not being yielded " : "", unlang_request_is_scheduled(request) ? "it already being scheduled" : "");
1209  return;
1210  }
1211 
1212  if (!unlang_request_is_scheduled(request)) {
1213  RDEBUG3("marking as resumable");
1214  }
1215 
1216  fr_assert(request->backlog != NULL);
1217 
1218  fr_heap_insert(request->backlog, request);
1219 }
