55 /*
56  * Check if account has expired, and if user may login now.
57  */
58 static unlang_action_t CC_HINT(nonnull) mod_authorize(rlm_rcode_t *p_result, UNUSED module_ctx_t const *mctx, request_t *request)
59 {
60  fr_pair_t *vp, *check_item = NULL;
61 
62  check_item = fr_pair_find_by_da(&request->control_pairs, attr_expiration);
63  if (check_item != NULL) {
64  uint32_t left;
65 
66  /*
67  * Has this user's password expired?
68  *
69  * If so, remove ALL reply attributes,
70  * and add our own Reply-Message, saying
71  * why they're being rejected.
72  */
73  if (check_item->vp_date <= fr_time_to_unix_time(request->packet->timestamp)) {
74  REDEBUG("Account expired at '%pV'", &check_item->data);
75 
76  RETURN_MODULE_DISALLOW;
77  }
78  RDEBUG2("Account will expire at '%pV'", &check_item->data);
79 
80  left = fr_time_to_sec(check_item->vp_date - request->packet->timestamp);
81 
82  /*
83  * Else the account hasn't expired, but it may do so
84  * in the future. Set Session-Timeout.
85  */
86  switch (pair_update_reply(&vp, attr_session_timeout)) {
87  case 1:
88  /* just update... */
89  if (vp->vp_uint32 > (uint32_t)left) {
90  vp->vp_uint32 = (uint32_t)left;
91  RDEBUG2("&reply.Session-Timeout := %pV", &vp->data);
92  }
93  break;
94 
95  case 0: /* no pre-existing */
96  vp->vp_uint32 = (uint32_t)left;
97  RDEBUG2("&reply.Session-Timeout := %pV", &vp->data);
98  break;
99 
100  default: /* malloc failure */
101  MEM(NULL);
102  }
103  } else {
104  RETURN_MODULE_NOOP;
105  }
106 
107  RETURN_MODULE_OK;
108 }

110 /*
111  * Compare the expiration date.
112  */
113 static int expirecmp(UNUSED void *instance, request_t *req, UNUSED fr_pair_t *request, fr_pair_t *check,
114  UNUSED fr_pair_t *check_list)
115 {
116  time_t now = 0;
117 
118  now = (req) ? fr_time_to_sec(req->packet->timestamp) : time(NULL);
119 
120  if (now <= fr_time_to_sec(check->vp_date)) return 0;
121 
122  return 1;
123 }
