44 /** Use Winbind to normalise a username
45  *
46  * @param[in] ctx The talloc context where the result is parented from
47  * @param[in] wb_ctx The winbind context
48  * @param[in] dom_name The domain of the user
49  * @param[in] name The username (without the domain) to be normalised
50  * @return
51  * - The username with the casing according to the Winbind remote server.
52  * - NULL if the username could not be found.
53  */
54 static char *wbclient_normalise_username(TALLOC_CTX *ctx, struct wbcContext *wb_ctx,
55  char const *dom_name, char const *name)
56 {
57  struct wbcDomainSid sid;
58  enum wbcSidType name_type;
59  wbcErr err;
60  char *res_domain = NULL;
61  char *res_name = NULL;
62  char *res = NULL;
63 
64  /* Step 1: Convert a name to a sid */
65  err = wbcCtxLookupName(wb_ctx, dom_name, name, &sid, &name_type);
66  if (!WBC_ERROR_IS_OK(err)) return NULL;
67 
68  /* Step 2: Convert the sid back to a name */
69  err = wbcCtxLookupSid(wb_ctx, &sid, &res_domain, &res_name, &name_type);
70  if (!WBC_ERROR_IS_OK(err)) return NULL;
71 
72  MEM(res = talloc_strdup(ctx, res_name));
73 
74  wbcFreeMemory(res_domain);
75  wbcFreeMemory(res_name);
76 
77  return res;
78 }

80 /** Check NTLM authentication direct to winbind via Samba's libwbclient library
81  *
82  * @return
83  * - 0 success.
84  * - -1 auth failure.
85  * - -648 password expired.
86  */
87 int do_auth_wbclient(rlm_mschap_t const *inst, request_t *request,
88  uint8_t const *challenge, uint8_t const *response,
89  uint8_t nthashhash[NT_DIGEST_LENGTH])
90 {
91  int ret = -1;
92  struct wbcContext *wb_ctx = NULL;
93  struct wbcAuthUserParams *authparams;
94  wbcErr err;
95  ssize_t slen;
96  struct wbcAuthUserInfo *info = NULL;
97  struct wbcAuthErrorInfo *error = NULL;
98  uint8_t resp[NT_LENGTH];
99 
100  /*
101  * wb_username must be set for this function to be called
102  */
103  fr_assert(inst->wb_username);
104 
105  MEM(authparams = talloc_zero_pooled_object(NULL, struct wbcAuthUserParams, 2, 1024));
106  /*
107  * Domain first so we don't leave holes in the pool
108  */
109  if (inst->wb_domain) {
110  slen = tmpl_aexpand(authparams, &authparams->domain_name, request, inst->wb_domain, NULL, NULL);
111  if (slen < 0) {
112  REDEBUG2("Unable to expand winbind_domain");
113  goto finish;
114  }
115  } else {
116  RWDEBUG2("No domain specified; authentication may fail because of this");
117  }
118 
119  /*
120  * Get the username and domain from the configuration
121  */
122  slen = tmpl_aexpand(authparams, &authparams->account_name, request, inst->wb_username, NULL, NULL);
123  if (slen < 0) {
124  REDEBUG2("Unable to expand winbind_username");
125  goto finish;
126  }
127 
128  /*
129  * Build the wbcAuthUserParams structure with what we know
130  */
131  authparams->level = WBC_AUTH_USER_LEVEL_RESPONSE;
132  authparams->password.response.nt_length = NT_LENGTH;
133 
134  memcpy(resp, response, NT_LENGTH);
135  authparams->password.response.nt_data = resp;
136 
137  memcpy(authparams->password.response.challenge, challenge, sizeof(authparams->password.response.challenge));
138 
139  authparams->parameter_control |= WBC_MSV1_0_ALLOW_MSVCHAPV2 |
140  WBC_MSV1_0_ALLOW_WORKSTATION_TRUST_ACCOUNT |
141  WBC_MSV1_0_ALLOW_SERVER_TRUST_ACCOUNT;
142 
143  /*
144  * Send auth request across to winbind
145  */
146  wb_ctx = fr_pool_connection_get(inst->wb_pool, request);
147  if (wb_ctx == NULL) {
148  RERROR("Unable to get winbind connection from pool");
149  goto finish;
150  }
151 
152  RDEBUG2("Sending authentication request user \"%pV\" domain \"%pV\"",
153  fr_box_strvalue_buffer(authparams->account_name),
154  fr_box_strvalue_buffer(authparams->domain_name));
155 
156  err = wbcCtxAuthenticateUserEx(wb_ctx, authparams, &info, &error);
157  if (err == WBC_ERR_AUTH_ERROR && inst->wb_retry_with_normalised_username) {
158  fr_pair_t *vp_response;
159  fr_pair_t *vp_challenge;
160  fr_pair_t *vp_chap_user_name;
161  char *normalised_username = NULL;
162 
163  normalised_username = wbclient_normalise_username(authparams, wb_ctx, authparams->domain_name,
164  authparams->account_name);
165  if (!normalised_username) goto release;
166 
167  RDEBUG2("Starting retry, normalised username \"%pV\" -> \"%pV\"",
168  fr_box_strvalue_buffer(authparams->account_name),
169  fr_box_strvalue_buffer(normalised_username));
170 
171  if (talloc_memcmp_bstr(authparams->account_name, normalised_username) == 0) goto release;
172 
173  talloc_const_free(authparams->account_name);
174  authparams->account_name = normalised_username;
175 
176  /* Set MS-CHAP-USER-NAME */
177  MEM(pair_update_request(&vp_chap_user_name, attr_ms_chap_user_name) >= 0);
178  fr_pair_value_bstrdup_buffer(vp_chap_user_name, normalised_username, true);
179 
180  RDEBUG2("Retrying authentication request user \"%pV\" domain \"%pV\"",
181  fr_box_strvalue_buffer(authparams->account_name),
182  fr_box_strvalue_buffer(normalised_username));
183 
184  /* Recalculate hash */
185  vp_challenge = fr_pair_find_by_da(&request->request_pairs, attr_ms_chap_challenge);
186  if (!vp_challenge) {
187  RERROR("Unable to get MS-CHAP-Challenge");
188  goto release;
189  }
190 
191  vp_response = fr_pair_find_by_da(&request->request_pairs, attr_ms_chap2_response);
192  if (!vp_response) {
193  RERROR("Unable to get MS-CHAP2-Response");
194  goto release;
195  }
196 
197  mschap_challenge_hash(authparams->password.response.challenge,
198  vp_response->vp_octets + 2,
199  vp_challenge->vp_octets,
200  vp_chap_user_name->vp_strvalue, vp_chap_user_name->vp_length);
201 
202  err = wbcCtxAuthenticateUserEx(wb_ctx, authparams, &info, &error);
203 release:
204  talloc_free(normalised_username);
205  }
206 
207  fr_pool_connection_release(inst->wb_pool, request, wb_ctx);
208 
209  /*
210  * Try and give some useful feedback on what happened. There are only
211  * a few errors that can actually be returned from wbcCtxAuthenticateUserEx.
212  */
213  switch (err) {
214  case WBC_ERR_SUCCESS:
215  ret = 0;
216  RDEBUG2("Authenticated successfully");
217  /* Grab the nthashhash from the result */
218  memcpy(nthashhash, info->user_session_key, NT_DIGEST_LENGTH);
219  break;
220 
221  case WBC_ERR_WINBIND_NOT_AVAILABLE:
222  RERROR("Unable to contact winbind!");
223  RDEBUG2("Check that winbind is running and that FreeRADIUS has");
224  RDEBUG2("permission to connect to the winbind privileged socket.");
225  break;
226 
227  case WBC_ERR_DOMAIN_NOT_FOUND:
228  REDEBUG2("Domain not found");
229  break;
230 
231  case WBC_ERR_AUTH_ERROR:
232  if (!error) {
233  REDEBUG2("Authentication failed");
234  break;
235  }
236 
237  /*
238  * The password needs to be changed, so set ret appropriately.
239  */
240  if (error->nt_status == NT_STATUS_PASSWORD_EXPIRED ||
241  error->nt_status == NT_STATUS_PASSWORD_MUST_CHANGE) {
242  ret = -648;
243  }
244 
245  /*
246  * Return the NT_STATUS human readable error string, if there is one.
247  */
248  if (error->display_string) {
249  REDEBUG2("%s [0x%X]", error->display_string, error->nt_status);
250  } else {
251  REDEBUG2("Authentication failed [0x%X]", error->nt_status);
252  }
253  break;
254 
255  default:
256  /*
257  * Only errors left are
258  * WBC_ERR_INVALID_PARAM
259  * WBC_ERR_NO_MEMORY
260  * neither of which are particularly likely.
261  */
262  if (error && error->display_string) {
263  REDEBUG2("libwbclient error: wbcErr %d (%s)", err, error->display_string);
264  } else {
265  REDEBUG2("libwbclient error: wbcErr %d", err);
266  }
267  break;
268  }
269 
270 finish:
271  talloc_free(authparams);
272  if (info) wbcFreeMemory(info);
273  if (error) wbcFreeMemory(error);
274 
275  return ret;
276 }
