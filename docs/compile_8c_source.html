165 static bool pass2_fixup_tmpl(TALLOC_CTX *ctx, CONF_ITEM const *ci, tmpl_t **vpt_p)
166 {
167  tmpl_t *vpt = *vpt_p;
168 
169  TMPL_VERIFY(vpt);
170 
171  /*
172  * Convert virtual &Attr-Foo to "%{Attr-Foo}"
173  */
174  if (tmpl_is_attr(vpt) && tmpl_da(vpt)->flags.virtual) {
175  if (tmpl_attr_to_xlat(ctx, vpt_p) < 0) {
176  return false;
177  }
178 
179  /*
180  * The VPT has been rewritten, so use the new one.
181  */
182  vpt = *vpt_p;
183  } /* it's now xlat, so we need to resolve it. */
184 
185  /*
186  * Fixup any other tmpl types
187  */
188  if (tmpl_resolve(vpt) < 0) {
189  cf_log_perr(ci, NULL);
190  return false;
191  }
192 
193  return true;
194 }

196 static bool pass2_fixup_map(fr_cond_t *c)
197 {
198  tmpl_t *vpt;
199  map_t *map;
200 
201  map = c->data.map; /* shorter */
202 
203  /*
204  * Auth-Type := foo
205  *
206  * Where "foo" is dynamically defined.
207  */
208  if (c->pass2_fixup == PASS2_FIXUP_TYPE) {
209  if (!fr_dict_enum_by_name(tmpl_da(map->lhs), map->rhs->name, -1)) {
210  cf_log_err(map->ci, "Invalid reference to non-existent %s %s { ... }",
211  tmpl_da(map->lhs)->name,
212  map->rhs->name);
213  return false;
214  }
215 
216  /*
217  * These guys can't have a paircmp fixup applied.
218  */
219  c->pass2_fixup = PASS2_FIXUP_NONE;
220  return true;
221  }
222 
223  if (c->pass2_fixup == PASS2_FIXUP_ATTR) {
224  fr_dict_attr_t const *cast = c->cast;
225 
226  /*
227  * Resolve the attribute references first
228  */
229  if (tmpl_is_attr_unresolved(map->lhs)) {
230  if (!pass2_fixup_tmpl(map, map->ci, &map->lhs)) return false;
231  if (!cast) cast = tmpl_da(map->lhs);
232  }
233 
234  if (tmpl_is_attr_unresolved(map->rhs)) {
235  if (!pass2_fixup_tmpl(map, map->ci, &map->rhs)) return false;
236  if (!cast) cast = tmpl_da(map->rhs);
237  }
238 
239  /*
240  * Then fixup the other side if it was unresolved
241  */
242  if (tmpl_is_unresolved(map->lhs)) {
243  switch (cast->type) {
244  case FR_TYPE_IPV4_ADDR:
245  if (strchr(c->data.map->lhs->name, '/') != NULL) {
246  c->cast = cast = fr_dict_attr_child_by_num(fr_dict_root(fr_dict_internal()),
247  FR_CAST_BASE + FR_TYPE_IPV4_PREFIX);
248  }
249  break;
250 
251  case FR_TYPE_IPV6_ADDR:
252  if (strchr(c->data.map->lhs->name, '/') != NULL) {
253  c->cast = cast = fr_dict_attr_child_by_num(fr_dict_root(fr_dict_internal()),
254  FR_CAST_BASE + FR_TYPE_IPV6_PREFIX);
255  }
256  break;
257 
258  default:
259  break;
260  }
261 
262  if (tmpl_cast_in_place(c->data.map->lhs, cast->type, cast) < 0) {
263  cf_log_err(map->ci, "Failed to parse data type %s from string: %pV",
264  fr_table_str_by_value(fr_value_box_type_table, cast->type, "<UNKNOWN>"),
265  fr_box_strvalue_len(map->lhs->name, map->lhs->len));
266 
267  return false;
268  }
269  } else if (tmpl_is_unresolved(map->rhs)) {
270  switch (cast->type) {
271  case FR_TYPE_IPV4_ADDR:
272  if (strchr(c->data.map->rhs->name, '/') != NULL) {
273  c->cast = cast = fr_dict_attr_child_by_num(fr_dict_root(fr_dict_internal()),
274  FR_CAST_BASE + FR_TYPE_IPV4_PREFIX);
275  }
276  break;
277 
278  case FR_TYPE_IPV6_ADDR:
279  if (strchr(c->data.map->rhs->name, '/') != NULL) {
280  c->cast = cast = fr_dict_attr_child_by_num(fr_dict_root(fr_dict_internal()),
281  FR_CAST_BASE + FR_TYPE_IPV6_PREFIX);
282  }
283  break;
284 
285  default:
286  break;
287  }
288 
289  if (tmpl_cast_in_place(c->data.map->rhs, cast->type, cast) < 0) {
290  cf_log_err(map->ci, "Failed to parse data type %s from string: %pV",
291  fr_table_str_by_value(fr_value_box_type_table, cast->type, "<UNKNOWN>"),
292  fr_box_strvalue_len(map->rhs->name, map->rhs->len));
293  return false;
294  }
295  }
296 
297  c->pass2_fixup = PASS2_FIXUP_NONE;
298  }
299 
300  /*
301  * Just in case someone adds a new fixup later.
302  */
303  fr_assert((c->pass2_fixup == PASS2_FIXUP_NONE) ||
304  (c->pass2_fixup == PASS2_PAIRCOMPARE));
305 
306  /*
307  * Precompile xlat's
308  */
309  if (tmpl_is_xlat_unresolved(map->lhs)) {
310  /*
311  * Compile the LHS to an attribute reference only
312  * if the RHS is a literal.
313  *
314  * @todo v3.1: allow anything anywhere.
315  */
316  if (!tmpl_is_unresolved(map->rhs)) {
317  if (!pass2_fixup_tmpl(map, map->ci, &map->lhs)) {
318  return false;
319  }
320  } else {
321  if (!pass2_fixup_tmpl(map, map->ci, &map->lhs)) {
322  return false;
323  }
324 
325  /*
326  * Attribute compared to a literal gets
327  * the literal cast to the data type of
328  * the attribute.
329  *
330  * The code in parser.c did this for
331  *
332  * &Attr == data
333  *
334  * But now we've just converted "%{Attr}"
335  * to &Attr, so we've got to do it again.
336  */
337  if (tmpl_is_attr(map->lhs)) {
338  if ((map->rhs->len > 0) ||
339  (map->op != T_OP_CMP_EQ) ||
340  (tmpl_da(map->lhs)->type == FR_TYPE_STRING) ||
341  (tmpl_da(map->lhs)->type == FR_TYPE_OCTETS)) {
342 
343  if (tmpl_cast_in_place(map->rhs, tmpl_da(map->lhs)->type, tmpl_da(map->lhs)) < 0) {
344  cf_log_err(map->ci, "Failed to parse data type %s from string: %pV",
345  fr_table_str_by_value(fr_value_box_type_table, tmpl_da(map->lhs)->type, "<UNKNOWN>"),
346  fr_box_strvalue_len(map->rhs->name, map->rhs->len));
347  return false;
348  } /* else the cast was successful */
349 
350  } else { /* RHS is empty, it's just a check for empty / non-empty string */
351  vpt = talloc_steal(c, map->lhs);
352  map->lhs = NULL;
353  talloc_free(c->data.map);
354 
355  /*
356  * "%{Foo}" == '' ---> !Foo
357  * "%{Foo}" != '' ---> Foo
358  */
359  c->type = COND_TYPE_EXISTS;
360  c->data.vpt = vpt;
361  c->negate = !c->negate;
362 
363  WARN("%s[%d]: Please change (\"%%{%s}\" %s '') to %c&%s",
364  cf_filename(cf_item_to_section(c->ci)),
365  cf_lineno(cf_item_to_section(c->ci)),
366  vpt->name, c->negate ? "==" : "!=",
367  c->negate ? '!' : ' ', vpt->name);
368 
369  /*
370  * No more RHS, so we can't do more optimizations
371  */
372  return true;
373  }
374  }
375  }
376  }
377 
378  if (tmpl_is_xlat_unresolved(map->rhs)) {
379  /*
380  * Convert the RHS to an attribute reference only
381  * if the LHS is an attribute reference, AND is
382  * of the same type as the RHS.
383  *
384  * We can fix this when the code in evaluate.c
385  * can handle strings on the LHS, and attributes
386  * on the RHS. For now, the code in parser.c
387  * forbids this.
388  */
389  if (tmpl_is_attr(map->lhs)) {
390  if (!pass2_fixup_tmpl(map, map->ci, &map->rhs)) return false;
391  } else {
392  if (!pass2_fixup_tmpl(map, map->ci, &map->rhs)) return false;
393  }
394  }
395 
396  if (tmpl_is_exec_unresolved(map->lhs)) {
397  if (!pass2_fixup_tmpl(map, map->ci, &map->lhs)) {
398  return false;
399  }
400  }
401 
402  if (tmpl_is_exec_unresolved(map->rhs)) {
403  if (!pass2_fixup_tmpl(map, map->ci, &map->rhs)) {
404  return false;
405  }
406  }
407 
408  /*
409  * Convert bare refs to %{Foreach-Variable-N}
410  */
411  if (tmpl_is_unresolved(map->lhs) &&
412  (strncmp(map->lhs->name, "Foreach-Variable-", 17) == 0)) {
413  char *fmt;
414  ssize_t slen;
415 
416  fmt = talloc_typed_asprintf(map->lhs, "%%{%s}", map->lhs->name);
417  slen = tmpl_afrom_substr(map, &vpt, &FR_SBUFF_IN(fmt, talloc_array_length(fmt) - 1),
418  T_DOUBLE_QUOTED_STRING,
419  NULL,
420  &(tmpl_rules_t){
421  .allow_unknown = true
422  });
423  if (slen < 0) {
424  char *spaces, *text;
425 
426  fr_canonicalize_error(map->ci, &spaces, &text, slen, fr_strerror());
427 
428  cf_log_err(map->ci, "Failed converting %s to xlat", map->lhs->name);
429  cf_log_err(map->ci, "%s", fmt);
430  cf_log_err(map->ci, "%s^ %s", spaces, text);
431 
432  talloc_free(spaces);
433  talloc_free(text);
434  talloc_free(fmt);
435 
436  return false;
437  }
438  talloc_free(map->lhs);
439  map->lhs = vpt;
440  }
441 
442 #ifdef HAVE_REGEX
443  if (tmpl_is_regex_xlat_unresolved(map->rhs)) {
444  if (!pass2_fixup_tmpl(map, map->ci, &map->rhs)) {
445  return false;
446  }
447  }
448  fr_assert(!tmpl_is_regex_xlat_unresolved(map->lhs));
449 #endif
450 
451  /*
452  * Convert &Packet-Type to "%{Packet-Type}", because
453  * these attributes don't really exist. The code to
454  * find an attribute reference doesn't work, but the
455  * xlat code does.
456  */
457  vpt = c->data.map->lhs;
458  if (tmpl_is_attr(vpt) && tmpl_da(vpt)->flags.virtual) {
459  if (tmpl_attr_to_xlat(c, &vpt) < 0) return false;
460  }
461 
462  /*
463  * @todo v3.1: do the same thing for the RHS...
464  */
465 
466  /*
467  * Only attributes can have a paircmp registered, and
468  * they can only be with the current request_t, and only
469  * with the request pairs.
470  */
471  if (!tmpl_is_attr(map->lhs) ||
472  (tmpl_request(map->lhs) != REQUEST_CURRENT) ||
473  (tmpl_list(map->lhs) != PAIR_LIST_REQUEST)) {
474  return true;
475  }
476 
477  if (!paircmp_find(tmpl_da(map->lhs))) return true;
478 
479  if (tmpl_is_regex_xlat_unresolved(map->rhs)) {
480  cf_log_err(map->ci, "Cannot compare virtual attribute %s via a regex", map->lhs->name);
481  return false;
482  }
483 
484  if (c->cast) {
485  cf_log_err(map->ci, "Cannot cast virtual attribute %s to %s", map->lhs->name,
486  fr_table_str_by_value(fr_value_box_type_table, c->cast->type, "<INVALID>"));
487  return false;
488  }
489 
490  if (map->op != T_OP_CMP_EQ) {
491  cf_log_err(map->ci, "Must use '==' for comparisons with virtual attribute %s", map->lhs->name);
492  return false;
493  }
494 
495  /*
496  * Mark it as requiring a paircmp() call, instead of
497  * fr_pair_cmp().
498  */
499  c->pass2_fixup = PASS2_PAIRCOMPARE;
500 
501  return true;
502 }

504 static bool pass2_cond_callback(fr_cond_t *c, UNUSED void *uctx)
505 {

507  /*
508  * These don't get optimized.
509  */
510  case COND_TYPE_TRUE:
511  case COND_TYPE_FALSE:
512  case COND_TYPE_RCODE:
513  return true;

516  * Call children.
517  */
518  case COND_TYPE_CHILD:
519  return pass2_cond_callback(c->data.child, uctx);
520 
521  /*
522  * Fix up the template.
523  */

525  fr_assert(!tmpl_is_regex_xlat_unresolved(c->data.vpt));

543 static bool pass2_fixup_update_map(map_t *map, tmpl_rules_t const *rules, fr_dict_attr_t const *parent)
544 {
545  if (tmpl_is_xlat_unresolved(map->lhs)) {
546  fr_assert(tmpl_xlat(map->lhs) == NULL);
547 
548  /*
549  * FIXME: compile to attribute && handle
550  * the conversion in map_to_vp().
551  */
552  if (!pass2_fixup_tmpl(map, map->ci, &map->lhs)) {
553  return false;
554  }
555  }
556 
557  if (tmpl_is_exec(map->lhs)) {
558  if (!pass2_fixup_tmpl(map, map->ci, &map->lhs)) {
559  return false;
560  }
561  }
562 
563  /*
564  * Deal with undefined attributes now.
565  */
566  if (tmpl_is_attr_unresolved(map->lhs)) {
567  if (!pass2_fixup_tmpl(map, map->ci, &map->lhs)) return false;
568  }
569 
570  /*
571  * Enforce parent-child relationships in nested maps.
572  */
573  if (parent) {
574  if (map->op != T_OP_EQ) {
575  cf_log_err(map->ci, "Invalid operator \"%s\" in nested map section. "
576  "Only '=' is allowed",
577  fr_table_str_by_value(fr_tokens_table, map->op, "<INVALID>"));
578  return false;
579  }
580  }
581 
582  if (map->rhs) {
583  if (tmpl_is_xlat_unresolved(map->rhs)) {
584  fr_assert(tmpl_xlat(map->rhs) == NULL);
585 
586  /*
587  * FIXME: compile to attribute && handle
588  * the conversion in map_to_vp().
589  */
590  if (!pass2_fixup_tmpl(map, map->ci, &map->rhs)) {
591  return false;
592  }
593  }
594 
595  fr_assert(!tmpl_is_regex_xlat_unresolved(map->rhs));
596 
597  if (tmpl_is_attr_unresolved(map->rhs)) {
598  if (!pass2_fixup_tmpl(map, map->ci, &map->rhs)) return false;
599  }
600 
601  if (tmpl_is_exec(map->rhs)) {
602  if (!pass2_fixup_tmpl(map, map->ci, &map->rhs)) {
603  return false;
604  }
605  }
606  }
607 
608  /*
609  * Sanity check sublists.
610  */
611  if (map->child) {
612  if (!tmpl_is_attr(map->lhs)) {
613  cf_log_err(map->ci, "Sublists can only be assigned to a known attribute");
614  return false;
615  }
616 
617  if ((tmpl_da(map->lhs)->type != FR_TYPE_GROUP) &&
618  (tmpl_da(map->lhs)->type != FR_TYPE_TLV)) {
619  cf_log_err(map->ci, "Sublists can only be assigned to attributes of type 'group' or 'tlv'");
620  return false;
621  }
622 
623  return pass2_fixup_update_map(map->child, rules, tmpl_da(map->lhs));
624  }
625 
626  return true;
627 }

1662  bool was_if = false;
1663  char const *skip_else = NULL;
1664 
1665  c = unlang_group_to_generic(g);
1666 
1667  /*
1668  * Loop over the children of this group.
1669  */
1670  while ((ci = cf_item_next(g->cs, ci))) {
1671  if (cf_item_is_data(ci)) continue;
1672 
1673  /*
1674  * Sections are keywords, or references to
1675  * modules with updated return codes.
1676  */
1677  if (cf_item_is_section(ci)) {
1678  char const *name = NULL;
1679  CONF_SECTION *subcs = cf_item_to_section(ci);
1680 
1681  /*
1682  * Skip precompiled blocks. This is
1683  * mainly for policies.
1684  */
1685  if (cf_data_find(subcs, unlang_group_t, NULL)) continue;
1686 
1687  /*
1688  * "actions" apply to the current group.
1689  * It's not a subgroup.
1690  */
1691  name = cf_section_name1(subcs);
1692  if (strcmp(name, "actions") == 0) {
1693  if (!compile_action_subsection(c, g->cs, subcs)) {
1694  talloc_free(c);
1695  return NULL;
1696  }
1697 
1698  continue;
1699  }
1700 
1701  /*
1702  * Special checks for "else" and "elsif".
1703  */
1704  if ((strcmp(name, "else") == 0) || (strcmp(name, "elsif") == 0)) {
1705  /*
1706  * We ran into one without a preceding "if" or "elsif".
1707  * That's not allowed.
1708  */
1709  if (!was_if) {
1710  cf_log_err(ci, "Invalid location for '%s'. There is no preceding "
1711  "'if' or 'elsif' statement", name);
1712  talloc_free(c);
1713  return NULL;
1714  }
1715 
1716  /*
1717  * There was a previous "if" or "elsif" which was always taken.
1718  * So we skip this "elsif" or "else".
1719  */
1720  if (skip_else) {
1721  cf_log_debug_prefix(ci, "Skipping contents of '%s' due to previous "
1722  "'%s' being always being taken.",
1723  name, skip_else);
1724  continue;
1725  }
1726  }
1727 
1728  /*
1729  * Otherwise it's a real keyword.
1730  */
1731  single = compile_item(c, unlang_ctx, ci);
1732  if (!single) {
1733  cf_log_err(ci, "Failed to parse \"%s\" subsection", cf_section_name1(subcs));
1734  talloc_free(c);
1735  return NULL;
1736  }
1737 
1738  goto add_child;
1739  }
1740 
1741  if (cf_item_is_pair(ci)) {
1742  char const *attr;
1743  CONF_PAIR *cp = cf_item_to_pair(ci);
1744 
1745  attr = cf_pair_attr(cp);
1746 
1747  /*
1748  * Bare "foo = bar" is disallowed.
1749  */
1750  if (cf_pair_value(cp) != NULL) {
1751  cf_log_err(cp, "Unknown keyword '%s', or invalid location", attr);
1752  talloc_free(c);
1753  return NULL;
1754  }
1755 
1756  /*
1757  * Compile the item as a module
1758  * reference, or as "break / return /
1759  * etc."
1760  */
1761  single = compile_item(c, unlang_ctx, ci);
1762  if (!single) {
1763  cf_log_err(ci, "Invalid keyword \"%s\".", attr);
1764  talloc_free(c);
1765  return NULL;
1766  }
1767 
1768  goto add_child;
1769  } /* was CONF_PAIR */
1770 
1771  cf_log_err(ci, "Internal sanity check failed in unlang compile.");
1772  talloc_free(c);
1773  return NULL;
1774 
1775  add_child:
1776  if (single == UNLANG_IGNORE) continue;
1777 
1778  /*
1779  * Do optimizations for "if" and "elsif"
1780  * conditions.
1781  */
1782  switch (single->type) {
1783  case UNLANG_TYPE_ELSIF:
1784  case UNLANG_TYPE_IF:
1785  was_if = true;
1786  {
1787  unlang_group_t *f;
1788  unlang_cond_t *gext;
1789 
1790  f = unlang_generic_to_group(single);
1791  gext = unlang_group_to_cond(f);
1792 
1793  switch (gext->cond->type) {
1794  case COND_TYPE_TRUE:
1795  skip_else = single->debug_name;
1796  break;
1797 
1798  case COND_TYPE_FALSE:
1799  /*
1800  * The condition never
1801  * matches, so we can
1802  * avoid putting it into
1803  * the unlang tree.
1804  */
1805  talloc_free(single);
1806  continue;
1807 
1808  default:
1809  break;
1810  }
1811  }
1812  break;
1813 
1814  default:
1815  was_if = false;
1816  skip_else = NULL;
1817  break;
1818  }
1819 
1820  /*
1821  * unlang_group_t is grown by adding a unlang_t to the end
1822  */
1823  fr_assert(g == talloc_parent(single));
1824  fr_assert(single->parent == unlang_group_to_generic(g));
1825  fr_assert(!single->next);
1826 
1827  *g->tail = single;
1828  g->tail = &single->next;
1829  g->num_children++;
1830 
1831  /*
1832  * If it's not possible to execute statement
1833  * after the current one, then just stop
1834  * processing the children.
1835  */
1836  if (g->self.closed) {
1837  cf_log_warn(ci, "Skipping remaining instructions due to '%s'",
1838  single->name);
1839  break;
1840  }
1841  }
1842 
1843  /*
1844  * Set the default actions, if they haven't already been
1845  * set by an "actions" section above.
1846  */
1847  compile_action_defaults(c, unlang_ctx);
1848 
1849  return c;
1850 }
