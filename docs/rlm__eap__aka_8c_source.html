118 /** Compile virtual server sections
119  *
120  * Called twice, once when a server with an eap-aka namespace is found, and once
121  * when an EAP-AKA module is instantiated.
122  *
123  * The first time is with actions == NULL and is to compile the sections and
124  * perform validation.
125  * The second time is to write out pointers to the compiled sections which the
126  * EAP-AKA module will use to execute unlang code.
127  *
128  */
129 static int mod_section_compile(eap_aka_sim_actions_t *actions, CONF_SECTION *server_cs)
130 {
131  int found;
132  tmpl_rules_t parse_rules;
133 
134  if (!fr_cond_assert(server_cs)) return -1;
135 
136  memset(&parse_rules, 0, sizeof(parse_rules));
137  parse_rules.dict_def = dict_eap_aka_sim;
138 
139  found = virtual_server_compile_sections(server_cs, compile_list, &parse_rules, actions);
140  if (found < 0) return -1;
141 
142  /*
143  * Warn if we couldn't find any actions.
144  */
145  if (!found) {
146  cf_log_warn(server_cs, "No \"eap-aka\" actions found in virtual server \"%s\"",
147  cf_section_name2(server_cs));
148  }
149 
150  return 0;
151 }

154 static int mod_instantiate(void *instance, CONF_SECTION *conf)
155 {
156  eap_aka_sim_common_conf_t *inst = talloc_get_type_abort(instance, eap_aka_sim_common_conf_t);
157 
158  if (mod_section_compile(&inst->actions, inst->virtual_server) < 0) return -1;
159 
160  /*
161  * If the user didn't specify a bidding value
162  * infer whether we need to send the bidding
163  * attribute, by whether the EAP module has
164  * has the AKA-Prime module enabled.
165  */
166  if (!inst->send_at_bidding_prefer_prime_is_set) {
167  CONF_SECTION *parent = cf_item_to_section(cf_parent(conf));
168  CONF_PAIR *cp = NULL;
169 
170  while ((cp = cf_pair_find_next(parent, cp, "type"))) {
171  if (strcmp(cf_pair_value(cp), "aka-prime") == 0) {
172  cf_log_debug(conf, "Setting 'prefer_aka_prime = yes', as EAP-AKA-Prime is enabled");
173  inst->send_at_bidding_prefer_prime = true;
174  break;
175  }
176  }
177  }
178 
179  return 0;
180 }

190 static int mod_load(void)
191 {
192  if (fr_aka_sim_init() < 0) return -1;
193 
194  if (virtual_namespace_register("eap-aka", dict_eap_aka_sim, mod_namespace_load) < 0) return -1;

196  fr_aka_sim_xlat_register();

198  return 0;
