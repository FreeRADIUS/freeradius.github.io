102 static int groupcmp(UNUSED void *instance, request_t *request, UNUSED fr_pair_t *req_vp,
103  fr_pair_t *check, UNUSED fr_pair_t *check_list)
104 {
105  struct passwd *pwd;
106  struct group *grp;
107  char **member;
108  int retval = -1;
109  fr_pair_t *username;
110 
111  /*
112  * No user name, can't compare.
113  */
114  username = fr_pair_find_by_da(&request->request_pairs, attr_user_name);
115  if (!username) return -1;
116 
117  if (rad_getpwnam(request, &pwd, username->vp_strvalue) < 0) {
118  RPEDEBUG("Failed resolving user name");
119  return -1;
120  }
121 
122  if (rad_getgrnam(request, &grp, check->vp_strvalue) < 0) {
123  RPEDEBUG("Failed resolving group name");
124  talloc_free(pwd);
125  return -1;
126  }
127 
128  /*
129  * The users default group isn't the one we're looking for,
130  * look through the list of group members.
131  */
132  if (pwd->pw_gid == grp->gr_gid) {
133  retval = 0;
134 
135  } else {
136  for (member = grp->gr_mem; *member && retval; member++) {
137  if (strcmp(*member, pwd->pw_name) == 0) {
138  retval = 0;
139  break;
140  }
141  }
142  }
143 
144  /* lifo */
145  talloc_free(grp);
146  talloc_free(pwd);
147 
148  return retval;
149 }

173  return -1;
174  }
175  talloc_free(unix_group);
176  }

183  * Pull the users password from where-ever, and add it to

186 static unlang_action_t CC_HINT(nonnull) mod_authorize(rlm_rcode_t *p_result, UNUSED module_ctx_t const *mctx, request_t *request)
187 {
188  char const *name;
189  char const *encrypted_pass;
190 #ifdef HAVE_GETSPNAM
191  struct spwd *spwd = NULL;
192 #endif
193  struct passwd *pwd;
194 #ifdef HAVE_GETUSERSHELL
195  char *shell;
196 #endif
197  fr_pair_t *vp;
198  fr_pair_t *username;
199 
200  /*
201  * We can only authenticate user requests which HAVE
202  * a User-Name attribute.
203  */
204  username = fr_pair_find_by_da(&request->request_pairs, attr_user_name);
205  if (!username) RETURN_MODULE_NOOP;
206 
207  name = username->vp_strvalue;
208  encrypted_pass = NULL;
209 
210  if ((pwd = getpwnam(name)) == NULL) {
211  RETURN_MODULE_NOTFOUND;
212  }
213  encrypted_pass = pwd->pw_passwd;
214 
215 #ifdef HAVE_GETSPNAM
216  /*
217  * See if there is a shadow password.
218  *
219  * Only query the _system_ shadow file if the encrypted
220  * password from the passwd file is < 10 characters (i.e.
221  * a valid password would never crypt() to it). This will
222  * prevents users from using NULL password fields as things
223  * stand right now.
224  */
225  if ((!encrypted_pass) || (strlen(encrypted_pass) < 10)) {
226  if ((spwd = getspnam(name)) == NULL) {
227  RETURN_MODULE_NOTFOUND;
228  }
229  encrypted_pass = spwd->sp_pwdp;
230  }
231 #endif /* HAVE_GETSPNAM */
232 
233 #ifdef DENY_SHELL
234  /*
235  * Users with a particular shell are denied access
236  */
237  if (strcmp(pwd->pw_shell, DENY_SHELL) == 0) {
238  REDEBUG("Invalid shell", name);
239  RETURN_MODULE_REJECT;
240  }
241 #endif
242 
243 #ifdef HAVE_GETUSERSHELL
244  /*
245  * Check /etc/shells for a valid shell. If that file
246  * contains /RADIUSD/ANY/SHELL then any shell will do.
247  */
248  while ((shell = getusershell()) != NULL) {
249  if (strcmp(shell, pwd->pw_shell) == 0 ||
250  strcmp(shell, "/RADIUSD/ANY/SHELL") == 0) {
251  break;
252  }
253  }
254  endusershell();
255  if (!shell) {
256  REDEBUG("[%s]: invalid shell [%s]", name, pwd->pw_shell);
257  RETURN_MODULE_REJECT;
258  }
259 #endif
260 
261 #if defined(HAVE_GETSPNAM) && !defined(M_UNIX)
262  /*
263  * Check if password has expired.
264  */
265  if (spwd && spwd->sp_lstchg > 0 && spwd->sp_max >= 0 &&
266  (fr_time_to_sec(request->packet->timestamp) / 86400) > (spwd->sp_lstchg + spwd->sp_max)) {
267  REDEBUG("[%s]: password has expired", name);
268  RETURN_MODULE_REJECT;
269  }
270  /*
271  * Check if account has expired.
272  */
273  if (spwd && spwd->sp_expire > 0 &&
274  (fr_time_to_sec(request->packet->timestamp) / 86400) > spwd->sp_expire) {
275  REDEBUG("[%s]: account has expired", name);
276  RETURN_MODULE_REJECT;
277  }
278 #endif
279 
280 #if defined(__FreeBSD__) || defined(bsdi) || defined(_PWF_EXPIRE)
281  /*
282  * Check if password has expired.
283  */
284  if ((pwd->pw_expire > 0) &&
285  (fr_time_to_sec(request->packet->timestamp) > pwd->pw_expire)) {
286  REDEBUG("[%s]: password has expired", name);
287  RETURN_MODULE_REJECT;
288  }
289 #endif
290 
291  /*
292  * We might have a passwordless account.
293  *
294  * FIXME: Maybe add Auth-Type := Accept?
295  */
296  if (encrypted_pass[0] == 0)
297  RETURN_MODULE_NOOP;
298 
299  MEM(pair_update_control(&vp, attr_crypt_password) >= 0);
300  fr_pair_value_strdup(vp, encrypted_pass);
301 
302  RETURN_MODULE_UPDATED;
303 }

306 /*
307  * UUencode 4 bits base64. We use this to turn a 4 byte field
308  * (an IP address) into 6 bytes of ASCII. This is used for the
309  * wtmp file if we didn't find a short name in the naslist file.
310  */
311 static char *uue(void *in)
312 {
313  int i;
314  static unsigned char res[7];
315  unsigned char *data = (unsigned char *)in;
316 
317  res[0] = ENC( data[0] >> 2 );
318  res[1] = ENC( ((data[0] << 4) & 060) + ((data[1] >> 4) & 017) );
319  res[2] = ENC( ((data[1] << 2) & 074) + ((data[2] >> 6) & 03) );
320  res[3] = ENC( data[2] & 077 );
321 
322  res[4] = ENC( data[3] >> 2 );
323  res[5] = ENC( (data[3] << 4) & 060 );
324  res[6] = 0;
325 
326  for(i = 0; i < 6; i++) {
327  if (res[i] == ' ') res[i] = '`';
328  if (res[i] < 32 || res[i] > 127)
329  printf("uue: protocol error ?!\n");
330  }
331  return (char *)res;
332 }

335 /*
336  * Unix accounting - write a wtmp file.
337  */
338 static unlang_action_t CC_HINT(nonnull) mod_accounting(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
339 {
340  rlm_unix_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_unix_t);
341  fr_pair_t *vp;
342  fr_cursor_t cursor;
343  FILE *fp;
344  struct utmp ut;
345  time_t t;
346  char buf[64];
347  char const *s;
348  int delay = 0;
349  int status = -1;
350  int nas_address = 0;
351  int framed_address = 0;
352 #ifdef USER_PROCESS
353  int protocol = -1;
354 #endif
355  uint32_t nas_port = 0;
356  bool port_seen = true;
357 
358 
359  /*
360  * No radwtmp. Don't do anything.
361  */
362  if (!inst->radwtmp) {
363  RDEBUG2("No radwtmp file configured. Ignoring accounting request");
364  RETURN_MODULE_NOOP;
365  }
366 
367  if (request->packet->socket.inet.src_ipaddr.af != AF_INET) {
368  RDEBUG2("IPv6 is not supported!");
369  RETURN_MODULE_NOOP;
370  }
371 
372  /*
373  * Which type is this.
374  */
375  if ((vp = fr_pair_find_by_da(&request->request_pairs, attr_acct_status_type)) == NULL) {
376  RDEBUG2("no Accounting-Status-Type attribute in request");
377  RETURN_MODULE_NOOP;
378  }
379  status = vp->vp_uint32;
380 
381  /*
382  * FIXME: handle FR_STATUS_ALIVE like 1.5.4.3 did.
383  */
384  if (status != FR_STATUS_START &&
385  status != FR_STATUS_STOP)
386  RETURN_MODULE_NOOP;
387 
388  /*
389  * We're only interested in accounting messages
390  * with a username in it.
391  */
392  if (fr_pair_find_by_da(&request->request_pairs, attr_user_name) == NULL)
393  RETURN_MODULE_NOOP;
394 
395  t = fr_time_to_sec(request->packet->timestamp);
396  memset(&ut, 0, sizeof(ut));
397 
398  /*
399  * First, find the interesting attributes.
400  */
401  for (vp = fr_cursor_init(&cursor, &request->request_pairs);
402  vp;
403  vp = fr_cursor_next(&cursor)) {
404  if (vp->da == attr_user_name) {
405  if (vp->vp_length >= sizeof(ut.ut_name)) {
406  memcpy(ut.ut_name, vp->vp_strvalue, sizeof(ut.ut_name));
407  } else {
408  strlcpy(ut.ut_name, vp->vp_strvalue, sizeof(ut.ut_name));
409  }
410 
411  } else if (vp->da == attr_login_ip_host ||
412  vp->da == attr_framed_ip_address) {
413  framed_address = vp->vp_ipv4addr;
414 
415 #ifdef USER_PROCESS
416  } else if (vp->da == attr_framed_protocol) {
417  protocol = vp->vp_uint32;
418 #endif
419  } else if (vp->da == attr_nas_ip_address) {
420  nas_address = vp->vp_ipv4addr;
421 
422  } else if (vp->da == attr_nas_port) {
423  nas_port = vp->vp_uint32;
424  port_seen = true;
425 
426  } else if (vp->da == attr_acct_delay_time) {
427  delay = vp->vp_ipv4addr;
428  }
429  }
430 
431  /*
432  * We don't store !root sessions, or sessions
433  * where we didn't see a NAS-Port attribute.
434  */
435  if (strncmp(ut.ut_name, "!root", sizeof(ut.ut_name)) == 0 || !port_seen)
436  RETURN_MODULE_NOOP;
437 
438  /*
439  * If we didn't find out the NAS address, use the
440  * originator's IP address.
441  */
442  if (nas_address == 0) {
443  nas_address = request->packet->socket.inet.src_ipaddr.addr.v4.s_addr;
444  }
445  s = request->client->shortname;
446  if (!s || s[0] == 0) s = uue(&(nas_address));
447 
448 #ifdef __linux__
449  /*
450  * Linux has a field for the client address.
451  */
452  ut.ut_addr = framed_address;
453 #endif
454  /*
455  * We use the tty field to store the terminal servers' port
456  * and address so that the tty field is unique.
457  */
458  snprintf(buf, sizeof(buf), "%03u:%s", nas_port, s);
459  strlcpy(ut.ut_line, buf, sizeof(ut.ut_line));
460 
461  /*
462  * We store the dynamic IP address in the hostname field.
463  */
464 #ifdef UT_HOSTSIZE
465  if (framed_address) {
466  inet_ntop(AF_INET, &framed_address, buf, sizeof(buf));
467  strlcpy(ut.ut_host, buf, sizeof(ut.ut_host));
468  }
469 #endif
470 #ifdef USE_UTMPX
471  ut.ut_xtime = t - delay;
472 #else
473  ut.ut_time = t - delay;
474 #endif
475 #ifdef USER_PROCESS
476  /*
477  * And we can use the ID field to store
478  * the protocol.
479  */
480  if (protocol == FR_PPP)
481  strcpy(ut.ut_id, "P");
482  else if (protocol == FR_SLIP)
483  strcpy(ut.ut_id, "S");
484  else
485  strcpy(ut.ut_id, "T");
486  ut.ut_type = status == FR_STATUS_STOP ? DEAD_PROCESS : USER_PROCESS;
487 #endif
488  if (status == FR_STATUS_STOP)
489  ut.ut_name[0] = 0;
490 
491  /*
492  * Write a RADIUS wtmp log file.
493  *
494  * Try to open the file if we can't, we don't write the
495  * wtmp file. If we can try to write. If we fail,
496  * return RLM_MODULE_FAIL ..
497  */
498  if ((fp = fopen(inst->radwtmp, "a")) != NULL) {
499  if ((fwrite(&ut, sizeof(ut), 1, fp)) != 1) {
500  fclose(fp);
501  RETURN_MODULE_FAIL;
502  }
503  fclose(fp);
504  } else
505  RETURN_MODULE_FAIL;
506 
507  RETURN_MODULE_OK;
508 }
