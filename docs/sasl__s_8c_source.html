53 /** Callback for fr_ldap_sasl_interactive_bind
54  *
55  * @param handle used for the SASL bind.
56  * @param flags data as provided to fr_ldap_sasl_interactive_bind.
57  * @param ctx Our context data, containing the identity, password, realm and various other things.
58  * @param sasl_callbacks Array of challenges to provide responses for.
59  * @return SASL_OK.
60  */
61 static int _sasl_interact(UNUSED LDAP *handle, UNUSED unsigned flags, void *ctx, void *sasl_callbacks)
62 {
63  fr_ldap_sasl_ctx_t *this = ctx;
64  request_t *request = this->request;
65  fr_ldap_config_t const *handle_config = this->conn->config;
66  sasl_interact_t *cb = sasl_callbacks;
67  sasl_interact_t *cb_p;
68 
69  for (cb_p = cb; cb_p->id != SASL_CB_LIST_END; cb_p++) {
70  ROPTIONAL(RDEBUG3, DEBUG3, "SASL challenge : %s", cb_p->challenge);
71  ROPTIONAL(RDEBUG3, DEBUG3, "SASL prompt : %s", cb_p->prompt);
72 
73  switch (cb_p->id) {
74  case SASL_CB_AUTHNAME:
75  cb_p->result = this->identity;
76  break;
77 
78  case SASL_CB_PASS:
79  cb_p->result = this->password;
80  break;
81 
82  case SASL_CB_USER:
83  cb_p->result = this->extra->proxy ? this->extra->proxy : this->identity;
84  break;
85 
86  case SASL_CB_GETREALM:
87  if (this->extra->realm) cb_p->result = this->extra->realm;
88  break;
89 
90  default:
91  break;
92  }
93  ROPTIONAL(RDEBUG3, DEBUG3, "SASL result : %s", cb_p->result ? (char const *)cb_p->result : "");
94  }
95  return SASL_OK;
96 }

98 /** Initiate an LDAP interactive bind
99  *
100  * @param[in] request Current request, this may be NULL.
101  * @param[in] conn to use. May change as this function calls functions
102  * which auto re-connect.
103  * @param[in] identity of the user.
104  * @param[in] password of the user.
105  * @param[in] sasl mechanism to use for bind, and additional parameters.
106  * @param[in] serverctrls Search controls to pass to the server. May be NULL.
107  * @param[in] clientctrls Search controls for fr_ldap_sasl_interactive. May be NULL.
108  * @param[in] timeout Maximum time bind is allowed to take.
109  * @return One of the LDAP_PROC_* (#fr_ldap_rcode_t) values.
110  */
111 fr_ldap_rcode_t fr_ldap_sasl_interactive(request_t *request,
112  fr_ldap_connection_t *conn, char const *identity,
113  char const *password, fr_ldap_sasl_t const *sasl,
114  LDAPControl **serverctrls, LDAPControl **clientctrls,
115  fr_time_delta_t timeout)
116 {
117  fr_ldap_rcode_t status;
118  int ret = 0;
119  int msgid;
120  char const *mech;
121  LDAPMessage *result = NULL;
122  fr_ldap_sasl_ctx_t sasl_ctx; /* SASL defaults */
123 
124  fr_ldap_config_t const *handle_config = conn->config;
125 
126  LDAPControl *our_serverctrls[LDAP_MAX_CONTROLS];
127  LDAPControl *our_clientctrls[LDAP_MAX_CONTROLS];
128 
129  fr_ldap_control_merge(our_serverctrls, our_clientctrls,
130  NUM_ELEMENTS(our_serverctrls),
131  NUM_ELEMENTS(our_clientctrls),
132  conn, serverctrls, clientctrls);
133 
134  sasl_ctx.request = request;
135  sasl_ctx.identity = identity;
136  sasl_ctx.password = password;
137  sasl_ctx.extra = sasl;
138  sasl_ctx.conn = conn;
139 
140  ROPTIONAL(RDEBUG2, DEBUG2, "Starting SASL mech(s): %s", sasl->mech);
141  for (;;) {
142  ret = ldap_sasl_interactive_bind(conn->handle, NULL, sasl->mech,
143  our_serverctrls, our_clientctrls,
144  LDAP_SASL_AUTOMATIC,
145  _sasl_interact, &sasl_ctx, result,
146  &mech, &msgid);
147 
148  /*
149  * If fr_ldap_sasl_interactive_bind indicates it didn't want
150  * to continue, then we're done.
151  *
152  * Calling ldap_result here, results in a timeout in some
153  * cases, so we need to figure out whether the bind was
154  * successful without the help of ldap_result.
155  */
156  if (ret != LDAP_SASL_BIND_IN_PROGRESS) {
157  status = fr_ldap_error_check(NULL, conn, NULL, identity);
158  break; /* Old result gets freed on after exit */
159  }
160 
161  ldap_msgfree(result); /* We always need to free the old message */
162 
163  /*
164  * If LDAP parse result indicates there was an error
165  * then we're done.
166  */
167  status = fr_ldap_result(&result, NULL, conn, msgid, 0, identity, timeout);
168  switch (status) {
169  case LDAP_PROC_SUCCESS: /* fr_ldap_sasl_interactive_bind should have indicated success */
170  case LDAP_PROC_CONTINUE:
171  break;
172 
173  default:
174  goto done;
175  }
176 
177  /*
178  * ...otherwise, the bind is still in progress.
179  */
180  ROPTIONAL(RDEBUG3, DEBUG3, "Continuing SASL mech %s...", mech);
181 
182  /*
183  * Write the servers response to the debug log
184  */
185  if (((request && RDEBUG_ENABLED3) || DEBUG_ENABLED3) && result) {
186  struct berval *srv_cred;
187 
188  if (ldap_parse_sasl_bind_result(conn->handle, result, &srv_cred, 0) == 0) {
189  ROPTIONAL(RDEBUG3, DEBUG3, "SASL response : %pV",
190  fr_box_strvalue_len(srv_cred->bv_val, srv_cred->bv_len));
191  ldap_memfree(srv_cred);
192  }
193  }
194  }
195 done:
196  ldap_msgfree(result);
197 
198  if (status != LDAP_PROC_SUCCESS) ROPTIONAL(RPEDEBUG, PERROR, "SASL bind failed");
199 
200  return status;
201 }
