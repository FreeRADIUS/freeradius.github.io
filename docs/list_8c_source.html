35 /*
36  * See if two packets are identical.
37  *
38  * Note that we do NOT compare the authentication vectors.
39  * That's because if the authentication vector is different,
40  * it means that the NAS has given up on the earlier request.
41  */
42 int fr_packet_cmp(fr_radius_packet_t const *a, fr_radius_packet_t const *b)
43 {
44  int ret;
45 
46  /*
47  * 256-way fanout.
48  */
49  if (a->id < b->id) return -1;
50  if (a->id > b->id) return +1;
51 
52  if (a->socket.fd < b->socket.fd) return -1;
53  if (a->socket.fd > b->socket.fd) return +1;
54 
55  /*
56  * Source ports are pretty much random.
57  */
58  ret = (int) a->socket.inet.src_port - (int) b->socket.inet.src_port;
59  if (ret != 0) return ret;
60 
61  /*
62  * Usually many client IPs, and few server IPs
63  */
64  ret = fr_ipaddr_cmp(&a->socket.inet.src_ipaddr, &b->socket.inet.src_ipaddr);
65  if (ret != 0) return ret;
66 
67  /*
68  * One socket can receive packets for multiple
69  * destination IPs, so we check that before checking the
70  * file descriptor.
71  */
72  ret = fr_ipaddr_cmp(&a->socket.inet.dst_ipaddr, &b->socket.inet.dst_ipaddr);
73  if (ret != 0) return ret;
74 
75  /*
76  * At this point, the order of comparing socket FDs
77  * and/or destination ports doesn't matter. One of those
78  * fields will make the socket unique, and the other is
79  * pretty much redundant.
80  */
81  ret = (int) a->socket.inet.dst_port - (int) b->socket.inet.dst_port;
82  return ret;
83 }

137 static fr_packet_socket_t *fr_socket_find(fr_packet_list_t *pl, int sockfd)
138 {
139  int i, start;
140 
141  i = start = SOCK2OFFSET(sockfd);
142 
143  do { /* make this hack slightly more efficient */
144  if (pl->sockets[i].socket.fd == sockfd) return &pl->sockets[i];

146  i = (i + 1) & SOCKOFFSET_MASK;
147  } while (i != start);
148

203 bool fr_packet_list_socket_add(fr_packet_list_t *pl, int sockfd, int proto,
204  fr_ipaddr_t *dst_ipaddr, uint16_t dst_port,
205  void *ctx)
206 {
207  int i, start;
208  struct sockaddr_storage src;
209  socklen_t sizeof_src;
210  fr_packet_socket_t *ps;
211 
212  if (!pl || !dst_ipaddr || (dst_ipaddr->af == AF_UNSPEC)) {
213  fr_strerror_printf("Invalid argument");
214  return false;
215  }
216 
217  if (pl->num_sockets >= MAX_SOCKETS) {
218  fr_strerror_printf("Too many open sockets");
219  return false;
220  }
221 
222  ps = NULL;
223  i = start = SOCK2OFFSET(sockfd);
224 
225  do {
226  if (pl->sockets[i].socket.fd == -1) {
227  ps = &pl->sockets[i];
228  break;
229  }
230 
231  i = (i + 1) & SOCKOFFSET_MASK;
232  } while (i != start);
233 
234  if (!ps) {
235  fr_strerror_printf("All socket entries are full");
236  return false;
237  }
238 
239  memset(ps, 0, sizeof(*ps));
240  ps->ctx = ctx;
241  ps->socket.proto = proto;
242 
243  /*
244  * Get address family, etc. first, so we know if we
245  * need to do udpfromto.
246  *
247  * FIXME: udpfromto also does this, but it's not
248  * a critical problem.
249  */
250  sizeof_src = sizeof(src);
251  memset(&src, 0, sizeof_src);
252  if (getsockname(sockfd, (struct sockaddr *) &src, &sizeof_src) < 0) {
253  fr_strerror_printf("%s", fr_syserror(errno));
254  return false;
255  }
256 
257  if (fr_ipaddr_from_sockaddr(&ps->socket.inet.src_ipaddr, &ps->socket.inet.src_port, &src, sizeof_src) < 0) {
258  fr_strerror_printf("Failed to get IP");
259  return false;
260  }
261 
262  ps->socket.inet.dst_ipaddr = *dst_ipaddr;
263  ps->socket.inet.dst_port = dst_port;
264 
265  ps->src_any = fr_ipaddr_is_inaddr_any(&ps->socket.inet.src_ipaddr);
266  if (ps->src_any < 0) return false;
267 
268  ps->dst_any = fr_ipaddr_is_inaddr_any(&ps->socket.inet.dst_ipaddr);
269  if (ps->dst_any < 0) return false;
270 
271  /*
272  * As the last step before returning.
273  */
274  ps->socket.fd = sockfd;
275  pl->num_sockets++;
276 
277  return true;
278 }

343 /*
344  * This presumes that the reply has dst_ipaddr && dst_port set up
345  * correctly (i.e. real IP, or "*").

347 fr_radius_packet_t **fr_packet_list_find_byreply(fr_packet_list_t *pl, fr_radius_packet_t *reply)
348 {
349  fr_radius_packet_t my_request, *request;
350  fr_packet_socket_t *ps;
351 
352  if (!pl || !reply) return NULL;
353 
354  ps = fr_socket_find(pl, reply->socket.fd);
355  if (!ps) return NULL;
356 
357  /*
358  * TCP sockets are always bound to the correct src/dst IP/port
359  */
360  if (ps->socket.proto == IPPROTO_TCP) {
361  fr_socket_addr_swap(&reply->socket, &ps->socket);
362  my_request.socket = ps->socket;
363  } else {
364  my_request.socket = ps->socket;
365 
366  if (!ps->src_any) my_request.socket.inet.src_ipaddr = reply->socket.inet.dst_ipaddr;
367  my_request.socket.inet.dst_ipaddr = reply->socket.inet.src_ipaddr;
368  my_request.socket.inet.dst_port = reply->socket.inet.src_port;
369  }
370 
371  /*
372  * Initialize request from reply, AND from the source
373  * IP & port of this socket. The client may have bound
374  * the socket to 0, in which case it's some random port,
375  * that is NOT in the original request->socket.inet.src_port.
376  */
377  my_request.socket.fd = reply->socket.fd;
378  my_request.id = reply->id;
379  request = &my_request;
380 
381  return rbtree_finddata(pl->tree, &request);
382 }

406 /*
407  * 1 == ID was allocated & assigned
408  * 0 == couldn't allocate ID.
409  *
410  * Note that this ALSO assigns a socket to use, and updates
411  * packet->request->socket.inet.src_ipaddr && packet->request->socket.inet.src_port
412  *
413  * In multi-threaded systems, the calls to id_alloc && id_free
414  * should be protected by a mutex. This does NOT have to be
415  * the same mutex as the one protecting the insert/find/yank
416  * calls!
417  *
418  * We assume that the packet has dst_ipaddr && dst_port
419  * already initialized. We will use those to find an
420  * outgoing socket. The request MAY also have src_ipaddr set.
421  *
422  * We also assume that the sender doesn't care which protocol
423  * should be used.
424  */
425 bool fr_packet_list_id_alloc(fr_packet_list_t *pl, int proto,
426  fr_radius_packet_t **request_p, void **pctx)
427 {
428  int i, j, k, fd, id, start_i, start_j, start_k;
429  int src_any = 0;
430  fr_packet_socket_t *ps= NULL;
431  fr_radius_packet_t *request = *request_p;
432 
433  if ((request->socket.inet.dst_ipaddr.af == AF_UNSPEC) ||
434  (request->socket.inet.dst_port == 0)) {
435  fr_strerror_printf("No destination address/port specified");
436  return false;
437  }
438 
439  /*
440  * Special case: unspec == "don't care"
441  */
442  if (request->socket.inet.src_ipaddr.af == AF_UNSPEC) {
443  memset(&request->socket.inet.src_ipaddr, 0, sizeof(request->socket.inet.src_ipaddr));
444  request->socket.inet.src_ipaddr.af = request->socket.inet.dst_ipaddr.af;
445  }
446 
447  src_any = fr_ipaddr_is_inaddr_any(&request->socket.inet.src_ipaddr);
448  if (src_any < 0) {
449  fr_strerror_printf("Can't check src_ipaddr");
450  return false;
451  }
452 
453  /*
454  * MUST specify a destination address.
455  */
456  if (fr_ipaddr_is_inaddr_any(&request->socket.inet.dst_ipaddr) != 0) {
457  fr_strerror_printf("Must specify a dst_ipaddr");
458  return false;
459  }
460 
461  /*
462  * FIXME: Go to an LRU system. This prevents ID re-use
463  * for as long as possible. The main problem with that
464  * approach is that it requires us to populate the
465  * LRU/FIFO when we add a new socket, or a new destination,
466  * which can be expensive.
467  *
468  * The LRU can be avoided if the caller takes care to free
469  * Id's only when all responses have been received, OR after
470  * a timeout.
471  *
472  * Right now, the random approach is almost OK... it's
473  * brute-force over all of the available ID's, BUT using
474  * random numbers for everything spreads the load a bit.
475  *
476  * The old method had a hash lookup on allocation AND
477  * on free. The new method has brute-force on allocation,
478  * and near-zero cost on free.
479  */
480 
481  id = fd = -1;
482  if (request->id >= 0 && request->id < 256)
483  id = request->id;
484  start_i = fr_rand() & SOCKOFFSET_MASK;
485 
486 #define ID_i ((i + start_i) & SOCKOFFSET_MASK)
487  for (i = 0; i < MAX_SOCKETS; i++) {
488  if (pl->sockets[ID_i].socket.fd == -1) continue; /* paranoia */
489 
490  ps = &(pl->sockets[ID_i]);
491 
492  /*
493  * This socket is marked as "don't use for new
494  * packets". But we can still receive packets
495  * that are outstanding.
496  */
497  if (ps->dont_use) continue;
498 
499  /*
500  * All IDs are allocated: ignore it.
501  */
502  if (ps->num_outgoing == 256) continue;
503 
504  if (ps->socket.proto != proto) continue;
505 
506  /*
507  * Address families don't match, skip it.
508  */
509  if (ps->socket.inet.src_ipaddr.af != request->socket.inet.dst_ipaddr.af) continue;
510 
511  /*
512  * MUST match dst port, if we have one.
513  */
514  if ((ps->socket.inet.dst_port != 0) &&
515  (ps->socket.inet.dst_port != request->socket.inet.dst_port)) continue;
516 
517  /*
518  * MUST match requested src port, if one has been given.
519  */
520  if ((request->socket.inet.src_port != 0) &&
521  (ps->socket.inet.src_port != request->socket.inet.src_port)) continue;
522 
523  /*
524  * We don't care about the source IP, but this
525  * socket is link local, and the requested
526  * destination is not link local. Ignore it.
527  */
528  if (src_any && (ps->socket.inet.src_ipaddr.af == AF_INET) &&
529  (((ps->socket.inet.src_ipaddr.addr.v4.s_addr >> 24) & 0xff) == 127) &&
530  (((request->socket.inet.dst_ipaddr.addr.v4.s_addr >> 24) & 0xff) != 127)) continue;
531 
532  /*
533  * We're sourcing from *, and they asked for a
534  * specific source address: ignore it.
535  */
536  if (ps->src_any && !src_any) continue;
537 
538  /*
539  * We're sourcing from a specific IP, and they
540  * asked for a source IP that isn't us: ignore
541  * it.
542  */
543  if (!ps->src_any && !src_any &&
544  (fr_ipaddr_cmp(&request->socket.inet.src_ipaddr,
545  &ps->socket.inet.src_ipaddr) != 0)) continue;
546 
547  /*
548  * UDP sockets are allowed to match
549  * destination IPs exactly, OR a socket
550  * with destination * is allowed to match
551  * any requested destination.
552  *
553  * TCP sockets must match the destination
554  * exactly. They *always* have dst_any=0,
555  * so the first check always matches.
556  */
557  if (!ps->dst_any &&
558  (fr_ipaddr_cmp(&request->socket.inet.dst_ipaddr,
559  &ps->socket.inet.dst_ipaddr) != 0)) continue;
560 
561  /*
562  * Otherwise, this socket is OK to use.
563  */
564 
565  /*
566  * An explicit ID was requested
567  */
568 
569  if (id != -1) {
570  if ((ps->id[(id >> 3) & 0x1f] & (1 << (id & 0x07))) != 0) continue;
571 
572  ps->id[(id >> 3) & 0x1f] |= (1 << (id & 0x07));
573  fd = i;
574  break;
575  }
576 
577  /*
578  * Look for a free Id, starting from a random number.
579  */
580  start_j = fr_rand() & 0x1f;
581 #define ID_j ((j + start_j) & 0x1f)
582  for (j = 0; j < 32; j++) {
583  if (ps->id[ID_j] == 0xff) continue;
584 
585 
586  start_k = fr_rand() & 0x07;
587 #define ID_k ((k + start_k) & 0x07)
588  for (k = 0; k < 8; k++) {
589  if ((ps->id[ID_j] & (1 << ID_k)) != 0) continue;
590 
591  ps->id[ID_j] |= (1 << ID_k);
592  id = (ID_j * 8) + ID_k;
593  fd = i;
594  break;
595  }
596  if (fd >= 0) break;
597  }
598 #undef ID_i
599 #undef ID_j
600 #undef ID_k
601  break;
602  }
603 
604  /*
605  * Ask the caller to allocate a new ID.
606  */
607  if (fd < 0) {
608  fr_strerror_printf("Failed finding socket, caller must allocate a new one");
609  return false;
610  }
611 
612  /*
613  * Set the ID, source IP, and source port.
614  */
615  request->id = id;
616 
617  request->socket.fd = ps->socket.fd;
618  request->socket.inet.src_ipaddr = ps->socket.inet.src_ipaddr;
619  request->socket.inet.src_port = ps->socket.inet.src_port;
620 
621  /*
622  * If we managed to insert it, we're done.
623  */
624  if (fr_packet_list_insert(pl, request_p)) {
625  if (pctx) *pctx = ps->ctx;
626  ps->num_outgoing++;
627  pl->num_outgoing++;
628  return true;
629  }
630 
631  /*
632  * Mark the ID as free. This is the one line from
633  * id_free() that we care about here.
634  */
635  ps->id[(request->id >> 3) & 0x1f] &= ~(1 << (request->id & 0x07));
636 
637  request->id = -1;
638  request->socket.fd = -1;
639  request->socket.inet.src_ipaddr.af = AF_UNSPEC;
640  request->socket.inet.src_port = 0;
641 
642  return false;
643 }

709  * Round-robins the receivers, without priority.
710  *
711  * FIXME: Add socket.fd, if -1, do round-robin, else do socket.fd
712  * IF in fdset.
713  */
714 fr_radius_packet_t *fr_packet_list_recv(fr_packet_list_t *pl, fd_set *set, uint32_t max_attributes, bool require_ma)
715 {
716  int start;
717  fr_radius_packet_t *packet;
718 
719  if (!pl || !set) return NULL;
720 
721  start = pl->last_recv;
722  do {
723  start++;
724  start &= SOCKOFFSET_MASK;
725 
726  if (pl->sockets[start].socket.fd == -1) continue;
727 
728  if (!FD_ISSET(pl->sockets[start].socket.fd, set)) continue;
729 
730  if (pl->sockets[start].socket.proto == IPPROTO_TCP) {
731  packet = fr_tcp_recv(pl->sockets[start].socket.fd, false);
732  } else
733  packet = fr_radius_packet_recv(NULL, pl->sockets[start].socket.fd, UDP_FLAGS_NONE,
734  max_attributes, require_ma);
735  if (!packet) continue;
736 
737  /*
738  * Call fr_packet_list_find_byreply(). If it
739  * doesn't find anything, discard the reply.
740  */
741 
742  pl->last_recv = start;
743  packet->socket.proto = pl->sockets[start].socket.proto;
744  return packet;
745  } while (start != pl->last_recv);
746 
747  return NULL;
748 }
