106 /** Retrieve an entry from the cache
107  *
108  * If a cache entry is found, but the cache entry needs to be deserialized, the driver
109  * is expected to allocate an appropriately sized #rlm_cache_entry_t, perform the deserialisation,
110  * and write a pointer to the new entry to out, returning #CACHE_OK.
111  *
112  * If the #rlm_cache_handle_t is inviable, the driver should return #CACHE_RECONNECT, to have
113  * it reinitialised/reconnected.
114  *
115  * @param[out] out Where to write a pointer to the retrieved entry (if there was one).
116  * @param[in] config for this instance of the rlm_cache module.
117  * @param[in] instance Driver specific instance data.
118  * @param[in] request The current request.
119  * @param[in] handle the driver gave us when we called #cache_acquire_t, or NULL if no
120  * #cache_acquire_t callback was provided.
121  * @param[in] key to use to lookup cache entry
122  * @param[in] key_len the length of the key string.
123  * @return
124  * - #CACHE_RECONNECT - If handle needs to be reinitialised/reconnected.
125  * - #CACHE_ERROR - If the lookup couldn't be completed.
126  * - #CACHE_OK - Lookup was successful.
127  * - #CACHE_MISS - No cached entry was found.
128  */
129 typedef cache_status_t (*cache_entry_find_t)(rlm_cache_entry_t **out, rlm_cache_config_t const *config,
130  void *instance, request_t *request, void *handle,
131  uint8_t const *key, size_t key_len);

133 /** Insert an entry into the cache
134  *
135  * Serialize (if necessary) the entry passed to us, and write it to the cache with
136  * the key c->key.
137  *
138  * The cache entry should not be freed by the driver, irrespective of success or failure.
139  * If the entry needs to be freed after insertion because a local copy should not be kept,
140  * the driver should provide a #cache_entry_free_t callback.
141  *
142  * If the #rlm_cache_handle_t is inviable, the driver should return #CACHE_RECONNECT, to have
143  * it reinitialised/reconnected.
144  *
145  * @note This callback is not optional.
146  *
147  * @note This callback *must* overwrite existing cache entries on insert.
148  *
149  * @param config for this instance of the rlm_cache module.
150  * @param instance Driver specific instance data.
151  * @param request The current request.
152  * @param handle the driver gave us when we called #cache_acquire_t, or NULL if no
153  * #cache_acquire_t callback was provided.
154  * @param c to insert.
155  * @return
156  * - #CACHE_RECONNECT - If handle needs to be reinitialised/reconnected.
157  * - #CACHE_ERROR - If the insert couldn't be completed.
158  * - #CACHE_OK - If the insert was successful.
159  */
160 typedef cache_status_t (*cache_entry_insert_t)(rlm_cache_config_t const *config, void *instance,
161  request_t *request, void *handle,
162  rlm_cache_entry_t const *c);
