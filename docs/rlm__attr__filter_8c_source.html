84 static void check_pair(request_t *request, fr_pair_t *check_item, fr_pair_t *reply_item, int *pass, int *fail)
85 {
86  int compare;
87 
88  if (check_item->op == T_OP_SET) return;
89 
90  compare = fr_pair_cmp(check_item, reply_item);
91  if (compare < 0) RPEDEBUG("Comparison failed");
92 
93  if (compare == 1) {
94  ++*(pass);
95  } else {
96  ++*(fail);
97  }
98 
99  RDEBUG3("%pP %s %pP", reply_item, compare == 1 ? "allowed by" : "disallowed by", check_item);
100 
101  return;
102 }

104 static int attr_filter_getfile(TALLOC_CTX *ctx, rlm_attr_filter_t *inst, char const *filename, PAIR_LIST **pair_list)
105 {
106  fr_cursor_t cursor;
107  int rcode;
108  PAIR_LIST *attrs = NULL;
109  PAIR_LIST *entry;
110  fr_pair_t *vp;
111 
112  rcode = pairlist_read(ctx, dict_radius, filename, &attrs, 1);
113  if (rcode < 0) {
114  return -1;
115  }
116 
117  /*
118  * Walk through the 'attrs' file list.
119  */
120 
121  entry = attrs;
122  while (entry) {
123  entry->check = entry->reply;
124  entry->reply = NULL;
125 
126  for (vp = fr_cursor_init(&cursor, &entry->check);
127  vp;
128  vp = fr_cursor_next(&cursor)) {
129  /*
130  * If it's NOT a vendor attribute,
131  * and it's NOT a wire protocol
132  * and we ignore Fall-Through,
133  * then bitch about it, giving a good warning message.
134  */
135  if (fr_dict_attr_is_top_level(vp->da) && (vp->da->attr > 1000)) {
136  WARN("[%s]:%d Check item \"%s\"\n\tfound in filter list for realm \"%s\".\n",
137  filename, entry->lineno, vp->da->name, entry->name);
138  }
139  }
140 
141  entry = entry->next;
142  }
143 
144  *pair_list = attrs;
145  return 0;
146 }

168 /*
169  * Common attr_filter checks
170  */
171 static unlang_action_t CC_HINT(nonnull(1,2)) attr_filter_common(rlm_rcode_t *p_result,
172  void const *instance, request_t *request,
173  fr_radius_packet_t *packet)
174 {
175  rlm_attr_filter_t const *inst = talloc_get_type_abort_const(instance, rlm_attr_filter_t);
176  fr_pair_t *vp;
177  fr_cursor_t input, check, out;
178  fr_pair_t *input_item, *check_item, *output;
179  PAIR_LIST *pl;
180  int found = 0;
181  int pass, fail = 0;
182  char const *keyname = NULL;
183  char buffer[256];
184  ssize_t slen;
185 
186  if (!packet) {
187  RETURN_MODULE_NOOP;
188  }
189 
190  slen = tmpl_expand(&keyname, buffer, sizeof(buffer), request, inst->key, NULL, NULL);
191  if (slen < 0) {
192  RETURN_MODULE_FAIL;
193  }
194  if ((keyname == buffer) && is_truncated((size_t)slen, sizeof(buffer))) {
195  REDEBUG("Key too long, expected < " STRINGIFY(sizeof(buffer)) " bytes, got %zi bytes", slen);
196  RETURN_MODULE_FAIL;
197  }
198 
199  /*
200  * Head of the output list
201  */
202  output = NULL;
203  fr_cursor_init(&out, &output);
204 
205  /*
206  * Find the attr_filter profile entry for the entry.
207  */
208  for (pl = inst->attrs; pl; pl = pl->next) {
209  int fall_through = 0;
210  int relax_filter = inst->relaxed;
211 
212  /*
213  * If the current entry is NOT a default,
214  * AND the realm does NOT match the current entry,
215  * then skip to the next entry.
216  */
217  if ((strcmp(pl->name, "DEFAULT") != 0) &&
218  (strcmp(keyname, pl->name) != 0)) {
219  continue;
220  }
221 
222  RDEBUG2("Matched entry %s at line %d", pl->name, pl->lineno);
223  found = 1;
224 
225  for (check_item = fr_cursor_init(&check, &pl->check);
226  check_item;
227  check_item = fr_cursor_next(&check)) {
228  if (check_item->da == attr_fall_through) {
229  if (check_item->vp_uint32 == 1) {
230  fall_through = 1;
231  continue;
232  }
233  } else if (check_item->da == attr_relax_filter) {
234  relax_filter = check_item->vp_uint32;
235  }
236 
237  /*
238  * If it is a SET operator, add the attribute to
239  * the output list without checking it.
240  */
241  if (check_item->op == T_OP_SET ) {
242  vp = fr_pair_copy(packet, check_item);
243  if (!vp) goto error;
244 
245  xlat_eval_pair(request, vp);
246  fr_cursor_append(&out, vp);
247  }
248  }
249 
250  /*
251  * Iterate through the input items, comparing
252  * each item to every rule, then moving it to the
253  * output list only if it matches all rules
254  * for that attribute. IE, Idle-Timeout is moved
255  * only if it matches all rules that describe an
256  * Idle-Timeout.
257  */
258  for (input_item = fr_cursor_init(&input, &packet->vps);
259  input_item;
260  input_item = fr_cursor_next(&input)) {
261  pass = fail = 0; /* reset the pass,fail vars for each reply item */
262 
263  /*
264  * Reset the check_item pointer to beginning of the list
265  */
266  for (check_item = fr_cursor_head(&check);
267  check_item;
268  check_item = fr_cursor_next(&check)) {
269  /*
270  * Vendor-Specific is special, and matches any VSA if the
271  * comparison is always true.
272  */
273  if ((check_item->da == attr_vendor_specific) &&
274  (fr_dict_vendor_num_by_da(input_item->da) != 0) &&
275  (check_item->op == T_OP_CMP_TRUE)) {
276  pass++;
277  continue;
278  }
279 
280  if (input_item->da == check_item->da) {
281  check_pair(request, check_item, input_item, &pass, &fail);
282  }
283  }
284 
285  RDEBUG3("Attribute \"%s\" allowed by %i rules, disallowed by %i rules",
286  input_item->da->name, pass, fail);
287  /*
288  * Only move attribute if it passed all rules, or if the config says we
289  * should copy unmatched attributes ('relaxed' mode).
290  */
291  if (fail == 0 && (pass > 0 || relax_filter)) {
292  if (!pass) {
293  RDEBUG3("Attribute \"%s\" allowed by relaxed mode", input_item->da->name);
294  }
295  vp = fr_pair_copy(packet, input_item);
296  if (!vp) {
297  goto error;
298  }
299  fr_cursor_append(&out, vp);
300  }
301  }
302 
303  /* If we shouldn't fall through, break */
304  if (!fall_through) {
305  break;
306  }
307  }
308 
309  /*
310  * No entry matched. We didn't do anything.
311  */
312  if (!found) {
313  fr_assert(!output);
314  RETURN_MODULE_NOOP;
315  }
316 
317  /*
318  * Replace the existing request list with our filtered one
319  */
320  fr_pair_list_free(&packet->vps);
321  packet->vps = output;
322 
323  RETURN_MODULE_UPDATED;
324 
325 error:
326  fr_pair_list_free(&output);
327  RETURN_MODULE_FAIL;
328 }
