38 /** Seed the random number generator
39  *
40  * May be called any number of times.
41  */
42 void fr_rand_seed(void const *data, size_t size)
43 {
44  uint32_t hash;
45 
46  /*
47  * Ensure that the pool is initialized.
48  */
49  if (!fr_rand_initialized) {
50  int fd;
51 
52  memset(&fr_rand_pool, 0, sizeof(fr_rand_pool));
53 
54  fd = open("/dev/urandom", O_RDONLY);
55  if (fd >= 0) {
56  size_t total;
57  ssize_t this;
58 
59  total = 0;
60  while (total < sizeof(fr_rand_pool.randrsl)) {
61  this = read(fd, fr_rand_pool.randrsl,
62  sizeof(fr_rand_pool.randrsl) - total);
63  if ((this < 0) && (errno != EINTR)) break;
64  if (this > 0) total += this;
65  }
66  close(fd);
67  } else {
68  fr_rand_pool.randrsl[0] = fd;
69  fr_rand_pool.randrsl[1] = time(NULL);
70  fr_rand_pool.randrsl[2] = errno;
71  }
72 
73  fr_rand_init(&fr_rand_pool, 1);
74  fr_rand_pool.randcnt = 0;
75  fr_rand_initialized = 1;
76  }
77 
78  if (!data) return;
79 
80  /*
81  * Hash the user data
82  */
83  hash = fr_rand();
84  if (!hash) hash = fr_rand();
85  hash = fr_hash_update(data, size, hash);
86 
87  fr_rand_pool.randmem[fr_rand_pool.randcnt] ^= hash;
88 }

91 /** Return a 32-bit random number
92  *
93  */
94 uint32_t fr_rand(void)
95 {
96  uint32_t num;
97 
98  /*
99  * Ensure that the pool is initialized.
100  */
101  if (!fr_rand_initialized) {
102  fr_rand_seed(NULL, 0);
103  }
104 
105  num = fr_rand_pool.randrsl[fr_rand_pool.randcnt++];
106  if (fr_rand_pool.randcnt >= 256) {
107  fr_rand_pool.randcnt = 0;
108  fr_isaac(&fr_rand_pool);
109  }
110 
111  return num;
112 }

114 void fr_rand_buffer(void *start, size_t length)
115 {
116  uint32_t x;
117  uint8_t *buffer = start;
118  size_t buflen = length;
119 
120  if (buflen > 4) {
121  size_t i;
122 
123  for (i = 0; i < buflen; i += 4) {
124  x = fr_rand();
125  memcpy(buffer + i, &x, sizeof(x));
126  }
127 
128  /*
129  * Keep only the last bytes in the word.
130  */
131  i = buflen & ~0x03;
132  buffer += i;
133  buflen &= 0x03;
134  }
135 
136  if (!buflen) return;
137 
138  x = fr_rand();
139 
140  memcpy(buffer, &x, buflen);
141 }

143 /** Generate a random string
144  *
145  * @note Character selection is not perfectly distributed, should not be used
146  * for cryptographic purposes.
147  *
148  * @param[out] out Where to write the string
149  * @param[in] len Length of the output buffer.
150  * @param[in] class to pick characters from (see function body).
151  */
152 void fr_rand_str(uint8_t *out, size_t len, char class)
153 {
154  uint8_t *p = out, *end = p + len;
155  unsigned int word, mod;
156  uint8_t byte;
157 
158  /*
159  * Lookup tables for randstr char classes
160  */
161  static char randstr_punc[] = "!\"#$%&'()*+,-./:;<=>?@[\\]^_`{|}~";
162  static char randstr_salt[] = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmopqrstuvwxyz/.";
163 
164  /*
165  * Characters humans rarely confuse. Reduces char set considerably
166  * should only be used for things such as one time passwords.
167  */
168  static char randstr_otp[] = "469ACGHJKLMNPQRUVWXYabdfhijkprstuvwxyz";
169 
170 /*
171  * yeah yeah not perfect distribution
172  * but close enough.
173  */
174 #define fill(_expr) \
175 while (p < end) { \
176  if ((mod = ((p - out) & (sizeof(word) - 1))) == 0) word = fr_rand(); \
177  byte = ((uint8_t *)&word)[mod]; \
178  *p++ = (_expr); \
179 }
180 
181  switch (class) {
182  /*
183  * Lowercase letters
184  */
185  case 'c':
186  fill('a' + (byte % 26))
187  return;
188 
189  /*
190  * Uppercase letters
191  */
192  case 'C':
193  fill('A' + (byte % 26))
194  return;
195 
196  /*
197  * Numbers
198  */
199  case 'n':
200  fill('0' + (byte % 10));
201  return;
202 
203  /*
204  * Alpha numeric
205  */
206  case 'a':
207  fill(randstr_salt[byte % (sizeof(randstr_salt) - 3)]);
208  return;
209 
210  /*
211  * Punctuation
212  */
213  case '!':
214  fill(randstr_punc[byte % (sizeof(randstr_punc) - 1)]);
215  return;
216 
217  /*
218  * Alpha numeric + punctuation
219  */
220  case '.':
221  fill('!' + (byte % 95));
222  break;
223 
224  /*
225  * Alpha numeric + salt chars './'
226  */
227  case 's':
228  fill(randstr_salt[byte % (sizeof(randstr_salt) - 1)]);
229  break;
230 
231  /*
232  * Chars suitable for One Time Password tokens.
233  * Alpha numeric with easily confused char pairs removed.
234  */
235  case 'o':
236  fill(randstr_otp[byte % (sizeof(randstr_otp) - 1)]);
237  break;
238 
239  /*
240  * Binary data
241  */
242  case 'b':
243  default:
244  fill(byte)
245  return;
246  }
247 }
