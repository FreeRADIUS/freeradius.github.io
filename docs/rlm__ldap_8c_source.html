285 /** Expand an LDAP URL into a query, and return a string result from that query.
286  *
287  * @ingroup xlat_functions
288  */
289 static ssize_t ldap_xlat(UNUSED TALLOC_CTX *ctx, char **out, size_t outlen,
290  void const *mod_inst, UNUSED void const *xlat_inst,
291  request_t *request, char const *fmt)
292 {
293  fr_ldap_rcode_t status;
294  size_t len = 0;
295  rlm_ldap_t const *inst = mod_inst;
296 
297  LDAPURLDesc *ldap_url;
298  LDAPMessage *result = NULL;
299  LDAPMessage *entry = NULL;
300 
301  struct berval **values;
302 
303  fr_ldap_connection_t *conn;
304  int ldap_errno;
305 
306  char const *url;
307  char const **attrs;
308 
309  LDAPControl *server_ctrls[] = { NULL, NULL };
310 
311  url = fmt;
312 
313  if (!ldap_is_ldap_url(url)) {
314  REDEBUG("String passed does not look like an LDAP URL");
315  return -1;
316  }
317 
318  if (ldap_url_parse(url, &ldap_url)){
319  REDEBUG("Parsing LDAP URL failed");
320  return -1;
321  }
322 
323  /*
324  * Nothing, empty string, "*" string, or got 2 things, die.
325  */
326  if (!ldap_url->lud_attrs || !ldap_url->lud_attrs[0] ||
327  !*ldap_url->lud_attrs[0] ||
328  (strcmp(ldap_url->lud_attrs[0], "*") == 0) ||
329  ldap_url->lud_attrs[1]) {
330  REDEBUG("Bad attributes list in LDAP URL. URL must specify exactly one attribute to retrieve");
331 
332  goto free_urldesc;
333  }
334 
335  conn = mod_conn_get(inst, request);
336  if (!conn) goto free_urldesc;
337 
338  memcpy(&attrs, &ldap_url->lud_attrs, sizeof(attrs));
339 
340  if (fr_ldap_parse_url_extensions(&server_ctrls[0], request, conn, ldap_url->lud_exts) < 0) goto free_socket;
341 
342  status = fr_ldap_search(&result, request, &conn, ldap_url->lud_dn, ldap_url->lud_scope,
343  ldap_url->lud_filter, attrs, server_ctrls, NULL);
344 
345 #ifdef HAVE_LDAP_CREATE_SORT_CONTROL
346  if (server_ctrls[0]) ldap_control_free(server_ctrls[0]);
347 #endif
348 
349  switch (status) {
350  case LDAP_PROC_SUCCESS:
351  break;
352 
353  default:
354  goto free_socket;
355  }
356 
357  fr_assert(conn);
358  fr_assert(result);
359 
360  entry = ldap_first_entry(conn->handle, result);
361  if (!entry) {
362  ldap_get_option(conn->handle, LDAP_OPT_RESULT_CODE, &ldap_errno);
363  REDEBUG("Failed retrieving entry: %s", ldap_err2string(ldap_errno));
364  len = -1;
365  goto free_result;
366  }
367 
368  values = ldap_get_values_len(conn->handle, entry, ldap_url->lud_attrs[0]);
369  if (!values) {
370  RDEBUG2("No \"%s\" attributes found in specified object", ldap_url->lud_attrs[0]);
371  goto free_result;
372  }
373 
374  if (values[0]->bv_len >= outlen) goto free_values;
375 
376  memcpy(*out, values[0]->bv_val, values[0]->bv_len + 1); /* +1 as strlcpy expects buffer size */
377  len = values[0]->bv_len;
378 
379 free_values:
380  ldap_value_free_len(values);
381 free_result:
382  ldap_msgfree(result);
383 free_socket:
384  ldap_mod_conn_release(inst, request, conn);
385 free_urldesc:
386  ldap_free_urldesc(ldap_url);
387 
388  return len;
389 }

406 /** Perform a search and map the result of the search to server attributes
407  *
408  * Unlike LDAP xlat, this can be used to process attributes from multiple entries.
409  *
410  * @todo For xlat expansions we need to parse the raw URL first, and then apply
411  * different escape functions to the different parts.
412  *
413  * @param[in] mod_inst #rlm_ldap_t
414  * @param[in] proc_inst unused.
415  * @param[in,out] request The current request.
416  * @param[in] url LDAP url specifying base DN and filter.
417  * @param[in] maps Head of the map list.
418  * @return
419  * - #RLM_MODULE_NOOP no rows were returned.
420  * - #RLM_MODULE_UPDATED if one or more #fr_pair_t were added to the #request_t.
421  * - #RLM_MODULE_FAIL if an error occurred.
422  */
423 static rlm_rcode_t mod_map_proc(void *mod_inst, UNUSED void *proc_inst, request_t *request,
424  fr_value_box_t **url, map_t const *maps)
425 {
426  rlm_rcode_t rcode = RLM_MODULE_UPDATED;
427  rlm_ldap_t *inst = talloc_get_type_abort(mod_inst, rlm_ldap_t);
428  fr_ldap_rcode_t status;
429 
430  LDAPURLDesc *ldap_url;
431 
432  LDAPMessage *result = NULL;
433  LDAPMessage *entry = NULL;
434  map_t const *map;
435  char const *url_str;
436 
437  fr_ldap_connection_t *conn;
438 
439  LDAPControl *server_ctrls[] = { NULL, NULL };
440 
441  fr_ldap_map_exp_t expanded; /* faster than allocing every time */
442 
443  /*
444  * FIXME - Maybe it can be NULL?
445  */
446  if (!*url) {
447  REDEBUG("LDAP URL cannot be (null)");
448  return RLM_MODULE_FAIL;
449  }
450 
451  if (fr_value_box_list_concat(request, *url, url, FR_TYPE_STRING, true) < 0) {
452  REDEBUG("Failed concatenating input");
453  return RLM_MODULE_FAIL;
454  }
455  url_str = (*url)->vb_strvalue;
456 
457  if (!ldap_is_ldap_url(url_str)) {
458  REDEBUG("Map query string does not look like a valid LDAP URI");
459  return RLM_MODULE_FAIL;
460  }
461 
462  if (ldap_url_parse(url_str, &ldap_url)){
463  REDEBUG("Parsing LDAP URL failed");
464  return RLM_MODULE_FAIL;
465  }
466 
467  /*
468  * Expand the RHS of the maps to get the name of the attributes.
469  */
470  if (fr_ldap_map_expand(&expanded, request, maps) < 0) {
471  rcode = RLM_MODULE_FAIL;
472  goto free_urldesc;
473  }
474 
475  conn = mod_conn_get(inst, request);
476  if (!conn) goto free_expanded;
477 
478  if (fr_ldap_parse_url_extensions(&server_ctrls[0], request, conn, ldap_url->lud_exts) < 0) goto free_socket;
479 
480  status = fr_ldap_search(&result, request, &conn, ldap_url->lud_dn, ldap_url->lud_scope,
481  ldap_url->lud_filter, expanded.attrs, server_ctrls, NULL);
482 
483 #ifdef HAVE_LDAP_CREATE_SORT_CONTROL
484  if (server_ctrls[0]) ldap_control_free(server_ctrls[0]);
485 #endif
486 
487  switch (status) {
488  case LDAP_PROC_SUCCESS:
489  break;
490 
491  case LDAP_PROC_NO_RESULT:
492  rcode = RLM_MODULE_NOOP;
493  goto free_socket;
494 
495  default:
496  rcode = RLM_MODULE_FAIL;
497  goto free_socket;
498  }
499 
500  fr_assert(conn);
501  fr_assert(result);
502 
503  for (entry = ldap_first_entry(conn->handle, result);
504  entry;
505  entry = ldap_next_entry(conn->handle, entry)) {
506  char *dn = NULL;
507  int i;
508 
509 
510  if (RDEBUG_ENABLED2) {
511  dn = ldap_get_dn(conn->handle, entry);
512  RDEBUG2("Processing \"%s\"", dn);
513  }
514 
515  RINDENT();
516  for (map = maps, i = 0;
517  map != NULL;
518  map = map->next, i++) {
519  int ret;
520  fr_ldap_result_t attr;
521 
522  attr.values = ldap_get_values_len(conn->handle, entry, expanded.attrs[i]);
523  if (!attr.values) {
524  /*
525  * Many LDAP directories don't expose the DN of
526  * the object as an attribute, so we need this
527  * hack, to allow the user to retrieve it.
528  */
529  if (strcmp(LDAP_VIRTUAL_DN_ATTR, expanded.attrs[i]) == 0) {

579 /** Perform LDAP-Group comparison checking
580  *
581  * Attempts to match users to groups using a variety of methods.
582  *
583  * @param instance of the rlm_ldap module.
584  * @param request Current request.
585  * @param thing Unknown.
586  * @param check Which group to check for user membership.
587  * @param check_list Unknown.
588  * @return
589  * - 1 on failure (or if the user is not a member).
590  * - 0 on success.

592 static int rlm_ldap_groupcmp(void *instance, request_t *request, UNUSED fr_pair_t *thing, fr_pair_t *check,

603 
604  fr_assert(inst->groupobj_base_dn);
605 
606  RDEBUG2("Searching for user in group \"%pV\"", &check->data);
607 
608  if (check->vp_length == 0) {
609  REDEBUG("Cannot do comparison (group name is empty)");
610  return 1;
611  }
612 
613  /*
614  * Check if we can do cached membership verification
615  */
616  check_is_dn = fr_ldap_util_is_dn(check->vp_strvalue, check->vp_length);
617  if (check_is_dn) {
618  char *norm;
619  size_t len;
620 
621  MEM(norm = talloc_array(check, char, talloc_array_length(check->vp_strvalue)));
622  len = fr_ldap_util_normalise_dn(norm, check->vp_strvalue);
623 
624  /*
625  * Will clear existing buffer (i.e. check->vp_strvalue)
626  */
627  fr_pair_value_bstrdup_buffer_shallow(check, norm, check->vp_tainted);
628 
629  /*
630  * Trim buffer to match normalised DN
631  */
632  fr_pair_value_bstr_realloc(check, NULL, len);
633  }
634  if ((check_is_dn && inst->cacheable_group_dn) || (!check_is_dn && inst->cacheable_group_name)) {
635  rlm_rcode_t our_rcode;
636 
637  rlm_ldap_check_cached(&our_rcode, inst, request, check);
638  switch (our_rcode) {
639  case RLM_MODULE_NOTFOUND:
640  found = false;
641  goto finish;
642 
643  case RLM_MODULE_OK:
644  found = true;
645  goto finish;
646  /*
647  * Fallback to dynamic search on failure
648  */
649  case RLM_MODULE_FAIL:
650  case RLM_MODULE_INVALID:
651  default:
652  break;
653  }
654  }
655 
656  conn = mod_conn_get(inst, request);
657  if (!conn) return 1;
658

660  * This is used in the default membership filter.
661  */
662  user_dn = rlm_ldap_find_user(inst, request, &conn, NULL, false, NULL, &rcode);

664  ldap_mod_conn_release(inst, request, conn);
665  return 1;
666  }
667 
668  fr_assert(conn);
669 
670  /*
671  * Check groupobj user membership
672  */
673  if (inst->groupobj_membership_filter) {
674  rlm_rcode_t our_rcode;
675 
676  rlm_ldap_check_groupobj_dynamic(&our_rcode, inst, request, &conn, check);
677  switch (our_rcode) {
678  case RLM_MODULE_NOTFOUND:
679  break;
680 
681  case RLM_MODULE_OK:
682  found = true;
683  FALL_THROUGH;
684 
685  default:
686  goto finish;
687  }
688  }
689 
690  fr_assert(conn);
691 
692  /*
693  * Check userobj group membership
694  */
695  if (inst->userobj_membership_attr) {
696  rlm_rcode_t our_rcode;
697 
698  rlm_ldap_check_userobj_dynamic(&our_rcode, inst, request, &conn, user_dn, check);
699  switch (our_rcode) {
700  case RLM_MODULE_NOTFOUND:
701  break;
702 
703  case RLM_MODULE_OK:
704  found = true;
705  FALL_THROUGH;
706 
707  default:
708  goto finish;
709  }
710  }
711 
712  fr_assert(conn);
713 
714 finish:
715  if (conn) ldap_mod_conn_release(inst, request, conn);
716 
717  if (!found) {
718  RDEBUG2("User is not a member of \"%pV\"", &check->data);
719 
720  return 1;
721  }
722 
723  return 0;
724 }

1204  int i, total = 0, last_pass = 0;
1205 
1206  char *expanded[LDAP_MAX_ATTRMAP];
1207  int last_exp = 0;
1208 
1209  char const *attr;
1210  char const *value;
1211 
1212  char const *dn;
1213  /*
1214  * Build our set of modifications using the update sections in
1215  * the config.
1216  */
1217  CONF_ITEM *ci;
1218  CONF_PAIR *cp;
1219  CONF_SECTION *cs;
1220  fr_token_t op;
1221  char path[FR_MAX_STRING_LEN];
1222 
1223  char *p = path;
1224 
1225  fr_assert(section);
1226 
1227  /*
1228  * Locate the update section were going to be using
1229  */
1230  if (section->reference[0] != '.') {
1231  *p++ = '.';
1232  }
1233 
1234  if (xlat_eval(p, (sizeof(path) - (p - path)) - 1, request, section->reference, NULL, NULL) < 0) {
1235  goto error;
1236  }
1237 
1238  ci = cf_reference_item(NULL, section->cs, path);
1239  if (!ci) {
1240  goto error;
1241  }
1242 
1243  if (!cf_item_is_section(ci)){
1244  REDEBUG("Reference must resolve to a section");
1245 
1246  goto error;
1247  }
1248 
1249  cs = cf_section_find(cf_item_to_section(ci), "update", NULL);
1250  if (!cs) {
1251  REDEBUG("Section must contain 'update' subsection");
1252 
1253  goto error;
1254  }
1255 
1256  /*
1257  * Iterate over all the pairs, building our mods array
1258  */
1259  for (ci = cf_item_next(cs, NULL); ci != NULL; ci = cf_item_next(cs, ci)) {
1260  bool do_xlat = false;
1261 
1262  if (total == LDAP_MAX_ATTRMAP) {
1263  REDEBUG("Modify map size exceeded");
1264 
1265  goto error;
1266  }
1267 
1268  if (!cf_item_is_pair(ci)) {
1269  REDEBUG("Entry is not in \"ldap-attribute = value\" format");
1270 
1271  goto error;
1272  }
1273 
1274  /*
1275  * Retrieve all the information we need about the pair
1276  */
1277  cp = cf_item_to_pair(ci);
1278  value = cf_pair_value(cp);
1279  attr = cf_pair_attr(cp);
1280  op = cf_pair_operator(cp);
1281 
1282  if (!value || (*value == '\0')) {
1283  RDEBUG2("Empty value string, skipping attribute \"%s\"", attr);
1284 
1285  continue;
1286  }
1287 
1288  switch (cf_pair_value_quote(cp)) {
1289  case T_BARE_WORD:
1290  case T_SINGLE_QUOTED_STRING:
1291  break;
1292 
1293  case T_BACK_QUOTED_STRING:
1294  case T_DOUBLE_QUOTED_STRING:
1295  do_xlat = true;
1296  break;
1297 
1298  default:
1299  fr_assert(0);
1300  goto error;
1301  }
1302 
1303  if (op == T_OP_CMP_FALSE) {
1304  passed[last_pass] = NULL;
1305  } else if (do_xlat) {
1306  char *exp = NULL;
1307 
1308  if (xlat_aeval(request, &exp, request, value, NULL, NULL) <= 0) {
1309  RDEBUG2("Skipping attribute \"%s\"", attr);
1310 
1311  talloc_free(exp);
1312 
1313  continue;
1314  }
1315 
1316  expanded[last_exp++] = exp;
1317  passed[last_pass] = exp;
1318  /*
1319  * Static strings
1320  */
1321  } else {
1322  memcpy(&(passed[last_pass]), &value, sizeof(passed[last_pass]));
1323  }
1324 
1325  passed[last_pass + 1] = NULL;
1326 
1327  mod_s[total].mod_values = &(passed[last_pass]);
1328 
1329  last_pass += 2;
1330 
1331  switch (op) {
1332  /*
1333  * T_OP_EQ is *NOT* supported, it is impossible to
1334  * support because of the lack of transactions in LDAP
1335  */
1336  case T_OP_ADD:
1337  mod_s[total].mod_op = LDAP_MOD_ADD;
1338  break;
1339 
1340  case T_OP_SET:
1341  mod_s[total].mod_op = LDAP_MOD_REPLACE;
1342  break;
1343 
1344  case T_OP_SUB:
1345  case T_OP_CMP_FALSE:
1346  mod_s[total].mod_op = LDAP_MOD_DELETE;
1347  break;
1348 
1349 #ifdef LDAP_MOD_INCREMENT
1350  case T_OP_INCRM:
1351  mod_s[total].mod_op = LDAP_MOD_INCREMENT;
1352  break;
1353 #endif
1354  default:
1355  REDEBUG("Operator '%s' is not supported for LDAP modify operations",
1356  fr_table_str_by_value(fr_tokens_table, op, "<INVALID>"));
1357 
1358  goto error;
1359  }
1360 
1361  /*
1362  * Now we know the value is ok, copy the pointers into
1363  * the ldapmod struct.
1364  */
1365  memcpy(&(mod_s[total].mod_type), &attr, sizeof(mod_s[total].mod_type));
1366 
1367  mod_p[total] = &(mod_s[total]);
1368  total++;
1369  }
1370 
1371  if (total == 0) {
1372  rcode = RLM_MODULE_NOOP;
1373  goto release;
1374  }
1375 
1376  mod_p[total] = NULL;
1377 
1378  conn = mod_conn_get(inst, request);
1379  if (!conn) RETURN_MODULE_FAIL;
1380 
1381 
1382  dn = rlm_ldap_find_user(inst, request, &conn, NULL, false, NULL, &rcode);
1383  if (!dn || (rcode != RLM_MODULE_OK)) {
1384  goto error;
1385  }
1386 
1387  status = fr_ldap_modify(request, &conn, dn, modify, NULL, NULL);
1388  switch (status) {
1389  case LDAP_PROC_SUCCESS:
1390  break;
1391 
1392  case LDAP_PROC_REJECT:
1393  case LDAP_PROC_BAD_DN:
1394  rcode = RLM_MODULE_INVALID;
1395  break;
1396 
1397  default:
1398  rcode = RLM_MODULE_FAIL;
1399  break;
1400  };
1401 
1402 release:
1403 error:
1404  /*
1405  * Free up any buffers we allocated for xlat expansion
1406  */
1407  for (i = 0; i < last_exp; i++) talloc_free(expanded[i]);
1408 
1409  ldap_mod_conn_release(inst, request, conn);
1410 
1411  RETURN_MODULE_RCODE(rcode);
1412 }

1554 /** Instantiate the module
1555  *
1556  * Creates a new instance of the module reading parameters from a configuration section.
1557  *
1558  * @param conf to parse.
1559  * @param instance configuration data.
1560  * @return
1561  * - 0 on success.
1562  * - < 0 on failure.
1563  */
1564 static int mod_instantiate(void *instance, CONF_SECTION *conf)
1565 {
1566  size_t i;
1567 
1568  CONF_SECTION *options, *update;
1569  rlm_ldap_t *inst = instance;
1570 
1571  inst->cs = conf;
1572 
1573  options = cf_section_find(conf, "options", NULL);
1574  if (!options || !cf_pair_find(options, "chase_referrals")) {
1575  inst->handle_config.chase_referrals_unset = true; /* use OpenLDAP defaults */
1576  }
1577 
1578  /*
1579  * If the configuration parameters can't be parsed, then fail.
1580  */
1581  if ((parse_sub_section(inst, conf, &inst->accounting, MOD_ACCOUNTING) < 0) ||
1582  (parse_sub_section(inst, conf, &inst->postauth, MOD_POST_AUTH) < 0)) {
1583  cf_log_err(conf, "Failed parsing configuration");
1584 
1585  goto error;
1586  }
1587 
1588  /*
1589  * Sanity checks for cacheable groups code.
1590  */
1591  if (inst->cacheable_group_name && inst->groupobj_membership_filter) {
1592  if (!inst->groupobj_name_attr) {
1593  cf_log_err(conf, "Configuration item 'group.name_attribute' must be set if cacheable "
1594  "group names are enabled");
1595 
1596  goto error;
1597  }
1598  }
1599 
1600  /*
1601  * If we have a *pair* as opposed to a *section*
1602  * then the module is referencing another ldap module's
1603  * connection pool.
1604  */
1605  if (!cf_pair_find(conf, "pool")) {
1606  if (!inst->handle_config.server_str) {
1607  cf_log_err(conf, "Configuration item 'server' must have a value");
1608  goto error;
1609  }
1610  }
1611 
1612 #ifndef WITH_SASL
1613  if (inst->user_sasl.mech) {
1614  cf_log_err(conf, "Configuration item 'user.sasl.mech' not supported. "
1615  "Linked libldap does not provide ldap_sasl_bind( function");
1616  goto error;
1617  }
1618 
1619  if (inst->handle_config.admin_sasl.mech) {
1620  cf_log_err(conf, "Configuration item 'sasl.mech' not supported. "
1621  "Linked libldap does not provide ldap_sasl_interactive_bind function");
1622  goto error;
1623  }
1624 #endif
1625 
1626 #ifndef HAVE_LDAP_CREATE_SORT_CONTROL
1627  if (inst->userobj_sort_by) {
1628  cf_log_err(conf, "Configuration item 'sort_by' not supported. "
1629  "Linked libldap does not provide ldap_create_sort_control function");
1630  goto error;
1631  }
1632 #endif
1633 
1634 #ifndef HAVE_LDAP_URL_PARSE
1635  if (inst->handle_config.use_referral_credentials) {
1636  cf_log_err(conf, "Configuration item 'use_referral_credentials' not supported. "
1637  "Linked libldap does not support URL parsing");
1638  goto error;
1639  }
1640 #endif
1641 
1642  /*
1643  * Initialise server with zero length string to
1644  * make code below simpler.
1645  */
1646  inst->handle_config.server = talloc_strdup(inst, "");
1647 
1648  /*
1649  * Now iterate over all the 'server' config items
1650  */
1651  for (i = 0; i < talloc_array_length(inst->handle_config.server_str); i++) {
1652  char const *value = inst->handle_config.server_str[i];
1653  size_t j;
1654 
1655  /*
1656  * Explicitly prevent multiple server definitions
1657  * being used in the same string.
1658  */
1659  for (j = 0; j < talloc_array_length(value) - 1; j++) {
1660  switch (value[j]) {
1661  case ' ':
1662  case ',':
1663  case ';':
1664  cf_log_err(conf, "Invalid character '%c' found in 'server' configuration item",
1665  value[j]);
1666  goto error;
1667 
1668  default:
1669  continue;
1670  }
1671  }
1672 
1673 #ifdef LDAP_CAN_PARSE_URLS
1674  /*
1675  * Split original server value out into URI, server and port
1676  * so whatever initialization function we use later will have
1677  * the server information in the format it needs.
1678  */
1679  if (ldap_is_ldap_url(value)) {
1680  LDAPURLDesc *ldap_url;
1681  bool set_port_maybe = true;
1682  int default_port = LDAP_PORT;
1683  char *p;
1684 
1685  if (ldap_url_parse(value, &ldap_url)){
1686  cf_log_err(conf, "Parsing LDAP URL \"%s\" failed", value);
1687  ldap_url_error:
1688  ldap_free_urldesc(ldap_url);
1689  return -1;
1690  }
1691 
1692  if (ldap_url->lud_dn && (ldap_url->lud_dn[0] != '\0')) {
1693  cf_log_err(conf, "Base DN cannot be specified via server URL");
1694  goto ldap_url_error;
1695  }
1696 
1697  if (ldap_url->lud_attrs && ldap_url->lud_attrs[0]) {
1698  cf_log_err(conf, "Attribute list cannot be specified via server URL");
1699  goto ldap_url_error;
1700  }
1701 
1702  /*
1703  * ldap_url_parse sets this to base by default.
1704  */
1705  if (ldap_url->lud_scope != LDAP_SCOPE_BASE) {
1706  cf_log_err(conf, "Scope cannot be specified via server URL");
1707  goto ldap_url_error;
1708  }
1709  ldap_url->lud_scope = -1; /* Otherwise LDAP adds ?base */
1710 
1711  /*
1712  * The public ldap_url_parse function sets the default
1713  * port, so we have to discover whether a port was
1714  * included ourselves.
1715  */
1716  if ((p = strchr(value, ']')) && (p[1] == ':')) { /* IPv6 */
1717  set_port_maybe = false;
1718  } else if ((p = strchr(value, ':')) && (strchr(p + 1, ':') != NULL)) { /* IPv4 */
1719  set_port_maybe = false;
1720  }
1721 
1722  /* We allow extensions */
1723 
1724 # ifdef HAVE_LDAP_INITIALIZE
1725  {
1726  char *url;
1727 
1728  /*
1729  * Figure out the default port from the URL
1730  */
1731  if (ldap_url->lud_scheme) {
1732  if (strcmp(ldap_url->lud_scheme, "ldaps") == 0) {
1733  if (inst->handle_config.start_tls == true) {
1734  cf_log_err(conf, "ldaps:// scheme is not compatible "
1735  "with 'start_tls'");
1736  goto ldap_url_error;
1737  }
1738  default_port = LDAPS_PORT;
1739 
1740  } else if (strcmp(ldap_url->lud_scheme, "ldapi") == 0) {
1741  set_port_maybe = false; /* Unix socket, no port */
1742  }
1743  }
1744 
1745  if (set_port_maybe) {
1746  /*
1747  * URL port overrides configured port.
1748  */
1749  ldap_url->lud_port = inst->handle_config.port;
1750 
1751  /*
1752  * If there's no URL port, then set it to the default
1753  * this is so debugging messages show explicitly
1754  * the port we're connecting to.
1755  */
1756  if (!ldap_url->lud_port) ldap_url->lud_port = default_port;
1757  }
1758 
1759  url = ldap_url_desc2str(ldap_url);
1760  if (!url) {
1761  cf_log_err(conf, "Failed recombining URL components");
1762  goto ldap_url_error;
1763  }
1764  inst->handle_config.server = talloc_asprintf_append(inst->handle_config.server,
1765  "%s ", url);
1766  free(url);
1767  }
1768 # else
1769  /*
1770  * No LDAP initialize function. Can't specify a scheme.
1771  */
1772  if (ldap_url->lud_scheme &&
1773  ((strcmp(ldap_url->lud_scheme, "ldaps") == 0) ||
1774  (strcmp(ldap_url->lud_scheme, "ldapi") == 0) ||
1775  (strcmp(ldap_url->lud_scheme, "cldap") == 0))) {
1776  cf_log_err(conf, "%s is not supported by linked libldap",
1777  ldap_url->lud_scheme);
1778  return -1;
1779  }
1780 
1781  /*
1782  * URL port over-rides the configured
1783  * port. But if there's no configured
1784  * port, we use the hard-coded default.
1785  */
1786  if (set_port_maybe) {
1787  ldap_url->lud_port = inst->handle_config.port;
1788  if (!ldap_url->lud_port) ldap_url->lud_port = default_port;
1789  }
1790 
1791  inst->handle_config.server = talloc_asprintf_append(inst->handle_config.server, "%s:%i ",
1792  ldap_url->lud_host ? ldap_url->lud_host :
1793  "localhost",
1794  ldap_url->lud_port);
1795 # endif
1796  /*
1797  * @todo We could set a few other top level
1798  * directives using the URL, like base_dn
1799  * and scope.
1800  */
1801  ldap_free_urldesc(ldap_url);
1802  /*
1803  * We need to construct an LDAP URI
1804  */
1805  } else
1806 #endif /* HAVE_LDAP_URL_PARSE && HAVE_LDAP_IS_LDAP_URL && LDAP_URL_DESC2STR */
1807  /*
1808  * If it's not an URL, or we don't have the functions necessary
1809  * to break apart the URL and recombine it, then just treat
1810  * server as a hostname.
1811  */
1812  {
1813 #ifdef HAVE_LDAP_INITIALIZE
1814  char const *p;
1815  char *q;
1816  int port = 0;
1817  size_t len;
1818 
1819  port = inst->handle_config.port;
1820 
1821  /*
1822  * We don't support URLs if the library didn't provide
1823  * URL parsing functions.
1824  */
1825  if (strchr(value, '/')) {
1826  bad_server_fmt:
1827 #ifdef LDAP_CAN_PARSE_URLS
1828  cf_log_err(conf, "Invalid 'server' entry, must be in format <server>[:<port>] or "
1829  "an ldap URI (ldap|cldap|ldaps|ldapi)://<server>:<port>");
1830 #else
1831  cf_log_err(conf, "Invalid 'server' entry, must be in format <server>[:<port>]");
1832 #endif
1833  return -1;
1834  }
1835 
1836  p = strrchr(value, ':');
1837  if (p) {
1838  port = (int)strtol((p + 1), &q, 10);
1839  if ((p == value) || ((p + 1) == q) || (*q != '\0')) goto bad_server_fmt;
1840  len = p - value;
1841  } else {
1842  len = strlen(value);
1843  }
1844  if (port == 0) port = LDAP_PORT;
1845 
1846  inst->handle_config.server = talloc_asprintf_append(inst->handle_config.server,
1847  "ldap://%.*s:%i ",
1848  (int) len, value, port);
1849 #else
1850  /*
1851  * ldap_init takes port, which can be overridden by :port so
