135 /** SIM AT on-the-wire format attribute sizes
136  *
137  * Holds the min/max sizes of all supported SIM AT attribute values as they
138  * would be found in a SIM AT packet.
139  *
140  * These sizes may be different than the sizes of INTERNAL formats, PRESENTATION
141  * formats and generic NETWORK formats.
142  */
143 size_t const fr_sim_attr_sizes[FR_TYPE_MAX + 1][2] = {
144  [FR_TYPE_INVALID] = {~0, 0},
145 
146  [FR_TYPE_STRING] = {0, ~0},
147  [FR_TYPE_OCTETS] = {0, ~0},
148 
149  [FR_TYPE_BOOL] = {2, 2},
150  [FR_TYPE_UINT8] = {1, 1},
151  [FR_TYPE_UINT16] = {2, 2},
152  [FR_TYPE_UINT32] = {4, 4},
153  [FR_TYPE_UINT64] = {8, 8},
154 
155  [FR_TYPE_TLV] = {2, ~0},
156 
157  [FR_TYPE_MAX] = {~0, 0} //!< Ensure array covers all types.
158 };

160 /** Return the on-the-wire length of an attribute value
161  *
162  * @param[in] vp to return the length of.
163  * @return the length of the attribute.
164  */
165 size_t fr_sim_attr_len(fr_pair_t const *vp)
166 {
167  switch (vp->vp_type) {
168  case FR_TYPE_VARIABLE_SIZE:
169  return vp->vp_length;
170 
171  default:
172  return fr_sim_attr_sizes[vp->vp_type][0];
173 
174  case FR_TYPE_STRUCTURAL:
175  fr_assert_fail(NULL);
176  return 0;
177  }
178 }

180 /** Return the number of bytes before the octets value
181  *
182  */
183 size_t fr_sim_octets_prefix_len(fr_dict_attr_t const *da)
184 {
185  if (da->flags.array) return 0; /* Array elements have no padding */
186  if (!da->flags.length) return 2; /* Variable length attributes need length field */
187  if (!(da->flags.length % 4)) return 2; /* Values that are multiples of four have 2 reserved bytes */
188  return 0; /* Everything else has zero padding bytes */
189 }

191 int fr_sim_init(void)
192 {
193  if (instance_count > 0) {
194  instance_count++;
195  return 0;
196  }
197 
198  if (fr_dict_autoload(libfreeradius_sim_dict) < 0) {
199  PERROR("Failed loading libfreeradius-sim dictionaries");
200  return -1;
201  }
202  if (fr_dict_attr_autoload(libfreeradius_sim_dict_attr) < 0) {
203  PERROR("Failed loading libfreeradius-sim attributes");
204  fr_dict_autofree(libfreeradius_sim_dict);
205  return -1;
206  }
207  instance_count++;
208 
209  return 0;
210 }
211 
212 void fr_sim_free(void)
213 {
214  if (--instance_count > 0) return;
215 
216  fr_dict_autofree(libfreeradius_sim_dict);
217 }
