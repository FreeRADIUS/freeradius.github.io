125 static void authentication_failed(request_t *request, char const *msg)
126 {
127  fr_pair_t *vp;
128 
129  RPEDEBUG("%s", msg);
130 
131  /*
132  * Set the server reply message. Note that we do not tell the user *why* they failed authentication.
133  */
134  if (!fr_pair_find_by_da(&request->reply_pairs, attr_tacacs_server_message)) {
135  MEM(pair_update_reply(&vp, attr_tacacs_server_message) >= 0);
136  fr_pair_value_strdup(vp, "Authentication failed");
137  }
138 
139  /*
140  * Set the status.
141  */
142  MEM(pair_update_reply(&vp, attr_tacacs_authentication_status) >= 0);
143  vp->vp_uint8 = FR_TAC_PLUS_AUTHEN_STATUS_FAIL;
144 }

147 static unlang_action_t mod_process(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
148 {
149  proto_tacacs_auth_t const *inst = talloc_get_type_abort_const(mctx->instance, proto_tacacs_auth_t);
150  fr_pair_t *vp;
151  fr_pair_t *auth_type;
152  rlm_rcode_t rcode;
153  CONF_SECTION *unlang;
154  fr_dict_enum_t const *dv = NULL;
155  fr_cursor_t cursor;
156  fr_tacacs_auth_request_ctx_t const *auth_ctx;
157  fr_tacacs_packet_hdr_t const *pkt = (fr_tacacs_packet_hdr_t const *) request->packet->data;
158 
159  REQUEST_VERIFY(request);
160 
161  /*
162  * Wrapper to distinguish between Authentication Start & Continue.
163  */
164  if (request->packet->code == FR_PACKET_TYPE_VALUE_AUTHENTICATION_START) {
165  auth_ctx = &inst->start;
166  } else {
167  auth_ctx = &inst->cont;
168  }
169 
170  switch (request->request_state) {
171  case REQUEST_INIT:
172  request->component = "tacacs";
173 
174  /*
175  * We always reply, unless specifically set to "Do not respond"
176  */
177  request->reply->code = FR_PACKET_TYPE_VALUE_AUTHENTICATION_REPLY;
178 
179  /*
180  * Grab the VPS and data associated with the
181  * TACACS-State attribute. This is a synthetic /
182  * internal attribute, which is composed of the
183  * listener followed by the session ID
184  *
185  * The session ID is unique per listener.
186  */
187  if (!request->parent) {
188  uint8_t buffer[sizeof(request->async->listen) + sizeof(pkt->session_id)];
189 
190  fr_assert(request->async->listen);
191  memcpy(buffer, &request->async->listen, sizeof(request->async->listen));
192  memcpy(buffer + sizeof(request->async->listen), &pkt->session_id, sizeof(pkt->session_id));
193 
194  vp = fr_pair_afrom_da(request->packet, attr_tacacs_state);
195  if (vp) {
196  fr_pair_value_memdup(vp, buffer, sizeof(buffer), false);
197  fr_pair_add(&request->request_pairs, vp);
198  }
199 
200  fr_state_to_request(inst->state_tree, request);
201  }
202 
203  /*
204  * Push the conf section into the unlang stack.
205  */
206  RDEBUG("Running 'recv %s' from file %s", cf_section_name2(auth_ctx->recv_request), cf_filename(auth_ctx->recv_request));
207  if (unlang_interpret_push_instruction(request, auth_ctx->unlang_request, RLM_MODULE_REJECT, UNLANG_TOP_FRAME) < 0) {
208  RETURN_MODULE_FAIL;
209  }
210 
211  request->request_state = REQUEST_RECV;
212  FALL_THROUGH;
213 
214  case REQUEST_RECV:
215  rcode = unlang_interpret(request);
216 
217  if (request->master_state == REQUEST_STOP_PROCESSING) {
218  *p_result = RLM_MODULE_HANDLED;
219  return UNLANG_ACTION_STOP_PROCESSING;
220  }
221 
222  if (rcode == RLM_MODULE_YIELD) return UNLANG_ACTION_YIELD;
223 
224  switch (rcode) {
225  case RLM_MODULE_NOOP:
226  case RLM_MODULE_NOTFOUND:
227  case RLM_MODULE_OK:
228  case RLM_MODULE_UPDATED:
229  break;
230 
231  case RLM_MODULE_HANDLED:
232  goto setup_send;
233 
234  case RLM_MODULE_FAIL:
235  case RLM_MODULE_INVALID:
236  case RLM_MODULE_REJECT:
237  case RLM_MODULE_DISALLOW:
238  default:
239  authentication_failed(request, "Failed to authenticate the user");
240  goto setup_send;
241  }
242 
243  /*
244  * Find TACACS-Authentication-Type, and complain if they have too many.
245  */
246  auth_type = NULL;
247  for (vp = fr_cursor_iter_by_da_init(&cursor, &request->control_pairs, attr_auth_type);
248  vp;
249  vp = fr_cursor_next(&cursor)) {
250  if (!auth_type) {
251  auth_type = vp;
252  continue;
253  }
254 
255  RWDEBUG("Ignoring extra %pP", vp);
256  }
257 
258  /*
259  * No Auth-Type, force it to reject.
260  */
261  if (!auth_type) {
262  vp = fr_pair_find_by_da(&request->request_pairs, attr_tacacs_authentication_type);
263  if (!vp) {
264  authentication_failed(request, "No Auth-Type or TACACS-Authentication-Type configured: rejecting authentication.");
265  goto setup_send;
266  }
267 
268  /*
269  * Look up the name of TACACS-Authentication-Type
270  */
271  dv = fr_dict_enum_by_value(vp->da, &vp->data);
272  if (!dv) {
273  authentication_failed(request, "Unknown value for TACACS-Authentication-Type: rejecting authentication.");
274  goto setup_send;
275  }
276 
277  /*
278  * Use that name to search for a matching Auth-Type which has been defined.
279  */
280  dv = fr_dict_enum_by_name(attr_auth_type, dv->name, -1);
281  if (!dv) {
282  authentication_failed(request, "No Auth-Type found to match TACACS-Authentication-Type: rejecting authentication.");
283  goto setup_send;
284  }
285  } else {
286  /*
287  * Find the appropriate Auth-Type by name.
288  */
289  vp = auth_type;
290  dv = fr_dict_enum_by_value(attr_auth_type, &vp->data);
291  }
292  if (!dv) {
293  authentication_failed(request, "Unknown Auth-Type found: rejecting the user");
294  goto setup_send;
295  }
296 
297  unlang = cf_section_find(request->server_cs, "authenticate", dv->name);
298  if (!unlang) {
299  authentication_failed(request, "No matching 'authenticate' section found: rejecting the user");
300  goto setup_send;
301  }
302 
303  RDEBUG("Running 'authenticate %s' from file %s", cf_section_name2(unlang), cf_filename(unlang));
304  if (unlang_interpret_push_section(request, unlang, RLM_MODULE_NOTFOUND, UNLANG_TOP_FRAME) < 0) {
305  RETURN_MODULE_FAIL;
306  }
307 
308  request->request_state = REQUEST_PROCESS;
309  FALL_THROUGH;
310 
311  case REQUEST_PROCESS:
312  rcode = unlang_interpret(request);
313 
314  if (request->master_state == REQUEST_STOP_PROCESSING) {
315  *p_result = RLM_MODULE_HANDLED;
316  return UNLANG_ACTION_STOP_PROCESSING;
317  }
318 
319  if (rcode == RLM_MODULE_YIELD) return UNLANG_ACTION_YIELD;
320 
321  switch (rcode) {
322  /*
323  * An authentication module FAIL
324  * return code, or any return code that
325  * is not expected from authentication,
326  * is the same as an explicit REJECT!
327  */
328  case RLM_MODULE_FAIL:
329  case RLM_MODULE_INVALID:
330  case RLM_MODULE_NOOP:
331  case RLM_MODULE_NOTFOUND:
332  case RLM_MODULE_REJECT:
333  case RLM_MODULE_UPDATED:
334  case RLM_MODULE_DISALLOW:
335  default:
336  authentication_failed(request, "Failed to authenticate the user");
337  goto setup_send;
338 
339  case RLM_MODULE_OK:
340  break;
341 
342  case RLM_MODULE_HANDLED:
343  goto setup_send;
344  }
345 
346  setup_send:
347  RDEBUG("Running 'send %s' from file %s", cf_section_name2(auth_ctx->send_reply), cf_filename(auth_ctx->send_reply));
348  if (unlang_interpret_push_instruction(request, auth_ctx->unlang_reply, RLM_MODULE_NOOP, UNLANG_TOP_FRAME) < 0) {
349  RETURN_MODULE_FAIL;
350  }
351 
352  request->request_state = REQUEST_SEND;
353  FALL_THROUGH;
354 
355  case REQUEST_SEND:
356  rcode = unlang_interpret(request);
357 
358  if (request->master_state == REQUEST_STOP_PROCESSING) {
359  *p_result = RLM_MODULE_HANDLED;
360  return UNLANG_ACTION_STOP_PROCESSING;
361  }
362 
363  if (rcode == RLM_MODULE_YIELD) return UNLANG_ACTION_YIELD;
364 
365  switch (rcode) {
366  case RLM_MODULE_FAIL:
367  case RLM_MODULE_INVALID:
368  case RLM_MODULE_REJECT:
369  case RLM_MODULE_DISALLOW:
370  default:
371  break;
372 
373  case RLM_MODULE_HANDLED:
374  case RLM_MODULE_NOOP:
375  case RLM_MODULE_NOTFOUND:
376  case RLM_MODULE_OK:
377  case RLM_MODULE_UPDATED:
378  /* reply code is already set */
379  break;
380  }
381 
382  request->reply->timestamp = fr_time();
383 
384  /*
385  * Save session-state list on Start discard it
386  * for "Continue".
387  */
388  if (!request->parent) {
389  /*
390  * Keep the state around for
391  * authorization and accounting packets.
392  */
393  if (pkt->seq_no >= 254) {
394  fr_state_discard(inst->state_tree, request);
395 
396  /*
397  * We can't create a valid response
398  */
399  } else if (fr_request_to_state(inst->state_tree, request) < 0) {
400  RWDEBUG("Failed saving state");
401  request->reply->code = FR_PACKET_TYPE_VALUE_DO_NOT_RESPOND;
402  }
403  }
404 
405  /*
406  * Check for "do not respond".
407  */
408  if (request->reply->code == FR_PACKET_TYPE_VALUE_DO_NOT_RESPOND) {
409  RDEBUG("Not sending reply to client.");
410  fr_state_discard(inst->state_tree, request);
411  break;
412  }
413  break;
414 
415  default:
416  RETURN_MODULE_FAIL;
417  }
418 
419  RETURN_MODULE_OK;
420 }

470 static int mod_instantiate(void *instance, UNUSED CONF_SECTION *process_app_cs)
471 {
472  proto_tacacs_auth_t *inst = instance;
473 
474  /*
475  * Usually we use the 'State' attribute. But, in this
476  * case we are using the listener followed by the
477  * TACACS-Session-ID as the state id. It is 32-bits of
478  * (allegedly) random value. It MUST be unique per TCP
479  * connection.
480  */
481  inst->state_tree = fr_state_tree_init(inst, attr_tacacs_state, main_config->spawn_workers, inst->max_session,
482  inst->session_timeout, inst->state_server_id);
483 
484  return 0;
485 }
