107 /*
108  * Compare prefix/suffix.
109  *
110  * If they compare:
111  * - if FR_STRIP_USER_NAME is present in check_list,
112  * strip the username of prefix/suffix.
113  * - if FR_STRIP_USER_NAME is not present in check_list,
114  * add a FR_STRIPPED_USER_NAME to the request.
115  */
116 static int prefix_suffix_cmp(UNUSED void *instance,
117  request_t *request,
118  fr_pair_t *req,
119  fr_pair_t *check,
120  fr_pair_t *check_list)
121 {
122  fr_pair_t *vp;
123  char const *name;
124  char rest[FR_MAX_STRING_LEN];
125  int len, namelen;
126  int ret = -1;
127  fr_pair_t *username;
128 
129  if (!request) return -1;
130 
131  username = fr_pair_find_by_da(&request->request_pairs, attr_stripped_user_name);
132  if (!username) username = fr_pair_find_by_da(&request->request_pairs, attr_user_name);
133  if (!username) return -1;
134 
135  VP_VERIFY(check);
136 
137  name = username->vp_strvalue;
138 
139  RDEBUG3("Comparing name \"%s\" and check value \"%pV\"", name, &check->data);
140 
141  len = strlen(check->vp_strvalue);
142 
143  if (check->da == attr_prefix) {
144  ret = strncmp(name, check->vp_strvalue, len);
145  if (ret == 0)
146  strlcpy(rest, name + len, sizeof(rest));
147  } else if (check->da == attr_suffix) {
148  namelen = strlen(name);
149  if (namelen >= len) {
150  ret = strcmp(name + namelen - len, check->vp_strvalue);
151  if (ret == 0) strlcpy(rest, name, namelen - len + 1);
152  }
153  }
154 
155  if (ret != 0) return ret;
156 
157  /*
158  * If Strip-User-Name == No, then don't do any more.
159  */
160  vp = fr_pair_find_by_da(&check_list, attr_strip_user_name);
161  if (vp && !vp->vp_uint32) return ret;
162 
163  /*
164  * See where to put the stripped user name.
165  */
166  vp = fr_pair_find_by_da(&check_list, attr_stripped_user_name);
167  if (!vp) {
168  /*
169  * If "request" is NULL, then the memory will be
170  * lost!
171  */
172  MEM(vp = fr_pair_afrom_da(request->packet, attr_stripped_user_name));
173  fr_pair_add(&req, vp);
174  }
175  fr_pair_value_strdup(vp, rest);
176 
177  return ret;
178 }

197 /*
198  * Generic comparisons, via xlat.
199  */
200 static int generic_cmp(UNUSED void *instance,
201  request_t *request,
202  fr_pair_t *req,
203  fr_pair_t *check,
204  UNUSED fr_pair_t *check_list)
205 {
206  VP_VERIFY(check);
207 
208  if ((check->op != T_OP_REG_EQ) && (check->op != T_OP_REG_NE)) {
209  int rcode;
210  char name[1024];
211  char value[1024];
212  fr_pair_t *vp;
213 
214  snprintf(name, sizeof(name), "%%{%s}", check->da->name);
215 
216  if (xlat_eval(value, sizeof(value), request, name, NULL, NULL) < 0) return 0;
217 
218  MEM(vp = fr_pair_afrom_da(req, check->da));
219  vp->op = check->op;
220  fr_pair_value_from_str(vp, value, -1, '"', false);
221 
222  /*
223  * Paircmp returns 0 for failed comparison, 1 for succeeded -1 for error.
224  */
225  rcode = fr_pair_cmp(check, vp);
226 
227  /*
228  * We're being called from paircmp_func,
229  * which wants 0 for success, and 1 for fail (sigh)
230  *
231  * We should really fix the API so that it is
232  * consistent. i.e. the comparison callbacks should
233  * return ONLY the resut of comparing A to B.
234  * The radius_callback_cmp function should then
235  * take care of using the operator to see if the
236  * condition (A OP B) is true or not.
237  *
238  * This would also allow "<", etc. to work in the
239  * callback functions...
240  *
241  * See rlm_ldap, ...groupcmp() for something that
242  * returns 0 for matched, and 1 for didn't match.
243  */
244  rcode = !rcode;
245  fr_pair_list_free(&vp);
246 
247  return rcode;
248  }
249 
250  /*
251  * Will do the xlat for us
252  */
253  return paircmp_pairs(request, check, NULL);
254 }

281 /** Compares check and vp by value.
282  *
283  * Does not call any per-attribute comparison function, but does honour
284  * check.operator. Basically does "vp.value check.op check.value".
285  *
286  * @param[in] request Current request.
287  * @param[in] check rvalue, and operator.
288  * @param[in] vp lvalue.
289  * @return
290  * - 0 if check and vp are equal
291  * - -1 if vp value is less than check value.
292  * - 1 is vp value is more than check value.
293  * - -2 on error.
294  */
295 #ifdef HAVE_REGEX
296 int paircmp_pairs(request_t *request, fr_pair_t *check, fr_pair_t *vp)
297 #else
298 int paircmp_pairs(UNUSED request_t *request, fr_pair_t *check, fr_pair_t *vp)
299 #endif
300 {
301  int ret = 0;
302 
303  /*
304  * Check for =* and !* and return appropriately
305  */
306  if (check->op == T_OP_CMP_TRUE) return 0;
307  if (check->op == T_OP_CMP_FALSE) return 1;
308 
309  if (!vp) {
310  REDEBUG("Non-Unary operations require two operands");
311  return -2;
312  }
313 
314 #ifdef HAVE_REGEX
315  if ((check->op == T_OP_REG_EQ) || (check->op == T_OP_REG_NE)) {
316  ssize_t slen;
317  regex_t *preg = NULL;
318  uint32_t subcaptures;
319  fr_regmatch_t *regmatch;
320 
321  char *expr = NULL, *value = NULL;
322  char const *expr_p, *value_p;
323 
324  if (check->vp_type == FR_TYPE_STRING) {
325  expr_p = check->vp_strvalue;
326  } else {
327  fr_value_box_aprint(check, &expr, &check->data, NULL);
328  expr_p = expr;
329  }
330 
331  if (vp->vp_type == FR_TYPE_STRING) {
332  value_p = vp->vp_strvalue;
333  } else {
334  fr_value_box_aprint(vp, &value, &vp->data, NULL);
335  value_p = value;
336  }
337 
338  if (!expr_p || !value_p) {
339  REDEBUG("Error stringifying operand for regular expression");
340 
341  regex_error:
342  talloc_free(preg);
343  talloc_free(expr);
344  talloc_free(value);
345  return -2;
346  }
347 
348  /*
349  * Include substring matches.
350  */
351  slen = regex_compile(request, &preg, expr_p, talloc_array_length(expr_p) - 1,
352  NULL, true, true);
353  if (slen <= 0) {
354  REMARKER(expr_p, -slen, "%s", fr_strerror());
355 
356  goto regex_error;
357  }
358 
359  subcaptures = regex_subcapture_count(preg);
360  if (!subcaptures) subcaptures = REQUEST_MAX_REGEX + 1; /* +1 for %{0} (whole match) capture group */
361  MEM(regmatch = regex_match_data_alloc(NULL, subcaptures));
362 
363  /*
364  * Evaluate the expression
365  */
366  slen = regex_exec(preg, value_p, talloc_array_length(value_p) - 1, regmatch);
367  if (slen < 0) {
368  RPERROR("Invalid regex");
369 
370  goto regex_error;
371  }
372 
373  if (check->op == T_OP_REG_EQ) {
374  /*
375  * Add in %{0}. %{1}, etc.
376  */
377  regex_sub_to_request(request, &preg, &regmatch);
378  ret = (slen == 1) ? 0 : -1;
379  } else {
380  ret = (slen != 1) ? 0 : -1;
381  }
382 
383  talloc_free(regmatch);
384  talloc_free(preg);
385  talloc_free(expr);
386  talloc_free(value);
387 
388  goto finish;
389  }
390 #endif
391 
392  /*
393  * Attributes must be of the same type.
394  *
395  * FIXME: deal with type mismatch properly if one side contain
396  * OCTETS or STRING by converting the other side to
397  * a string
398  *
399  */
400  if (vp->vp_type != check->vp_type) return -1;
401 
402  /*
403  * Not a regular expression, compare the types.
404  */
405  switch (check->vp_type) {
406  case FR_TYPE_OCTETS:
407  if (vp->vp_length != check->vp_length) {
408  ret = 1; /* NOT equal */
409  break;
410  }
411  ret = memcmp(vp->vp_strvalue, check->vp_strvalue, vp->vp_length);
412  break;
413 
414  case FR_TYPE_STRING:
415  ret = strcmp(vp->vp_strvalue, check->vp_strvalue);
416  break;
417 
418  case FR_TYPE_UINT8:
419  ret = vp->vp_uint8 - check->vp_uint8;
420  break;
421 
422  case FR_TYPE_UINT16:
423  ret = vp->vp_uint16 - check->vp_uint16;
424  break;
425 
426  case FR_TYPE_UINT32:
427  ret = vp->vp_uint32 - check->vp_uint32;
428  break;
429 
430  case FR_TYPE_UINT64:
431  /*
432  * Don't want integer overflow!
433  */
434  if (vp->vp_uint64 < check->vp_uint64) {
435  ret = -1;
436  } else if (vp->vp_uint64 > check->vp_uint64) {
437  ret = +1;
438  } else {
439  ret = 0;
440  }
441  break;
442 
443  case FR_TYPE_INT32:
444  if (vp->vp_int32 < check->vp_int32) {
445  ret = -1;
446  } else if (vp->vp_int32 > check->vp_int32) {
447  ret = +1;
448  } else {
449  ret = 0;
450  }
451  break;
452 
453  case FR_TYPE_DATE:
454  ret = vp->vp_date - check->vp_date;
455  break;
456 
457  case FR_TYPE_IPV4_ADDR:
458  ret = ntohl(vp->vp_ipv4addr) - ntohl(check->vp_ipv4addr);
459  break;
460 
461  case FR_TYPE_IPV6_ADDR:
462  ret = memcmp(vp->vp_ip.addr.v6.s6_addr, check->vp_ip.addr.v6.s6_addr,
463  sizeof(vp->vp_ip.addr.v6.s6_addr));
464  break;
465 
466  case FR_TYPE_IPV4_PREFIX:
467  case FR_TYPE_IPV6_PREFIX:
468  ret = fr_pair_cmp_op(check->op, vp, check);
469  if (ret == -1) return -2; // error
470  if (check->op == T_OP_LT || check->op == T_OP_LE)
471  ret = (ret == 1) ? -1 : 1;
472  else if (check->op == T_OP_GT || check->op == T_OP_GE)
473  ret = (ret == 1) ? 1 : -1;
474  else if (check->op == T_OP_CMP_EQ)
475  ret = (ret == 1) ? 0 : -1;
476  break;
477 
478  case FR_TYPE_IFID:
479  ret = memcmp(vp->vp_ifid, check->vp_ifid, sizeof(vp->vp_ifid));
480  break;
481 
482  default:
483  break;
484  }
485 
486 finish:
487  if (ret > 0) return 1;
488  if (ret < 0) return -1;
489  return 0;
490 }

492 /** Compare check and vp. May call the attribute comparison function.
493  *
494  * Unlike paircmp_pairs() this function will call any attribute-specific
495  * comparison functions registered.
496  *
497  * @param[in] request Current request.
498  * @param[in] request_list list pairs.
499  * @param[in] check item to compare.
500  * @param[in] check_list list.
501  * @return
502  * - 0 if check and vp are equal.
503  * - -1 if vp value is less than check value.
504  * - 1 is vp value is more than check value.
505  */
506 static int paircmp_func(request_t *request,
507  fr_pair_t *request_list,
508  fr_pair_t *check,
509  fr_pair_t *check_list)
510 {
511  paircmp_t *c;
512 
513  VP_VERIFY(check);
514 
515  /*
516  * Check for =* and !* and return appropriately
517  */
518  if (check->op == T_OP_CMP_TRUE) return 0;
519  if (check->op == T_OP_CMP_FALSE) return 1;
520 
521  /*
522  * See if there is a special compare function.
523  *
524  * FIXME: use new RB-Tree code.
525  */
526  for (c = cmp; c; c = c->next) {
527  if (c->da == check->da) {
528  return (c->compare)(c->instance, request, request_list, check, check_list);
529  }
530  }
531 
532  if (!request) return -1; /* doesn't exist, don't compare it */
533 
534  return paircmp_pairs(request, check, request_list);
535 }

537 /** Compare two pair lists except for the password information.
538  *
539  * For every element in "check" at least one matching copy must be present
540  * in "reply".
541  *
542  * @param[in] request Current request.
543  * @param[in] request_list request valuepairs.
544  * @param[in] check Check/control valuepairs.
545  * @return 0 on match.
546  */
547 int paircmp(request_t *request,
548  fr_pair_t *request_list,
549  fr_pair_t *check)
550 {
551  fr_cursor_t cursor;
552  fr_pair_t *check_item;
553  fr_pair_t *auth_item;
554  fr_dict_attr_t const *from;
555 
556  int result = 0;
557  int compare;
558  bool first_only;
559 
560  for (check_item = fr_cursor_init(&cursor, &check);
561  check_item;
562  check_item = fr_cursor_next(&cursor)) {
563  /*
564  * If the user is setting a configuration value,
565  * then don't bother comparing it to any attributes
566  * sent to us by the user. It ALWAYS matches.
567  */
568  if ((check_item->op == T_OP_SET) ||
569  (check_item->op == T_OP_ADD)) {
570  continue;
571  }
572 
573  /*
574  * Attributes we skip during comparison.
575  * These are "server" check items.
576  */
577  if ((check_item->da == attr_crypt_password) ||
578  (check_item->da == attr_auth_type) ||
579  (check_item->da == attr_strip_user_name)) {
580  continue;
581  }
582 
583  /*
584  * IF the password attribute exists, THEN
585  * we can do comparisons against it. If not,
586  * then the request did NOT contain a
587  * User-Password attribute, so we CANNOT do
588  * comparisons against it.
589  *
590  * This hack makes CHAP-Password work..
591  */
592  if (check_item->da == attr_user_password) {
593  if (check_item->op == T_OP_CMP_EQ) {
594  WARN("Found User-Password == \"...\"");
595  WARN("Are you sure you don't mean Cleartext-Password?");
596  WARN("See \"man rlm_pap\" for more information");
597  }
598  if (fr_pair_find_by_num(&request_list, 0, FR_USER_PASSWORD) == NULL) continue;
599  }
600 
601  /*
602  * See if this item is present in the request.
603  */
604  first_only = other_attr(check_item->da, &from);
605 
606  auth_item = request_list;
607 
608  try_again:
609  if (!first_only) {
610  while (auth_item != NULL) {
611  if ((auth_item->da == from) || (!from)) break;
612 
613  auth_item = auth_item->next;
614  }
615  }
616 
617  /*
618  * Not found, it's not a match.
619  */
620  if (auth_item == NULL) {
621  /*
622  * Didn't find it. If we were *trying*
623  * to not find it, then we succeeded.
624  */
625  if (check_item->op == T_OP_CMP_FALSE) {
626  continue;
627  } else {
628  return -1;
629  }
630  }
631 
632  /*
633  * Else we found it, but we were trying to not
634  * find it, so we failed.
635  */
636  if (check_item->op == T_OP_CMP_FALSE) return -1;
637 
638  /*
639  * We've got to xlat the string before doing
640  * the comparison.
641  */
642  xlat_eval_pair(request, check_item);
643 
644  /*
645  * OK it is present now compare them.
646  */
647  compare = paircmp_func(request, auth_item, check_item, check);
648  switch (check_item->op) {
649  case T_OP_EQ:
650  default:
651  RWDEBUG("Invalid operator '%s' for item %s: reverting to '=='",
652  fr_table_str_by_value(fr_tokens_table, check_item->op, "<INVALID>"), check_item->da->name);
653  FALL_THROUGH;
654  case T_OP_CMP_TRUE:
655  case T_OP_CMP_FALSE:
656  case T_OP_CMP_EQ:
657  if (compare != 0) result = -1;
658  break;
659 
660  case T_OP_NE:
661  if (compare == 0) result = -1;
662  break;
663 
664  case T_OP_LT:
665  if (compare >= 0) result = -1;
666  break;
667 
668  case T_OP_GT:
669  if (compare <= 0) result = -1;
670  break;
671 
672  case T_OP_LE:
673  if (compare > 0) result = -1;
674  break;
675 
676  case T_OP_GE:
677  if (compare < 0) result = -1;
678  break;
679 
680 #ifdef HAVE_REGEX
681  case T_OP_REG_EQ:
682  case T_OP_REG_NE:
683  if (compare != 0) result = -1;
684  break;
685 #endif
686  } /* switch over the operator of the check item */
687 
688  /*
689  * This attribute didn't match, but maybe there's
690  * another of the same attribute, which DOES match.
691  */
692  if ((result != 0) && (!first_only)) {
693  auth_item = auth_item->next;
694  result = 0;
695  goto try_again;
696  }
697 
698  } /* for every entry in the check item list */
699 
700  return result;
701 }
