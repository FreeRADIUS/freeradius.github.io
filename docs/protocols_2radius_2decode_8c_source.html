35 static void memcpy_bounded(void * restrict dst, const void * restrict src, size_t n, const void * restrict end)
36 {
37  size_t len = n;
38 
39  if (!fr_cond_assert(n <= 65535)) {
40  return;
41  }
42 
43  if (!fr_cond_assert(src <= end)) {
44  return;
45  }
46 
47  if (len == 0) return;
48 
49  if (!fr_cond_assert(((uint8_t const * restrict) src + len) <= (uint8_t const * restrict) end)) {
50  len = (uint8_t const * restrict) end - (uint8_t const * restrict) src;
51  }
52 
53  memcpy(dst, src, len);
54 }

57 /** Decode Tunnel-Password encrypted attributes
58  *
59  * Defined in RFC-2868, this uses a two char SALT along with the
60  * initial intermediate value, to differentiate it from the
61  * above.
62  */
63 ssize_t fr_radius_decode_tunnel_password(uint8_t *passwd, size_t *pwlen,
64  char const *secret, uint8_t const *vector, bool tunnel_password_zeros)
65 {
66  fr_md5_ctx_t *md5_ctx, *md5_ctx_old;
67  uint8_t digest[RADIUS_AUTH_VECTOR_LENGTH];
68  int secretlen;
69  size_t i, n, encrypted_len, embedded_len;
70 
71  encrypted_len = *pwlen;
72 
73  /*
74  * We need at least a salt.
75  */
76  if (encrypted_len < 2) {
77  fr_strerror_printf("Tunnel password is too short");
78  return -1;
79  }
80 
81  /*
82  * There's a salt, but no password. Or, there's a salt
83  * and a 'data_len' octet. It's wrong, but at least we
84  * can figure out what it means: the password is empty.
85  *
86  * Note that this means we ignore the 'data_len' field,
87  * if the attribute length tells us that there's no
88  * more data. So the 'data_len' field may be wrong,
89  * but that's ok...
90  */
91  if (encrypted_len <= 3) {
92  passwd[0] = 0;
93  *pwlen = 0;
94  return 0;
95  }
96 
97  encrypted_len -= 2; /* discount the salt */
98 
99  /*
100  * Use the secret to setup the decryption digest
101  */
102  secretlen = talloc_array_length(secret) - 1;
103 
104  md5_ctx = fr_md5_ctx_alloc(false);
105  md5_ctx_old = fr_md5_ctx_alloc(true);
106 
107  fr_md5_update(md5_ctx, (uint8_t const *) secret, secretlen);
108  fr_md5_ctx_copy(md5_ctx_old, md5_ctx); /* save intermediate work */
109 
110  /*
111  * Set up the initial key:
112  *
113  * b(1) = MD5(secret + vector + salt)
114  */
115  fr_md5_update(md5_ctx, vector, RADIUS_AUTH_VECTOR_LENGTH);
116  fr_md5_update(md5_ctx, passwd, 2);
117 
118  embedded_len = 0;
119  for (n = 0; n < encrypted_len; n += AUTH_PASS_LEN) {
120  size_t base;
121  size_t block_len = AUTH_PASS_LEN;
122 
123  /*
124  * Ensure we don't overflow the input on MD5
125  */
126  if ((n + 2 + AUTH_PASS_LEN) > *pwlen) {
127  block_len = *pwlen - n - 2;
128  }
129 
130  if (n == 0) {
131  base = 1;
132 
133  fr_md5_final(digest, md5_ctx);
134  fr_md5_ctx_copy(md5_ctx, md5_ctx_old);
135 
136  /*
137  * A quick check: decrypt the first octet
138  * of the password, which is the
139  * 'data_len' field. Ensure it's sane.
140  */
141  embedded_len = passwd[2] ^ digest[0];
142  if (embedded_len > encrypted_len) {
143  fr_strerror_printf("Tunnel Password is too long for the attribute "
144  "(shared secret is probably incorrect!)");
145  fr_md5_ctx_free(&md5_ctx);
146  fr_md5_ctx_free(&md5_ctx_old);
147  return -1;
148  }
149 
150  fr_md5_update(md5_ctx, passwd + 2, block_len);
151 
152  } else {
153  base = 0;
154 
155  fr_md5_final(digest, md5_ctx);
156 
157  fr_md5_ctx_copy(md5_ctx, md5_ctx_old);
158  fr_md5_update(md5_ctx, passwd + n + 2, block_len);
159  }
160 
161  for (i = base; i < block_len; i++) {
162  passwd[n + i - 1] = passwd[n + i + 2] ^ digest[i];
163  }
164  }
165 
166  fr_md5_ctx_free(&md5_ctx);
167  fr_md5_ctx_free(&md5_ctx_old);
168 
169  /*
170  * Check trailing bytes
171  */
172  if (tunnel_password_zeros) for (i = embedded_len; i < (encrypted_len - 1); i++) { /* -1 for length field */
173  if (passwd[i] != 0) {
174  fr_strerror_printf("Trailing garbage in Tunnel Password "
175  "(shared secret is probably incorrect!)");
176 
177  return -1;
178  }
179  }
180 
181  *pwlen = embedded_len;
182 
183  passwd[embedded_len] = '\0';
184 
185  return embedded_len;
186 }

188 /** Decode password
189  *
190  */
191 ssize_t fr_radius_decode_password(char *passwd, size_t pwlen, char const *secret, uint8_t const *vector)
192 {
193  fr_md5_ctx_t *md5_ctx, *md5_ctx_old;
194  uint8_t digest[RADIUS_AUTH_VECTOR_LENGTH];
195  int i;
196  size_t n, secretlen;
197 
198  /*
199  * The RFC's say that the maximum is 128.
200  * The buffer we're putting it into above is 254, so
201  * we don't need to do any length checking.
202  */
203  if (pwlen > 128) pwlen = 128;
204 
205  /*
206  * Catch idiots.
207  */
208  if (pwlen == 0) goto done;
209 
210  /*
211  * Use the secret to setup the decryption digest
212  */
213  secretlen = talloc_array_length(secret) - 1;
214 
215  md5_ctx = fr_md5_ctx_alloc(false);
216  md5_ctx_old = fr_md5_ctx_alloc(true);
217 
218  fr_md5_update(md5_ctx, (uint8_t const *) secret, secretlen);
219  fr_md5_ctx_copy(md5_ctx_old, md5_ctx); /* save intermediate work */
220 
221  /*
222  * The inverse of the code above.
223  */
224  for (n = 0; n < pwlen; n += AUTH_PASS_LEN) {
225  if (n == 0) {
226  fr_md5_update(md5_ctx, vector, RADIUS_AUTH_VECTOR_LENGTH);
227  fr_md5_final(digest, md5_ctx);
228 
229  fr_md5_ctx_copy(md5_ctx, md5_ctx_old);
230  if (pwlen > AUTH_PASS_LEN) {
231  fr_md5_update(md5_ctx, (uint8_t *) passwd, AUTH_PASS_LEN);
232  }
233  } else {
234  fr_md5_final(digest, md5_ctx);
235 
236  fr_md5_ctx_copy(md5_ctx, md5_ctx_old);
237  if (pwlen > (n + AUTH_PASS_LEN)) {
238  fr_md5_update(md5_ctx, (uint8_t *) passwd + n, AUTH_PASS_LEN);
239  }
240  }
241 
242  for (i = 0; i < AUTH_PASS_LEN; i++) passwd[i + n] ^= digest[i];
243  }
244 
245  fr_md5_ctx_free(&md5_ctx);
246  fr_md5_ctx_free(&md5_ctx_old);
247 
248  done:
249  passwd[pwlen] = '\0';
250  return strlen(passwd);
251 }

256 int fr_radius_decode_tlv_ok(uint8_t const *data, size_t length, size_t dv_type, size_t dv_length)
257 {
258  uint8_t const *end = data + length;
259 
260  FR_PROTO_TRACE("Checking TLV %u/%u", (unsigned int) dv_type, (unsigned int) dv_length);
261 
262  FR_PROTO_HEX_DUMP(data, length, "tlv_ok");
263 
264  if ((dv_length > 2) || (dv_type == 0) || (dv_type > 4)) {
265  fr_strerror_printf("%s: Invalid arguments", __FUNCTION__);
266  return -1;
267  }
268 
269  while (data < end) {
270  size_t attrlen;
271 
272  if ((data + dv_type + dv_length) > end) {
273  fr_strerror_printf("Attribute header overflow");
274  return -1;
275  }
276 
277  switch (dv_type) {
278  case 4:
279  if ((data[0] == 0) && (data[1] == 0) &&
280  (data[2] == 0) && (data[3] == 0)) {
281  zero:
282  fr_strerror_printf("Invalid attribute 0");
283  return -1;
284  }
285 
286  if (data[0] != 0) {
287  fr_strerror_printf("Invalid attribute > 2^24");
288  return -1;
289  }
290  break;
291 
292  case 2:
293  if ((data[0] == 0) && (data[1] == 0)) goto zero;
294  break;
295 
296  case 1:
297  /*
298  * Zero is allowed, because the Colubris
299  * people are dumb and use it.
300  */
301  break;
302 
303  default:
304  fr_strerror_printf("Internal sanity check failed");
305  return -1;
306  }
307 
308  switch (dv_length) {
309  case 0:
310  return 0;
311 
312  case 2:
313  if (data[dv_type] != 0) {
314  fr_strerror_printf("Attribute is longer than 256 octets");
315  return -1;
316  }
317  FALL_THROUGH;
318  case 1:
319  attrlen = data[dv_type + dv_length - 1];
320  break;
321 
322 
323  default:
324  fr_strerror_printf("Internal sanity check failed");
325  return -1;
326  }
327 
328  if (attrlen < (dv_type + dv_length)) {
329  fr_strerror_printf("Attribute header has invalid length");
330  return -1;
331  }
332 
333  if (attrlen > length) {
334  fr_strerror_printf("Attribute overflows container");
335  return -1;
336  }
337 
338  data += attrlen;
339  length -= attrlen;
340  }
341 
342  return 0;
343 }

345 /** Convert a "concatenated" attribute to one long VP
346  *
347  */
348 static ssize_t decode_concat(TALLOC_CTX *ctx, fr_cursor_t *cursor,
349  fr_dict_attr_t const *parent, uint8_t const *data,
350  size_t const packet_len)
351 {
352  size_t total;
353  uint8_t attr;
354  uint8_t const *ptr = data;
355  uint8_t const *end = data + packet_len;
356  uint8_t *p;
357  fr_pair_t *vp;
358 
359  total = 0;
360  attr = ptr[0];
361 
362  /*
363  * See how many consecutive attributes there are.
364  */
365  while (ptr < end) {
366  if (ptr[1] <= 2) return -1;
367  if ((ptr + ptr[1]) > end) return -1;
368 
369  total += ptr[1] - 2;
370 
371  ptr += ptr[1];
372 
373  if (ptr == end) break;
374 
375  /*
376  * Attributes MUST be consecutive.
377  */
378  if (ptr[0] != attr) break;
379  }
380 
381  /*
382  * Reset the end of the data we're trying to parse
383  */
384  end = ptr;
385 
386  /*
387  * If there's no data, just return that we skipped the
388  * attribute header.
389  */
390  if (!total) return 2;
391 
392  vp = fr_pair_afrom_da(ctx, parent);
393  if (!vp) return -1;
394 
395  if (fr_pair_value_mem_alloc(vp, &p, total, true) != 0) {
396  talloc_free(vp);
397  return -1;
398  }
399 
400  ptr = data;
401  while (ptr < end) {
402  memcpy_bounded(p, ptr + 2, ptr[1] - 2, end);
403  p += ptr[1] - 2;
404  ptr += ptr[1];
405  }
406  fr_cursor_append(cursor, vp);
407  return ptr - data;
408 }

565 /** Convert a fragmented extended attr to a VP
566  *
567  * Format is:
568  *
569  * attr
570  * length
571  * extended-attr
572  * flag
573  * data...
574  *
575  * But for the first fragment, we get passed a pointer to the "extended-attr"
576  */
577 static ssize_t decode_extended(TALLOC_CTX *ctx, fr_cursor_t *cursor, fr_dict_t const *dict,
578  fr_dict_attr_t const *parent,
579  uint8_t const *data, size_t attr_len, size_t packet_len,
580  void *decoder_ctx)
581 {
582  ssize_t ret;
583  size_t fraglen;
584  uint8_t *head, *tail;
585  uint8_t const *frag, *end;
586  uint8_t const *attr;
587  int fragments;
588  bool last_frag;
589 
590  /*
591  * data = Ext-Attr Flag ...
592  */
593 
594  if (attr_len < 3) return -1;
595 
596  /*
597  * No continuation, just decode the attributre in place.
598  */
599  if ((data[1] & 0x80) == 0) {
600  ret = fr_radius_decode_pair_value(ctx, cursor, dict,
601  parent, data + 2, attr_len - 2, attr_len - 2, decoder_ctx);
602  if (ret < 0) return -1;
603  return attr_len;
604  }
605 
606  /*
607  * Calculate the length of all of the fragments. For
608  * now, they MUST be contiguous in the packet, and they
609  * MUST be all of the same TYPE and EXTENDED-TYPE
610  */
611  attr = data - 2;
612  fraglen = attr_len - 2;
613  frag = data + attr_len;
614  end = data + packet_len;
615  fragments = 1;
616  last_frag = false;
617 
618  while (frag < end) {
619  if (last_frag ||
620  (frag[0] != attr[0]) ||
621  (frag[1] < 4) || /* too short for long-extended */
622  (frag[2] != attr[2]) ||
623  ((frag + frag[1]) > end)) { /* overflow */
624  end = frag;
625  break;
626  }
627 
628  last_frag = ((frag[3] & 0x80) == 0);
629 
630  fraglen += frag[1] - 4;
631  frag += frag[1];
632  fragments++;
633  }
634 
635  head = tail = talloc_array(ctx, uint8_t, fraglen);
636  if (!head) return -1;
637 
638  FR_PROTO_TRACE("Fragments %d, total length %d", fragments, (int) fraglen);
639 
640  /*
641  * And again, but faster and looser.
642  *
643  * We copy the first fragment, followed by the rest of
644  * the fragments.
645  */
646  frag = attr;
647 
648  while (fragments > 0) {
649  if (frag[1] > 4) memcpy_bounded(tail, frag + 4, frag[1] - 4, end);
650  tail += frag[1] - 4;
651  frag += frag[1];
652  fragments--;
653  }
654 
655  FR_PROTO_HEX_DUMP(head, fraglen, "long-extended fragments");
656 
657  ret = fr_radius_decode_pair_value(ctx, cursor, dict,
658  parent, head, fraglen, fraglen, decoder_ctx);
659  talloc_free(head);
660  if (ret < 0) return ret;
661 
662  return end - data;
663 }

711  if (ret < 0) return ret;
712 
713  return attr_len;
714  }
715 
716  /*
717  * Calculate the length of all of the fragments. For
718  * now, they MUST be contiguous in the packet, and they
719  * MUST be all of the same VSA, WiMAX, and WiMAX-attr.
720  *
721  * The first fragment doesn't have a RADIUS attribute
722  * header.
723  */
724  wimax_len = 0;
725  attr = data + 4;
726  end = data + packet_len;
727 
728  while (attr < end) {
729  /*
730  * Not enough room for Attribute + length +
731  * continuation, it's bad.
732  */
733  if ((end - attr) < 3) return -1;
734 
735  /*
736  * Must have non-zero data in the attribute.
737  */
738  if (attr[1] <= 3) return -1;
739 
740  /*
741  * If the WiMAX attribute overflows the packet,
742  * it's bad.
743  */
744  if ((attr + attr[1]) > end) return -1;
745 
746  /*
747  * Check the continuation flag.
748  */
749  more = ((attr[2] & 0x80) != 0);
750 
751  /*
752  * Or, there's no more data, in which case we
753  * shorten "end" to finish at this attribute.
754  */
755  if (!more) end = attr + attr[1];
756 
757  /*
758  * There's more data, but we're at the end of the
759  * packet. The attribute is malformed!
760  */
761  if (more && ((attr + attr[1]) == end)) return -1;
762 
763  /*
764  * Add in the length of the data we need to
765  * concatenate together.
766  */
767  wimax_len += attr[1] - 3;
768 
769  /*
770  * Go to the next attribute, and stop if there's
771  * no more.
772  */
773  attr += attr[1];
774  if (!more) break;
775 
776  /*
777  * data = VID VID VID VID WiMAX-Attr WimAX-Len Continuation ...
778  *
779  * attr = Vendor-Specific VSA-Length VID VID VID VID WiMAX-Attr WimAX-Len Continuation ...
780  *
781  */
782 
783  /*
784  * No room for Vendor-Specific + length +
785  * Vendor(4) + attr + length + continuation + data
786  */
787  if ((end - attr) < 9) return -1;
788 
789  if (attr[0] != FR_VENDOR_SPECIFIC) return -1;
790  if (attr[1] < 9) return -1;
791  if ((attr + attr[1]) > end) return -1;
792  if (memcmp(data, attr + 2, 4) != 0) return -1; /* not WiMAX Vendor ID */
793 
794  if (attr[1] != (attr[7] + 6)) return -1; /* WiMAX attr doesn't exactly fill the VSA */
795 
796  if (data[4] != attr[6]) return -1; /* different WiMAX attribute */
797 
798  /*
799  * Skip over the Vendor-Specific header, and
800  * continue with the WiMAX attributes.
801  */
802  attr += 6;
803  }
804 
805  /*
806  * No data in the WiMAX attribute, make a "raw" one.
807  */
808  if (!wimax_len) return -1;
809 
810  head = tail = talloc_array(ctx, uint8_t, wimax_len);
811  if (!head) return -1;
812 
813  /*
814  * Copy the data over, this time trusting the attribute
815  * contents.
816  */
817  attr = data;
818  while (attr < end) {
819  memcpy_bounded(tail, attr + 4 + 3, attr[4 + 1] - 3, end);
820  tail += attr[4 + 1] - 3;
821  attr += 4 + attr[4 + 1]; /* skip VID+WiMax header */
822  attr += 2; /* skip Vendor-Specific header */
823  }
824 
825  FR_PROTO_HEX_DUMP(head, wimax_len, "Wimax fragments");
826 
827  ret = fr_radius_decode_pair_value(ctx, cursor, dict,
828  da, head, wimax_len, wimax_len, decoder_ctx);
829  talloc_free(head);
830  if (ret < 0) return ret;
831 
832  return end - data;
833 }

940  packet_len -= 4;

956  }
957 
958  data += vsa_len;
959  attr_len -= vsa_len;
960  packet_len -= vsa_len;
961  total += vsa_len;
962  }
963  fr_cursor_head(&tlv_cursor);
964  fr_cursor_tail(cursor);
965  fr_cursor_merge(cursor, &tlv_cursor);
966 
967  /*

969  * decode_vsa_internal, the hierachy between that unknown
970  * attribute and first known attribute was cloned
971  * meaning we can now free the unknown vendor.
972  */
973 
974  return total;
975 }
