193 /*
194  * Replace %<whatever> in a string.
195  *
196  * %P pool_name
197  * %I param
198  * %J lease_duration
199  *
200  */
201 static int sqlippool_expand(char * out, int outlen, char const * fmt,
202  rlm_sqlippool_t const *data, char * param, int param_len)
203 {
204  char *q;
205  char const *p;
206  char tmp[40]; /* For temporary storing of integers */
207 
208  q = out;
209  for (p = fmt; *p ; p++) {
210  int freespace;
211  int c;
212 
213  /* Calculate freespace in output */
214  freespace = outlen - (q - out);
215  if (freespace <= 1)
216  break;
217 
218  c = *p;
219  if (c != '%') {
220  *q++ = *p;
221  continue;
222  }
223 
224  if (*++p == '\0') {
225  break;
226  }
227 
228  if (c == '%') {
229  switch (*p) {
230  case 'P': /* pool name */
231  strlcpy(q, data->pool_name, freespace);
232  q += strlen(q);
233  break;
234  case 'I': /* IP address */
235  if (param && param_len > 0) {
236  if (param_len > freespace) {
237  strlcpy(q, param, freespace);
238  q += strlen(q);
239  }
240  else {
241  memcpy(q, param, param_len);
242  q += param_len;
243  }
244  }
245  break;
246  case 'J': /* lease duration */
247  sprintf(tmp, "%u", data->lease_duration);
248  strlcpy(q, tmp, freespace);
249  q += strlen(q);
250  break;
251 
252  default:
253  *q++ = '%';
254  *q++ = *p;
255  break;
256  }
257  }
258  }
259  *q = '\0';
260 
261 #if 0
262  DEBUG2("sqlippool_expand: \"%s\"", out);
263 #endif
264 
265  return strlen(out);
266 }

268 /** Perform a single sqlippool query
269  *
270  * Mostly wrapper around sql_query which does some special sqlippool sequence substitutions and expands
271  * the format string.
272  *
273  * @param fmt sql query to expand.
274  * @param handle sql connection handle.
275  * @param data Instance of rlm_sqlippool.
276  * @param request Current request.
277  * @param param ip address string.
278  * @param param_len ip address string len.
279  * @return
280  * - 0 on success.
281  * - < 0 on error.
282  */
283 static int sqlippool_command(char const *fmt, rlm_sql_handle_t **handle,
284  rlm_sqlippool_t const *data, request_t *request,
285  char *param, int param_len)
286 {
287  char query[MAX_QUERY_LEN];
288  char *expanded = NULL;
289 
290  int ret;
291  int affected;
292 
293  /*
294  * If we don't have a command, do nothing.
295  */
296  if (!fmt || !*fmt) return 0;
297 
298  /*
299  * No handle? That's an error.
300  */
301  if (!handle || !*handle) return -1;
302 
303  /*
304  * @todo this needs to die (should just be done in xlat expansion)
305  */
306  sqlippool_expand(query, sizeof(query), fmt, data, param, param_len);
307 
308  if (xlat_aeval(request, &expanded, request, query, data->sql_inst->sql_escape_func, *handle) < 0) return -1;
309 
310  ret = data->sql_inst->sql_query(data->sql_inst, request, handle, expanded);
311  if (ret < 0){
312  talloc_free(expanded);
313  return -1;
314  }
315  talloc_free(expanded);
316 
317  /*
318  * No handle, we can't continue.
319  */
320  if (!*handle) return -1;
321 
322  affected = (data->sql_inst->driver->sql_affected_rows)(*handle, data->sql_inst->config);
323 
324  (data->sql_inst->driver->sql_finish_query)(*handle, data->sql_inst->config);
325 
326  return affected;
327 }

336 /*
337  * Query the database expecting a single result row
338  */
339 static int CC_HINT(nonnull (1, 3, 4, 5)) sqlippool_query1(char *out, int outlen, char const *fmt,
340  rlm_sql_handle_t **handle, rlm_sqlippool_t *data,
341  request_t *request, char *param, int param_len)
342 {
343  char query[MAX_QUERY_LEN];
344  char *expanded = NULL;
345 
346  int rlen, retval;
347 
348  rlm_sql_row_t row;
349 
350  /*
351  * @todo this needs to die (should just be done in xlat expansion)
352  */
353  sqlippool_expand(query, sizeof(query), fmt, data, param, param_len);
354 
355  *out = '\0';
356 
357  /*
358  * Do an xlat on the provided string
359  */
360  if (xlat_aeval(request, &expanded, request, query, data->sql_inst->sql_escape_func, *handle) < 0) {
361  return 0;
362  }
363  retval = data->sql_inst->sql_select_query(data->sql_inst, request, handle, expanded);
364  talloc_free(expanded);
365 
366  if ((retval != 0) || !*handle) {
367  REDEBUG("database query error on '%s'", query);
368  return 0;
369  }
370 
371  if (data->sql_inst->sql_fetch_row(&row, data->sql_inst, request, handle) < 0) {
372  REDEBUG("Failed fetching query result");
373  goto finish;
374  }
375 
376  if (!row) {
377  REDEBUG("SQL query did not return any results");
378  goto finish;
379  }
380 
381  if (!row[0]) {
382  REDEBUG("The first column of the result was NULL");
383  goto finish;
384  }
385 
386  rlen = strlen(row[0]);
387  if (rlen >= outlen) {
388  RDEBUG2("insufficient string space");
389  goto finish;
390  }
391 
392  strcpy(out, row[0]);
393  retval = rlen;
394 
395 finish:
396  (data->sql_inst->driver->sql_finish_select_query)(*handle, data->sql_inst->config);
397 
398  return retval;
399 }

402  * Do any per-module initialization that is separate to each

419  inst->pool_name = talloc_typed_strdup(inst, pool_name);
420  } else {
421  inst->pool_name = talloc_typed_strdup(inst, "ippool");
422  }
423  sql_inst = module_by_name(NULL, inst->sql_instance_name);
424  if (!sql_inst) {

432  cf_log_perr(conf, "Failed resolving attribute");

434  }
435 
436  switch (inst->allocated_address_da->type) {
437  default:
438  cf_log_err(conf, "Cannot use non-IP attributes for 'allocated_address_attr = %s'", inst->allocated_address_attr);
439  return -1;
440 
441  case FR_TYPE_IPV4_ADDR:
442  case FR_TYPE_IPV4_PREFIX:
443  case FR_TYPE_IPV6_ADDR:
444  case FR_TYPE_IPV6_PREFIX:
445  break;
446  }
447 
448  if (inst->requested_address) {
449  if (!tmpl_is_xlat(inst->requested_address)) {

453  }
454

456 
457  if (strcmp(cf_section_name1(inst->sql_inst->cs), "sql") != 0) {
458  cf_log_err(conf, "Module \"%s\" is not an instance of the rlm_sql module",
459  inst->sql_instance_name);
460  return -1;
461  }
462 
463  return 0;

490 /*
491  * Allocate an IP number from the pool.
492  */
493 static unlang_action_t CC_HINT(nonnull) mod_alloc(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
494 {
495  rlm_sqlippool_t *inst = talloc_get_type_abort(mctx->instance, rlm_sqlippool_t);
496  char allocation[FR_MAX_STRING_LEN];
497  int allocation_len;
498  fr_pair_t *vp;
499  rlm_sql_handle_t *handle;
500 
501  /*
502  * If there is a Framed-IP-Address attribute in the reply do nothing
503  */
504  if (fr_pair_find_by_da(&request->reply_pairs, inst->allocated_address_da) != NULL) {
505  RDEBUG2("%s already exists", inst->allocated_address_da->name);
506 
507  return do_logging(p_result, inst, request, inst->log_exists, RLM_MODULE_NOOP);
508  }
509 
510  if (fr_pair_find_by_da(&request->control_pairs, attr_pool_name) == NULL) {
511  RDEBUG2("No %s defined", attr_pool_name->name);
512 
513  return do_logging(p_result, inst, request, inst->log_nopool, RLM_MODULE_NOOP);
514  }
515 
516  handle = fr_pool_connection_get(inst->sql_inst->pool, request);
517  if (!handle) {
518  REDEBUG("Failed reserving SQL connection");
519  RETURN_MODULE_FAIL;
520  }
521 
522  if (inst->sql_inst->sql_set_user(inst->sql_inst, request, NULL) < 0) {
523  RETURN_MODULE_FAIL;
524  }
525 
526  DO_PART(alloc_begin);
527 
528  /*
529  * If there is a query for finding the existing IP
530  * run that first
531  */
532  if (inst->alloc_existing && *inst->alloc_existing) {
533  allocation_len = sqlippool_query1(allocation, sizeof(allocation),
534  inst->alloc_existing, &handle,
535  inst, request, (char *) NULL, 0);
536  if (!handle) RETURN_MODULE_FAIL;
537  } else {
538  allocation_len = 0;
539  }
540 
541  /*
542  * If no existing IP was found and we have a requested IP address
543  * and a query to find whether it is available then try that
544  */
545  if ((allocation_len == 0) && inst->alloc_requested && *inst->alloc_requested) {
546  char buffer[128];
547  char *ip = NULL;
548  ssize_t slen;
549 
550  slen = tmpl_expand(&ip, buffer, sizeof(buffer), request, inst->requested_address, NULL, NULL);
551  if (slen < 0) RETURN_MODULE_FAIL;
552 
553  if (slen > 0) {
554  allocation_len = sqlippool_query1(allocation, sizeof(allocation),
555  inst->alloc_requested, &handle,
556  inst, request, (char *) NULL, 0);
557  if (!handle) RETURN_MODULE_FAIL;
558  }
559  }
560 
561  /*
562  * If no existing IP was found (or no query was run),
563  * run the query to find a free IP
564  */
565  if (allocation_len == 0) {
566  allocation_len = sqlippool_query1(allocation, sizeof(allocation),
567  inst->alloc_find, &handle,
568  inst, request, (char *) NULL, 0);
569  if (!handle) RETURN_MODULE_FAIL;
570  }
571 
572  /*
573  * Nothing found...
574  */
575  if (allocation_len == 0) {
576  DO_PART(alloc_commit);
577 
578  /*
579  *Should we perform pool-check ?
580  */
581  if (inst->pool_check && *inst->pool_check) {
582 
583  /*
584  *Ok, so the allocate-find query found nothing ...
585  *Let's check if the pool exists at all
586  */
587  allocation_len = sqlippool_query1(allocation, sizeof(allocation),
588  inst->pool_check, &handle, inst, request,
589  (char *) NULL, 0);
590  if (!handle) RETURN_MODULE_FAIL;
591 
592  fr_pool_connection_release(inst->sql_inst->pool, request, handle);
593 
594  if (allocation_len) {
595 
596  /*
597  * Pool exists after all... So,
598  * the failure to allocate the IP
599  * address was most likely due to
600  * the depletion of the pool. In
601  * that case, we should return
602  * NOTFOUND
603  */
604  RDEBUG2("pool appears to be full");
605  return do_logging(p_result, inst, request, inst->log_failed, RLM_MODULE_NOTFOUND);
606  }
607 
608  /*
609  * Pool doesn't exist in the table. It
610  * may be handled by some other instance of
611  * sqlippool, so we should just ignore this
612  * allocation failure and return NOOP

616 
617  }

623  }
624

626  * See if we can create the VP from the returned data. If not,
627  * error out. If so, add it to the list.
628  */

630  if (fr_pair_value_from_str(vp, allocation, allocation_len, '\0', true) < 0) {

632

634  fr_pool_connection_release(inst->sql_inst->pool, request, handle);
635  return do_logging(p_result, inst, request, inst->log_failed, RLM_MODULE_NOOP);
636  }
637 
638  RDEBUG2("Allocated IP %s", allocation);
639  fr_pair_add(&request->reply_pairs, vp);

648  RETURN_MODULE_FAIL;
649  }

652

703  } else {

802 /*

806 static unlang_action_t CC_HINT(nonnull) mod_accounting(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
807 {

814  RDEBUG2("Could not find account status type in packet");
815  RETURN_MODULE_NOOP;
816  }
817  acct_status_type = vp->vp_uint32;
818 
819 
820  switch (acct_status_type) {
821  case FR_STATUS_START:
822  case FR_STATUS_ALIVE:
823  return mod_update(p_result, mctx, request);
824 
825  case FR_STATUS_STOP:
826  return mod_release(p_result, mctx, request);

829  case FR_STATUS_ACCOUNTING_OFF:
830  return mod_bulk_release(p_result, mctx, request);

832  default:
833  /* We don't care about any other accounting packet */
834  RETURN_MODULE_NOOP;
835  }
