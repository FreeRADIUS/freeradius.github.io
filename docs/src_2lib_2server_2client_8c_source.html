67  int ret;

159 {

174 /** Add a client to a RADCLIENT_LIST
175  *
176  * @param clients list to add client to, may be NULL if global client list is being used.
177  * @param client to add.
178  * @return
179  * - true on success.
180  * - false on failure.
181  */
182 bool client_add(RADCLIENT_LIST *clients, RADCLIENT *client)
183 {
184 #ifdef WITH_TRIE
185  fr_trie_t *trie;
186 #else
187 #endif
188  RADCLIENT *old;
189  char buffer[FR_IPADDR_PREFIX_STRLEN];
190 
191  if (!client) return false;
192 
193  /*
194  * Hack to fixup wildcard clients
195  *
196  * If the IP is all zeros, with a 32 or 128 bit netmask
197  * assume the user meant to configure 0.0.0.0/0 instead
198  * of 0.0.0.0/32 - which would require the src IP of
199  * the client to be all zeros.
200  */
201  if (fr_ipaddr_is_inaddr_any(&client->ipaddr) == 1) switch (client->ipaddr.af) {
202  case AF_INET:
203  if (client->ipaddr.prefix == 32) client->ipaddr.prefix = 0;
204  break;
205 
206  case AF_INET6:
207  if (client->ipaddr.prefix == 128) client->ipaddr.prefix = 0;
208  break;
209 
210  default:
211  fr_assert(0);
212  }
213 
214  fr_inet_ntop_prefix(buffer, sizeof(buffer), &client->ipaddr);
215  DEBUG3("Adding client %s (%s) to prefix tree %i", buffer, client->longname, client->ipaddr.prefix);
216 
217  /*
218  * If "clients" is NULL, it means add to the global list,
219  * unless we're trying to add it to a virtual server...
220  */
221  if (!clients) {
222  if (client->server != NULL) {
223  CONF_SECTION *cs;
224  CONF_SECTION *subcs;
225 
226  if (!client->cs) {
227  ERROR("Failed to find configuration section in client. Ignoring 'virtual_server' directive");
228  return false;
229  }
230 
231  cs = cf_section_find(cf_root(client->cs), "server", client->server);
232  if (!cs) {
233  ERROR("Failed to find virtual server %s", client->server);
234  return false;
235  }
236 
237  /*
238  * If this server has no "listen" section, add the clients
239  * to the global client list.
240  */
241  subcs = cf_section_find(cs, "listen", NULL);
242  if (!subcs) goto global_clients;
243 
244  /*
245  * If the client list already exists, use that.
246  * Otherwise, create a new client list.
247  */
248  clients = cf_data_value(cf_data_find(cs, RADCLIENT_LIST, NULL));
249  if (!clients) {
250  clients = client_list_init(cs);
251  if (!clients) {
252  ERROR("Out of memory");
253  return false;
254  }
255 
256  if (!cf_data_add(cs, clients, NULL, true)) {
257  ERROR("Failed to associate clients with virtual server %s", client->server);
258  talloc_free(clients);
259  return false;
260  }
261  }
262 
263  } else {
264  global_clients:
265  /*
266  * Initialize the global list, if not done already.
267  */
268  if (!root_clients) {
269  root_clients = client_list_init(NULL);
270  if (!root_clients) return false;
271  }
272  clients = root_clients;
273  }
274  }
275 
276 #define namecmp(a) ((!old->a && !client->a) || (old->a && client->a && (strcmp(old->a, client->a) == 0)))
277 
278 #ifdef WITH_TRIE
279  trie = clients_trie(clients, &client->ipaddr, client->proto);
280 
281  /*
282  * Cannot insert the same client twice.
283  */
284  old = fr_trie_match(trie, &client->ipaddr.addr, client->ipaddr.prefix);
285 
286 #else /* WITH_TRIE */
287 
288  if (!clients->tree[client->ipaddr.prefix]) {
289  clients->tree[client->ipaddr.prefix] = rbtree_talloc_alloc(clients, client_cmp, RADCLIENT,
290  NULL, RBTREE_FLAG_NONE);
291  if (!clients->tree[client->ipaddr.prefix]) {
292  return false;
293  }
294  }
295 
296  old = rbtree_finddata(clients->tree[client->ipaddr.prefix], client);
297 #endif
298  if (old) {
299  /*
300  * If it's a complete duplicate, then free the new
301  * one, and return "OK".
302  */
303  if (namecmp(longname) && namecmp(secret) &&
304  namecmp(shortname) && namecmp(nas_type) &&
305  namecmp(server) &&
306  (old->message_authenticator == client->message_authenticator)) {
307  WARN("Ignoring duplicate client %s", client->longname);
308  client_free(client);
309  return true;
310  }
311 
312  ERROR("Failed to add duplicate client %s", client->shortname);
313  client_free(client);
314  return false;
315  }
316 #undef namecmp
317 
318 #ifdef WITH_TRIE
319  /*
320  * Other error adding client: likely is fatal.
321  */
322  if (fr_trie_insert(trie, &client->ipaddr.addr, client->ipaddr.prefix, client) < 0) {
323  client_free(client);
324  return false;
325  }
326 #else
327  if (!rbtree_insert(clients->tree[client->ipaddr.prefix], client)) {
328  client_free(client);
329  return false;
330  }
331 #endif
332 
333  /*
334  * @todo - do we want to do this for dynamic clients?
335  */
336  (void) talloc_steal(clients, client); /* reparent it */
337 
338  return true;
339 }

375 /*
376  * Find a client in the RADCLIENTS list.
377  */
378 RADCLIENT *client_find(RADCLIENT_LIST const *clients, fr_ipaddr_t const *ipaddr, int proto)
379 {
380 #ifdef WITH_TRIE
381  fr_trie_t *trie;
382 #else
383  int i, max;
384  RADCLIENT my_client, *client;
385 #endif
386 
387  if (!clients) clients = root_clients;
388 
389  if (!clients || !ipaddr) return NULL;
390 
391 #ifdef WITH_TRIE
392  trie = clients_trie(clients, ipaddr, proto);
393 
394  return fr_trie_lookup(trie, &ipaddr->addr, ipaddr->prefix);
395 #else
396 
397  if (proto == AF_INET) {
398  max = 32;
399  } else {
400  max = 128;
401  }
402 
403  if (max > ipaddr->prefix) max = ipaddr->prefix;
404 
405  my_client.proto = proto;
406  for (i = max; i >= 0; i--) {
407  if (!clients->tree[i]) continue;
408 
409  my_client.ipaddr = *ipaddr;
410  fr_ipaddr_mask(&my_client.ipaddr, i);
411  client = rbtree_finddata(clients->tree[i], &my_client);
412  if (client) {
413  return client;
414  }
415  }
416 
417  return NULL;
418 #endif
419 }

469 RADCLIENT_LIST *client_list_parse_section(CONF_SECTION *section, int proto, TLS_UNUSED bool tls_required)
470 {
471  bool global = false;
472  CONF_SECTION *cs = NULL;
473  RADCLIENT *c = NULL;
474  RADCLIENT_LIST *clients = NULL;
475  CONF_SECTION *server_cs = NULL;
476 
477  /*
478  * Be forgiving. If there's already a clients, return
479  * it. Otherwise create a new one.
480  */
481  clients = cf_data_value(cf_data_find(section, RADCLIENT_LIST, NULL));
482  if (clients) return clients;
483 
484  /*
485  * Parent the client list from the section.
486  */
487  clients = client_list_init(section);
488  if (!clients) return NULL;
489 
490  /*
491  * If the section is hung off the config root, this is
492  * the global client list, else it's virtual server
493  * specific client list.
494  */
495  if (cf_root(section) == section) global = true;
496 
497  if (strcmp("server", cf_section_name1(section)) == 0) server_cs = section;
498 
499  /*
500  * Iterate over all the clients in the section, adding
501  * them to the client list.
502  */
503  while ((cs = cf_section_find_next(section, cs, "client", CF_IDENT_ANY))) {
504  /*
505  * Check this before parsing the client.
506  */
507  if (proto) {
508  CONF_PAIR *cp;
509  int client_proto = IPPROTO_UDP;
510 
511  cp = cf_pair_find(cs, "proto");
512  if (cp) {
513  char const *value = cf_pair_value(cp);
514 
515  if (!value) {
516  cf_log_err(cs, "'proto' field must have a value");
517  talloc_free(clients);
518  return NULL;
519  }
520 
521  if (strcmp(value, "udp") == 0) {
522  /* do nothing */
523 
524  } else if (strcmp(value, "tcp") == 0) {
525  client_proto = IPPROTO_TCP;
526 #ifdef WITH_TLS
527  } else if (strcmp(value, "tls") == 0) {
528  client_proto = IPPROTO_TCP;
529 #endif
530  } else if (strcmp(value, "*") == 0) {
531  client_proto = IPPROTO_IP; /* fake for dual */
532  } else {
533  cf_log_err(cs, "Unknown proto \"%s\".", value);
534  talloc_free(clients);
535  return NULL;
536  }
537  }
538 
539  /*
540  * We don't have "proto = *", so the
541  * protocol MUST match what the caller
542  * asked for. Otherwise, we ignore the
543  * client.
544  */
545  if ((client_proto != IPPROTO_IP) && (proto != client_proto)) continue;
546  }
547 
548 
549  c = client_afrom_cs(cs, cs, server_cs);
550  if (!c) {
551  error:
552  client_free(c);
553  talloc_free(clients);
554  return NULL;
555  }
556 
557 #ifdef WITH_TLS
558  /*
559  * TLS clients CANNOT use non-TLS listeners.
560  * non-TLS clients CANNOT use TLS listeners.
561  */
562  if (tls_required != c->tls_required) {
563  cf_log_err(cs, "Client does not have the same TLS configuration as the listener");
564  goto error;
565  }
566 #endif
567 
568  if (!client_add(clients, c)) {
569  cf_log_err(cs, "Failed to add client %s", cf_section_name2(cs));
570  goto error;
571  }
572 
573  }
574 
575  /*
576  * Associate the clients structure with the section.
577  */
578  if (!cf_data_add(section, clients, NULL, false)) {
579  cf_log_err(section, "Failed to associate clients with section %s", cf_section_name1(section));
580  talloc_free(clients);
581  return NULL;
582  }
583 
584  /*
585  * Replace the global list of clients with the new one.
586  * The old one is still referenced from the original
587  * configuration, and will be freed when that is freed.
588  */
589  if (global) root_clients = clients;
590 
591  return clients;
592 }

594 /** Create a client CONF_SECTION using a mapping section to map values from a result set to client attributes
595  *
596  * If we hit a CONF_SECTION we recurse and process its CONF_PAIRS too.
597  *
598  * @note Caller should free CONF_SECTION passed in as out, on error.
599  * Contents of that section will be in an undefined state.
600  *
601  * @param[in,out] out Section to perform mapping on. Either the root of the client config, or a parent section
602  * (when this function is called recursively).
603  * Should be alloced with cf_section_alloc, or if there's a separate template section, the
604  * result of calling cf_section_dup on that section.
605  * @param[in] map section.
606  * @param[in] func to call to retrieve CONF_PAIR values. Must return a talloced buffer containing the value.
607  * @param[in] data to pass to func, usually a result pointer.
608  * @return
609  * - 0 on success.
610  * - -1 on failure.
611  */
612 int client_map_section(CONF_SECTION *out, CONF_SECTION const *map, client_value_cb_t func, void *data)
613 {
614  CONF_ITEM const *ci;
615 
616  for (ci = cf_item_next(map, NULL);
617  ci != NULL;
618  ci = cf_item_next(map, ci)) {
619  CONF_PAIR const *cp;
620  CONF_PAIR *old;
621  char *value;
622  char const *attr;
623 
624  /*
625  * Recursively process map subsection
626  */
627  if (cf_item_is_section(ci)) {
628  CONF_SECTION *cs, *cc;
629 
630  cs = cf_item_to_section(ci);
631  /*
632  * Use pre-existing section or alloc a new one
633  */
634  cc = cf_section_find(out, cf_section_name1(cs), cf_section_name2(cs));
635  if (!cc) {
636  cc = cf_section_alloc(out, out, cf_section_name1(cs), cf_section_name2(cs));
637  cf_section_add(out, cc);
638  if (!cc) return -1;
639  }
640 
641  if (client_map_section(cc, cs, func, data) < 0) return -1;
642  continue;
643  }
644 
645  cp = cf_item_to_pair(ci);
646  attr = cf_pair_attr(cp);
647 
648  /*
649  * The callback can return 0 (success) and not provide a value
650  * in which case we skip the mapping pair.
651  *
652  * Or return -1 in which case we error out.
653  */
654  if (func(&value, cp, data) < 0) {
655  cf_log_err(out, "Failed performing mapping \"%s\" = \"%s\"", attr, cf_pair_value(cp));
656  return -1;
657  }
658  if (!value) continue;
659 
660  /*
661  * Replace an existing CONF_PAIR
662  */
663  old = cf_pair_find(out, attr);
664  if (old) {
665  cf_pair_replace(out, old, value);
666  talloc_free(value);
667  continue;
668  }
669 
670  /*
671  * ...or add a new CONF_PAIR
672  */
673  cp = cf_pair_alloc(out, attr, value, T_OP_SET, T_BARE_WORD, T_SINGLE_QUOTED_STRING);
674  if (!cp) {
675  cf_log_err(out, "Failed allocing pair \"%s\" = \"%s\"", attr, value);
676  talloc_free(value);
677  return -1;
678  }
679  talloc_free(value);
680  cf_item_add(out, cf_pair_to_item(cp));
681  }
682 
683  return 0;
684 }

686 /** Allocate a new client from a config section
687  *
688  * @param ctx to allocate new clients in.
689  * @param cs to process as a client.
690  * @param server_cs The virtual server that this client belongs to.
691  * @return new RADCLIENT struct.
692  */
693 RADCLIENT *client_afrom_cs(TALLOC_CTX *ctx, CONF_SECTION *cs, CONF_SECTION *server_cs)
694 {
695  RADCLIENT *c;

826  goto error;
827  }
828  }

832  * we will use this address instead of the src from the
833  * request.
834  */

852  goto error;
853  }

860  */
861  if (c->response_window) {

863  FR_TIME_DELTA_BOUND_CHECK("response_window", c->response_window, <=, fr_time_delta_from_sec(60));
864  FR_TIME_DELTA_BOUND_CHECK("response_window", c->response_window, <=, main_config->max_request_time);
865  }
866

870  * omitted. When omitted, it's hard-coded to
871  * "radsec". See RFC 6614.
872  */
873  if (c->tls_required) {
874  c->secret = talloc_typed_strdup(cs, "radsec");
875  }
876 #endif
877 
878  if ((c->proto == IPPROTO_TCP) || (c->proto == IPPROTO_IP)) {
879  if ((c->limit.idle_timeout > 0) && (c->limit.idle_timeout < 5))
880  c->limit.idle_timeout = 5;

882  c->limit.lifetime = 5;
883  if ((c->limit.lifetime > 0) && (c->limit.idle_timeout > c->limit.lifetime))
884  c->limit.idle_timeout = 0;
885  }
