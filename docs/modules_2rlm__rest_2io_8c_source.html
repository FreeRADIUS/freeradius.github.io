29 /** Handle asynchronous cancellation of a request
30  *
31  * If we're signalled that the request has been cancelled (FR_SIGNAL_CANCEL).
32  * Cleanup any pending state and release the connection handle back into the pool.
33  */
34 void rest_io_module_action(module_ctx_t const *mctx, request_t *request, void *rctx, fr_state_signal_t action)
35 {
36  fr_curl_io_request_t *randle = talloc_get_type_abort(rctx, fr_curl_io_request_t);
37  rlm_rest_thread_t *t = talloc_get_type_abort(mctx->thread, rlm_rest_thread_t);
38  CURLMcode ret;
39 
40  if (action != FR_SIGNAL_CANCEL) return;
41 
42  RDEBUG2("Forcefully cancelling pending REST request");
43 
44  ret = curl_multi_remove_handle(t->mhandle->mandle, randle->candle); /* Gracefully terminate the request */
45  if (ret != CURLM_OK) {
46  RERROR("Failed removing curl handle from multi-handle: %s (%i)", curl_multi_strerror(ret), ret);
47  /* Not much we can do */
48  }
49  t->mhandle->transfers--;
50 
51  rest_request_cleanup(mctx->instance, randle);
52  fr_pool_connection_release(t->pool, request, randle);
53 }
