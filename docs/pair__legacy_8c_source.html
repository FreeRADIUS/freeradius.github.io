38 /** Create a new valuepair
39  *
40  * If attr and vendor match a dictionary entry then a VP with that #fr_dict_attr_t
41  * will be returned.
42  *
43  * If attr or vendor are uknown will call dict_attruknown to create a dynamic
44  * #fr_dict_attr_t of #FR_TYPE_OCTETS.
45  *
46  * Which type of #fr_dict_attr_t the #fr_pair_t was created with can be determined by
47  * checking @verbatim vp->da->flags.is_unknown @endverbatim.
48  *
49  * @param[in] ctx for allocated memory, usually a pointer to a #fr_radius_packet_t.
50  * @param[in] attr number.
51  * @param[in] vendor number.
52  * @return
53  * - A new #fr_pair_t.
54  * - NULL on error.
55  */
56 fr_pair_t *fr_pair_afrom_num(TALLOC_CTX *ctx, unsigned int vendor, unsigned int attr)
57 {
58  fr_dict_attr_t const *da;
59  fr_dict_attr_t const *parent;
60 
61  if (vendor == 0) {
62  da = fr_dict_attr_child_by_num(fr_dict_root(fr_dict_internal()), attr);
63  goto alloc;
64  }
65 
66  parent = fr_dict_attr_child_by_num(fr_dict_root(fr_dict_internal()), FR_VENDOR_SPECIFIC);
67  if (!parent) return NULL;
68 
69  parent = fr_dict_attr_child_by_num(parent, vendor);
70  if (!parent) return NULL;
71 
72  da = fr_dict_attr_child_by_num(parent, attr);
73 
74 alloc:
75  if (!da) {
76  fr_pair_t *vp;
77 
78  vp = fr_pair_alloc(ctx);
79  if (!vp) return NULL;
80 
81  /*
82  * Ensure that the DA is parented by the VP.
83  */
84  da = fr_dict_unknown_afrom_fields(vp, fr_dict_root(fr_dict_internal()), vendor, attr);
85  if (!da) {
86  talloc_free(vp);
87  return NULL;
88  }
89 
90  vp->da = da;
91  fr_value_box_init(&vp->data, da->type, da, false);
92 
93  return vp;
94  }
95 
96  return fr_pair_afrom_da(ctx, da);
97 }

99 /** Mark a valuepair for xlat expansion
100  *
101  * Copies xlat source (unprocessed) string to valuepair value, and sets value type.
102  *
103  * @param vp to mark for expansion.
104  * @param value to expand.
105  * @return
106  * - 0 if marking succeeded.
107  * - -1 if #fr_pair_t already had a value, or OOM.
108  */
109 int fr_pair_mark_xlat(fr_pair_t *vp, char const *value)
110 {
111  char *raw;
112 
113  /*
114  * valuepair should not already have a value.
115  */
116  if (vp->type != VT_NONE) {
117  fr_strerror_printf("Pair already has a value");
118  return -1;
119  }
120 
121  raw = talloc_typed_strdup(vp, value);
122  if (!raw) {
123  fr_strerror_printf("Out of memory");
124  return -1;
125  }
126 
127  vp->type = VT_XLAT;
128  vp->xlat = raw;
129  vp->vp_length = 0;
130 
131  return 0;
132 }

134 /** Create a valuepair from an ASCII attribute and value
135  *
136  * Where the attribute name is in the form:
137  * - Attr-%d
138  * - Attr-%d.%d.%d...
139  *
140  * @param ctx for talloc
141  * @param dict to user for partial resolution.
142  * @param attribute name to parse.
143  * @param value to parse (must be a hex string).
144  * @param op to assign to new valuepair.
145  * @return new #fr_pair_t or NULL on error.
146  */
147 static fr_pair_t *fr_pair_make_unknown(TALLOC_CTX *ctx, fr_dict_t const *dict,
148  char const *attribute, char const *value,
149  fr_token_t op)
150 {
151  fr_pair_t *vp;
152  fr_dict_attr_t *n;
153 
154  vp = fr_pair_alloc(ctx);
155  if (!vp) return NULL;
156 
157  if (fr_dict_unknown_afrom_oid_str(vp, &n, fr_dict_root(dict), attribute) <= 0) {
158  talloc_free(vp);
159  return NULL;
160  }
161  vp->da = n;
162 
163  /*
164  * No value, but ensure that we still set up vp->data properly.
165  */
166  if (!value) {
167  value = "";
168 
169  } else if (strncasecmp(value, "0x", 2) != 0) {
170  /*
171  * Unknown attributes MUST be of type 'octets'
172  */
173  fr_strerror_printf("Unknown attribute \"%s\" requires a hex "
174  "string, not \"%s\"", attribute, value);
175  talloc_free(vp);
176  return NULL;
177  }
178 
179  if (fr_pair_value_from_str(vp, value, -1, '"', false) < 0) {
180  talloc_free(vp);
181  return NULL;
182  }
183 
184  vp->op = (op == 0) ? T_OP_EQ : op;
185  return vp;
186 }

188 /** Create a #fr_pair_t from ASCII strings
189  *
190  * Converts an attribute string identifier (with an optional tag qualifier)
191  * and value string into a #fr_pair_t.
192  *
193  * The string value is parsed according to the type of #fr_pair_t being created.
194  *
195  * @param[in] ctx for talloc.
196  * @param[in] dict to look attributes up in.
197  * @param[in] vps list where the attribute will be added (optional)
198  * @param[in] attribute name.
199  * @param[in] value attribute value (may be NULL if value will be set later).
200  * @param[in] op to assign to new #fr_pair_t.
201  * @return a new #fr_pair_t.
202  */
203 fr_pair_t *fr_pair_make(TALLOC_CTX *ctx, fr_dict_t const *dict, fr_pair_list_t *vps,
204  char const *attribute, char const *value, fr_token_t op)
205 {
206  fr_dict_attr_t const *da;
207  fr_pair_t *vp;
208  char const *attrname = attribute;
209 
210  /*
211  * It's not found in the dictionary, so we use
212  * another method to create the attribute.
213  */
214  if (fr_dict_attr_by_qualified_name(&da, dict, attrname, true) != FR_DICT_ATTR_OK) {
215  vp = fr_pair_make_unknown(ctx, dict, attrname, value, op);
216  if (!vp) return NULL;
217 
218  if (vps) fr_pair_add(vps, vp);
219  return vp;
220  }
221 
222 #ifndef NDEBUG
223  if (!da) return NULL;
224 #endif
225 
226  if (da->type == FR_TYPE_GROUP) {
227  fr_strerror_printf("Attributes of type 'group' are not supported");
228  return NULL;
229  }
230 
231  vp = fr_pair_afrom_da(ctx, da);
232  if (!vp) return NULL;
233  vp->op = (op == 0) ? T_OP_EQ : op;
234 
235  switch (vp->op) {
236  case T_OP_CMP_TRUE:
237  case T_OP_CMP_FALSE:
238  fr_pair_value_clear(vp);
239  value = NULL; /* ignore it! */
240  break;
241 
242  /*
243  * Regular expression comparison of integer attributes
244  * does a STRING comparison of the names of their
245  * integer attributes.
246  */
247  case T_OP_REG_EQ: /* =~ */
248  case T_OP_REG_NE: /* !~ */
249  {
250 #ifndef HAVE_REGEX
251  fr_strerror_printf("Regular expressions are not supported");
252  return NULL;
253 #else
254  ssize_t slen;
255  regex_t *preg;
256 
257  /*
258  * Someone else will fill in the value.
259  */
260  if (!value) break;
261 
262  talloc_free(vp);
263 
264  slen = regex_compile(ctx, &preg, value, strlen(value), NULL, false, true);
265  if (slen <= 0) {
266  fr_strerror_printf_push("Error at offset %zu compiling regex for %s", -slen, attribute);
267  return NULL;
268  }
269  talloc_free(preg);
270 
271  vp = fr_pair_make(ctx, dict, NULL, attribute, NULL, op);
272  if (!vp) return NULL;
273 
274  if (fr_pair_mark_xlat(vp, value) < 0) {
275  talloc_free(vp);
276  return NULL;
277  }
278 
279  value = NULL; /* ignore it */
280  break;
281 #endif
282  }
283  default:
284  break;
285  }
286 
287  /*
288  * We probably want to fix fr_pair_value_from_str to accept
289  * octets as values for any attribute.
290  */
291  if (value && (fr_pair_value_from_str(vp, value, -1, '\"', true) < 0)) {
292  talloc_free(vp);
293  return NULL;
294  }
295 
296  if (vps) fr_pair_add(vps, vp);
297  return vp;
298 }

300 /** Read one line of attribute/value pairs into a list.
301  *
302  * The line may specify multiple attributes separated by commas.
303  *
304  * @note If the function returns #T_INVALID, an error has occurred and
305  * @note the valuepair list should probably be freed.
306  *
307  * @param[in] ctx for talloc
308  * @param[in] dict to resolve attributes in.
309  * @param[in] buffer to read valuepairs from.
310  * @param[in] list where the parsed fr_pair_ts will be appended.
311  * @param[in,out] token The last token we parsed
312  * @param[in] depth the nesting depth for FR_TYPE_GROUP
313  * @return
314  * - <= 0 on failure.
315  * - The number of bytes of name consumed on success.
316  */
317 static ssize_t fr_pair_list_afrom_substr(TALLOC_CTX *ctx, fr_dict_t const *dict, char const *buffer, fr_pair_list_t *list, fr_token_t *token, int depth)
318 {
319  fr_pair_t *vp, *head, **tail;
320  char const *p, *next;
321  fr_token_t last_token = T_INVALID;
322  fr_pair_t_RAW raw;
323  fr_dict_attr_t const *root = fr_dict_root(dict);
324 
325  /*
326  * We allow an empty line.
327  */
328  if (buffer[0] == 0) {
329  *token = T_EOL;
330  return 0;
331  }
332 
333  head = NULL;
334  tail = &head;
335 
336  p = buffer;
337  while (true) {
338  ssize_t slen;
339  fr_dict_attr_t const *da;
340  fr_dict_attr_t *da_unknown = NULL;
341 
342  fr_skip_whitespace(p);
343 
344  /*
345  * Stop at the end of the input, returning
346  * whatever token was last read.
347  */
348  if (!*p) break;
349 
350  if (*p == '#') {
351  last_token = T_EOL;
352  break;
353  }
354 
355  /*
356  * Parse the name.
357  */
358  slen = fr_dict_attr_by_qualified_name_substr(NULL, &da, dict, &FR_SBUFF_IN(p, strlen(p)), true);
359  if (slen <= 0) {
360 
361  slen = fr_dict_unknown_afrom_oid_substr(ctx, &da_unknown, root, p);
362  if (slen <= 0) {
363  fr_strerror_printf("Invalid attribute name %s", p);
364  p += -slen;
365 
366  error:
367  fr_pair_list_free(&head);
368  *token = T_INVALID;
369  return -(p - buffer);
370  }
371 
372  da = da_unknown;
373  }
374 
375  next = p + slen;
376 
377  if ((size_t) (next - p) >= sizeof(raw.l_opand)) {
378  fr_dict_unknown_free(&da);
379  fr_strerror_printf("Attribute name too long");
380  goto error;
381  }
382 
383  memcpy(raw.l_opand, p, next - p);
384  raw.l_opand[next - p] = '\0';
385  raw.r_opand[0] = '\0';
386 
387  p = next;
388  fr_skip_whitespace(p);
389 
390  /*
391  * There must be an operator here.
392  */
393  raw.op = gettoken(&p, raw.r_opand, sizeof(raw.r_opand), false);
394  if ((raw.op < T_EQSTART) || (raw.op > T_EQEND)) {
395  fr_dict_unknown_free(&da);
396  fr_strerror_printf("Expecting operator");
397  goto error;
398  }
399 
400  fr_skip_whitespace(p);
401 
402  /*
403  * Allow grouping attributes.
404  */
405  if ((da->type == FR_TYPE_GROUP) || (da->type == FR_TYPE_TLV)) {
406  fr_pair_t *child = NULL;
407 
408  if (*p != '{') {
409  fr_strerror_printf("Group list for %s MUST start with '{'", da->name);
410  goto error;
411  }
412  p++;
413 
414  vp = fr_pair_afrom_da(ctx, da);
415  if (!vp) goto error;
416 
417  slen = fr_pair_list_afrom_substr(vp, dict, p, &child, &last_token, depth + 1);
418  if (slen <= 0) {
419  talloc_free(vp);
420  goto error;
421  }
422 
423  if (last_token != T_RCBRACE) {
424  failed_group:
425  fr_strerror_printf("Failed to end group list with '}'");
426  talloc_free(vp);
427  goto error;
428  }
429 
430  p += slen;
431  fr_skip_whitespace(p);
432  if (*p != '}') goto failed_group;
433  p++;
434  vp->vp_group = child;
435 
436  } else {
437  fr_token_t quote;
438  char const *q;
439 
440  /*
441  * Free the unknown attribute, we don't need it any more.
442  */
443  fr_dict_unknown_free(&da);
444  da_unknown = NULL;
445 
446  /*
447  * Get the RHS thing.
448  */
449  quote = gettoken(&p, raw.r_opand, sizeof(raw.r_opand), false);
450  if (quote == T_EOL) {
451  fr_strerror_printf("Failed to get value");
452  goto error;
453  }
454 
455  switch (quote) {
456  /*
457  * Perhaps do xlat's
458  */
459  case T_DOUBLE_QUOTED_STRING:
460  /*
461  * Only report as double quoted if it contained valid
462  * a valid xlat expansion.
463  */
464  q = strchr(raw.r_opand, '%');
465  if (q && (q[1] == '{')) {
466  raw.quote = quote;
467  } else {
468  raw.quote = T_SINGLE_QUOTED_STRING;
469  }
470  break;
471 
472  case T_SINGLE_QUOTED_STRING:
473  case T_BACK_QUOTED_STRING:
474  case T_BARE_WORD:
475  raw.quote = quote;
476  break;
477 
478  default:
479  fr_strerror_printf("Failed to find expected value on right hand side in %s", da->name);
480  goto error;
481  }
482 
483  fr_skip_whitespace(p);
484 
485  /*
486  * Regular expressions get sanity checked by pair_make().
487  *
488  * @todo - note that they will also be escaped,
489  * so we may need to fix that later.
490  */
491  if ((raw.op == T_OP_REG_EQ) || (raw.op == T_OP_REG_NE)) {
492  vp = fr_pair_make(ctx, dict, NULL, raw.l_opand, raw.r_opand, raw.op);
493  if (!vp) goto error;
494  } else {
495  /*
496  * All other attributes get the name
497  * parsed.
498  */
499  vp = fr_pair_make(ctx, dict, NULL, raw.l_opand, NULL, raw.op);
500  if (!vp) goto error;
501 
502  /*
503  * We don't care what the value is, so
504  * ignore it.
505  */
506  if ((raw.op == T_OP_CMP_TRUE) || (raw.op == T_OP_CMP_FALSE)) goto next;
507 
508  /*
509  * fr_pair_raw_from_str() only returns this when
510  * the input looks like it needs to be xlat'd.
511  */
512  if (raw.quote == T_DOUBLE_QUOTED_STRING) {
513  if (fr_pair_mark_xlat(vp, raw.r_opand) < 0) {
514  talloc_free(vp);
515  goto error;
516  }
517 
518  /*
519  * Parse it ourselves. The RHS
520  * might NOT be tainted, but we
521  * don't know. So just mark it
522  * as such to be safe.
523  */
524  } else if (fr_pair_value_from_str(vp, raw.r_opand, -1, '"', true) < 0) {
525  talloc_free(vp);
526  goto error;
527  }
528  }
529  }
530 
531  next:
532  /*
533  * Free the unknown attribute, we don't need it any more.
534  */
535  fr_dict_unknown_free(&da);
536 
537  *tail = vp;
538  tail = &((*tail)->next);
539 
540  /*
541  * Now look for EOL, hash, etc.
542  */
543  if (!*p || (*p == '#') || (*p == '\n')) {
544  last_token = T_EOL;
545  break;
546  }
547 
548  /*
549  * Check for nested groups.
550  */
551  if ((depth > 0) && (p[0] == ' ') && (p[1] == '}')) p++;
552 
553  /*
554  * Stop at '}', too, if we're inside of a group.
555  */
556  if ((depth > 0) && (*p == '}')) {
557  last_token = T_RCBRACE;
558  break;
559  }
560 
561  if (*p != ',') {
562  fr_strerror_printf("Expected ',', got '%c' at offset %zu", *p, p - buffer);
563  goto error;
564  }
565  p++;
566  last_token = T_COMMA;
567  }
568 
569  if (head) fr_pair_add(list, head);
570 
571  /*
572  * And return the last token which we read.
573  */
574  *token = last_token;
575  return p - buffer;
576 }

599 /*
600  * Read valuepairs from the fp up to End-Of-File.
601  */
602 int fr_pair_list_afrom_file(TALLOC_CTX *ctx, fr_dict_t const *dict, fr_pair_list_t *out, FILE *fp, bool *pfiledone)
603 {
604  fr_token_t last_token = T_EOL;
605  bool found = false;
606  fr_cursor_t cursor;
607  char buf[8192];
608 
609  fr_cursor_init(&cursor, out);
610 
611  while (fgets(buf, sizeof(buf), fp) != NULL) {
612  fr_cursor_t append;
613  fr_pair_t *vp;
614 
615  /*
616  * If we get a '\n' by itself, we assume that's
617  * the end of that VP list.
618  */
619  if (buf[0] == '\n') {
620  if (found) {
621  *pfiledone = false;
622  return 0;
623  }
624  continue;
625  }
626 
627  /*
628  * Comments get ignored
629  */
630  if (buf[0] == '#') continue;
631 
632  /*
633  * Read all of the attributes on the current line.
634  *
635  * If we get nothing but an EOL, it's likely OK.
636  */
637  vp = NULL;
638  last_token = fr_pair_list_afrom_str(ctx, dict, buf, &vp);
639  if (!vp) {
640  if (last_token == T_EOL) break;
641 
642  /*
643  * Didn't read anything, but the previous
644  * line wasn't EOL. The input file has a
645  * format error.
646  */
647  *pfiledone = false;
648  vp = fr_cursor_head(&cursor);
649  if (vp) fr_pair_list_free(&vp);
650  *out = NULL;
651  return -1;
652  }
653 
654  found = true;
655  fr_cursor_init(&append, &vp);
656  fr_cursor_merge(&cursor, &append);
657  (void) fr_cursor_tail(&cursor);
658  }
659 
660  *pfiledone = true;
661  return 0;
662 }

665 /** Move pairs from source list to destination list respecting operator
666  *
667  * @note This function does some additional magic that's probably not needed
668  * in most places. Consider using radius_pairmove in server code.
669  *
670  * @note fr_pair_list_free should be called on the head of the source list to free
671  * unmoved attributes (if they're no longer needed).
672  *
673  * @param[in,out] to destination list.
674  * @param[in,out] from source list.
675  *
676  * @see radius_pairmove
677  */
678 void fr_pair_list_move(fr_pair_list_t *to, fr_pair_list_t *from)
679 {
680  fr_pair_t *i, *found;
681  fr_pair_t *head_new, **tail_new;
682  fr_pair_t **tail_from;
683 
684  if (!to || !from || !*from) return;
685 
686  /*
687  * We're editing the "to" list while we're adding new
688  * attributes to it. We don't want the new attributes to
689  * be edited, so we create an intermediate list to hold
690  * them during the editing process.
691  */
692  head_new = NULL;
693  tail_new = &head_new;
694 
695  /*
696  * We're looping over the "from" list, moving some
697  * attributes out, but leaving others in place.
698  */
699  tail_from = from;
700  while ((i = *tail_from) != NULL) {
701  fr_pair_t *j;
702 
703  VP_VERIFY(i);
704 
705  /*
706  * We never move Fall-Through.
707  */
708  if (fr_dict_attr_is_top_level(i->da) && (i->da->attr == FR_FALL_THROUGH)) {
709  tail_from = &(i->next);
710  continue;
711  }
712 
713  /*
714  * Unlike previous versions, we treat all other
715  * attributes as normal. i.e. there's no special
716  * treatment for passwords or Hint.
717  */
718 
719  switch (i->op) {
720  /*
721  * Anything else are operators which
722  * shouldn't occur. We ignore them, and
723  * leave them in place.
724  */
725  default:
