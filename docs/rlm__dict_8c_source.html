31  *
32  * @ingroup xlat_functions

34 static ssize_t xlat_dict_attr_by_num(TALLOC_CTX *ctx, char **out, UNUSED size_t outlen,
35  UNUSED void const *mod_inst, UNUSED void const *xlat_inst,
36  request_t *request, char const *fmt)
37 {
38  char *q;
39  unsigned int number;
40  fr_dict_attr_t const *da;
41 
42  *out = NULL;
43 
44  number = (unsigned int)strtoul(fmt, &q, 10);
45  if ((q == fmt) || (*q != '\0')) {
46  REDEBUG("Trailing garbage \"%s\" in attribute number string \"%s\"", q, fmt);
47  return -1;
48  }
49 
50  da = fr_dict_attr_child_by_num(fr_dict_root(request->dict), number);
51  if (!da) {
52  REDEBUG("No attribute found with number %u", number);
53  return -1;
54  }
55 
56  *out = talloc_typed_strdup(ctx, da->name);
57 
58  return talloc_array_length(*out) - 1;
59 }
