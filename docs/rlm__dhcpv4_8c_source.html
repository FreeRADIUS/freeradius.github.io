53 /** Decode DHCP option data
54  *
55  * Creates DHCP attributes based on the given binary option data
56  *
57  * Example:
58 @verbatim
59 %{dhcpv4_decode:%{Tmp-Octets-0}}
60 @endverbatim
61  *
62  * @ingroup xlat_functions
63  */
64 static xlat_action_t dhcpv4_decode_xlat(TALLOC_CTX *ctx, fr_cursor_t *out,
65  request_t *request, UNUSED void const *xlat_inst, UNUSED void *xlat_thread_inst,
66  fr_value_box_t **in)
67 {
68  fr_cursor_t in_cursor, cursor;
69  fr_value_box_t *vb, *vb_decoded;
70  fr_pair_t *vp, *head = NULL;
71  int decoded = 0;
72 
73  fr_cursor_init(&cursor, &head);
74 
75  for (vb = fr_cursor_talloc_init(&in_cursor, in, fr_value_box_t);
76  vb;
77  vb = fr_cursor_next(&in_cursor)) {
78  uint8_t const *p, *end;
79  ssize_t len;
80  fr_pair_t *vps = NULL;
81  fr_cursor_t options_cursor;
82 
83  if (vb->type != FR_TYPE_OCTETS) {
84  RWDEBUG("Skipping value \"%pV\", expected value of type %s, got type %s",
85  vb,
86  fr_table_str_by_value(fr_value_box_type_table, FR_TYPE_OCTETS, "<INVALID>"),
87  fr_table_str_by_value(fr_value_box_type_table, vb->type, "<INVALID>"));
88  continue;
89  }
90 
91  fr_cursor_init(&options_cursor, &vps);
92 
93  p = vb->vb_octets;
94  end = vb->vb_octets + vb->vb_length;
95 
96  /*
97  * Loop over all the options data
98  */
99  while (p < end) {
100  len = fr_dhcpv4_decode_option(request->packet, &options_cursor, dict_dhcpv4,
101  p, end - p, NULL);
102  if (len <= 0) {
103  RPERROR("DHCP option decoding failed");
104  fr_pair_list_free(&head);
105  return XLAT_ACTION_FAIL;
106  }
107  p += len;
108  }
109  fr_cursor_head(&options_cursor);
110  fr_cursor_merge(&cursor, &options_cursor);
111  }
112 
113  for (vp = fr_cursor_head(&cursor);
114  vp;
115  vp = fr_cursor_next(&cursor)) {
116  RDEBUG2("dhcp_option: &%pP", vp);
117  decoded++;
118  }
119 
120  fr_pair_list_move(&(request->request_pairs), &head);
121 
122  /* Free any unmoved pairs */
123  fr_pair_list_free(&head);
124 
125  /* create a value box to hold the decoded count */
126  MEM(vb_decoded = fr_value_box_alloc(ctx, FR_TYPE_UINT16, NULL, false));
127  vb_decoded->vb_uint16 = decoded;
128  fr_cursor_append(out, vb_decoded);
129 
130  return XLAT_ACTION_DONE;
131 }

133 /** Encode DHCP option data
134  *
135  * Returns octet string created from the provided DHCP attributes
136  *
137  * Example:
138 @verbatim
139 %{dhcpv4_encode:&request[*]}
140 @endverbatim
141  *
142  * @ingroup xlat_functions
143  */
144 static xlat_action_t dhcpv4_encode_xlat(TALLOC_CTX *ctx, fr_cursor_t *out,
145  request_t *request, UNUSED void const *xlat_inst, UNUSED void *xlat_thread_inst,
146  fr_value_box_t **in)
147 {
148  fr_cursor_t *cursor;
149  bool tainted = false;
150  fr_value_box_t *encoded;
151 
152  uint8_t binbuf[2048];
153  uint8_t *p = binbuf, *end = p + sizeof(binbuf);
154  ssize_t len = 0;
155 
156  if (!*in) return XLAT_ACTION_DONE;
157 
158  if (fr_value_box_list_concat(ctx, *in, in, FR_TYPE_STRING, true) < 0) {
159  RPEDEBUG("Failed concatenating input string for attribute reference");
160  return XLAT_ACTION_FAIL;
161  }
162 
163  if (xlat_fmt_to_cursor(NULL, &cursor, &tainted, request, (*in)->vb_strvalue) < 0) return XLAT_ACTION_FAIL;
164 
165  if (!fr_cursor_head(cursor)) return XLAT_ACTION_DONE; /* Nothing to encode */
166 
167  while (fr_cursor_filter_current(cursor, fr_dhcpv4_is_encodable, NULL)) {
168  len = fr_dhcpv4_encode_option(&FR_DBUFF_TMP(p, end), cursor,
169  &(fr_dhcpv4_ctx_t){ .root = fr_dict_root(dict_dhcpv4) });
170  if (len < 0) {
171  RPEDEBUG("DHCP option encoding failed");
172  talloc_free(cursor);
173  return XLAT_ACTION_FAIL;
174  }
175  p += len;
176  }
177  talloc_free(cursor);
178 
179  /*
180  * Pass the options string back
181  */
182  MEM(encoded = fr_value_box_alloc_null(ctx));
183  fr_value_box_memdup(encoded, encoded, NULL, binbuf, (size_t)len, tainted);
184  fr_cursor_append(out, encoded);
185 
186  return XLAT_ACTION_DONE;
187 }

189 static int dhcp_load(void)
