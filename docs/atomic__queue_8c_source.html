67 /** Create fixed-size atomic queue
68  *
69  * @note the queue must be freed explicitly by the ctx being freed, or by using
70  * the #fr_atomic_queue_free function.
71  *
72  * @param[in] ctx The talloc ctx to allocate the queue in.
73  * @param[in] size The number of entries in the queue.
74  * @return
75  * - NULL on error.
76  * - fr_atomic_queue_t *, a pointer to the allocated and initialized queue.
77  */
78 fr_atomic_queue_t *fr_atomic_queue_alloc(TALLOC_CTX *ctx, size_t size)
79 {
80  size_t i;
81  int64_t seq;
82  fr_atomic_queue_t *aq;
83  TALLOC_CTX *chunk;
84 
85  if (size == 0) return NULL;
86 
87  /*
88  * Allocate a contiguous blob for the header and queue.
89  * This helps with memory locality.
90  *
91  * Since we're allocating a blob, we should also set the
92  * name of the data, too.
93  */
94  chunk = talloc_aligned_array(ctx, (void **)&aq, CACHE_LINE_SIZE,
95  sizeof(*aq) + (size) * sizeof(aq->entry[0]));
96  if (!chunk) return NULL;
97  aq->chunk = chunk;
98 
99  talloc_set_name_const(chunk, "fr_atomic_queue_t");
100 
101  /*
102  * Initialize the array. Data is NULL, and indexes are
103  * the array entry number.
104  */
105  for (i = 0; i < size; i++) {
106  seq = i;
107 
108  aq->entry[i].data = NULL;
109  store(aq->entry[i].seq, seq);
110  }
111 
112  aq->size = size;
113 
114  /*
115  * Set the head / tail indexes, and force other CPUs to
116  * see the writes.
117  */
118  store(aq->head, 0);
119  store(aq->tail, 0);
120  atomic_thread_fence(memory_order_seq_cst);
121 
122  return aq;
123 }

138 /** Push a pointer into the atomic queue
139  *
140  * @param[in] aq The atomic queue to add data to.
141  * @param[in] data to push.
142  * @return
143  * - true on successful push
144  * - false on queue full
145  */
146 bool fr_atomic_queue_push(fr_atomic_queue_t *aq, void *data)
147 {
148  int64_t head;
149  fr_atomic_queue_entry_t *entry;
150 
151  if (!data) return false;
152 
153  head = load(aq->head);
154 
155  /*
156  * Try to find the current head.
157  */
158  for (;;) {
159  int64_t seq, diff;
160 
161  entry = &aq->entry[ head % aq->size ];
162  seq = aquire(entry->seq);
163  diff = (seq - head);
164 
165  /*
166  * head is larger than the current entry, the queue is full.
167  */
168  if (diff < 0) {
169 #if 0
170  fr_atomic_queue_debug(aq, stderr);
171 #endif
172  return false;
173  }
174 
175  /*
176  * Someone else has already written to this entry. Get the new head pointer, and continue.
177  */
178  if (diff > 0) {
179  head = load(aq->head);
180  continue;
181  }
182 
183  /*
184  * We have the possibility that we can write to
185  * this entry. Try it. If the write succeeds,
186  * we're done. If the write fails, re-load the
187  * current head entry, and continue.
188  */
189  if (cas_incr(aq->head, head)) {
190  break;
191  }
192  }
193 
194  /*
195  * Store the data in the queue, and increment the entry
196  * with the new index, and make the write visible to
197  * other CPUs.
198  */
199  entry->data = data;
200  store(entry->seq, head + 1);
201  return true;
202 }

205 /** Pop a pointer from the atomic queue
206  *
207  * @param[in] aq the atomic queue to retrieve data from.
208  * @param[out] p_data where to write the data.
209  * @return
210  * - true on successful pop
211  * - false on queue empty
212  */
213 bool fr_atomic_queue_pop(fr_atomic_queue_t *aq, void **p_data)
214 {
215  int64_t tail, seq;
216  fr_atomic_queue_entry_t *entry;
217 
218  if (!p_data) return false;
219 
220  tail = load(aq->tail);
221 
222  for (;;) {
223  int64_t diff;
224 
225  entry = &aq->entry[ tail % aq->size ];
226  seq = aquire(entry->seq);
227 
228  diff = (seq - (tail + 1));
229 
230  /*
231  * tail is smaller than the current entry, the queue is full.
232  */
233  if (diff < 0) {
234  return false;
235  }
236 
237  if (diff > 0) {
238  tail = load(aq->tail);
239  continue;
240  }
241 
242  if (cas_incr(aq->tail, tail)) {
243  break;
244  }
245  }
246 
247  /*
248  * Copy the pointer to the caller BEFORE updating the
249  * queue entry.
250  */
251  *p_data = entry->data;
252 
253  /*
254  * Set the current entry to past the end of the queue.
255  * i.e. it's unused.
256  */
257  seq = tail + aq->size;
258  store(entry->seq, seq);
259 
260  return true;
261 }
