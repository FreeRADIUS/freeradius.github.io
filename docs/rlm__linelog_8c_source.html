196 static void *mod_conn_create(TALLOC_CTX *ctx, void *instance, fr_time_delta_t timeout)
197 {
198  rlm_linelog_t const *inst = instance;
199  linelog_conn_t *conn;
200  int sockfd = -1;
201 
202  switch (inst->log_dst) {
203  case LINELOG_DST_UNIX:
204  DEBUG2("Opening UNIX socket at \"%s\"", inst->unix_sock.path);
205  sockfd = fr_socket_client_unix(inst->unix_sock.path, true);
206  if (sockfd < 0) {
207  PERROR("Failed opening UNIX socket");
208  return NULL;
209  }
210  break;
211 
212  case LINELOG_DST_TCP:
213  DEBUG2("Opening TCP connection to %pV:%u", fr_box_ipaddr(inst->tcp.dst_ipaddr), inst->tcp.port);
214 
215  sockfd = fr_socket_client_tcp(NULL, &inst->tcp.dst_ipaddr, inst->tcp.port, true);
216  if (sockfd < 0) {
217  PERROR("Failed opening TCP socket");
218  return NULL;
219  }
220  break;
221 
222  case LINELOG_DST_UDP:
223  DEBUG2("Opening UDP connection to %pV:%u", fr_box_ipaddr(inst->udp.dst_ipaddr), inst->udp.port);
224 
225  sockfd = fr_socket_client_udp(NULL, NULL, &inst->udp.dst_ipaddr, inst->udp.port, true);
226  if (sockfd < 0) {
227  PERROR("Failed opening UDP socket");
228  return NULL;
229  }
230  break;
231 
232  /*
233  * Are not connection oriented destinations
234  */
235  case LINELOG_DST_INVALID:
236  case LINELOG_DST_FILE:
237  case LINELOG_DST_SYSLOG:
238  fr_assert(0);
239  return NULL;
240  }
241 
242  if (errno == EINPROGRESS) {
243  if (timeout) {
244  DEBUG2("Waiting for connection to complete...");
245  } else {
246  DEBUG2("Blocking until connection complete...");
247  }
248  if (fr_socket_wait_for_connect(sockfd, timeout) < 0) {
249  PERROR("Failed connecting to log destination");
250  close(sockfd);
251  return NULL;
252  }
253  }
254  DEBUG2("Connection successful");
255 
256  /*
257  * Set blocking operation as we have no timeout set
258  */
259  if (!timeout && (fr_blocking(sockfd) < 0)) {
260  ERROR("Failed setting nonblock flag on fd");
261  close(sockfd);
262  return NULL;
263  }
264 
265  conn = talloc_zero(ctx, linelog_conn_t);
266  conn->sockfd = sockfd;
267  talloc_set_destructor(conn, _mod_conn_free);
268 
269  return conn;
270 }

285 static int mod_instantiate(void *instance, CONF_SECTION *conf)
286 {
287  rlm_linelog_t *inst = instance;
288  char prefix[100];
289 
290  /*
291  * Escape filenames only if asked.
292  */
293  if (inst->file.escape) {
294  inst->file.escape_func = rad_filename_escape;
295  } else {
296  inst->file.escape_func = rad_filename_make_safe;
297  }
298 
299  inst->log_dst = fr_table_value_by_str(linefr_log_dst_table, inst->log_dst_str, LINELOG_DST_INVALID);
300  if (inst->log_dst == LINELOG_DST_INVALID) {
301  cf_log_err(conf, "Invalid log destination \"%s\"", inst->log_dst_str);
302  return -1;
303  }
304 
305  if (!inst->log_src && !inst->log_ref) {
306  cf_log_err(conf, "Must specify a log format, or reference");
307  return -1;
308  }
309 
310  inst->name = cf_section_name2(conf);
311  if (!inst->name) inst->name = cf_section_name1(conf);
312 
313  snprintf(prefix, sizeof(prefix), "rlm_linelog (%s)", inst->name);
314 
315  /*
316  * Setup the logging destination
317  */
318  switch (inst->log_dst) {
319  case LINELOG_DST_FILE:
320  {
321  if (!inst->file.name) {
322  cf_log_err(conf, "No value provided for 'file.filename'");
323  return -1;
324  }
325 
326  inst->file.ef = module_exfile_init(inst, conf, 256, 30, true, NULL, NULL);
327  if (!inst->file.ef) {
328  cf_log_err(conf, "Failed creating log file context");
329  return -1;
330  }
331 
332  if (inst->file.group_str) {
333  char *endptr;
334 
335  inst->file.group = strtol(inst->file.group_str, &endptr, 10);
336  if (*endptr != '\0') {
337  if (rad_getgid(inst, &(inst->file.group), inst->file.group_str) < 0) {
338  cf_log_err(conf, "Unable to find system group \"%s\"",
339  inst->file.group_str);
340  return -1;
341  }
342  }
343  }
344  }
345  break;
346 
347  case LINELOG_DST_SYSLOG:
348  {
349  int num;
350 
351 #ifndef HAVE_SYSLOG_H
352  cf_log_err(conf, "Syslog output is not supported on this system");
353  return -1;
354 #else
355  if (inst->syslog.facility) {
356  num = fr_table_value_by_str(syslog_facility_table, inst->syslog.facility, -1);
357  if (num < 0) {
358  cf_log_err(conf, "Invalid syslog facility \"%s\"", inst->syslog.facility);
359  return -1;
360  }
361  inst->syslog.priority |= num;
362  }
363 
364  num = fr_table_value_by_str(syslog_severity_table, inst->syslog.severity, -1);
365  if (num < 0) {
366  cf_log_err(conf, "Invalid syslog severity \"%s\"", inst->syslog.severity);
367  return -1;
368  }
369  inst->syslog.priority |= num;
370 #endif
371  }
372  break;
373 
374  case LINELOG_DST_UNIX:
375 #ifndef HAVE_SYS_UN_H
376  cf_log_err(conf, "Unix sockets are not supported on this sytem");
377  return -1;
378 #else
379  inst->pool = module_connection_pool_init(cf_section_find(conf, "unix", NULL),
380  inst, mod_conn_create, NULL, prefix, NULL, NULL);
381  if (!inst->pool) return -1;
382 #endif
383  break;
384 
385  case LINELOG_DST_UDP:
386  inst->pool = module_connection_pool_init(cf_section_find(conf, "udp", NULL),
387  inst, mod_conn_create, NULL, prefix, NULL, NULL);
388  if (!inst->pool) return -1;
389  break;
390 
391  case LINELOG_DST_TCP:
392  inst->pool = module_connection_pool_init(cf_section_find(conf, "tcp", NULL),
393  inst, mod_conn_create, NULL, prefix, NULL, NULL);
394  if (!inst->pool) return -1;
395  break;
396 
397  case LINELOG_DST_INVALID:
398  fr_assert(0);
399  break;
400  }
401 
402  inst->delimiter_len = talloc_array_length(inst->delimiter) - 1;
403  inst->cs = conf;
404 
405  return 0;
406 }

408 /** Escape unprintable characters
409  *
410  * - Newline is escaped as ``\\n``.
411  * - Return is escaped as ``\\r``.
412  * - All other unprintables are escaped as @verbatim <oct><oct><oct> @endverbatim.
413  *
414  * @param request The current request.
415  * @param out Where to write the escaped string.
416  * @param outlen Length of the output buffer.
417  * @param in String to escape.
418  * @param arg unused.
419  */
420 /*
421  * Escape unprintable characters.
422  */
423 static size_t linelog_escape_func(UNUSED request_t *request,
424  char *out, size_t outlen, char const *in,
425  UNUSED void *arg)
426 {
427  if (outlen == 0) return 0;
428 
429  if (outlen == 1) {
430  *out = '\0';
431  return 0;
432  }
433 
434 
435  return fr_snprint(out, outlen, in, -1, 0);
436 }

438 /** Write a linelog message
439  *
440  * Write a log message to syslog or a flat file.
441  *
442  * @param[in] p_result the result of the module call:
443  * - #RLM_MODULE_NOOP if no message to log.
444  * - #RLM_MODULE_FAIL if we failed writing the message.
445  * - #RLM_MODULE_OK on success.
446  * @param[in] mctx module calling context.
447  * @param[in] request The current request.
448  */
449 static unlang_action_t CC_HINT(nonnull) mod_do_linelog(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
450 {
451  rlm_linelog_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_linelog_t);
452  linelog_conn_t *conn;
453  fr_time_delta_t timeout = 0;
454  char buff[4096];
455 
456  char *p = buff;
457  char const *value;
458  tmpl_t empty, *vpt = NULL, *vpt_p = NULL;
459  rlm_rcode_t rcode = RLM_MODULE_OK;
460  ssize_t slen;
461 
462  struct iovec vector_s[2];
463  struct iovec *vector = NULL, *vector_p;
464  size_t vector_len;
465  bool with_delim;
466 
467  buff[0] = '.'; /* force to be in current section (by default) */
468  buff[1] = '\0';
469  buff[2] = '\0';
470 
471  /*
472  * Expand log_ref to a config path, using the module
473  * configuration section as the root.
474  */
475  if (inst->log_ref) {
476  CONF_ITEM *ci;
477  CONF_PAIR *cp;
478  char const *tmpl_str;
479  char const *path;
480 
481  if (tmpl_expand(&path, buff + 1, sizeof(buff) - 1,
482  request, inst->log_ref, linelog_escape_func, NULL) < 0) {
483  RETURN_MODULE_FAIL;
484  }
485 
486  if (path != buff + 1) strlcpy(buff + 1, path, sizeof(buff) - 1);
487 
488  if (buff[1] == '.') p++;
489 
490  /*
491  * Don't go back up.
492  */
493  if (buff[2] == '.') {
494  REDEBUG("Invalid path \"%s\"", p);
495  RETURN_MODULE_FAIL;
496  }
497 
498  ci = cf_reference_item(NULL, inst->cs, p);
499  if (!ci) {
500  RDEBUG2("Path \"%s\" doesn't exist", p);
501  goto default_msg;
502  }
503 
504  if (!cf_item_is_pair(ci)) {
505  REDEBUG("Path \"%s\" resolves to a section (should be a pair)", p);
506  RETURN_MODULE_FAIL;
507  }
508 
509  cp = cf_item_to_pair(ci);
510  tmpl_str = cf_pair_value(cp);
511  if (!tmpl_str || (tmpl_str[0] == '\0')) {
512  RDEBUG2("Path \"%s\" resolves to an empty config pair", p);
513  vpt_p = tmpl_init_shallow(&empty, TMPL_TYPE_DATA, T_DOUBLE_QUOTED_STRING, "", 0);
514  fr_value_box_init_null(&empty.data.literal);
515  fr_value_box_strdup_shallow(&empty.data.literal, NULL, "", false);
516  goto build_vector;
517  }
518 
519  /*
520  * Alloc a template from the value of the CONF_PAIR
521  * using request as the context (which will hopefully avoid an alloc).
522  */
523  slen = tmpl_afrom_substr(request, &vpt,
524  &FR_SBUFF_IN(tmpl_str, talloc_array_length(tmpl_str) - 1),
525  cf_pair_value_quote(cp),
526  NULL,
527  &(tmpl_rules_t){
528  .dict_def = request->dict,
529  .allow_unknown = true,
530  .allow_unresolved = false,
531  .at_runtime = true
532  });
533  if (slen <= 0) {
534  REMARKER(tmpl_str, -slen, "%s", fr_strerror());
535  RETURN_MODULE_FAIL;
536  }
537  vpt_p = vpt;
538  } else {
539  default_msg:
540  /*
541  * Use the default format string
542  */
543  if (!inst->log_src) {
544  RDEBUG2("No default message configured");
545  RETURN_MODULE_NOOP;
546  }
547  /*
548  * Use the pre-parsed format template
549  */
550  RDEBUG2("Using default message");
551  vpt_p = inst->log_src;
552  }
553 
554 build_vector:
555  with_delim = (inst->log_dst != LINELOG_DST_SYSLOG) && (inst->delimiter_len > 0);
556 
557  /*
558  * Log all the things!
559  */
560  switch (vpt_p->type) {
561  case TMPL_TYPE_ATTR:
562  case TMPL_TYPE_LIST:
563  {
564  #define VECTOR_INCREMENT 20
565  fr_cursor_t cursor;
566  tmpl_cursor_ctx_t cc;
567  fr_pair_t *vp;
568  int alloced = VECTOR_INCREMENT, i;
569 
570  MEM(vector = talloc_array(request, struct iovec, alloced));
571  for (vp = tmpl_cursor_init(NULL, NULL, &cc, &cursor, request, vpt_p), i = 0;
572  vp;
573  vp = fr_cursor_next(&cursor), i++) {
574  /* need extra for line terminator */
575  if ((with_delim && ((i + 1) >= alloced)) ||
576  (i >= alloced)) {
577  alloced += VECTOR_INCREMENT;
578  MEM(vector = talloc_realloc(request, vector, struct iovec, alloced));
579  }
580 
581  switch (vp->vp_type) {
582  case FR_TYPE_OCTETS:
583  case FR_TYPE_STRING:
584  vector[i].iov_len = vp->vp_length;
585  vector[i].iov_base = vp->vp_ptr;
586  break;
587 
588  default:
589  vector[i].iov_len = fr_value_box_aprint(vector, &p, &vp->data, NULL);
590  vector[i].iov_base = p;
591  break;
592  }
593 
594  /*
595  * Add the line delimiter string
596  */
597  if (with_delim) {
598  i++;
599  memcpy(&vector[i].iov_base, &(inst->delimiter), sizeof(vector[i].iov_base));
600  vector[i].iov_len = inst->delimiter_len;
601  }
602  }
603  tmpl_cursor_clear(&cc);
604  vector_p = vector;
605  vector_len = i;
606  }
607  break;
608 
609  /*
610  * Log a single thing.
611  */
612  default:
613  slen = tmpl_expand(&value, buff, sizeof(buff), request, vpt_p, linelog_escape_func, NULL);
614  if (slen < 0) {
