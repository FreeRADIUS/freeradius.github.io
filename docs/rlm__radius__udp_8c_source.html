332 /*
333  * Status-Server checks. Manually build the packet, and
334  * all of its associated glue.
335  */
336 static void CC_HINT(nonnull) status_check_alloc(fr_event_list_t *el, udp_handle_t *h)
337 {
338  udp_request_t *u;
339  request_t *request;
340  rlm_radius_udp_t const *inst = h->inst;
341  map_t *map;
342 
343  fr_assert(!h->status_u && !h->status_r && !h->status_request);
344 
345  u = talloc_zero(h, udp_request_t);
346 
347  /*
348  * Status checks are prioritized over any other packet
349  */
350  u->priority = ~(uint32_t) 0;
351  u->status_check = true;
352 
353  /*
354  * Allocate outside of the free list.
355  * There appears to be an issue where
356  * the thread destructor runs too
357  * early, and frees the freelist's
358  * head before the module destructor
359  * runs.
360  */
361  request = request_local_alloc(u);
362  request->async = talloc_zero(request, fr_async_t);
363  talloc_const_free(request->name);
364  request->name = talloc_strdup(request, h->module_name);
365 
366  request->el = el;
367  request->packet = fr_radius_alloc(request, false);
368  request->reply = fr_radius_alloc(request, false);
369 
370  /*
371  * Create the VPs, and ignore any errors
372  * creating them.
373  */
374  for (map = inst->parent->status_check_map; map != NULL; map = map->next) {
375  /*
376  * Skip things which aren't attributes.
377  */
378  if (!tmpl_is_attr(map->lhs)) continue;
379 
380  /*
381  * Ignore internal attributes.
382  */
383  if (tmpl_da(map->lhs)->flags.internal) continue;
384 
385  /*
386  * Ignore signalling attributes. They shouldn't exist.
387  */
388  if ((tmpl_da(map->lhs) == attr_proxy_state) ||
389  (tmpl_da(map->lhs) == attr_message_authenticator)) continue;
390 
391  /*
392  * Allow passwords only in Access-Request packets.
393  */
394  if ((inst->parent->status_check != FR_CODE_ACCESS_REQUEST) &&
395  (tmpl_da(map->lhs) == attr_user_password)) continue;
396 
397  (void) map_to_request(request, map, map_to_vp, NULL);
398  }
399 
400  /*
401  * Ensure that there's a NAS-Identifier, if one wasn't
402  * already added.
403  */
404  if (!fr_pair_find_by_da(&request->request_pairs, attr_nas_identifier)) {
405  fr_pair_t *vp;
406 
407  MEM(pair_add_request(&vp, attr_nas_identifier) >= 0);
408  fr_pair_value_strdup(vp, "status check - are you alive?");
409  }
410 
411  /*
412  * Always add an Event-Timestamp, which will be the time
413  * at which the first packet is sent. Or for
414  * Status-Server, the time of the current packet.
415  */
416  if (!fr_pair_find_by_da(&request->request_pairs, attr_event_timestamp)) {
417  MEM(pair_add_request(NULL, attr_event_timestamp) >= 0);
418  }
419 
420  /*
421  * Initialize the request IO ctx. Note that we don't set
422  * destructors.
423  */
424  u->code = inst->parent->status_check;
425  request->packet->code = u->code;
426 
427  DEBUG3("%s - Status check packet type will be %s", h->module_name, fr_packet_codes[u->code]);
428  log_request_pair_list(L_DBG_LVL_3, request, request->request_pairs, NULL);
429 
430  MEM(h->status_r = talloc_zero(request, udp_result_t));
431  h->status_u = u;
432  h->status_request = request;
433 }

462 /** Status check timedout
463  *
464  * Setup retries, or fail the connection.
465  */
466 static void conn_status_check_timeout(fr_event_list_t *el, fr_time_t now, void *uctx)
467 {
468  fr_connection_t *conn = talloc_get_type_abort(uctx, fr_connection_t);
469  udp_handle_t *h;
470  udp_request_t *u;
471 
472  /*
473  * Connection must be in the connecting state when this fires
474  */
475  fr_assert(conn->state == FR_CONNECTION_STATE_CONNECTING);
476 
477  h = talloc_get_type_abort(conn->h, udp_handle_t);
478  u = h->status_u;
479 
480  /*
481  * We're only interested in contiguous, good, replies.
482  */
483  u->num_replies = 0;
484 
485  switch (fr_retry_next(&u->retry, now)) {
486  case FR_RETRY_MRD:
487  DEBUG("%s - Reached maximum_retransmit_duration, failing status checks",
488  h->module_name);
489  goto fail;
490 
491  case FR_RETRY_MRC:
492  DEBUG("%s - Reached maximum_retransmit_count, failing status checks",
493  h->module_name);
494  fail:
495  fr_connection_signal_reconnect(conn, FR_CONNECTION_FAILED);
496  return;
497 
498  case FR_RETRY_CONTINUE:
499  if (fr_event_fd_insert(h, el, h->fd, conn_writable_status_check, NULL,
500  conn_error_status_check, conn) < 0) {
501  PERROR("%s - Failed inserting FD event", h->module_name);
502  fr_connection_signal_reconnect(conn, FR_CONNECTION_FAILED);
503  }
504  return;
505  }
506 
507  fr_assert(0);
508 }

524 /** Read the incoming status-check response. If it's correct mark the connection as connected
525  *
526  */
527 static void conn_readable_status_check(fr_event_list_t *el, UNUSED int fd, UNUSED int flags, void *uctx)
528 {
529  fr_connection_t *conn = talloc_get_type_abort(uctx, fr_connection_t);
530  udp_handle_t *h = talloc_get_type_abort(conn->h, udp_handle_t);
531  fr_trunk_t *trunk = h->thread->trunk;
532  rlm_radius_t const *inst = h->inst->parent;
533  udp_request_t *u = h->status_u;
534  ssize_t slen;
535  fr_pair_t *reply = NULL;
536  uint8_t code = 0;
537 
538  slen = read(h->fd, h->buffer, h->buflen);
539  if (slen == 0) return;
540 
541  if (slen < 0) {
542  switch (errno) {
543 #if defined(EWOULDBLOCK) && (EWOULDBLOCK != EAGAIN)
544  case EWOULDBLOCK:
545 #endif
546  case EAGAIN:
547  case EINTR:
548  return; /* Wait to be signalled again */
549 
550  default:
551  break;
552  }
553 
554  ERROR("%s - Failed reading response from socket: %s",
555  h->module_name, fr_syserror(errno));
556  fr_connection_signal_reconnect(conn, FR_CONNECTION_FAILED);
557  return;
558  }
559 
560  /*
561  * Where we just return in this function, we're letting
562  * the response timer take care of progressing the
563  * connection attempt.
564  */
565  if (slen < RADIUS_HEADER_LENGTH) {
566  ERROR("%s - Packet too short, expected at least %zu bytes got %zd bytes",
567  h->module_name, (size_t)RADIUS_HEADER_LENGTH, slen);
568  return;
569  }
570 
571  if (u->id != h->buffer[1]) {
572  ERROR("%s - Received response with incorrect or expired ID. Expected %u, got %u",
573  h->module_name, u->id, h->buffer[1]);
574  return;
575  }
576 
577  if (decode(h, &reply, &code,
578  h, h->status_request, h->status_u, u->packet + RADIUS_AUTH_VECTOR_OFFSET,
579  h->buffer, slen) != DECODE_FAIL_NONE) return;
580 
581  fr_pair_list_free(&reply); /* FIXME - Do something with these... */
582 
583  /*
584  * Process the error, and count this as a success.
585  * This is usually used for dynamic configuration
586  * on startup.
587  */
588  if (code == FR_CODE_PROTOCOL_ERROR) protocol_error_reply(u, NULL, h);
589 
590  /*
591  * Last trunk event was a failure, be more careful about
592  * bringing up the connection (require multiple responses).
593  */
594  if ((trunk->last_failed && (trunk->last_failed > trunk->last_connected)) &&
595  (u->num_replies < inst->num_answers_to_alive)) {
596  /*
597  * Leave the timer in place. This timer is BOTH when we
598  * give up on the current status check, AND when we send
599  * the next status check.
600  */
601  DEBUG("%s - Received %u / %u replies for status check, on connection - %s",
602  h->module_name, u->num_replies, inst->num_answers_to_alive, h->name);
603  DEBUG("%s - Next status check packet will be in %pVs",
604  h->module_name, fr_box_time_delta(u->retry.next - fr_time()));
605 
606  /*
607  * Set the timer for the next retransmit.
608  */
609  if (fr_event_timer_at(h, el, &u->ev, u->retry.next, conn_status_check_again, conn) < 0) {
610  fr_connection_signal_reconnect(conn, FR_CONNECTION_FAILED);
611  }
612  return;
613  }
614 
615  /*
616  * It's alive!
617  */
618  status_check_reset(h, u);
619 
620  DEBUG("%s - Connection open - %s", h->module_name, h->name);
621 
622  fr_connection_signal_connected(conn);
623 }

626  *

628 static void conn_writable_status_check(fr_event_list_t *el, UNUSED int fd, UNUSED int flags, void *uctx)
629 {

718  * @param[in] conn to initialise.
719  * @param[in] uctx A #udp_thread_t

722 {

725  udp_thread_t *thread = talloc_get_type_abort(uctx, udp_thread_t);

733  h->src_port = 0;
734  h->max_packet_size = h->inst->max_packet_size;
735  h->last_idle = fr_time();

739  * to the iovec structs in coalesced, so we
740  * just need to setup the iovec, and pass how
741  * many messages we want to send to sendmmsg.

762  talloc_free(h);

764  }
765 
766  /*
767  * Set the connection name.
768  */
769  h->name = fr_asprintf(h, "proto udp local %pV port %u remote %pV port %u",
770  fr_box_ipaddr(h->src_ipaddr), h->src_port,
771  fr_box_ipaddr(h->inst->dst_ipaddr), h->inst->dst_port);
772 
773  talloc_set_destructor(h, _udp_handle_free);
774 
775 #ifdef SO_RCVBUF

782  }
783  }
784 #endif
785 
786 #ifdef SO_SNDBUF
787  {
788  int opt;
789  socklen_t socklen = sizeof(int);
790 
791  if (h->inst->send_buff_is_set) {
792  opt = h->inst->send_buff;
793  if (setsockopt(fd, SOL_SOCKET, SO_SNDBUF, &opt, sizeof(int)) < 0) {
794  WARN("%s - Failed setting 'SO_SNDBUF', write performance may be sub-optimal: %s",
795  h->module_name, fr_syserror(errno));
796  }
797  }
798 
799  if (getsockopt(fd, SOL_SOCKET, SO_SNDBUF, &opt, &socklen) < 0) {
800  WARN("%s - Failed getting 'SO_SNDBUF', write performance may be sub-optimal: %s",
801  h->module_name, fr_syserror(errno));
802 
803  /*
804  * This controls how many packets we attempt
805  * to send at once. Nothing bad happens if
806  * we get it wrong, but the user may see
807  * ENOBUFS errors at high packet rates.
808  */
809  h->send_buff_actual = h->inst->send_buff_is_set ?
810  h->inst->send_buff : h->max_packet_size * h->inst->max_send_coalesce;
811 
812  WARN("%s - Max coalesced outbound data will be %zu bytes", h->module_name,
813  h->send_buff_actual);
814  } else {
815 #ifdef __linux__
816  /*
817  * Linux doubles the buffer when you set it
818  * to account for "overhead".
819  */
820  h->send_buff_actual = ((size_t)opt) / 2;
821 #else
822  h->send_buff_actual = (size_t)opt;
823 #endif
824  }
825  }
826 #else
827  h->send_buff_actual = h->inst->send_buff_is_set ?
828  h->inst_send_buff : h->max_packet_size * h->inst->max_send_coalesce;
829 
830  WARN("%s - Modifying 'SO_SNDBUF' value is not supported on this system, "
831  "write performance may be sub-optimal", h->module_name);
832  WARN("%s - Max coalesced outbound data will be %zu bytes", h->module_name, h->inst->send_buff_actual);
833 #endif
834 
835  h->fd = fd;
836 
837  /*
838  * If we're doing status checks, then we want at least
839  * one positive response before signalling that the
840  * connection is open.
841  *
842  * To do this we install special I/O handlers that
843  * only signal the connection as open once we get a
844  * status-check response.
845  */
846  if (h->inst->parent->status_check) {
847  status_check_alloc(conn->el, h);
848 
849  /*
850  * Start status checking.
851  *
852  * If we've had no recent failures we need exactly
853  * one response to bring the connection online,
854  * otherwise we need inst->num_answers_to_alive
855  */
856  if (fr_event_fd_insert(h, conn->el, h->fd, NULL,
857  conn_writable_status_check, conn_error_status_check, conn) < 0) goto fail;
858  /*
859  * If we're not doing status-checks, signal the connection
860  * as open as soon as it becomes writable.
861  */
862  } else {
863  fr_connection_signal_on_fd(conn, fd);
864  }
865 
866  *h_out = h;
867 
868  // @todo - initialize the tracking memory, etc.
869  // i.e. histograms (or hyperloglog) of packets, so we can see
870  // which connections / home servers are fast / slow.
871 
872  return FR_CONNECTION_STATE_CONNECTING;
873 }

1269 static int encode(rlm_radius_udp_t const *inst, request_t *request, udp_request_t *u, uint8_t id)
1270 {

1272  uint8_t *msg = NULL;
1273  int message_authenticator = u->require_ma * (RADIUS_MESSAGE_AUTHENTICATOR_LENGTH + 2);
1274  int proxy_state = 6;
1275 
1276  fr_assert(inst->parent->allowed[u->code]);
1277  fr_assert(!u->packet);
1278 
1279  /*
1280  * Try to retransmit, unless there are special
1281  * circumstances.
1282  */
1283  u->can_retransmit = true;
1284 
1285  /*
1286  * This is essentially free, as this memory was
1287  * pre-allocated as part of the treq.
1288  */
1289  u->packet_len = inst->max_packet_size;
1290  MEM(u->packet = talloc_array(u, uint8_t, u->packet_len));
1291 
1292  /*
1293  * All proxied Access-Request packets MUST have a
1294  * Message-Authenticator, otherwise they're insecure.
1295  * Same goes for Status-Server.
1296  *
1297  * And we set the authentication vector to a random
1298  * number...
1299  */
1300  switch (u->code) {
1301  case FR_CODE_ACCESS_REQUEST:
1302  case FR_CODE_STATUS_SERVER:
1303  {
1304  size_t i;
1305  uint32_t hash, base;
1306 
1307  message_authenticator = RADIUS_MESSAGE_AUTHENTICATOR_LENGTH + 2;
1308 
1309  base = fr_rand();
1310  for (i = 0; i < RADIUS_AUTH_VECTOR_LENGTH; i += sizeof(uint32_t)) {
1311  hash = fr_rand() ^ base;
1312  memcpy(u->packet + RADIUS_AUTH_VECTOR_OFFSET + i, &hash, sizeof(hash));
1313  }
1314  }
1315  FALL_THROUGH;
1316 
1317  default:
1318  break;
1319  }
1320 
1321 
1322  /*
1323  * If we're sending a status check packet, update any
1324  * necessary timestamps. Also, don't add Proxy-State, as
1325  * we're originating the packet.
1326  */
1327  if (u->status_check) {
1328  fr_pair_t *vp;
1329 
1330  proxy_state = 0;
1331  vp = fr_pair_find_by_da(&request->request_pairs, attr_event_timestamp);
1332  if (vp) vp->vp_date = fr_time_to_unix_time(u->retry.updated);
1333 
1334  if (u->code == FR_CODE_STATUS_SERVER) u->can_retransmit = false;
1335 
1336  } else if (inst->parent->originate) {
1337  /*
1338  * We're originating packets instead of proxying
1339  * them. We don't add a Proxy-State attribute.
1340  */
1341  proxy_state = 0;
1342  }
1343 
1344  /*
1345  * We should have at mininum 64-byte packets, so don't
1346  * bother doing run-time checks here.
1347  */
1348  fr_assert(u->packet_len >= (size_t) (RADIUS_HEADER_LENGTH + proxy_state + message_authenticator));
1349 
1350  /*
1351  * Encode it, leaving room for Proxy-State and
1352  * Message-Authenticator if necessary.
1353  */
1354  packet_len = fr_radius_encode(u->packet, u->packet_len - (proxy_state + message_authenticator), NULL,
1355  inst->secret, talloc_array_length(inst->secret) - 1,
1356  u->code, id, request->request_pairs);
1357  if (fr_pair_encode_is_error(packet_len)) {
1358  RPERROR("Failed encoding packet");
1359 
1360  error:
1361  TALLOC_FREE(u->packet);
1362  return -1;
1363  }
1364 
1365  if (packet_len < 0) {
1366  size_t have;
1367  size_t need;
1368 
1369  have = u->packet_len - (proxy_state + message_authenticator);
1370  need = have - packet_len;
1371 
1372  if (need > RADIUS_MAX_PACKET_SIZE) {
1373  RERROR("Failed encoding packet. Have %zu bytes of buffer, need %zu bytes",
1374  have, need);
1375  } else {
1376  RERROR("Failed encoding packet. Have %zu bytes of buffer, need %zu bytes. "
1377  "Increase 'max_packet_size'", have, need);
1378  }
1379 
1380  goto error;
1381  }
1382  /*
1383  * The encoded packet should NOT over-run the input buffer.
1384  */
1385  fr_assert((size_t) (packet_len + proxy_state + message_authenticator) <= u->packet_len);
1386 
1387  /*
1388  * Add Proxy-State to the tail end of the packet.
1389  *
1390  * We need to add it here, and NOT in
1391  * request->request_pairs, because multiple modules
1392  * may be sending the packets at the same time.
1393  */
1394  if (proxy_state) {
1395  uint8_t *attr = u->packet + packet_len;
1396  fr_pair_t *vp;
1397  fr_cursor_t cursor;
1398  int count = 0;
1399 
1400  /*
1401  * Count how many Proxy-State attributes have
1402  * *our* magic number. Note that we also add a
1403  * counter to each Proxy-State, so we're double
1404  * sure that it's a loop.
1405  */
1406  if (DEBUG_ENABLED) {
1407  for (vp = fr_cursor_iter_by_da_init(&cursor, &request->request_pairs, attr_proxy_state);
1408  vp;
1409  vp = fr_cursor_next(&cursor)) {
1410  if ((vp->vp_length == 5) && (memcmp(vp->vp_octets, &inst->parent->proxy_state, 4) == 0)) {
1411  count++;
1412  }
1413  }
1414 
1415  /*
1416  * Some configurations may proxy to
1417  * ourselves for tests / simplicity. But
1418  * warn if there are a large number of
1419  * identical Proxy-State attributes.
1420  */
1421  if (count >= 4) RWARN("Potential proxy loop detected! Please recheck your configuration.");
1422  }
1423 
1424  attr[0] = (uint8_t)attr_proxy_state->attr;
1425  attr[1] = 7;
1426  memcpy(attr + 2, &inst->parent->proxy_state, 4);
1427  attr[6] = count & 0xff;
1428  packet_len += 7;
1429 
1430  MEM(vp = fr_pair_afrom_da(u->packet, attr_proxy_state));
1431  fr_pair_value_memdup(vp, attr + 2, 5, true);
1432  fr_pair_add(&u->extra, vp);
1433  }
1434 
1435  /*
1436  * Add Message-Authenticator manually.
1437  *
1438  * Note that the length check will always pass, due to
1439  * the buflen manipulation done above.
1440  */
1441  if (message_authenticator) {
1442  msg = u->packet + packet_len;
1443 
1444  msg[0] = (uint8_t) attr_message_authenticator->attr;
1445  msg[1] = RADIUS_MESSAGE_AUTHENTICATOR_LENGTH + 2;
1446  memset(msg + 2, 0, RADIUS_MESSAGE_AUTHENTICATOR_LENGTH);
1447 
1448  packet_len += msg[1];
1449  }
1450 
1451  /*
1452  * Update the packet header based on the new attributes.
1453  */
1454  u->packet[2] = (packet_len >> 8) & 0xff;
1455  u->packet[3] = packet_len & 0xff;
1456  u->packet_len = packet_len;
1457 
1458  /*
1459  * Ensure that we update the Acct-Delay-Time based on the
1460  * time difference between now, and when we originally
1461  * received the request.
1462  */
1463  if ((u->code == FR_CODE_ACCOUNTING_REQUEST) &&
1464  (fr_pair_find_by_da(&request->request_pairs, attr_acct_delay_time) != NULL)) {
1465  uint8_t *attr, *end;
1466  uint32_t delay;
1467  fr_time_t now;
1468 
1469  /*
1470  * Change Acct-Delay-Time in the packet, but not
1471  * in the debug output. Oh well. We don't want
1472  * to edit the incoming VPs, and we want to
1473  * update the encoded version of Acct-Delay-Time.
1474  * So we just walk through the packet to find it.
1475  */
1476  end = u->packet + packet_len;
1477 
1478  for (attr = u->packet + RADIUS_HEADER_LENGTH;
1479  attr < end;
1480  attr += attr[1]) {
1481  if (attr[0] != attr_acct_delay_time->attr) continue;
1482  if (attr[1] != 6) continue;
1483 
1484  now = u->retry.updated;
1485 
1486  /*
1487  * Add in the time between when
1488  * we received the packet, and
1489  * when we're sending the packet.
1490  */
1491  memcpy(&delay, attr + 2, 4);
1492  delay = ntohl(delay);
1493  delay += fr_time_delta_to_sec(now - u->recv_time);
1494  delay = htonl(delay);
1495  memcpy(attr + 2, &delay, 4);
1496  break;
1497  }
1498 
1499  u->can_retransmit = false;
1500  }
1501 
1502  /*
1503  * Only certain types of packet, and those with a
1504  * message_authenticator need signing.
1505  */
1506  if (message_authenticator) goto sign;
1507  switch (u->code) {
1508  case FR_CODE_ACCOUNTING_REQUEST:
1509  case FR_CODE_DISCONNECT_REQUEST:
1510  case FR_CODE_COA_REQUEST:
1511  sign:
1512  /*
1513  * Now that we're done mangling the packet, sign it.
1514  */
1515  if (fr_radius_sign(u->packet, NULL, (uint8_t const *) inst->secret,
1516  talloc_array_length(inst->secret) - 1) < 0) {
1517  RERROR("Failed signing packet");
1518  goto error;
1519  }
1520  break;
1521 
1522  default:
1523  break;
1524 
1525  }
1526  return 0;
1527 }

1542 /** See if the connection is zombied.
1543  *
1544  * We check for zombie when major events happen:
1545  *
1546  * 1) request hits its final timeout
1547  * 2) request timer hits, and it needs to be retransmitted
1548  * 3) a DUP packet comes in, and the request needs to be retransmitted
1549  * 4) we're sending a packet.
1550  *
1551  * There MIGHT not be retries configured, so we MUST check for zombie
1552  * when any new packet comes in. Similarly, there MIGHT not be new
1553  * packets, but retries are configured, so we have to check there,
1554  * too.
1555  *
1556  * Also, the socket might not be writable for a while. There MIGHT
1557  * be a long time between getting the timer / DUP signal, and the
1558  * request finally being written to the socket. So we need to check
1559  * for zombie at BOTH the timeout and the mux / write function.
1560  *
1561  * @return
1562  * - true if a connection state change was triggered.
1563  * The connection is likely now a zombie or was reconnected.
1564  * - false if the connection did not change state. It may
1565  * still be a zombie, but it was a zombie when this
1566  */
1567 static bool check_for_zombie(fr_event_list_t *el, fr_trunk_connection_t *tconn, fr_time_t now)
1568 {
1569  udp_handle_t *h = talloc_get_type_abort(tconn->conn->h, udp_handle_t);
1570 
1571  /*
1572  * We're replicating, and don't care about the health of
1573  * the home server, and this function should not be called.
1574  */
1575  fr_assert(!h->inst->replicate);
1576 
1577  /*
1578  * If there's already a zombie check started, don't do
1579  * another one.
1580  *
1581  * Or if we never sent a packet, we don't know (or care)
1582  * if the home server is up.
1583  *
1584  * Or if we had sent packets, and then went idle.
1585  *
1586  * Or we had replies, and then went idle.
1587  *
1588  * We do checks for both sent && replied, because we
1589  * could have sent packets without getting replies (and
1590  * then mark it zombie), or we could have gotten some
1591  * replies which then stopped coming back (and then mark
1592  * it zombie).
1593  */
1594  if (h->status_checking || h->zombie_ev || !h->last_sent || (h->last_sent <= h->last_idle) ||
1595  (h->last_reply && (h->last_reply <= h->last_idle))) {
1596  return false;
1597  }
1598 
1599  if (now == 0) now = fr_time();
1600 
1601  /*
1602  * We've sent a packet since we last went idle, and/or
1603  * we've received replies since we last went idle.
1604  *
1605  * If we have a reply, then set the zombie timeout from
1606  * when we received the last reply.
1607  *
1608  * If we haven't seen a reply, then set the zombie
1609  * timeout from when we first started sending packets.
1610  */
1611  if (h->last_reply) {
1612  if ((h->last_reply + h->inst->parent->zombie_period) >= now) return false;
1613  DEBUG2("%s - We have passed 'zombie_period' time since the last reply on connection %s",
1614  h->module_name, h->name);
1615  } else {
1616  if ((h->first_sent + h->inst->parent->zombie_period) >= now) return false;
1617  DEBUG2("%s - We have passed 'zombie_period' time since we first sent a packet, and "
1618  "there have been no replies on connection %s", h->module_name, h->name);
1619  }
1620 
1621  /*
1622  * No status checks: this connection is dead.
1623  *
1624  * We will requeue this packet on another
1625  * connection.
1626  */
1627  if (!h->inst->parent->status_check) {
1628  fr_time_t when;
1629 
1630  WARN("%s - Connection failed. Reviving it in %pVs", h->module_name,
1631  fr_box_time_delta(h->inst->parent->revive_interval));
1632  fr_trunk_connection_signal_inactive(tconn);
1633  (void) fr_trunk_connection_requests_requeue(tconn, FR_TRUNK_REQUEST_STATE_ALL, 0, false);
1634 
1635  when = now + h->inst->parent->revive_interval;
1636  if (fr_event_timer_at(h, el, &h->zombie_ev, when, revive_timer, tconn) < 0) {
1637  fr_trunk_connection_signal_reconnect(tconn, FR_CONNECTION_FAILED);
1638  return true;
1639  }
1640 
1641  return true;
1642  }
1643 
1644  /*
1645  * Mark the connection as inactive, but keep sending
1646  * packets on it.
1647  */
1648  WARN("%s - Entering Zombie state - connection %s", h->module_name, h->name);
1649  h->status_checking = true;
1650 
1651  /*
1652  * Move ALL requests to other connections!
1653  */
1654  fr_trunk_connection_signal_inactive(tconn);
1655  (void) fr_trunk_connection_requests_requeue(tconn, FR_TRUNK_REQUEST_STATE_ALL, 0, false);
1656 
1657  /*
1658  * Queue up the status check packet. It will be sent
1659  * when the connection is writable.
1660  */
1661  h->status_u->retry.start = 0;
1662  h->status_r->treq = NULL;
1663 
1664  if (fr_trunk_request_enqueue_on_conn(&h->status_r->treq, tconn, h->status_request,
1665  h->status_u, h->status_r, true) != FR_TRUNK_ENQUEUE_OK) {
1666  fr_trunk_connection_signal_reconnect(tconn, FR_CONNECTION_FAILED);
1667  }
1668 
1669  return true;
1670 }

1676 {

1679  udp_request_t *u = talloc_get_type_abort(treq->preq, udp_request_t);

1696  * been assigned to another connection
1697  * or freed.
1698  *
1699  * In any case we must not continue to

1708  * contiguous, good, replies.
1709  */

1718  * request_mux() ? That way we can catch the case of
1719  * packets sitting in the queue for extended periods of
1720  * time, and still run the timers.

1722  case FR_RETRY_CONTINUE:
1723  fr_trunk_request_requeue(treq);
1724  return;
1725

1729

1731  RDEBUG("Reached maximum_retransmit_count, failing request");
1732  break;
1733  }

1736  fr_trunk_request_signal_complete(treq);
1737 
1738  if (!u->status_check) return;
1739 
1740  WARN("%s - No response to status check, marking connection as dead - %s", h->module_name, h->name);
1741 
1742  h->status_checking = false;
1743  fr_trunk_connection_signal_reconnect(tconn, FR_CONNECTION_FAILED);
1744 }

1746 static void request_mux(fr_event_list_t *el,
1747  fr_trunk_connection_t *tconn, fr_connection_t *conn, UNUSED void *uctx)
1748 {
1749  udp_handle_t *h = talloc_get_type_abort(conn->h, udp_handle_t);
1750  rlm_radius_udp_t const *inst = h->inst;
1751  int sent;
1752  uint16_t i, queued;
1753  size_t total_len = 0;
1754 
1755  /*
1756  * If the connection just became a zombie
1757  * don't try and enqueue things on it!
1758  */
1759  if (check_for_zombie(el, tconn, 0)) return;
1760 
1761  /*
1762  * Encode multiple packets in preparation
1763  * for transmission with sendmmsg.
