59  * The first comparison returns +1 for a>b, and -1 for a<b

62  * Use STABLE_COMPARE when you don't really care about ordering,

127 /*
128  * The callback should be declared as:
129  * int callback(void *context, void *data)
130  *
131  * The "context" is some user-defined context.
132  * The "data" is the pointer to the user data in the node,
133  * NOT the node itself.
134  *
135  * It should return 0 if all is OK, and !0 for any error.
136  * The walking will stop on any error.
137  *
138  * Except with RBTREE_DELETE_ORDER, where the callback should return <0 for
139  * errors, and may return 1 to delete the current node and halt,
140  * or 2 to delete the current node and continue. This may be
141  * used to batch-delete select nodes from a locked rbtree.
142  */
143 int rbtree_walk(rbtree_t *tree, rb_order_t order, rb_walker_t compare, void *uctx);
