50 /** Wrap freeReplyObject so we consistently check for NULL pointers
51  *
52  * Older versions such as 0.10 (which ship with Ubuntu <= 14.10)
53  * don't check for NULL pointer before attempting to free, so we
54  * get a NULL pointer dereference in some cases.
55  *
56  * Rather than go back through the many calls to freeReplyObject
57  * and attempt to determine code paths that may result in it being
58  * called on a NULL pointer, we use this to always check.
59  */
60 static inline void fr_redis_reply_free(redisReply **reply)
61 {
62  if (*reply) freeReplyObject(*reply);
63  *reply = NULL;
64 }

66 static inline void fr_redis_pipeline_free(redisReply *reply[], size_t num)
67 {
68  size_t i;
69  for (i = 0; i < num; i++) fr_redis_reply_free(&(reply[i]));
70 }
