42 /*
43  * null JSON values are represented as NULL pointers by JSON-C
44  *
45  * Which would be ok, except functions such as json_object_get_type
46  * and json_object_is_type* SEGV when passed a NULL pointer.
47  *
48  * There's a json_type_null in the type enum, but for some reason
49  * it's not used. Newer versions of json-c don't have this issue.
50  * Old versions of json-c SEGV when passed a null JSON object.
51  *
52  */
53 #define fr_json_object_get_type(_obj) ((_obj) ? json_object_get_type(_obj) : json_type_null)
54 #define fr_json_object_is_type(_obj, _type) ((_obj) ? json_object_is_type(_obj, _type) : (_type == json_type_null))
