70 /** Callback for fr_ldap_sasl_interactive_bind
71  *
72  * @param[in] handle used for the SASL bind.
73  * @param[in] flags data as provided to fr_ldap_sasl_interactive_bind.
74  * @param[in] uctx Our context data, containing the identity, password, realm and various other things.
75  * @param[in] sasl_callbacks Array of challenges to provide responses for.
76  * @return SASL_OK.
77  */
78 static int _sasl_interact(UNUSED LDAP *handle, UNUSED unsigned flags, void *uctx, void *sasl_callbacks)
79 {
80  fr_ldap_sasl_ctx_t *sasl_ctx = talloc_get_type_abort(uctx, fr_ldap_sasl_ctx_t);;
81  sasl_interact_t *cb = sasl_callbacks;
82  sasl_interact_t *cb_p;
83 
84  for (cb_p = cb; cb_p->id != SASL_CB_LIST_END; cb_p++) {
85  DEBUG3("SASL challenge : %s", cb_p->challenge);
86  DEBUG3("SASL prompt : %s", cb_p->prompt);
87 
88  switch (cb_p->id) {
89  case SASL_CB_AUTHNAME:
90  cb_p->result = sasl_ctx->identity;
91  break;
92 
93  case SASL_CB_PASS:
94  cb_p->result = sasl_ctx->password;
95  break;
96 
97  case SASL_CB_USER:
98  cb_p->result = sasl_ctx->proxy ? sasl_ctx->proxy : sasl_ctx->identity;
99  break;
100 
101  case SASL_CB_GETREALM:
102  cb_p->result = sasl_ctx->realm;
103  break;
104 
105  default:
106  break;
107  }
108  DEBUG3("SASL result : %s", cb_p->result ? (char const *)cb_p->result : "");
109  }
110  return SASL_OK;
111 }

113 /** Parse a sasl bind response from a server
114  *
115  * @param[in] el the event occurred in.
116  * @param[in] fd the event occurred on.
117  * @param[in] flags from kevent.
118  * @param[in] uctx bind_ctx containing credentials, and connection config/handle.
119  */
120 static void _ldap_sasl_bind_io_read(fr_event_list_t *el, int fd, UNUSED int flags, void *uctx)
121 {
122  fr_ldap_sasl_ctx_t *sasl_ctx = talloc_get_type_abort(uctx, fr_ldap_sasl_ctx_t);
123  fr_ldap_connection_t *c = sasl_ctx->c;
124  fr_ldap_rcode_t status;
125 
126  /*
127  * If LDAP parse result indicates there was an error
128  * then we're done.
129  */
130  status = fr_ldap_result(&sasl_ctx->result, NULL, c, sasl_ctx->msgid, LDAP_MSG_ALL, sasl_ctx->identity, 0);
131  switch (status) {
132  case LDAP_PROC_SUCCESS:
133  case LDAP_PROC_CONTINUE:
134  {
135  struct berval *srv_cred;
136  int ret;
137 
138  /*
139  * Free the old result (if there is one)
140  */
141  if (sasl_ctx->result) {
142  ldap_msgfree(sasl_ctx->result);
143  sasl_ctx->result = NULL;
144  }
145 
146  ret = ldap_parse_sasl_bind_result(c->handle, sasl_ctx->result, &srv_cred, 0);
147  if (ret != LDAP_SUCCESS) {
148  ERROR("SASL decode failed (bind failed): %s", ldap_err2string(ret));
149  error:
150  talloc_free(sasl_ctx);
151  fr_ldap_state_error(c); /* Restart the connection state machine */
152  return;
153  }
154 
155  DEBUG3("SASL response : %pV", fr_box_strvalue_len(srv_cred->bv_val, srv_cred->bv_len));
156  ldap_memfree(srv_cred);
157 
158  /*
159  * If we need to continue, wait until the
160  * socket is writable, and then call
161  * ldap_sasl_interactive_bind again.
162  */
163  DEBUG3("Continuing SASL mech %s...", sasl_ctx->rmech);
164 
165  ret = fr_event_fd_insert(sasl_ctx, el, fd,
166  NULL,
167  _ldap_sasl_bind_io_write, /* Need to write more SASL stuff */
168  _ldap_sasl_bind_io_error,
169  sasl_ctx);
170  if (!fr_cond_assert(ret == 0)) goto error;
171  }
172  return;
173 
174  default:
175  PERROR("SASL bind failed");
176  goto error;
177  }
178 }

180 /** Progress an interactive SASL bind
181  *
182  * @param[in] el the event occurred in.
183  * @param[in] fd the event occurred on.
184  * @param[in] flags from kevent.
185  * @param[in] uctx Connection config and handle.
186  */
187 static void _ldap_sasl_bind_io_write(fr_event_list_t *el, int fd, UNUSED int flags, void *uctx)
188 {
189  fr_ldap_sasl_ctx_t *sasl_ctx = talloc_get_type_abort(uctx, fr_ldap_sasl_ctx_t);
190  fr_ldap_connection_t *c = sasl_ctx->c;
191 
192  int ret = 0;
193 
194  LDAPControl *our_serverctrls[LDAP_MAX_CONTROLS];
195  LDAPControl *our_clientctrls[LDAP_MAX_CONTROLS];
196 
197  fr_ldap_control_merge(our_serverctrls, our_clientctrls,
198  NUM_ELEMENTS(our_serverctrls),
199  NUM_ELEMENTS(our_clientctrls),
200  c, sasl_ctx->serverctrls, sasl_ctx->clientctrls);
201 
202  DEBUG2("Starting SASL mech(s): %s", sasl_ctx->mechs);
203 
204  /*
205  * Set timeout to be 0.0, which is the magic
206  * non-blocking value.
207  */
208  (void) ldap_set_option(c->handle, LDAP_OPT_NETWORK_TIMEOUT, &fr_time_delta_to_timeval(0));
209  ret = ldap_sasl_interactive_bind(c->handle, NULL, sasl_ctx->mechs,
210  our_serverctrls, our_clientctrls,
211  LDAP_SASL_AUTOMATIC,
212  _sasl_interact, sasl_ctx, sasl_ctx->result,
213  &sasl_ctx->rmech, &sasl_ctx->msgid);
214  fr_ldap_connection_timeout_reset(c);
215  switch (ret) {
216  /*
217  * If the handle was not connected, this operation
218  * can return either LDAP_X_CONNECTING or
219  * LDAP_SASL_BIND_IN_PROGRESS
220  * depending on how fast the connection came up
221  * and whether it was connectionless.
222  */
223  case LDAP_X_CONNECTING:
224  ret = ldap_get_option(c->handle, LDAP_OPT_DESC, &fd);
225  if (!fr_cond_assert(ret == LDAP_OPT_SUCCESS)) {
226  error:
227  talloc_free(sasl_ctx);
228  fr_ldap_connection_timeout_reset(c);
229  fr_ldap_state_error(c); /* Restart the connection state machine */
230  return;
231  }
232 
233  ret = fr_event_fd_insert(sasl_ctx, el, fd,
234  NULL,
235  _ldap_sasl_bind_io_write, /* We'll be called again when the conn is open */
236  _ldap_sasl_bind_io_error,
237  sasl_ctx);
238  if (!fr_cond_assert(ret == 0)) goto error;
239  break;
240 
241  /*
242  * Want to read more SASL stuff...
243  */
244  case LDAP_SASL_BIND_IN_PROGRESS:
245  ret = fr_event_fd_insert(sasl_ctx, el, fd,
246  _ldap_sasl_bind_io_read,
247  NULL,
248  _ldap_sasl_bind_io_error,
249  sasl_ctx);
250  if (!fr_cond_assert(ret == 0)) goto error;
251  break;
252 
253  /*
254  * We're done, woohoo!
255  */
256  case LDAP_SUCCESS:
257  talloc_free(sasl_ctx);
258  fr_ldap_state_next(c);
259  break;
260 
261  default:
262  ERROR("ldap sasl bind failed: %s", ldap_err2string(ret));
263  goto error;
264  }
265 }

279 /** Install I/O handlers for the bind operation
280  *
281  * @param[in] c connection to StartTLS on.
282  * @param[in] mechs Space delimited list of sasl mechs to try.
283  * @param[in] identity SASL identity to bind with.
284  * @param[in] password Password credential to pass to SASL.
285  * @param[in] proxy identity. May be NULL.
286  * @param[in] realm SASL realm.
287  * @param[in] serverctrls Extra controls to pass to the server.
288  * @param[in] clientctrls Extra controls to pass to libldap.
289  * @return
290  * - 0 on success.
291  * - -1 on failure.
292  */
293 int fr_ldap_sasl_bind_async(fr_ldap_connection_t *c,
294  char const *mechs,
295  char const *identity,
296  char const *password,
297  char const *proxy,
298  char const *realm,
299  LDAPControl **serverctrls, LDAPControl **clientctrls)
300 {
301  int fd = -1;
302  fr_ldap_sasl_ctx_t *sasl_ctx;
303  fr_event_list_t *el;
304 
305  DEBUG2("Starting SASL bind operation");
306 
307  MEM(sasl_ctx = talloc_zero(c, fr_ldap_sasl_ctx_t));
308  talloc_set_destructor(sasl_ctx, _sasl_ctx_free);
309 
310  sasl_ctx->c = c;
311  sasl_ctx->mechs = mechs;
312  sasl_ctx->identity = identity;
313  sasl_ctx->password = password;
314  sasl_ctx->proxy = proxy;
315  sasl_ctx->realm = realm;
316  sasl_ctx->serverctrls = serverctrls;
317  sasl_ctx->clientctrls = clientctrls;
318 
319  el = c->conn->el;
320 
321  if (ldap_get_option(c->handle, LDAP_OPT_DESC, &fd) == LDAP_SUCCESS) {
322  int ret;
323 
324  ret = fr_event_fd_insert(sasl_ctx, el, fd,
325  NULL,
326  _ldap_sasl_bind_io_write,
327  _ldap_sasl_bind_io_error,
328  sasl_ctx);
329  if (!fr_cond_assert(ret == 0)) {
330  talloc_free(sasl_ctx);
331  return -1;
332  }
333  } else {
334  _ldap_sasl_bind_io_write(el, -1, 0, sasl_ctx);
335  }
336 
337  return 0;
338 }
