99 /*
100  * Allow for quotation marks.
101  */
102 static bool buf2entry(rlm_csv_t *inst, char *buf, char **out)
103 {
104  char *p, *q;
105 
106  if (*buf != '"') {
107  *out = strchr(buf + 1, *inst->delimiter);
108 
109  if (!*out) { /* mash CR / LF */
110  for (p = buf; *p != '\0'; p++) {
111  if (*p < ' ') {
112  *p = '\0';
113  break;
114  }
115  }
116  }
117 
118  return true;
119  }
120 
121  p = buf + 1;
122  q = buf;
123 
124  while (*p) {
125  if (*p < ' ') {
126  *q = '\0';
127  *out = NULL;
128  return true;
129  }
130 
131  /*
132  * Double quotes to single quotes.
133  */
134  if ((*p == '"') && (p[1] == '"')) {
135  *(q++) = '"';
136  p += 2;
137  continue;
138  }
139 
140  /*
141  * Double quotes and EOL mean we're done.
142  */
143  if ((*p == '"') && (p[1] < ' ')) {
144  *(q++) = '\0';
145 
146  *out = NULL;
147  return true;
148  }
149 
150  /*
151  * Double quotes and delimiter: point to the delimiter.
152  */
153  if ((*p == '"') && (p[1] == *inst->delimiter)) {
154  *(q++) = '\0';
155 
156  *out = p + 1;
157  return true;
158  }
159 
160  /*
161  * Everything else gets copied over verbatim
162  */
163  *(q++) = *(p++);
164  *q = '\0';
165  }
166 
167  return false;
168 }

188 static bool insert_entry(CONF_SECTION *conf, rlm_csv_t *inst, rlm_csv_entry_t *e, int lineno)
189 {
190  rlm_csv_entry_t *old;
191 
192  fr_assert(e != NULL);
193 
194  old = find_entry(inst, e->key);
195  if (old) {
196  if (!inst->allow_multiple_keys && !inst->multiple_index_fields) {
197  cf_log_err(conf, "%s[%d]: Multiple entries are disallowed", inst->filename, lineno);
198  goto fail;
199  }
200 
201  /*
202  * Save this entry at the tail of the matching
203  * entries.
204  */
205  while (old->next) old = old->next;
206  old->next = e;
207  return true;
208  }
209 
210  if ((inst->key_data_type == FR_TYPE_IPV4_ADDR) || (inst->key_data_type == FR_TYPE_IPV4_PREFIX)) {
211  if (fr_trie_insert(inst->trie, &e->key->vb_ip.addr.v4.s_addr, e->key->vb_ip.prefix, e) < 0) {
212  cf_log_err(conf, "Failed inserting entry for file %s line %d: %s",
213  inst->filename, lineno, fr_strerror());
214  fail:
215  talloc_free(e);
216  return false;
217  }
218 
219  } else if ((inst->key_data_type == FR_TYPE_IPV6_ADDR) || (inst->key_data_type == FR_TYPE_IPV6_PREFIX)) {
220  if (fr_trie_insert(inst->trie, &e->key->vb_ip.addr.v6.s6_addr, e->key->vb_ip.prefix, e) < 0) {
221  cf_log_err(conf, "Failed inserting entry for file %s line %d: %s",
222  inst->filename, lineno, fr_strerror());
223  goto fail;
224  }
225 
226  } else if (!rbtree_insert(inst->tree, e)) {
227  cf_log_err(conf, "Failed inserting entry for file %s line %d: duplicate entry",
228  inst->filename, lineno);
229  goto fail;
230  }
231 
232  return true;
233 }

236 static bool duplicate_entry(CONF_SECTION *conf, rlm_csv_t *inst, rlm_csv_entry_t *old, char *p, int lineno)
237 {
238  int i;
239  fr_type_t type = inst->key_data_type;
240  rlm_csv_entry_t *e;
241 
242  MEM(e = (rlm_csv_entry_t *)talloc_zero_array(inst, uint8_t,
243  sizeof(*e) + (inst->used_fields * sizeof(e->data[0]))));
244  talloc_set_type(e, rlm_csv_entry_t);
245 
246  e->key = talloc_zero(e, fr_value_box_t);
247  if (fr_value_box_from_str(e->key, e->key, &type, NULL, p, -1, 0, false) < 0) {
248  cf_log_err(conf, "Failed parsing key field in file %s line %d - %s", inst->filename, lineno,
249  fr_strerror());
250  return false;
251  }
252 
253  /*
254  * Copy the other fields;
255  */
256  for (i = 0; i < inst->used_fields; i++) {
257  if (old->data[i]) e->data[i] = old->data[i]; /* no need to dup it, it's never freed... */
258  }
259 
260  return insert_entry(conf, inst, e, lineno);
261 }

263 /*
264  * Convert a buffer to a CSV entry
265  */
266 static bool file2csv(CONF_SECTION *conf, rlm_csv_t *inst, int lineno, char *buffer)
267 {
268  rlm_csv_entry_t *e;
269  int i;
270  char *p, *q;
271 
272  MEM(e = (rlm_csv_entry_t *)talloc_zero_array(inst, uint8_t,
273  sizeof(*e) + (inst->used_fields * sizeof(e->data[0]))));
274  talloc_set_type(e, rlm_csv_entry_t);
275 
276  for (p = buffer, i = 0; p != NULL; p = q, i++) {
277  if (!buf2entry(inst, p, &q)) {
278  cf_log_err(conf, "Malformed entry in file %s line %d", inst->filename, lineno);
279  return false;
280  }
281 
282  if (q) *(q++) = '\0';
283 
284  if (i >= inst->num_fields) {
285  cf_log_err(conf, "Too many fields at file %s line %d", inst->filename, lineno);
286  return false;
287  }
288 
289  /*
290  * This is the key field.
291  */
292  if (i == inst->index_field) {
293  fr_type_t type = inst->key_data_type;
294 
295  /*
296  * Check for /etc/group style keys.
297  */
298  if (inst->multiple_index_fields) {
299  char *l;
300 
301  /*
302  * Silently omit empty entries.
303  */
304  if (!*p) {
305  talloc_free(e);
306  return true;
307  }
308 
309  /*
310  * Check & smash ','. duplicate
311  * 'e', and insert it into the
312  * hash table / trie.
313  */
314  l = strchr(p, ',');
315  while (l) {
316  *l = '\0';
317 
318  if (!duplicate_entry(conf, inst, e, p, lineno)) goto fail;
319 
320  if (!l) break;
321  p = l + 1;
322  l = strchr(p, ',');
323  }
324  }
325 
326  /*
327  * Set the last entry to use 'e'
328  */
329  e->key = talloc_zero(e, fr_value_box_t);
330  if (fr_value_box_from_str(e->key, e->key, &type, NULL, p, -1, 0, false) < 0) {
331  cf_log_err(conf, "Failed parsing key field in file %s line %d - %s", inst->filename, lineno,
332  fr_strerror());
333  fail:
334  talloc_free(e);
335  return false;
336  }
337  continue;
338  }
339 
340  /*
341  * This field is unused. Ignore it.
342  */
343  if (inst->field_offsets[i] < 0) continue;
344 
345  /*
346  * Try to parse fields as data types if the data type is defined.
347  */
348  if (inst->field_types[i] != FR_TYPE_INVALID) {
349  fr_value_box_t box;
350  fr_type_t type = inst->field_types[i];
351 
352  if (fr_value_box_from_str(e, &box, &type, NULL, p, -1, 0, false) < 0) {
353  cf_log_err(conf, "Failed parsing field '%s' in file %s line %d - %s", inst->field_names[i],
354  inst->filename, lineno, fr_strerror());
355  goto fail;
356  }
357 
358  fr_value_box_clear(&box);
359  }
360 
361  MEM(e->data[inst->field_offsets[i]] = talloc_typed_strdup(e, p));
362  }
363 
364  if (i < inst->num_fields) {
365  cf_log_err(conf, "Too few fields in file %s at line %d (%d < %d)", inst->filename, lineno, i, inst->num_fields);
366  goto fail;
367  }
368 
369  return insert_entry(conf, inst, e, lineno);
370 }

373 static int fieldname2offset(rlm_csv_t const *inst, char const *field_name, int *array_offset)
374 {
375  int i;
376 
377  /*
378  * Find out which field the RHS maps to.
379  *
380  * For maps of less than 32 entries or so, an
381  * array is faster than more complex solutions.
382  */
383  for (i = 0; i < inst->num_fields; i++) {
384  if (strcmp(field_name, inst->field_names[i]) == 0) {
385  if (array_offset) *array_offset = i;
386  return inst->field_offsets[i];
387  }
388  }
389 
390  return -1;
391 }

395 /*
396  * Verify one map entry.
397  */
398 static int csv_map_verify(map_t *map, void *instance)
399 {
400  rlm_csv_t *inst = instance;
401  int offset;
402  fr_type_t type = FR_TYPE_INVALID;
403 
404  /*
405  * Destinations where we can put the fr_pair_ts we
406  * create using CSV values.
407  */
408  switch (map->lhs->type) {
409  case TMPL_TYPE_ATTR:
410  type = tmpl_da(map->lhs)->type;
411  break;
412 
413  case TMPL_TYPE_LIST:
414  break;
415 
416  case TMPL_TYPE_ATTR_UNRESOLVED:
417  cf_log_err(map->ci, "Unknown attribute %s", tmpl_attr_unresolved(map->lhs));
418  return -1;
419 
420  default:
421  cf_log_err(map->ci, "Left hand side of map must be an attribute or list, not a %s",
422  fr_table_str_by_value(tmpl_type_table, map->lhs->type, "<INVALID>"));
423  return -1;
424  }
425 
426  /*
427  * Sources we can use to get the name of the attribute
428  * we're retrieving from LDAP.
429  */
430  switch (map->rhs->type) {
431  case TMPL_TYPE_UNRESOLVED:
432  offset = -1;
433  if (fieldname2offset(inst, map->rhs->name, &offset) < 0) {
434  cf_log_err(map->ci, "Unknown field '%s'", map->rhs->name);
435  return -1;
436  }
437 
438  if ((type == FR_TYPE_INVALID) || (offset < 0)) break;
439 
440  /*
441  * Try to set the data type of the field. But if
442  * they map the same field to two different data
443  * types, that's an error.
444  */
445  if (inst->field_types[offset] == FR_TYPE_INVALID) {
446  inst->field_types[offset] = type;
447  break;
448  }
449 
450  if (inst->field_types[offset] != type) {
451  cf_log_err(map->ci, "Field '%s' maps to two different data types, making it impossible to parse it.", map->rhs->name);
452  return -1;
453  }
454 
455  break;
456 
457  case TMPL_TYPE_ATTR_UNRESOLVED:
458  cf_log_err(map->ci, "Unknown attribute %s", tmpl_attr_unresolved(map->rhs));
459  return -1;
460 
461  default:
462  cf_log_err(map->ci, "Right hand side of map must be a field name, not a %s",
463  fr_table_str_by_value(tmpl_type_table, map->rhs->type, "<INVALID>"));
464  return -1;
465  }
466 
467  /*
468  * Only some operators are allowed.
469  */
470  switch (map->op) {
471  case T_OP_SET:
472  case T_OP_EQ:
473  case T_OP_SUB:
474  case T_OP_ADD:
475  case T_OP_LT:
476  case T_OP_GT:
477  case T_OP_LE:
478  case T_OP_GE:
479  break;
480 
481  default:
482  cf_log_err(map->ci, "Operator \"%s\" not allowed for CSV mappings",
483  fr_table_str_by_value(fr_tokens_table, map->op, "<INVALID>"));
484  return -1;
485  }
486 
487  return 0;
488 }

490 /*
491  * Verify the result of the map.
492  */
493 static int csv_maps_verify(CONF_SECTION *cs, void *mod_inst, UNUSED void *proc_inst,
494  tmpl_t const *src, map_t const *maps)
495 {
496  map_t const *map;
497 
498  if (!src) {
499  cf_log_err(cs, "Missing key expansion");
500 
501  return -1;
502  }
503 
504  for (map = maps;
505  map != NULL;
506  map = map->next) {
507  map_t *unconst_map;
508 
509  memcpy(&unconst_map, &map, sizeof(map));
510 
511  /*
512  * This function doesn't change the map, so it's OK.
513  */
514  if (csv_map_verify(unconst_map, mod_inst) < 0) return -1;
515  }
516 
517  return 0;
518 }

520 /*
521  * Do any per-module initialization that is separate to each
522  * configured instance of the module. e.g. set up connections
523  * to external databases, read configuration files, set up
524  * dictionary entries, etc.
525  *
526  * If configuration information is given in the config section
527  * that must be referenced in later calls, store a handle to it
528  * in *instance otherwise put a null pointer there.
529  */
530 static int mod_bootstrap(void *instance, CONF_SECTION *conf)
531 {
532  rlm_csv_t *inst = instance;
533  int i;
534  char const *p;
535  char *q;
536  char *fields;
537 
538  inst->name = cf_section_name2(conf);
539  if (!inst->name) inst->name = cf_section_name1(conf);
540 
541  if (inst->delimiter[1]) {
542  cf_log_err(conf, "'delimiter' must be one character long");
543  return -1;
544  }
545 
546  if (inst->key) {
547  inst->key_data_type = tmpl_expanded_type(inst->key);
548  switch (inst->key_data_type) {
549  case FR_TYPE_VALUE:
550  break;
551 
552  case FR_TYPE_INVALID:
553  cf_log_err(conf, "Can't determine key data_type. 'key' must be an expression of type "
554  "xlat, attr, or data");
555  return -1;
556 
557  default:
558  cf_log_err(conf, "Invalid key data type '%s'",
559  fr_table_str_by_value(fr_value_box_type_table, inst->key_data_type, "<INVALID>"));
560  return -1;
561  }
562  } else {
563  inst->key_data_type = FR_TYPE_STRING;
564  }
565 
566  /*
567  * IP addresses go into tries. Everything else into binary tries.
568  */
569  if ((inst->key_data_type == FR_TYPE_IPV4_ADDR) || (inst->key_data_type == FR_TYPE_IPV4_PREFIX) ||
570  (inst->key_data_type == FR_TYPE_IPV6_ADDR) || (inst->key_data_type == FR_TYPE_IPV6_PREFIX)) {
571  MEM(inst->trie = fr_trie_alloc(inst));
572  } else {
573  MEM(inst->tree = rbtree_talloc_alloc(inst, csv_entry_cmp, rlm_csv_entry_t, NULL, 0));
574  }
575 
576  if ((*inst->index_field_name == ',') || (*inst->index_field_name == *inst->delimiter)) {
577  cf_log_err(conf, "Field names cannot begin with the '%c' character", *inst->index_field_name);
578  return -1;
579  }
580 
581  if (inst->delimiter[1] != '\0') {
582  cf_log_err(conf, "The 'delimiter' field MUST be one character long");
583  return -1;
584  }
585 
586  /*
587  * If there is a header in the file, then read that first.
588  */
589  if (inst->header) {
590  FILE *fp;
591  char buffer[8192];
592 
593  fp = fopen(inst->filename, "r");
594  if (!fp) {
595  cf_log_err(conf, "Error opening filename %s: %s", inst->filename, fr_syserror(errno));
596  return -1;
597  }
598 
599  p = fgets(buffer, sizeof(buffer), fp);
600  if (!p) {
601  error_eof:
602  cf_log_err(conf, "Error reading filename %s: Unexpected EOF", inst->filename);
603  fclose(fp);
604  return -1;
605  }
606 
607  q = strchr(buffer, '\n');
608  if (!q) goto error_eof;
609 
610  *q = '\0';
611 
612  /*
613  * Over-write whatever is in the config with the
614  * header from the file.
615  */
616  inst->fields = talloc_strdup(inst, buffer);
617  fclose(fp);
618  }
619 
620  /*
621  * Parse the field names AFTER opening the file. Because
622  * the field names might be taken from the header.
623  */
624  for (p = inst->fields; p != NULL; p = strchr(p + 1, *inst->delimiter)) {
625  inst->num_fields++;
626  }
627 
628  if (inst->num_fields < 2) {
629  cf_log_err(conf, "The CSV file MUST have at least a key field and data field");
630  return -1;
631  }
632 
633  MEM(inst->field_names = talloc_zero_array(inst, const char *, inst->num_fields));
634  MEM(inst->field_offsets = talloc_array(inst, int, inst->num_fields));
635  MEM(inst->field_types = talloc_array(inst, fr_type_t, inst->num_fields));
636 
637  for (i = 0; i < inst->num_fields; i++) {
638  inst->field_offsets[i] = -1; /* unused */
639  inst->field_types[i] = FR_TYPE_INVALID;
640  }
641 
642  /*
643  * Get a writable copy of the fields definition
644  */
645  MEM(fields = talloc_typed_strdup(inst, inst->fields));
646 
647  /*
648  * Mark up the field names. Note that they can be empty,
649  * in which case they don't map to anything.
650  */
651  inst->index_field = -1;
652 
653  /*
654  * Parse the field names
655  */
656  i = 0;
657  p = q = fields;
658  while (*q) {
659  bool last_field;
660 
661  /*
662  * Skip the field name
663  */
664  while (*q && (*q != *inst->delimiter)) {
665  if ((*q == '\'') || (*q == '"')) {
666  cf_log_err(conf, "Field %d name cannot have quotation marks.",
667  i + 1);
668  return -1;
669  }
670 
671  if (*q < ' ') {
672  *q = '\0';
673  break;
674  }
675 
676  if (isspace((int) *q)) {
677  cf_log_err(conf, "Field %d name cannot have spaces.",
678  i + 1);
679  return -1;
680  }
681 
682  q++;
683  }
684 
685  /*
686  * Check for the last field.
687  */
688  if (!*q) {
689  last_field = true;
690  } else {
691  *q = '\0';
692  last_field = false;
693  }
694 
695  /*
696  * Track which field is the key, and which fields
697  * map to which entries.
698  *
699  * Some fields are unused, so there isn't a 1-1
700  * mapping betweeen CSV file fields, and fields
701  * in the map.
702  */
703  if (last_field && (*p == ',') && (strcmp(p + 1, inst->index_field_name) == 0)) {
704  inst->index_field = i;
705  p++;
706 
707  inst->multiple_index_fields = true;
708 
709  } else if (strcmp(p, inst->index_field_name) == 0) {
710  inst->index_field = i;
711 
712  } else if ((*p == ',') || (*p == *inst->delimiter)) {
713  cf_log_err(conf, "Field names MUST NOT begin with the '%c' character", *p);
714  return -1;
715 
716  } else {
717  inst->field_offsets[i] = inst->used_fields++;
718  }
719 
720  /*
721  * Save the field names, even when the field names are empty.
722  */
723  inst->field_names[i] = p;
724 
725  if (last_field) break;
726 
727  q++;
728  i++;
729  p = q;
730  }
731 
732  if (inst->index_field < 0) {
733  cf_log_err(conf, "index_field '%s' does not appear in the list of field names",
734  inst->index_field_name);
735  return -1;
736  }
737 
738  /*
739  * Set the data type of the index field.
740  */
741  inst->field_types[inst->index_field] = inst->key_data_type;
742 
743  /*
744  * And register the `map csv <key> { ... }` function.
745  */
746  map_proc_register(inst, inst->name, mod_map_proc, csv_maps_verify, 0);
747 
748  return 0;

752 /** Instantiate the module
753  *
754  * Creates a new instance of the module reading parameters from a configuration section.
755  *
756  * @param conf to parse.
757  * @param instance configuration data.
758  * @return
759  * - 0 on success.
760  * - < 0 on failure.
761  */
762 static int mod_instantiate(void *instance, CONF_SECTION *conf)
763 {
764  rlm_csv_t *inst = instance;
765  CONF_SECTION *cs;
766  int lineno;
767  FILE *fp;
768  tmpl_rules_t parse_rules = {
769  .allow_foreign = true /* Because we don't know where we'll be called */
770  };
771  char buffer[8192];
772 
773  /*
774  * "update" without "key" is invalid, as we can't run the
775  * module.
776  *
777  * "key" without "update" means we just ignore the key.
778  */
779  cs = cf_section_find(conf, "update", CF_IDENT_ANY);
780  if (cs) {
781  if (!inst->key) {
782  cf_log_err(conf, "There is no 'key' defined for the 'update' section");
783  return -1;
784  }
785 
786  /*
787  * Parse the "update" section. This parsing includes
788  * setting inst->field_types[], base on the attribute
789  * mappings.
790  */
791  if (map_afrom_cs(inst, &inst->map, cs,
792  &parse_rules, &parse_rules, csv_map_verify, inst,
793  CSV_MAX_ATTRMAP) < 0) {
794  return -1;
795  }
796  } else if (inst->key) {
797  cf_log_warn(conf, "Ignoring 'key', as no 'update' section has been defined.");
798  }
799 
800  /*
801  * Re-open the file and read it all.
802  */
803  fp = fopen(inst->filename, "r");
804  if (!fp) {
805  cf_log_err(conf, "Error opening filename %s: %s", inst->filename, fr_syserror(errno));
806  return -1;
807  }
808  lineno = 1;
809 
810  /*
811  * If there is a header in the file, then read that first.
812  * This time we just ignore it.
813  */
814  if (inst->header) {
815  char *p = fgets(buffer, sizeof(buffer), fp);
816  if (!p) {
817  cf_log_err(conf, "Error reading filename %s: Unexpected EOF", inst->filename);
818  fclose(fp);
819  return -1;
820  }
821  lineno++;
822  }
823 
824  /*
825  * Read the rest of the file.
826  */
827  while (fgets(buffer, sizeof(buffer), fp) != NULL) {
828  if (!file2csv(conf, inst, lineno, buffer)) {
829  fclose(fp);
830  return -1;
831  }
832 
833  lineno++;
834  }
835  fclose(fp);
836 
837  return 0;
838 }

894 /** Perform a search and map the result of the search to server attributes
895  *
896  * @param[in] inst #rlm_csv_t.
897  * @param[in,out] request The current request.
898  * @param[in] key key to look for
899  * @param[in] maps Head of the map list.
900  * @return
901  * - #RLM_MODULE_NOOP no rows were returned.
902  * - #RLM_MODULE_UPDATED if one or more #fr_pair_t were added to the #request_t.
903  * - #RLM_MODULE_FAIL if an error occurred.
904  */
905 static rlm_rcode_t mod_map_apply(rlm_csv_t const *inst, request_t *request,
906  fr_value_box_t const *key, map_t const *maps)
907 {
908  rlm_rcode_t rcode = RLM_MODULE_UPDATED;
909  rlm_csv_entry_t *e;
910  map_t const *map;
911 
912  e = find_entry(inst, key);
913  if (!e) {
914  rcode = RLM_MODULE_NOOP;
915  goto finish;
916  }
917 
918 redo:
919  RINDENT();
920  for (map = maps;
921  map != NULL;
922  map = map->next) {
923  int field;
924  char *field_name;
925 
926  /*
927  * Avoid memory allocations if possible.
928  */
929  if (!tmpl_is_unresolved(map->rhs)) {
930  if (tmpl_aexpand(request, &field_name, request, map->rhs, NULL, NULL) < 0) {
931  REXDENT();
932  REDEBUG("Failed expanding RHS at %s", map->lhs->name);
933  rcode = RLM_MODULE_FAIL;
934  goto finish;
935  }
936  } else {
937  memcpy(&field_name, &map->rhs->name, sizeof(field_name)); /* const */
938  }
939 
940  field = fieldname2offset(inst, field_name, NULL);
941 
942  if (field_name != map->rhs->name) talloc_free(field_name);
943 
944  if (field < 0) {
945  REXDENT();
946  REDEBUG("No such field name %s", map->rhs->name);
947  rcode = RLM_MODULE_FAIL;
948  goto finish;
949  }
950 
951  /*
952  * Pass the raw data to the callback, which will
953  * create the VP and add it to the map.
954  */
955  if (map_to_request(request, map, csv_map_getvalue, e->data[field]) < 0) {
956  REXDENT();
957  rcode = RLM_MODULE_FAIL;
958  goto finish;
959  }
960  }
961 
962  REXDENT();
963 
964  if (e->next) {
965  e = e->next;
966  goto redo;
967  }
968 
969 finish:
970  return rcode;
971 }
