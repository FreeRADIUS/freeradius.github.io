76 static unlang_action_t eap_tls_success_with_prf(rlm_rcode_t *p_result, request_t *request, eap_session_t *eap_session)
77 {
78 #if OPENSSL_VERSION_NUMBER >= 0x10101000L
79  eap_tls_session_t *eap_tls_session = talloc_get_type_abort(eap_session->opaque, eap_tls_session_t);
80  fr_tls_session_t *tls_session = eap_tls_session->tls_session;
81 
82  /*
83  * Set the PRF label based on the TLS version negotiated
84  * in the handshake.
85  */
86  switch (SSL_SESSION_get_protocol_version(SSL_get_session(tls_session->ssl))) {
87  case SSL2_VERSION: /* Should never happen */
88  case SSL3_VERSION: /* Should never happen */
89  fr_assert(0);
90  RETURN_MODULE_INVALID;
91 
92  case TLS1_VERSION:
93  case TLS1_1_VERSION:
94  case TLS1_2_VERSION:
95 #endif
96  {
97  static char const keying_prf_label[] = "client EAP encryption";
98 
99  if (eap_tls_success(request, eap_session,
100  keying_prf_label, sizeof(keying_prf_label) - 1,
101  NULL, 0) < 0) RETURN_MODULE_FAIL;
102  }
103 #if OPENSSL_VERSION_NUMBER >= 0x10101000L
104  break;
105 
106  case TLS1_3_VERSION:
107  default:
108  {
109  static char const keying_prf_label[] = "EXPORTER_EAP_TLS_Key_Material";
110  static char const sessid_prf_label[] = "EXPORTER_EAP_TLS_Method-Id";
111 
112  if (eap_tls_success(request, eap_session,
113  keying_prf_label, sizeof(keying_prf_label) - 1,
114  sessid_prf_label, sizeof(sessid_prf_label) - 1) < 0) RETURN_MODULE_FAIL;
115  }
116  break;
117  }
118 #endif
119  RETURN_MODULE_OK;
120 }

139 static unlang_action_t eap_tls_virtual_server(rlm_rcode_t *p_result, rlm_eap_tls_t *inst, request_t *request,
140  eap_session_t *eap_session)
141 {
142  CONF_SECTION *server_cs;
143  CONF_SECTION *section;
144  fr_pair_t *vp;
145 
146  /* set the virtual server to use */
147  vp = fr_pair_find_by_da(&request->control_pairs, attr_virtual_server);
148  if (vp) {
149  server_cs = virtual_server_find(vp->vp_strvalue);
150  if (!server_cs) {
151  REDEBUG2("Virtual server \"%pV\" not found", &vp->data);
152  error:
153  eap_tls_fail(request, eap_session);
154  RETURN_MODULE_INVALID;
155  }
156  } else {
157  server_cs = virtual_server_find(inst->virtual_server);
158  fr_assert(server_cs);
159  }
160 
161  section = cf_section_find(server_cs, "recv", "Access-Request");
162  if (!section) {
163  REDEBUG2("Failed finding 'recv Access-Request { ... }' section of virtual server %s",
164  cf_section_name2(server_cs));
165  goto error;
166  }
167 
168  if (!unlang_section(section)) {
169  REDEBUG("Failed to find pre-compiled unlang for section %s %s { ... }",
170  cf_section_name1(server_cs), cf_section_name2(server_cs));
171  goto error;
172  }
173 
174  RDEBUG2("Validating certificate");
175 
176  /*
177  * Catch the interpreter on the way back up the stack
178  */
179  if (unlang_interpret_push_function(request, NULL, eap_tls_virtual_server_result, eap_session) < 0) {
180  RETURN_MODULE_FAIL;
181  }
182 
183  /*
184  * Push unlang instructions for the virtual server section
185  */
186  if (unlang_interpret_push_section(request, section, RLM_MODULE_NOOP, UNLANG_SUB_FRAME) < 0) {
187  RETURN_MODULE_FAIL;
188  }
189 
190  *p_result = RLM_MODULE_YIELD;
191 
192  return UNLANG_ACTION_YIELD;
193 }

195 static unlang_action_t mod_process(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
196 {
197  rlm_eap_tls_t *inst = talloc_get_type_abort(mctx->instance, rlm_eap_tls_t);
198  eap_tls_status_t status;
199 
200  eap_session_t *eap_session = eap_session_get(request->parent);
201  eap_tls_session_t *eap_tls_session = talloc_get_type_abort(eap_session->opaque, eap_tls_session_t);
202  fr_tls_session_t *tls_session = eap_tls_session->tls_session;
203 
204  status = eap_tls_process(request, eap_session);
205  if ((status == EAP_TLS_INVALID) || (status == EAP_TLS_FAIL)) {
206  REDEBUG("[eap-tls process] = %s", fr_table_str_by_value(eap_tls_status_table, status, "<INVALID>"));
207  } else {
208  RDEBUG2("[eap-tls process] = %s", fr_table_str_by_value(eap_tls_status_table, status, "<INVALID>"));
209  }
210 
211  switch (status) {
212  /*
213  * EAP-TLS handshake was successful, return an
214  * EAP-TLS-Success packet here.
215  *
216  * If a virtual server was configured, check that
217  * it accepts the certificates, too.
218  */
219  case EAP_TLS_ESTABLISHED:
220  if (inst->virtual_server) return eap_tls_virtual_server(p_result, inst, request, eap_session);
221  return eap_tls_success_with_prf(p_result, request, eap_session);
222 
223 
224  /*
225  * The TLS code is still working on the TLS
226  * exchange, and it's a valid TLS request.
227  * do nothing.
228  */
229  case EAP_TLS_HANDLED:
230  RETURN_MODULE_HANDLED;
231 
232  /*
233  * Handshake is done, proceed with decoding tunneled
234  * data.
235  */
236  case EAP_TLS_RECORD_RECV_COMPLETE:
237  REDEBUG("Received unexpected tunneled data after successful handshake");
238  eap_tls_fail(request, eap_session);
239 
240  RETURN_MODULE_INVALID;
241 
242  /*
243  * Anything else: fail.
244  *
245  * Also, remove the session from the cache so that
246  * the client can't re-use it.
247  */
248  default:
249  fr_tls_cache_deny(tls_session);
250 
251  RETURN_MODULE_REJECT;
252  }
253 }

258 static unlang_action_t mod_session_init(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
259 {
260  rlm_eap_tls_t *inst = talloc_get_type_abort(mctx->instance, rlm_eap_tls_t);
261  eap_session_t *eap_session = eap_session_get(request->parent);
262  eap_tls_session_t *eap_tls_session;
263 
264  fr_pair_t *vp;
265  bool client_cert;
266 
267  eap_session->tls = true;
268 
269  /*
270  * EAP-TLS-Require-Client-Cert attribute will override
271  * the require_client_cert configuration option.
272  */
273  vp = fr_pair_find_by_da(&request->control_pairs, attr_eap_tls_require_client_cert);
274  if (vp) {
275  client_cert = vp->vp_uint32 ? true : false;
276  } else {
277  client_cert = inst->req_client_cert;
278  }
279 
280  /*
281  * EAP-TLS always requires a client certificate.
282  */
283  eap_session->opaque = eap_tls_session = eap_tls_session_init(request, eap_session, inst->tls_conf, client_cert);
284  if (!eap_tls_session) RETURN_MODULE_FAIL;
285 
286  eap_tls_session->include_length = inst->include_length;
287 
288  /*
289  * TLS session initialization is over. Now handle TLS
290  * related handshaking or application data.
291  */
292  if (eap_tls_start(request, eap_session) < 0) {
293  talloc_free(eap_tls_session);
294  RETURN_MODULE_FAIL;
295  }
296 
297  eap_session->process = mod_process;
298 
299  RETURN_MODULE_HANDLED;
300 }

341  */
342 static int mod_section_compile(eap_tls_actions_t *actions, CONF_SECTION *server_cs)
343 {
344  int found;
345  tmpl_rules_t parse_rules;
346 
347  if (!fr_cond_assert(server_cs)) return -1;
348 
349  memset(&parse_rules, 0, sizeof(parse_rules));
350  parse_rules.dict_def = dict_freeradius;
351 
352  found = virtual_server_compile_sections(server_cs, compile_list, &parse_rules, actions);
353  if (found < 0) return -1;
354 
355  /*
356  * Warn if we couldn't find any actions.
357  */
358  if (!found) {
359  cf_log_warn(server_cs, "No \"eap-tls\" actions found in virtual server \"%s\"",
360  cf_section_name2(server_cs));
361  }
362 
363  return 0;
364 }
