36 /** Attribute extension - Holds children for an attribute
37  *
38  * Children are possible for:
39  *
40  * #FR_TYPE_TLV, #FR_TYPE_VENDOR, #FR_TYPE_VSA, #FR_TYPE_STRUCT
41  *
42  * *or* where the parent->parent->type is
43  * #FR_TYPE_STRUCT, and "parent" is a "key"
44  * field. Note that these attributes therefore
45  * cannot have VALUEs, as the child defines their
46  * VALUE. See dict_attr_can_have_children() for details.

101 /* Retrieve an extension structure for a dictionary attribute
102  *
103  * @param[in] da to retrieve structure from.
104  * @param[in] ext to retrieve.
105  * @return
106  * - NULL if the extension wasn't found.
107  * - A pointer to the start of the extension.
108  */
109 static inline void *fr_dict_attr_ext(fr_dict_attr_t const *da, fr_dict_attr_ext_t ext)
110 {
111  if (!da->ext[ext]) return NULL;
112 
113  return FR_EXT_PTR(da, ext, ext);
114 }

116 /** Return whether a da has a given extension or not

124 static inline bool fr_dict_attr_has_ext(fr_dict_attr_t const *da, fr_dict_attr_ext_t ext)

163 /** Return the vendor number for an attribute
164  *
165  * @param[in] da The dictionary attribute to find the
166  * vendor for.
167  * @return
168  * - 0 this isn't a vendor specific attribute.
169  * - The vendor PEN.
170  */
171 static inline uint32_t fr_dict_vendor_num_by_da(fr_dict_attr_t const *da)
172 {
173  fr_dict_attr_ext_vendor_t *ext;
174 
175  if (da->type == FR_TYPE_VENDOR) return da->attr;
176 
177  ext = fr_dict_attr_ext(da, FR_DICT_ATTR_EXT_VENDOR);
178  if (!ext || !ext->vendor) return 0;
179 
180  return ext->vendor->attr;
181 }
