170 /*
171  * Return negative numbers to put 'one' at the top of the heap.
172  * Return positive numbers to put 'two' at the top of the heap.
173  */
174 static int8_t pending_packet_cmp(void const *one, void const *two)
175 {
176  fr_io_pending_packet_t const *a = talloc_get_type_abort_const(one, fr_io_pending_packet_t);
177  fr_io_pending_packet_t const *b = talloc_get_type_abort_const(two, fr_io_pending_packet_t);
178  int ret;
179 
180  /*
181  * Higher priority elements are larger than lower
182  * priority elements. So if "a" is larger than "b", we
183  * wish to prefer "a".
184  */
185  ret = COMPARE_PREFER_LARGER(a->priority, b->priority);
186  if (ret != 0) return ret;
187 
188  /*
189  * Smaller numbers mean packets were received earlier.
190  * We want to process packets in time order. So if "a"
191  * is smaller than "b", we wish to prefer "a".
192  *
193  * After that, it doesn't really matter what order the
194  * packets go in. Since we'll never have two identical
195  * "recv_time" values, the code should never get here.
196  */
197  return COMPARE_PREFER_SMALLER(a->recv_time, b->recv_time);
198 }

222 static int8_t address_cmp(void const *one, void const *two)
223 {
224  int ret;
225  fr_io_address_t const *a = talloc_get_type_abort_const(one, fr_io_address_t);
226  fr_io_address_t const *b = talloc_get_type_abort_const(two, fr_io_address_t);;
227 
228  ret = STABLE_COMPARE(a->socket.inet.src_port, b->socket.inet.src_port);
229  if (ret != 0) return ret;
230 
231  ret = STABLE_COMPARE(a->socket.inet.dst_port, b->socket.inet.dst_port);
232  if (ret != 0) return ret;
233 
234  ret = STABLE_COMPARE(a->socket.inet.ifindex, b->socket.inet.ifindex);
235  if (ret != 0) return ret;
236 
237  ret = fr_ipaddr_cmp(&a->socket.inet.src_ipaddr, &b->socket.inet.src_ipaddr);
238  if (ret != 0) return ret;
239 
240  return fr_ipaddr_cmp(&a->socket.inet.dst_ipaddr, &b->socket.inet.dst_ipaddr);
241 }

319 static fr_io_pending_packet_t *pending_packet_pop(fr_io_thread_t *thread)
320 {
321  fr_io_client_t *client;
322  fr_io_pending_packet_t *pending;
323 
324  client = fr_heap_pop(thread->pending_clients);
325  if (!client) {
326  /*
327  * 99% of the time we don't have pending clients.
328  * So we might as well free this, so that the
329  * caller doesn't keep checking us for every packet.
330  */
331  talloc_free(thread->pending_clients);
332  thread->pending_clients = NULL;
333  return NULL;
334  }
335 
336  pending = fr_heap_pop(client->pending);
337  fr_assert(pending != NULL);
338 
339  /*
340  * If the client has more packets pending, add it back to
341  * the heap.
342  */
343  if (fr_heap_num_elements(client->pending) > 0) {
344  if (fr_heap_insert(thread->pending_clients, client) < 0) {
345  fr_assert(0 == 1);
346  }
347  }
348 
349  fr_assert(thread->num_pending_packets > 0);
350  thread->num_pending_packets--;
351 
352  return pending;
353 }

407 /** Count the number of connections used by active clients.
408  *
409  * Unfortunately, we also count NAK'd connections, too, even if they
410  * are closed. The alternative is to walk through all connections
411  * for each client, which would be a long time.
412  */
413 static int count_connections(void *ctx, UNUSED uint8_t const *key, UNUSED size_t keylen, void *data)
414 {
415  fr_io_client_t *client = talloc_get_type_abort(data, fr_io_client_t);
416  int connections;
417 
418  /*
419  * This client has no connections, skip the mutex lock.
420  */
421  if (!client->ht) return 0;
422 
423  fr_assert(client->use_connected);
424 
425  pthread_mutex_lock(&client->mutex);
426  connections = fr_hash_table_num_elements(client->ht);
427  pthread_mutex_unlock(&client->mutex);
428 
429  *((uint32_t *) ctx) += connections;
430 
431  return 0;
432 }

453 /** Create a new connection.
454  *
455  * Called ONLY from the master socket.
456  */
457 static fr_io_connection_t *fr_io_connection_alloc(fr_io_instance_t const *inst,
458  fr_io_thread_t *thread,
459  fr_io_client_t *client, int fd,
460  fr_io_address_t *address,
461  fr_io_connection_t *nak)
462 {
463  int ret;
464  fr_io_connection_t *connection;
465  dl_module_inst_t *dl_inst = NULL;
466  fr_listen_t *li;
467  RADCLIENT *radclient;
468 
469  /*
470  * Reload the app_io module as a "new" library. This
471  * causes the link count for the library to be correct.
472  * It also allocates a new instance data for it, too.
473  * Passing CONF_SECTION of NULL ensures that there's no
474  * config for it, as we'll just clone it's contents from
475  * the original. It also means that detach should be
476  * called when the instance data is freed.
477  */
478  if (!nak) {
479  if (inst->max_connections) {
480  /*
481  * We've hit the connection limit. Walk
482  * over all clients with connections, and
483  * count the number of connections used.
484  */
485  if (thread->num_connections >= inst->max_connections) {
486  thread->num_connections = 0;
487 
488  (void) fr_trie_walk(thread->trie, &thread->num_connections, count_connections);
489 
490  if ((thread->num_connections + 1) >= inst->max_connections) {
491  DEBUG("Too many open connections. Ignoring dynamic client %s. Discarding packet.", client->radclient->shortname);
492  return NULL;
493  }
494  }
495  }
496 
497  if (dl_module_instance(NULL, &dl_inst, NULL, inst->dl_inst, inst->transport, DL_MODULE_TYPE_SUBMODULE) < 0) {
498  DEBUG("Failed to find proto_%s_%s", inst->app->name, inst->transport);
499  return NULL;
500  }
501  fr_assert(dl_inst != NULL);
502  } else {
503  dl_inst = talloc_init_const("nak");
504  }
505 
506  MEM(connection = talloc_zero(dl_inst, fr_io_connection_t));
507  MEM(connection->address = talloc_memdup(connection, address, sizeof(*address)));
508  (void) talloc_set_name_const(connection->address, "fr_io_address_t");
509 
510  connection->parent = client;
511  connection->dl_inst = dl_inst;
512 
513  MEM(connection->client = talloc_named(NULL, sizeof(fr_io_client_t), "fr_io_client_t"));
514  memset(connection->client, 0, sizeof(*connection->client));
515 
516  MEM(connection->client->radclient = radclient = radclient_clone(connection->client, client->radclient));
517 
518  talloc_set_destructor(connection->client, _client_free);
519  talloc_set_destructor(connection, connection_free);
520 
521  connection->client->pending_id = -1;
522  connection->client->alive_id = -1;
523  connection->client->connection = connection;
524 
525  /*
526  * Create the packet tracking table for this client.
527  *
528  * #todo - unify the code with static clients?
529  */
530  if (inst->app_io->track_duplicates) {
531  MEM(connection->client->table = rbtree_talloc_alloc(client, track_connected_cmp, fr_io_track_t,
532  NULL, RBTREE_FLAG_NONE));
533  }
534 
535  /*
536  * Set this radclient to be dynamic, and active.
537  */
538  radclient->dynamic = true;
539  radclient->active = true;
540 
541  /*
542  * address->socket.inet.client points to a "static" client. We want
543  * to clean up everything associated with the connection
544  * when it closes. So we need to point to our own copy
545  * of the client here.
546  */
547  connection->address->radclient = connection->client->radclient;
548  connection->client->inst = inst;
549  connection->client->thread = thread;
550 
551  /*
552  * Create a heap for packets which are pending for this
553  * client.
554  */
555  MEM(connection->client->pending = fr_heap_alloc(connection->client, pending_packet_cmp,
556  fr_io_pending_packet_t, heap_id));
557 
558  /*
559  * Clients for connected sockets are always a /32 or /128.
560  */
561  connection->client->src_ipaddr = address->socket.inet.src_ipaddr;
562  connection->client->network = address->socket.inet.src_ipaddr;
563 
564  /*
565  * Don't initialize mutex or hash table.
566  * Connections cannot spawn other connections.
567  */
568 
569  /*
570  * If this client state is pending, then the connection
571  * state is pending, too. That allows NAT gateways to be
572  * defined dynamically, AND for them to have multiple
573  * connections, each with a different client. This
574  * allows for different shared secrets to be used for
575  * different connections. Once the client gets defined
576  * for this connection, it will be either "connected" or
577  * not. If connected, then the parent client remains
578  * PENDING. Otherwise, the parent client is moved to
579  * DYNAMIC
580  *
581  * If this client state is static or dynamic,
582  * then we're just using connected sockets behind
583  * that client. The connections here all use the
584  * same shared secret, but they use different
585  * sockets, so they allow for sharing of IO
586  * across CPUs / threads.
587  */
588  switch (client->state) {
589  case PR_CLIENT_PENDING:
590  connection->client->state = PR_CLIENT_PENDING;
591 
592  /*
593  * Needed for rlm_radius, which refuses to proxy packets
594  * that define a dynamic client.
595  */
596  radclient->active = false;
597  break;
598 
599  case PR_CLIENT_STATIC:
600  case PR_CLIENT_DYNAMIC:
601  connection->client->state = PR_CLIENT_CONNECTED;
602  break;
603 
604  case PR_CLIENT_INVALID:
605  case PR_CLIENT_NAK:
606  case PR_CLIENT_CONNECTED:
607  fr_assert(0 == 1);
608  talloc_free(dl_inst);
609  return NULL;
610  }
611 
612  if (!nak) {
613  /*
614  * Get the child listener.
615  */
616  MEM(li = connection->child = talloc(connection, fr_listen_t));
617  memcpy(li, thread->listen, sizeof(*li));
618 
619  /*
620  * Glue in the actual app_io
621  */
622  li->connected = true;
623  li->app_io = thread->child->app_io;
624  li->thread_instance = connection;
625  li->app_io_instance = dl_inst->data;
626  li->track_duplicates = thread->child->app_io->track_duplicates;
627 
628  /*
629  * Create writable thread instance data.
630  */
631  connection->child->thread_instance = talloc_zero_array(NULL, uint8_t,
632  inst->app_io->thread_inst_size);
633  talloc_set_destructor(connection->child, fr_io_listen_free);
634  talloc_set_name(connection->child->thread_instance, "proto_%s_thread_t",
635  inst->app_io->name);
636 
637  /*
638  * This is "const", and the user can't
639  * touch it. So we just re-use the same
640  * configuration everywhere.
641  */
642  connection->child->app_io_instance = inst->app_io_instance;
643 
644  /*
645  * Create the listener, based on our listener.
646  */
647  MEM(li = connection->listen = talloc(connection, fr_listen_t));
648 
649  /*
650  * Note that our instance is effectively 'const'.
651  *
652  * i.e. we can't add things to it. Instead, we have to
653  * put all variable data into the connection.
654  */
655  memcpy(li, thread->listen, sizeof(*li));
656 
657  /*

673  DEBUG("Failed setting connection for socket.");
674  talloc_free(dl_inst);

679  * UDP sockets: open a new socket, and then
680  * connect it to the client. This emulates the
681  * behavior of accept().

704  close(fd);

706  talloc_free(dl_inst);
707  return NULL;
708  }
709  } else {
710  connection->child->fd = fd;
711  }
712 
713  /*
714  * Set the new FD, and get the module to set it's connection name.
715  */
716  if (inst->app_io->fd_set(connection->child, fd) < 0) {
717  DEBUG3("Failed setting FD to %s", inst->app_io->name);
718  close(fd);
719  return NULL;
720  }
721 
722  li->fd = fd;

725  connection->name = fr_asprintf(connection, "proto_%s from client %pV port "
726  "%u to server %pV port %u",
727  inst->app_io->name,
728  fr_box_ipaddr(connection->address->socket.inet.src_ipaddr),
729  connection->address->socket.inet.src_port,
730  fr_box_ipaddr(connection->address->socket.inet.dst_ipaddr),
731  connection->address->socket.inet.dst_port);
732  } else {
733  connection->name = inst->app_io->get_name(connection->child);
734  }
735

737  * Set the names for the listeners.
738  */
739  connection->listen->name = connection->name;
740  connection->child->name = connection->name;
741  }
742

744  * Add the connection to the set of connections for this
745  * client.
746  */

748  if (nak) (void) fr_hash_table_delete(client->ht, nak);
749  ret = fr_hash_table_insert(client->ht, connection);
750  client->ready_to_delete = false;
751  pthread_mutex_unlock(&client->mutex);
752 
753  if (ret < 0) {
754  ERROR("proto_%s - Failed inserting connection into tracking table. Closing it, and discarding all packets for connection %s.", inst->app_io->name, connection->name);
755  goto cleanup;
756  }
757 
758  /*
759  * It's a NAK client. Set the state to NAK, and don't
760  * add it to the scheduler.
761  */
762  if (nak) {
763  connection->name = talloc_strdup(connection, nak->name);
764  connection->client->state = PR_CLIENT_NAK;
765  connection->el = nak->el;
766  return connection;
767  }
768

770  connection->nr = fr_schedule_listen_add(thread->sc, connection->listen);
771  if (!connection->nr) {
772  ERROR("proto_%s - Failed inserting connection into scheduler. Closing it, and diuscarding all packets for connection %s.", inst->app_io->name, connection->name);
773  pthread_mutex_lock(&client->mutex);
774  (void) fr_hash_table_delete(client->ht, connection);
775  pthread_mutex_unlock(&client->mutex);
776 
777  cleanup:
778  talloc_free(dl_inst);
779  return NULL;
780  }

784  * NOT decrement this counter when a connection
785  * closes, as the close is done in a child
786  * thread. Instead, we just let counter hit the
787  * limit, and then walk over the clients to reset
788  * the count.
789  */
790  thread->num_connections++;
791 
792  return connection;

846 static fr_io_track_t *fr_io_track_add(fr_io_client_t *client,
847  fr_io_address_t *address,
848  uint8_t const *packet, size_t packet_len,
849  fr_time_t recv_time, bool *is_dup)
850 {
851  size_t len;
852  fr_io_track_t *track, *old;
853  fr_io_address_t *my_address;
854 
855  *is_dup = false;
856 
857  /*
858  * Allocate a new tracking structure. Most of the time
859  * there are no duplicates, so this is fine.
860  */
861  MEM(track = talloc_zero_pooled_object(client, fr_io_track_t, 1, sizeof(*track) + sizeof(track->address) + 64));
862  MEM(track->address = my_address = talloc_zero(track, fr_io_address_t));
863 
864  memcpy(my_address, address, sizeof(*address));
865  my_address->radclient = client->radclient;
866 
867  track->client = client;
868  if (client->connection) {
869  track->address = client->connection->address;
870  }
871 
872  track->timestamp = recv_time;
873  track->packets = 1;
874 
875  /*
876  * We're not tracking duplicates, so just return the
877  * tracking entry. This tracks src/dst IP/port, client,
878  * receive time, etc.
879  */
880  if (!client->inst->app_io->track_duplicates) {
881  talloc_set_destructor(track, track_free);
882  return track;
883  }
884 
885  /*
886  * We are checking for duplicates, see if there is a dup
887  * already in the tree.
888  */
889  track->packet = client->inst->app_io->track(track, packet, packet_len);
890  if (!track->packet) {
891  talloc_free(track);
892  return NULL;
893  }
894 
895  /*
896  * No existing duplicate. Return the new tracking entry.
897  */
898  old = rbtree_finddata(client->table, track);
899  if (!old) goto do_insert;
900 
901  fr_assert(old->client == client);
902 
903  /*
904  * It cannot be both in the free list and in the tracking table.
905  *
906  * 2020-08-17, this assertion fails randomly in travis.
907  * Which means that "track" was in the free list, *and*
908  * in the rbtree.
909  */
910  fr_assert(old != track);
911 
912  /*
913  * The new packet has the same dedup fields as the old
914  * one, BUT it may be a conflicting packet. Check for
915  * that via a simple memcmp().
916  *
917  * It's an exact duplicate. Drop the new one and
918  * use the old one.
919  *
920  * If there's a cached reply, the caller will take care
921  * of sending it to the network layer.
922  */
923  len = talloc_array_length(old->packet);
924  if ((len == talloc_array_length(track->packet)) &&
925  (memcmp(old->packet, track->packet, len) == 0)) {
926  fr_assert(old != track);
927 
928  /*
929  * Ignore duplicates while the client is
930  * still pending.
931  */
932  if (client->state == PR_CLIENT_PENDING) {
933  DEBUG("Ignoring duplicate packet while client %s is still pending dynamic definition",
934  client->radclient->shortname);
935  return NULL;
936  }
937 
938  *is_dup = true;
939  old->packets++;
940  talloc_free(track);
941 
942  /*
943  * Retransmits can sit in the outbound queue for
944  * a while. We don't want to time out this
945  * struct while the packet is in the outbound
946  * queue.
947  */
948  if (old->ev) (void) fr_event_timer_delete(&old->ev);
949  return old;
950  }
951 
952  /*
953  * Else it's a conflicting packet. Which is OK if we
954  * already have a reply. We just delete the old entry,
955  * and insert the new one.
956  *
957  * If there's no reply, then the old request is still
958  * "live". Delete the old one from the tracking tree,
959  * and return the new one.
960  */
961  if (old->reply_len || old->do_not_respond) {
962  talloc_free(old);
963 
964  } else {
965  fr_assert(client == old->client);
966 
967  if (!rbtree_deletebydata(client->table, old)) {
968  fr_assert(0);
969  }
970  if (old->ev) (void) fr_event_timer_delete(&old->ev);
971 
972  talloc_set_destructor(old, track_free);
973 
974  old->discard = true; /* don't send any reply, there's nowhere for it to go */
975  }
976 
977 do_insert:
978  if (!rbtree_insert(client->table, track)) {
979  fr_assert(0);
980  }
981 
982  talloc_set_destructor(track, track_dedup_free);
983  return track;
984 }

992  * Note that we don't check timestamps, replies, etc. If

1089  uint8_t *buffer, size_t buffer_len, size_t *leftover, uint32_t *priority, bool *is_dup)
1090 {

1124  */

1851 static void client_expiry_timer(fr_event_list_t *el, fr_time_t now, void *uctx)
1852 {
1853  fr_io_client_t *client = talloc_get_type_abort(uctx, fr_io_client_t);
1854  fr_io_instance_t const *inst;
1855  fr_io_connection_t *connection;
1856  fr_time_delta_t delay;
1857  int packets, connections;
1858 
1859  /*
1860  * No event list? We don't need to expire the client.
1861  */
1862  if (!el) return;
1863 
1864  DEBUG("TIMER - checking status of client %s", client->radclient->shortname);
1865 
1866  // @todo - print out what we plan on doing next
1867  connection = client->connection;
1868  inst = client->inst;
1869 
1870  fr_assert(client->state != PR_CLIENT_STATIC);
1871 
1872  /*
1873  * Called from the read or write functions with
1874  * now==0, to signal that we have to *set* the timer.
1875  */
1876  if (!now) {
1877  switch (client->state) {
1878  case PR_CLIENT_CONNECTED:
1879  fr_assert(connection != NULL);
1880  delay = inst->idle_timeout;
1881  break;
1882 
1883  case PR_CLIENT_DYNAMIC:
1884  delay = inst->idle_timeout;
1885  break;
1886 
1887  case PR_CLIENT_NAK:
1888  fr_assert(!connection);
1889  delay = inst->nak_lifetime;
1890  break;
1891 
1892  default:
1893  fr_assert(0 == 1);
1894  return;
1895  }
1896 
1897  goto reset_timer;
1898  }
1899 
1900  /*
1901  * Count active packets AND pending packets.
1902  */
1903  packets = client->packets;
1904  if (client->pending) packets += fr_heap_num_elements(client->pending);
1905 
1906  /*
1907  * It's a negative cache entry. Just delete it.
1908  */
1909  if (client->state == PR_CLIENT_NAK) {
1910  delete_client:
1911  fr_assert(packets == 0);
1912 
1913  /*
1914  * It's a connected socket. Remove it from the
1915  * parents list of connections, and delete it.
1916  */
1917  if (connection) {
1918  fr_io_client_t *parent = connection->parent;
1919 
1920  pthread_mutex_lock(&parent->mutex);
1921  (void) fr_hash_table_delete(parent->ht, connection);
1922  pthread_mutex_unlock(&parent->mutex);
1923 
1924  /*
1925  * Mark the connection as dead, and tell
1926  * the network side to stop reading from
1927  * it.
1928  */
1929  connection->dead = true;
1930  fr_network_listen_read(connection->nr, connection->listen);
1931  return;
1932  }
1933 
1934  talloc_free(client);
1935  return;
1936  }
1937 
1938  /*
1939  * It's a dynamically defined client. If no one is using
1940  * it, clean it up after an idle timeout.
1941  */
1942  if ((client->state == PR_CLIENT_DYNAMIC) ||
1943  (client->state == PR_CLIENT_CONNECTED)) {
1944  if (packets > 0) {
1945  client->ready_to_delete = false;
1946  return;
1947  }
1948 
1949  /*
1950  * No packets, check / set idle timeout.
1951  */
1952  goto idle_timeout;
1953  }
1954 
1955  /*
1956  * The client is pending definition. It's either a
1957  * dynamic client which has timed out, OR it's a
1958  * "place-holder" client for connected sockets.
1959  */
1960  fr_assert(client->state == PR_CLIENT_PENDING);
1961 
1962  /*
1963  * This is a dynamic client pending definition.
1964  * But it's taken too long to define, so we just
1965  * delete the client, and all packets for it. A
1966  * new packet will cause the dynamic definition
1967  * to be run again.
1968  */
1969  if (!client->use_connected) {
1970  if (!packets) {
1971  goto delete_client;
1972  }
1973 
1974  /*
1975  * Tell the writer to NOT dynamically define the
1976  * client. We've run into a problem. Then,
1977  * return. The writer will take care of calling
1978  * us again when it notices that a PENDING client
1979  * is ready to delete.
1980  *
1981  * TBH... that shouldn't happen? We should rely
1982  * on the write to do this all of the time...
1983  */
1984  client->ready_to_delete = true;
1985  return;
1986  }
1987 
1988  fr_assert(!connection);
1989  fr_assert(client->ht != NULL);
1990 
1991  /*
1992  * Find out how many connections are using this
1993  * client.
1994  */
1995  pthread_mutex_lock(&client->mutex);
1996  connections = fr_hash_table_num_elements(client->ht);
1997  pthread_mutex_unlock(&client->mutex);
1998 
1999  /*
2000  * No connections are using this client. If
2001  * we've passed the idle timeout, then just
2002  * delete it. Otherwise, set an idle timeout (as
2003  * above);
2004  */
2005  if (!connections) {
2006 idle_timeout:
2007  /*
2008  * We didn't receive any packets during the
2009  * idle_timeout, just delete it.
2010  */
2011  if (client->ready_to_delete) {
2012  if (connection) {
2013  DEBUG("proto_%s - idle timeout for connection %s", inst->app_io->name, connection->name);
2014  } else {
2015  DEBUG("proto_%s - idle timeout for client %s", inst->app_io->name, client->radclient->shortname);
2016  }
2017  goto delete_client;
2018  }
2019 
2020  /*
2021  * No packets and no idle timeout set, go set
2022  * idle timeut.
2023  */
2024  client->ready_to_delete = true;
2025  delay = inst->idle_timeout;
2026  goto reset_timer;
2027  }
2028 
2029  /*
2030  * There are live sub-connections. Poll again after a
2031  * long period of time. Once all of the connections are
2032  * closed, we can then delete this client.
2033  *
2034  * @todo - maybe just leave it? we want to be able to
2035  * clean up this client after a while tho... especially
2036  * if the total number of clients is limited.
2037  */
2038  client->ready_to_delete = false;
2039  delay = inst->check_interval;
2040 
2041 reset_timer:
2042  if (fr_event_timer_in(client, el, &client->ev,
2043  delay, client_expiry_timer, client) < 0) {
2044  ERROR("proto_%s - Failed adding timeout for dynamic client %s. It will be permanent!",
