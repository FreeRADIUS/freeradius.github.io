61 /** Create a new memcached handle
62  *
63  */
64 static void *mod_conn_create(TALLOC_CTX *ctx, void *instance, fr_time_delta_t timeout)
65 {
66  rlm_cache_memcached_t *driver = instance;
67  rlm_cache_memcached_handle_t *mandle;
68 
69  memcached_st *sandle;
70  memcached_return_t ret;
71 
72  sandle = memcached(driver->options, talloc_array_length(driver->options) -1);
73  if (!sandle) {
74  ERROR("Failed creating memcached connection");
75 
76  return NULL;
77  }
78 
79  ret = memcached_behavior_set(sandle, MEMCACHED_BEHAVIOR_CONNECT_TIMEOUT, fr_time_delta_to_msec(timeout));
80  if (ret != MEMCACHED_SUCCESS) {
81  ERROR("%s: %s", memcached_strerror(sandle, ret), memcached_last_error_message(sandle));
82  error:
83  memcached_free(sandle);
84  return NULL;
85  }
86 
87  ret = memcached_version(sandle);
88  if (ret != MEMCACHED_SUCCESS) {
89  ERROR("%s: %s", memcached_strerror(sandle, ret), memcached_last_error_message(sandle));
90  goto error;
91  }
92 
93  mandle = talloc_zero(ctx, rlm_cache_memcached_handle_t);
94  mandle->handle = sandle;
95  talloc_set_destructor(mandle, _mod_conn_free);
96 
97  return mandle;
98 }

110 static int mod_instantiate(void *instance, CONF_SECTION *conf)
111 {
112  rlm_cache_memcached_t *driver = instance;
113  memcached_return_t ret;
114  char buffer[256];

116 
117  fr_assert(config);
118 
119  snprintf(buffer, sizeof(buffer), "rlm_cache (%s)", config->name);
120 
121  ret = libmemcached_check_configuration(driver->options, talloc_array_length(driver->options) -1,
122  buffer, sizeof(buffer));
123  if (ret != MEMCACHED_SUCCESS) {
124  ERROR("%s", buffer);
125  return -1;

130  if (!driver->pool) return -1;
131 
132  talloc_link_ctx(driver, driver->pool); /* Ensure pool is freed */
133 
134  if (config->max_entries > 0) {
135  ERROR("max_entries is not supported by this driver");
136  return -1;
137  }
138  return 0;
139 }

156 /** Locate a cache entry in memcached
157  *
158  * @copydetails cache_entry_find_t
159  */
160 static cache_status_t cache_entry_find(rlm_cache_entry_t **out,
161  UNUSED rlm_cache_config_t const *config, UNUSED void *instance,
162  request_t *request, void *handle, uint8_t const *key, size_t key_len)
163 {
164  rlm_cache_memcached_handle_t *mandle = handle;
165 
166  memcached_return_t mret;
167  size_t len;
168  int ret;
169  uint32_t flags;
170 
171  char *from_store;
172 
173  rlm_cache_entry_t *c;
174 
175  from_store = memcached_get(mandle->handle, (char const *)key, key_len, &len, &flags, &mret);
176  if (!from_store) {
177  if (mret == MEMCACHED_NOTFOUND) return CACHE_MISS;
178 
179  RERROR("Failed retrieving entry: %s: %s", memcached_strerror(mandle->handle, mret),
180  memcached_last_error_message(mandle->handle));
181 
182  return CACHE_ERROR;
183  }
184  RDEBUG2("Retrieved %zu bytes from memcached", len);
185  RDEBUG2("%s", from_store);
186 
187  c = talloc_zero(NULL, rlm_cache_entry_t);
188  ret = cache_deserialize(c, request->dict, from_store, len);
189  free(from_store);
190  if (ret < 0) {
191  RPERROR("Invalid entry");
192  talloc_free(c);
193  return CACHE_ERROR;
194  }
195  c->key = talloc_memdup(c, key, key_len);
196  c->key_len = key_len;
197 
198  *out = c;
199 
200  return CACHE_OK;
201 }

203 /** Insert a new entry into the data store
204  *
205  * @copydetails cache_entry_insert_t
206  */
207 static cache_status_t cache_entry_insert(UNUSED rlm_cache_config_t const *config, UNUSED void *instance,
208  request_t *request, void *handle, const rlm_cache_entry_t *c)
209 {
210  rlm_cache_memcached_handle_t *mandle = handle;
211 
212  memcached_return_t ret;
213 
214  TALLOC_CTX *pool;
215  char *to_store;
216 
217  pool = talloc_pool(NULL, 1024);
218  if (!pool) return CACHE_ERROR;
219 
220  if (cache_serialize(pool, &to_store, c) < 0) {
221  talloc_free(pool);
222 
223  return CACHE_ERROR;
224  }
225 
226  ret = memcached_set(mandle->handle, (char const *)c->key, c->key_len,
227  to_store ? to_store : "",
228  to_store ? talloc_array_length(to_store) - 1 : 0, c->expires, 0);
229  talloc_free(pool);
230  if (ret != MEMCACHED_SUCCESS) {
231  RERROR("Failed storing entry: %s: %s", memcached_strerror(mandle->handle, ret),
232  memcached_last_error_message(mandle->handle));
233 
234  return CACHE_ERROR;
235  }
236 
237  return CACHE_OK;
238 }

240 /** Call delete the cache entry from memcached
241  *
242  * @copydetails cache_entry_expire_t
243  */
244 static cache_status_t cache_entry_expire(UNUSED rlm_cache_config_t const *config, UNUSED void *instance,
245  request_t *request, void *handle, uint8_t const *key, size_t key_len)
246 {
247  rlm_cache_memcached_handle_t *mandle = handle;
248 
249  memcached_return_t ret;
250 
251  ret = memcached_delete(mandle->handle, (char const *)key, key_len, 0);
252  switch (ret) {
253  case MEMCACHED_SUCCESS:
254  return CACHE_OK;
255 
256  case MEMCACHED_DATA_DOES_NOT_EXIST:
257  return CACHE_MISS;
258 
259  default:
260  RERROR("Failed deleting entry: %s", memcached_last_error_message(mandle->handle));
261  return CACHE_ERROR;
262  }
263 }
