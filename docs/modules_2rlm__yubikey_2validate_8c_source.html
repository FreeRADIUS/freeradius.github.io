60 {

63 
64  int count = 0;
65 
66  if (!inst->client_id) {
67  ERROR("validation.client_id must be set (to a valid id) when validation is enabled");

72  if (!inst->api_key || !*inst->api_key || is_zero(inst->api_key)) {

74 
75  return -1;
76  }
77 
78  DEBUG("Initialising ykclient");
79 
80  status = ykclient_global_init();
81  if (status != YKCLIENT_OK) {
82 yk_error:

84 
85  return -1;
86  }
87 
88  status = ykclient_init(&inst->ykc);
89  if (status != YKCLIENT_OK) goto yk_error;
90 
91  servers = cf_section_find(conf, "servers", CF_IDENT_ANY);
92  if (servers) {
93  CONF_PAIR *uri, *first;
94  /*
95  * If there were no uris configured we just use the default
96  * ykclient uris which point to the yubico servers.
97  */
98  first = uri = cf_pair_find(servers, "uri");
99  if (!uri) {
100  goto init;
101  }
102 
103  while (uri) {
104  count++;
105  uri = cf_pair_find_next(servers, uri, "uri");
106  }
107  inst->uris = talloc_zero_array(inst, char const *, count);
108 
109  uri = first;
110  count = 0;
111  while (uri) {
112  inst->uris[count++] = cf_pair_value(uri);
113  uri = cf_pair_find_next(servers, uri, "uri");
114  }
115  if (count) {
116  status = ykclient_set_url_templates(inst->ykc, count, inst->uris);
117  if (status != YKCLIENT_OK) {
118  goto yk_error;
119  }
120  }
121  }
122

125  if (status != YKCLIENT_OK) {

150 unlang_action_t rlm_yubikey_validate(rlm_rcode_t *p_result, rlm_yubikey_t const *inst, request_t *request, char const *passcode)
151 {
152  rlm_rcode_t rcode = RLM_MODULE_OK;
153  ykclient_rc status;
154  ykclient_handle_t *yandle;
155 
156  yandle = fr_pool_connection_get(inst->pool, request);
157  if (!yandle) RETURN_MODULE_FAIL;
158 
159  /*
160  * The libcurl multi-handle interface will tear down the TCP sockets for any partially completed
161  * requests when their easy handle is removed from the multistack.
162  *
163  * For performance reasons ykclient will stop processing the request immediately after receiving
164  * a response from one of the servers. If we then immediately call ykclient_handle_cleanup
165  * the connections are destroyed and will need to be re-established the next time the handle
166  * is used.
167  *
168  * To try and prevent this from happening, we leave cleanup until the *next* time
169  * the handle is used, by which time the requests will of hopefully completed and the connections
170  * can be re-used.
171  *
172  */
173  ykclient_handle_cleanup(yandle);
174 
175  status = ykclient_request_process(inst->ykc, yandle, passcode);
176  if (status != YKCLIENT_OK) {
177  REDEBUG("%s", ykclient_strerror(status));
178  switch (status) {
179  case YKCLIENT_BAD_OTP:
180  case YKCLIENT_REPLAYED_OTP:
181  rcode = RLM_MODULE_REJECT;
182  break;
183 
184  case YKCLIENT_NO_SUCH_CLIENT:
185  rcode = RLM_MODULE_NOTFOUND;
186  break;
187 
188  default:
189  rcode = RLM_MODULE_FAIL;
190  }
191  }
192 
193  fr_pool_connection_release(inst->pool, request, yandle);
194 
195  RETURN_MODULE_RCODE(rcode);
196 }
