57 /** Check the reply for errors
58  *
59  * @param conn used to issue the command.
60  * @param reply to process.
61  * @return
62  * - REDIS_RCODE_TRY_AGAIN - If the operation should be retries.
63  * - REDIS_RCODE_MOVED - If the key has been permanently moved.
64  * - REDIS_RCODE_ASK - If the key has been temporarily moved.
65  * - REDIS_RCODE_SUCCESS - if no errors.
66  * - REDIS_RCODE_ERROR - on command/server error.
67  * - REDIS_RCODE_NO_SCRIPT - script specified by evalsha doesn't exist.
68  * - REDIS_RCODE_RECONNECT - on connection error (probably needs reconnecting).
69  */
70 fr_redis_rcode_t fr_redis_command_status(fr_redis_conn_t *conn, redisReply *reply)
71 {
72  size_t i = 0;
73 
74  if (!reply) switch (conn->handle->err) {
75  case REDIS_OK:
76  break;
77 
78  case REDIS_ERR_IO:
79  case REDIS_ERR_EOF:
80  case REDIS_ERR_OTHER:
81  fr_strerror_printf("Connection error: %s", conn->handle->errstr);
82  return REDIS_RCODE_RECONNECT;
83 
84  default:
85  case REDIS_ERR_PROTOCOL:
86  fr_strerror_printf("Command error: %s", conn->handle->errstr);
87  return REDIS_RCODE_ERROR;
88  }
89 
90  if (reply) switch (reply->type) {
91  case REDIS_REPLY_STATUS:
92  return REDIS_RCODE_SUCCESS;
93 
94  case REDIS_REPLY_ERROR:
95  fr_strerror_printf("Server error: %s", reply->str);
96  if (strncmp(REDIS_ERROR_MOVED_STR, reply->str, sizeof(REDIS_ERROR_MOVED_STR) - 1) == 0) {
97  return REDIS_RCODE_MOVE;
98  }
99  if (strncmp(REDIS_ERROR_ASK_STR, reply->str, sizeof(REDIS_ERROR_ASK_STR) - 1) == 0) {
100  return REDIS_RCODE_ASK;
101  }
102  if (strncmp(REDIS_ERROR_TRY_AGAIN_STR, reply->str, sizeof(REDIS_ERROR_TRY_AGAIN_STR) - 1) == 0) {
103  return REDIS_RCODE_TRY_AGAIN;
104  }
105  if (strncmp(REDIS_ERROR_NO_SCRIPT_STR, reply->str, sizeof(REDIS_ERROR_NO_SCRIPT_STR) - 1) == 0) {
106  return REDIS_RCODE_NO_SCRIPT;
107  }
108  return REDIS_RCODE_ERROR;
109 
110  /*
111  * Recurse to check for nested errors
112  */
113  case REDIS_REPLY_ARRAY:
114  for (i = 0; i < reply->elements; i++) {
115  int ret;
116 
117  ret = fr_redis_command_status(conn, reply->element[i]);
118  if (ret < 0) return ret;
119  }
120  break;
121 
122  default:
123  break;
124  }
125  return REDIS_RCODE_SUCCESS;
126 }

128 /** Print the response data in a useful treelike form
129  *
130  * @param lvl to print data at.
131  * @param reply to print.
132  * @param request The current request.
133  * @param idx Response number.
134  */
135 void fr_redis_reply_print(fr_log_lvl_t lvl, redisReply *reply, request_t *request, int idx)
136 {
137  size_t i = 0;
138 
139  if (!reply) return;
140 
141  switch (reply->type) {
142  case REDIS_REPLY_ERROR:
143  REDEBUG("(%i) error : %s", idx, reply->str);
144  break;
145 
146  case REDIS_REPLY_STATUS:
147  RDEBUGX(lvl, "(%i) status : %s", idx, reply->str);
148  break;
149 
150  case REDIS_REPLY_STRING:
151  RDEBUGX(lvl, "(%i) string : %s", idx, reply->str);
152  break;
153 
154  case REDIS_REPLY_INTEGER:
155  RDEBUGX(lvl, "(%i) integer : %lld", idx, reply->integer);
156  break;
157 
158  case REDIS_REPLY_NIL:
159  RDEBUGX(lvl, "(%i) nil", idx);
160  break;
161 
162  case REDIS_REPLY_ARRAY:
163  RDEBUGX(lvl, "(%i) array[%zu]", idx, reply->elements);
164  for (i = 0; i < reply->elements; i++) {
165  RINDENT();
166  fr_redis_reply_print(lvl, reply->element[i], request, i);
167  REXDENT();
168  }
169  break;
170  }

173 /** Convert a string or integer type to #fr_value_box_t of specified type
174  *
175  * Will work with REDIS_REPLY_STRING (which is converted to #FR_TYPE_STRING
176  * then cast to dst_type), or REDIS_REPLY_INTEGER (which is converted to
177  * #FR_TYPE_UINT64, then cast to dst_type).
178  *
179  * @note Any unsupported types will trigger an assert. You must check the
180  * reply type prior to calling this function.
181  *
182  * @param[in,out] ctx to allocate any buffers in.
183  * @param[out] out Where to write the cast type.
184  * @param[in] reply to process.
185  * @param[in] dst_type to convert to.
186  * @param[in] dst_enumv Used to convert string types to integers for attributes
187  * with enumerated values.
188  * @return
189  * - 1 if we received a NIL reply. Out will remain uninitialized.
190  * - 0 on success.
191  * - -1 on cast or parse failure.
192  */
193 int fr_redis_reply_to_value_box(TALLOC_CTX *ctx, fr_value_box_t *out, redisReply *reply,
194  fr_type_t dst_type, fr_dict_attr_t const *dst_enumv)
195 {
196  fr_value_box_t in;
197 
198  memset(&in, 0, sizeof(in));
199 
200  switch (reply->type) {
201  case REDIS_REPLY_NIL:
202  return 1;
203 
204  /*
205  * Try and convert the integer to the smallest
206  * and simplest type possible, to give the cast
207  * the greatest chance of success.
208  */
209  case REDIS_REPLY_INTEGER:
210  if (reply->integer < INT32_MIN) { /* 64bit signed */
211  fr_value_box_shallow(&in, (int64_t) reply->integer, true);
212  }
213  else if (reply->integer < INT16_MIN) { /* 32bit signed */
214  fr_value_box_shallow(&in, (int32_t) reply->integer, true);
215  }
216  else if (reply->integer < INT8_MIN) { /* 16bit signed */
217  fr_value_box_shallow(&in, (int16_t) reply->integer, true);
218  }
219  else if (reply->integer < 0) { /* 8bit signed */
220  fr_value_box_shallow(&in, (int8_t) reply->integer, true);
221  }
222  else if (reply->integer > UINT32_MAX) { /* 64bit unsigned */
223  fr_value_box_shallow(&in, (uint64_t) reply->integer, true);
224  }
225  else if (reply->integer > UINT16_MAX) { /* 32bit unsigned */
226  fr_value_box_shallow(&in, (uint32_t) reply->integer, true);
227  }
228  else if (reply->integer > UINT8_MAX) { /* 16bit unsigned */
229  fr_value_box_shallow(&in, (uint16_t) reply->integer, true);
230  }
231  else { /* 8bit unsigned */
232  fr_value_box_shallow(&in, (uint8_t) reply->integer, true);
233  }
234  break;
235 
236  case REDIS_REPLY_STRING:
237  case REDIS_REPLY_STATUS:
238  case REDIS_REPLY_ERROR:
239  fr_value_box_bstrndup_shallow(&in, NULL, reply->str, reply->len, true);
240  break;
241 
242  case REDIS_REPLY_ARRAY:
243  fr_assert(0);
244  }
245 
246  if (fr_value_box_cast(ctx, out, dst_type, dst_enumv, &in) < 0) return -1;
247 
248  return 0;
249 }

251 /** Convert a pair of redis reply objects to a map
252  *
253  * The maps can then be applied using #map_to_request.
254  *
255  * @param[in,out] ctx to allocate maps in.
256  * @param[out] out Where to write the head of the new maps list.
257  * @param[in] request The current request.
258  * @param[in] key to process.
259  * @param[in] op to process.
260  * @param[in] value to process.
261  * @return
262  * - 0 on success.
263  * - -1 on failure.
264  */
265 int fr_redis_reply_to_map(TALLOC_CTX *ctx, map_t **out, request_t *request,
266  redisReply *key, redisReply *op, redisReply *value)
267 {
268  map_t *map = NULL;
269  ssize_t slen;
270 
271  *out = NULL;
272 
273  if (key->type != REDIS_REPLY_STRING) {
274  REDEBUG("Bad key type, expected string, got %s",
275  fr_table_str_by_value(redis_reply_types, key->type, "<UNKNOWN>"));
276  error:
277  TALLOC_FREE(map);
278  return -1;
279  }
280 
281  if (op->type != REDIS_REPLY_STRING) {
282  REDEBUG("Bad key type, expected string, got %s",
283  fr_table_str_by_value(redis_reply_types, op->type, "<UNKNOWN>"));
284  goto error;
285  }
286 
287  RDEBUG3("Got key : %s", key->str);
288  RDEBUG3("Got op : %s", op->str);
289  RDEBUG3("Got value : %pV", fr_box_strvalue_len(value->str, value->len));
290 
291  MEM(map = talloc_zero(ctx, map_t));
292  slen = tmpl_afrom_attr_str(map, NULL, &map->lhs, key->str, &(tmpl_rules_t){ .dict_def = request->dict });
293  if (slen < 0) {
294  REMARKER(key->str, -slen, "%s", fr_strerror());
295  goto error;
296  }
297 
298  map->op = fr_table_value_by_str(fr_tokens_table, op->str, T_INVALID);
299  if (map->op == T_INVALID) {
300  REDEBUG("Invalid operator \"%s\"", op->str);
301  goto error;
302  }
303 
304  switch (value->type) {
305  case REDIS_REPLY_STRING:
306  case REDIS_REPLY_INTEGER:
307  {
308  fr_value_box_t vpt;
309 
310  /* Logs own errors */
311  if (fr_redis_reply_to_value_box(map, &vpt, value,
312  tmpl_da(map->lhs)->type, tmpl_da(map->lhs)) < 0) {
313  RPEDEBUG("Failed converting Redis data");
314  goto error;
315  }
316 
317  /* This will only fail only memory allocation errors */
318  if (tmpl_afrom_value_box(map, &map->rhs, &vpt, true) < 0) goto error;
319  }
320  break;
321 
322  default:
323  REDEBUG("Bad value type, expected string or integer, got %s",
324  fr_table_str_by_value(redis_reply_types, value->type, "<UNKNOWN>"));
325  goto error;
326 
327  }
328  MAP_VERIFY(map);
329 
330  *out = map;
331 
332  return 0;
333 }

335 /** Add a single map pair to an existing command string as three elements
336  *
337  * - Integer types will be encoded as integers.
338  * - Strings and octets will be encoded in their raw form.
339  * - Other types will be converted to their printable form and will be encoded as strings.
340  *
341  * @note lhs must be a #TMPL_TYPE_ATTR.
342  * @note rhs must be a #TMPL_TYPE_DATA.
343  *
344  * @param pool to allocate any buffers in.
345  * @param out Where to write pointers to the member of the tuple. Unused elements should be
346  * a multiple of three, and it should have at least three unused elements.
347  * @param out_len Where to write the size of the data pointed to by the equivalent index
348  * in the out array.
349  * @param map to convert.
350  * @return
351  * 0 on success.
352  * -1 on failure.
353  */
354 int fr_redis_tuple_from_map(TALLOC_CTX *pool, char const *out[], size_t out_len[], map_t *map)
355 {
356  char *new;
357 
358  char key_buf[256];
359  char *key;
360  size_t key_len;
361  ssize_t slen;
362 
363  fr_assert(tmpl_is_attr(map->lhs));
364  fr_assert(tmpl_is_data(map->rhs));
365 
366  slen = tmpl_print(&FR_SBUFF_OUT(key_buf, sizeof(key_buf)), map->lhs, TMPL_ATTR_REF_PREFIX_NO, NULL);
367  if (slen < 0) {
368  fr_strerror_printf("Key too long. Must be < " STRINGIFY(sizeof(key_buf)) " "
369  "bytes, got %zu bytes", (size_t)(slen * -1));
370  return -1;
371  }
372  key_len = (size_t)slen;
373  key = talloc_bstrndup(pool, key_buf, key_len);
374  if (!key) return -1;
375 
376  switch (tmpl_value_type(map->rhs)) {
377  case FR_TYPE_STRING:
378  case FR_TYPE_OCTETS:
379  out[2] = tmpl_value(map->rhs)->datum.ptr;
380  out_len[2] = tmpl_value_length(map->rhs);
381  break;
382 
383  /*
384  * For everything else we get the string representation
385  */
386  default:
387  fr_value_box_aprint(pool, &new, tmpl_value(map->rhs), NULL);
388  if (!new) {
389  talloc_free(key);
390  return -1;
391  }
392  out[2] = new;
393  out_len[2] = talloc_array_length(new) - 1;
394  break;
395  }
396 
397  out[0] = key;
398  out_len[0] = key_len;
399  out[1] = fr_table_str_by_value(fr_tokens_table, map->op, NULL);
400  out_len[1] = strlen(out[1]);
401 
402  return 0;
403 }

405 /** Simplifies handling of pipelined commands with Redis cluster
406  *
407  * Retrieve all available pipelined responses, and write them to the array.
408  *
409  * On encountering an error, all previously retrieved responses are freed, and the reply
410  * containing the error is written to the first element of out. All responses after the
411  * error are also freed.
412  *
413  * If the number of responses != pipelined, that's also an error, a very serious one,
414  * in libhiredis or Redis. We can't really do much here apart from error out.
415  *
416  * @param[out] pipelined Number of pipelined commands we sent to the server.
417  * @param[out] rcode Status of the first errored response, or REDIS_RCODE_SUCCESS
418  * if all responses were processed.
419  * @param[out] out Where to write the replies from pipelined commands.
420  * Will contain exactly 1 element on error WHICH MUST BE FREED,
421  * else the number passed in pipelined.
422  * @param[in] out_len number of elements in out.
423  * @param[in] conn the pipelined commands were issued on.
424  * @return
425  * - #REDIS_RCODE_SUCCESS on success.
426  * - #REDIS_RCODE_ERROR on command/response mismatch or command error.
427  * - REDIS_RCODE_* on other errors;
428  */
429 fr_redis_rcode_t fr_redis_pipeline_result(unsigned int *pipelined, fr_redis_rcode_t *rcode,
430  redisReply *out[], size_t out_len,
431  fr_redis_conn_t *conn)
432 {
433  size_t i;
434  redisReply **out_p = out;
435  fr_redis_rcode_t status = REDIS_RCODE_SUCCESS;
436  redisReply *reply = NULL;
437 
438  fr_assert(out_len >= (size_t)*pipelined);
439 
440  fr_strerror(); /* Clear any outstanding errors */
441 
442  if ((size_t) *pipelined > out_len) {
443  for (i = 0; i < (size_t)*pipelined; i++) {
444  if (redisGetReply(conn->handle, (void **)&reply) != REDIS_OK) break;
445  fr_redis_reply_free(&reply);
446  }
447 
448  *pipelined = 0; /* all outstanding responses should be cleared */
449 
450  fr_strerror_printf("Too many pipelined commands");
451  out[0] = NULL;
452  return REDIS_RCODE_ERROR;
453  }
454 
455  for (i = 0; i < (size_t)*pipelined; i++) {
456  bool maybe_more = false;
457 
458  /*
459  * we don't need to check the return code here,
460  * as it's also stored in the conn->handle.
461  */
462  reply = NULL; /* redisGetReply doesn't NULLify reply on error *sigh* */
463  if (redisGetReply(conn->handle, (void **)&reply) == REDIS_OK) maybe_more = true;
464  status = fr_redis_command_status(conn, reply);
465  *out_p++ = reply;
466 
467  /*
468  * Bail out of processing responses,
469  * free the remaining ones (leaving this one intact)
470  * pass control back to the cluster code.
471  */
472  if (maybe_more && (status != REDIS_RCODE_SUCCESS)) {
473  size_t j;
474  error:
475  /*
476  * Append the hiredis error
477  */
478  if (conn->handle->errstr[0]) fr_strerror_printf_push("%s", conn->handle->errstr);
479 
480  /*
481  * Free everything that came before the bad reply
482  */
483  for (j = 0; j < i; j++) {
484  fr_redis_reply_free(&out[j]);
485  out[j] = NULL;
486  }
487 
488  /*
489  * ...and drain the rest of the pipelined responses
490  */
491  for (j = i + 1; j < (size_t)*pipelined; j++) {
492  redisReply *to_clear;
493 
494  if (redisGetReply(conn->handle, (void **)&to_clear) != REDIS_OK) break;
495  fr_redis_reply_free(&to_clear);
496  }
497 
498  out[0] = reply;
499 
500  *rcode = status;
501  *pipelined = 0; /* all outstanding responses should be cleared */
502 
503  return reply ? 1 : 0;
504  }
505  }
506 
507  if (i != (size_t)*pipelined) {
508  fr_strerror_printf("Expected %u responses, got %zu", *pipelined, i);
509  status = REDIS_RCODE_ERROR;
510  goto error;
511  }
512 
513  *rcode = status;
514 
515  *pipelined = 0; /* all outstanding responses should be cleared */
516 
517  return i;
518 }

520 /** Get the version of Redis running on the remote server
521  *
522  * This can be useful for some modules, as it allows adaptive behaviour, or early termination.
523  *
524  * @param[out] out Where to write the version string.
525  * @param[in] out_len Length of the version string buffer.
526  * @param[in] conn Used to query the version string.
527  * @return
528  * - #REDIS_RCODE_SUCCESS on success.
529  * - #REDIS_RCODE_ERROR on command/response mismatch or command error.
530  * - REDIS_RCODE_* on other errors;
531  */
532 fr_redis_rcode_t fr_redis_get_version(char *out, size_t out_len, fr_redis_conn_t *conn)
533 {
534  redisReply *reply;
535  fr_redis_rcode_t status;
536  char *p, *q;
537 
538  fr_assert(out_len > 0);
539  out[0] = '\0';
540 
541  reply = redisCommand(conn->handle, "INFO SERVER");
542  status = fr_redis_command_status(conn, reply);
543  if (status != REDIS_RCODE_SUCCESS) return status;
544 
545  if (reply->type != REDIS_REPLY_STRING) {
546  fr_strerror_printf("Bad value type, expected string or integer, got %s",
547  fr_table_str_by_value(redis_reply_types, reply->type, "<UNKNOWN>"));
548  error:
549  fr_redis_reply_free(&reply);
550  return REDIS_RCODE_ERROR;
551  }
552 
553  p = strstr(reply->str, "redis_version:");
554  if (!p) {
555  fr_strerror_printf("Response did not contain version string");
556  goto error;
557  }
558 
559  p = strchr(p, ':');
560  fr_assert(p);
561  p++;
562 
563  q = strstr(p, "\r\n");
564  if (!q) q = p + strlen(p);
565 
566  if ((size_t)(q - p) >= out_len) {
567  fr_strerror_printf("Version string %zu bytes, expected < %zu bytes", q - p, out_len);
568  goto error;
569  }
570  strlcpy(out, p, (q - p) + 1);
571 
572  fr_redis_reply_free(&reply);
573 
574  return REDIS_RCODE_SUCCESS;
575 }

577 /** Convert version string into a 32bit unsigned integer for comparisons
578  *
579  * @param[in] version string to parse.
580  * @return 32bit unsigned integer representing the version string.
581  */
582 uint32_t fr_redis_version_num(char const *version)
583 {
584  unsigned long num;
585  uint32_t ret;
586  char const *p = version;
587  char *q;
588 
589  num = strtoul(p, &q, 10);
590  if (num > UINT8_MAX) {
591  fr_strerror_printf("Major version number %lu greater than " STRINGIFY(UINT8_MAX), num);
592  return 0;
593  }
594 
595  if ((p == q) || (q[0] != '.')) {
596  fr_strerror_printf("Trailing garbage in Redis version \"%s\"", q);
597  return 0;
598  }
599  ret = num << 24;
600  p = q + 1;
601 
602  num = strtoul(p, &q, 10);
603  if (num > UINT8_MAX) {
604  fr_strerror_printf("Minor version number %lu greater than " STRINGIFY(UINT8_MAX), num);
605  return 0;
606  }
607 
608  if ((p == q) || (q[0] != '.')) {
609  fr_strerror_printf("Trailing garbage in Redis version \"%s\"", q);
610  return 0;
611  }
612  ret |= num << 16;
613  p = q + 1;
614 
615  num = strtoul(p, &q, 10);
616  if (num > UINT16_MAX) {
617  fr_strerror_printf("Minor version number %lu greater than " STRINGIFY(UINT16_MAX), num);
618  return 0;
619  }
