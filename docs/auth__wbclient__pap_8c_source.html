38 /** PAP authentication direct to winbind via Samba's libwbclient library
39  *
40  * @param[in] inst Module instance
41  * @param[in] request The current request
42  * @param[in] password the User-Password
43  *
44  * @return
45  * - 0 Success
46  * - -1 Authentication failure
47  * - -648 Password expired
48  *
49  */
50 int do_auth_wbclient_pap(rlm_winbind_t const *inst, request_t *request, fr_pair_t *password)
51 {
52  int ret = -1;
53  struct wbcContext *wb_ctx;
54  struct wbcAuthUserParams authparams;
55  wbcErr err;
56  int len;
57  struct wbcAuthUserInfo *info = NULL;
58  struct wbcAuthErrorInfo *error = NULL;
59  char user_name_buf[500];
60  char domain_name_buf[500];
61 
62  /*
63  * Clear the auth parameters - this is important, as
64  * there are options that will cause wbcAuthenticateUserEx
65  * to bomb out if not zero.
66  */
67  memset(&authparams, 0, sizeof(authparams));
68 
69  /*
70  * wb_username must be set for this function to be called
71  */
72  fr_assert(inst->wb_username);
73 
74  /*
75  * Get the username and domain from the configuration
76  */
77  len = tmpl_expand(&authparams.account_name, user_name_buf, sizeof(user_name_buf),
78  request, inst->wb_username, NULL, NULL);
79  if (len < 0) {
80  REDEBUG2("Unable to expand winbind_username");
81  goto done;
82  }
83 
84  if (inst->wb_domain) {
85  len = tmpl_expand(&authparams.domain_name, domain_name_buf, sizeof(domain_name_buf),
86  request, inst->wb_domain, NULL, NULL);
87  if (len < 0) {
88  REDEBUG2("Unable to expand winbind_domain");
89  goto done;
90  }
91  } else {
92  RWDEBUG2("No domain specified; authentication may fail because of this");
93  }
94 
95 
96  /*
97  * Build the wbcAuthUserParams structure with what we know
98  */
99  authparams.level = WBC_AUTH_USER_LEVEL_PLAIN;
100  authparams.password.plaintext = password->data.vb_strvalue;
101 
102  /*
103  * Parameters documented as part of the MSV1_0_SUBAUTH_LOGON structure
104  * at https://msdn.microsoft.com/aa378767.aspx
105  */
106  authparams.parameter_control |= WBC_MSV1_0_CLEARTEXT_PASSWORD_ALLOWED |
107  WBC_MSV1_0_ALLOW_WORKSTATION_TRUST_ACCOUNT |
108  WBC_MSV1_0_ALLOW_SERVER_TRUST_ACCOUNT;
109 
110  /*
111  * Send auth request across to winbind
112  */
113  wb_ctx = fr_pool_connection_get(inst->wb_pool, request);
114  if (wb_ctx == NULL) {
115  RERROR("Unable to get winbind connection from pool");
116  goto done;
117  }
118 
119  RDEBUG2("Sending authentication request user='%s' domain='%s'", authparams.account_name,
120  authparams.domain_name);
121 
122  err = wbcCtxAuthenticateUserEx(wb_ctx, &authparams, &info, &error);
123 
124  fr_pool_connection_release(inst->wb_pool, request, wb_ctx);
125 
126 
127  /*
128  * Try and give some useful feedback on what happened. There are only
129  * a few errors that can actually be returned from wbcCtxAuthenticateUserEx.
130  */
131  switch (err) {
132  case WBC_ERR_SUCCESS:
133  ret = 0;
134  RDEBUG2("Authenticated successfully");
135  break;
136 
137  case WBC_ERR_WINBIND_NOT_AVAILABLE:
138  RERROR("Unable to contact winbindd");
139  RDEBUG2("Check that winbind is running and that FreeRADIUS has");
140  RDEBUG2("permission to connect to the winbind privileged socket");
141  break;
142 
143  case WBC_ERR_DOMAIN_NOT_FOUND:
144  REDEBUG2("Domain not found");
145  break;
146 
147  case WBC_ERR_AUTH_ERROR:
148  if (!error) {
149  REDEBUG2("Authentication failed");
150  break;
151  }
152 
153  /*
154  * The password needs to be changed, set ret appropriately.
155  */
156  if (error->nt_status == NT_STATUS_PASSWORD_EXPIRED ||
157  error->nt_status == NT_STATUS_PASSWORD_MUST_CHANGE) {
158  ret = -648;
159  }
160 
161  /*
162  * Return the NT_STATUS human readable error string, if there is one.
163  */
164  if (error->display_string) {
165  REDEBUG2("%s [0x%X]", error->display_string, error->nt_status);
166  } else {
167  REDEBUG2("Unknown authentication failure [0x%X]", error->nt_status);
168  }
169  break;
170 
171  default:
172  /*
173  * Only errors left are
174  * WBC_ERR_INVALID_PARAM
175  * WBC_ERR_NO_MEMORY
176  * neither of which are particularly likely.
177  */
178  if (error && error->display_string) {
179  REDEBUG2("Failed authenticating user: %s (%s)", error->display_string, wbcErrorString(err));
180  } else {
181  REDEBUG2("Failed authenticating user: Winbind error (%s)", wbcErrorString(err));
182  }
183  break;
184  }
185 
186 
187 done:
188  if (info) wbcFreeMemory(info);
189  if (error) wbcFreeMemory(error);
190 
191  return ret;
192 }
