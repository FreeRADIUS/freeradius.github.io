96 /** Check whether the string is all whitespace
97  *
98  * @return
99  * - true if the entirety of the string is whitespace.
100  * - false if the string contains non whitespace.
101  */
102 static inline bool is_whitespace(char const *value)
103 {
104 #ifdef __clang_analyzer__
105  if (*value == '\0') return false; /* clang analyzer doesn't seem to know what isspace does */
106 #endif
107  do {
108  if (!isspace(*value)) return false;
109  } while (*++value);
110 
111  return true;
112 }

114 /** Check whether the string is made up of printable UTF8 chars
115  *
116  * @param value to check.
117  * @param len of value.
118  *
119  * @return
120  * - true if the string is printable.
121  * - false if the string contains non printable chars
122  */
123  static inline bool is_printable(void const *value, size_t len)
124  {
125  uint8_t const *p = value;
126  size_t clen;
127  size_t i;
128 
129  for (i = 0; i < len; i++) {
130  clen = fr_utf8_char(p, len - i);
131  if (clen == 0) return false;
132  i += (size_t)clen;
133  p += clen;
134  }
135  return true;
136  }

171 /** Find the highest order bit in an unsigned 64 bit integer
172  *
173  * @return 0-64 indicating the position of the highest bit,
174  * with 0 indicating no high bits, 1 indicating the 1st
175  * bit and 64 indicating the last bit.
176  */
177 static inline uint8_t fr_high_bit_pos(uint64_t num)
178 {
179 #ifdef HAVE_BUILTIN_CLZLL
180  return (64 - __builtin_clzll(num));
181 #else
182  uint8_t ret = 1;
183  if (num == 0) return 0;
184  while (num >>= 1) ret++;
185  return ret;
186 #endif
187 }
