75 /** Group comparison for Winbind-Group
76  *
77  * @param instance Instance of this module
78  * @param request The current request
79  * @param req The request list
80  * @param check Value pair containing group to be searched
81  * @param check_list Unknown
82  *
83  * @return
84  * - 0 user is in group
85  * - 1 failure or user is not in group
86  */
87 static int winbind_group_cmp(void *instance, request_t *request, UNUSED fr_pair_t *req, fr_pair_t *check,
88  UNUSED fr_pair_t *check_list)
89 {
90  rlm_winbind_t *inst = instance;
91  int ret = 1;
92  struct wbcContext *wb_ctx;
93  wbcErr err;
94  uint32_t num_groups, i;
95  gid_t *wb_groups = NULL;
96 
97  char const *domain = NULL;
98  size_t domain_len = 0;
99  char const *user = NULL;
100  char *user_buff = NULL;
101  char const *username;
102  char *username_buff = NULL;
103  fr_pair_t *vp_username;
104 
105  ssize_t slen;
106  size_t backslash = 0;
107 
108  vp_username = fr_pair_find_by_da(&request->request_pairs, attr_user_name);
109  if (!vp_username) return -1;
110 
111  RINDENT();
112 
113  if (check->vp_length == 0) {
114  REDEBUG("Group name is empty, nothing to check!");
115  goto error;
116  }
117 
118  /*
119  * Work out what username to check groups for, made up from
120  * either winbind_domain and either group_search_username or
121  * just User-Name.
122  */
123 
124  /*
125  * Include the domain in the username?
126  */
127  if (inst->group_add_domain && inst->wb_domain) {
128  slen = tmpl_aexpand(request, &domain, request, inst->wb_domain, NULL, NULL);
129  if (slen < 0) {
130  REDEBUG("Unable to expand group_search_username");
131  goto error;
132  }
133  domain_len = (size_t)slen;
134  }
135 
136  /*
137  * Sort out what User-Name we are going to use.
138  */
139  if (inst->group_username) {
140  slen = tmpl_aexpand(request, &user_buff, request, inst->group_username, NULL, NULL);
141  if (slen < 0) {
142  REDEBUG("Unable to expand group_search_username");
143  goto error;
144  }
145  user = user_buff;
146  } else {
147  /*
148  * This is quite unlikely to work without a domain, but
149  * we've not been given much else to work on.
150  */
151  if (!domain) {
152  RWDEBUG("Searching group with plain username, this will probably fail");
153  RWDEBUG("Ensure winbind_domain and group_search_username are both correctly set");
154  }
155  user = vp_username->vp_strvalue;
156  }
157 
158  if (domain) {
159  username = username_buff = talloc_typed_asprintf(request, "%s\\%s", domain, user);
160  } else {
161  username = user;
162  }
163 
164  /*
165  * Get a libwbclient connection from the pool
166  */
167  wb_ctx = fr_pool_connection_get(inst->wb_pool, request);
168  if (wb_ctx == NULL) {
169  RERROR("Unable to get winbind connection from the pool");
170  goto error;
171  }
172 
173  REDEBUG2("Trying to find user \"%s\" in group \"%pV\"", username, &check->data);
174 
175  err = wbcCtxGetGroups(wb_ctx, username, &num_groups, &wb_groups);
176  switch (err) {
177  case WBC_ERR_SUCCESS:
178  ret = 0;
179  REDEBUG2("Successfully retrieved user's groups");
180  break;
181 
182  case WBC_ERR_WINBIND_NOT_AVAILABLE:
183  RERROR("Failed retrieving groups: Unable to contact winbindd"); /* Global error */
184  break;
185 
186  case WBC_ERR_DOMAIN_NOT_FOUND:
187  /* Yeah, weird. libwbclient returns this if the username is unknown */
188  REDEBUG("Failed retrieving groups: User or Domain not found");
189  break;
190 
191  case WBC_ERR_UNKNOWN_USER:
192  REDEBUG("Failed retrieving groups: User cannot be found");
193  break;
194 
195  default:
196  REDEBUG("Failed retrieving groups: %s", wbcErrorString(err));
197  break;
198  }
199 
200  if (!num_groups) REDEBUG2("No groups returned");
201 
202  if (ret) goto finish;
203  ret = 1;
204 
205  /*
206  * See if any of the groups match
207  */
208 
209  /*
210  * We try and find where the '\' is in the returned group, which saves
211  * looking for it each time. There seems to be no way to get a list of
212  * groups without the domain in them, but at least the backslash is
213  * always going to be in the same place.
214  *
215  * Maybe there should be an option to include the domain in the compared
216  * group name in case people have multiple domains?
217  */
218  if (domain_len > 0) backslash = domain_len - 1;
219 
220  for (i = 0; i < num_groups; i++) {
221  struct group *group;
222  char *group_name;
223 
224  bool found = false;
225 
226  /* Get the group name from the (fake winbind) gid */
227  err = wbcCtxGetgrgid(wb_ctx, wb_groups[i], &group);
228  if (err != WBC_ERR_SUCCESS) {
229  REDEBUG("Failed resolving GID %i: %s", wb_groups[i], wbcErrorString(err));
230  if (wb_groups[i] == UINT32_MAX) {
231  REDEBUG("GID appears to be winbind placeholder value, idmap likely failed");
232  }
233  continue;
234  }
235 
236  REDEBUG2("Resolved GID %i to name \"%s\"", wb_groups[i], group->gr_name);
237 
238  /* Find the backslash in the returned group name */
239  if ((backslash < strlen(group->gr_name)) && (group->gr_name[backslash] == '\\')) {
240  group_name = group->gr_name + backslash + 1;
241  } else if ((group_name = strchr(group->gr_name, '\\'))) {
242  group_name++;
243  backslash = group_name - (group->gr_name - 1);
244  } else {
245  group_name = group->gr_name;
246  }
247 
248  /* See if the group matches */
249  REDEBUG2("Checking plain group name \"%s\"", group_name);
250  if (!strcasecmp(group_name, check->vp_strvalue)) {
251  REDEBUG2("Found matching group: %s", group_name);
252  found = true;
253  ret = 0;
254  }
255  wbcFreeMemory(group);
256 
257  /* Short-circuit to save unnecessary enumeration */
258  if (found) break;
259  }
260 
261  if (ret) REDEBUG2("No groups found that match");
262 
263 finish:
264  wbcFreeMemory(wb_groups);
265  fr_pool_connection_release(inst->wb_pool, request, wb_ctx);
266 
267 error:
268  talloc_free(user_buff);
269  talloc_free(username_buff);
270  talloc_const_free(domain);
271  REXDENT();
272 
273  return ret;
274 }

317 /** Bootstrap this module
318  *
319  * Register pair compare function for Winbind-Group fake attribute
320  *
321  * @param[in] conf Module configuration
322  * @param[in] instance This module's instance
323  *
324  * @return
325  * - 0 success
326  * - -1 failure
327  */
328 static int mod_bootstrap(void *instance, CONF_SECTION *conf)
329 {
330  rlm_winbind_t *inst = instance;
331  char const *group_attribute;
332  char buffer[256];
333 
334  inst->name = cf_section_name2(conf);
335  if (!inst->name) inst->name = cf_section_name1(conf);
336 
337  if (inst->group_attribute) {
338  group_attribute = inst->group_attribute;
339  } else if (cf_section_name2(conf)) {
340  snprintf(buffer, sizeof(buffer), "%s-Winbind-Group", inst->name);
341  group_attribute = buffer;
342  } else {
343  group_attribute = "Winbind-Group";
344  }
345 
346  if (paircmp_register_by_name(group_attribute, attr_user_name, false,
347  winbind_group_cmp, inst) < 0) {
348  PERROR("Error registering group comparison");
349  return -1;
350  }
351 
352  return 0;
353 }

356 /** Instantiate this module
357  *
358  * @param[in] conf Module configuration
359  * @param[in] instance This module's instance
360  *
361  * @return
362  * - 0 instantiation succeeded
363  * - -1 instantiation failed
364  */
365 static int mod_instantiate(void *instance, CONF_SECTION *conf)
366 {
367  rlm_winbind_t *inst = instance;
368  struct wbcInterfaceDetails *wb_info = NULL;
369 
370  if (!inst->wb_username) {
371  cf_log_err(conf, "winbind_username must be defined to use rlm_winbind");
372  return -1;
373  }
374 
375  inst->wb_pool = module_connection_pool_init(conf, inst, mod_conn_create, NULL, NULL, NULL, NULL);
376  if (!inst->wb_pool) {
377  cf_log_err(conf, "Unable to initialise winbind connection pool");
378  return -1;
379  }
380 
381  inst->auth_type = fr_dict_enum_by_name(attr_auth_type, inst->name, -1);
382  if (!inst->auth_type) {
383  WARN("Failed to find 'authenticate %s {...}' section. Winbind authentication will likely not work",
384  inst->name);
385  }
386 
387  /*
388  * If the domain has not been specified, try and find
389  * out what it is from winbind.
390  */
391  if (!inst->wb_domain) {
392  wbcErr err;
393  struct wbcContext *wb_ctx;
394 
395  cf_log_err(conf, "winbind_domain unspecified; trying to get it from winbind");
396 
397  wb_ctx = wbcCtxCreate();
398  if (!wb_ctx) {
399  /* this should be very unusual */
400  cf_log_err(conf, "Unable to get libwbclient context, cannot get domain");
401  goto no_domain;
402  }
403 
404  err = wbcCtxInterfaceDetails(wb_ctx, &wb_info);
405  wbcCtxFree(wb_ctx);
406 
407  if (err != WBC_ERR_SUCCESS) {
408  cf_log_err(conf, "libwbclient returned wbcErr code %d; unable to get domain name.", err);
409  cf_log_err(conf, "Is winbind running and does the winbind_privileged socket have");
410  cf_log_err(conf, "the correct permissions?");
411  goto no_domain;
412  }
413 
414  if (!wb_info->netbios_domain) {
415  cf_log_err(conf, "winbind returned blank domain name");
416  goto no_domain;
417  }
418 
419  tmpl_afrom_substr(instance, &inst->wb_domain,
420  &FR_SBUFF_IN(wb_info->netbios_domain, strlen(wb_info->netbios_domain)),
421  T_SINGLE_QUOTED_STRING,
422  NULL,
423  &(tmpl_rules_t){
424  .allow_unknown = true,
425  .allow_unresolved = true
426  });
427 
428  cf_log_err(conf, "Using winbind_domain '%s'", inst->wb_domain->name);
429 
430 no_domain:
431  wbcFreeMemory(wb_info);
432  }
433 
434  return 0;
435 }

455 /** Authorize for libwbclient/winbind authentication
456  *
457  * Checks there is a password available so we can authenticate
458  * against winbind and, if so, sets Auth-Type to ourself.
459  *
460  * @param[out] p_result The result of the module call:
461  * - #RLM_MODULE_NOOP unable to use winbind authentication

464  * @param[in] request The current request.

466 static unlang_action_t CC_HINT(nonnull) mod_authorize(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)

470 
471  vp = fr_pair_find_by_da(&request->request_pairs, attr_user_password);
472  if (!vp) {
473  REDEBUG2("No User-Password found in the request; not doing winbind authentication.");
474  RETURN_MODULE_NOOP;
475  }
476 
477  if (!inst->auth_type) {

485  RETURN_MODULE_OK;

489 /** Authenticate the user via libwbclient and winbind
490  *
491  * @param[out] p_result The result of the module call.
492  * @param[in] mctx Module instance data.
493  * @param[in] request The current request
494  */
495 static unlang_action_t CC_HINT(nonnull) mod_authenticate(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
496 {
497  rlm_winbind_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_winbind_t);
498  fr_pair_t *username, *password;
499 
500  username = fr_pair_find_by_da(&request->request_pairs, attr_user_name);
501  password = fr_pair_find_by_da(&request->request_pairs, attr_user_password);
502 
503  /*
504  * We can only authenticate user requests which HAVE
505  * a User-Name attribute.
506  */
507  if (!username) {
508  REDEBUG("Attribute \"User-Name\" is required for authentication");
509  RETURN_MODULE_INVALID;
510  }
511 
512  if (!password) {
513  REDEBUG("Attribute \"User-Password\" is required for authentication");
514  RETURN_MODULE_INVALID;
515  }
516 
517  /*
518  * Make sure the supplied password isn't empty
519  */
520  if (password->vp_length == 0) {
521  REDEBUG("User-Password must not be empty");
522  RETURN_MODULE_INVALID;
523  }
524 
525  /*
526  * Log the password
527  */
528  if (RDEBUG_ENABLED3) {
529  RDEBUG("Login attempt with password \"%pV\"", &password->data);
530  } else {
531  RDEBUG2("Login attempt with password");
532  }
533 
534  /*
535  * Authenticate and return OK if successful. No need for
536  * many debug outputs or errors as the auth function is
537  * chatty enough.
538  */
539  if (do_auth_wbclient_pap(inst, request, password) == 0) {
540  REDEBUG2("User authenticated successfully using winbind");
541  RETURN_MODULE_OK;
542  }
543 
544  RETURN_MODULE_REJECT;
545 }
