30 /** Initialize src/lib/server/
31  *
32  * This is just so that the callers don't need to call a million functions.
33  *
34  * @param cs The root configuration section.
35  * @return
36  * - 0 on success.
37  * - -1 on failure.
38  */
39 int server_init(CONF_SECTION *cs)
40 {
41  /*
42  * Initialise the trigger rate limiting tree
43  */
44  if (trigger_exec_init(cs) < 0) return -1;
45 
46  /*
47  * Instantiate "permanent" paircmps
48  */
49  if (paircmp_init() < 0) return -1;
50 
51  /*
52  * Set up dictionaries and attributes for password comparisons
53  */
54  if (password_init() < 0) return -1;
55 
56  /*
57  * Initialize Auth-Type, etc. in the virtual servers
58  * before loading the modules. Some modules need those
59  * to be defined.
60  */
61  if (virtual_servers_bootstrap(cs) < 0) return -1;
62 
63  /*
64  * Bootstrap the modules. This links to them, and runs
65  * their "bootstrap" routines.
66  *
67  * After this step, all dynamic attributes, xlats, etc. are defined.
68  */
69  if (modules_bootstrap(cs) < 0) return -1;
70 
71  /*
72  * And then load the virtual servers.
73  */
74  if (virtual_servers_instantiate() < 0) return -1;
75 
76  /*
77  * Instantiate the modules
78  */
79  if (modules_instantiate() < 0) return -1;
80 
81  /*
82  * Call xlat instantiation functions (after the xlats have been compiled)
83  */
84  if (xlat_instantiate() < 0) return -1;
85 
86  return 0;
87 }

89 /** Free src/lib/server/
90  *
91  * This is just so that the callers don't need to call a million functions.
92  */
93 void server_free(void)
94 {
95  /*
96  * Free password dictionaries
97  */
98  password_free();
99 
100  /*
101  * Free xlat instance data, and call any detach methods
102  */
103  xlat_instances_free();
104 
105  /*
106  * Detach modules, connection pools, registered xlats / paircmps / maps.
107  */
108  modules_free();
109 
110  /*
111  * The only paircmps remaining are the ones registered by the server core.
112  */
113  paircmp_free();
114 
115  /*
116  * The only xlats remaining are the ones registered by the server core.
117  */
118  xlat_free();
119 
120  /*
121  * The only maps remaining are the ones registered by the server core.
122  */
123  map_proc_free();
124 
125  /*
126  * Free information associated with the virtual servers.
127  */
128  virtual_servers_free();
129 
130  /*
131  * Now we're sure no more triggers can fire, free the
132  * trigger tree.
133  */
134  trigger_exec_free();
135 
136 }
