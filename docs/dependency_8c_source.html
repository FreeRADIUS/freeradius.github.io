49 /** Check built and linked versions of OpenSSL match
50  *
51  * OpenSSL version number consists of:
52  * MNNFFPPS: major minor fix patch status
53  *
54  * Where status >= 0 && < 10 means beta, and status 10 means release.
55  *
56  * https://wiki.openssl.org/index.php/Versioning
57  *
58  * Startup check for whether the linked version of OpenSSL matches the
59  * version the server was built against.
60  *
61  * @return
62  * - 0 if ok.
63  * - -1 if not ok.
64  */
65 int ssl_check_consistency(void)
66 {
67  long ssl_linked;
68 
69  ssl_linked = SSLeay();
70 
71  /*
72  * Major and minor versions mismatch, that's bad.
73  */
74  if ((ssl_linked & 0xfff00000) != (ssl_built & 0xfff00000)) goto mismatch;
75 
76  /*
77  * 1.1.0 and later export all of the APIs we need, so we
78  * don't care about mismatches in fix / patch / status
79  * fields. If the major && minor fields match, that's
80  * good enough.
81  */
82  if ((ssl_linked & 0xfff00000) >= 0x10100000) return 0;
83 
84  /*
85  * Before 1.1.0, we need all kinds of stupid checks to
86  * see if it might work.
87  */
88 
89  /*
90  * Status mismatch always triggers error.
91  */
92  if ((ssl_linked & 0x0000000f) != (ssl_built & 0x0000000f)) {
93  mismatch:
94  ERROR("libssl version mismatch. built: %lx linked: %lx",
95  (unsigned long) ssl_built,
96  (unsigned long) ssl_linked);
97 
98  return -1;
99  }
100 
101  /*
102  * Use the OpenSSH approach and relax fix checks after version
103  * 1.0.0 and only allow moving backwards within a patch
104  * series.
105  */
106  if (ssl_built & 0xf0000000) {
107  if ((ssl_built & 0xfffff000) != (ssl_linked & 0xfffff000) ||
108  (ssl_built & 0x00000ff0) > (ssl_linked & 0x00000ff0)) goto mismatch;
109  /*
110  * Before 1.0.0 we require the same major minor and fix version
111  * and ignore the patch number.
112  */
113  } else if ((ssl_built & 0xfffff000) != (ssl_linked & 0xfffff000)) goto mismatch;
114 
115  return 0;
116 }

118 /** Convert a version number to a text string
119  *
120  * @note Not thread safe.
121  *
122  * @param v version to convert.
123  * @return pointer to a static buffer containing the version string.
124  */
125 char const *ssl_version_by_num(uint32_t v)
126 {
127  /* 2 (%s) + 1 (.) + 2 (%i) + 1 (.) + 2 (%i) + 1 (c) + 8 (%s) + \0 */
128  static char buffer[18];
129  char *p = buffer;
130 
131  p += sprintf(p, "%u.%u.%u",
132  (0xf0000000 & v) >> 28,
133  (0x0ff00000 & v) >> 20,
134  (0x000ff000 & v) >> 12);
135 
136  if ((0x00000ff0 & v) >> 4) {
137  *p++ = (char) (0x60 + ((0x00000ff0 & v) >> 4));
138  }
139 
140  *p++ = ' ';
141 
142  /*
143  * Development (0)
144  */
145  if ((0x0000000f & v) == 0) {
146  strcpy(p, "dev");
147  /*
148  * Beta (1-14)
149  */
150  } else if ((0x0000000f & v) <= 14) {
151  sprintf(p, "beta %u", 0x0000000f & v);
152  } else {
153  strcpy(p, "release");
154  }
155 
156  return buffer;
157 }

171 /** Convert two openssl version numbers into a range string
172  *
173  * @note Not thread safe.
174  *
175  * @param low version to convert.
176  * @param high version to convert.
177  * @return pointer to a static buffer containing the version range string.
178  */
179 char const *ssl_version_range(uint32_t low, uint32_t high)
180 {
181  /* 18 (version) + 3 ( - ) + 18 (version) */
182  static char buffer[40];
183  char *p = buffer;
184 
185  p += strlcpy(p, ssl_version_by_num(low), sizeof(buffer));
186  p += strlcpy(p, " - ", sizeof(buffer) - (p - buffer));
187  strlcpy(p, ssl_version_by_num(high), sizeof(buffer) - (p - buffer));
188 
189  return buffer;
190 }

228 /** Check if the application linking to the library has the correct magic number
229  *
230  * @param magic number as defined by RADIUSD_MAGIC_NUMBER
231  * @returns
232  * - 0 on success.
233  * - -1 on prefix mismatch.
234  * - -2 on version mismatch.
235  * - -3 on commit mismatch.
236  */
237 int rad_check_lib_magic(uint64_t magic)
238 {
239  if (MAGIC_PREFIX(magic) != MAGIC_PREFIX(libmagic)) {
240  ERROR("Application and libfreeradius-server magic number (prefix) mismatch."
241  " application: %x library: %x",
242  MAGIC_PREFIX(magic), MAGIC_PREFIX(libmagic));
243  return -1;
244  }
245 
246  if (MAGIC_VERSION(magic) != MAGIC_VERSION(libmagic)) {
247  ERROR("Application and libfreeradius-server magic number (version) mismatch."
248  " application: %lx library: %lx",
249  (unsigned long) MAGIC_VERSION(magic), (unsigned long) MAGIC_VERSION(libmagic));
250  return -2;
251  }
252 
253  if (MAGIC_COMMIT(magic) != MAGIC_COMMIT(libmagic)) {
254  ERROR("Application and libfreeradius-server magic number (commit) mismatch."
255  " application: %lx library: %lx",
256  (unsigned long) MAGIC_COMMIT(magic), (unsigned long) MAGIC_COMMIT(libmagic));
257  return -3;
258  }
259 
260  return 0;
261 }

320 int dependency_version_number_add(CONF_SECTION *cs, char const *name, char const *version)
321 {
322  CONF_PAIR *old;
323 
324  if (!cs) cs = default_version_cs;
325  if (!fr_cond_assert_msg(cs, "dependency_version_numbers_init() must be called before calling %s", __FUNCTION__)) {
326  return -1;
327  }
328 
329  old = cf_pair_find(cs, name);
330  if (!old) {
331  CONF_PAIR *cp;
332 
333  cp = cf_pair_alloc(cs, name, version, T_OP_EQ, T_BARE_WORD, T_SINGLE_QUOTED_STRING);

337  } else {
338  WARN("Replacing user version.%s (%s) with %s", name, cf_pair_value(old), version);
339 
340  cf_pair_replace(cs, old, version);
341  }
342 
343  return 0;
