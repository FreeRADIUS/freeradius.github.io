36 typedef enum {

38  FR_TIME_TRACKING_RUNNING, //!< We're currently tracking time in the
39  ///< running state.

44 /** A structure to track the time spent processing a request.
45  *
46  * The same structure is used by threads to track when they are
47  * running / waiting. The functions modifying fr_time_tracking_t all
48  * take an explicit "when" parameter. This parameter allows the
49  * thread to update a requests tracking structure, and then use that
50  * same fr_time_t to update the threads tracking structure.
51  *
52  * While fr_time() is fast, it is also called very often. We should
53  * therefore be careful to call it only when necessary.
54  */
55 typedef struct fr_time_tracking_s fr_time_tracking_t;
56 struct fr_time_tracking_s {
57  fr_time_tracking_state_t state; //!< What state we're currently in.
58  ///< only used for the leaf node.
59  fr_time_t last_changed; //!< last time we changed a field
60 
61  fr_time_t started; //!< Last time this tracked entity or a child
62  ///< entered the running state, or entered
63  ///< a time tracked parent.
64 
65  fr_time_t last_yielded; //!< Last time this tracked entity or a child
66  ///< yielded.
67 
68  fr_time_t last_resumed; //!< Last time this tracked entity or a child
69  ///< resumed;
70 
71  fr_time_t ended; //!< Last time this tracked entity or a child
72  ///< left the running state, or popped a time
73  ///< tracked parent.
74 
75  fr_time_delta_t running_total; //!< total time spent running
76  fr_time_delta_t waiting_total; //!< total time spent waiting
77 
78  fr_time_tracking_t *parent; //!< To update with our time tracking data when
79  ///< tracking is complete.
80 };

82 /** We use a monotonic time source
83  *
84  */
85 #define ASSERT_ON_TIME_TRAVEL(_tt, _now) \
86 do { \
87  fr_assert((_tt)->last_changed <= (_now)); \
88  fr_assert((_tt)->started <= (_now)); \
89  fr_assert((_tt)->ended <= (_now)); \
90  fr_assert((_tt)->last_yielded <= (_now)); \
91  fr_assert((_tt)->last_resumed <= (_now)); \
92 } while(0);

94 /** Set the last time a tracked entity started in its list of parents
95  *
96  */
97 #define UPDATE_PARENT_START_TIME(_tt, _now) \
98 do { \
99  fr_time_tracking_t *_parent; \
100  for (_parent = (_tt)->parent; _parent; _parent = _parent->parent) { \
101  _parent->started = _now; \
102  _parent->last_changed = _now; \
103  } \
104 } while (0)
105 
106 /** Update total run time up the list of parents
107  *
108  */
109 #define UPDATE_PARENT_RUN_TIME(_tt, _run_time, _event, _now) \
110 do { \
111  fr_time_tracking_t *_parent; \
112  for (_parent = (_tt)->parent; _parent; _parent = _parent->parent) { \
113  _parent->_event = _now; \
114  _parent->last_changed = _now; \
115  _parent->running_total += _run_time; \
116  } \
117 } while (0)

121  */

140 /** Start time tracking for a tracked entity
141  *
142  * Should be called when the tracked entity starts running.
143  *
144  * @param[in] parent to update when time tracking ends.
145  * @param[in] tt the time tracked entity.
146  * @param[in] now the current time.
147  */
148 static inline CC_HINT(nonnull) void fr_time_tracking_start(fr_time_tracking_t *parent,
149  fr_time_tracking_t *tt, fr_time_t now)
150 {
151  fr_assert(tt->state == FR_TIME_TRACKING_STOPPED);
152  fr_assert(!tt->parent);
153 
154  ASSERT_ON_TIME_TRAVEL(tt, now);
155 
156  tt->state = FR_TIME_TRACKING_RUNNING;
157  tt->started = tt->last_changed = tt->last_resumed = now;
158 
159  tt->parent = parent;
160 
161  UPDATE_PARENT_START_TIME(tt, now);
162 }
163 
164 /** Tracked entity entered a deeper time tracked code area
165  *
166  * @param[in] parent we entered. Must be a direct descendent of the
167  * current tt->parent.
168  * @param[in] tt the time tracked entity.
169  * @param[in] now the current time.
170  */
171 static inline CC_HINT(nonnull) void fr_time_tracking_push(fr_time_tracking_t *parent,
172  fr_time_tracking_t *tt, fr_time_t now)
173 {
174  fr_time_delta_t run_time;
175 
176  fr_assert(parent->parent = tt->parent);
177 
178  fr_assert(tt->state == FR_TIME_TRACKING_RUNNING);
179  run_time = now - tt->last_changed;
180  tt->last_changed = parent->started = now;
181 
182  UPDATE_PARENT_RUN_TIME(tt, run_time, last_changed, now);
183 
184  tt->parent = parent;
185 }
186 
187 /** Tracked entity left a tracked nested code area
188  *
189  * Updates parent to point to the current time tracking parent.
190  *
191  * @param[in] tt the time tracked entity.
192  * @param[in] now the current time.
193  */
194 static inline CC_HINT(nonnull) void fr_time_tracking_pop(fr_time_tracking_t *tt, fr_time_t now)
195 {
196  fr_time_delta_t run_time;
197 
198  fr_assert(tt->state == FR_TIME_TRACKING_RUNNING);
199  run_time = now - tt->last_changed;
200  tt->last_changed = tt->parent->ended = now;
201 
202  tt->running_total += run_time;
203  UPDATE_PARENT_RUN_TIME(tt, run_time, last_changed, now);
204 
205  tt->parent = tt->parent->parent;
206 }

208 /** Transition to the yielded state, recording the time we just spent running
209  *
210  * @param[in] tt the time tracked entity.
211  * @param[in] now the current time.
212  */
213 static inline CC_HINT(nonnull) void fr_time_tracking_yield(fr_time_tracking_t *tt, fr_time_t now)
214 {
215  fr_time_delta_t run_time;
216 
217  ASSERT_ON_TIME_TRAVEL(tt, now);
218 
219  fr_assert(tt->state == FR_TIME_TRACKING_RUNNING);
220  tt->state = FR_TIME_TRACKING_YIELDED;
221  tt->last_yielded = tt->last_changed = now;
222 
223  run_time = now - tt->last_resumed;
224  tt->running_total += run_time;
225  UPDATE_PARENT_RUN_TIME(tt, run_time, last_yielded, now);
226 }

229  *
230  * @param[in] tt the time tracked entity.
231  * @param[in] now the current time.
232  */
233 static inline CC_HINT(nonnull) void fr_time_tracking_resume(fr_time_tracking_t *tt, fr_time_t now)
234 {
235  fr_time_delta_t wait_time;
236 
237  ASSERT_ON_TIME_TRAVEL(tt, now);
238 
239  fr_assert(tt->state == FR_TIME_TRACKING_YIELDED);
240  tt->state = FR_TIME_TRACKING_RUNNING;
241  tt->last_resumed = tt->last_changed = now;
242

244  tt->waiting_total += wait_time;
245  UPDATE_PARENT_WAIT_TIME(tt, wait_time, last_resumed, now);
246 }

251 /** End time tracking for this entity
252  *
253  * @param[in,out] predicted Update our prediction of how long requests should run for.
254  * @param[in] tt the time tracking structure.
255  * @param[in] now the current time.
256  */
257 static inline void fr_time_tracking_end(fr_time_delta_t *predicted,
258  fr_time_tracking_t *tt, fr_time_t now)
259 {
260  fr_time_delta_t run_time;
261 
262  fr_assert(tt->state == FR_TIME_TRACKING_RUNNING);
263  ASSERT_ON_TIME_TRAVEL(tt, now);
264 
265  tt->state = FR_TIME_TRACKING_STOPPED;
266  tt->ended = tt->last_changed = now;
267 
268  run_time = now - tt->last_resumed;
269  tt->running_total += run_time;
270  UPDATE_PARENT_RUN_TIME(tt, run_time, ended, now);
271 
272  if (predicted) *predicted = !(*predicted) ? tt->running_total : RTT((*predicted), tt->running_total);
273 
274  tt->parent = NULL;
275 }
