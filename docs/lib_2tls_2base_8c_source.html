183 /*
184  * If we're linking against OpenSSL, then it is the
185  * duty of the application, if it is multithreaded,
186  * to provide OpenSSL with appropriate thread id
187  * and mutex locking functions
188  *
189  * Note: this only implements static callbacks.
190  * OpenSSL does not use dynamic locking callbacks
191  * right now, but may in the future, so we will have
192  * to add them at some point.
193  */
194 static pthread_mutex_t *global_mutexes = NULL;

196 static unsigned long _thread_id(void)
197 {
198  unsigned long ret;
199  pthread_t thread = pthread_self();
200 
201  if (sizeof(ret) >= sizeof(thread)) {
202  memcpy(&ret, &thread, sizeof(thread));
203  } else {
204  memcpy(&ret, &thread, sizeof(ret));
205  }
206 
207  return ret;
208 }

221 static void _global_mutex(int mode, int n, UNUSED char const *file, UNUSED int line)
222 {
223  if (mode & CRYPTO_LOCK) {
224  pthread_mutex_lock(&(global_mutexes[n]));
225  } else {
226  pthread_mutex_unlock(&(global_mutexes[n]));
227  }

230 /** Free the static mutexes we allocated for OpenSSL
231  *
232  */
233 static int _global_mutexes_free(pthread_mutex_t *mutexes)
234 {
235  size_t i;
236 
237  /*
238  * Ensure OpenSSL doesn't use the locks
239  */
240  CRYPTO_set_id_callback(NULL);
241  CRYPTO_set_locking_callback(NULL);
242 
243  /*
244  * Destroy all the mutexes
245  */
246  for (i = 0; i < talloc_array_length(mutexes); i++) pthread_mutex_destroy(&(mutexes[i]));
247 
248  return 0;
249 }

336 #ifdef ENABLE_OPENSSL_VERSION_CHECK
337 /** Check for vulnerable versions of libssl
338  *
339  * @param acknowledged The highest CVE number a user has confirmed is not present in the system's
340  * libssl.
341  * @return 0 if the CVE specified by the user matches the most recent CVE we have, else -1.
342  */
343 int fr_openssl_version_check(char const *acknowledged)
344 {
345  uint64_t v;
346  bool bad = false;
347  size_t i;
348 
349  /*
350  * Didn't get passed anything, that's an error.
351  */
352  if (!acknowledged || !*acknowledged) {
353  ERROR("Refusing to start until 'allow_vulnerable_openssl' is given a value");
354  return -1;
355  }
356 
357  if (strcmp(acknowledged, "yes") == 0) return 0;
358 
359  /* Check for bad versions */
360  v = (uint64_t) SSLeay();
361 
362  for (i = 0; i < (NUM_ELEMENTS(fr_openssl_defects)); i++) {
363  fr_openssl_defect_t *defect = &fr_openssl_defects[i];
364 
365  if ((v >= defect->low) && (v <= defect->high)) {
366  /*
367  * If the CVE is acknowledged, allow it.
368  */
369  if (!bad && (strcmp(acknowledged, defect->id) == 0)) return 0;
370 
371  ERROR("Refusing to start with libssl version %s (in range %s)",
372  ssl_version(), ssl_version_range(defect->low, defect->high));
373  ERROR("Security advisory %s (%s)", defect->id, defect->name);
374  ERROR("%s", defect->comment);
375 
376  /*
377  * Only warn about the first one...
378  */
379  if (!bad) {
380  INFO("Once you have verified libssl has been correctly patched, "
381  "set security.allow_vulnerable_openssl = '%s'", defect->id);
382  bad = true;
383  }
384  }
385  }
386 
387  if (bad) return -1;
388 
389  return 0;
390 }
