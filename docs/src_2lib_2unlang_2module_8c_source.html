41 /** Wrap an #fr_event_timer_t providing data needed for unlang events
42  *
43  */
44 typedef struct {
45  request_t *request; //!< Request this event pertains to.
46  int fd; //!< File descriptor to wait on.
47  unlang_module_timeout_t timeout; //!< Function to call on timeout.
48  unlang_module_fd_event_t fd_read; //!< Function to call when FD is readable.
49  unlang_module_fd_event_t fd_write; //!< Function to call when FD is writable.
50  unlang_module_fd_event_t fd_error; //!< Function to call when FD has errored.

53  void const *ctx; //!< ctx data to pass to callbacks.
54  fr_event_timer_t const *ev; //!< Event in this worker's event heap.
55 } unlang_module_event_t;

81  *
82  * @return 0
83  */
84 static int _unlang_event_free(unlang_module_event_t *ev)
85 {
86  if (ev->request) (void) request_data_get(ev->request, ev->ctx, UNLANG_TYPE_MODULE);
87 
88  if (ev->ev) {
89  (void) fr_event_timer_delete(&(ev->ev));
90  return 0;

122 /** Set a timeout for the request.
123  *
124  * Used when a module needs wait for an event. Typically the callback is set, and then the
125  * module returns unlang_module_yield().
126  *
127  * @note The callback is automatically removed on unlang_interpret_resumable().
128  *
129  * param[in] request the current request.
130  * param[in] callback to call.
131  * param[in] ctx for the callback.
132  * param[in] timeout when to call the timeout (i.e. now + timeout).
133  * @return
134  * - 0 on success.
135  * - <0 on error.
136  */
137 int unlang_module_timeout_add(request_t *request, unlang_module_timeout_t callback,
138  void const *ctx, fr_time_t when)
139 {
140  unlang_stack_t *stack = request->stack;
141  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
142  unlang_module_event_t *ev;
143  unlang_module_t *mc;
144  unlang_frame_state_module_t *state = talloc_get_type_abort(frame->state,
145  unlang_frame_state_module_t);
146 
147  fr_assert(stack->depth > 0);
148  fr_assert(frame->instruction->type == UNLANG_TYPE_MODULE);
149  mc = unlang_generic_to_module(frame->instruction);
150 
151  ev = talloc_zero(request, unlang_module_event_t);
152  if (!ev) return -1;
153 
154  ev->request = request;
155  ev->fd = -1;
156  ev->timeout = callback;
157  ev->inst = mc->instance->dl_inst->data;
158  ev->thread = state->thread;
159  ev->ctx = ctx;
160 
161  if (fr_event_timer_at(request, request->el, &ev->ev,
162  when, unlang_module_event_timeout_handler, ev) < 0) {
163  RPEDEBUG("Failed inserting event");
164  talloc_free(ev);
165  return -1;
166  }
167 
168  (void) request_data_talloc_add(request, ctx, UNLANG_TYPE_MODULE, unlang_module_event_t, ev, true, false, false);
169 
170  talloc_set_destructor(ev, _unlang_event_free);
171 
172  return 0;
173 }

239 /** Set a callback for the request.
240  *
241  * Used when a module needs to read from an FD. Typically the callback is set, and then the
242  * module returns unlang_module_yield().
243  *
244  * @note The callback is automatically removed on unlang_interpret_resumable().
245  *
246  * @param[in] request The current request.
247  * @param[in] read callback. Used for receiving and demuxing/decoding data.
248  * @param[in] write callback. Used for writing and encoding data.
249  * Where a 3rd party library is used, this should be the function
250  * issuing queries, and writing data to the socket. This should
251  * not be done in the module itself.
252  * This allows write operations to be retried in some instances,
253  * and means if the write buffer is full, the request is kept in
254  * a suspended state.
255  * @param[in] error callback. If the fd enters an error state. Should cleanup any
256  * handles wrapping the file descriptor, and any outstanding requests.
257  * @param[in] ctx for the callback.
258  * @param[in] fd to watch.
259  * @return
260  * - 0 on success.
261  * - <0 on error.
262  */
263 int unlang_module_fd_add(request_t *request,
264  unlang_module_fd_event_t read,
265  unlang_module_fd_event_t write,
266  unlang_module_fd_event_t error,
267  void const *ctx, int fd)
268 {
269  unlang_stack_t *stack = request->stack;
270  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
271  unlang_module_event_t *ev;
272  unlang_module_t *mc;
273  unlang_frame_state_module_t *state = talloc_get_type_abort(frame->state,
274  unlang_frame_state_module_t);
275 
276  fr_assert(stack->depth > 0);
277 
278  fr_assert(frame->instruction->type == UNLANG_TYPE_MODULE);
279  mc = unlang_generic_to_module(frame->instruction);
280 
281  ev = talloc_zero(request, unlang_module_event_t);
282  if (!ev) return -1;
283 
284  ev->request = request;
285  ev->fd = fd;
286  ev->fd_read = read;
287  ev->fd_write = write;
288  ev->fd_error = error;
289  ev->inst = mc->instance->dl_inst->data;
290  ev->thread = state->thread;
291  ev->ctx = ctx;
292 
293  /*
294  * Register for events on the file descriptor
295  */
296  if (fr_event_fd_insert(request, request->el, fd,
297  ev->fd_read ? unlang_event_fd_read_handler : NULL,
298  ev->fd_write ? unlang_event_fd_write_handler : NULL,
299  ev->fd_error ? unlang_event_fd_error_handler: NULL,
300  ev) < 0) {
301  talloc_free(ev);
302  return -1;
303  }
304 
305  (void) request_data_talloc_add(request, ctx, fd, unlang_module_event_t, ev, true, false, false);
306  talloc_set_destructor(ev, _unlang_event_free);
307 
308  return 0;
309 }

330 /** Push a module or submodule onto the stack for evaluation
331  *
332  * @param[out] out Where to write the result of calling the module.
333  * @param[in] request The current request.
334  * @param[in] module_instance Instance of the module to call.
335  * @param[in] method to call.
336  * @param[in] top_frame Set to UNLANG_TOP_FRAME if the interpreter should return.
337  * Set to UNLANG_SUB_FRAME if the interprer should continue.
338  * @return
339  * - 0 on success.
340  * - -1 on failure.
341  */
342 int unlang_module_push(rlm_rcode_t *out, request_t *request,
343  module_instance_t *module_instance, module_method_t method, bool top_frame)
344 {
345  unlang_stack_t *stack = request->stack;
346  unlang_stack_frame_t *frame;
347  unlang_frame_state_module_t *state;
348  unlang_module_t *mc;
349 
350  /*
351  * We need to have a unlang_module_t to push on the
352  * stack. The only sane way to do it is to attach it to
353  * the frame state.
354  */
355  MEM(mc = talloc(stack, unlang_module_t)); /* Still gets allocated from the stack pool */
356  *mc = (unlang_module_t){
357  .self = {
358  .type = UNLANG_TYPE_MODULE,
359  .name = module_instance->name,
360  .debug_name = module_instance->name,
361  .actions = {
362  [RLM_MODULE_REJECT] = 0,
363  [RLM_MODULE_FAIL] = MOD_ACTION_RETURN, /* Exit out of nested levels */
364  [RLM_MODULE_OK] = 0,
365  [RLM_MODULE_HANDLED] = 0,
366  [RLM_MODULE_INVALID] = 0,
367  [RLM_MODULE_DISALLOW] = 0,
368  [RLM_MODULE_NOTFOUND] = 0,
369  [RLM_MODULE_NOOP] = 0,
370  [RLM_MODULE_UPDATED] = 0
371  }
372  },
373  .instance = module_instance,
374  .method = method
375  };
376 
377  /*
378  * Push a new module frame onto the stack
379  */
380  if (unlang_interpret_push(request, unlang_module_to_generic(mc),
381  RLM_MODULE_UNKNOWN, UNLANG_NEXT_STOP, top_frame) < 0) {
382  return -1;
383  }
384 
385  frame = &stack->frame[stack->depth];
386  state = frame->state;
387  *state = (unlang_frame_state_module_t){
388  .p_result = out,
389  .thread = module_thread(module_instance)
390  };
391 
392  /*
393  * Bind the temporary unlang_module_t to the frame state.
394  *
395  * There aren't _that_ many children in the stack context.
396  * so we should be ok.
397  *
398  * Hopefully in future versions of talloc the O(n) problem
399  * will be fixed for stealing.
400  */
401  talloc_steal(state, mc);
402 
403  return 0;
404 }

452  *
453  * In order to use the async unlang processor the calling module needs to establish
454  * a resumption point, as the call to an xlat function may require yielding control
455  * back to the interpreter.
456  *
457  * To simplify the calling conventions, this function is provided to first push a
458  * resumption stack frame for the module, and then push an xlat stack frame.
459  *
460  * After pushing those frames the function updates the stack pointer to jump over
461  * the resumption frame and execute the xlat interpreter.
462  *
463  * When the xlat interpreter finishes, and pops the xlat frame, the unlang interpreter
464  * will then call the module resumption frame, allowing the module to continue exectuion.
465  *
466  * @param[in] ctx To allocate talloc value boxes and values in.
467  * @param[out] out Where to write the result of the expansion.
468  * @param[in] request The current request.
469  * @param[in] exp XLAT expansion to evaluate.
470  * @param[in] resume function to call when the XLAT expansion is complete.
471  * @param[in] signal function to call if a signal is received.
472  * @param[in] rctx to pass to the resume() and signal() callbacks.
473  * @return
474  * - UNLANG_ACTION_YIELD
475  */
476 unlang_action_t unlang_module_yield_to_xlat(TALLOC_CTX *ctx, fr_value_box_t **out,

478  unlang_module_resume_t resume,
479  unlang_module_signal_t signal, void *rctx)
480 {
481  /*
482  * Push the resumption point BEFORE pushing the xlat onto

484  */

486

488  * Push the xlat function
489  */

609 static void unlang_module_signal(request_t *request, fr_state_signal_t action)
610 {
611  unlang_stack_t *stack = request->stack;
612  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
613  unlang_frame_state_module_t *state = talloc_get_type_abort(frame->state, unlang_frame_state_module_t);
614  unlang_module_t *mc = unlang_generic_to_module(frame->instruction);
615 
616  if (!state->signal) return;
617 
618  request->module = mc->instance->name;
619  state->signal(&(module_ctx_t){
620  .instance = mc->instance->dl_inst->data,
621  .thread = state->thread->data
622  },
623  request, state->rctx, action);
624  /*
625  * One fewer caller for this module. Since this module
626  * has been cancelled, decrement the active callers and
627  * ignore any future signals.
628  */
629  if (action == FR_SIGNAL_CANCEL) {
630  state->thread->active_callers--;
631  state->signal = NULL;
632  }
633 }

651 /** Wrapper to call a module's resumption function
652  *
653  */
654 static unlang_action_t unlang_module_resume(rlm_rcode_t *p_result, request_t *request)
655 {
656  unlang_stack_t *stack = request->stack;
657  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
658  unlang_frame_state_module_t *state = talloc_get_type_abort(frame->state, unlang_frame_state_module_t);
659  unlang_module_t *mc = unlang_generic_to_module(frame->instruction);
660  char const *caller;
661  rlm_rcode_t rcode = RLM_MODULE_NOOP;
662  int stack_depth = stack->depth;
663  unlang_action_t ua;
664 
665  fr_assert(state->resume != NULL);
666 
667  /*
668  * Lock is noop unless instance->mutex is set.
669  */
670  caller = request->module;
671  request->module = mc->instance->name;
672 
673  safe_lock(mc->instance);
674  ua = state->resume(&rcode,
675  &(module_ctx_t){
676  .instance = mc->instance->dl_inst->data,
677  .thread = state->thread->data
678  }, request, state->rctx);
679  safe_unlock(mc->instance);
680 
681  request->rcode = rcode;
682  request->module = caller;
683 
684  /*
685  * It is now marked as "stop" when it wasn't before, we
686  * must have been blocked. We do NOT decrement
687  * "active_callers" if it's already been cancelled in a
688  * signal.
689  */
690  if ((ua == UNLANG_ACTION_STOP_PROCESSING) || (request->master_state == REQUEST_STOP_PROCESSING)) {
691  RWARN("Module %s or worker signalled to stop processing request", mc->instance->module->name);
692  if (state->p_result) *state->p_result = rcode;
693 
694  if (state->signal) state->thread->active_callers--;
695 
696  *p_result = rcode;
697  return UNLANG_ACTION_STOP_PROCESSING;
698  }
699 
700  RDEBUG2("%s (%s)", frame->instruction->name ? frame->instruction->name : "",
701  fr_table_str_by_value(mod_rcode_table, rcode, "<invalid>"));
702 
703  if (ua == UNLANG_ACTION_YIELD) {
704  if (stack_depth < stack->depth) return UNLANG_ACTION_PUSHED_CHILD;
705  fr_assert(stack_depth == stack->depth);
706  *p_result = rcode;
707  return UNLANG_ACTION_YIELD;
708  }
709 
710  state->thread->active_callers--;
711 
712  /*
713  * The module is done. But, running it pushed one or
714  * more asynchronous calls onto the stack. These need to
715  * be run before the next module runs.
716  */
717  if (ua == UNLANG_ACTION_PUSHED_CHILD) {
718  state->rcode = rcode;
719  repeatable_set(frame);
720  frame->process = unlang_module_resume_final;
721  return UNLANG_ACTION_PUSHED_CHILD;
722  }
723 
724  request->rcode = rcode;
725  if (state->p_result) *state->p_result = rcode;
726 
727  *p_result = rcode;
728  return ua; /* Usually UNLANG_ACTION_CALCULATE_RESULT */
729 }

731 /** Yield a request back to the interpreter from within a module
732  *
733  * This passes control of the request back to the unlang interpreter, setting
734  * callbacks to execute when the request is 'signalled' asynchronously, or whatever
735  * timer or I/O event the module was waiting for occurs.
736  *
737  * @note The module function which calls #unlang_module_yield should return control
738  * of the C stack to the unlang interpreter immediately after calling #unlang_module_yield.
739  * A common pattern is to use ``return unlang_module_yield(...)``.
740  *
741  * @param[in] request The current request.
742  * @param[in] resume Called on unlang_interpret_resumable().
743  * @param[in] signal Called on unlang_action().
744  * @param[in] rctx to pass to the callbacks.
745  * @return
746  * - UNLANG_ACTION_YIELD.
747  */
748 unlang_action_t unlang_module_yield(request_t *request,
749  unlang_module_resume_t resume, unlang_module_signal_t signal, void *rctx)
750 {
751  unlang_stack_t *stack = request->stack;
752  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
753  unlang_frame_state_module_t *state = talloc_get_type_abort(frame->state, unlang_frame_state_module_t);
754 
755  REQUEST_VERIFY(request); /* Check the yielded request is sane */
756 
757  state->rctx = rctx;
758  state->resume = resume;
759  state->signal = signal;
760 
761  /*
762  * We set the repeatable flag here, so that the resume
763  * function is always called going back up the stack.
764  * This setting is normally done in the intepreter.
765  * However, the caller of this function may call us, and
766  * then push *other* things onto the stack. Which means
767  * that the interpreter never gets a chance to set this
768  * flag.
769  */
770  frame->process = unlang_module_resume;
771  repeatable_set(frame);
772 
773  return UNLANG_ACTION_YIELD;
774 }

776 static unlang_action_t unlang_module(rlm_rcode_t *p_result, request_t *request)
777 {
778  unlang_module_t *mc;
779  unlang_stack_t *stack = request->stack;
780  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
781  unlang_t *instruction = frame->instruction;
782  unlang_frame_state_module_t *state = talloc_get_type_abort(frame->state, unlang_frame_state_module_t);
783  int stack_depth = stack->depth;
784  char const *caller;
785  rlm_rcode_t rcode = RLM_MODULE_NOOP;
786  unlang_action_t ua;
787 
788 #ifndef NDEBUG
789  int unlang_indent = request->log.unlang_indent;
790 #endif
791 
792  /*
793  * Process a stand-alone child, and fall through
794  * to dealing with it's parent.
795  */
796  mc = unlang_generic_to_module(instruction);
797  fr_assert(mc);
798 
799  RDEBUG4("[%i] %s - %s (%s)", stack->depth, __FUNCTION__,
800  mc->instance->name, mc->instance->module->name);
801 
802  /*
803  * Grab the thread/module specific data if any exists.
804  */
805  state->thread = module_thread(mc->instance);
806  state->p_result = NULL;
807  fr_assert(state->thread != NULL);
808 
809  /*
810  * Return administratively configured return code
811  */
812  if (mc->instance->force) {
813  rcode = mc->instance->code;
814  ua = UNLANG_ACTION_CALCULATE_RESULT;
815  goto done;
816  }
817 
818  /*
819  * For logging unresponsive children.
820  */
821  state->thread->total_calls++;
822 
823  caller = request->module;
824  request->module = mc->instance->name;
825  safe_lock(mc->instance); /* Noop unless instance->mutex set */
826  ua = mc->method(&rcode,
827  &(module_ctx_t){
828  .instance = mc->instance->dl_inst->data,
829  .thread = state->thread->data
830  },
831  request);
832  safe_unlock(mc->instance);
833  request->module = caller;
834 
835  /*
836  * It is now marked as "stop" when it wasn't before, we
837  * must have been blocked.
838  */
839  if ((ua == UNLANG_ACTION_STOP_PROCESSING) || (request->master_state == REQUEST_STOP_PROCESSING)) {
840  RWARN("Module %s became unblocked", mc->instance->module->name);
841  if (state->p_result) *state->p_result = rcode;
842 
843  *p_result = rcode;
844  return UNLANG_ACTION_STOP_PROCESSING;
845  }
846 
847  /*
848  * Must be left at RDEBUG() level otherwise RDEBUG becomes pointless
849  */
850  RDEBUG("%s (%s)", instruction->name ? instruction->name : "",
851  fr_table_str_by_value(mod_rcode_table, rcode, "<invalid>"));
852 
853 
854  if (ua == UNLANG_ACTION_YIELD) {
855  state->thread->active_callers++;
856  if (stack_depth < stack->depth) return UNLANG_ACTION_PUSHED_CHILD;
857  fr_assert(stack_depth == stack->depth);
858  frame->process = unlang_module_resume;
859  return UNLANG_ACTION_YIELD;
860  }
861 
862 done:
863  fr_assert(unlang_indent == request->log.unlang_indent);
864  fr_assert(rcode >= RLM_MODULE_REJECT);
865  fr_assert(rcode < RLM_MODULE_NUMCODES);
866 
867  /*
868  * The module is done. But, running it pushed one or
869  * more asynchronous calls onto the stack. These need to
870  * be run before the next module runs.
871  */
872  if (ua == UNLANG_ACTION_PUSHED_CHILD) {
873  state->rcode = rcode;
874  repeatable_set(frame);
875  frame->process = unlang_module_resume_final;
876  return UNLANG_ACTION_PUSHED_CHILD;
877  }
878 
879  request->rcode = rcode;
880  if (state->p_result) *state->p_result = rcode;
881 
882  *p_result = rcode;
883  return ua;
884 }
