27 /** Copy a known or unknown attribute to produce an unknown attribute with the specified name
28  *
29  * Will copy the complete hierarchy down to the first known attribute.
30  */
31 fr_dict_attr_t *fr_dict_unknown_acopy(TALLOC_CTX *ctx, fr_dict_attr_t const *da, char const *new_name)
32 {
33  fr_dict_attr_t *n;
34  fr_dict_attr_t const *parent;
35  fr_dict_attr_flags_t flags = da->flags;
36  fr_type_t type = da->type;
37 
38  /*
39  * Set the unknown flag, and clear other flags which are
40  * no longer relevant.
41  */
42  flags.is_unknown = 1;
43  flags.array = 0;
44  flags.has_value = 0;
45 
46  /*
47  * Allocate an attribute.
48  */
49  n = dict_attr_alloc_null(ctx);
50  if (!n) return NULL;
51 
52  /*
53  * We want to have parent / child relationships, AND to
54  * copy all unknown parents, AND to free the unknown
55  * parents when this 'da' is freed. We therefore talloc
56  * the parent from the 'da'.
57  */
58  if (da->parent->flags.is_unknown) {
59  parent = fr_dict_unknown_acopy(n, da->parent, NULL);
60  if (!parent) {
61  talloc_free(n);
62  return NULL;
63  }
64 
65  } else {
66  parent = da->parent;
67  }
68 
69  /*
70  * VENDOR and TLV are structural, and can have unknown
71  * children. But they're left alone. All other base
72  * types are mangled to OCTETs.
73  */
74  switch (type) {
75  case FR_TYPE_VENDOR:
76  case FR_TYPE_TLV:
77  case FR_TYPE_VSA:
78  break;
79 
80  default:
81  type = FR_TYPE_OCTETS;
82  break;
83  }
84 
85  /*
86  * Initialize the rest of the fields.
87  */
88  dict_attr_init(&n, parent, new_name ? new_name : da->name, da->attr, type, &flags);
89  dict_attr_ext_copy_all(&n, da);
90  DA_VERIFY(n);
91 
92  return n;
93 }

95 /** Converts an unknown to a known by adding it to the internal dictionaries.
96  *
97  * Does not free old #fr_dict_attr_t, that is left up to the caller.
98  *
99  * @param[in] dict of protocol context we're operating in.
100  * If NULL the internal dictionary will be used.
101  * @param[in] old unknown attribute to add.
102  * @return
103  * - Existing #fr_dict_attr_t if old was found in a dictionary.
104  * - A new entry representing old.
105  */
106 fr_dict_attr_t const *fr_dict_unknown_add(fr_dict_t *dict, fr_dict_attr_t const *old)
107 {
108  fr_dict_attr_t const *da;
109  fr_dict_attr_t const *parent;
110  fr_dict_attr_flags_t flags;
111 
112  if (unlikely(dict->read_only)) {
113  fr_strerror_printf("%s dictionary has been marked as read only", fr_dict_root(dict)->name);
114  return NULL;
115  }
116 
117  da = fr_dict_attr_by_name(dict, old->name);
118  if (da) return da;
119 
120  /*
121  * Define the complete unknown hierarchy
122  */
123  if (old->parent && old->parent->flags.is_unknown) {
124  parent = fr_dict_unknown_add(dict, old->parent);
125  if (!parent) {
126  fr_strerror_printf_push("Failed adding parent \"%s\"", old->parent->name);
127  return NULL;
128  }
129  } else {
130 #ifdef __clang_analyzer__
131  if (!old->parent) return NULL;
132 #endif
133  parent = old->parent;
134  }
135 
136  memcpy(&flags, &old->flags, sizeof(flags));
137  flags.is_unknown = 0;
138 
139  /*
140  * If this is a vendor, we skip most of the sanity
141  * checks and add it to the vendor hash, and add it
142  * as a child attribute to the Vendor-Specific
143  * container.
144  */
145  if (old->type == FR_TYPE_VENDOR) {
146  fr_dict_attr_t *mutable, *n;
147 
148  if (dict_vendor_add(dict, old->name, old->attr) < 0) return NULL;
149 
150  n = dict_attr_alloc(dict->pool, parent, old->name, old->attr, old->type, &flags);
151  if (unlikely(!n)) return NULL;
152 
153  /*
154  * Setup parenting for the attribute
155  */
156  memcpy(&mutable, &old->parent, sizeof(mutable));
157  if (dict_attr_child_add(mutable, n) < 0) return NULL;
158 
159  return n;
160  }
161 
162  /*
163  * Look up the attribute by number. If it doesn't exist,
164  * add it both by name and by number. If it does exist,
165  * add it only by name.
166  */
167  da = fr_dict_attr_child_by_num(parent, old->attr);
168  if (da) {
169  fr_dict_attr_t *n;
170 
171  n = dict_attr_alloc(dict->pool, parent, old->name, old->attr, old->type, &flags);
172  if (!n) return NULL;
173 
174  /*
175  * Add the unknown by NAME. e.g. if the admin does "Attr-26", we want
176  * to return "Attr-26", and NOT "Vendor-Specific". The rest of the server
177  * is responsible for converting "Attr-26 = 0x..." to an actual attribute,
178  * if it so desires.
179  */
180  if (dict_attr_add_by_name(dict, n) < 0) {
181  talloc_free(n);
182  return NULL;
183  }
184 
185  return n;
186  }
187 
188 #ifdef __clang_analyzer__
189  if (!old->name) return NULL;
190 #endif
191 
192  /*
193  * Add the attribute by both name and number.
194  *
195  * Fixme - Copy extensions?
196  */
197  if (fr_dict_attr_add(dict, parent, old->name, old->attr, old->type, &flags) < 0) return NULL;
198 
199  /*
200  * For paranoia, return it by name.
201  */
202  return fr_dict_attr_by_name(dict, old->name);
203 }

229 /** Build an unknown vendor, parented by a VSA attribute
230  *
231  * This allows us to complete the path back to the dictionary root in the case
232  * of unknown attributes with unknown vendors.
233  *
234  * @note Will return known vendors attributes where possible. Do not free directly,
235  * use #fr_dict_unknown_free.
236  *
237  * @param[in] ctx to allocate the vendor attribute in.
238  * @param[out] out Where to write point to new unknown dict attr
239  * representing the unknown vendor.
240  * @param[in] parent of the VSA attribute.
241  * @param[in] vendor id.
242  * @return
243  * - 0 on success.
244  * - -1 on failure.
245  */
246 int fr_dict_unknown_vendor_afrom_num(TALLOC_CTX *ctx, fr_dict_attr_t **out,
247  fr_dict_attr_t const *parent, unsigned int vendor)
248 {
249  fr_dict_attr_flags_t flags = {
250  .is_unknown = 1,
251  .type_size = 1,
252  .length = 1
253  };
254 
255  if (!fr_cond_assert(parent)) {
256  fr_strerror_printf("%s: Invalid argument - parent was NULL", __FUNCTION__);
257  return -1;
258  }
259 
260  *out = NULL;
261 
262  /*
263  * Vendor attributes can occur under VSA attribute.
264  */
265  switch (parent->type) {
266  case FR_TYPE_VSA:
267  if (!fr_cond_assert(!parent->flags.is_unknown)) return -1;
268 
269  *out = dict_attr_alloc(ctx, parent, NULL, vendor, FR_TYPE_VENDOR, &flags);
270 
271  return 0;
272 
273  case FR_TYPE_VENDOR:
274  if (!fr_cond_assert(!parent->flags.is_unknown)) return -1;
275  fr_strerror_printf("Unknown vendor cannot be parented by another vendor");
276  return -1;
277 
278  default:
279  fr_strerror_printf("Unknown vendors can only be parented by 'vsa' or 'evs' "
280  "attributes, not '%s'", fr_table_str_by_value(fr_value_box_type_table, parent->type, "?Unknown?"));
281  return -1;
282  }
283 }

285 /** Allocates an unknown attribute
286  *
287  * @note If vendor != 0, an unknown vendor (may) also be created, parented by
288  * the correct VSA attribute. This is accessible via da->parent,
289  * and will be use the unknown da as its talloc parent.
290  *
291  * @param[in] ctx to allocate DA in.
292  * @param[in] parent of the unknown attribute (may also be unknown).
293  * @param[in] attr number.
294  * @param[in] vendor number.
295  * @return 0 on success.
296  */
297 fr_dict_attr_t const *fr_dict_unknown_afrom_fields(TALLOC_CTX *ctx, fr_dict_attr_t const *parent,
298  unsigned int vendor, unsigned int attr)
299 {
300  fr_dict_attr_t const *da;
301  fr_dict_attr_t *n;
302  fr_dict_attr_t *new_parent = NULL;
303  fr_dict_attr_flags_t flags = {
304  .is_unknown = 1
305  };
306 
307  if (!fr_cond_assert(parent)) {
308  fr_strerror_printf("%s: Invalid argument - parent was NULL", __FUNCTION__);
309  return NULL;
310  }
311 
312  /*
313  * If there's a vendor specified, we check to see
314  * if the parent is a VSA, and if it is
315  * we either lookup the vendor to get the correct
316  * attribute, or bridge the gap in the tree, with an
317  * unknown vendor.
318  *
319  * We need to do the check, as the parent could be
320  * a TLV, in which case the vendor should be known
321  * and we don't need to modify the parent.
322  */
323  if (vendor && (parent->type == FR_TYPE_VSA)) {
324  da = fr_dict_attr_child_by_num(parent, vendor);
325  if (!da) {
326  if (fr_dict_unknown_vendor_afrom_num(ctx, &new_parent, parent, vendor) < 0) return NULL;
327  da = new_parent;
328  }
329  parent = da;
330 
331  /*
332  * Need to clone the unknown hierachy, as unknown
333  * attributes must parent the complete heirachy,
334  * and cannot share any parts with any other unknown
335  * attributes.
336  */
337  } else if (parent->flags.is_unknown) {
338  new_parent = fr_dict_unknown_acopy(ctx, parent, NULL);
339  parent = new_parent;
340  }
341 
342  n = dict_attr_alloc(ctx, parent, NULL, attr, FR_TYPE_OCTETS, &flags);
343  if (!n) return NULL;
344 
345  /*
346  * The config files may reference the unknown by name.
347  * If so, use the pre-defined name instead of an unknown
348  * one!
349  */
350  da = fr_dict_attr_by_name(dict_by_da(parent), n->name);
351  if (da) {
352  fr_dict_unknown_free(&parent);
353  parent = n;
354  fr_dict_unknown_free(&parent);
355  return da;
356  }
357 
358  /*
359  * Ensure the parent is freed at the same time as the
360  * unknown DA. This should be OK as we never parent
361  * multiple unknown attributes off the same parent.
362  */
363  if (new_parent && new_parent->flags.is_unknown) talloc_steal(n, new_parent);
364 
365  return n;
366 }

407 /** Create a fr_dict_attr_t from an ASCII attribute and value
408  *
409  * Where the attribute name is in the form:
410  * - Attr-%d
411  * - Attr-%d.%d.%d...
412  *
413  * @copybrief fr_dict_unknown_afrom_fields
414  *
415  * @note If vendor != 0, an unknown vendor (may) also be created, parented by
416  * the correct VSA attribute. This is accessible via vp->parent,
417  * and will be use the unknown da as its talloc parent.
418  *
419  * @param[in] ctx to alloc new attribute in.
420  * @param[out] out Where to write the head of the chain unknown
421  * dictionary attributes.
422  * @param[in] parent Attribute to use as the root for resolving OIDs in.
423  * Usually the root of a protocol dictionary.
424  * @param[in] oid_str of attribute.
425  * @return
426  * - The number of bytes parsed on success.
427  * - <= 0 on failure. Negative offset indicates parse error position.
428  */
429 ssize_t fr_dict_unknown_afrom_oid_str(TALLOC_CTX *ctx, fr_dict_attr_t **out,
430  fr_dict_attr_t const *parent, char const *oid_str)
431 {
432  char const *p = oid_str, *end = oid_str + strlen(oid_str);
433  fr_dict_attr_t const *our_parent = parent;
434  fr_dict_attr_t *n = NULL, *our_da;
435  fr_dict_attr_flags_t flags = {
436  .is_unknown = true
437  };
438 
439  if (!fr_cond_assert(parent)) {
440  fr_strerror_printf("%s: Invalid argument - parent was NULL", __FUNCTION__);
441  return -1;
442  }
443 
444  *out = NULL;
445 
446  if (fr_dict_valid_oid_str(oid_str, -1) < 0) return -1;
447 
448  /*
449  * All unknown attributes are of the form "Attr-#.#.#.#"
450  */
451  if (strncasecmp(p, "Attr-", 5) != 0) {
452  fr_strerror_printf("Unknown attribute '%s'", oid_str);
453  return 0;
454  }
455  p += 5;
456 
457  /*
458  * Allocate the final attribute first, so that any
459  * unknown parents can be freed when this da is freed.
460  *
461  * See fr_dict_unknown_acopy() for more details.
462  *
463  * Note also that we copy the input name, even if it is
464  * not normalized.
465  *
466  * While the name of this attribute is "Attr-#.#.#", one
467  * or more of the leading components may, in fact, be
468  * known.
469  */
470  n = dict_attr_alloc_null(ctx);
471 
472  /*
473  * Parse the name of this attribute
474  */
475  do {
476  unsigned int num;
477  fr_dict_attr_t const *da = NULL;
478 
479  if (fr_dict_oid_component(&num, &p) < 0) {
480  error:
481  talloc_free(n);
482  return -(p - oid_str);
483  }
484 
485  switch (*p) {
486  /*
487  * Structural attribute
488  */
489  case '.':
490  if (!our_parent) goto is_root;
491 
492  da = fr_dict_attr_child_by_num(our_parent, num);
493  if (!da) { /* Unknown component */
494  switch (our_parent->type) {
495  case FR_TYPE_VSA:
496  da = fr_dict_attr_child_by_num(our_parent, num);
497  if (!fr_cond_assert(!da || (da->type == FR_TYPE_VENDOR))) goto error;
498 
499  if (!da) {
500  if (fr_dict_unknown_vendor_afrom_num(n, &our_da,
501  our_parent, num) < 0) {
502  goto error;
503  }
504  da = our_da;
505  }
506  break;
507 
508  case FR_TYPE_TLV:
509  is_root:
510  if (fr_dict_unknown_attr_afrom_num(n, &our_da, our_parent, num) < 0) {
511  goto error;
512  }
513  da = our_da;
514  break;
515 
516  /*
517  * Can't have a FR_TYPE_STRING inside a
518  * FR_TYPE_STRING (for example)
519  */
520  default:
521  fr_strerror_printf("Parent OID component (%s) in \"%.*s\" specified a "
522  "non-structural type (%s)", our_parent->name,
523  (int)(p - oid_str), oid_str,
524  fr_table_str_by_value(fr_value_box_type_table,
525  our_parent->type, "<INVALID>"));
526  goto error;
527  }
528  }
529  our_parent = da;
530  break;
531 
532  /*
533  * Leaf attribute
534  */
535  case '\0':
536  dict_attr_init(&n, our_parent, oid_str, num, FR_TYPE_OCTETS, &flags);
537  break;
538  }
539  p++;
540  } while (p < end);
541 
542  /*
543  * @todo - if we really care about normalization, re-print the name here, normalized.
544  */
545 
546  DA_VERIFY(n);
547 
548  *out = n;
549 
550  return end - oid_str;
551 }

553 /** Create a dictionary attribute by name embedded in another string
554  *
555  * Find the first invalid attribute name char in the string pointed to by name.
556  *
557  * Copy the characters between the start of the name string and the first none
558  * #fr_dict_attr_allowed_chars char to a buffer and initialise da as an unknown
559  * attribute.
560  *
561  * @param[in] ctx To allocate unknown #fr_dict_attr_t in.
562  * @param[out] out Where to write the head of the chain unknown
563  * dictionary attributes.
564  * @param[in] parent Attribute to use as the root for resolving OIDs in.
565  * Usually the root of a protocol dictionary.
566  * @param[in] name string start.
567  * @return
568  * - <= 0 on failure.
569  * - The number of bytes of name consumed on success.
570  */
571 ssize_t fr_dict_unknown_afrom_oid_substr(TALLOC_CTX *ctx, fr_dict_attr_t **out,
572  fr_dict_attr_t const *parent, char const *name)
573 {
574  char const *p;
575  size_t len;
576  char buffer[FR_DICT_ATTR_MAX_NAME_LEN + 1];
577  ssize_t slen;
578 
579  if (!name || !*name) return 0;
580 
581  /*
582  * Advance p until we get something that's not part of
583  * the dictionary attribute name.
584  */
585  for (p = name; fr_dict_attr_allowed_chars[(uint8_t)*p] || (*p == '.') || (*p == '-'); p++);
586 
587  len = p - name;
588  if (len > FR_DICT_ATTR_MAX_NAME_LEN) {
589  fr_strerror_printf("Attribute name too long");
590  return 0;
591  }
592  if (len == 0) {
593  fr_strerror_printf("Invalid attribute name");
594  return 0;
595  }
596  strlcpy(buffer, name, len + 1);
597 
598  slen = fr_dict_unknown_afrom_oid_str(ctx, out, parent, buffer);
599  if (slen <= 0) return slen;
600 
601  return p - name;
602 }

605 /** Check to see if we can convert a nested TLV structure to known attributes
606  *
607  * @param[in] dict to search in.
608  * @param[in] da Nested tlv structure to convert.
609  * @return
610  * - NULL if we can't.
611  * - Known attribute if we can.
612  */
613 fr_dict_attr_t const *fr_dict_attr_known(fr_dict_t const *dict, fr_dict_attr_t const *da)
614 {
615  INTERNAL_IF_NULL(dict, NULL);
616 
617  if (!da->flags.is_unknown) return da; /* It's known */
618 
619  if (da->parent) {
620  fr_dict_attr_t const *parent;
621 
622  parent = fr_dict_attr_known(dict, da->parent);
623  if (!parent) return NULL;
624 
625  return fr_dict_attr_child_by_num(parent, da->attr);
626  }
627 
628  if (dict->root == da) return dict->root;
629  return NULL;
630 }
