177 /** SIM AT on-the-wire format attribute sizes
178  *
179  * Holds the min/max sizes of all supported SIM AT attribute values as they
180  * would be found in a SIM AT packet.
181  *
182  * These sizes may be different than the sizes of INTERNAL formats, PRESENTATION
183  * formats and generic NETWORK formats.
184  */
185 size_t const fr_aka_sim_attr_sizes[FR_TYPE_MAX + 1][2] = {
186  [FR_TYPE_INVALID] = {~0, 0},
187 
188  [FR_TYPE_STRING] = {0, ~0},
189  [FR_TYPE_OCTETS] = {0, ~0},
190 
191  [FR_TYPE_BOOL] = {2, 2},
192  [FR_TYPE_UINT8] = {1, 1},
193  [FR_TYPE_UINT16] = {2, 2},
194  [FR_TYPE_UINT32] = {4, 4},
195  [FR_TYPE_UINT64] = {8, 8},
196 
197  [FR_TYPE_TLV] = {2, ~0},
198 
199  [FR_TYPE_MAX] = {~0, 0} //!< Ensure array covers all types.
200 };

202 /** Return the on-the-wire length of an attribute value
203  *
204  * @param[in] vp to return the length of.
205  * @return the length of the attribute.
206  */
207 size_t fr_aka_sim_attr_len(fr_pair_t const *vp)
208 {
209  switch (vp->vp_type) {
210  case FR_TYPE_VARIABLE_SIZE:
211  return vp->vp_length;
212 
213  default:
214  return fr_aka_sim_attr_sizes[vp->vp_type][0];
215 
216  case FR_TYPE_STRUCTURAL:
217  if (!fr_cond_assert(0)) return 0;
218  }
219 }
220 
221 /** Return the number of bytes before the octets value
222  *
223  */
224 size_t fr_aka_sim_octets_prefix_len(fr_dict_attr_t const *da)
225 {
226  if (da->flags.array) return 0; /* Array elements have no padding */
227  if (!da->flags.length) return 2; /* Variable length attributes need length field */
228  if (!(da->flags.length % 4)) return 2; /* Values that are multiples of four have 2 reserved bytes */
229  return 0; /* Everything else has zero padding bytes */
230 }

232 int fr_aka_sim_init(void)
233 {
234  if (instance_count > 0) {
235  instance_count++;
236  return 0;
237  }
238 
239  if (fr_dict_autoload(libfreeradius_aka_sim_dict) < 0) {
240  PERROR("Failed loading libfreeradius-eap-aka-sim dictionaries");
241  return -1;
242  }
243  if (fr_dict_attr_autoload(libfreeradius_aka_sim_dict_attr) < 0) {
244  PERROR("Failed loading libfreeradius-eap-aka-sim attributes");
245  fr_dict_autofree(libfreeradius_aka_sim_dict);
246  return -1;
247  }
248  instance_count++;
249 
250  return 0;
251 }
