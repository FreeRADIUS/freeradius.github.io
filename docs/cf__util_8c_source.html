41 /** Return the next child that's of the specified type
42  *
43  * @param[in] parent to return children from.
44  * @param[in] prev child to start searching from.
45  * @param[in] type to search for.
46  * @return
47  * - The next #CONF_ITEM that's a child of ci matching type.
48  * - NULL if no #CONF_ITEM matches that criteria.
49  */
50 static CONF_ITEM *cf_next(CONF_ITEM const *parent, CONF_ITEM const *prev, CONF_ITEM_TYPE type)
51 {
52  CONF_ITEM *ci;
53 
54  for (ci = prev ? prev->next : parent->child;
55  ci;
56  ci = ci->next) {
57  if (ci->type == type) return ci;
58  }
59 
60  return NULL;
61 }

65 /** Return the next child that's of the specified type with the specified identifiers
66  *
67  * @param[in] parent The section we're searching in.
68  * @param[in] type of #CONF_ITEM we're searching for.
69  * @param[in] ident1 The first identifier.
70  * @param[in] ident2 The second identifier. Special value CF_IDENT_ANY
71  * can be used to match any ident2 value.
72  * @return
73  * - The first matching item.
74  * - NULL if no items matched.
75  */
76 static CONF_ITEM *cf_find(CONF_ITEM const *parent, CONF_ITEM_TYPE type, char const *ident1, char const *ident2)
77 {
78  CONF_SECTION cs_find;
79  CONF_PAIR cp_find;
80  CONF_DATA cd_find;
81  CONF_ITEM *find;
82 
83  if (!parent) return NULL;
84  if (!parent->child) return NULL; /* No children */
85 
86  if (!ident1) return cf_next(parent, NULL, type);
87 
88  switch (type) {
89  case CONF_ITEM_SECTION:
90  memset(&cs_find, 0, sizeof(cs_find));
91  cs_find.item.type = CONF_ITEM_SECTION;
92  cs_find.name1 = ident1;
93  if (!IS_WILDCARD(ident2)) cs_find.name2 = ident2;
94 
95  find = (CONF_ITEM *)&cs_find;
96  break;
97 
98  case CONF_ITEM_PAIR:
99  fr_assert((ident2 == NULL) || IS_WILDCARD(ident2));
100 
101  memset(&cp_find, 0, sizeof(cp_find));
102  cp_find.item.type = CONF_ITEM_PAIR;
103  cp_find.attr = ident1;
104 
105  find = (CONF_ITEM *)&cp_find;
106  break;
107 
108  case CONF_ITEM_DATA:
109  memset(&cd_find, 0, sizeof(cd_find));
110  cd_find.item.type = CONF_ITEM_DATA;
111  cd_find.type = ident1;
112  if (!IS_WILDCARD(ident2)) cd_find.name = ident2;
113 
114  find = (CONF_ITEM *)&cd_find;
115  break;
116 
117  default:
118  fr_assert_fail(NULL);
119  return NULL;
120  }
121 
122  /*
123  * No ident1, iterate over the child list
124  */
125  if (IS_WILDCARD(ident1)) {
126  CONF_ITEM *ci;
127 
128  for (ci = parent->child;
129  ci != NULL;
130  ci = ci->next) {
131  if (find->type != ci->type) continue;
132 
133  if (cf_ident2_cmp(find, ci) == 0) break;
134  }
135 
136  return ci;
137  }
138 
139  /*
140  * No ident2, use the ident1 tree.
141  */
142  if (IS_WILDCARD(ident2)) return rbtree_finddata(parent->ident1, find);
143 
144  /*
145  * Both ident1 and ident2 use the ident2 tree.
146  */
147  return rbtree_finddata(parent->ident2, find);
148 }

150 /** Return the next child that's of the specified type with the specified identifiers
151  *
152  * @param[in] parent The section we're searching in.
153  * @param[in] prev item we found, or NULL to start from the beginning.
154  * @param[in] type of #CONF_ITEM we're searching for.
155  * @param[in] ident1 The first identifier.
156  * @param[in] ident2 The second identifier. Special value CF_IDENT_ANY
157  * can be used to match any ident2 value.
158  * @return
159  * - The first matching item.
160  * - NULL if no items matched.
161  */
162 static CONF_ITEM *cf_find_next(CONF_ITEM const *parent, CONF_ITEM const *prev,
163  CONF_ITEM_TYPE type, char const *ident1, char const *ident2)
164 {
165  CONF_SECTION cs_find;
166  CONF_PAIR cp_find;
167  CONF_DATA cd_find;
168  CONF_ITEM *find;
169  CONF_ITEM *ci;
170 
171  if (!parent) return NULL;
172 
173  if (!prev) {
174  if (!ident1) return cf_next(parent, NULL, type);
175  return cf_find(parent, type, ident1, ident2);
176  }
177  if (!ident1) return cf_next(parent, prev, type);
178 
179  switch (type) {
180  case CONF_ITEM_SECTION:
181  memset(&cs_find, 0, sizeof(cs_find));
182  cs_find.item.type = CONF_ITEM_SECTION;
183  cs_find.name1 = ident1;
184  if (!IS_WILDCARD(ident2)) cs_find.name2 = ident2;
185 
186  find = (CONF_ITEM *)&cs_find;
187  break;
188 
189  case CONF_ITEM_PAIR:
190  fr_assert((ident2 == NULL) || IS_WILDCARD(ident2));
191 
192  memset(&cp_find, 0, sizeof(cp_find));
193  cp_find.item.type = CONF_ITEM_PAIR;
194  cp_find.attr = ident1;
195 
196  find = (CONF_ITEM *)&cp_find;
197  break;
198 
199  case CONF_ITEM_DATA:
200  memset(&cd_find, 0, sizeof(cd_find));
201  cd_find.item.type = CONF_ITEM_DATA;
202  cd_find.type = ident1;
203  if (!IS_WILDCARD(ident2)) cd_find.name = ident2;
204 
205  find = (CONF_ITEM *)&cd_find;
206  break;
207 
208  default:
209  fr_assert_fail(NULL);
210  return NULL;
211  }
212 
213  if (IS_WILDCARD(ident1)) {
214  for (ci = prev->next;
215  ci && (cf_ident2_cmp(ci, find) != 0);
216  ci = ci->next);
217 
218  return ci;
219  }
220 
221  if (IS_WILDCARD(ident2)) {
222  for (ci = prev->next;
223  ci && (_cf_ident1_cmp(ci, find) != 0);
224  ci = ci->next) {
225  fr_assert(ci->next != ci);
226  }
227 
228  return ci;
229  }
230 
231  for (ci = prev->next;
232  ci && (_cf_ident2_cmp(ci, find) != 0);
233  ci = ci->next);
234 
235  return ci;
236 }

238 /** Compare the first identifier of a child
239  *
240  * For CONF_ITEM_PAIR this is 'attr'.
241  * For CONF_ITEM_SECTION this is 'name1'.
242  * For CONF_ITEM_DATA this is 'type'.
243  *
244  * @param[in] a First CONF_ITEM to compare.
245  * @param[in] b Second CONF_ITEM to compare.
246  * @return
247  * - >0 if a > b.
248  * - <0 if a < b.
249  * - 0 if a == b.
250  */
251 static inline int _cf_ident1_cmp(void const *a, void const *b)
252 {
253  CONF_ITEM_TYPE type;
254 
255  {
256  CONF_ITEM const *one = a;
257  CONF_ITEM const *two = b;
258 
259  if (one->type > two->type) return +1;
260  if (one->type < two->type) return -1;
261 
262  type = one->type;
263  }
264 
265  switch (type) {
266  case CONF_ITEM_PAIR:
267  {
268  CONF_PAIR const *one = a;
269  CONF_PAIR const *two = b;
270 
271  return strcmp(one->attr, two->attr);
272  }
273 
274  case CONF_ITEM_SECTION:
275  {
276  CONF_SECTION const *one = a;
277  CONF_SECTION const *two = b;
278 
279  return strcmp(one->name1, two->name1);
280  }
281 
282  case CONF_ITEM_DATA:
283  {
284  CONF_DATA const *one = a;
285  CONF_DATA const *two = b;
286 
287  return strcmp(one->type, two->type);
288  }
289 
290  default:
291  fr_assert_fail(NULL);
292  return 0;
293  }
294 }

296 /** Compare only the second identifier of a child
297  *
298  * For CONF_ITEM_SECTION this is 'name2'.
299  * For CONF_ITEM_DATA this is 'name'.
300  *
301  * @param[in] a First CONF_ITEM to compare.
302  * @param[in] b Second CONF_ITEM to compare.
303  * @return
304  * - >0 if a > b.
305  * - <0 if a < b.
306  * - 0 if a == b.
307  */
308 static inline int cf_ident2_cmp(void const *a, void const *b)
309 {
310  CONF_ITEM const *ci = a;
311 
312  switch (ci->type) {
313  case CONF_ITEM_PAIR:
314  return 0;
315 
316  case CONF_ITEM_SECTION:
317  {
318  CONF_SECTION const *one = a;
319  CONF_SECTION const *two = b;
320 
321  if (!two->name2 && one->name2) return +1;
322  if (two->name2 && !one->name2) return -1;
323  if (!two->name2 && !one->name2) return 0;
324 
325  return strcmp(one->name2, two->name2);
326  }
327 
328  case CONF_ITEM_DATA:
329  {
330  CONF_DATA const *one = a;
331  CONF_DATA const *two = b;
332 
333  if (!two->name && one->name) return +1;
334  if (two->name && !one->name) return -1;
335  if (!two->name && !one->name) return 0;
336 
337  return strcmp(one->name, two->name);
338  }
339 
340  default:
341  fr_assert_fail(NULL);
342  return 0;
343  }
344 }

346 /** Compare the first and second identifiers of a child
347  *
348  * For CONF_ITEM_SECTION this is 'name2'.
349  * For CONF_ITEM_DATA this is 'name'.
350  *
351  * @param[in] a First CONF_ITEM to compare.
352  * @param[in] b Second CONF_ITEM to compare.
353  * @return
354  * - >0 if a > b.
355  * - <0 if a < b.
356  * - 0 if a == b.
357  */
358 static int _cf_ident2_cmp(void const *a, void const *b)
359 {
360  int ret;
361 
362  ret = _cf_ident1_cmp(a, b);
363  if (ret != 0) return ret;
364 
365  return cf_ident2_cmp(a, b);
366 }

369  *
370  * @param[in] parent to add child to.
371  * @param[in] child to add.
372  */
373 void _cf_item_add(CONF_ITEM *parent, CONF_ITEM *child)
374 {
375  fr_cursor_t to_merge;
376  CONF_ITEM *ci;
377 
378  fr_assert(parent != child);
379 
380  if (!parent || !child) return;
381 
382  /*
383  * New child, add child trees.
384  */
385  if (!parent->ident1) parent->ident1 = rbtree_alloc(parent, _cf_ident1_cmp, NULL, RBTREE_FLAG_NONE);
386  if (!parent->ident2) parent->ident2 = rbtree_alloc(parent, _cf_ident2_cmp, NULL, RBTREE_FLAG_NONE);
387 
388  fr_cursor_init(&to_merge, &child);
389 
390  for (ci = fr_cursor_head(&to_merge);
391  ci;
392  ci = fr_cursor_next(&to_merge)) {
393  rbtree_insert(parent->ident1, ci);
394  rbtree_insert(parent->ident2, ci); /* NULL ident2 is still a value */
395  fr_cursor_append(&parent->cursor, ci); /* Append to the list of children */
396  }
397 }

399 /** Remove item from parent and fixup trees
400  *
401  * @param[in] parent to remove child from.
402  * @param[in] child to remove.
403  * @return
404  * - The item removed.
405  * - NULL if the item wasn't set.
406  */
407 CONF_ITEM *cf_remove(CONF_ITEM *parent, CONF_ITEM *child)
408 {
409  CONF_ITEM *found;
410  bool in_ident1, in_ident2;
411 
412  if (!parent || !parent->child) return NULL;
413  if (parent != child->parent) return NULL;
414 
415  for (found = fr_cursor_head(&parent->cursor);
416  found && (child != found);
417  found = fr_cursor_next(&parent->cursor));
418 
419  if (!found) return NULL;
420 
421  /*
422  * Fixup the linked list
423  */
424  found = fr_cursor_remove(&parent->cursor);
425  if (!fr_cond_assert(found == child)) return NULL;
426 
427  in_ident1 = (rbtree_finddata(parent->ident1, child) == child);
428  if (in_ident1 && (!rbtree_deletebydata(parent->ident1, child))) {
429  fr_assert(0);
430  return NULL;
431  }
432 
433  in_ident2 = (rbtree_finddata(parent->ident2, child) == child);
434  if (in_ident2 && (!rbtree_deletebydata(parent->ident2, child))) {
435  fr_assert(0);
436  return NULL;
437  }
438 
439  /*
440  * Look for twins
441  */
442  for (found = fr_cursor_head(&parent->cursor);
443  found && (in_ident1 || in_ident2);
444  found = fr_cursor_next(&parent->cursor)) {
445  if (in_ident1 && (_cf_ident1_cmp(found, child) == 0)) {
446  rbtree_insert(parent->ident1, child);
447  in_ident1 = false;
448  }
449 
450  if (in_ident2 && (_cf_ident2_cmp(found, child) == 0)) {
451  rbtree_insert(parent->ident2, child);
452  in_ident2 = false;
453  }
454  }
455 
456  return child;
457 }

739

745  parent->item.lineno,
746  parent,
747  buffer, sizeof(buffer), name2, -1, NULL);
748 
749  if (!name2) {

751  return NULL;
752  }
753  }
754  }
755

763  if (lineno) cf_lineno_set(cs, lineno);

766  if (name2) {

769  }
770  talloc_set_destructor(cs, _cf_section_free);
771 
772  if (parent) {
773  CONF_DATA const *cd;
774  CONF_PARSER *rule;
775 
776  cs->depth = parent->depth + 1;
777  cf_item_add(parent, &(cs->item));
778

780  if (cd) {
781  rule = cf_data_value(cd);
782 
783  /*
784  * The parent has an ON_READ rule. Check
785  * if that rule has a child which matches
786  * this section.
787  */
788  if ((FR_BASE_TYPE(rule->type) == FR_TYPE_SUBSECTION) &&
789  ((rule->type & FR_TYPE_ON_READ) != 0) &&
790  rule->subcs) {

792 
793  for (rule_p = rule->subcs; rule_p->name; rule_p++) {
794  if ((FR_BASE_TYPE(rule_p->type) == FR_TYPE_SUBSECTION) &&
795  ((rule_p->type & FR_TYPE_ON_READ) != 0) &&
796  (strcmp(rule_p->name, name1) == 0)) {

798  cd->item.filename, cd->item.lineno) < 0) {
799  error:
800  talloc_free(cs);
801  return NULL;
802  }
803 
804  if (rule_p->func(ctx, NULL, NULL,
805  cf_section_to_item(cs), rule_p) < 0) goto error;
806  return cs;
807  }
808  }
809  }
810  }
811

813  * Call any ON_READ callback. And push this rule
814  * to the child section, so that the child can
815  * pick it up.
816  */
817  cd = cf_data_find(CF_TO_ITEM(parent), CONF_PARSER, name1);
818  if (cd) {
819  rule = cf_data_value(cd);
820  if (rule->func &&
821  (FR_BASE_TYPE(rule->type) == FR_TYPE_SUBSECTION) &&
822  ((rule->type & FR_TYPE_ON_READ) != 0)) {
823  if (cf_section_rules_push(cs, rule) < 0) goto error;
824  if (rule->func(ctx, NULL, NULL, cf_section_to_item(cs), rule) < 0) goto error;
825  }
826  }
827  }
828 
829  return cs;
