218 /** Decode the packet, and set the request->process function
219  *
220  */
221 static int mod_decode(void const *instance, request_t *request, uint8_t *const data, size_t data_len)
222 {
223  proto_detail_t const *inst = talloc_get_type_abort_const(instance, proto_detail_t);
224  int num, lineno;
225  uint8_t const *p, *end;
226  fr_pair_t *vp;
227  fr_cursor_t cursor;
228  time_t timestamp = 0;
229 
230  RHEXDUMP3(data, data_len, "proto_detail decode packet");
231 
232  request->packet->code = inst->code;
233 
234  /*
235  * Set default addresses
236  */
237  request->packet->socket.fd = -1;
238  request->packet->socket.inet.src_ipaddr.af = AF_INET;
239  request->packet->socket.inet.src_ipaddr.addr.v4.s_addr = htonl(INADDR_NONE);
240  request->packet->socket.inet.dst_ipaddr = request->packet->socket.inet.src_ipaddr;
241 
242  request->reply->socket.inet.src_ipaddr = request->packet->socket.inet.src_ipaddr;
243  request->reply->socket.inet.dst_ipaddr = request->packet->socket.inet.src_ipaddr;
244 
245  end = data + data_len;
246 
247  MPRINT("HEADER %s", data);
248 
249  if (sscanf((char const *) data, "%*s %*s %*d %*d:%*d:%*d %d", &num) != 1) {
250  REDEBUG("Malformed header '%s'", (char const *) data);
251  return -1;
252  }
253 
254  /*
255  * Skip the header
256  */
257  for (p = data; p < end; p++) {
258  if (!*p) break;
259  }
260 
261  lineno = 1;
262  fr_cursor_init(&cursor, &request->request_pairs);
263  fr_cursor_tail(&cursor); /* Ensure we only free what we add on error */
264 
265  /*
266  * Parse each individual line.
267  */
268  while (p < end) {
269  /*
270  * Each record begins with a zero byte. If the
271  * next byte is also zero, that's the end of
272  * record indication.
273  */
274  if ((end - p) < 2) break;
275  if (!p[1]) break;
276 
277  /*
278  * Already checked in the "read" routine. But it
279  * doesn't hurt to re-check it here.
280  */
281  if ((*p != '\0') && (*p != '\t')) {
282  REDEBUG("Malformed line %d", lineno);
283  error:
284  fr_cursor_free_list(&cursor);
285  return -1;
286  }
287 
288  p += 2;
289 
290  MPRINT("LINE :%s", p);
291 
292  /*
293  * Skip this for backwards compatability.
294  */
295  if (strncasecmp((char const *) p, "Request-Authenticator", 21) == 0) goto next;
296 
297  /*
298  * The original time at which we received the
299  * packet. We need this to properly calculate
300  * Acct-Delay-Time.
301  */
302  if (strncasecmp((char const *) p, "Timestamp = ", 12) == 0) {
303  p += 12;
304 
305  timestamp = atoi((char const *) p);
306 
307  vp = fr_pair_afrom_da(request->packet, attr_packet_original_timestamp);
308  if (vp) {
309  vp->vp_date = ((fr_time_t) timestamp) * NSEC;
310  vp->type = VT_DATA;
311  fr_cursor_append(&cursor, vp);
312  }
313  goto next;
314  }
315 
316  /*
317  * This should also have been caught.
318  */
319  if (strncasecmp((char const *) p, "Donestamp", 9) == 0) {
320  goto next;
321  }
322 
323  /*
324  * The parsing function appends the created VPs
325  * to the input list, so we need to set 'vp =
326  * NULL'. We don't want to have multiple cursor
327  * functions walking over the list.
328  */
329  vp = NULL;
330  if ((fr_pair_list_afrom_str(request->packet, request->dict, (char const *) p, &vp) > 0) && vp) {
331  fr_cursor_append(&cursor, vp);
332  } else {
333  RWDEBUG("Ignoring line %d - :%s", lineno, p);
334  }
335 
336  /*
337  * Set the original src/dst ip/port
338  */
339  if (vp) {
340  if ((vp->da == attr_packet_src_ip_address) ||
341  (vp->da == attr_packet_src_ipv6_address)) {
342  request->packet->socket.inet.src_ipaddr = vp->vp_ip;
343  } else if ((vp->da == attr_packet_dst_ip_address) ||
344  (vp->da == attr_packet_dst_ipv6_address)) {
345  request->packet->socket.inet.dst_ipaddr = vp->vp_ip;
346  } else if (vp->da == attr_packet_src_port) {
347  request->packet->socket.inet.src_port = vp->vp_uint16;
348  } else if (vp->da == attr_packet_dst_port) {
349  request->packet->socket.inet.dst_port = vp->vp_uint16;
350  } else if (vp->da == attr_protocol) {
351  request->dict = fr_dict_by_protocol_num(vp->vp_uint32);
352  if (!request->dict) {
353  REDEBUG("Invalid protocol: %pP", vp);
354  goto error;
355  }
356  }
357  }
358 
359  next:
360  lineno++;
361  while ((p < end) && (*p)) p++;
362  }
363 
364  /*
365  * Let the app_io take care of populating additional fields in the request
366  */
367  return inst->app_io->decode(inst->app_io_instance, request, data, data_len);
368 }

393 /** Open listen sockets/connect to external event source
394  *
395  * @param[in] instance Ctx data for this application.

413  talloc_set_destructor(li, fr_io_listen_free);

420  li->app = &proto_detail;
421  li->app_instance = instance;
422  li->server_cs = inst->server_cs;
423 
424  /*

429

431  * Open the file.
432  */

435  talloc_free(li);

437  }
438

449  return -1;
450  }
451 
452  inst->listen = li;

454  }

460  talloc_free(li);
461  return -1;
462  }
463 
464  inst->listen = li; /* Probably won't need it, but doesn't hurt */
465  inst->sc = sc;
466 
467  return 0;

471 /** Instantiate the application
472  *
473  * Instantiate I/O and type submodules.
474  *
475  * @param[in] instance Ctx data for this application.
476  * @param[in] conf Listen section parsed to give us isntance.
477  * @return

481 static int mod_instantiate(void *instance, CONF_SECTION *conf)
482 {
483  proto_detail_t *inst = talloc_get_type_abort(instance, proto_detail_t);
484  CONF_PAIR *cp = NULL;
485 
486  /*
487  * Instantiate the I/O module. But DON'T instantiate the
488  * work submodule. We leave that until later.
489  */
490  if (inst->app_io->instantiate &&
491  (inst->app_io->instantiate(inst->app_io_instance,
492  inst->app_io_conf) < 0)) {
493  cf_log_err(conf, "Instantiation failed for \"%s\"", inst->app_io->name);
494  return -1;
495  }
496 
497  /*
498  * Instantiate the process module.
499  */
500  while ((cp = cf_pair_find_next(conf, cp, "type")) != NULL) {
501  fr_app_worker_t const *app_process;
502 
503 #ifdef __clang_analyzer__
504  DEBUG("Instantiating %s", cf_pair_value(cp));
505 #endif
506 
507  app_process = (fr_app_worker_t const *)inst->type_submodule->module->common;
508  if (app_process->instantiate && (app_process->instantiate(inst->type_submodule->data,
509  inst->type_submodule->conf) < 0)) {
510  cf_log_err(conf, "Instantiation failed for \"%s\"", app_process->name);
511  return -1;
512  }
513 
514  break;
515  }
516 
517  /*
518  * These configuration items are not printed by default,
519  * because normal people shouldn't be touching them.
520  */
521  if (!inst->max_packet_size) inst->max_packet_size = inst->app_io->default_message_size;
522 
523  if (!inst->num_messages) inst->num_messages = 2;
524 
525  FR_INTEGER_BOUND_CHECK("num_messages", inst->num_messages, >=, 2);
526  FR_INTEGER_BOUND_CHECK("num_messages", inst->num_messages, <=, 65535);
527 
528  FR_INTEGER_BOUND_CHECK("max_packet_size", inst->max_packet_size, >=, 1024);
529  FR_INTEGER_BOUND_CHECK("max_packet_size", inst->max_packet_size, <=, 65536);
530 
531  if (!inst->priority) inst->priority = PRIORITY_NORMAL;
532 
533  /*
534  * If the IO is "file" and not the worker, instantiate the worker now.
535  */
536  if (strcmp(inst->io_submodule->module->dl->name, "proto_detail_work") != 0) {
537  if (inst->work_io->instantiate && (inst->work_io->instantiate(inst->work_io_instance,
538  inst->work_io_conf) < 0)) {
539  cf_log_err(inst->work_io_conf, "Instantiation failed for \"%s\"", inst->work_io->name);
540  return -1;
541  }
542  }
543 
544  return 0;
545 }

547 /** Bootstrap the application
548  *
549  * Bootstrap I/O and type submodules.
550  *
551  * @param[in] instance Ctx data for this application.
552  * @param[in] conf Listen section parsed to give us instance.
553  * @return
554  * - 0 on success.
555  * - -1 on failure.
556  */
557 static int mod_bootstrap(void *instance, CONF_SECTION *conf)
558 {
559  proto_detail_t *inst = talloc_get_type_abort(instance, proto_detail_t);
560  CONF_PAIR *cp = NULL;
561 
562  /*
563  * The listener is inside of a virtual server.
564  */
565  inst->server_cs = cf_item_to_section(cf_parent(conf));
566  inst->cs = conf;
567  inst->self = &proto_detail;
568 
569  virtual_server_dict_set(inst->server_cs, inst->dict, false);
570 
571  /*
572  * Bootstrap the process module.
573  */
574  while ((cp = cf_pair_find_next(conf, cp, "type"))) {
575  dl_module_t const *module = talloc_get_type_abort_const(inst->type_submodule->module,
576  dl_module_t);
577  fr_app_worker_t const *app_process = (fr_app_worker_t const *)(module->common);
578 
579  if (app_process->bootstrap && (app_process->bootstrap(inst->type_submodule->data,
580  inst->type_submodule->conf) < 0)) {
581  cf_log_err(conf, "Bootstrap failed for \"%s\"", app_process->name);
582  return -1;
583  }
584  }
585 
586  /*
587  * No IO module, it's an empty listener. That's not
588  * allowed for the detail file reader.
589  */
590  if (!inst->io_submodule) {
591  cf_log_err(conf, "Virtual server for detail files requires a 'transport' configuration");
592  return -1;
593  }
594 
595  /*
596  * Bootstrap the I/O module
597  */
598  inst->app_io = (fr_app_io_t const *) inst->io_submodule->module->common;
599  inst->app_io_instance = inst->io_submodule->data;
