108 static int type_parse(TALLOC_CTX *ctx, void *out, void *parent,
109  CONF_ITEM *ci, UNUSED CONF_PARSER const *rule)
110 {
111  static char const *type_lib_table[FR_VQP_MAX_CODE] = {
112  [FR_PACKET_TYPE_VALUE_JOIN_REQUEST] = "process",
113  [FR_PACKET_TYPE_VALUE_RECONFIRM_REQUEST] = "process",
114  };
115  int code;
116  proto_vmps_t *inst = talloc_get_type_abort(parent, proto_vmps_t);
117 
118  code = fr_app_process_type_parse(ctx, out, ci, attr_packet_type,
119  type_lib_table, NUM_ELEMENTS(type_lib_table), "proto_vmps");
120  if (code < 0) return -1;
121 
122  /*
123  * Set the allowed codes so that the IO functions can
124  * filter them as necessary.
125  */
126  inst->code_allowed[code] = true;
127  return 0;
128 }

141 static int transport_parse(TALLOC_CTX *ctx, void *out, UNUSED void *parent,
142  CONF_ITEM *ci, UNUSED CONF_PARSER const *rule)
143 {
144  char const *name = cf_pair_value(cf_item_to_pair(ci));
145  dl_module_inst_t *parent_inst;
146  proto_vmps_t *inst;
147  CONF_SECTION *listen_cs = cf_item_to_section(cf_parent(ci));
148  CONF_SECTION *transport_cs;
149 
150  transport_cs = cf_section_find(listen_cs, name, NULL);
151 
152  /*
153  * Allocate an empty section if one doesn't exist
154  * this is so defaults get parsed.
155  */
156  if (!transport_cs) transport_cs = cf_section_alloc(listen_cs, listen_cs, name, NULL);
157 
158  parent_inst = cf_data_value(cf_data_find(listen_cs, dl_module_inst_t, "proto_vmps"));
159  fr_assert(parent_inst);
160 
161  /*
162  * Set the allowed codes so that we can compile them as
163  * necessary.
164  */
165  inst = talloc_get_type_abort(parent_inst->data, proto_vmps_t);
166  inst->io.transport = name;
167 
168  return dl_module_instance(ctx, out, transport_cs, parent_inst, name, DL_MODULE_TYPE_SUBMODULE);
169 }

174 static int mod_decode(void const *instance, request_t *request, uint8_t *const data, size_t data_len)
175 {
176  proto_vmps_t const *inst = talloc_get_type_abort_const(instance, proto_vmps_t);
177  fr_io_track_t const *track = talloc_get_type_abort_const(request->async->packet_ctx, fr_io_track_t);
178  fr_io_address_t const *address = track->address;
179  RADCLIENT const *client;
180  fr_radius_packet_t *packet = request->packet;
181  fr_cursor_t cursor;
182 
183  fr_assert(data[0] < FR_RADIUS_MAX_PACKET_CODE);
184 
185  RHEXDUMP3(data, data_len, "proto_vmps decode packet");
186 
187  /*
188  * Set the request dictionary so that we can do
189  * generic->protocol attribute conversions as
190  * the request runs through the server.
191  */
192  request->dict = dict_vmps;
193 
194  client = address->radclient;
195 
196  /*
197  * Hacks for now until we have a lower-level decode routine.
198  */
199  request->packet->code = data[0];
200  request->packet->id = data[1];
201  request->reply->id = data[1];
202  memcpy(request->packet->vector, data + 4, sizeof(request->packet->vector));
203 
204  request->packet->data = talloc_memdup(request->packet, data, data_len);
205  request->packet->data_len = data_len;
206 
207  /*
208  * Note that we don't set a limit on max_attributes here.
209  * That MUST be set and checked in the underlying
210  * transport, via a call to fr_vmps_ok().
211  */
212  fr_cursor_init(&cursor, &packet->vps);
213  if (fr_vmps_decode(packet, packet->data, packet->data_len, &cursor, &packet->code) < 0) {
214  RPEDEBUG("Failed decoding packet");
215  return -1;
216  }
217 
218  /*
219  * Set the rest of the fields.
220  */
221  memcpy(&request->client, &client, sizeof(client)); /* const issues */
222 
223  request->packet->socket = address->socket;
224  fr_socket_addr_swap(&request->reply->socket, &address->socket);
225 
226  request->config = main_config;
227  REQUEST_VERIFY(request);
228 
229  if (!inst->io.app_io->decode) return 0;
230 
231  /*
232  * Let the app_io do anything it needs to do.
233  */
234  return inst->io.app_io->decode(inst->io.app_io_instance, request, data, data_len);
235 }

237 static ssize_t mod_encode(void const *instance, request_t *request, uint8_t *buffer, size_t buffer_len)
238 {
239  proto_vmps_t const *inst = talloc_get_type_abort_const(instance, proto_vmps_t);
240  fr_io_track_t const *track = talloc_get_type_abort_const(request->async->packet_ctx, fr_io_track_t);
241  fr_io_address_t const *address = track->address;
242  ssize_t data_len;
243  RADCLIENT const *client;
244  fr_cursor_t cursor;
245 
246  /*
247  * The packet timed out. Tell the network side that the packet is dead.
248  */
249  if (buffer_len == 1) {
250  *buffer = true;
251  return 1;
252  }
253 
254  /*
255  * "Do not respond"
256  */
257  if ((request->reply->code == FR_CODE_DO_NOT_RESPOND) ||
258  (request->reply->code == 0) || (request->reply->code >= FR_RADIUS_MAX_PACKET_CODE)) {
259  *buffer = false;
260  return 1;
261  }
262 
263  client = address->radclient;
264  fr_assert(client);
265 
266  /*
267  * Dynamic client stuff
268  */
269  if (client->dynamic && !client->active) {
270  RADCLIENT *new_client;
271 
272  fr_assert(buffer_len >= sizeof(client));
273 
274  /*
275  * Allocate the client. If that fails, send back a NAK.
276  *
277  * @todo - deal with NUMA zones? Or just deal with this
278  * client being in different memory.
279  *
280  * Maybe we should create a CONF_SECTION from the client,
281  * and pass *that* back to mod_write(), which can then
282  * parse it to create the actual client....
283  */
284  new_client = client_afrom_request(NULL, request);
285  if (!new_client) {
286  PERROR("Failed creating new client");
287  buffer[0] = true;
288  return 1;
289  }
290 
291  memcpy(buffer, &new_client, sizeof(new_client));
292  return sizeof(new_client);
293  }
294 
295  /*
296  * If the app_io encodes the packet, then we don't need
297  * to do that.
298  */
299  if (inst->io.app_io->encode) {
300  data_len = inst->io.app_io->encode(inst->io.app_io_instance, request, buffer, buffer_len);
301  if (data_len > 0) return data_len;
302  }
303 
304  /*
305  * Overwrite the src ip address on the outbound packet
306  * with the one specified by the client. This is useful
307  * to work around broken DSR implementations and other
308  * routing issues.
309  */
310  if (client->src_ipaddr.af != AF_UNSPEC) {
311  request->reply->socket.inet.src_ipaddr = client->src_ipaddr;
312  }
313 
314  fr_cursor_talloc_iter_init(&cursor, &request->reply_pairs, fr_proto_next_encodable, dict_vmps, fr_pair_t);
315 
316  data_len = fr_vmps_encode(&FR_DBUFF_TMP(buffer, buffer_len), request->packet->data,
317  request->reply->code, request->reply->id, &cursor);
318  if (data_len < 0) {
319  RPEDEBUG("Failed encoding VMPS reply");
320  return -1;
321  }
322 
323  RHEXDUMP3(buffer, data_len, "proto_vmps encode packet");
324 
325  return data_len;
326 }

328 static void mod_entry_point_set(void const *instance, request_t *request)
329 {
330  proto_vmps_t const *inst = talloc_get_type_abort_const(instance, proto_vmps_t);
331  dl_module_inst_t *type_submodule;
332  fr_io_track_t *track = request->async->packet_ctx;
333 
334  fr_assert(request->packet->code != 0);
335  fr_assert(request->packet->code <= FR_VQP_MAX_CODE);
336 
337  request->server_cs = inst->io.server_cs;
338 
339  /*
340  * 'track' can be NULL when there's no network listener.
341  */
342  if (inst->io.app_io && (track->dynamic == request->async->recv_time)) {
343  fr_app_worker_t const *app_process;
344 
345  app_process = (fr_app_worker_t const *) inst->io.dynamic_submodule->module->common;
346 
347  request->async->process = app_process->entry_point;
348  request->async->process_inst = inst->io.dynamic_submodule;
349  track->dynamic = 0;
350  return;
351  }
352 
353  type_submodule = inst->type_submodule_by_code[request->packet->code];
354  if (!type_submodule) {
355  REDEBUG("No module available to handle packet code %i", request->packet->code);
356  return;
357  }
358 
359  request->async->process = ((fr_app_worker_t const *)type_submodule->module->common)->entry_point;
360  request->async->process_inst = type_submodule->data;
361 }

365 {
366  proto_vmps_t const *inst = talloc_get_type_abort_const(instance, proto_vmps_t);

368  /*

370  */
371  if (!inst->priorities[buffer[1]]) return 0;
372 
373  /*
374  * @todo - if we cared, we could also return -1 for "this
375  * is a bad packet". But that's really only for
376  * mod_inject, as we assume that app_io->read() always
377  * returns good packets.
378  */
379 
380  /*
381  * Return the configured priority.
382  */
383  return inst->priorities[buffer[1]];
384 }

424  NUM_ELEMENTS(inst->type_submodule_by_code),

455 /** Bootstrap the application
456  *
457  * Bootstrap I/O and type submodules.
458  *
459  * @param[in] instance Ctx data for this application.

465 static int mod_bootstrap(void *instance, CONF_SECTION *conf)
466 {
467  proto_vmps_t *inst = talloc_get_type_abort(instance, proto_vmps_t);
468 
469  /*
470  * Ensure that the server CONF_SECTION is always set.
471  */
472  inst->io.server_cs = cf_item_to_section(cf_parent(conf));
473 
474  fr_assert(dict_vmps != NULL);
475  fr_assert(attr_packet_type != NULL);
476 
477  /*
478  * Bootstrap the app_process modules.
479  */
480  if (fr_app_process_bootstrap(inst->io.server_cs, inst->type_submodule, conf) < 0) return -1;
481 
482  /*
483  * No IO module, it's an empty listener.
484  */
485  if (!inst->io.submodule) return 0;
486 
487  /*
488  * These timers are usually protocol specific.
489  */
490  FR_TIME_DELTA_BOUND_CHECK("idle_timeout", inst->io.idle_timeout, >=, fr_time_delta_from_sec(1));
491  FR_TIME_DELTA_BOUND_CHECK("idle_timeout", inst->io.idle_timeout, <=, fr_time_delta_from_sec(600));
492 
493  FR_TIME_DELTA_BOUND_CHECK("nak_lifetime", inst->io.nak_lifetime, >=, fr_time_delta_from_sec(1));
494  FR_TIME_DELTA_BOUND_CHECK("nak_lifetime", inst->io.nak_lifetime, <=, fr_time_delta_from_sec(600));
495 
496  /*
497  * Tell the master handler about the main protocol instance.
498  */
499  inst->io.app = &proto_vmps;
500  inst->io.app_instance = inst;
501 
502  /*
503  * We will need this for dynamic clients and connected sockets.
504  */
505  inst->io.dl_inst = dl_module_instance_by_data(inst);
506  fr_assert(inst != NULL);
507 
508  /*
509  * Bootstrap the master IO handler.
510  */
511  return fr_master_app_io.bootstrap(&inst->io, conf);
512 }
