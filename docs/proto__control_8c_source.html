70 /** Wrapper around dl_instance
71  *
72  * @param[in] ctx to allocate data in (instance of proto_control).
73  * @param[out] out Where to write a dl_module_inst_t containing the module handle and instance.
74  * @param[in] parent Base structure address.
75  * @param[in] ci #CONF_PAIR specifying the name of the type module.
76  * @param[in] rule unused.
77  * @return
78  * - 0 on success.
79  * - -1 on failure.
80  */
81 static int transport_parse(TALLOC_CTX *ctx, void *out, UNUSED void *parent, CONF_ITEM *ci, UNUSED CONF_PARSER const *rule)
82 {
83  char const *name = cf_pair_value(cf_item_to_pair(ci));
84  dl_module_inst_t *parent_inst;
85  proto_control_t *inst;
86  CONF_SECTION *listen_cs = cf_item_to_section(cf_parent(ci));
87  CONF_SECTION *transport_cs;
88 
89  transport_cs = cf_section_find(listen_cs, name, NULL);
90 
91  /*
92  * Allocate an empty section if one doesn't exist
93  * this is so defaults get parsed.
94  */
95  if (!transport_cs) transport_cs = cf_section_alloc(listen_cs, listen_cs, name, NULL);
96 
97  parent_inst = cf_data_value(cf_data_find(listen_cs, dl_module_inst_t, "proto_control"));
98  fr_assert(parent_inst);
99 
100  /*
101  * Set the allowed codes so that we can compile them as
102  * necessary.
103  */
104  inst = talloc_get_type_abort(parent_inst->data, proto_control_t);
105  inst->io.transport = name;
106 
107  return dl_module_instance(ctx, out, transport_cs, parent_inst, name, DL_MODULE_TYPE_SUBMODULE);
108 }

131 /** Instantiate the application
132  *
133  * Instantiate I/O and type submodules.
134  *
135  * @param[in] instance Ctx data for this application.
136  * @param[in] conf Listen section parsed to give us instance.
137  * @return
138  * - 0 on success.

141 static int mod_instantiate(void *instance, CONF_SECTION *conf)
142 {
143  proto_control_t *inst = talloc_get_type_abort(instance, proto_control_t);
144 
145  fr_assert(inst->io.submodule != NULL);
146 
147  /*
148  * These configuration items are not printed by default,
149  * because normal people shouldn't be touching them.
150  */
151  if (!inst->max_packet_size && inst->io.app_io) inst->max_packet_size = inst->io.app_io->default_message_size;

153  if (!inst->num_messages) inst->num_messages = 256;

156  FR_INTEGER_BOUND_CHECK("num_messages", inst->num_messages, <=, 65535);

159  FR_INTEGER_BOUND_CHECK("max_packet_size", inst->max_packet_size, <=, 65535);
160 
161  /*
162  * Instantiate the master io submodule
163  */
164  return fr_master_app_io.instantiate(&inst->io, conf);
165 }

168 /** Bootstrap the application
169  *
170  * Bootstrap I/O and type submodules.
171  *
172  * @param[in] instance Ctx data for this application.
173  * @param[in] conf Listen section parsed to give us instance.
174  * @return
175  * - 0 on success.
176  * - -1 on failure.
177  */
178 static int mod_bootstrap(void *instance, CONF_SECTION *conf)
179 {
180  proto_control_t *inst = talloc_get_type_abort(instance, proto_control_t);
181 
182  /*
183  * Ensure that the server CONF_SECTION is always set.
184  */
185  inst->io.server_cs = cf_item_to_section(cf_parent(conf));
186 
187  /*
188  * No IO module, it's an empty listener.
189  */
190  if (!inst->io.submodule) {
191  cf_log_err(conf, "The control server MUST have a 'listener' section.");
192  return -1;
193  }
194 
195  /*
196  * These timers are usually protocol specific.
197  */
198  FR_TIME_DELTA_BOUND_CHECK("idle_timeout", inst->io.idle_timeout, >=, fr_time_delta_from_sec(1));
199  FR_TIME_DELTA_BOUND_CHECK("idle_timeout", inst->io.idle_timeout, <=, fr_time_delta_from_sec(600));
200 
201  FR_TIME_DELTA_BOUND_CHECK("nak_lifetime", inst->io.nak_lifetime, >=, fr_time_delta_from_sec(1));
202  FR_TIME_DELTA_BOUND_CHECK("nak_lifetime", inst->io.nak_lifetime, <=, fr_time_delta_from_sec(600));
203 
204  /*
205  * Tell the master handler about the main protocol instance.
206  */
207  inst->io.app = &proto_control;
208  inst->io.app_instance = inst;
209 
210  /*
211  * We will need this for dynamic clients and connected sockets.
212  */
213  inst->io.dl_inst = dl_module_instance_by_data(inst);
214  fr_assert(inst != NULL);
215 
216  /*
217  * Bootstrap the master IO handler.
218  */
219  return fr_master_app_io.bootstrap(&inst->io, conf);
220 }
