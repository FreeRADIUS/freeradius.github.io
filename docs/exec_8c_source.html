57 static void fr_exec_pair_to_env(request_t *request, fr_pair_t *input_pairs, char **envp, size_t envlen, bool shell_escape)
58 {
59  char *p;
60  size_t i;
61  fr_cursor_t cursor;
62  fr_dict_attr_t const *da;
63  fr_pair_t *vp;
64  char buffer[1024];
65 
66  /*
67  * Set up the environment variables in the
68  * parent, so we don't call libc functions that
69  * hold mutexes. They might be locked when we fork,
70  * and will remain locked in the child.
71  */
72  for (vp = fr_cursor_init(&cursor, &input_pairs), i = 0;
73  vp && (i < envlen - 1);
74  vp = fr_cursor_next(&cursor)) {
75  size_t n;
76 
77  /*
78  * Hmm... maybe we shouldn't pass the
79  * user's password in an environment
80  * variable...
81  */
82  snprintf(buffer, sizeof(buffer), "%s=", vp->da->name);
83  if (shell_escape) {
84  for (p = buffer; *p != '='; p++) {
85  if (*p == '-') {
86  *p = '_';
87  } else if (isalpha((int) *p)) {
88  *p = toupper(*p);
89  }
90  }
91  }
92 
93  n = strlen(buffer);
94  fr_pair_print_value_quoted(&FR_SBUFF_OUT(buffer + n, sizeof(buffer) - n), vp,
95  shell_escape ? T_DOUBLE_QUOTED_STRING : T_BARE_WORD);
96 
97  DEBUG3("export %s", buffer);
98  envp[i++] = talloc_typed_strdup(envp, buffer);
99  }
100 
101  if (request) {
102  da = fr_dict_attr_child_by_num(fr_dict_root(fr_dict_internal()), FR_EXEC_EXPORT);
103  if (da) {
104  for (vp = fr_cursor_iter_by_da_init(&cursor, &request->control_pairs, da);
105  vp && (i < (envlen - 1));
106  vp = fr_cursor_next(&cursor)) {
107  DEBUG3("export %pV", &vp->data);
108  memcpy(&envp[i++], &vp->vp_strvalue, sizeof(*envp));
109  }
110 
111  /*
112  * NULL terminate for execve
113  */
114  envp[i] = NULL;
115  }
116  }
117 }

121  *
122  * We try to be fail-safe here. So if ANYTHING
123  * goes wrong, we exit with status 1.

125 static NEVER_RETURNS void fr_exec_child(request_t *request, char **argv, char **envp,
126  bool exec_wait, int *input_fd, int *output_fd,
127  int to_child[static 2], int from_child[static 2])
128 {
129  int devnull;
130 
131  /*
132  * Open STDIN to /dev/null
133  */
134  devnull = open("/dev/null", O_RDWR);
135  if (devnull < 0) {
136  ERROR("Failed opening /dev/null: %s\n", fr_syserror(errno));
137 
138  /*
139  * Where the status code is interpreted as a module rcode
140  * one is subtracted from it, to allow 0 to equal success
141  *
142  * 2 is RLM_MODULE_FAIL + 1
143  */
144  exit(2);
145  }
146 
147  /*
148  * Only massage the pipe handles if the parent
149  * has created them.
150  */
151  if (exec_wait) {
152  if (input_fd) {
153  close(to_child[1]);
154  dup2(to_child[0], STDIN_FILENO);
155  } else {
156  dup2(devnull, STDIN_FILENO);
157  }
158 
159  if (output_fd) {
160  close(from_child[0]);
161  dup2(from_child[1], STDOUT_FILENO);
162  } else {
163  dup2(devnull, STDOUT_FILENO);
164  }
165 
166  } else { /* no pipe, STDOUT should be /dev/null */
167  dup2(devnull, STDIN_FILENO);
168  dup2(devnull, STDOUT_FILENO);
169  }
170 
171  /*
172  * If we're not debugging, then we can't do
173  * anything with the error messages, so we throw
174  * them away.
175  *
176  * If we are debugging, then we want the error
177  * messages to go to the STDERR of the server.
178  */
179  if (!request || !RDEBUG_ENABLED) dup2(devnull, STDERR_FILENO);
180  close(devnull);
181 
182  /*
183  * The server may have MANY FD's open. We don't
184  * want to leave dangling FD's for the child process
185  * to play funky games with, so we close them.
186  */
187  closefrom(3);
188 
189  /*
190  * I swear the signature for execve is wrong and should
191  * take 'char const * const argv[]'.
192  *
193  * Note: execve(), unlike system(), treats all the space
194  * delimited arguments as literals, so there's no need
195  * to perform additional escaping.
196  */
197  execve(argv[0], argv, envp);
198  printf("Failed to execute \"%s\": %s", argv[0], fr_syserror(errno)); /* fork output will be captured */
199 
200  /*
201  * Where the status code is interpreted as a module rcode
202  * one is subtracted from it, to allow 0 to equal success
203  *
204  * 2 is RLM_MODULE_FAIL + 1
205  */
206  exit(2);
207 }

209 /** Start a process
210  *
211  * @param cmd Command to execute. This is parsed into argv[] parts, then each individual argv
212  * part is xlat'ed.
213  * @param request Current reuqest
214  * @param exec_wait set to true to read from or write to child.
215  * @param[in,out] input_fd pointer to int, receives the stdin file descriptor. Set to NULL
216  * and the child will have /dev/null on stdin.
217  * @param[in,out] output_fd pinter to int, receives the stdout file descriptor. Set to NULL
218  * and child will have /dev/null on stdout.
219  * @param input_pairs list of value pairs - these will be put into the environment variables
220  * of the child.
221  * @param shell_escape values before passing them as arguments.
222  * @return
223  * - PID of the child process.
224  * - -1 on failure.
225  */
226 pid_t radius_start_program(char const *cmd, request_t *request, bool exec_wait,
227  int *input_fd, int *output_fd,
228  fr_pair_t *input_pairs, bool shell_escape)
229 {
230  int to_child[2] = {-1, -1};
231  int from_child[2] = {-1, -1};
232  pid_t pid;
233  int argc;
234  int i;
235  char const **argv_p;
236  char *argv[MAX_ARGV], **argv_start = argv;
237  char argv_buf[4096];
238 #define MAX_ENVP 1024
239  char **envp;
240 
241  /*
242  * Stupid array decomposition...
243  *
244  * If we do memcpy(&argv_p, &argv, sizeof(argv_p)) src ends up being a char **
245  * pointing to the value of the first element.
246  */
247  memcpy(&argv_p, &argv_start, sizeof(argv_p));
248  argc = rad_expand_xlat(request, cmd, MAX_ARGV, argv_p, true, sizeof(argv_buf), argv_buf);
249  if (argc <= 0) {
250  ROPTIONAL(RPEDEBUG, PERROR, "Invalid command '%s'", cmd);
251  return -1;
252  }
253 
254  if (DEBUG_ENABLED3) {
255  for (i = 0; i < argc; i++) DEBUG3("arg[%d] %s", i, argv[i]);
256  }
257 
258  /*
259  * Open a pipe for child/parent communication, if necessary.
260  */
261  if (exec_wait) {
262  if (input_fd) {
263  if (pipe(to_child) != 0) {
264  ERROR("Couldn't open pipe to child: %s", fr_syserror(errno));
265  return -1;
266  }
267  }
268  if (output_fd) {
269  if (pipe(from_child) != 0) {
270  ERROR("Couldn't open pipe from child: %s", fr_syserror(errno));
271  /* safe because these either need closing or are == -1 */
272  close(to_child[0]);
273  close(to_child[1]);
274  return -1;
275  }
276  }
277  }
278 
279  MEM(envp = talloc_zero_array(request, char *, MAX_ENVP));
280  envp[0] = NULL;
281  if (input_pairs) fr_exec_pair_to_env(request, input_pairs, envp, MAX_ENVP, shell_escape);
282 
283  pid = fork();
284  if (pid == 0) {
285  fr_exec_child(request, argv, envp, exec_wait, input_fd, output_fd, to_child, from_child);
286  }
287 
288  /*
289  * Free child environment variables
290  */
291  talloc_free(envp);
292 
293  /*
294  * Parent process.
295  */
296  if (pid < 0) {
297  ERROR("Couldn't fork %s: %s", argv[0], fr_syserror(errno));
298  if (exec_wait) {
299  /* safe because these either need closing or are == -1 */
300  close(to_child[0]);
301  close(to_child[1]);
302  close(from_child[0]);
303  close(from_child[1]);
304  }
305  return -1;
306  }
307 
308  /*
309  * We're done. Do any necessary cleanups.
310  */
311  if (exec_wait) {
312  /*
313  * Close the ends of the pipe(s) the child is using
314  * return the ends of the pipe(s) our caller wants
315  *
316  */
317  if (input_fd) {
318  *input_fd = to_child[1];
319  close(to_child[0]);
320  }
321  if (output_fd) {
322  *output_fd = from_child[0];
323  close(from_child[1]);
324  }
325 
326  } else {
327  (void) fr_event_pid_wait(request->el, request->el, NULL, pid, NULL, NULL);
328  }
329 
330  return pid;
331 }

334 /** Read from the child process.
335  *
336  * @param fd file descriptor to read from.
337  * @param pid pid of child, will be reaped if it dies.
338  * @param timeout amount of time to wait, in seconds.
339  * @param answer buffer to write into.
340  * @param left length of buffer.
341  * @return
342  * - -1 on failure.
343  * - Length of output.
344  */
345 int radius_readfrom_program(int fd, pid_t pid, fr_time_delta_t timeout,
346  char *answer, int left)
347 {
348  int done = 0;
349  int status;
350  fr_time_t start;
351 
352  fr_nonblock(fd);
353 
354  /*
355  * Minimum timeout period is one section
356  */
357  if (timeout < NSEC) timeout = fr_time_delta_from_sec(1);
358 
359  /*
360  * Read from the pipe until we doesn't get any more or
361  * until the message is full.
362  */
363  start = fr_time();
364  while (1) {
365  int rcode;
366  fd_set fds;
367  fr_time_delta_t elapsed;
368 
369  FD_ZERO(&fds);
370  FD_SET(fd, &fds);
371 
372  elapsed = fr_time() - start;
373  if (elapsed >= timeout) goto too_long;
374 
375  rcode = select(fd + 1, &fds, NULL, NULL, &fr_time_delta_to_timeval(timeout - elapsed));
376  if (rcode == 0) {
377  too_long:
378  DEBUG("Child PID %u is taking too much time: forcing failure and killing child.", pid);
379  kill(pid, SIGTERM);
380  close(fd); /* should give SIGPIPE to child, too */
381 
382  /*
383  * Clean up the child entry.
384  */
385  waitpid(pid, &status, 0);
386  return -1;
387  }
388  if (rcode < 0) {
389  if (errno == EINTR) continue;
390  break;
391  }
392 
393 #ifdef O_NONBLOCK
394  /*
395  * Read as many bytes as possible. The kernel
396  * will return the number of bytes available.
397  */
398  status = read(fd, answer + done, left);
399 #else
400  /*
401  * There's at least 1 byte ready: read it.
402  * This is a terrible hack for non-blocking IO.
403  */
404  status = read(fd, answer + done, 1);
405 #endif
406 
407  /*
408  * Nothing more to read: stop.
409  */
410  if (status == 0) {
411  break;
412  }
413 
414  /*
415  * Error: See if we have to continue.
416  */
417  if (status < 0) {
418  /*
419  * We were interrupted: continue reading.
420  */
421  if (errno == EINTR) {
422  continue;
423  }
424 
425  /*
426  * There was another error. Most likely
427  * The child process has finished, and
428  * exited.
429  */
430  break;
431  }
432 
433  done += status;
434  left -= status;
435  if (left <= 0) break;
436  }
437 
438  /* Strip trailing new lines */
439  while ((done > 0) && (answer[done - 1] == '\n')) {
440  answer[--done] = '\0';
441  }
442 
443  return done;
444 }

446 /** Execute a program.
447  *
448  * @param[in,out] ctx to allocate new fr_pair_t (s) in.
449  * @param[out] out buffer to append plaintext (non valuepair) output.
450  * @param[in] outlen length of out buffer.
451  * @param[out] output_pairs list of value pairs - Data on child's stdout will be parsed and
452  * added into this list of value pairs.
453  * @param[in] request Current request (may be NULL).
454  * @param[in] cmd Command to execute. This is parsed into argv[] parts, then each individual argv
455  * part is xlat'ed.
456  * @param[in] input_pairs list of value pairs - these will be available in the environment of the
457  * child.
458  * @param[in] exec_wait set to 1 if you want to read from or write to child.
459  * @param[in] shell_escape values before passing them as arguments.
460  * @param[in] timeout amount of time to wait, in seconds.
461  * @return
462  * - 0 if exec_wait==0.
463  * - exit code if exec_wait!=0.
464  * - -1 on failure.
465  */
466 int radius_exec_program(TALLOC_CTX *ctx, char *out, size_t outlen, fr_pair_t **output_pairs,
467  request_t *request, char const *cmd, fr_pair_t *input_pairs,
468  bool exec_wait, bool shell_escape, fr_time_delta_t timeout)
469 
470 {
471  pid_t pid;
472  int from_child;
473  char *p;
474  pid_t child_pid;
475  int comma = 0;
476  int status, ret = 0;
477  ssize_t len;
478  char answer[4096];
479 
480  RDEBUG2("Executing: %s", cmd);
481 
482  if (out) *out = '\0';
483 
484  pid = radius_start_program(cmd, request, exec_wait, NULL, &from_child, input_pairs, shell_escape);
485  if (pid < 0) {
486  return -1;
487  }
488 
489  if (!exec_wait) {
490  return 0;
491  }
492 
493  len = radius_readfrom_program(from_child, pid, timeout, answer, sizeof(answer));
494  if (len < 0) {
495  /*
496  * Failure - radius_readfrom_program will
497  * have called close(from_child) for us
498  */
499  RERROR("Failed to read from child output");
500  return -1;
501 
502  }
503  answer[len] = '\0';
504 
505  /*
506  * Make sure that the writer can't block while writing to
507  * a pipe that no one is reading from anymore.
508  */
509  close(from_child);
510 
511  if (len == 0) {
512  goto wait;
513  }
514 
515  /*
516  * Parse the output, if any.
517  */
518  if (output_pairs) {
519  fr_pair_t *vps = NULL;
520 
521  /*
522  * HACK: Replace '\n' with ',' so that
523  * fr_pair_list_afrom_str() can parse the buffer in
524  * one go (the proper way would be to
525  * fix fr_pair_list_afrom_str(), but oh well).
526  */
527  for (p = answer; *p; p++) {
528  if (*p == '\n') {
529  *p = comma ? ' ' : ',';
530  p++;
531  comma = 0;
532  }
533  if (*p == ',') {
534  comma++;
535  }
536  }
537 
538  /*
539  * Replace any trailing comma by a NUL.
540  */
541  if (answer[len - 1] == ',') {
542  answer[--len] = '\0';
543  }
544 
545  if (fr_pair_list_afrom_str(ctx, request->dict, answer, &vps) == T_INVALID) {
546  RPERROR("Failed parsing output from: %s", cmd);
547  if (out) strlcpy(out, answer, len);
548  ret = -1;
549  }
550 
551  /*
552  * We want to mark the new attributes as tainted,
553  * but not the existing ones.
554  */
555  fr_pair_list_tainted(&vps);
556  fr_pair_add(output_pairs, vps);
557 
558  } else if (out) {
559  /*
560  * We've not been told to extract output pairs,
561  * just copy the programs output to the out
562  * buffer.
563  */
564  strlcpy(out, answer, outlen);
565  }
566 
567 wait:
568  child_pid = waitpid(pid, &status, 0);
569  if (child_pid == 0) {
570  RERROR("Timeout waiting for child");
571 
572  return -2;
573  }
574 
575  if (child_pid == pid) {
576  if (WIFEXITED(status)) {
577  status = WEXITSTATUS(status);
578  if ((status != 0) || (ret < 0)) {
579  RERROR("Program returned code (%d) and output \"%pV\"", status,
580  fr_box_strvalue_len(answer, len));
581  } else {
582  RDEBUG2("Program returned code (%d) and output \"%pV\"", status,
583  fr_box_strvalue_len(answer, len));
584  }
585 
586  return ret < 0 ? ret : status;
587  }
588  }
589 
590  RERROR("Abnormal child exit: %s", fr_syserror(errno));
591 
592  return -1;
593 }

596 /** Execute a program without waiting for the program to finish.
597  *
598  * @param request the request
599  * @param vb as returned by xlat_frame_eval()
600  * @param env_pairs VPs to put into into the environment. May be NULL.
601  * @return
602  * - <0 on error
603  * - 0 on success
604  *
605  * @todo - maybe take an fr_cursor_t instead of env_pairs? That
606  * would allow finer-grained control over the attributes to put into
607  * the environment.
608  */
609 int fr_exec_nowait(request_t *request, fr_value_box_t *vb, fr_pair_t *env_pairs)
610 {
611  int argc;
612  char **envp;
613  char **argv;
614  pid_t pid;
615  fr_value_box_t *first;
616 
617  /*
618  * Ensure that we don't do anything stupid.
619  */
620  first = fr_value_box_list_get(vb, 0);
621  if (first->type == FR_TYPE_GROUP) first = first->vb_group;
622  if (first->tainted) {
623  fr_strerror_printf("Program to run comes from tainted source - %pV", first);
624  return -1;
625  }
626 
627  /*
628  * Get the environment variables.
629  */
630  if (env_pairs) {
631  MEM(envp = talloc_zero_array(request, char *, MAX_ENVP));
632  fr_exec_pair_to_env(request, env_pairs, envp, MAX_ENVP, true);
633  } else {
634  MEM(envp = talloc_zero_array(request, char *, 1));
635  envp[0] = NULL;
636  }
637 
638  argc = fr_value_box_list_flatten_argv(request, &argv, vb);
639  if (argc < 0) {
640  talloc_free(envp);
641  return -1;
642  }
643 
644  if (DEBUG_ENABLED3) {
645  int i;
646 
647  for (i = 0; i < argc; i++) RDEBUG3("arg[%d] %s", i, argv[i]);
648  }
649 
650  pid = fork();
651 
652  /*
653  * The child never returns from calling fr_exec_child();
654  */
655  if (pid == 0) {
656  int unused[2];
657 
658  fr_exec_child(request, argv, envp, false, NULL, NULL, unused, unused);
659  }
660 
661  /*
662  * Parent process. Do all necessary cleanups.
663  */
664  talloc_free(envp);
665 
666  if (pid < 0) {
667  ERROR("Couldn't fork %s: %s", argv[0], fr_syserror(errno));
668  talloc_free(argv);
669  return -1;
670  }
671 
672  /*
673  * Ensure that we can clean up any child processes. We
674  * don't want them left over as zombies.
675  */
676  if (fr_event_pid_wait(request->el, request->el, NULL, pid, NULL, NULL) < 0) return -1;
677 
678  return 0;
679 }

681 /** Execute a program assuming that the caller waits for it to finish.
682  *
683  * The caller takes responsibility for calling waitpid() on the returned PID.
684  *
685  * The caller takes responsibility for reading from the returned FD,
686  * and closing it.
687  *
688  * @param request the request
689  * @param vb as returned by xlat_frame_eval()
690  * @param env_pairs VPs to put into into the environment. May be NULL.
691  * @param[out] pid_p The PID of the child
692  * @param[out] input_fd The stdin FD of the child
693  * @param[out] output_fd The stdout FD of the child
694  * @return
695  * - <0 on error
696  * - 0 on success
697  *
698  * @todo - maybe take an fr_cursor_t instead of env_pairs? That
699  * would allow finer-grained control over the attributes to put into
700  * the environment.
701  */
702 int fr_exec_wait_start(request_t *request, fr_value_box_t *vb, fr_pair_t *env_pairs, pid_t *pid_p, int *input_fd, int *output_fd)
703 {
704  int argc;
705  char **envp;
706  char **argv;
707  pid_t pid;
708  fr_value_box_t *first;
709  int to_child[2] = {-1, -1};
710  int from_child[2] = {-1, -1};
711 
712  /*
713  * Ensure that we don't do anything stupid.
714  */
715  first = fr_value_box_list_get(vb, 0);
