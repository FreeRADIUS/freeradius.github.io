98 /*
99  * Query the database executing a command with no result rows
100  */
101 static int rediswho_command(rlm_rediswho_t const *inst, request_t *request, char const *fmt)
102 {
103  fr_redis_conn_t *conn;
104 
105  int ret = -1;
106 
107  fr_redis_cluster_state_t state;
108  fr_redis_rcode_t status;
109  redisReply *reply = NULL;
110  int s_ret;
111 
112  uint8_t const *key = NULL;
113  size_t key_len = 0;
114 
115  int argc;
116  char const *argv[MAX_REDIS_ARGS];
117  char argv_buf[MAX_REDIS_COMMAND_LEN];
118 
119  if (!fmt || !*fmt) return 0;
120 
121  argc = rad_expand_xlat(request, fmt, MAX_REDIS_ARGS, argv, false, sizeof(argv_buf), argv_buf);
122  if (argc < 0) {
123  RPEDEBUG("Invalid command: %s", fmt);
124  return -1;
125  }
126 
127  /*
128  * If we've got multiple arguments, the second one is usually the key.
129  * The Redis docs say commands should be analysed first to get key
130  * positions, but this involves sending them to the server, which is
131  * just as expensive as sending them to the wrong server and receiving
132  * a redirect.
133  */
134  if (argc > 1) {
135  key = (uint8_t const *)argv[1];
136  key_len = strlen((char const *)key);
137  }
138 
139  for (s_ret = fr_redis_cluster_state_init(&state, &conn, inst->cluster, request, key, key_len, false);
140  s_ret == REDIS_RCODE_TRY_AGAIN; /* Continue */
141  s_ret = fr_redis_cluster_state_next(&state, &conn, inst->cluster, request, status, &reply)) {
142  reply = redisCommandArgv(conn->handle, argc, argv, NULL);
143  status = fr_redis_command_status(conn, reply);
144  }
145  if (s_ret != REDIS_RCODE_SUCCESS) {
146  RERROR("Failed inserting accounting data");
147  error:
148  fr_redis_reply_free(&reply);
149  return -1;
150  }
151  if (!fr_cond_assert(reply)) goto error;
152 
153  /*
154  * Write the response to the debug log
155  */
156  fr_redis_reply_print(L_DBG_LVL_2, reply, request, 0);
157 
158  switch (reply->type) {
159  case REDIS_REPLY_ERROR:
160  break;
161 
162  case REDIS_REPLY_INTEGER:
163  if (reply->integer > 0) ret = reply->integer;
164  break;
165 
166  /*
167  * We don't know to interpret this, the user has probably messed
168  * up the queries, so print an error message and fail.
169  */
170  default:
171  REDEBUG("Expected type \"integer\" got type \"%s\"",
172  fr_table_str_by_value(redis_reply_types, reply->type, "<UNKNOWN>"));
173  break;
174  }
175  fr_redis_reply_free(&reply);
176 
177  return ret;
178 }

180 static unlang_action_t mod_accounting_all(rlm_rcode_t *p_result, rlm_rediswho_t const *inst, request_t *request,
181  char const *insert,
182  char const *trim,
183  char const *expire)
184 {
185  int ret;
186 
187  ret = rediswho_command(inst, request, insert);
188  if (ret < 0) RETURN_MODULE_FAIL;
189 
190  /* Only trim if necessary */
191  if ((inst->trim_count >= 0) && (ret > inst->trim_count)) {
192  if (rediswho_command(inst, request, trim) < 0) RETURN_MODULE_FAIL;
193  }
194 
195  if (rediswho_command(inst, request, expire) < 0) RETURN_MODULE_FAIL;
196  RETURN_MODULE_OK;
197 }

199 static unlang_action_t CC_HINT(nonnull) mod_accounting(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
200 {
201  rlm_rediswho_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_rediswho_t);
202  rlm_rcode_t rcode;
203  fr_pair_t *vp;
204  fr_dict_enum_t *dv;
205  CONF_SECTION *cs;
206  char const *insert, *trim, *expire;
207 
208  vp = fr_pair_find_by_da(&request->request_pairs, attr_acct_status_type);
209  if (!vp) {
210  RDEBUG2("Could not find account status type in packet");
211  RETURN_MODULE_NOOP;
212  }
213 
214  dv = fr_dict_enum_by_value(vp->da, &vp->data);
215  if (!dv) {
216  RDEBUG2("Unknown Acct-Status-Type %u", vp->vp_uint32);
217  RETURN_MODULE_NOOP;
218  }
219 
220  cs = cf_section_find(inst->cs, dv->name, NULL);
221  if (!cs) {
222  RDEBUG2("No subsection %s", dv->name);
223  RETURN_MODULE_NOOP;
224  }
225 
226  insert = cf_pair_value(cf_pair_find(cs, "insert"));
227  trim = cf_pair_value(cf_pair_find(cs, "trim"));
228  expire = cf_pair_value(cf_pair_find(cs, "expire"));
229 
230  return mod_accounting_all(&rcode, inst, request, insert, trim, expire);
231 }
