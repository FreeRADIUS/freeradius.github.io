52 static unlang_action_t mod_process(rlm_rcode_t *p_result, UNUSED module_ctx_t const *mctx, request_t *request)
53 {
54  fr_pair_t *vp;
55  rlm_rcode_t rcode;
56  CONF_SECTION *unlang;
57  fr_dict_enum_t const *dv;
58 
59  REQUEST_VERIFY(request);
60 
61  switch (request->request_state) {
62  case REQUEST_INIT:
63  if (request->parent && RDEBUG_ENABLED) {
64  RDEBUG("Received %s ID %i", fr_packet_codes[request->packet->code], request->packet->id);
65  log_request_pair_list(L_DBG_LVL_1, request, request->request_pairs, "");
66  }
67 
68  request->component = "radius";
69 
70  unlang = cf_section_find(request->server_cs, "recv", "Accounting-Request");
71  if (!unlang) {
72  REDEBUG("Failed to find 'recv Accounting-Request' section");
73  request->reply->code = FR_CODE_DO_NOT_RESPOND;
74  goto send_reply;
75  }
76 
77  RDEBUG("Running 'recv Accounting-Request' from file %s", cf_filename(unlang));
78  if (unlang_interpret_push_section(request, unlang, RLM_MODULE_NOOP, UNLANG_TOP_FRAME) < 0) {
79  RETURN_MODULE_FAIL;
80  }
81 
82  request->request_state = REQUEST_RECV;
83  FALL_THROUGH;
84 
85  case REQUEST_RECV:
86  rcode = unlang_interpret(request);
87 
88  if (request->master_state == REQUEST_STOP_PROCESSING) {
89  *p_result = RLM_MODULE_HANDLED;
90  return UNLANG_ACTION_STOP_PROCESSING;
91  }
92 
93  if (rcode == RLM_MODULE_YIELD) return UNLANG_ACTION_YIELD;
94 
95  switch (rcode) {
96  /*
97  * The module has a number of OK return codes.
98  */
99  case RLM_MODULE_NOOP:
100  case RLM_MODULE_OK:
101  case RLM_MODULE_UPDATED:
102  case RLM_MODULE_HANDLED:
103  request->reply->code = FR_CODE_ACCOUNTING_RESPONSE;
104  break;
105 
106  /*
107  * The module failed, or said the request is
108  * invalid, therefore we stop here.
109  */
110  case RLM_MODULE_FAIL:
111  case RLM_MODULE_INVALID:
112  case RLM_MODULE_NOTFOUND:
113  case RLM_MODULE_REJECT:
114  case RLM_MODULE_DISALLOW:
115  default:
116  RDEBUG("The 'recv Accounting-Request' section returned %s - not sending a response",
117  fr_table_str_by_value(rcode_table, rcode, "???"));
118  request->reply->code = FR_CODE_DO_NOT_RESPOND;
119  goto send_reply;
120  }
121 
122  /*
123  * Run accounting foo { ... }
124  */
125  vp = fr_pair_find_by_da(&request->request_pairs, attr_acct_status_type);
126  if (!vp) goto setup_send;
127 
128  dv = fr_dict_enum_by_value(vp->da, &vp->data);
129  if (!dv) goto setup_send;
130 
131  unlang = cf_section_find(request->server_cs, "accounting", dv->name);
132  if (!unlang) {
133  RDEBUG2("No 'accounting %s { ... }' section found - skipping...", dv->name);
134  goto setup_send;
135  }
136 
137  RDEBUG("Running 'accounting %s' from file %s", cf_section_name2(unlang), cf_filename(unlang));
138  if (unlang_interpret_push_section(request, unlang, RLM_MODULE_NOOP, UNLANG_TOP_FRAME) < 0) {
139  RETURN_MODULE_FAIL;
140  }
141 
142  request->request_state = REQUEST_PROCESS;
143  FALL_THROUGH;
144 
145  case REQUEST_PROCESS:
146  rcode = unlang_interpret(request);
147 
148  if (request->master_state == REQUEST_STOP_PROCESSING) {
149  *p_result = RLM_MODULE_HANDLED;
150  return UNLANG_ACTION_STOP_PROCESSING;
151  }
152 
153  if (rcode == RLM_MODULE_YIELD) return UNLANG_ACTION_YIELD;
154 
155  switch (rcode) {
156  /*
157  * The module has a number of OK return codes.
158  */
159  case RLM_MODULE_NOOP:
160  case RLM_MODULE_OK:
161  case RLM_MODULE_UPDATED:
162  case RLM_MODULE_HANDLED:
163  break;
164 
165  /*
166  * The module failed, or said the request is
167  * invalid, therefore we stop here.
168  */
169  case RLM_MODULE_FAIL:
170  case RLM_MODULE_INVALID:
171  case RLM_MODULE_NOTFOUND:
172  case RLM_MODULE_REJECT:
173  case RLM_MODULE_DISALLOW:
174  default:
175  RDEBUG("The 'accounting' section returned %s - not sending a response",
176  fr_table_str_by_value(rcode_table, rcode, "???"));
177  request->reply->code = FR_CODE_DO_NOT_RESPOND;
178  goto send_reply;
179  }
180 
181  setup_send:
182  /*
183  * Allow for over-ride of reply code.
184  */
185  vp = fr_pair_find_by_da(&request->reply_pairs, attr_packet_type);
186  if (vp) request->reply->code = vp->vp_uint32;
187 
188  dv = fr_dict_enum_by_value(attr_packet_type, fr_box_uint32(request->reply->code));
189  unlang = NULL;
190  if (dv) unlang = cf_section_find(request->server_cs, "send", dv->name);
191 
192  if (!unlang) goto send_reply;
193 
194  RDEBUG("Running 'send %s' from file %s", cf_section_name2(unlang), cf_filename(unlang));
195  if (unlang_interpret_push_section(request, unlang, RLM_MODULE_NOOP, UNLANG_TOP_FRAME) < 0) {
196  RETURN_MODULE_FAIL;
197  }
198 
199  request->request_state = REQUEST_SEND;
200  FALL_THROUGH;
201 
202  case REQUEST_SEND:
203  rcode = unlang_interpret(request);
204 
205  if (request->master_state == REQUEST_STOP_PROCESSING) {
206  *p_result = RLM_MODULE_HANDLED;
207  return UNLANG_ACTION_STOP_PROCESSING;
208  }
209 
210  if (rcode == RLM_MODULE_YIELD) return UNLANG_ACTION_YIELD;
211 
212  switch (rcode) {
213  case RLM_MODULE_NOOP:
214  case RLM_MODULE_OK:
215  case RLM_MODULE_UPDATED:
216  case RLM_MODULE_HANDLED:
217  /* reply is already set */
218  break;
219 
220  default:
221  RDEBUG("The 'send Accounting-Response' section returned %s - not sending a response",
222  fr_table_str_by_value(rcode_table, rcode, "???"));
223  request->reply->code = FR_CODE_DO_NOT_RESPOND;
224  break;
225  }
226 
227  send_reply:
228  request->reply->timestamp = fr_time();
229  /*
230  * Check for "do not respond".
231  */
232  if (request->reply->code == FR_CODE_DO_NOT_RESPOND) {
233  RDEBUG("Not sending reply to client.");
234  break;
235  }
236 
237  if (request->parent && RDEBUG_ENABLED) {
238  RDEBUG("Sending %s ID %i", fr_packet_codes[request->reply->code], request->reply->id);
239  log_request_pair_list(L_DBG_LVL_1, request, request->reply_pairs, "");
240  }
241  break;
242 
243  default:
244  RETURN_MODULE_FAIL;
245  }
246 
247  RETURN_MODULE_OK;
248 }
