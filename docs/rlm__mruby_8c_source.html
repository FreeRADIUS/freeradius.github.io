72 static void mruby_parse_config(mrb_state *mrb, CONF_SECTION *cs, int lvl, mrb_value hash)
73 {
74  int indent_section = (lvl + 1) * 4;
75  int indent_item = (lvl + 2) * 4;
76  CONF_ITEM *ci = NULL;
77 
78  if (!cs) return;
79 
80  DEBUG("%*s%s {", indent_section, " ", cf_section_name1(cs));
81 
82  while ((ci = cf_item_next(cs, ci))) {
83  if (cf_item_is_section(ci)) {
84  CONF_SECTION *sub_cs = cf_item_to_section(ci);
85  char const *key = cf_section_name1(sub_cs);
86  mrb_value sub_hash, mrubyKey;
87 
88  if (!key) continue;
89 
90  mrubyKey = mrb_str_new_cstr(mrb, key);
91 
92  if (!mrb_nil_p(mrb_hash_get(mrb, hash, mrubyKey))) {
93  WARN("rlm_mruby: Ignoring duplicate config section '%s'", key);
94  continue;
95  }
96 
97  sub_hash = mrb_hash_new(mrb);
98  mrb_hash_set(mrb, hash, mrubyKey, sub_hash);
99 
100  mruby_parse_config(mrb, sub_cs, lvl + 1, sub_hash);
101  } else if (cf_item_is_pair(ci)) {
102  CONF_PAIR *cp = cf_item_to_pair(ci);
103  const char *key = cf_pair_attr(cp);
104  const char *value = cf_pair_value(cp);
105  mrb_value mrubyKey, mrubyValue;
106 
107  if (!key || !value) continue;
108 
109  mrubyKey = mrb_str_new_cstr(mrb, key);
110  mrubyValue = mrb_str_new_cstr(mrb, value);
111 
112  if (!mrb_nil_p(mrb_hash_get(mrb, hash, mrubyKey))) {
113  WARN("rlm_mruby: Ignoring duplicate config item '%s'", key);
114  continue;
115  }
116 
117  mrb_hash_set(mrb, hash, mrubyKey, mrubyValue);
118 
119  DEBUG("%*s%s = %s", indent_item, " ", key, value);
120  }
121  }
122 
123  DEBUG("%*s}", indent_section, " ");
124 }

132 static int mod_instantiate(void *instance, CONF_SECTION *conf)
133 {
134  rlm_mruby_t *inst = instance;
135  mrb_state *mrb;
136  CONF_SECTION *cs;
137  FILE *f;
138  mrb_value status;
139 
140  mrb = inst->mrb = mrb_open();
141  if (!mrb) {
142  ERROR("mruby initialization failed");
143  return -1;
144  }
145 
146  /* Define the freeradius module */
147  DEBUG("Creating module %s", inst->module_name);
148  inst->mruby_module = mrb_define_module(mrb, inst->module_name);
149  if (!inst->mruby_module) {
150  ERROR("Creating module %s failed", inst->module_name);
151  return -1;
152  }
153 
154  /* Define the log method */
155  mrb_define_class_method(mrb, inst->mruby_module, "log", mruby_log, MRB_ARGS_REQ(2));
156 
157 #define A(x) mrb_define_const(mrb, inst->mruby_module, #x, mrb_fixnum_value(x));
158  /* Define the logging constants */
159  A(L_DBG);
160  A(L_WARN);
161  A(L_INFO);
162  A(L_ERR);
163  A(L_WARN);
164  A(L_DBG_WARN);
165  A(L_DBG_ERR);
166  A(L_DBG_WARN_REQ);
167  A(L_DBG_ERR_REQ);
168 
169  /* Define the return value constants */
170  A(RLM_MODULE_REJECT)
171  A(RLM_MODULE_FAIL)
172  A(RLM_MODULE_OK)
173  A(RLM_MODULE_HANDLED)
174  A(RLM_MODULE_INVALID)
175  A(RLM_MODULE_DISALLOW)
176  A(RLM_MODULE_NOTFOUND)
177  A(RLM_MODULE_NOOP)
178  A(RLM_MODULE_UPDATED)
179  A(RLM_MODULE_NUMCODES)
180 #undef A
181 
182  /* Convert a FreeRADIUS config structure into a mruby hash */
183  inst->mrubyconf_hash = mrb_hash_new(mrb);
184  cs = cf_section_find(conf, "config", NULL);
185  if (cs) mruby_parse_config(mrb, cs, 0, inst->mrubyconf_hash);
186 
187  /* Define the Request class */
188  inst->mruby_request = mruby_request_class(mrb, inst->mruby_module);
189 
190  DEBUG("Loading file %s...", inst->filename);
191  f = fopen(inst->filename, "r");
192  if (!f) {
193  ERROR("Opening file failed");
194  return -1;
195  }
196 
197  status = mrb_load_file(mrb, f);
198  if (mrb_undef_p(status)) {
199  ERROR("Parsing file failed");
200  return -1;
201  }
202  fclose(f);
203 
204  status = mrb_funcall(mrb, mrb_obj_value(inst->mruby_module), "instantiate", 0);
205  if (mrb_undef_p(status)) {
206  ERROR("Running instantiate failed");
207  return -1;
208  }
209 
210  return 0;
211 }

213 static int mruby_vps_to_array(request_t *request, mrb_value *out, mrb_state *mrb, fr_pair_t **vps)
214 {
215  mrb_value res;
216  fr_pair_t *vp;
217  fr_cursor_t cursor;
218 
219  res = mrb_ary_new(mrb);
220  for (vp = fr_cursor_init(&cursor, vps); vp; vp = fr_cursor_next(&cursor)) {
221  mrb_value tmp, key, val, to_cast;
222 
223  tmp = mrb_ary_new_capa(mrb, 2);
224  key = mrb_str_new(mrb, vp->da->name, strlen(vp->da->name));
225 
226  /*
227  * The only way to create floats, doubles, bools etc,
228  * is to feed mruby the string representation and have
229  * it convert to its internal types.
230  */
231  switch (vp->vp_type) {
232  case FR_TYPE_STRING:
233  case FR_TYPE_OCTETS:
234  to_cast = mrb_str_new(mrb, vp->vp_ptr, vp->vp_length);
235  break;
236 
237  case FR_TYPE_BOOL:
238 #ifndef NDEBUG
239  to_cast = mrb_nil_value(); /* Not needed but clang flags it */
240 #endif
241  break;
242 
243  default:
244  {
245  char *in;
246  size_t len;
247 
248  len = fr_value_box_aprint(request, &in, &vp->data, NULL);
249  to_cast = mrb_str_new(mrb, in, len);
250  talloc_free(in);
251  }
252  break;
253  }
254 
255  switch (vp->vp_type) {
256  case FR_TYPE_STRING:
257  case FR_TYPE_OCTETS:
258  case FR_TYPE_IPV4_ADDR:
259  case FR_TYPE_IPV4_PREFIX:
260  case FR_TYPE_IPV6_ADDR:
261  case FR_TYPE_IPV6_PREFIX:
262  case FR_TYPE_IFID:
263  case FR_TYPE_ETHERNET:
264  val = to_cast; /* No conversions required */
265  break;
266 
267  case FR_TYPE_BOOL:
268  val = vp->vp_bool ? mrb_obj_value(mrb->true_class) : mrb_obj_value(mrb->false_class);
269  break;
270 
271  case FR_TYPE_UINT8:
272  case FR_TYPE_UINT16:
273  case FR_TYPE_UINT32:
274  case FR_TYPE_UINT64:
275  case FR_TYPE_INT8:
276  case FR_TYPE_INT16:
277  case FR_TYPE_INT32:
278  case FR_TYPE_INT64:
279  case FR_TYPE_DATE:
280  case FR_TYPE_TIME_DELTA:
281  case FR_TYPE_SIZE:
282  val = mrb_convert_type(mrb, to_cast, MRB_TT_FIXNUM, "Fixnum", "to_int");
283  break;
284 
285  case FR_TYPE_FLOAT32:
286  case FR_TYPE_FLOAT64:
287  val = mrb_convert_type(mrb, to_cast, MRB_TT_FLOAT, "Float", "to_f");
288  break;
289 
290  case FR_TYPE_NON_VALUES:
291  fr_assert(0);
292  return -1;
293  }
294 
295  mrb_ary_push(mrb, tmp, key);
296  mrb_ary_push(mrb, tmp, val);
297  mrb_ary_push(mrb, res, tmp);
298 
299  *out = res;
300  }
301 
302  return 0;
303 }

305 static void add_vp_tuple(TALLOC_CTX *ctx, request_t *request, fr_pair_t **vps, mrb_state *mrb, mrb_value value, char const *function_name)
306 {
307  int i;
308 
309  for (i = 0; i < RARRAY_LEN(value); i++) {
310  mrb_value tuple = mrb_ary_entry(value, i);
311  mrb_value key, val;
312  char const *ckey, *cval;
313  fr_pair_t *vp;
314  tmpl_t *dst;
315  fr_token_t op = T_OP_EQ;
316 
317  /* This tuple should be an array of length 2 */
318  if (mrb_type(tuple) != MRB_TT_ARRAY) {
319  REDEBUG("add_vp_tuple, %s: non-array passed at index %i", function_name, i);
320  continue;
321  }
322 
323  if (RARRAY_LEN(tuple) != 2 && RARRAY_LEN(tuple) != 3) {
324  REDEBUG("add_vp_tuple, %s: array with incorrect length passed at index "
325  "%i, expected 2 or 3, got %"PRId64, function_name, i, RARRAY_LEN(tuple));
326  continue;
327  }
328 
329  key = mrb_ary_entry(tuple, 0);
330  val = mrb_ary_entry(tuple, -1);
331  if (mrb_type(key) != MRB_TT_STRING) {
332  REDEBUG("add_vp_tuple, %s: tuple element %i must have a string as first element", function_name, i);
333  continue;
334  }
335 
336  ckey = mrb_str_to_cstr(mrb, key);
337  cval = mrb_str_to_cstr(mrb, mrb_obj_as_string(mrb, val));
338  if (ckey == NULL || cval == NULL) {
339  REDEBUG("%s: string conv failed", function_name);
340  continue;
341  }
342 
343 
344  if (RARRAY_LEN(tuple) == 3) {
345  if (mrb_type(mrb_ary_entry(tuple, 1)) != MRB_TT_STRING) {
346  REDEBUG("Invalid type for operator, expected string, falling back to =");
347  } else {
348  char const *cop = mrb_str_to_cstr(mrb, mrb_ary_entry(tuple, 1));
349  if (!(op = fr_table_value_by_str(fr_tokens_table, cop, 0))) {
350  REDEBUG("Invalid operator: %s, falling back to =", cop);
351  op = T_OP_EQ;
352  }
353  }
354  }
355  DEBUG("%s: %s %s %s", function_name, ckey, fr_table_str_by_value(fr_tokens_table, op, "="), cval);
356 
357  if (tmpl_afrom_attr_str(request, NULL, &dst, ckey,
358  &(tmpl_rules_t){
359  .dict_def = request->dict,
360  .list_def = PAIR_LIST_REPLY
361  }) <= 0) {
362  ERROR("Failed to find attribute %s", ckey);
363  continue;
364  }
365 
366  if (radius_request(&request, tmpl_request(dst)) < 0) {
367  ERROR("Attribute name %s refers to outer request but not in a tunnel, skipping...", ckey);
368  talloc_free(dst);
369  continue;
370  }
371 
372  MEM(vp = fr_pair_afrom_da(ctx, tmpl_da(dst)));
373  talloc_free(dst);
374 
375  vp->op = op;
376  if (fr_pair_value_from_str(vp, cval, -1, '\0', false) < 0) {
377  REDEBUG("%s: %s %s %s failed", function_name, ckey, fr_table_str_by_value(fr_tokens_table, op, "="), cval);
378  } else {
379  DEBUG("%s: %s %s %s OK", function_name, ckey, fr_table_str_by_value(fr_tokens_table, op, "="), cval);
380  }
381 
382  radius_pairmove(request, vps, vp, false);
383  }
384 }

400 static rlm_rcode_t CC_HINT(nonnull) do_mruby(request_t *request, rlm_mruby_t const *inst, char const *function_name)
401 {
402  mrb_state *mrb = inst->mrb;
403  mrb_value mruby_request, mruby_result;
404 
405  mruby_request = mrb_obj_new(mrb, inst->mruby_request, 0, NULL);
406  mrb_iv_set(mrb, mruby_request, mrb_intern_cstr(mrb, "@frconfig"), inst->mrubyconf_hash);
407  mruby_set_vps(request, mrb, mruby_request, "@request", &request->request_pairs);
408  mruby_set_vps(request, mrb, mruby_request, "@reply", &request->reply_pairs);
409  mruby_set_vps(request, mrb, mruby_request, "@control", &request->control_pairs);
410  mruby_set_vps(request, mrb, mruby_request, "@session_state", &request->state);
411 
412 DIAG_OFF(class-varargs)
413  mruby_result = mrb_funcall(mrb, mrb_obj_value(inst->mruby_module), function_name, 1, mruby_request);
414 DIAG_ON(class-varargs)
415 
416  /* Two options for the return value:
417  * - a fixnum: convert to rlm_rcode_t, and return that
418  * - an array: this should have exactly three items in it. The first one
419  * should be a fixnum, this will once again be converted to
420  * rlm_rcode_t and eventually returned. The other two items
421  * should be arrays. The items of the first array should be
422  * merged into reply, the second array into control.
423  */
424  switch (mrb_type(mruby_result)) {
425  /* If it is a Fixnum: return that value */
426  case MRB_TT_FIXNUM:
427  return (rlm_rcode_t)mrb_int(mrb, mruby_result);
428 
429  case MRB_TT_ARRAY:
430  /* Must have exactly three items */
431  if (RARRAY_LEN(mruby_result) != 3) {
432  ERROR("Expected array to have exactly three values, got %" PRId64 " instead", RARRAY_LEN(mruby_result));
433  RETURN_MODULE_FAIL;
434  }
435 
436  /* First item must be a Fixnum, this will be the return type */
437  if (mrb_type(mrb_ary_entry(mruby_result, 0)) != MRB_TT_FIXNUM) {
438  ERROR("Expected first array element to be a Fixnum, got %s instead", RSTRING_PTR(mrb_obj_as_string(mrb, mrb_ary_entry(mruby_result, 0))));
439  RETURN_MODULE_FAIL;
440  }
441 
442  /* Second and third items must be Arrays, these will be the updates for reply and control */
443  if (mrb_type(mrb_ary_entry(mruby_result, 1)) != MRB_TT_ARRAY) {
444  ERROR("Expected second array element to be an Array, got %s instead", RSTRING_PTR(mrb_obj_as_string(mrb, mrb_ary_entry(mruby_result, 1))));
445  RETURN_MODULE_FAIL;
446  } else if (mrb_type(mrb_ary_entry(mruby_result, 2)) != MRB_TT_ARRAY) {
447  ERROR("Expected third array element to be an Array, got %s instead", RSTRING_PTR(mrb_obj_as_string(mrb, mrb_ary_entry(mruby_result, 2))));
448  RETURN_MODULE_FAIL;
449  }
450 
451  add_vp_tuple(request->reply, request, &request->reply_pairs, mrb, mrb_ary_entry(mruby_result, 1), function_name);
452  add_vp_tuple(request, request, &request->control_pairs, mrb, mrb_ary_entry(mruby_result, 2), function_name);
453  return (rlm_rcode_t)mrb_int(mrb, mrb_ary_entry(mruby_result, 0));
454 
455  default:
456  /* Invalid return type */
457  ERROR("Expected return to be a Fixnum or an Array, got %s instead", RSTRING_PTR(mrb_obj_as_string(mrb, mruby_result)));
458  RETURN_MODULE_FAIL;
459  }
460 }
