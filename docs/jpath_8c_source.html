95 /** Escapes special chars
96  *
97  * Escapes any characters that may have special meaning within a jpath expression
98  *
99  * @param request Current request (unused may be NULL).
100  * @param out Where to write the escaped string.
101  * @param outlen Length of the output buffer.
102  * @param in data to escape.
103  * @param arg uctx data, not used.
104  * @return the number of chars written to out.
105  */
106 size_t fr_jpath_escape_func(UNUSED request_t *request, char *out, size_t outlen, char const *in, UNUSED void *arg)
107 {
108  char const *p = in;
109  char *q = out, *end = out + outlen;
110 
111  if (outlen == 0) return 0;
112 
113  while (*p && (q < end)) {
114  if (memchr(escape_chars, *p, sizeof(escape_chars))) {
115  if ((q + 1) >= end) break;
116  q[0] = '\\';
117  q[1] = *p++;
118  q += 2;
119  continue;
120  }
121  *q++ = *p++;
122  }
123  *q = '\0';
124 
125  return q - end;
126 }

128 /** Recursive function for jpath_expr_evaluate
129  *
130  * @param[in,out] ctx to allocate fr_value_box_t in.
131  * @param[out] tail Where to write fr_value_box_t (**).
132  * @param[in] dst_type FreeRADIUS type to convert to.
133  * @param[in] dst_enumv Enumeration values to allow string to integer conversions.
134  * @param[in] object current node in the json tree.
135  * @param[in] jpath to evaluate.
136  * @return
137  * - 1 on match.
138  * - 0 on no match.
139  * - -1 on error.
140  */
141 static int jpath_evaluate(TALLOC_CTX *ctx, fr_value_box_t ***tail,
142  fr_type_t dst_type, fr_dict_attr_t const *dst_enumv,
143  json_object *object, fr_jpath_node_t const *jpath)
144 {
145  fr_value_box_t *value;
146  fr_jpath_node_t const *node;
147  jpath_selector_t const *selector;
148  bool child_matched = false;
149  int ret = 0;
150 
151  /*
152  * Iterate over the nodes, we only recurse for
153  * more complex operations.
154  */
155  for (node = jpath; node; node = node->next) switch (node->selector->type) {
156  case JPATH_SELECTOR_FIELD:
157  if (!fr_json_object_is_type(object, json_type_object)) return 0;
158  if (!json_object_object_get_ex(object, node->selector->field, &object)) return 0;
159  continue;
160 
161  case JPATH_SELECTOR_INDEX:
162  case JPATH_SELECTOR_SLICE:
163  /*
164  * There may be multiple selectors per node
165  */
166  for (selector = node->selector; selector; selector = selector->next) switch (selector->type) {
167  case JPATH_SELECTOR_INDEX:
168  {
169  struct array_list *array_obj; /* Because array_list is a global... */
170 
171  fr_assert(selector->slice[0] != SELECTOR_INDEX_UNSET);
172 
173  if (!fr_json_object_is_type(object, json_type_array)) return 0;
174  array_obj = json_object_get_array(object);
175  if ((selector->slice[0] < 0) ||
176  (selector->slice[0] >= (int32_t)(array_obj->length & INT32_MAX))) continue;
177 
178  ret = jpath_evaluate(ctx, tail, dst_type, dst_enumv,
179  array_obj->array[selector->slice[0]], node->next);
180  if (ret < 0) return ret;
181  if (ret == 1) child_matched = true;
182  }
183  break;
184 
185  case JPATH_SELECTOR_SLICE:
186  {
187  struct array_list *array_obj;
188  int32_t start, end, step, i;
189 
190  if (!fr_json_object_is_type(object, json_type_array)) return 0;
191  array_obj = json_object_get_array(object);
192 
193  /*
194  * This logic may seem slightly odd, but it perfectly
195  * emulates python array slicing behaviour AFAICT
196  */
197  step = selector->slice[2];
198  if (step == SELECTOR_INDEX_UNSET) step = 1;
199 
200  start = selector->slice[0];
201  if (start == SELECTOR_INDEX_UNSET) start = (step < 0) ?
202  (int32_t)((array_obj->length - 1) & INT32_MAX) : 0;
203  else if (start < 0) start = array_obj->length + start;
204 
205  end = selector->slice[1];
206  if (end == SELECTOR_INDEX_UNSET) end = (step < 0) ?
207  -1 : (int32_t)((array_obj->length - 1) & INT32_MAX);
208  else if (end < 0) end = array_obj->length + end;
209 
210  /*
211  * Descending
212  */
213  if (step < 0) for (i = start; (i > end) && (i >= 0); i += step) {
214  fr_assert((i >= 0) && (i < (int32_t)(array_obj->length & INT32_MAX)));
215  ret = jpath_evaluate(ctx, tail, dst_type, dst_enumv,
216  array_obj->array[i], node->next);
217  if (ret < 0) return ret;
218  if (ret == 1) child_matched = true;
219  /*
220  * Ascending
221  */
222  } else for (i = start; (i < end) && (i < (int32_t)(array_obj->length & INT32_MAX)); i += step) {
223  fr_assert((i >= 0) && (i < (int32_t)(array_obj->length & INT32_MAX)));
224  ret = jpath_evaluate(ctx, tail, dst_type, dst_enumv,
225  array_obj->array[i], node->next);
226  if (ret < 0) return ret;
227  if (ret == 1) child_matched = true;
228  }
229  }
230  break;
231 
232  default:
233  fr_assert(0);
234  return -1;
235  }
236  return child_matched ? 1 : 0;
237 
238  /*
239  * Iterate over fields or array indices
240  */
241  case JPATH_SELECTOR_WILDCARD:
242  {
243  int i;
244 
245  if (fr_json_object_is_type(object, json_type_array)) {
246  struct array_list *array_obj;
247 
248  array_obj = json_object_get_array(object);
249  for (i = 0; i < (int32_t)(array_obj->length & INT32_MAX); i++) {
250  ret = jpath_evaluate(ctx, tail, dst_type, dst_enumv,
251  array_obj->array[i], node->next);
252  if (ret < 0) return ret;
253  if (ret == 1) child_matched = true;
254  }
255  return child_matched ? 1 : 0;
256  } else if (fr_json_object_is_type(object, json_type_object)) {
257  json_object_object_foreach(object, field_name, field_value) {
258 #ifndef NDEBUG
259  fr_assert(field_name);
260 #else
261  UNUSED_VAR(field_name);
262 #endif
263  ret = jpath_evaluate(ctx, tail, dst_type, dst_enumv,
264  field_value, node->next);
265  if (ret < 0) return ret;
266  if (ret == 1) child_matched = true;
267  }
268  return child_matched ? 1 : 0;
269  } else return 0;
270  }
271 
272  /*
273  * @todo Brute force it more efficiently.
274  */
275  case JPATH_SELECTOR_RECURSIVE_DESCENT:
276  {
277  int i;
278 
279  if (fr_json_object_is_type(object, json_type_array)) {
280  struct array_list *array_obj;
281 
282  /*
283  * Descend into each element of the array
284  */
285  array_obj = json_object_get_array(object);
286  for (i = 0; i < (int32_t)(array_obj->length & INT32_MAX); i++) {
287  ret = jpath_evaluate(ctx, tail, dst_type, dst_enumv,
288  array_obj->array[i], node);
289  if (ret < 0) return ret;
290  if (ret == 1) child_matched = true;
291  }
292 
293  /*
294  * On the way back up, evaluate the object's fields
295  */
296  ret = jpath_evaluate(ctx, tail, dst_type, dst_enumv,
297  object, node->next);
298  if (ret < 0) return ret;
299  if (ret == 1) child_matched = true;
300 
301  return child_matched ? 1 : 0;
302  } else if (fr_json_object_is_type(object, json_type_object)) {
303  /*
304  * Descend into each field of the object
305  */
306  json_object_object_foreach(object, field_name, field_value) {
307 #ifndef NDEBUG
308  fr_assert(field_name);
309 #else
310  UNUSED_VAR(field_name);
311 #endif
312  ret = jpath_evaluate(ctx, tail, dst_type, dst_enumv,
313  field_value, node);
314  if (ret < 0) return ret;
315  if (ret == 1) child_matched = true;
316  }
317 
318  /*
319  * On the way back up, evaluate the object's fields
320  */
321  ret = jpath_evaluate(ctx, tail, dst_type, dst_enumv,
322  object, node->next);
323  if (ret < 0) return ret;
324  if (ret == 1) child_matched = true;
325 
326  return child_matched ? 1 : 0;
327  }
328 
329  /*
330  * Descend down to the level of the leaf
331  *
332  * Parser guarantees that the recursive descent operator
333  * is never the last in a jpath sequence.
334  */
335  return jpath_evaluate(ctx, tail, dst_type, dst_enumv, object, node->next);
336  }
337 
338  case JPATH_SELECTOR_FILTER_EXPRESSION:
339  case JPATH_SELECTOR_EXPRESSION:
340  case JPATH_SELECTOR_INVALID:
341  case JPATH_SELECTOR_ROOT:
342  case JPATH_SELECTOR_CURRENT:
343  fr_assert(0);
344  return -1; /* Not yet implemented */
345  }
346 
347  /*
348  * We've reached the end of the jpath sequence
349  * we now attempt conversion of the leaf to
350  * the specified value.
351  */
352  value = fr_value_box_alloc_null(ctx);
353  if (fr_json_object_to_value_box(value, value, object, dst_enumv, true) < 0) {
354  talloc_free(value);
355  return -1;
356  }
357 
358  if (fr_value_box_cast_in_place(value, value, dst_type, dst_enumv) < 0) {
359  talloc_free(value);
360  return -1;
361  }
362 
363  **tail = value;
364  *tail = &(**tail)->next;
365  return 1;
366 }

407 /** Print a node list to a string for debugging
408  *
409  * Will not be identical to the original parsed string, but should be sufficient
410  * for testing purposes.
411  *
412  * @param ctx to allocate string in.
413  * @param head of the node list.
414  * @return the string representation of the node list.
415  */
416 char *fr_jpath_asprint(TALLOC_CTX *ctx, fr_jpath_node_t const *head)
417 {
418  fr_jpath_node_t const *node;
419  jpath_selector_t *selector;
420  char *p;
421 
422  p = talloc_zero_array(ctx, char, 1);
423 
424  for (node = head; node; node = node->next) switch (node->selector->type) {
425  case JPATH_SELECTOR_ROOT:
426  p = talloc_strdup_append_buffer(p, "$");
427  break;
428 
429  case JPATH_SELECTOR_CURRENT:
430  p = talloc_strdup_append_buffer(p, "@");
431  break;
432 
433  case JPATH_SELECTOR_WILDCARD:
434  p = talloc_strdup_append_buffer(p, ".*");
435  break;
436 
437  case JPATH_SELECTOR_FIELD:
438  {
439  char buffer[257];
440 
441  fr_jpath_escape_func(NULL, buffer, sizeof(buffer), node->selector->field, NULL);
442  p = talloc_asprintf_append_buffer(p, ".%s", buffer);
443  }
444  break;
445 
446  /*
447  * Bracketed selectors can contain a mix of types
448  */
449  case JPATH_SELECTOR_INDEX:
450  case JPATH_SELECTOR_SLICE:
451  case JPATH_SELECTOR_FILTER_EXPRESSION:
452  case JPATH_SELECTOR_EXPRESSION:
453  p = talloc_strdup_append_buffer(p, "[");
454  for (selector = node->selector; selector; selector = selector->next) switch (selector->type) {
455  case JPATH_SELECTOR_INDEX:
456  p = talloc_asprintf_append_buffer(p, "%i%s", selector->slice[0], selector->next ? "," : "");
457  break;
458 
459  case JPATH_SELECTOR_SLICE:
460  if (selector->slice[0] == SELECTOR_INDEX_UNSET) {
461  p = talloc_strdup_append_buffer(p, ":");
462  } else {
463  p = talloc_asprintf_append_buffer(p, "%i:", selector->slice[0]);
464  }
465  if (selector->slice[1] == SELECTOR_INDEX_UNSET) {
466  p = talloc_strdup_append_buffer(p, ":");
467  } else if (selector->slice[2] == SELECTOR_INDEX_UNSET) {
468  p = talloc_asprintf_append_buffer(p, "%i", selector->slice[1]);
469  } else {
470  p = talloc_asprintf_append_buffer(p, "%i:", selector->slice[1]);
471  }
472  if (selector->slice[2] != SELECTOR_INDEX_UNSET) {
473  p = talloc_asprintf_append_buffer(p, "%i", selector->slice[2]);
474  }
475  if (selector->next) p = talloc_strdup_append_buffer(p, ",");
476  break;
477 
478  default:
479  fr_assert(0); /* Not yet implemented */
480  break;
481  }
482  p = talloc_strdup_append_buffer(p, "]");
483  break;
484 
485  case JPATH_SELECTOR_RECURSIVE_DESCENT:
486  if (node->next) switch (node->next->selector->type) {
487  case JPATH_SELECTOR_SLICE:
488  case JPATH_SELECTOR_INDEX:
489  p = talloc_strdup_append_buffer(p, "..");
490  break;
491 
492  default:
493  p = talloc_strdup_append_buffer(p, ".");
494  break;
495  }
496  break;
497 
498  case JPATH_SELECTOR_INVALID:
499  fr_assert(0);
500  return NULL;
501  }
502 
503  return p;
504 }

532 /** Parse index/slice notation
533  *
534  * Expects in to point to a buffer containing:
535  *
536  @verbatim
537  [<int0>:<int1>:<int2>]
538  @endverbatim
539  *
540  * Where each of the integers and its accompanying delimiter is optional.
541  *
542  * @param selector to populate with index/slice info.
543  * @param in input.
544  * @param inlen length of in.
545  * @return
546  * - > 0 on success.
547  * - <= 0 on error (* -1 to get offset error ocurred at).
548  */
549 static ssize_t jpath_array_parse(jpath_selector_t *selector, char const *in, size_t inlen)
550 {
551  int idx = 0;
552  int32_t num;
553  char const *p, *end = in + inlen;
554  char *q;
555  ssize_t ret;
556 
557  char buffer[33]; /* Max (uin32_t * 3) + 2 + 1 */
558 
559  selector->type = JPATH_SELECTOR_INDEX;
560  selector->slice[0] = SELECTOR_INDEX_UNSET;
561  selector->slice[1] = SELECTOR_INDEX_UNSET;
562  selector->slice[2] = SELECTOR_INDEX_UNSET;
563 
564  /*
565  * Scan forward until we find a delimiter or terminator
566  */
567  for (p = in; p < end; p++) if ((p[0] == ',') || (p[0] == ']')) break;
568  if (p == end) {
569  fr_strerror_printf("Missing selector delimiter ',' or terminator ']'");
570  return -inlen;
571  }
572 
573  ret = (p - in);
574  if (ret == 0) {
575  fr_strerror_printf("Empty selector");
576  return 0;
577  }
578 
579  if (inlen > sizeof(buffer)) { /* - 1 for ] */
580  fr_strerror_printf("Selector too long");
581  return -inlen;
582  }
583 
584  /*
585  * Have to use an intermediary buffer because strtol
586  * doesn't accept a terminating address via endptr.
587  */
588  memcpy(&buffer, in, p - in);
589  buffer[p - in] = '\0';
590  p = buffer;
591 
592  /*
593  * Index or start
594  */
595  num = (int32_t)strtol(p, &q, 10);
596  if (q > p) switch (q[0]) {
597  default:
598  no_term:
599  fr_strerror_printf("Expected num, ':' or ']'");
600  return buffer - q;
601 
602  case ':': /* More integers to parse */
603  selector->slice[idx] = num;
604  break;
605 
606  case '\0': /* Array index */
607  selector->slice[idx] = num;
608  return ret;
609  }
610  if (q[0] != ':') goto no_term;
611  idx++;
612  p = q + 1;
613 
614  selector->type = JPATH_SELECTOR_SLICE;
615 
616  /*
617  * End
618  */
619  num = (int32_t)strtol(p, &q, 10);
620  if (q > p) switch (q[0]) {
621  default:
622  goto no_term;
623 
624  case ':': /* More integers to parse */
625  selector->slice[idx] = num;
626  break;
627 
628  case '\0': /* Array End */
629  selector->slice[idx] = num;
630  return ret;
631  }
632  if (q[0] != ':') goto no_term;
633  idx++;
634  p = q + 1;
635 
636  /*
637  * Step
638  */
639  num = (int32_t)strtol(p, &q, 10);
640  if (q[0] != '\0') {
641  fr_strerror_printf("Expected num or ']'");
642  return buffer - q;
643  }
644  if (q > p) {
645  if (num == 0) {
646  fr_strerror_printf("Step cannot be 0");
647  return buffer - p;
648  }
649  selector->slice[idx] = num;
650  }
651  return ret;
652 }

654 /** Parse a jpath field
655  *
656  */
657 static size_t jpath_field_parse(fr_jpath_node_t *node, char const *in, size_t inlen)
658 {
659  char buffer[128];
660 
661  char const *p = in, *end = p + inlen;
662  char *buff_p = buffer, *buff_end = buff_p + sizeof(buffer);
663 
664  /*
665  * Field name with optional selector
666  */
667  while (p < end) {
668  int clen;
669 
670  if (buff_p == buff_end) {
671  name_too_big:
672  fr_strerror_printf("Exceeded maximum field name length");
673  return in - p;
674  }
675 
676  clen = fr_utf8_char((uint8_t const *)p, end - p);
677  if (clen == 0) {
678  fr_strerror_printf("Bad UTF8 char");

687  memcpy(buff_p, p, clen);
688  buff_p += clen;
689  p += clen;
690  continue;
691  }
692 
693  switch (p[0]) { /* Normal char */
694  default:
695  *buff_p++ = *p++;
696  continue;
697 
698  /*
699  * Escape sequence
700  */
701  case '\\':
702  if (++p == end) return p - in;
703 
704  if (memchr(escape_chars, p[0], sizeof(escape_chars))) {
705  *buff_p++ = *p++;
706  continue;
707  }
708  *buff_p++ = '\\';
709  continue;
710 
711  /*
712  * Things that mark the end of a field
713  */
714  case '[':
715  case ']': /* Not really, but it's probably not right */
716  case '.':
717  break;
718  }
719  break;
720  }
721 
722  if (buff_p == buffer) {
723  fr_strerror_printf("Empty field specifier");
724  return 0;
725  }
726  node->selector->field = talloc_bstrndup(node, buffer, buff_p - buffer);
727  node->selector->type = JPATH_SELECTOR_FIELD;
728 
729  return p - in;

732 /** parse a jpath selector
733  *
734  */
735 static size_t jpath_selector_parse(fr_jpath_node_t *node, char const *in, size_t inlen)
736 {
737  ssize_t slen;
738  char const *p = in, *end = p + inlen;
739 
740  jpath_selector_t **stail;
741  jpath_selector_t *selector;
742 
743  stail = &node->selector->next;
744  selector = node->selector;
745 
746  if (++p == end) { /* Skip past [ */
747  missing_terminator:
748  fr_strerror_printf("Missing selector terminator ']'");
749  return in - p;
750  }
751 
752  while (p < end) {
753  /*
754  * What kind of selector is it?
755  */
756  switch (p[0]) {
757  case '?': /* Filter expression */
758  slen = jpath_filter_expr_parse(selector, p, end - p);
759  break;
760 
761  case '(': /* Expression */
762  slen = jpath_expr_parse(selector, p, end - p);
763  break;
764 
765  default: /* Index or slice */
766  slen = jpath_array_parse(selector, p, end - p);
767  break;
768  }
769  if (slen <= 0) {
770  p += -slen;
771  return in - p; /* Error */
772  }
773  p += slen;
774  if (p == end) goto missing_terminator;
775  fr_assert(p < end);
776 
777  /*
778  * Things that terminate a selector
779  *
780  * - ']' a selector terminator.
781  * - ',' another selector.
782  */
783  if (p[0] == ']') break; /* We're done */
784  if (p[0] != ',') { /* There's more... */
785  fr_strerror_printf("Expected selector delimiter ','"
786  "or selector terminator ']'");
787  return in - p; /* Error */
788  }
789  if (++p == end) goto missing_terminator;
790 
791  /*
792  * Link in an additional selector
793  */
794  *stail = selector = talloc_zero(node, jpath_selector_t);
795  if (!selector) {
796  fr_strerror_printf("Failed allocating selector");
797  return in - p;
798  }
799  stail = &selector->next;
800  }
801  if (p[0] != ']') goto missing_terminator;
802 
803  p++; /* Skip past ] */
804 
805  return p - in;
806 }

808 /** Parse a jpath string
809  *
810  * Based on the syntax described here http://goessner.net/articles/JsonPath/
811  *
812  * Implements parser for everything except unions and expressions
813  *
814  * @return
815  * - > 0 on success.
816  * - <= 0 on error (* -1 to get offset error ocurred at).
817  */
818 ssize_t fr_jpath_parse(TALLOC_CTX *ctx, fr_jpath_node_t **head, char const *in, size_t inlen)
819 {
820  TALLOC_CTX *tail_ctx = ctx;
821 
822  ssize_t slen;
823  fr_jpath_node_t *node, **tail = head;
824 
825  char const *p = in, *end = p + inlen;
826 
827  *head = NULL;
828 
829 #define NODE_NEW(_node) \
830 do { \
831  tail_ctx = *tail = (_node) = talloc_zero(tail_ctx, fr_jpath_node_t); \
832  (_node)->selector = talloc_zero((_node), jpath_selector_t); \
833  tail = &(_node)->next; \
834 } while (0)
835 
836  if (inlen < 1) {
837  bad_start:
838  fr_strerror_printf("Expected root specifier '$', or current node specifier '@'");
839  return 0;
840  }
841 
842  /*
843  * Start of the jpath expression
844  *
845  * - '$' JSON root node specifier
846  * - '@' Current node specifier
847  */
848  switch (p[0]) {
849  case '$':
850  NODE_NEW(node);
851  node->selector->type = JPATH_SELECTOR_ROOT;
852  p++;
853  break;
854 
855  case '@':
856  NODE_NEW(node);
857  node->selector->type = JPATH_SELECTOR_CURRENT;
858  p++;
859  break;
860 
861  default:
862  goto bad_start;
863  }
864 
865  /*
866  * Valid successions of '$' or '@'
867  *
868  * - '[' Start of a selector
869  * - '.' Start of a field specifier
870  */
871  while (p < end) {
872  NODE_NEW(node);
873 
874  switch (p[0]) {
875  case '.':
876  if (++p == end) {
877  fr_strerror_printf("Expected recursive descent '..' "
878  "wildcard '*' or field specifier");
879  error:
880  TALLOC_FREE(*head);
881  return in - p;
882  }
883 
884  /*
885  * Valid successions of '.'
886  *
887  * - '.' recursive descent.
888  * - '*' wildcard.
889  * - <name> (fieldname).
890  *
891  * This should probably be in its own function, but never mind...
892  */
893  switch (p[0]) {
894  case '.':
895  if ((p != end) && (p[1] == '.')) {
896  fr_strerror_printf("Recursive descent must not be "
897  "followed by child delimiter '.'");
898  p++;
899  goto error;
900  }
901  node->selector->type = JPATH_SELECTOR_RECURSIVE_DESCENT;
902 
903  if ((p + 1) == end) {
904  fr_strerror_printf("Path may not end in recursive descent");
905  goto error;
906  }
907 
908  /*
909  * If and only if, the next char is the beginning
910  * of a selector, advance the pointer.
911  *
912  * Otherwise we leave it pointing to the second '.'
913  * allowing .* and .<field>
914  */
915  if (p[1] == '[') p++;
916  continue;
917 
918  case '*':
919  node->selector->type = JPATH_SELECTOR_WILDCARD;
920  p++;
921  continue;
922 
923  default:
924  /*
925  * Field specifier is the only other valid possibility
926  */
927  slen = jpath_field_parse(node, p, (end - p));
928  if (slen <= 0) {
929  p += -(slen);
930  goto error;
931  }
932  p += slen;
933  if (p == end) return p - in; /* The end of string! */
934  fr_assert(p < end);
935  }
936  break;
937 
938  case '[':
939  slen = jpath_selector_parse(node, p, (end - p));
940  if (slen <= 0) {
941  p += -(slen);
942  goto error;
943  }
944  p += slen;
945  if (p == end) return p - in; /* The end of string! */
946  fr_assert(p < end);
947  break;
948 
949  default:
950  fr_strerror_printf("Expected field specifier '.' or selector '['");
951  goto error;
952  }
953  }
954 
955  return p - in;
956 }
