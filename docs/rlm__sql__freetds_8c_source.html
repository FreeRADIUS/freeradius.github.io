52  *
53  * Callback for any errors raised by the Client-Library. Will overwrite any previous errors associated

56  * @param context The FreeTDS library context.
57  * @param conn DB connection handle.
58  * @param emsgp Pointer to the error structure.

61 static CS_RETCODE CS_PUBLIC clientmsg_callback(CS_CONTEXT *context, UNUSED CS_CONNECTION *conn, CS_CLIENTMSG *emsgp)
62 {
63  rlm_sql_freetds_conn_t *this = NULL;
64  int len = 0;
65 
66  /*
67  * Not actually an error, but the client wanted to tell us something...
68  */
69  if (emsgp->severity == CS_SV_INFORM) {
70  INFO("%s", emsgp->msgstring);
71 
72  return CS_SUCCEED;
73  }
74 
75  if ((cs_config(context, CS_GET, CS_USERDATA, &this, sizeof(this), &len) != CS_SUCCEED) || !this) {
76  ERROR("failed retrieving context userdata");
77 
78  return CS_SUCCEED;
79  }
80 
81  if (this->error) TALLOC_FREE(this->error);
82 
83  this->error = talloc_typed_asprintf(this, "client error: severity(%ld), number(%ld), origin(%ld), layer(%ld): %s",
84  (long)CS_SEVERITY(emsgp->severity), (long)CS_NUMBER(emsgp->msgnumber),
85  (long)CS_ORIGIN(emsgp->msgnumber), (long)CS_LAYER(emsgp->msgnumber),
86  emsgp->msgstring);
87 
88  if (emsgp->osstringlen > 0) {
89  this->error = talloc_asprintf_append(this->error, ". os error: number(%ld): %s",

91  }
92 
93  return CS_SUCCEED;

186 /*************************************************************************
187  *
188  * Function: sql_query
189  *
190  * Purpose: Issue a non-SELECT query (ie: update/delete/insert) to
191  * the database.
192  *
193  *************************************************************************/
194 static sql_rcode_t sql_query(rlm_sql_handle_t *handle, UNUSED rlm_sql_config_t *config, char const *query)
195 {
196  rlm_sql_freetds_conn_t *conn = handle->conn;
197 
198  CS_RETCODE results_ret;
199  CS_INT result_type;
200 
201  if (ct_cmd_alloc(conn->db, &conn->command) != CS_SUCCEED) {
202  ERROR("Unable to allocate command structure (ct_cmd_alloc())");
203 
204  return RLM_SQL_ERROR;
205  }
206 
207  if (ct_command(conn->command, CS_LANG_CMD, query, CS_NULLTERM, CS_UNUSED) != CS_SUCCEED) {
208  ERROR("Unable to initialise command structure (ct_command())");
209 
210  return RLM_SQL_ERROR;
211  }
212 
213  if (ct_send(conn->command) != CS_SUCCEED) {
214  ERROR("Unable to send command (ct_send())");
215 
216  return RLM_SQL_ERROR;
217  }
218 
219  /*
220  * We'll make three calls to ct_results, first to get a success indicator, secondly to get a
221  * done indicator, and thirdly to get a "nothing left to handle" status.
222  */
223 
224  /*
225  * First call to ct_results, we need returncode CS_SUCCEED and result_type CS_CMD_SUCCEED.
226  */
227  if ((results_ret = ct_results(conn->command, &result_type)) == CS_SUCCEED) {
228  if (result_type != CS_CMD_SUCCEED) {
229  if (result_type == CS_ROW_RESULT) {
230  ERROR("sql_query processed a query returning rows. "
231  "Use sql_select_query instead!");
232  }
233  ERROR("Result failure or unexpected result type from query");
234 
235  return RLM_SQL_ERROR;
236  }
237  } else {
238  switch (results_ret) {
239  case CS_FAIL: /* Serious failure, freetds requires us to cancel and maybe even close db */
240  ERROR("Failure retrieving query results");
241 
242  if (ct_cancel(NULL, conn->command, CS_CANCEL_ALL) == CS_FAIL) {
243  INFO("Cleaning up");
244 
245  return RLM_SQL_RECONNECT;
246  }
247  conn->command = NULL;
248 
249  return RLM_SQL_ERROR;
250  default:
251  ERROR("Unexpected return value from ct_results()");
252 
253  return RLM_SQL_ERROR;
254  }
255  }
256 
257  /*
258  * Second call to ct_results, we need returncode CS_SUCCEED
259  * and result_type CS_CMD_DONE.
260  */
261  if ((results_ret = ct_results(conn->command, &result_type)) == CS_SUCCEED) {
262  if (result_type != CS_CMD_DONE) {
263  ERROR("Result failure or unexpected result type from query");
264 
265  return RLM_SQL_ERROR;
266  }
267  } else {
268  switch (results_ret) {
269  case CS_FAIL: /* Serious failure, freetds requires us to cancel and maybe even close db */
270  ERROR("Failure retrieving query results");
271  if (ct_cancel(NULL, conn->command, CS_CANCEL_ALL) == CS_FAIL) return RLM_SQL_RECONNECT;
272 
273  conn->command = NULL;
274  return RLM_SQL_ERROR;
275 
276  default:
277  ERROR("Unexpected return value from ct_results()");
278 
279  return RLM_SQL_ERROR;
280  }
281  }
282 
283  /*
284  * Third call to ct_results, we need returncode CS_END_RESULTS result_type will be ignored.
285  */
286  results_ret = ct_results(conn->command, &result_type);
287  switch (results_ret) {
288  case CS_FAIL: /* Serious failure, freetds requires us to cancel and maybe even close db */
289  ERROR("Failure retrieving query results");
290  if (ct_cancel(NULL, conn->command, CS_CANCEL_ALL) == CS_FAIL) return RLM_SQL_RECONNECT;
291  conn->command = NULL;
292 
293  return RLM_SQL_ERROR;
294 
295  case CS_END_RESULTS: /* This is where we want to end up */
296  break;
297 
298  default:
299  ERROR("Unexpected return value from ct_results()");
300 
301  return RLM_SQL_ERROR;
302  }
303 
304  return RLM_SQL_OK;
305 }

329 /*************************************************************************
330  *
331  * Function: sql_fields
332  *
333  * Purpose: Return name of regular result columns.
334  *
335  *************************************************************************/
336 static sql_rcode_t sql_fields(char const **out[], rlm_sql_handle_t *handle, UNUSED rlm_sql_config_t *config)
337 {
338  rlm_sql_freetds_conn_t *conn = handle->conn;
339  CS_DATAFMT datafmt;
340  int fields, i;
341  char const **names;
342 
343  /* Get number of elements in row result */
344  if (ct_res_info(conn->command, CS_NUMDATA, (CS_INT *)&fields, CS_UNUSED, NULL) != CS_SUCCEED) {
345  ERROR("sql_fields() Error retrieving column count");
346 
347  return RLM_SQL_ERROR;
348  }
349 
350  if (fields <= 0) return RLM_SQL_ERROR;
351 
352  MEM(names = talloc_array(handle, char const *, fields));
353 
354  for (i = 0; i < fields; i++) {
355  int col = i + 1;
356  char *p;
357 
358  /*
359  ** Get the column description. ct_describe() fills the
360  ** datafmt parameter with a description of the column.
361  */
362  if (ct_describe(conn->command, col, &datafmt) != CS_SUCCEED) {
363  ERROR("sql_fields() Problems with ct_describe(), column %d", col);
364  talloc_free(names);
365  return RLM_SQL_ERROR;
366  }
367 
368  if (datafmt.namelen > 0) {
369  MEM(p = talloc_array(names, char, (size_t)datafmt.namelen + 1));
370  strlcpy(p, datafmt.name, (size_t)datafmt.namelen + 1);
371  names[i] = p;
372  }
373  }
374 
375  *out = names;
376 
377  return RLM_SQL_OK;
378 }

425 /** Execute a query when we expected a result set
426  *
427  * @note Only the first row from queries returning several rows will be returned by this function,
428  * consecutive rows will be discarded.
429  *
430  */
431 static sql_rcode_t sql_select_query(rlm_sql_handle_t *handle, rlm_sql_config_t *config, char const *query)
432 {
433  rlm_sql_freetds_conn_t *conn = handle->conn;
434 
435  CS_RETCODE results_ret;
436  CS_INT result_type;
437  CS_DATAFMT descriptor;
438 
439  int colcount,i;
440  char **rowdata;
441 
442  if (!conn->db) {
443  ERROR("socket not connected");
444 
445  return RLM_SQL_ERROR;
446  }
447 
448  if (ct_cmd_alloc(conn->db, &conn->command) != CS_SUCCEED) {
449  ERROR("unable to allocate command structure (ct_cmd_alloc())");
450 
451  return RLM_SQL_ERROR;
452  }
453 
454  if (ct_command(conn->command, CS_LANG_CMD, query, CS_NULLTERM, CS_UNUSED) != CS_SUCCEED) {
455  ERROR("unable to initiate command structure (ct_command()");
456 
457  return RLM_SQL_ERROR;
458  }
459 
460  if (ct_send(conn->command) != CS_SUCCEED) {
461  ERROR("unable to send command (ct_send())");
462  return RLM_SQL_ERROR;
463  }
464 
465  results_ret = ct_results(conn->command, &result_type);
466  switch (results_ret) {
467  case CS_SUCCEED:
468  switch (result_type) {
469  case CS_ROW_RESULT:
470 
471  /*
472  * Set up a target buffer for the results data, and associate the buffer with the results,
473  * but the actual fetching takes place in sql_fetch_row.
474  * The layer above MUST call sql_fetch_row and/or sql_finish_select_query
475  * or this socket will be unusable and may cause segfaults
476  * if reused later on.
477  */
478 
479  /*
480  * Set up the DATAFMT structure that describes our target array
481  * and tells freetds what we want future ct_fetch calls to do.
482  */
483  descriptor.datatype = CS_CHAR_TYPE; /* The target buffer is a string */
484  descriptor.format = CS_FMT_NULLTERM; /* Null termination please */
485  descriptor.maxlength = MAX_DATASTR_LEN; /* The string arrays are this large */
486  descriptor.count = 1; /* Fetch one row of data */
487  descriptor.locale = NULL; /* Don't do NLS stuff */
488 
489  colcount = sql_num_fields(handle, config); /* Get number of elements in row result */
490 
491  rowdata = talloc_zero_array(conn, char *, colcount + 1); /* Space for pointers */
492  rowdata[colcount] = NULL;
493 
494  for (i = 0; i < colcount; i++) {
495  /* Space to hold the result data */
496  rowdata[i] = talloc_array(rowdata, char, MAX_DATASTR_LEN + 1);
497 
498  /* Associate the target buffer with the data */
499  if (ct_bind(conn->command, i + 1, &descriptor, rowdata[i], NULL, NULL) != CS_SUCCEED) {
500  talloc_free(rowdata);
501 
502  ERROR("ct_bind() failed)");
503 
504  return RLM_SQL_ERROR;
505  }
506 
507  }
508 
509  rowdata[i] = NULL; /* Terminate the array */
510  conn->results = rowdata;
511  break;
512 
513  case CS_CMD_SUCCEED:
514  case CS_CMD_DONE:
515  ERROR("query returned no data");
516  break;
517 
518  default:
519 
520  ERROR("unexpected result type from query");
521  sql_finish_select_query(handle, config);
522 
523  return RLM_SQL_ERROR;
524  }
525  break;
526 
527  case CS_FAIL:
528 
529  /*
530  * Serious failure, freetds requires us to cancel the results and maybe even close the db.
531  */
532 
533  ERROR("failure retrieving query results");
534 
535  if (ct_cancel(NULL, conn->command, CS_CANCEL_ALL) == CS_FAIL) {
536  ERROR("cleaning up");
537 
538  return RLM_SQL_RECONNECT;
539  }
540  conn->command = NULL;
541 
542  return RLM_SQL_ERROR;
543 
544  default:
545  ERROR("unexpected return value from ct_results()");
546 
547  return RLM_SQL_ERROR;
548  }
549 
550  return RLM_SQL_OK;
551 }

567 static sql_rcode_t sql_fetch_row(rlm_sql_row_t *out, rlm_sql_handle_t *handle, UNUSED rlm_sql_config_t *config)
568 {
569  rlm_sql_freetds_conn_t *conn = handle->conn;
570  CS_INT ret, count;
571 
572  *out = NULL;
573  handle->row = NULL;
574 
575  ret = ct_fetch(conn->command, CS_UNUSED, CS_UNUSED, CS_UNUSED, &count);
576  switch (ret) {
577  case CS_FAIL:
578  /*
579  * Serious failure, freetds requires us to cancel the results and maybe even close the db.
580  */
581  ERROR("failure fetching row data");
582  if (ct_cancel(NULL, conn->command, CS_CANCEL_ALL) == CS_FAIL) {
583  ERROR("cleaning up");
584  } else {
585  conn->command = NULL;
586  }
587 
588  return RLM_SQL_RECONNECT;
589 
590  case CS_END_DATA:
591  return RLM_SQL_NO_MORE_ROWS;
592 
593  case CS_SUCCEED:
594  *out = handle->row = conn->results;
595 
596  return RLM_SQL_OK;
597 
598  case CS_ROW_FAIL:
599  ERROR("recoverable failure fetching row data");
600 
601  return RLM_SQL_RECONNECT;
602 
603  default:
604  ERROR("unexpected returncode from ct_fetch");
605 
606  return RLM_SQL_ERROR;
607  }
608 }

642 static int _sql_socket_destructor(rlm_sql_freetds_conn_t *conn)
643 {
644  DEBUG2("socket destructor called, closing socket");
645 
646  if (conn->command) {
647  ct_cancel(NULL, conn->command, CS_CANCEL_ALL);
648  if (ct_cmd_drop(conn->command) != CS_SUCCEED) {
649  ERROR("freeing command structure failed");
650 
651  return RLM_SQL_ERROR;
652  }
653  }
654 
655  if (conn->db) {
656  /*
657  * We first try gracefully closing the connection (which informs the server)
658  * Then if that fails we force the connection closure.
659  *
660  * Sybase docs says this may fail because of pending results, but we
661  * should not have any pending results at this point, so something else must
662  * of gone wrong.
663  */
664  if (ct_close(conn->db, CS_UNUSED) != CS_SUCCEED) {
665  ct_close(conn->db, CS_FORCE_CLOSE);
666  }
667 
668  ct_con_drop(conn->db);
669  }
670 
671  if (conn->context) {
672  ct_exit(conn->context, CS_UNUSED);
673  cs_ctx_drop(conn->context);

679 static sql_rcode_t sql_socket_init(rlm_sql_handle_t *handle, rlm_sql_config_t *config, fr_time_delta_t timeout)
680 {
681  rlm_sql_freetds_conn_t *conn;
682  unsigned int timeout_ms = fr_time_delta_to_msec(timeout);
683 
684  MEM(conn = handle->conn = talloc_zero(handle, rlm_sql_freetds_conn_t));
685  talloc_set_destructor(conn, _sql_socket_destructor);
686 
687  /*
688  * Allocate a CS context structure. This should really only be done once, but because of
689  * the db pooling design of rlm_sql, we'll have to go with one context per db
690  */
691  if (cs_ctx_alloc(CS_VERSION_100, &conn->context) != CS_SUCCEED) {
692  ERROR("unable to allocate CS context structure (cs_ctx_alloc())");
693 
694  goto error;
695  }
696 
697  /*
698  * Initialize ctlib
699  */
700  if (ct_init(conn->context, CS_VERSION_100) != CS_SUCCEED) {
701  ERROR("unable to initialize Client-Library");
702 
703  goto error;
704  }
705 
706  if (ct_config(conn->context, CS_GET, CS_LOGIN_TIMEOUT, (CS_VOID *)&timeout_ms, CS_UNUSED, NULL) != CS_SUCCEED) {
707  ERROR("Setting connection timeout failed");
708 
709  goto error;
710  }
711 
712  /*
713  * Install callback functions for error-handling
714  */
715  if (cs_config(conn->context, CS_SET, CS_MESSAGE_CB, (CS_VOID *)csmsg_callback, CS_UNUSED, NULL) != CS_SUCCEED) {
716  ERROR("unable to install CS Library error callback");
717 
718  goto error;
719  }
720 
721  if (cs_config(conn->context, CS_SET, CS_USERDATA,
722  (CS_VOID *)&handle->conn, sizeof(handle->conn), NULL) != CS_SUCCEED) {
723  ERROR("unable to set userdata pointer");
724 
725  goto error;
726  }
727 
728  if (ct_callback(conn->context, NULL, CS_SET, CS_CLIENTMSG_CB, (CS_VOID *)clientmsg_callback) != CS_SUCCEED) {
729  ERROR("unable to install client message callback");
730 
731  goto error;
732  }
733 
734  if (ct_callback(conn->context, NULL, CS_SET, CS_SERVERMSG_CB, (CS_VOID *)servermsg_callback) != CS_SUCCEED) {
735  ERROR("unable to install server message callback");
736 
737  goto error;
738  }
739 
740  /*
741  * Allocate a ctlib db structure
742  */
743  if (ct_con_alloc(conn->context, &conn->db) != CS_SUCCEED) {
744  ERROR("unable to allocate db structure");
745 
746  goto error;
747  }
748 
749  /*
750  * Set User and Password properties for the db
751  */
752  {
753  CS_VOID *login, *password;
754  CS_CHAR *server;
755  char database[128];
756 
757  memcpy(&login, &config->sql_login, sizeof(login));
758  if (ct_con_props(conn->db, CS_SET, CS_USERNAME, login, strlen(config->sql_login), NULL) != CS_SUCCEED) {
759  ERROR("unable to set username for db");
760 
761  goto error;
762  }
763 
764  memcpy(&password, &config->sql_password, sizeof(password));
765  if (ct_con_props(conn->db, CS_SET, CS_PASSWORD,
766  password, strlen(config->sql_password), NULL) != CS_SUCCEED) {
767  ERROR("unable to set password for db");
768 
769  goto error;
770  }
771 
772  /*
773  * Connect to the database
774  */
775  memcpy(&server, &config->sql_server, sizeof(server));
776  if (ct_connect(conn->db, server, strlen(config->sql_server)) != CS_SUCCEED) {
777  ERROR("unable to establish db to symbolic servername %s",
778  config->sql_server);
779 
780  goto error;
781  }
782 
783  /*
784  * There doesn't appear to be a way to set the database with the API, so use an
785  * sql statement when we first open the connection.
786  */
787  snprintf(database, sizeof(database), "USE %s;", config->sql_db);
788  if (sql_query(handle, config, database) != RLM_SQL_OK) {
789  goto error;
790  }
791 
792  sql_finish_query(handle, config);
793  }
794 
795  return RLM_SQL_OK;
796 
797 error:
798  if (conn->context) {
799  sql_log_entry_t error;
800 
801  if (sql_error(NULL, &error, 1, handle, config) > 0) ERROR("%s", error.msg);
802  }
803 
804  return RLM_SQL_ERROR;
805 }
