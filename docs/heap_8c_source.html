29 /*
30  * A heap entry is made of a pointer to the object, which
31  * contains the key. The heap itself is an array of pointers.
32  *
33  * Heaps normally support only ordered insert, and extraction
34  * of the minimum element. The heap entry can contain an "int"
35  * field that holds the entries position in the heap. The offset
36  * of the field is held inside of the heap structure.
37  */
38

51 /*
52  * First node in a heap is element 0. Children of i are 2i+1 and
53  * 2i+2. These macros wrap the logic, so the code is more
54  * descriptive.
55  */
56 #define HEAP_PARENT(_x) (((_x) - 1 ) / 2)
57 #define HEAP_LEFT(_x) (2 * (_x) + 1)
58 /* #define HEAP_RIGHT(_x) (2 * (_x) + 2 ) */
59 #define HEAP_SWAP(_a, _b) { void *_tmp = _a; _a = _b; _b = _tmp; }

99 /** Insert a new element into the heap
100  *
101  * Insert element in heap. Normally, p != NULL, we insert p in a
102  * new position and bubble up. If p == NULL, then the element is
103  * already in place, and key is the position where to start the
104  * bubble-up.
105  *
106  * Returns -1 on failure (cannot allocate new heap entry)
107  *
108  * If offset > 0 the position (index, int) of the element in the
109  * heap is also stored in the element itself at the given offset
110  * in bytes.
111  *
112  * @param[in] hp The heap to insert an element into.
113  * @param[in] data Data to insert into the heap.
114  * @return
115  * - 0 on success.
116  * - -1 on failure (heap full or malloc error).
117  */
118 int fr_heap_insert(fr_heap_t *hp, void *data)
119 {
120  int32_t child;
121 
122  /*
123  * On insert, the heap_id MUST be either:
124  *
125  * -1 = the node was added / removed from the heap
126  * and the heap code set the ID to -1
127  * 0 = the node was just allocated via an "alloc_zero"
128  * function
129  */
130  child = index_get(hp, data);
131  if ((child > 0) || ((child == 0) && (hp->num_elements > 0) && (data == hp->p[0]))) {
132  fr_strerror_printf("Node is already in the heap");
133  return -1;
134  }
135 
136  child = hp->num_elements;
137 
138 #ifndef TALLOC_GET_TYPE_ABORT_NOOP
139  if (hp->type) (void)_talloc_get_type_abort(data, hp->type, __location__);
140 #endif
141 
142  /*
143  * Heap is full. Double it's size.
144  */
145  if ((size_t)child == hp->size) {
146  void **n;
147  size_t n_size = hp->size * 2;
148 
149  /*
150  * heap_id is a 32-bit signed integer. If the heap will
151  * grow to contain more than 2B elements, disallow
152  * integer overflow. Tho TBH, that should really never
153  * happen.
154  */
155  if (n_size > INT32_MAX) {
156  if (hp->size == INT32_MAX) {
157  fr_strerror_printf("Heap is full");
158  return -1;
159  } else {
160  n_size = INT32_MAX;
161  }
162  }
163 
164  n = talloc_realloc(hp, hp->p, void *, n_size);
165  if (!n) {
166  fr_strerror_printf("Failed expanding heap to %zu elements (%zu bytes)",
167  n_size, (n_size * sizeof(void *)));
168  return -1;
169  }
170  hp->size = n_size;
171  hp->p = n;
172  }
173 
174  hp->p[child] = data;
175  hp->num_elements++;
176 
177  fr_heap_bubble(hp, child);
178 
179  return 0;
180 }

182 static void fr_heap_bubble(fr_heap_t *hp, int32_t child)
183 {
184  /*
185  * Bubble up the element.
186  */
187  while (child > 0) {
188  int32_t parent = HEAP_PARENT(child);
189 
190  /*
191  * Parent is smaller than the child. We're done.
192  */
193  if (hp->cmp(hp->p[parent], hp->p[child]) < 0) break;
194 
195  /*
196  * Child is smaller than the parent, repeat.
197  */
198  HEAP_SWAP(hp->p[child], hp->p[parent]);
199  OFFSET_SET(hp, child);
200  child = parent;
201  }
202  OFFSET_SET(hp, child);
203 }

206 /** Remove a node from the heap
207  *
208  * @param[in] hp The heap to extract an element from.
209  * @param[in] data Data to extract from the heap.
210  * @return
211  * - 0 on success.
212  * - -1 on failure (no elements or data not found).
213  */
214 int fr_heap_extract(fr_heap_t *hp, void *data)
215 {
216  int32_t parent, child, max;
217 
218  /*
219  * Extract element. Default is the first one (pop)
220  */
221  if (!data) {
222  if (unlikely((hp->num_elements == 0) || !hp->p[0])) {
223  fr_strerror_printf("Tried to extract element from empty heap");
224  return -1;
225  }
226  parent = 0;
227  } else { /* extract from the middle */
228  parent = index_get(hp, data);
229 
230  /*
231  * Out of bounds.
232  */
233  if (unlikely((parent < 0) || (parent >= hp->num_elements))) {
234  fr_strerror_printf("Heap parent (%i) out of bounds (0-%i)", parent, hp->num_elements);
235  return -1;
236  }
237 
238  if (unlikely(data != hp->p[parent])) {
239  fr_strerror_printf("Invalid heap index. Expected data %p at offset %i, got %p", data,
240  parent, hp->p[parent]);
241  return -1;
242  }
243  }
244  max = hp->num_elements - 1;
245 
246  OFFSET_RESET(hp, parent);
247  child = HEAP_LEFT(parent);
248  while (child <= max) {
249  /*
250  * Maybe take the right child.
251  */
252  if ((child != max) &&
253  (hp->cmp(hp->p[child + 1], hp->p[child]) < 0)) {
254  child = child + 1;
255  }
256  hp->p[parent] = hp->p[child];
257  OFFSET_SET(hp, parent);
258  parent = child;
259  child = HEAP_LEFT(child);
260  }
261  hp->num_elements--;
262 
263  /*
264  * We didn't end up at the last element in the heap.
265  * This element has to be re-inserted.
266  */
267  if (parent != max) {
268  /*
269  * Fill hole with last entry and bubble up,
270  * reusing the insert code
271  */
272  hp->p[parent] = hp->p[max];
273 
274  fr_heap_bubble(hp, parent);
275  }
276 
277  return 0;
278 }
