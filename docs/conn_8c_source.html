37 /** Gets an LDAP socket from the connection pool
38  *
39  * Retrieve a socket from the connection pool, or NULL on error (of if no sockets are available).
40  *
41  * @param inst rlm_ldap configuration.
42  * @param request Current request (may be NULL).
43  */
44 fr_ldap_connection_t *mod_conn_get(rlm_ldap_t const *inst, request_t *request)
45 {
46  fr_ldap_connection_t *conn;
47 
48  conn = fr_pool_connection_get(inst->pool, request);
49 
50  fr_assert(!conn || conn->config);
51 
52 #ifdef LDAP_CONTROL_X_SESSION_TRACKING
53  /*
54  * Add optional session tracking controls,
55  * that contain values of some attributes
56  * in the request.
57  */
58  if ((conn != NULL) && (request != NULL) && inst->session_tracking) {
59  if (fr_ldap_control_add_session_tracking(conn, request) < 0) {
60  fr_pool_connection_release(inst->pool, request, conn);
61  return NULL;
62  }
63  }
64 #endif
65  return conn;
66 }

68 /** Releases an LDAP socket back to the connection pool
69  *
70  * If the socket was rebound chasing a referral onto another server then we destroy it.
71  * If the socket was rebound to another user on the same server, we let the next caller rebind it.
72  *
73  * @param inst rlm_ldap configuration.
74  * @param request The current request.
75  * @param conn to release.
76  */
77 void ldap_mod_conn_release(rlm_ldap_t const *inst, request_t *request, fr_ldap_connection_t *conn)
78 {
79  /*
80  * Could have already been free'd due to a previous error.
81  */
82  if (!conn) return;
83 
84  /*
85  * Clear any client/server controls associated with the connection.
86  */
87  fr_ldap_control_clear(conn);
88 
89  /*
90  * We chased a referral to another server.
91  *
92  * This connection is no longer part of the pool which is
93  * connected to and bound to the configured server.
94  * Close it.
95  *
96  * Note that we do NOT close it if it was bound to another user.
97  * Instead, we let the next caller do the rebind.
98  */
99  if (conn->referred) {
100  fr_pool_connection_close(inst->pool, request, conn);
101  return;
102  }
103 
104  fr_pool_connection_release(inst->pool, request, conn);
105  return;
106 }

108 /** Create and return a new connection
109  *
110  * Create a new ldap connection and allocate memory for a new rlm_handle_t
111  */
112 void *ldap_mod_conn_create(TALLOC_CTX *ctx, void *instance, fr_time_delta_t timeout)
113 {
114  fr_ldap_rcode_t status;
115  fr_ldap_connection_t *conn;
116  fr_ldap_config_t const *handle_config = instance; /* Not talloced */
117 
118  conn = fr_ldap_connection_alloc(ctx);
119  if (!conn) return NULL;
120 
121  if (fr_ldap_connection_configure(conn, handle_config) < 0) {
122  talloc_free(conn);
123  return NULL;
124  }
125 
126  fr_ldap_connection_timeout_set(conn, timeout);
127  if (handle_config->start_tls) {
128  if (ldap_start_tls_s(conn->handle, NULL, NULL) != LDAP_SUCCESS) {
129  int ldap_errno;
130 
131  ldap_get_option(conn->handle, LDAP_OPT_ERROR_NUMBER, &ldap_errno);
132 
133  ERROR("Could not start TLS: %s", ldap_err2string(ldap_errno));
134 
135  error:
136  talloc_free(conn);
137 
138  return NULL;
139  }
140  }
141 
142  status = fr_ldap_bind(NULL,
143  &conn,
144  conn->config->admin_identity, conn->config->admin_password,
145  &(conn->config->admin_sasl),
146  timeout,
147  NULL, NULL);
148  if (status != LDAP_PROC_SUCCESS) goto error;
149  fr_ldap_connection_timeout_reset(conn);
150 
151  /*
152  * Only error out on memory allocation errors
153  */
154  if (fr_ldap_directory_alloc(conn, &conn->directory, &conn) < 0) goto error;
155 
156  return conn;
157 }
