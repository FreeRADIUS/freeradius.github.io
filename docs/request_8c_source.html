47  *

52 static void request_init(char const *file, int line, request_t *request)
53 {

55  request->magic = REQUEST_MAGIC;
56 #endif
57 
58  request->request_state = REQUEST_INIT;
59  request->master_state = REQUEST_ACTIVE;
60 
61  /*
62  * Initialise the stack
63  */
64  MEM(request->stack = unlang_interpret_stack_alloc(request));
65 
66  /*
67  * Initialise the request data list
68  */
69  request_data_list_init(&request->data);
70 
71  /*
72  * Initialise the state_ctx
73  */
74  if (!request->state_ctx) {
75  request->state_ctx = talloc_init_const("session-state");
76 #ifndef NDEBUG

78 #endif

83  */

85  if (!request->log.dst) {
86  request->log.dst = talloc_zero(request, log_dst_t);
87  } else {
88  memset(request->log.dst, 0, sizeof(*request->log.dst));
89  }
90  request->log.dst->func = vlog_request;
91  request->log.dst->uctx = &default_log;

93  request->module = NULL;
94  request->component = "<core>";
95

99

102  */
103  request->alloc_file = file;

105

107 }

109 /** Callback for freeing a request struct
110  *
111  * @param[in] request to free or return to the free list.
112  * @return
113  * - 0 in the request was freed.
114  * - -1 if the request was inserted into the free list.
115  */
116 static int _request_free(request_t *request)
117 {
118  fr_assert(!request->ev);
119 
120  /*
121  * Reinsert into the free list if it's not already
122  * in the free list.
123  *
124  * If it *IS* already in the free list, then free it.
125  */
126  if (unlikely(fr_dlist_entry_in_list(&request->free_entry))) {
127  fr_dlist_entry_unlink(&request->free_entry); /* Don't trust the list head to be available */
128  goto really_free;
129  }
130 
131  /*
132  * We keep a buffer of <active> + N requests per
133  * thread, to avoid spurious allocations.
134  */
135  if (fr_dlist_num_elements(request_free_list) <= 256) {
136  TALLOC_CTX *state_ctx;
137  fr_dlist_head_t *free_list;
138 
139  /*
140  * Ensure any data associated
141  * with the state ctx is freed.
142  */
143  state_ctx = request->state_ctx;
144  if (request->state_ctx) {
145  fr_assert(!request->parent || (request->state_ctx != request->parent->state_ctx));
146  talloc_free_children(request->state_ctx);
147  }
148  free_list = request_free_list;
149 
150  /*
151  * Reinitialise the request
152  */
153  talloc_free_children(request);
154  memset(request, 0, sizeof(*request));
155  request->component = "free_list";
156  request->state_ctx = state_ctx; /* Use the old, now cleared, state_ctx */
157 
158  /*
159  * Reinsert into the free list
160  */
161  fr_dlist_insert_head(free_list, request);
162  request_free_list = free_list;
163 
164  return -1; /* Prevent free */
165  }
166 
167  /*
168  * Ensure anything that might reference the request is
169  * freed before it is.
170  */
171  talloc_free_children(request);
172 
173 really_free:
174  /*
175  * state_ctx is parented separately.
176  *
177  * The reason why it's OK to do this, is if the state attributes
178  * need to persist across requests, they will already have been
179  * moved to a fr_state_entry_t, with the state pointers in the
180  * request being set to NULL, before the request is freed.
181  *
182  * Our state ctx can be the same as the parents if
183  * request_data_restore_to_child() was called. That
184  * function resets the childs state_ctx to be the same as
185  * the parents. Adding this check here means that we
186  * don't need to call request_detach() on a child which
187  * will be freed immediately after the detach.
188  *
189  * Note also that we do NOT call TALLOC_FREE(), which
190  * sets state_ctx=NULL. We don't control the order in
191  * which talloc frees the children. And the parents
192  * state_ctx pointer needs to stick around so that all of
193  * the children can check it.
194  *
195  * If this assertion hits, it means that someone didn't
196  * call fr_state_store_in_parent()
197  */
198  if (request->state_ctx) {
199  fr_assert(!request->parent || (request->state_ctx != request->parent->state_ctx));
200  talloc_free(request->state_ctx);
201  }
202 
203 #ifndef NDEBUG
204  request->magic = 0x01020304; /* set the request to be nonsense */
205 #endif
206 
207  return 0;
208 }

225 /** Create a new request_t data structure
226  *
227  */
228 request_t *_request_alloc(char const *file, int line, TALLOC_CTX *ctx)
229 {
230  request_t *request;
231  fr_dlist_head_t *free_list;
232 
233  /*
234  * Setup the free list, or return the free
235  * list for this thread.
236  */
237  if (unlikely(!request_free_list)) {
238  MEM(free_list = talloc(NULL, fr_dlist_head_t));
239  fr_dlist_init(free_list, request_t, free_entry);
240  fr_thread_local_set_destructor(request_free_list, _request_free_list_free_on_exit, free_list);
241  } else {
242  free_list = request_free_list;
243  }
244 
245  request = fr_dlist_head(free_list);
246  if (!request) {
247  /*
248  * Only allocate requests in the NULL
249  * ctx. There's no scenario where it's
250  * appropriate to allocate them in a
251  * pool, and using a strict talloc
252  * hierarchy means that child requests
253  * cannot be returned to a free list
254  * and would have to be freed.
255  */
256  MEM(request = talloc_zero_pooled_object(NULL, request_t,
257  1 + /* Stack pool */
258  UNLANG_STACK_MAX + /* Stack Frames */
259  2 + /* packets */
260  10, /* extra */
261  (UNLANG_FRAME_PRE_ALLOC * UNLANG_STACK_MAX) + /* Stack memory */
262  (sizeof(fr_radius_packet_t) * 2) + /* packets */
263  128 /* extra */
264  ));
265  talloc_set_destructor(request, _request_free);
266  } else {
267  /*
268  * Remove from the free list, as we're
269  * about to use it!
270  */
271  fr_dlist_remove(free_list, request);
272  }
273 
274  request_init(file, line, request);
275 
276  /*
277  * Bind lifetime to a parent.
278  *
279  * If the parent is freed the destructor
280  * will fire, and return the request
281  * to a "top level" free list.
282  */
283  if (ctx) talloc_link_ctx(ctx, request);
284 
285  return request;
286 }

288 /** Allocate a request that's not in the free list
289  *
290  * This can be useful if modules need a persistent request for their own purposes
291  * which needs to be outside of the normal free list, so that it can be freed
292  * when the module requires, not when the thread destructor runs.
293  */
294 request_t *_request_local_alloc(char const *file, int line, TALLOC_CTX *ctx)
295 {
296  request_t *request;
297 
298  MEM(request = talloc_zero(ctx, request_t));
299 
300  request_init(file, line, request);
301 
302  /*
303  * Bind state_ctx to the request Apple
304  * Instruments seems to require this to not
305  * count the state_ctx as leaked.
306  *
307  * It's unclear if this is a genuine leak
308  * the previous code used a destructor
309  * to free the state_ctx.
310  */
311  talloc_steal(request, request->state_ctx);
312 
313  return request;
314 }

316 static request_t *request_init_fake(char const *file, int line, request_t *request, request_t *fake)
317 {
318  fake->number = request->child_number++;
319  fake->name = talloc_typed_asprintf(fake, "%s.%" PRIu64 , request->name, fake->number);
320 
321  fake->seq_start = 0; /* children always start with their own sequence */
322 
323  fake->parent = request;
324  fake->dict = request->dict;
325  fake->config = request->config;
326  fake->client = request->client;
327 
328  /*
329  * For new server support.
330  *
331  * FIXME: Key instead off of a "virtual server" data structure.
332  *
333  * FIXME: Permit different servers for inner && outer sessions?
334  */
335  fake->server_cs = request->server_cs;
336 
337  fake->packet = fr_radius_alloc(fake, true);
338  if (!fake->packet) {
339  talloc_free(fake);
340  return NULL;
341  }
342 
343  fake->reply = fr_radius_alloc(fake, false);
344  if (!fake->reply) {
345  talloc_free(fake);
346  return NULL;
347  }
348 
349  fake->master_state = REQUEST_ACTIVE;
350 
351  /*
352  * Fill in the fake request.
353  */
354  fake->packet->socket = request->packet->socket;
355  fake->packet->socket.inet.dst_port = 0;
356  fake->packet->socket.fd = -1;
357 
358  /*
359  * This isn't STRICTLY required, as the fake request MUST NEVER
360  * be put into the request list. However, it's still reasonable
361  * practice.
362  */
363  fake->packet->id = fake->number & 0xff;
364  fake->packet->code = request->packet->code;
365  fake->packet->timestamp = request->packet->timestamp;
366 
367  /*
368  * Required for new identity support
369  */
370  fake->listener = request->listener;
371 
372  /*
373  * Fill in the fake reply, based on the fake request.
374  */
375  fr_socket_addr_swap(&fake->reply->socket, &fake->packet->socket);
376  fake->reply->id = fake->packet->id;
377  fake->reply->code = 0; /* UNKNOWN code */
378 
379  /*
380  * Copy debug information.
381  */
382  memcpy(&(fake->log), &(request->log), sizeof(fake->log));
383  fake->log.unlang_indent = 0; /* Apart from the indent which we reset */
384  fake->log.module_indent = 0; /* Apart from the indent which we reset */
385 
386  /*
387  * Allocation information
388  */
389  fake->alloc_file = file;
390  fake->alloc_line = line;
391 
392  return fake;
393 }

421 request_t *_request_alloc_detachable(char const *file, int line, request_t *request, fr_dict_t const *namespace)

433  * Ensure that we use our own version of the logging

437  memcpy(fake->log.dst, request->log.dst, sizeof(*fake->log.dst));

439  /*

444  */

448  }
449 
450  return fake;

453 /** Unlink a subrequest from its parent
454  *
455  * @note This should be used for requests in preparation for freeing them.
456  *
457  * @param[in] fake request to unlink.
458  * @param[in] will_free Caller super pinky swears to free
459  * the request ASAP, and that it wont
460  * touch persistable request data,
461  * request->state_ctx or request->state.
462  * @return
463  * - 0 on success.
464  * - -1 on failure.
465  */
466 int request_detach(request_t *fake, bool will_free)
467 {
468  request_t *request = fake->parent;
469 
470  fr_assert(request != NULL);
471 
472  /*
473  * Unlink the child from the parent.
474  */
475  request_data_get(request, fake, 0);
476 
477  /*
478  * Fixup any sate or persistent
479  * request data.
480  */
481  fr_state_detach(fake, will_free);
482 
483  fake->parent = NULL;
484 
485  while (!request->backlog) {
486  fr_assert(request->parent != NULL);
487  request = request->parent;
488  }
489 
490  fake->backlog = request->backlog;
491 
492  return 0;
493 }
