32 /** Allocate a new fr_radius_packet_t
33  *
34  * @param ctx the context in which the packet is allocated. May be NULL if
35  * the packet is not associated with a request_t.
36  * @param new_vector if true a new request authenticator will be generated.
37  * @return
38  * - New fr_radius_packet_t.
39  * - NULL on error.
40  */
41 fr_radius_packet_t *fr_radius_alloc(TALLOC_CTX *ctx, bool new_vector)
42 {
43  fr_radius_packet_t *rp;
44 
45  rp = talloc_zero(ctx, fr_radius_packet_t);
46  if (!rp) {
47  fr_strerror_printf("out of memory");
48  return NULL;
49  }
50  rp->id = -1;
51 
52  if (new_vector) {
53  fr_rand_buffer(rp->vector, sizeof(rp->vector));
54  }
55 
56  return rp;
57 }

59 /** Allocate a new fr_radius_packet_t response
60  *
61  * @param ctx the context in which the packet is allocated. May be NULL if
62  * the packet is not associated with a request_t.
63  * @param packet The request packet.
64  * @return
65  * - New fr_radius_packet_t.
66  * - NULL on error.
67  */
68 fr_radius_packet_t *fr_radius_alloc_reply(TALLOC_CTX *ctx, fr_radius_packet_t *packet)
69 {
70  fr_radius_packet_t *reply;
71 
72  if (!packet) return NULL;
73 
74  reply = fr_radius_alloc(ctx, false);
75  if (!reply) return NULL;
76 
77  /*
78  * Initialize the fields from the request.
79  */
80  fr_socket_addr_swap(&reply->socket, &packet->socket);
81  reply->id = packet->id;
82  reply->code = 0; /* UNKNOWN code */
83  memset(reply->vector, 0,
84  sizeof(reply->vector));
85  reply->vps = NULL;
86  reply->data = NULL;
87  reply->data_len = 0;
88 
89  return reply;
90 }
