47 /** Resolve a named service to a port
48  *
49  * @param[in] proto The protocol. Either IPPROTO_TCP or IPPROTO_UDP.
50  * @param[in] port_name The service name, i.e. "radius".
51  * @return
52  * - > 0 the port port_name resolves to.
53  * - < 0 on error.
54  */
55 static int socket_port_from_service(int proto, char const *port_name)
56 {
57  struct servent *service;
58  char const *proto_name;
59 
60  if (!port_name) {
61  fr_strerror_printf("No port specified");
62  return -1;
63  }
64 
65  switch (proto) {
66  case IPPROTO_UDP:
67  proto_name = "udp";
68  break;
69 
70  case IPPROTO_TCP:
71  proto_name = "tcp";
72  break;
73 
74 #ifdef IPPROTO_SCTP
75  case IPPROTO_SCTP:
76  proto_name = "sctp";
77  break;
78 #endif
79 
80  default:
81  fr_strerror_printf("Unrecognised proto %i", proto);
82  return -1;
83  }
84 
85  service = getservbyname(port_name, proto_name);
86  if (!service) {
87  fr_strerror_printf("Unknown service %s", port_name);
88  return -1;
89  }
90 
91  return ntohs(service->s_port);
92 }

222  *

231  error:

235  }
236  //Optionally, if blocking operation is required
237  if (fr_blocking(sockfd) < 0) goto error;

241  * @param async Whether to set the socket to nonblocking, allowing use of
242  * #fr_socket_wait_for_connect.
243  * @return
244  * - Socket FD on success.
245  * - -1 on failure.
246  */
247 int fr_socket_client_unix(char const *path, bool async)
248 {

250  size_t len;
251  socklen_t socklen;
252  struct sockaddr_un saremote;
253

262  if (sockfd < 0) {
263  fr_strerror_printf("Failed creating UNIX socket: %s", fr_syserror(errno));
264  return -1;
265  }
266 
267  if (async && (fr_nonblock(sockfd) < 0)) {
268  close(sockfd);
269  return -1;
270  }
271 
272  saremote.sun_family = AF_UNIX;
273  memcpy(saremote.sun_path, path, len + 1); /* SUN_LEN does strlen */
274 
275  socklen = SUN_LEN(&saremote);
276 
277  /*
278  * Although we ignore SIGPIPE, some operating systems
279  * like BSD and OSX ignore the ignoring.
280  *
281  * Fortunately, those operating systems usually support
282  * SO_NOSIGPIPE, to prevent them raising the signal in
283  * the first place.
284  */
285 #ifdef SO_NOSIGPIPE
286  {
287  int set = 1;
288 
289  setsockopt(sockfd, SOL_SOCKET, SO_NOSIGPIPE, (void *)&set, sizeof(int));
290  }
291 #endif
292 
293  if (connect(sockfd, (struct sockaddr *)&saremote, socklen) < 0) {
294  /*
295  * POSIX says the only time we will get this,
296  * is if the socket has been marked as
297  * nonblocking. This is not an error, the caller
298  * must check the state of errno, and wait for
299  * the connection to complete.
300  */
301  if (errno == EINPROGRESS) return sockfd;
302 
303  close(sockfd);
304  fr_strerror_printf("Failed connecting to %s: %s", path, fr_syserror(errno));
305 
306  return -1;
307  }
308  return sockfd;
309 }

331  error:

335  }
336  //Optionally, if blocking operation is required
337  if (fr_blocking(sockfd) < 0) goto error;
338  @endcode
339  *
340  * @param[in,out] src_ipaddr to bind socket to, may be NULL if socket is not bound to any specific
341  * address. If non-null, the bound IP is copied here, too.
342  * @param[out] src_port The source port we were bound to, may be NULL.
343  * @param dst_ipaddr Where to send datagrams.
344  * @param dst_port Where to send datagrams.
345  * @param async Whether to set the socket to nonblocking, allowing use of
346  * #fr_socket_wait_for_connect.
347  * @return
348  * - FD on success.
349  * - -1 on failure.

351 int fr_socket_client_udp(fr_ipaddr_t *src_ipaddr, uint16_t *src_port, fr_ipaddr_t const *dst_ipaddr, uint16_t dst_port, bool async)
352 {
353  int sockfd;
354  struct sockaddr_storage salocal;
355  socklen_t salen;
356 
357  if (!dst_ipaddr) return -1;
358 
359  sockfd = socket(dst_ipaddr->af, SOCK_DGRAM, 0);
360  if (sockfd < 0) {
361  fr_strerror_printf("Error creating UDP socket: %s", fr_syserror(errno));
362  return -1;
363  }
364 
365  if (async && (fr_nonblock(sockfd) < 0)) {
366  error:
367  close(sockfd);
368  return -1;
369  }
370 
371  /*
372  * Allow the caller to bind us to a specific source IP.
373  */
374  if (src_ipaddr && (src_ipaddr->af != AF_UNSPEC)) {
375  /*

377  */

379 
380  if (fr_ipaddr_to_sockaddr(&salocal, &salen, src_ipaddr, 0) < 0) {
381  close(sockfd);

383  }
384 
385  if (bind(sockfd, (struct sockaddr *) &salocal, salen) < 0) {
386  fr_strerror_printf("Failure binding to IP: %s", fr_syserror(errno));
387  close(sockfd);
388  return -1;
389  }
390  }

394  * like BSD and OSX ignore the ignoring.
395  *
396  * Fortunately, those operating systems usually support
397  * SO_NOSIGPIPE, to prevent them raising the signal in
398  * the first place.
399  */
400 #ifdef SO_NOSIGPIPE

412  */
413  if (src_port) {

419  if (getsockname(sockfd, (struct sockaddr *) &salocal, &salen) < 0) {

423  }
424

427  */
428  if (!src_ipaddr) src_ipaddr = &my_ipaddr;
429  if (!src_port) src_port = &my_port;
430 
431  if (fr_ipaddr_from_sockaddr(src_ipaddr, src_port, &salocal, salen) < 0) {
432  close(sockfd);
433  return -1;
434  }
435  }
436 
437  /*
438  * And now get our destination
439  */
440  if (fr_ipaddr_to_sockaddr(&salocal, &salen, dst_ipaddr, dst_port) < 0) {

442  return -1;

446  /*
447  * POSIX says the only time we will get this,
448  * is if the socket has been marked as
449  * nonblocking. This is not an error, the caller
450  * must check the state of errno, and wait for
451  * the connection to complete.
452  */
453  if (errno == EINPROGRESS) return sockfd;
454 
455  fr_strerror_printf("Failed connecting socket: %s", fr_syserror(errno));
456  close(sockfd);
457  return -1;
458  }
459 
460  return sockfd;
461 }

566 /** Wait for a socket to be connected, with an optional timeout
567  *
568  * @note On error the caller is expected to ``close(sockfd)``.
569  *
570  * @param sockfd the socket to wait on.
571  * @param timeout How long to wait for socket to open.
572  * @return
573  * - 0 on success.
574  * - -1 on connection error.
575  * - -2 on timeout.
576  * - -3 on select error.
577  */
578 int fr_socket_wait_for_connect(int sockfd, fr_time_delta_t timeout)
579 {
580  int ret;
581  fd_set error_set;
582  fd_set write_set; /* POSIX says sockets are open when they become writable */
583 
584  FD_ZERO(&error_set);
585  FD_ZERO(&write_set);
586 
587  FD_SET(sockfd, &error_set);
588  FD_SET(sockfd, &write_set);
589 
590  /* Don't let signals mess up the select */
591  do {
592  ret = select(sockfd + 1, NULL, &write_set, &error_set, &fr_time_delta_to_timeval(timeout));
593  } while ((ret == -1) && (errno == EINTR));
594 
595  switch (ret) {
596  case 1: /* ok (maybe) */
597  {
598  int error;
599  socklen_t socklen = sizeof(error);
600 
601  if (getsockopt(sockfd, SOL_SOCKET, SO_ERROR, (void *)&error, &socklen)) {
602  fr_strerror_printf("Failed connecting socket: %s", fr_syserror(errno));
603  return -1;
604  }
605 
606  if (FD_ISSET(sockfd, &error_set)) {
607  fr_strerror_printf("Failed connecting socket: Unknown error");
608  return -1;
609  }
610  }
611  return 0;
612 
613  case 0: /* timeout */
614  if (!fr_cond_assert(timeout)) return -1;
615  fr_strerror_printf("Connection timed out after %pVs", fr_box_time_delta(timeout));
616  return -2;
617 
618  case -1: /* select error */
619  fr_strerror_printf("Failed waiting for connection: %s", fr_syserror(errno));
620  return -3;
621 
622  default:
623  (void)fr_cond_assert(0);
624  return -1;
625  }
626 }

628 /** Open an IPv4/IPv6 unconnected UDP socket
629  *
630  * Function name is a bit of a misnomer as it can also be used to create client sockets too,
631  * such is the nature of UDP.
632  *
633  * @param[in] src_ipaddr The IP address to listen on
634  * @param[in,out] src_port the port to listen on. If *port == 0, the resolved
635  * service port will be written here.
636  * @param[in] port_name if *port == 0, the name of the port
637  * @param[in] async whether we block or not on reads and writes
638  * @return
639  * - Socket FD on success.
640  * - -1 on failure.
641  */
642 int fr_socket_server_udp(fr_ipaddr_t const *src_ipaddr, uint16_t *src_port, char const *port_name, bool async)
643 {
644  int sockfd;
645  uint16_t my_port = 0;
646 
647  if (src_port) my_port = *src_port;
648 
649  /*
650  * Check IP looks OK
651  */
652  if (!src_ipaddr || ((src_ipaddr->af != AF_INET) && (src_ipaddr->af != AF_INET6))) {
653  fr_strerror_printf("No address specified");
654  return -1;
655  }
656 
657  /*
658  * Check we have a port value or stuff we can resolve to a port
659  */
660  if (!my_port && port_name) {
661  int ret;
662 
663  ret = socket_port_from_service(IPPROTO_UDP, port_name);
664  if (ret < 0) return -1;
665 
666  my_port = ret;
667  }
668 
669  /*
670  * Open the socket
671  */
672  sockfd = socket(src_ipaddr->af, SOCK_DGRAM, IPPROTO_UDP);
673  if (sockfd < 0) {
674  fr_strerror_printf("Failed creating UDP socket: %s", fr_syserror(errno));
675  return -1;
676  }
677 
678  /*
679  * Make it non-blocking if asked
680  */
681  if (async && (fr_nonblock(sockfd) < 0)) {
682  error:
683  close(sockfd);
684  return -1;
685  }
686 
687  /*
688  * Don't allow child processes to inherit the socket
689  */
690  if (socket_dont_inherit(sockfd) < 0) goto error;
691 
692  /*
693  * Initialize udpfromto for UDP sockets.
694  */
695  if (udpfromto_init(sockfd) != 0) {
696  fr_strerror_printf("Failed initializing udpfromto: %s", fr_syserror(errno));
697  goto error;
698  }
699 
700  /*
701  * Make sure we don't get v4 and v6 packets on inaddr_any sockets.
702  */
703  if (socket_inaddr_any_v6only(sockfd, src_ipaddr) < 0) goto error;
704 
705  /*
706  * Ensure don't fragment bit is set
707  */

709

713

716  * when the packet was received, not when it was read
717  * from the socket.

719  if (setsockopt(sockfd, SOL_SOCKET, SO_TIMESTAMP, &on, sizeof(int)) < 0) {
720  close(sockfd);

722  return -1;
723  }

732 /** Open an IPv4/IPv6 TCP socket
733  *
734  * @param[in] src_ipaddr The IP address to listen on
735  * @param[in,out] src_port the port to listen on. If *port == 0, the resolved
736  * service port will be written here.
737  * NULL if any port is allowed.
738  * @param[in] port_name if *port == 0, the name of the port
739  * @param[in] async whether we block or not on reads and writes
740  * @return
741  * - Socket FD on success.
742  * - -1 on failure.
743  */
744 int fr_socket_server_tcp(fr_ipaddr_t const *src_ipaddr, uint16_t *src_port, char const *port_name, bool async)
745 {
746  int sockfd;
747  uint16_t my_port = 0;
748 
749  if (src_port) my_port = *src_port;
750 
751  /*
752  * Check IP looks OK
753  */
754  if (!src_ipaddr || ((src_ipaddr->af != AF_INET) && (src_ipaddr->af != AF_INET6))) {
755  fr_strerror_printf("No address specified");
756  return -1;
757  }
758 
759  /*
760  * Check we have a port value or stuff we can resolve to a port
761  */
762  if (!my_port && port_name) {
763  int ret;
764 
765  ret = socket_port_from_service(IPPROTO_TCP, port_name);
766  if (ret < 0) return -1;
767 
768  my_port = ret;
769  }
770 
771  /*
772  * Open the socket
773  */
774  sockfd = socket(src_ipaddr->af, SOCK_STREAM, IPPROTO_TCP);
775  if (sockfd < 0) {
776  fr_strerror_printf("Failed creating TCP socket: %s", fr_syserror(errno));
777  return -1;
778  }
779 
780  /*
781  * Make it non-blocking if asked
782  */
783  if (async && (fr_nonblock(sockfd) < 0)) {
784  error:
785  close(sockfd);
786  return -1;
787  }
788 
789  /*
790  * Don't allow child processes to inherit the socket
791  */
792  if (socket_dont_inherit(sockfd) < 0) goto error;
793 
794  /*
795  * Make sure we don't get v4 and v6 packets on inaddr_any sockets.
796  */
797  if (socket_inaddr_any_v6only(sockfd, src_ipaddr) < 0) goto error;
798 
799  {
800  int on = 1;
801 
802  if (setsockopt(sockfd, SOL_SOCKET, SO_REUSEADDR, &on, sizeof(on)) < 0) {
803  close(sockfd);
804  fr_strerror_printf("Failed to reuse address: %s", fr_syserror(errno));
805  return -1;
806  }
807  }
808 
809  if (src_port) *src_port = my_port;
810 
811  return sockfd;
812 }

814 /** Bind a UDP/TCP v4/v6 socket to a given ipaddr src port, and interface.
815  *
816  * Use one of:
817  * - fr_socket_client_udp - for a connected socket.
818  * - fr_socket_server_udp - for non-connected socket.
819  * - fr_socket_server_tcp
820  * ...to open a file descriptor, then call this function to bind the socket to an IP address.
821  *
822  * @param[in] sockfd the socket which opened by fr_socket_server_*.
823  * @param[in,out] src_ipaddr The IP address to bind to. NULL to just bind to an interface.
824  * @param[in] src_port the port to bind to. NULL if any port is allowed.
825  * @param[in] interface to bind to.
826  * @return
827  * - 0 on success
828  * - -1 on failure.
829  */
830 int fr_socket_bind(int sockfd, fr_ipaddr_t const *src_ipaddr, uint16_t *src_port, char const *interface)
831 {
832  int ret;
833  uint16_t my_port = 0;
834  fr_ipaddr_t my_ipaddr;
835  struct sockaddr_storage salocal;
836  socklen_t salen;
837 
838  /*
839  * Clear the thread local error stack as we may
840  * push multiple errors onto the stack, and this
841  * is likely to be the function which returns
842  * the "original" error.
843  */
844  (void)fr_strerror();
845 
846  if (src_port) my_port = *src_port;
847  if (src_ipaddr) {
848  my_ipaddr = *src_ipaddr;
849  } else {
850  my_ipaddr = (fr_ipaddr_t) {
851  .af = AF_UNSPEC
852  };
853  }
854 
855 #ifdef HAVE_CAPABILITY_H
856  /*
857  * If we're binding to a special port as non-root, then
858  * check capabilities. If we're root, we already have
859  * equivalent capabilities so we don't need to check.
860  */
861  if (src_port && (*src_port < 1024) && (geteuid() != 0)) {
862  (void)fr_cap_enable(CAP_NET_BIND_SERVICE, CAP_EFFECTIVE);
863  }
864 #endif
865 
866  /*
867  * Bind to a device BEFORE touching IP addresses.
868  */
869  if (interface) {
870 #ifdef HAVE_NET_IF_H
871  uint32_t scope_id;
872 
873  scope_id = if_nametoindex(interface);
874  if (!scope_id) {
875  fr_strerror_printf_push("Failed finding interface %s: %s",
876  interface, fr_syserror(errno));
877  return -1;
878  }
879 
880  /*
881  * If the scope ID hasn't already been set, then
882  * set it. This allows us to get the scope from the interface name.
883  */
884  if ((my_ipaddr.scope_id != 0) && (scope_id != my_ipaddr.scope_id)) {
885  fr_strerror_printf_push("Cannot bind to interface %s: Socket is already bound "
886  "to another interface", interface);
887  return -1;
888  }
889 #endif
890 
891 #ifdef SO_BINDTODEVICE
892  /*
893  * The caller didn't specify a scope_id, but we
894  * have one from above. Call "bind to device",
895  * and set the scope_id.
896  */
897  if (!my_ipaddr.scope_id) {
898  ret = setsockopt(sockfd, SOL_SOCKET, SO_BINDTODEVICE, interface, strlen(interface));
899  if (ret < 0) {
900  fr_strerror_printf_push("Bind failed on interface %s: %s",
901  interface, fr_syserror(errno));
902  return -1;
903  } /* else it worked. */
904 
905  /*
906  * Set the scope ID.
907  */
908  my_ipaddr.scope_id = scope_id;
909  }
910  /*
911  * SO_BINDTODEVICE succeeded, so we're always
912  * bound to the socket.
913  */
914 
915 #else
916  struct ifaddrs *list = NULL;
917  bool bound = false;
918 
919  /*
920  * Troll through all interfaces to see if there's
921  */
922  if (getifaddrs(&list) == 0) {
923  struct ifaddrs *i;
924 
925  for (i = list; i != NULL; i = i->ifa_next) {
926  if (i->ifa_addr && i->ifa_name && (strcmp(i->ifa_name, interface) == 0)) {
927  /*
928  * IPv4, and there's either no src_ip, OR src_ip is INADDR_ANY,
929  * it's a match.
930  *
931  * We also update my_ipaddr to point to this particular IP,
932  * so that we can later bind() to it. This gets us the same
933  * effect as SO_BINDTODEVICE.
934  */
935  if ((i->ifa_addr->sa_family == AF_INET) &&
936  (!src_ipaddr || fr_ipaddr_is_inaddr_any(src_ipaddr))) {
937  (void) fr_ipaddr_from_sockaddr(&my_ipaddr, NULL,
938  (struct sockaddr_storage *) i->ifa_addr,
939  sizeof(struct sockaddr_in));
940  my_ipaddr.scope_id = scope_id;
941  bound = true;
942  break;
943  }
944 
945  /*
946  * The caller specified a source IP, and we find a matching
947  * address family. Allow it.
948  *
949  * Note that we do NOT check for matching IPs here. If we did,
950  * then binding to an interface and the *wrong* IP would get us
951  * a "bind to device is unsupported" message.
952  *
953  * Instead we say "yes, we found a matching interface", and then
954  * allow the bind() call below to run. If that fails, we get a
955  * "Can't assign requested address" error, which is more informative.
956  */
957  if (src_ipaddr && (src_ipaddr->af == i->ifa_addr->sa_family)) {
958  my_ipaddr.scope_id = scope_id;
959  bound = true;
960  break;
961  }
962  }
963  }
964 
965  freeifaddrs(list);
966 
967  if (!bound) {
968  /*
969  * IPv4: no link local addresses,
970  * and no bind to device.
971  */
972  fr_strerror_printf_push("Bind to interface %s failed: Unable to match interface with the given IP address.",
973  interface);
974  return -1;
975  }
976  } else {
977  fr_strerror_printf_push("Bind to interface %s failed, unable to get list of interfaces: %s",
978  interface, fr_syserror(errno));
979  return -1;
980  }
981 #endif
982  } /* else no interface was passed in */
983 
984  /*
985  * Don't bind to an IP address if there's no src IP address.
986  */
987  if (my_ipaddr.af == AF_UNSPEC) goto done;
988 
989  /*
990  * Set up sockaddr stuff.
991  */
992  if (fr_ipaddr_to_sockaddr(&salocal, &salen, &my_ipaddr, my_port) < 0) return -1;
993 
994  ret = bind(sockfd, (struct sockaddr *) &salocal, salen);
995  if (ret < 0) {
996  fr_strerror_printf_push("Bind failed with source address %pV:%pV on interface %s: %s",
997  src_ipaddr ? fr_box_ipaddr(*src_ipaddr) : fr_box_strvalue("*"),
998  src_port ? fr_box_int16(*src_port) : fr_box_strvalue("*"),
999  interface ? interface : "*",
1000  fr_syserror(errno));
1001  return ret;
1002  }
1003 
1004  if (!src_port) goto done;
1005 
1006  /*
1007  * FreeBSD jail issues. We bind to 0.0.0.0, but the
1008  * kernel instead binds us to a 1.2.3.4. So once the
1009  * socket is bound, ask it what it's IP address is.
1010  *
1011  * @todo - Uh... we don't update src_ipaddr with the new
1012  * IP address. This means that we don't tell the caller
1013  * what IP address we're bound to. That seems wrong.
1014  */
1015  salen = sizeof(salocal);
1016  memset(&salocal, 0, salen);
1017  if (getsockname(sockfd, (struct sockaddr *) &salocal, &salen) < 0) {
1018  fr_strerror_printf_push("Failed getting socket name: %s", fr_syserror(errno));
1019  return -1;
1020  }
1021 
1022  if (fr_ipaddr_from_sockaddr(&my_ipaddr, &my_port, &salocal, salen) < 0) return -1;
1023  *src_port = my_port;
1024 
1025 done:
1026 #ifdef HAVE_CAPABILITY_H
1027  /*
1028  * Clear any errors we may have produced in the
1029  * capabilities check.
1030  */
1031  fr_strerror();
1032 #endif
1033  return 0;
1034 }
