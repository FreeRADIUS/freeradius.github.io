44 /*
45  * Run a virtual server auth and postauth
46  *
47  */
48 unlang_action_t rad_virtual_server(rlm_rcode_t *p_result, request_t *request)
49 {
50  fr_pair_t *vp, *username, *parent_username = NULL;
51  rlm_rcode_t final;
52 
53  RDEBUG("Virtual server %s received request", cf_section_name2(request->server_cs));
54  log_request_pair_list(L_DBG_LVL_1, request, request->request_pairs, NULL);
55 
56  username = fr_pair_find_by_num(&request->request_pairs, 0, FR_STRIPPED_USER_NAME);
57  if (!username) username = fr_pair_find_by_num(&request->request_pairs, 0, FR_USER_NAME);
58 
59  if (request->parent) {
60  parent_username = fr_pair_find_by_num(&request->parent->request_pairs, 0, FR_STRIPPED_USER_NAME);
61  if (!parent_username) parent_username = fr_pair_find_by_num(&request->parent->request_pairs, 0, FR_USER_NAME);
62  }
63 
64  /*
65  * Complain about possible issues related to tunnels.
66  */
67  if (username && parent_username) {
68  /*
69  * Look at the full User-Name with realm.
70  */
71  if (parent_username->da->attr == FR_STRIPPED_USER_NAME) {
72  vp = fr_pair_find_by_num(&request->parent->request_pairs, 0, FR_USER_NAME);
73  if (!vp) goto runit;
74  } else {
75  vp = parent_username;
76  }
77 
78  /*
79  * If the names aren't identical, we do some detailed checks.
80  */
81  if (strcmp(vp->vp_strvalue, username->vp_strvalue) != 0) {
82  char const *outer, *inner;
83 
84  outer = strchr(vp->vp_strvalue, '@');
85 
86  /*
87  * If there's no realm, or there's a user identifier before
88  * the realm name, check the user identifier.
89  *
90  * It SHOULD be "anonymous", or "anonymous@realm"
91  */
92  if (outer) {
93  if ((outer != vp->vp_strvalue) &&
94  ((vp->vp_length < 10) || (memcmp(vp->vp_strvalue, "anonymous@", 10) != 0))) {
95  RWDEBUG("Outer User-Name is not anonymized. User privacy is compromised.");
96  } /* else it is anonymized */
97 
98  /*
99  * Check when there's no realm, and without the trailing '@'
100  */
101  } else if ((vp->vp_length < 9) || (memcmp(vp->vp_strvalue, "anonymous", 9) != 0)) {
102  RWDEBUG("Outer User-Name is not anonymized. User privacy is compromised.");
103 
104  } /* else the user identifier is anonymized */
105 
106  /*
107  * Look for an inner realm, which may or may not exist.
108  */
109  inner = strchr(username->vp_strvalue, '@');
110  if (outer && inner) {
111  outer++;
112  inner++;
113 
114  /*
115  * The realms are different, do
116  * more detailed checks.
117  */
118  if (strcmp(outer, inner) != 0) {
119  size_t outer_len, inner_len;
120 
121  outer_len = vp->vp_length;
122  outer_len -= (outer - vp->vp_strvalue);
123 
124  inner_len = username->vp_length;
125  inner_len -= (inner - username->vp_strvalue);
126 
127  /*
128  * Inner: secure.example.org
129  * Outer: example.org
130  */
131  if (inner_len > outer_len) {
132  char const *suffix;
133 
134  suffix = inner + (inner_len - outer_len) - 1;
135 
136  if ((*suffix != '.') ||
137  (strcmp(suffix + 1, outer) != 0)) {
138  RWDEBUG("Possible spoofing: Inner realm '%s' is not a "
139  "subdomain of the outer realm '%s'", inner, outer);
140  }
141 
142  } else {
143  RWDEBUG("Possible spoofing: Inner realm and "
144  "outer realms are different");
145  }
146  }
147  }
148 
149  } else {
150  RWDEBUG("Outer and inner identities are the same. User privacy is compromised.");
151  }
152  }
153 
154 runit:
155  if (!request->async) {
156 #ifdef __clang_analyzer__
157  if (!request->parent) RETURN_MODULE_FAIL;
158 #endif
159  fr_assert(request->parent != NULL);
160 
161  request->async = talloc_memdup(request, request->parent->async, sizeof(fr_async_t));
162  talloc_set_name_const(request->async, talloc_get_name(request->parent->async));
163  }
164 
165  RDEBUG("server %s {", cf_section_name2(request->server_cs));
166  request->async->process(&final, &(module_ctx_t){ .instance = request->async->process_inst }, request);
167  RDEBUG("} # server %s", cf_section_name2(request->server_cs));
168 
169  fr_cond_assert(final == RLM_MODULE_OK);
170 
171  if (!request->reply->code ||
172  (request->reply->code == FR_CODE_ACCESS_REJECT)) {
173  RETURN_MODULE_REJECT;
174  }
175 
176  if (request->reply->code == FR_CODE_ACCESS_CHALLENGE) {
177  RETURN_MODULE_HANDLED;
178  }
179 
180  RETURN_MODULE_OK;
181 }
