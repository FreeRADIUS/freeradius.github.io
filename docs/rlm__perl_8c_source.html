149 static void **rlm_perl_get_handles(pTHX)
150 {
151  I32 i;
152  AV *librefs = get_av(dl_librefs, false);
153  AV *modules = get_av(dl_modules, false);
154  void **handles;
155 
156  if (!librefs) return NULL;
157 
158  if (!(AvFILL(librefs) >= 0)) {
159  return NULL;
160  }
161 
162  MEM(handles = talloc_array(NULL, void *, AvFILL(librefs) + 2));
163  for (i = 0; i <= AvFILL(librefs); i++) {
164  void *handle;
165  SV *handle_sv = *av_fetch(librefs, i, false);
166  if (!handle_sv) {
167  ERROR("Could not fetch $%s[%d]!", dl_librefs, (int)i);
168  continue;
169  }
170  handle = (void *)SvIV(handle_sv);
171 
172  if (handle) handles[i] = handle;
173  }
174 
175  av_clear(modules);
176  av_clear(librefs);
177 
178  handles[i] = (void *)0;
179 
180  return handles;
181 }

183 static void rlm_perl_close_handles(void **handles)

185  int i;
186 
187  if (!handles) {

192  DEBUG("Close %p", handles[i]);
193  dlclose(handles[i]);
194  }
195 
196  talloc_free(handles);
197 }

351 /** Call perl code using an xlat
352  *
353  * @ingroup xlat_functions
354  */
355 static ssize_t perl_xlat(UNUSED TALLOC_CTX *ctx, char **out, size_t outlen,
356  void const *mod_inst, UNUSED void const *xlat_inst,
357  request_t *request, char const *fmt)
358 {
359 
360  rlm_perl_t *inst;
361  char *tmp;
362  char const *p, *q;
363  int count;
364  size_t ret = 0;
365  STRLEN n_a;
366 
367  memcpy(&inst, &mod_inst, sizeof(inst));
368 
369 #ifdef USE_ITHREADS
370  PerlInterpreter *interp;
371 
372  pthread_mutex_lock(&inst->clone_mutex);
373  interp = rlm_perl_clone(inst->perl, inst->thread_key);
374  {
375  dTHXa(interp);
376  PERL_SET_CONTEXT(interp);
377  }
378  pthread_mutex_unlock(&inst->clone_mutex);
379 #else
380  PERL_SET_CONTEXT(inst->perl);
381 #endif
382  {
383  dSP;
384  ENTER;SAVETMPS;
385 
386  PUSHMARK(SP);
387 
388  p = q = fmt;
389  while (*p == ' ') {
390  p++;
391  q++;
392  }
393  while (*q) {
394  if (*q == ' ') {
395  XPUSHs(sv_2mortal(newSVpvn(p, q - p)));
396  p = q + 1;
397 
398  /*
399  * Don't use an empty string
400  */
401  while (*p == ' ') p++;
402  q = p;
403  }
404  q++;
405  }
406 
407  /*
408  * And the last bit.
409  */
410  if (*p) {
411  XPUSHs(sv_2mortal(newSVpvn(p, strlen(p))));
412  }
413 
414  PUTBACK;
415 
416  count = call_pv(inst->func_xlat, G_SCALAR | G_EVAL);
417 
418  SPAGAIN;
419  if (SvTRUE(ERRSV)) {
420  REDEBUG("Exit %s", SvPV(ERRSV,n_a));
421  (void)POPs;
422  } else if (count > 0) {
423  tmp = POPp;
424  strlcpy(*out, tmp, outlen);
425  ret = strlen(*out);
426 
427  RDEBUG2("Len is %zu , out is %s freespace is %zu", ret, *out, outlen);
428  }
429 
430  PUTBACK ;
431  FREETMPS ;
432  LEAVE ;
433 
434  }
435 
436  return ret;
437 }

439 /*
440  * Parse a configuration section, and populate a HV.
441  * This function is recursively called (allows to have nested hashes.)
442  */
443 static void perl_parse_config(CONF_SECTION *cs, int lvl, HV *rad_hv)
444 {
445  if (!cs || !rad_hv) return;
446 
447  int indent_section = (lvl + 1) * 4;
448  int indent_item = (lvl + 2) * 4;
449 
450  DEBUG("%*s%s {", indent_section, " ", cf_section_name1(cs));
451 
452  CONF_ITEM *ci = NULL;
453 
454  while ((ci = cf_item_next(cs, ci))) {
455  /*
456  * This is a section.
457  * Create a new HV, store it as a reference in current HV,
458  * Then recursively call perl_parse_config with this section and the new HV.
459  */
460  if (cf_item_is_section(ci)) {
461  CONF_SECTION *sub_cs = cf_item_to_section(ci);
462  char const *key = cf_section_name1(sub_cs); /* hash key */
463  HV *sub_hv;
464  SV *ref;
465 
466  if (!key) continue;
467 
468  if (hv_exists(rad_hv, key, strlen(key))) {
469  WARN("Ignoring duplicate config section '%s'", key);
470  continue;
471  }
472 
473  sub_hv = newHV();
474  ref = newRV_inc((SV*) sub_hv);
475 
476  (void)hv_store(rad_hv, key, strlen(key), ref, 0);
477 
478  perl_parse_config(sub_cs, lvl + 1, sub_hv);
479  } else if (cf_item_is_pair(ci)){
480  CONF_PAIR *cp = cf_item_to_pair(ci);
481  char const *key = cf_pair_attr(cp); /* hash key */
482  char const *value = cf_pair_value(cp); /* hash value */
483 
484  if (!key || !value) continue;
485 
486  /*
487  * This is an item.
488  * Store item attr / value in current HV.
489  */
490  if (hv_exists(rad_hv, key, strlen(key))) {
491  WARN("Ignoring duplicate config item '%s'", key);
492  continue;
493  }
494 
495  (void)hv_store(rad_hv, key, strlen(key), newSVpvn(value, strlen(value)), 0);
496 
497  DEBUG("%*s%s = %s", indent_item, " ", key, value);
498  }
499  }
500 
501  DEBUG("%*s}", indent_section, " ");
502 }

518 /*
519  * Do any per-module initialization that is separate to each
520  * configured instance of the module. e.g. set up connections
521  * to external databases, read configuration files, set up
522  * dictionary entries, etc.
523  *
524  * If configuration information is given in the config section
525  * that must be referenced in later calls, store a handle to it
526  * in *instance otherwise put a null pointer there.
527  *
528  * Setup a hashes wich we will use later
529  * parse a module and give it a chance to live
530  *
531  */
532 static int mod_instantiate(void *instance, CONF_SECTION *conf)
533 {
534  rlm_perl_t *inst = instance;
535  AV *end_AV;
536 
537  char const **embed_c; /* Stupid Perl and lack of const consistency */
538  char **embed;
539  char **envp = NULL;
540  int exitstatus = 0, argc=0;
541  char arg[] = "0";
542 
543  CONF_SECTION *cs;
544 
545 #ifdef USE_ITHREADS
546  /*
547  * Create pthread key. This key will be stored in instance
548  */
549  pthread_mutex_init(&inst->clone_mutex, NULL);
550 
551  MEM(inst->thread_key = talloc_zero(inst, pthread_key_t));
552  rlm_perl_make_key(inst->thread_key);
553 #endif
554 
555  /*
556  * Setup the argument array we pass to the perl interpreter
557  */
558  MEM(embed_c = talloc_zero_array(inst, char const *, 4));
559  memcpy(&embed, &embed_c, sizeof(embed));
560  embed_c[0] = NULL;
561  if (inst->perl_flags) {
562  embed_c[1] = inst->perl_flags;
563  embed_c[2] = inst->module;
564  embed_c[3] = arg;
565  argc = 4;
566  } else {
567  embed_c[1] = inst->module;
568  embed_c[2] = arg;
569  argc = 3;
570  }
571 
572  /*
573  * Create tweak the server's environment to support
574  * perl. Docs say only call this once... Oops.
575  */
576  if (!perl_sys_init3_called) {
577  PERL_SYS_INIT3(&argc, &embed, &envp);
578  perl_sys_init3_called = 1;
579  }
580 
581  /*
582  * Allocate a new perl interpreter to do the parsing
583  */
584  if ((inst->perl = perl_alloc()) == NULL) {
585  ERROR("No memory for allocating new perl interpretor!");
586  return -1;
587  }
588  perl_construct(inst->perl); /* ...and initialise it */
589 
590 #ifdef USE_ITHREADS
591  PL_perl_destruct_level = 2;
592 
593  {
594  dTHXa(inst->perl);
595  }
596  PERL_SET_CONTEXT(inst->perl);
597 #endif
598 
599 #if PERL_REVISION >= 5 && PERL_VERSION >=8
600  PL_exit_flags |= PERL_EXIT_DESTRUCT_END;
601 #endif
602 
603  exitstatus = perl_parse(inst->perl, xs_init, argc, embed, NULL);
604 
605  end_AV = PL_endav;
606  PL_endav = (AV *)NULL;
607 
608  if (exitstatus) {
609  ERROR("Perl_parse failed: %s not found or has syntax errors", inst->module);
610  return -1;
611  }
612 
613  /* parse perl configuration sub-section */
614  cs = cf_section_find(conf, "config", NULL);
615  if (cs) {
616  inst->rad_perlconf_hv = get_hv("RAD_PERLCONF", 1);
617  perl_parse_config(cs, 0, inst->rad_perlconf_hv);
618  }
619 
620  inst->perl_parsed = true;
621  perl_run(inst->perl);
622 
623  PL_endav = end_AV;
624 
625  return 0;
626 }

628 static void perl_vp_to_svpvn_element(request_t *request, AV *av, fr_pair_t const *vp,
629  int *i, const char *hash_name, const char *list_name)
630 {
631 
632  SV *sv;
633 
634  switch (vp->vp_type) {
635  case FR_TYPE_STRING:
636  RDEBUG2("$%s{'%s'}[%i] = &%s.%s -> '%s'", hash_name, vp->da->name, *i,
637  list_name, vp->da->name, vp->vp_strvalue);
638  sv = newSVpvn(vp->vp_strvalue, vp->vp_length);
639  break;
640 
641  case FR_TYPE_OCTETS:
642  RDEBUG2("$%s{'%s'}[%i] = &%s.%s -> 0x%pH", hash_name, vp->da->name, *i,
643  list_name, vp->da->name, &vp->data);
644  sv = newSVpvn((char const *)vp->vp_octets, vp->vp_length);
645  break;
646 
647  default:
648  {
649  char buffer[1024];
650  ssize_t slen;
651 
652  slen = fr_pair_print_value_quoted(&FR_SBUFF_OUT(buffer, sizeof(buffer)), vp, T_BARE_WORD);
653  if (slen < 0) return;
654 
655  RDEBUG2("$%s{'%s'}[%i] = &%s.%s -> '%pV'", hash_name, vp->da->name, *i,
656  list_name, vp->da->name, fr_box_strvalue_len(buffer, (size_t)slen));
657  sv = newSVpvn(buffer, (size_t)slen);
658  }
659  break;
660  }
661 
662  if (!sv) return;
663  SvTAINT(sv);
664  av_push(av, sv);
665  (*i)++;

668 /*

674 static void perl_store_vps(UNUSED TALLOC_CTX *ctx, request_t *request, fr_pair_t **vps, HV *rad_hv,
675  const char *hash_name, const char *list_name)
676 {

680

682 
683  RINDENT();
684  fr_pair_list_sort(vps, fr_pair_cmp_by_da);
685  for (vp = fr_cursor_init(&cursor, vps);
686  vp;
687  vp = fr_cursor_next(&cursor)) {
688  fr_pair_t *next;
689  char const *name;
690  name = vp->da->name;
691 
692  /*
693  * We've sorted by type, then tag, so attributes of the
694  * same type/tag should follow on from each other.
695  */
696  if ((next = fr_cursor_next_peek(&cursor)) && ATTRIBUTE_EQ(vp, next)) {
697  int i = 0;

751 static int pairadd_sv(TALLOC_CTX *ctx, request_t *request, fr_pair_t **vps, char *key, SV *sv, fr_token_t op,
752  const char *hash_name, const char *list_name)
753 {
