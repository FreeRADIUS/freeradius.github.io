93 /** Ensure contents are quoted correctly for a JSON document
94  *
95  * @ingroup xlat_functions
96  *
97  * @param ctx talloc context
98  * @param out Where to write the output
99  * @param request The current request.
100  * @param xlat_inst unused

102  * @param in list of value boxes as input
103  * @return XLAT_ACTION_DONE or XLAT_ACTION_FAIL
104  */
105 static xlat_action_t json_quote_xlat(TALLOC_CTX *ctx, fr_cursor_t *out, request_t *request,
106  UNUSED void const *xlat_inst, UNUSED void *xlat_thread_inst,
107  fr_value_box_t **in)
108 {
109  fr_value_box_t *vb;
110  char *tmp;
111 
112  if (!in) {
113  REDEBUG("Nothing to quote");
114  return XLAT_ACTION_FAIL;
115  }
116 
117  if (fr_value_box_list_concat(ctx, *in, in, FR_TYPE_STRING, true) < 0) {
118  RPEDEBUG("Failed concatenating input");
119  return XLAT_ACTION_FAIL;
120  }
121 
122  MEM(vb = fr_value_box_alloc_null(ctx));
123 
124  if (!(tmp = fr_json_from_string(vb, (*in)->vb_strvalue, false))) {

127  return XLAT_ACTION_FAIL;
128  }
129  fr_value_box_bstrdup_buffer_shallow(NULL, vb, NULL, tmp, false);
130 
131  fr_cursor_append(out, vb);
132 
133  return XLAT_ACTION_DONE;

136 /** Determine if a jpath expression is valid
137  *
138  * @ingroup xlat_functions
139  *
140  * @param ctx to allocate expansion buffer in.
141  * @param mod_inst data.
142  * @param xlat_inst data.
143  * @param out Where to write the output (in the format @verbatim<bytes parsed>[:error]@endverbatim).
144  * @param outlen How big out is.
145  * @param request The current request.
146  * @param fmt jpath expression to parse.
147  * @return number of bytes written to out.
148  */
149 static ssize_t jpath_validate_xlat(UNUSED TALLOC_CTX *ctx, char **out, size_t outlen,
150  UNUSED void const *mod_inst, UNUSED void const *xlat_inst,
151  request_t *request, char const *fmt)
152 {
153  fr_jpath_node_t *head;
154  ssize_t slen, ret;
155  char *jpath_str;
156 
157  slen = fr_jpath_parse(request, &head, fmt, strlen(fmt));
158  if (slen <= 0) {
159  fr_assert(head == NULL);
160  return snprintf(*out, outlen, "%zd:%s", -(slen), fr_strerror());
161  }
162  fr_assert(talloc_get_type_abort(head, fr_jpath_node_t));
163 
164  jpath_str = fr_jpath_asprint(request, head);
165  ret = snprintf(*out, outlen, "%zu:%s", (size_t) slen, jpath_str);
166  talloc_free(head);
167  talloc_free(jpath_str);
168 
169  return ret;
170 }

172 /** Convert given attributes to a JSON document
173  *
174  * Usage is `%{json_encode:attr tmpl list}`
175  *
176  * @ingroup xlat_functions
177  *
178  * @param ctx talloc context
179  * @param out where to write the output
180  * @param request the current request
181  * @param xlat_inst xlat instance data
182  * @param xlat_thread_inst unused
183  * @param in list of value boxes as input
184  * @return XLAT_ACTION_DONE or XLAT_ACTION_FAIL
185  */
186 static xlat_action_t json_encode_xlat(TALLOC_CTX *ctx, fr_cursor_t *out, request_t *request,
187  void const *xlat_inst, UNUSED void *xlat_thread_inst,
188  fr_value_box_t **in)
189 {
190  rlm_json_t const *inst = talloc_get_type_abort_const(*((void const * const *)xlat_inst),
191  rlm_json_t);
192  fr_json_format_t const *format = inst->format;
193 
194  ssize_t slen;
195  tmpl_t *vpt = NULL;
196  fr_cursor_t cursor;
197  fr_cursor_t filter;
198  fr_pair_t *json_vps = NULL;
199  fr_pair_t *vps = NULL;
200  bool negate;
201  char *json_str = NULL;
202  fr_value_box_t *vb;
203  fr_sbuff_t sbuff;
204 
205  if (!in) {
206  REDEBUG("Missing attribute(s)");
207  return XLAT_ACTION_FAIL;
208  }
209 
210  if (fr_value_box_list_concat(ctx, *in, in, FR_TYPE_STRING, true) < 0) {
211  RPEDEBUG("Failed concatenating input");
212  return XLAT_ACTION_FAIL;
213  }
214 
215  sbuff = FR_SBUFF_IN((*in)->vb_strvalue, (*in)->vb_length);
216  fr_sbuff_adv_past_whitespace(&sbuff, SIZE_MAX);
217 
218  /*
219  * Iterate through the list of attribute templates in the xlat. For each
220  * one we either add it to the list of attributes for the JSON document
221  * or, if prefixed with '!', remove from the JSON list.
222  */
223  while (fr_sbuff_extend(&sbuff)) {
224  negate = false;
225 
226  /* Check if we should be removing attributes */
227  if (fr_sbuff_next_if_char(&sbuff, '!')) negate = true;
228 
229  /* Decode next attr template */
230  slen = tmpl_afrom_attr_substr(ctx, NULL, &vpt,
231  &sbuff,
232  &json_arg_parse_rules,
233  &(tmpl_rules_t){ .dict_def = request->dict });
234  if (slen <= 0) {
235  fr_sbuff_set(&sbuff, (size_t)(slen * -1));
236  REMARKER(fr_sbuff_start(&sbuff), fr_sbuff_used(&sbuff), "%s", fr_strerror());
237  error:
238  fr_pair_list_free(&json_vps);
239  talloc_free(vpt);
240  return XLAT_ACTION_FAIL;
241  }
242 
243  /*
244  * Get attributes from the template.
245  * Missing attribute isn't an error (so -1, not 0).
246  */
247  if (tmpl_copy_pairs(ctx, &vps, request, vpt) < -1) {
248  RPEDEBUG("Error copying attributes");
249  goto error;
250  }
251 
252  if (negate) {
253  /* Remove all template attributes from JSON list */
254  for (fr_pair_t *vp = fr_cursor_init(&filter, &vps);
255  vp;
256  vp = fr_cursor_next(&filter)) {
257 
258  fr_pair_t *vpm = fr_cursor_init(&cursor, &json_vps);
259  while (vpm) {
260  if (vp->da == vpm->da) {
261  talloc_free(fr_cursor_remove(&cursor));
262  vpm = fr_cursor_current(&cursor);
263  continue;
264  }
265  vpm = fr_cursor_next(&cursor);
266  }
267  }
268 
269  fr_pair_list_free(&vps);
270  } else {
271  /* Add template VPs to JSON list */
272  fr_pair_add(&json_vps, vps);
273  }
274 
275  TALLOC_FREE(vpt);
276 
277  /* Jump forward to next attr */
278  fr_sbuff_adv_past_whitespace(&sbuff, SIZE_MAX);
279  }
280 
281  /*
282  * Given the list of attributes we now have in json_vps,
283  * convert them into a JSON document and append it to the
284  * return cursor.
285  */
286  MEM(vb = fr_value_box_alloc_null(ctx));
287 
288  json_str = fr_json_afrom_pair_list(vb, json_vps, format);
289  if (!json_str) {
290  REDEBUG("Failed to generate JSON string");
291  goto error;
292  }
293  fr_value_box_bstrdup_buffer_shallow(NULL, vb, NULL, json_str, false);
294 
295  fr_cursor_append(out, vb);
296  fr_pair_list_free(&json_vps);
297 
298  return XLAT_ACTION_DONE;
299 }

302 /** Pre-parse and validate literal jpath expressions for maps
303  *
304  * @param[in] cs #CONF_SECTION that defined the map instance.
305  * @param[in] mod_inst module instance (unused).
306  * @param[in] proc_inst the cache structure to fill.
307  * @param[in] src Where to get the JSON data from.
308  * @param[in] maps set of maps to translate to jpaths.
309  * @return
310  * - 0 on success.
311  * - -1 on failure.
312  */
313 static int mod_map_proc_instantiate(CONF_SECTION *cs, UNUSED void *mod_inst, void *proc_inst,
314  tmpl_t const *src, map_t const *maps)
315 {
316  rlm_json_jpath_cache_t *cache_inst = proc_inst;
317  map_t const *map;
318  ssize_t slen;
319  rlm_json_jpath_cache_t *cache = cache_inst, **tail = &cache->next;
320 
321  if (!src) {
322  cf_log_err(cs, "Missing JSON source");
323 
324  return -1;
325  }
326 
327  for (map = maps; map; map = map->next) {
328  CONF_PAIR *cp = cf_item_to_pair(map->ci);
329  char const *p;
330 
331 #ifndef HAVE_JSON_OBJECT_GET_INT64
332  if (tmpl_is_attr(map->lhs) && (tmpl_da(map->lhs)->type == FR_TYPE_UINT64)) {
333  cf_log_err(cp, "64bit integers are not supported by linked json-c. "
334  "Upgrade to json-c >= 0.10 to use this feature");
335  return -1;
336  }
337 #endif
338 
339  switch (map->rhs->type) {
340  case TMPL_TYPE_UNRESOLVED:
341  p = map->rhs->name;
342  slen = fr_jpath_parse(cache, &cache->jpath, p, map->rhs->len);
343  if (slen <= 0) {
344  char *spaces, *text;
345 
346  error:
347  fr_canonicalize_error(cache, &spaces, &text, slen, fr_strerror());
348 
349  cf_log_err(cp, "Syntax error");
350  cf_log_err(cp, "%s", p);
351  cf_log_err(cp, "%s^ %s", spaces, text);
352 
353  talloc_free(spaces);
354  talloc_free(text);
355  return -1;
356  }
357  break;
358 
359  case TMPL_TYPE_DATA:
360  if (tmpl_value_type(map->rhs) != FR_TYPE_STRING) {
361  cf_log_err(cp, "Right side of map must be a string");
362  return -1;
363  }
364  p = tmpl_value(map->rhs)->vb_strvalue;
365  slen = fr_jpath_parse(cache, &cache->jpath, p, tmpl_value_length(map->rhs));
366  if (slen <= 0) goto error;
367  break;
368 
369  default:
370  continue;
371  }
372 
373  /*
374  * Slightly weird... This is here because our first
375  * list member was pre-allocated and passed to the
376  * instantiation callback.
377  */
378  if (map->next) {
379  *tail = cache = talloc_zero(cache, rlm_json_jpath_cache_t);
380  tail = &cache->next;
381  }
382  }
383 
384  return 0;
385 }

387 /** Converts a string value into a #fr_pair_t
388  *
389  * @param[in,out] ctx to allocate #fr_pair_t (s).
390  * @param[out] out where to write the resulting #fr_pair_t.
391  * @param[in] request The current request.
392  * @param[in] map to process.
393  * @param[in] uctx The json tree/jpath expression to evaluate.
394  * @return
395  * - 0 on success.
396  * - -1 on failure.
397  */
398 static int _json_map_proc_get_value(TALLOC_CTX *ctx, fr_pair_t **out, request_t *request,
399  map_t const *map, void *uctx)
400 {
401  fr_pair_t *vp;
402  fr_cursor_t cursor;
403  rlm_json_jpath_to_eval_t *to_eval = uctx;
404  fr_value_box_t *head, *value;
405  int ret;
406 
407  *out = NULL;
408 
409  ret = fr_jpath_evaluate_leaf(request, &head, tmpl_da(map->lhs)->type, tmpl_da(map->lhs),
410  to_eval->root, to_eval->jpath);
411  if (ret < 0) {
412  RPEDEBUG("Failed evaluating jpath");
413  return -1;
414  }
415  if (ret == 0) return 0;
416  fr_assert(head);
417 
418  for (fr_cursor_init(&cursor, out), value = head;
419  value;
420  fr_cursor_append(&cursor, vp), value = value->next) {
421  MEM(vp = fr_pair_afrom_da(ctx, tmpl_da(map->lhs)));
422  vp->op = map->op;
423 
424  if (fr_value_box_steal(vp, &vp->data, value) < 0) {
425  RPEDEBUG("Copying data to attribute failed");
426  talloc_free(vp);
427  talloc_free(*out);
428  return -1;
429  }
430  }
431  return 0;
432 }

434 /** Parses a JSON string, and executes jpath queries against it to map values to attributes
435  *
436  * @param mod_inst unused.
437  * @param proc_inst cached jpath sequences.
438  * @param request The current request.
439  * @param json JSON string to parse.
440  * @param maps Head of the map list.
441  * @return
442  * - #RLM_MODULE_NOOP no rows were returned or columns matched.
443  * - #RLM_MODULE_UPDATED if one or more #fr_pair_t were added to the #request_t.
444  * - #RLM_MODULE_FAIL if a fault occurred.
445  */
446 static rlm_rcode_t mod_map_proc(UNUSED void *mod_inst, void *proc_inst, request_t *request,
447  fr_value_box_t **json, map_t const *maps)
448 {
449  rlm_rcode_t rcode = RLM_MODULE_UPDATED;
450  struct json_tokener *tok;
451 
452  rlm_json_jpath_cache_t *cache = proc_inst;
453  map_t const *map;
454 
455  rlm_json_jpath_to_eval_t to_eval;
456 
457  char const *json_str = NULL;
458 
459  if (!*json) {
460  REDEBUG("JSON map input cannot be (null)");
461  return RLM_MODULE_FAIL;
462  }
463 
464  if (fr_value_box_list_concat(request, *json, json, FR_TYPE_STRING, true) < 0) {
465  REDEBUG("Failed concatenating input");
466  return RLM_MODULE_FAIL;
467  }
468  json_str = (*json)->vb_strvalue;
469 
470  if ((talloc_array_length(json_str) - 1) == 0) {
471  REDEBUG("JSON map input length must be > 0");
472  return RLM_MODULE_FAIL;
473  }
474 
475  tok = json_tokener_new();
476  to_eval.root = json_tokener_parse_ex(tok, json_str, (int)(talloc_array_length(json_str) - 1));
477  if (!to_eval.root) {
478  REMARKER(json_str, tok->char_offset, "%s", json_tokener_error_desc(json_tokener_get_error(tok)));
479  rcode = RLM_MODULE_FAIL;
480  goto finish;
481  }
482 
483  for (map = maps; map; map = map->next) {
484  switch (map->rhs->type) {
485  /*
486  * Cached types
487  */
488  case TMPL_TYPE_UNRESOLVED:
489  case TMPL_TYPE_DATA:
490  to_eval.jpath = cache->jpath;
491 
492  if (map_to_request(request, map, _json_map_proc_get_value, &to_eval) < 0) {
493  rcode = RLM_MODULE_FAIL;
494  goto finish;
495  }
496  cache = cache->next;
497  break;
498 
499  /*
500  * Dynamic types
501  */
502  default:
503  {
504  ssize_t slen;
505  fr_jpath_node_t *node;
506  char *to_parse;
507 
508  if (tmpl_aexpand(request, &to_parse, request, map->rhs, fr_jpath_escape_func, NULL) < 0) {
509  RPERROR("Failed getting jpath data");
510  rcode = RLM_MODULE_FAIL;
511  goto finish;
512  }
513  slen = fr_jpath_parse(request, &node, to_parse, talloc_array_length(to_parse) - 1);
514  if (slen <= 0) {
515  REMARKER(to_parse, -(slen), "%s", fr_strerror());
516  talloc_free(to_parse);
517  rcode = RLM_MODULE_FAIL;
518  goto finish;
519  }
520  to_eval.jpath = node;
521 
522  if (map_to_request(request, map, _json_map_proc_get_value, &to_eval) < 0) {
523  talloc_free(node);
524  talloc_free(to_parse);
525  rcode = RLM_MODULE_FAIL;
526  goto finish;
527  }
528  talloc_free(node);
529  }
530  break;
531  }
532  }
533 
534 
535 finish:
536  json_object_put(to_eval.root);
537  json_tokener_free(tok);
538 
539  return rcode;
540 }

542 static int mod_bootstrap(void *instance, CONF_SECTION *conf)
543 {
544  rlm_json_t *inst = talloc_get_type_abort(instance, rlm_json_t);
545  xlat_t const *xlat;
546  char *name;
547  fr_json_format_t *format = inst->format;
548 
549  inst->name = cf_section_name2(conf);
550  if (!inst->name) inst->name = cf_section_name1(conf);
551 
552  xlat_register(instance, "jsonquote", json_quote_xlat, false);
553  xlat_register_legacy(instance, "jpathvalidate", jpath_validate_xlat, NULL, NULL, 0, XLAT_DEFAULT_BUF_LEN);
554 
555  name = talloc_asprintf(inst, "%s_encode", inst->name);
556  xlat = xlat_register(instance, name, json_encode_xlat, false);
557  xlat_async_instantiate_set(xlat, json_xlat_instantiate,
558  rlm_json_t *, NULL, inst);
559  talloc_free(name);
560 
561  /*
562  * Check the output format type and warn on unused
563  * format options
564  */
565  format->output_mode = fr_table_value_by_str(fr_json_format_table, format->output_mode_str, JSON_MODE_UNSET);
566  if (format->output_mode == JSON_MODE_UNSET) {
567  cf_log_err(conf, "output_mode value \"%s\" is invalid", format->output_mode_str);
568  return -1;
569  }
570  fr_json_format_verify(format, true);
571 
572  if (map_proc_register(instance, "json", mod_map_proc,
573  mod_map_proc_instantiate, sizeof(rlm_json_jpath_cache_t)) < 0) return -1;
574  return 0;
575 }
