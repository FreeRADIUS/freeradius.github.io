17 /**
18  * $Id: 9e7a8de172a740ec47c006f994e0d1c4a27b7b6b $
19  *
20  * @brief Multi-packet state handling
21  * @file src/lib/server/state.c
22  *
23  * @ingroup AVP
24  *
25  * For each round of a multi-round authentication method such as EAP,
26  * or a 2FA method such as OTP, a state entry will be created. The state
27  * entry holds data that should be available during the complete lifecycle
28  * of the authentication attempt.
29  *
30  * When a request is complete, #fr_request_to_state is called to transfer
31  * ownership of the state fr_pair_ts and state_ctx (which the fr_pair_ts
32  * are allocated in) to a #fr_state_entry_t. This #fr_state_entry_t holds the
33  * value of the State attribute, that will be send out in the response.
34  *
35  * When the next request is received, #fr_state_to_request is called to transfer
36  * the fr_pair_ts and state ctx to the new request.
37  *
38  * The ownership of the state_ctx and state fr_pair_ts is transferred as below:
39  *
40  * @verbatim
41  request -> state_entry -> request -> state_entry -> request -> free()
42  \-> reply \-> reply \-> access-reject/access-accept
43  * @endverbatim
44  *
45  * @copyright 2014 The FreeRADIUS server project
46  */

58 /** Holds a state value, and associated fr_pair_ts and data
59  *
60  */
61 typedef struct {
62  uint64_t id; //!< State number within state tree.
63  union {
64  /** Server ID components
65  *
66  * State values should be unique to a given server
67  */
68  struct state_comp {
69  uint8_t tries; //!< Number of rounds so far in this state sequence.
70  uint8_t tx; //!< Bits changed in the tries counter for this round.
71  uint8_t r_0; //!< Random component.
72  uint8_t server_id; //!< Configured server ID. Used for debugging
73  //!< to locate authentication sessions originating
74  //!< from a particular backend authentication server.
75 
76  uint32_t server_hash; //!< Hash of the current virtual server, xor'd with
77  //!< r1, r2, r3, r4 after the original state value
78  //!< is sent, but before the state entry is inserted
79  //!< into the tree. The receiving virtual server
80  //!< xor's its hash with the received state before
81  //!< performing the lookup. This means one virtual
82  //!< server can't act on a state entry generated by
83  //!< another, even though the state tree is global
84  //!< to all virtual servers.
85 
86  uint8_t vx_0; //!< Random component.
87  uint8_t r_5; //!< Random component.
88  uint8_t vx_1; //!< Random component.
89  uint8_t r_6; //!< Random component.
90 
91  uint8_t vx_2; //!< Random component.
92  uint8_t r_7; //!< Random component.
93  uint8_t r_8; //!< Random component.
94  uint8_t r_9; //!< Random component.
95  } state_comp;
96 
97  uint8_t state[sizeof(struct state_comp)]; //!< State value in binary.
98  };
99 
100  uint64_t seq_start; //!< Number of first request in this sequence.
101  time_t cleanup; //!< When this entry should be cleaned up.
102  fr_dlist_t list; //!< Entry in the list of things to expire.
103 
104  int tries;
105 
106  TALLOC_CTX *ctx; //!< ctx to parent any data that needs to be
107  //!< tied to the lifetime of the request progression.
108  fr_pair_t *vps; //!< session-state fr_pair_ts, parented by ctx.
109 
110  fr_dlist_head_t data; //!< Persistable request data, also parented by ctx.
111 
112  request_t *thawed; //!< The request that thawed this entry.
113 } fr_state_entry_t;

142 {

148 /** Free the state tree
149  *
150  */
151 static int _state_tree_free(fr_state_tree_t *state)
152 {
153  fr_state_entry_t *entry;
154 
155  if (state->thread_safe) pthread_mutex_destroy(&state->mutex);
156 
157  DEBUG4("Freeing state tree %p", state);
158 
159  while ((entry = fr_dlist_head(&state->to_expire))) {
160  DEBUG4("Freeing state entry %p (%"PRIu64")", entry, entry->id);
161  state_entry_unlink(state, entry);
162  talloc_free(entry);
163  }
164 
165  /*
166  * Free the rbtree
167  */
168  talloc_free(state->tree);
169 
170  return 0;
171 }

173 /** Initialise a new state tree
174  *
175  * @param[in] ctx to link the lifecycle of the state tree to.
176  * @param[in] da Attribute used to store and retrieve state from.
177  * @param[in] thread_safe Whether we should mutex protect the state tree.
178  * @param[in] max_sessions we track state for.
179  * @param[in] timeout How long to wait before cleaning up entries.
180  * @param[in] server_id ID byte to use in load-balancing operations.
181  * @return
182  * - A new state tree.
183  * - NULL on failure.
184  */
185 fr_state_tree_t *fr_state_tree_init(TALLOC_CTX *ctx, fr_dict_attr_t const *da, bool thread_safe,
186  uint32_t max_sessions, uint32_t timeout, uint8_t server_id)
187 {
188  fr_state_tree_t *state;
189 
190  state = talloc_zero(NULL, fr_state_tree_t);
191  if (!state) return 0;
192 
193  state->max_sessions = max_sessions;
194  state->timeout = timeout;
195 
196  /*
197  * Create a break in the contexts.
198  * We still want this to be freed at the same time
199  * as the parent, but we also need it to be thread
200  * safe, and multiple threads could be using the
201  * tree.
202  */
203  talloc_link_ctx(ctx, state);
204 
205  if (thread_safe && (pthread_mutex_init(&state->mutex, NULL) != 0)) {
206  talloc_free(state);
207  return NULL;
208  }
209 
210  fr_dlist_talloc_init(&state->to_expire, fr_state_entry_t, list);
211 
212  /*
213  * We need to do controlled freeing of the
214  * rbtree, so that all the state entries
215  * are freed before it's destroyed. Hence
216  * it being parented from the NULL ctx.
217  */
218  state->tree = rbtree_talloc_alloc(NULL, state_entry_cmp, fr_state_entry_t, NULL, 0);
219  if (!state->tree) {
220  talloc_free(state);
221  return NULL;
222  }
223  talloc_set_destructor(state, _state_tree_free);
224 
225  state->da = da; /* Remember which attribute we use to load/store state */
226  state->server_id = server_id;
227  state->thread_safe = thread_safe;
228 
229  return state;
230 }

233  *
234  */
235 static void state_entry_unlink(fr_state_tree_t *state, fr_state_entry_t *entry)
236 {

239  */
240  (void) talloc_get_type_abort(entry, fr_state_entry_t);
241 
242  fr_dlist_remove(&state->to_expire, entry);

245

249 /** Frees any data associated with a state
250  *
251  */
252 static int _state_entry_free(fr_state_entry_t *entry)
253 {
254 #ifdef WITH_VERIFY_PTR
255  fr_cursor_t cursor;
256  fr_pair_t *vp;
257 
258  /*
259  * Verify all state attributes are parented
260  * by the state context.
261  */
262  if (entry->ctx) {
263  for (vp = fr_cursor_init(&cursor, &entry->vps);
264  vp;
265  vp = fr_cursor_next(&cursor)) {
266  fr_assert(entry->ctx == talloc_parent(vp));
267  }
268  }
269 
270  /*
271  * Ensure any request data is parented by us
272  * so we know it'll be cleaned up.
273  */
274  (void)fr_cond_assert(request_data_verify_parent(entry->ctx, &entry->data));
275 #endif
276 
277  /*
278  * Should also free any state attributes
279  */
280  if (entry->ctx) TALLOC_FREE(entry->ctx);
281 
282  DEBUG4("State ID %" PRIu64 " freed", entry->id);
283 
284  return 0;
285 }

287 /** Create a new state entry
288  *
289  * @note Called with the mutex held.
290  */
291 static fr_state_entry_t *state_entry_create(fr_state_tree_t *state, request_t *request,
292  fr_radius_packet_t *packet, fr_state_entry_t *old)
293 {
294  size_t i;
295  uint32_t x;
296  time_t now = time(NULL);
297  fr_pair_t *vp;
298  fr_state_entry_t *entry, *next;
299 
300  uint8_t old_state[sizeof(old->state)];
301  int old_tries = 0;
302  uint64_t timed_out = 0;
303  bool too_many = false;
304  fr_dlist_head_t to_free;
305 
306  fr_dlist_init(&to_free, fr_state_entry_t, list);
307 
308  /*
309  * Clean up old entries.
310  */
311  for (entry = fr_dlist_head(&state->to_expire);
312  entry != NULL;
313  entry = next) {
314  (void)talloc_get_type_abort(entry, fr_state_entry_t); /* Allow examination */
315  next = fr_dlist_next(&state->to_expire, entry); /* Advance *before* potential unlinking */
316 
317  if (entry == old) continue;
318 
319  /*
320  * Too old, we can delete it.
321  */
322  if (entry->cleanup < now) {
323  state_entry_unlink(state, entry);
324  fr_dlist_insert_tail(&to_free, entry);
325  timed_out++;
326  continue;
327  }
328 
329  break;
330  }
331 
332  state->timed_out += timed_out;
333 
334  if (!old && (rbtree_num_elements(state->tree) >= (uint32_t) state->max_sessions)) too_many = true;
335 
336  /*
337  * Record the information from the old state, we may base the
338  * new state off the old one.
339  *
340  * Once we release the mutex, the state of old becomes indeterminate
341  * so we have to grab the values now.
342  */
343  if (old) {
344  old_tries = old->tries;
345 
346  memcpy(old_state, old->state, sizeof(old_state));
347 
348  /*
349  * The old one isn't used any more, so we can free it.
350  */
351  if (fr_dlist_empty(&old->data)) {
352  state_entry_unlink(state, old);
353  fr_dlist_insert_tail(&to_free, old);
354  }
355  }
356  PTHREAD_MUTEX_UNLOCK(&state->mutex);
357 
358  if (timed_out > 0) RWDEBUG("Cleaning up %"PRIu64" timed out state entries", timed_out);
359 
360  /*
361  * Now free the unlinked entries.
362  *
363  * We do it here as freeing may involve significantly more
364  * work than just freeing the data.
365  *
366  * If there's request data that was persisted it will now
367  * be freed also, and it may have complex destructors associated
368  * with it.
369  */
370  while ((entry = fr_dlist_head(&to_free)) != NULL) {
371  fr_dlist_remove(&to_free, entry);
372  talloc_free(entry);
373  }
374 
375  /*
376  * Have to do this post-cleanup, else we end up returning with
377  * a list full of entries to free with none of them being
378  * freed which is bad...
379  */
380  if (too_many) {
381  RERROR("Failed inserting state entry - At maximum ongoing session limit (%u)",
382  state->max_sessions);
383  PTHREAD_MUTEX_LOCK(&state->mutex); /* Caller expects this to be locked */
384  return NULL;
385  }
386 
387  /*
388  * Allocation doesn't need to occur inside the critical region
389  * and would add significantly to contention.
390  */
391  entry = talloc_zero(NULL, fr_state_entry_t);
392  if (!entry) {
393  PTHREAD_MUTEX_LOCK(&state->mutex); /* Caller expects this to be locked */
394  return NULL;
395  }
396 
397  request_data_list_init(&entry->data);
398  talloc_set_destructor(entry, _state_entry_free);
399  entry->id = state->id++;
400 
401  /*
402  * Limit the lifetime of this entry based on how long the
403  * server takes to process a request. Doing it this way
404  * isn't perfect, but it's reasonable, and it's one less
405  * thing for an administrator to configure.
406  */
407  entry->cleanup = now + state->timeout;
408 
409  /*
410  * Some modules create their own magic
411  * state attributes. If a state value already exists
412  * int the reply, we use that in preference to the
413  * old state.
414  */
415  vp = fr_pair_find_by_da(&packet->vps, state->da);
416  if (vp) {
417  if (DEBUG_ENABLED && (vp->vp_length > sizeof(entry->state))) {
418  WARN("State too long, will be truncated. Expected <= %zd bytes, got %zu bytes",
419  sizeof(entry->state), vp->vp_length);
420  }
421 
422  /*
423  * Assume our own State first.
424  */
425  if (vp->vp_length == sizeof(entry->state)) {
426  memcpy(entry->state, vp->vp_octets, sizeof(entry->state));
427 
428  /*
429  * Too big? Get the MD5 hash, in order
430  * to depend on the entire contents of State.
431  */
432  } else if (vp->vp_length > sizeof(entry->state)) {
433  fr_md5_calc(entry->state, vp->vp_octets, vp->vp_length);
434 
435  /*
436  * Too small? Use the whole thing, and
437  * set the rest of my_entry.state to zero.
438  */
439  } else {
440  memcpy(entry->state, vp->vp_octets, vp->vp_length);
441  memset(&entry->state[vp->vp_length], 0, sizeof(entry->state) - vp->vp_length);
442  }
443  } else {
444  /*
445  * 16 octets of randomness should be enough to
446  * have a globally unique state.
447  */
448  if (old) {
449  memcpy(entry->state, old_state, sizeof(entry->state));
450  entry->tries = old_tries + 1;
451  /*
452  * Base the new state on the old state if we had one.
453  */
454  } else {
455  for (i = 0; i < sizeof(entry->state) / sizeof(x); i++) {
456  x = fr_rand();
457  memcpy(entry->state + (i * 4), &x, sizeof(x));
458  }
459  }
460 
461  entry->state_comp.tries = entry->tries + 1;
462 
463  entry->state_comp.tx = entry->state_comp.tries ^ entry->tries;
464 
465  entry->state_comp.vx_0 = entry->state_comp.r_0 ^
466  ((((uint32_t) HEXIFY(RADIUSD_VERSION)) >> 16) & 0xff);
467  entry->state_comp.vx_1 = entry->state_comp.r_0 ^
468  ((((uint32_t) HEXIFY(RADIUSD_VERSION)) >> 8) & 0xff);
469  entry->state_comp.vx_2 = entry->state_comp.r_0 ^
470  (((uint32_t) HEXIFY(RADIUSD_VERSION)) & 0xff);
471 
472  /*
473  * Allow a portion of the State attribute to be set,
474  * this is useful for debugging purposes.
475  */
476  entry->state_comp.server_id = state->server_id;
477 
478  MEM(vp = fr_pair_afrom_da(packet, state->da));
479  fr_pair_value_memdup(vp, entry->state, sizeof(entry->state), false);
480  fr_pair_add(&packet->vps, vp);
481  }
482 
483  DEBUG4("State ID %" PRIu64 " created, value 0x%pH, expires %" PRIu64 "s",
484  entry->id, fr_box_octets(entry->state, sizeof(entry->state)), (uint64_t)entry->cleanup - now);
485 
486  PTHREAD_MUTEX_LOCK(&state->mutex);
487 
488  /*
489  * XOR the server hash with four bytes of random data.
490  * We XOR is again before resolving, to ensure state lookups
491  * only succeed in the virtual server that created the state
492  * value.
493  */
494  *((uint32_t *)(&entry->state_comp.server_hash)) ^= fr_hash_string(cf_section_name2(request->server_cs));
495 
496  if (!rbtree_insert(state->tree, entry)) {
497  RERROR("Failed inserting state entry - Insertion into state tree failed");
498  fr_pair_delete_by_da(&packet->vps, state->da);
499  talloc_free(entry);
500  return NULL;
501  }
502 
503  /*
504  * Link it to the end of the list, which is implicitely
505  * ordered by cleanup time.
506  */
507  fr_dlist_insert_tail(&state->to_expire, entry);
508 
509  return entry;
510 }

512 /** Find the entry, based on the State attribute
513  *
514  */
515 static fr_state_entry_t *state_entry_find(fr_state_tree_t *state, request_t *request, fr_value_box_t const *vb)
516 {
517  fr_state_entry_t *entry, my_entry;
518 
519  /*
520  * Assume our own State first.
521  */
522  if (vb->vb_length == sizeof(my_entry.state)) {
523  memcpy(my_entry.state, vb->vb_octets, sizeof(my_entry.state));
524 
525  /*
526  * Too big? Get the MD5 hash, in order
527  * to depend on the entire contents of State.
528  */
529  } else if (vb->vb_length > sizeof(my_entry.state)) {
530  fr_md5_calc(my_entry.state, vb->vb_octets, vb->vb_length);
531 
532  /*
533  * Too small? Use the whole thing, and
534  * set the rest of my_entry.state to zero.
535  */
536  } else {
537  memcpy(my_entry.state, vb->vb_octets, vb->vb_length);
538  memset(&my_entry.state[vb->vb_length], 0, sizeof(my_entry.state) - vb->vb_length);
539  }
540 
541  /*
542  * Make it unique for different virtual servers handling the same request
543  */
544  my_entry.state_comp.server_hash ^= fr_hash_string(cf_section_name2(request->server_cs));
545 
546  entry = rbtree_finddata(state->tree, &my_entry);
547 
548  if (entry) (void) talloc_get_type_abort(entry, fr_state_entry_t);
549 
550  return entry;
551 }

553 /** Called when sending an Access-Accept/Access-Reject to discard state information
554  *
555  */
556 void fr_state_discard(fr_state_tree_t *state, request_t *request)
557 {
558  fr_state_entry_t *entry;
559  fr_pair_t *vp;
560 
561  vp = fr_pair_find_by_da(&request->request_pairs, state->da);
562  if (!vp) return;
563 
564  PTHREAD_MUTEX_LOCK(&state->mutex);
565  entry = state_entry_find(state, request, &vp->data);
566  if (!entry) {
567  PTHREAD_MUTEX_UNLOCK(&state->mutex);
568  return;
569  }
570  state_entry_unlink(state, entry);
571  PTHREAD_MUTEX_UNLOCK(&state->mutex);
572 
573  /*
574  * If fr_state_to_request was never called, this ensures
575  * the state owned by entry is freed, otherwise this is
576  * mostly a NOOP, other than freeing the memory held by
577  * the entry.
578  */
579  TALLOC_FREE(entry);
580 
581  /*
582  * If fr_state_to_request was called, then the request
583  * holds the state data, and we need to destroy it
584  * and return the request to the state it was in when
585  * it was first alloced, just in case a user does something
586  * stupid like add additional session-state attributes
587  * in one of the later sections.
588  */
589  TALLOC_FREE(request->state_ctx);
590  request->state_pairs = NULL;
591 
592  MEM(request->state_ctx = talloc_init_const("session-state"));
593 
594  RDEBUG3("%s - discarded", state->da->name);
595 
596  return;
597 }

599 /** Copy a pointer to the head of the list of state fr_pair_ts (and their ctx) into the request
600  *
601  * @note Does not copy the actual fr_pair_ts. The fr_pair_ts and their context
602  * are transferred between state entries as the conversation progresses.
603  *
604  * @note Called with the mutex free.
605  */
606 void fr_state_to_request(fr_state_tree_t *state, request_t *request)
607 {
608  fr_state_entry_t *entry;
609  TALLOC_CTX *old_ctx = NULL;
610  fr_pair_t *vp;
611 
612  fr_assert(request->state_pairs == NULL);
613 
614  /*
615  * No State, don't do anything.
616  */
617  vp = fr_pair_find_by_da(&request->request_pairs, state->da);
618  if (!vp) {
619  RDEBUG3("No &request.State attribute, can't restore &session-state");
620  if (request->seq_start == 0) request->seq_start = request->number; /* Need check for fake requests */
621  return;
622  }
623 
624  PTHREAD_MUTEX_LOCK(&state->mutex);
625  entry = state_entry_find(state, request, &vp->data);
626  if (entry) {
627  (void)talloc_get_type_abort(entry, fr_state_entry_t);
628  if (entry->thawed) {
629  REDEBUG("State entry has already been thawed by a request %"PRIu64, entry->thawed->number);
630  PTHREAD_MUTEX_UNLOCK(&state->mutex);
631  return;
632  }
633  if (request->state_ctx) old_ctx = request->state_ctx; /* Store for later freeing */
634 
635  fr_assert(entry->ctx);
636 
637  request->seq_start = entry->seq_start;
638  request->state_ctx = entry->ctx;
639  request->state_pairs = entry->vps;
640  request_data_restore(request, &entry->data);
641 
642  entry->ctx = NULL;
643  entry->vps = NULL;
644  entry->thawed = request;
645  }
646  PTHREAD_MUTEX_UNLOCK(&state->mutex);
647 
648  if (request->state) {
649  RDEBUG2("Restored &session-state");
650  log_request_pair_list(L_DBG_LVL_2, request, request->state, "&session-state.");
651  }

665 /** Transfer ownership of the state fr_pair_ts and ctx, back to a state entry
666  *
667  * Put request->state_pairs into the State attribute. Put the State attribute
668  * into the vps list. Delete the original entry, if it exists
669  *
670  * Also creates a new state entry.
671  */
672 int fr_request_to_state(fr_state_tree_t *state, request_t *request)
673 {
674  fr_state_entry_t *entry, *old = NULL;
675  fr_dlist_head_t data;
676  fr_pair_t *vp;
677 
678  request_data_list_init(&data);
679  request_data_by_persistance(&data, request, true);
680 
681  if (!request->state_pairs && fr_dlist_empty(&data)) return 0;
682 
683  if (request->state) {
684  RDEBUG2("Saving &session-state");
685  log_request_pair_list(L_DBG_LVL_2, request, request->state, "&session-state.");
686  }
687 
688  vp = fr_pair_find_by_da(&request->request_pairs, state->da);
689 
690  PTHREAD_MUTEX_LOCK(&state->mutex);
691  if (vp) old = state_entry_find(state, request, &vp->data);
692 
693  entry = state_entry_create(state, request, request->reply, old);
694  if (!entry) {
695  PTHREAD_MUTEX_UNLOCK(&state->mutex);
696  RERROR("Creating state entry failed");
697  request_data_restore(request, &data); /* Put it back again */
698  return -1;
699  }
700 
701  fr_assert(entry->ctx == NULL);
702  fr_assert(request->state_ctx);
703 
704  entry->seq_start = request->seq_start;
705  entry->ctx = request->state_ctx;
706  entry->vps = request->state;
707  fr_dlist_move(&entry->data, &data);
708 
709  request->state_ctx = NULL;
710  request->state_pairs = NULL;
711 
712  PTHREAD_MUTEX_UNLOCK(&state->mutex);
713 
714  RDEBUG3("%s - saved", state->da->name);
715  REQUEST_VERIFY(request);
716 
717  return 0;
718 }

721  *

728 void fr_state_store_in_parent(request_t *request, void const *unique_ptr, int unique_int)

731  "Child request must have request->parent set when storing state")) return;

736  * Shove this into the child to make
737  * it easier to store/restore the

744  * add the state list into request_data_t

747  request_data_talloc_add(request, (void *)fr_state_store_in_parent, 0, fr_pair_t,

791 /** Move all request data and session-state VPs into a new state_ctx
792  *
793  * If we don't do this on detach, session-state VPs and persistable
794  * request data will be freed when the parent's state_ctx is freed.
795  * If the parent was freed before the child, we'd get all kinds of
796  * use after free nastiness.
797  *
798  * @param[in] request to detach.
799  * @param[in] will_free Caller super pinky swears to free
800  * the request ASAP, and that it wont
801  * touch persistable request data,
802  * request->state_ctx or request->state.
803  */
804 void fr_state_detach(request_t *request, bool will_free)
805 {
806  fr_pair_t *vps = NULL;
807  TALLOC_CTX *new_state_ctx;
808 
809  if (unlikely(request->parent == NULL)) return;
810 
811  if (will_free) {
812  fr_pair_list_free(&request->state);
813 
814  /*
815  * The non-persistable stuff is
816  * prented directly by the request
817  */
818  request_data_persistable_free(request);
819 
820  /*
821  * Parent will take care of freeing
822  * honestly this should probably
823  * be an assert.
824  */
825  if (request->state_ctx == request->parent->state_ctx) request->state_ctx = NULL;
826  return;
827  }
828 
829  MEM(new_state_ctx = talloc_init_const("session-state"));
830  request_data_by_persistance_reparent(new_state_ctx, NULL, request, true);
831  request_data_by_persistance_reparent(new_state_ctx, NULL, request, false);
832 
833  (void) fr_pair_list_copy(new_state_ctx, &vps, &request->state);
834  fr_pair_list_free(&request->state);
835 
836  request->state_pairs = vps;
837 
838  /*
839  * ...again, should probably
840  * not happen and should probably
841  * be an assert.
842  */
843  if (request->state_ctx != request->parent->state_ctx) talloc_free(request->state_ctx);
844  request->state_ctx = new_state_ctx;
845 }
