92 static int inject_read_cb(struct osmo_fd *fd, unsigned int what)
93 {
94  struct msgb *msg;
95  struct xua_msg_part *data, *link;
96  struct bsc_data *bsc;
97  struct xua_msg *m2ua;
98  struct mtp_link_set *out_set;
99  uint8_t buf[4096];
100 
101  bsc = fd->data;
102 
103  int rc = read(fd->fd, buf, sizeof(buf));
104  if (rc <= 0) {
105  LOGP(DINP, LOGL_ERROR, "Failed to read from the console.\n");
106  return -1;
107  }
108 
109  if (!bsc->allow_inject) {
110  LOGP(DINP, LOGL_ERROR, "Injecting messages is not allowed.\n");
111  return -1;
112  }
113 
114  m2ua = xua_from_msg(M2UA_VERSION, rc, buf);
115  if (!m2ua) {
116  LOGP(DINP, LOGL_ERROR, "Failed to parse M2UA.\n");
117  return -1;
118  }
119 
120  if (m2ua->hdr.msg_class == M2UA_CLS_MAUP && m2ua->hdr.msg_type == M2UA_MAUP_DATA) {
121  data = xua_msg_find_tag(m2ua, M2UA_TAG_DATA);
122  if (!data) {
123  LOGP(DINP, LOGL_ERROR, "MAUP Data without data.\n");
124  goto exit;
125  }
126 
127  if (data->len > 2048) {
128  LOGP(DINP, LOGL_ERROR, "Data is too big for this configuration.\n");
129  goto exit;
130  }
131 
132  link = xua_msg_find_tag(m2ua, MUA_TAG_IDENT_TEXT);
133  if (!link) {
134  LOGP(DINP, LOGL_ERROR, "Interface Identifier Text is mandantory.\n");
135  goto exit;
136  }
137 
138  if (link->len > 255) {
139  LOGP(DINP, LOGL_ERROR, "Spec violation. Ident text should be shorter than 255.\n");
140  goto exit;
141  }
142 
143  out_set = find_link_set(bsc, link->len, (const char *) link->dat);
144  if (!out_set) {
145  LOGP(DINP, LOGL_ERROR, "Identified linkset does not exist.\n");
146  goto exit;
147  }
148 
149  msg = msgb_alloc(2048, "inject-data");
150  if (!msg) {
151  LOGP(DINP, LOGL_ERROR, "Failed to allocate storage.\n");
152  goto exit;
153  }
154 
155  msg->l2h = msgb_put(msg, data->len);
156  memcpy(msg->l2h, data->dat, data->len);
157 
158  /* we are diretcly going to the output. no checking of anything */
159  if (mtp_link_set_send(out_set, msg) != 0) {
160  LOGP(DINP, LOGL_ERROR, "Failed to send message.\n");
161  msgb_free(msg);
162  }
163  }
164 
165 exit:
166  xua_msg_free(m2ua);
167  return 0;
168 }

170 static int inject_init(struct bsc_data *bsc)
171 {
172  int fd;
173  struct sockaddr_in addr;
174 
175  fd = socket(PF_INET, SOCK_DGRAM, 0);
176  if (fd < 0)
177  return -1;
178 
179  memset(&addr, 0, sizeof(addr));
180  addr.sin_family = AF_INET;
181  addr.sin_port = htons(5001);
182 
183  if (bind(fd, (struct sockaddr *) &addr, sizeof(addr)) != 0) {
184  LOGP(DINP, LOGL_ERROR, "Failed to bind to port 5001.\n");
185  close(fd);
186  return -1;
187  }
188 
189  bsc->inject_fd.fd = fd;
190  bsc->inject_fd.when = BSC_FD_READ;
191  bsc->inject_fd.cb = inject_read_cb;
192  bsc->inject_fd.data = bsc;
193 
194  if (osmo_fd_register(&bsc->inject_fd) != 0) {
195  LOGP(DINP, LOGL_ERROR, "Failed to register.\n");
196  close(fd);
197  return -1;
198  }
199 
200  return 0;
201 }
