212 /** Calls to convert verify_mode strings into macros
213  *
214  * @param[in] ctx to allocate data in.
215  * @param[out] out the verify_mode macro representing the mode.
216  * @param[in] ci #CONF_PAIR specifying the name of the mode.
217  * @param[in] rule unused.
218  * @return
219  * - 0 on success.
220  * - -1 on failure.
221  */
222 static int chain_verify_mode_parse(UNUSED TALLOC_CTX *ctx, void *out, UNUSED void *parent,
223  CONF_ITEM *ci, UNUSED CONF_PARSER const *rule)
224 {
225  fr_tls_chain_verify_mode_t type;
226  char const *type_str;
227 
228  type_str = cf_pair_value(cf_item_to_pair(ci));
229  type = fr_table_value_by_str(chain_verify_mode_table, type_str, 0);
230  if (type == 0) {
231  cf_log_err(ci, "Invalid mode \"%s\", expected 'hard', 'soft' or 'none'", type_str);
232  return -1;
233  }
234 
235  *((int *)out) = type;
236 
237  return 0;
238 }

240 /** Calls to convert format strings to OpenSSL macros
241  *
242  * @param[in] ctx to allocate data in.
243  * @param[out] out the OpenSSL macro representing the format.
244  * @param[in] ci #CONF_PAIR specifying the name of the format.
245  * @param[in] rule unused.
246  * @return
247  * - 0 on success.
248  * - -1 on failure.
249  */
250 static int certificate_format_type_parse(UNUSED TALLOC_CTX *ctx, void *out, UNUSED void *parent,
251  CONF_ITEM *ci, UNUSED CONF_PARSER const *rule)
252 {
253  int type;
254  char const *type_str;
255 
256  type_str = cf_pair_value(cf_item_to_pair(ci));
257  type = fr_table_value_by_str(certificate_format_table, type_str, -1);
258  if (type == -1) {
259  cf_log_err(ci, "Invalid format \"%s\", expected either 'PEM' or 'ASN1'", type_str);
260  return -1;
261  }
262 
263  *((int *)out) = type;
264 
265  return 0;
266 }

276 /** Use cert_admin to retrieve the password for the private key
277  *
278  */
279 static int conf_cert_admin_password(fr_tls_conf_t *conf)
280 {
281  size_t i, cnt;
282 
283  if (!conf->chains) return 0;
284 
285  cnt = talloc_array_length(conf->chains);
286  for (i = 0; i < cnt; i++) {
287  char cmd[256];
288  char *password;
289  long const max_password_len = 128;
290  FILE *cmd_pipe;
291 
292  if (!conf->chains[i]->password) continue;
293 
294  if (strncmp(conf->chains[i]->password, special_string, strlen(special_string)) != 0) continue;
295 
296  snprintf(cmd, sizeof(cmd) - 1, "/usr/sbin/certadmin --get-private-key-passphrase \"%s\"",
297  conf->chains[i]->private_key_file);
298 
299  DEBUG2("Getting private key passphrase using command \"%s\"", cmd);
300 
301  cmd_pipe = popen(cmd, "r");
302  if (!cmd_pipe) {
303  ERROR("%s command failed: Unable to get private_key_password", cmd);
304  ERROR("Error reading private_key_file %s", conf->chains[i]->private_key_file);
305  return -1;
306  }
307 
308  password = talloc_array(conf, char, max_password_len);
309  if (!password) {
310  ERROR("Can't allocate space for private_key_password");
311  ERROR("Error reading private_key_file %s", conf->chains[i]->private_key_file);
312  pclose(cmd_pipe);
313  return -1;
314  }
315 
316  fgets(password, max_password_len, cmd_pipe);
317  pclose(cmd_pipe);
318 
319  /* Get rid of newline at end of password. */
320  password[strlen(password) - 1] = '\0';
321 
322  DEBUG3("Password from command = \"%s\"", password);
323  talloc_const_free(conf->chains[i]->password);
324  conf->chains[i]->password = password;
325  }
326 
327  return 0;
328 }
329 #endif

334  * OCSP responses
335  *
336  * - Load the trusted CAs
337  * - Load the trusted issuer certificates
338  */
339 static X509_STORE *conf_ocsp_revocation_store(fr_tls_conf_t *conf)
340 {
341  X509_STORE *store = NULL;
342 
343  store = X509_STORE_new();
344 
345  /* Load the CAs we trust */
346  if (conf->ca_file || conf->ca_path)
347  if (!X509_STORE_load_locations(store, conf->ca_file, conf->ca_path)) {
348  fr_tls_log_error(NULL, "Error reading Trusted root CA list \"%s\"", conf->ca_file);
349  return NULL;
350  }
351 
352 #ifdef X509_V_FLAG_CRL_CHECK_ALL
353  if (conf->check_crl) X509_STORE_set_flags(store, X509_V_FLAG_CRL_CHECK | X509_V_FLAG_CRL_CHECK_ALL);
354 #endif
355

357 }

360 /*
361  * Free TLS client/server config
362  * Should not be called outside this code, as a callback is
363  * added to automatically free the data when the CONF_SECTION
364  * is freed.
365  */
366 static int _conf_server_free(fr_tls_conf_t *conf)
367 {
368  uint32_t i;
369 
370  for (i = 0; i < conf->ctx_count; i++) SSL_CTX_free(conf->ctx[i]);
371 
372 #ifdef HAVE_OPENSSL_OCSP_H
373  if (conf->ocsp.store) X509_STORE_free(conf->ocsp.store);
374  conf->ocsp.store = NULL;
375  if (conf->staple.store) X509_STORE_free(conf->staple.store);
376  conf->staple.store = NULL;
377 #endif
378 
379 #ifndef NDEBUG
380  memset(conf, 0, sizeof(*conf));
381 #endif
382  return 0;
383 }

400 fr_tls_conf_t *fr_tls_conf_parse_server(CONF_SECTION *cs)
401 {
402  fr_tls_conf_t *conf;
403  uint32_t i;
404 
405  /*
406  * If cs has already been parsed there should be a cached copy
407  * of conf already stored, so just return that.
408  */
409  conf = cf_data_value(cf_data_find(cs, fr_tls_conf_t, NULL));
410  if (conf) {
411  DEBUG("Using cached TLS configuration from previous invocation");
412  return conf;
413  }
414 
415  if (cf_section_rules_push(cs, fr_tls_server_config) < 0) return NULL;
416 
417  conf = fr_tls_conf_alloc(cs);
418 
419  if ((cf_section_parse(conf, conf, cs) < 0) ||
420  (cf_section_parse_pass2(conf, cs) < 0)) {
421  error:
422  talloc_free(conf);
423  return NULL;
424  }
425 
426  /*
427  * Save people from their own stupidity.
428  */
429  if (conf->fragment_size < 100) conf->fragment_size = 100;
430 
431 #ifdef __APPLE__
432  if (conf_cert_admin_password(conf) < 0) goto error;
433 #endif
434 
435  conf->ctx_count = fr_tls_max_threads * 2; /* Reduce contention */
436  if (!conf->ctx_count) conf->ctx_count = 1;
437 
438  /*
439  * Initialize TLS
440  */
441  conf->ctx = talloc_zero_array(conf, SSL_CTX *, conf->ctx_count);
442  for (i = 0; i < conf->ctx_count; i++) {
443  conf->ctx[i] = fr_tls_ctx_alloc(conf, false);
444  if (conf->ctx[i] == NULL) goto error;
445  }
446 
447 #ifdef HAVE_OPENSSL_OCSP_H
448  /*
449  * @fixme: This is all pretty terrible.
450  * The stores initialized here are for validating
451  * OCSP responses. They have nothing to do with
452  * verifying other certificates.
453  */
454 
455  /*
456  * Initialize OCSP Revocation Store
457  */
458  if (conf->ocsp.enable) {
459  conf->ocsp.store = conf_ocsp_revocation_store(conf);
460  if (conf->ocsp.store == NULL) goto error;
461  }
462 
463  if (conf->staple.enable) {
464  conf->staple.store = conf_ocsp_revocation_store(conf);
465  if (conf->staple.store == NULL) goto error;
466  }
467 #endif /*HAVE_OPENSSL_OCSP_H*/
468 
469  if (conf->verify_tmp_dir) {
470  if (chmod(conf->verify_tmp_dir, S_IRWXU) < 0) {
471  ERROR("Failed changing permissions on %s: %s",
472  conf->verify_tmp_dir, fr_syserror(errno));
473  goto error;
474  }
475  }
476 
477  if (conf->verify_client_cert_cmd && !conf->verify_tmp_dir) {
478  ERROR("You MUST set the verify directory in order to use verify_client_cmd");
479  goto error;
480  }
481 
482  if (conf->session_cache_server) {
483  CONF_SECTION *server_cs;
484 
485  server_cs = virtual_server_find(conf->session_cache_server);
486  if (!server_cs) {
487  ERROR("No such virtual server '%s'", conf->session_cache_server);
488  goto error;
489  }
490 
491  if (fr_tls_cache_compile(&conf->session_cache, server_cs) < 0) goto error;
492  }
493 
494  if (conf->ocsp.cache_server) {
495  CONF_SECTION *server_cs;
496 
497  server_cs = virtual_server_find(conf->ocsp.cache_server);
498  if (!server_cs) {
499  ERROR("No such virtual server '%s'", conf->ocsp.cache_server);
500  goto error;
