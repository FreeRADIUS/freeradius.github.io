89 /** Handle authorization requests using Couchbase document data
90  *
91  * Attempt to fetch the document assocaited with the requested user by
92  * using the deterministic key defined in the configuration. When a valid
93  * document is found it will be parsed and the containing value pairs will be
94  * injected into the request.
95  *
96  * @param[out] p_result Operation status (#rlm_rcode_t).
97  * @param[in] mctx module calling context.
98  * @param[in] request The authorization request.
99  */
100 static unlang_action_t mod_authorize(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
101 {
102  rlm_couchbase_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_couchbase_t); /* our module instance */
103  rlm_couchbase_handle_t *handle = NULL; /* connection pool handle */
104  char buffer[MAX_KEY_SIZE];
105  char const *dockey; /* our document key */
106  lcb_error_t cb_error = LCB_SUCCESS; /* couchbase error holder */
107  rlm_rcode_t rcode = RLM_MODULE_OK; /* return code */
108  ssize_t slen;
109 
110  /* assert packet as not null */
111  fr_assert(request->packet != NULL);
112 
113  /* attempt to build document key */
114  slen = tmpl_expand(&dockey, buffer, sizeof(buffer), request, inst->user_key, NULL, NULL);
115  if (slen < 0) RETURN_MODULE_FAIL;
116  if ((dockey == buffer) && is_truncated((size_t)slen, sizeof(buffer))) {
117  REDEBUG("Key too long, expected < " STRINGIFY(sizeof(buffer)) " bytes, got %zi bytes", slen);
118  RETURN_MODULE_FAIL;
119  }
120 
121  /* get handle */
122  handle = fr_pool_connection_get(inst->pool, request);
123 
124  /* check handle */
125  if (!handle) RETURN_MODULE_FAIL;
126 
127  /* set couchbase instance */
128  lcb_t cb_inst = handle->handle;
129 
130  /* set cookie */
131  cookie_t *cookie = handle->cookie;
132 
133  /* fetch document */
134  cb_error = couchbase_get_key(cb_inst, cookie, dockey);
135 
136  /* check error */
137  if (cb_error != LCB_SUCCESS || !cookie->jobj) {
138  /* log error */
139  RERROR("failed to fetch document or parse return");
140  /* set return */
141  rcode = RLM_MODULE_FAIL;
142  /* return */
143  goto finish;
144  }
145 
146  /* debugging */
147  RDEBUG3("parsed user document == %s", json_object_to_json_string(cookie->jobj));
148 
149  {
150  TALLOC_CTX *pool = talloc_pool(request, 1024); /* We need to do lots of allocs */
151  fr_cursor_t maps, vlms;
152  map_t *map_head = NULL, *map;
153  vp_list_mod_t *vlm_head = NULL, *vlm;
154 
155  fr_cursor_init(&maps, &map_head);
156 
157  /*
158  * Convert JSON data into maps
159  */
160  if ((mod_json_object_to_map(pool, &maps, request, cookie->jobj, PAIR_LIST_CONTROL) < 0) ||
161  (mod_json_object_to_map(pool, &maps, request, cookie->jobj, PAIR_LIST_REPLY) < 0) ||
162  (mod_json_object_to_map(pool, &maps, request, cookie->jobj, PAIR_LIST_REQUEST) < 0) ||
163  (mod_json_object_to_map(pool, &maps, request, cookie->jobj, PAIR_LIST_STATE) < 0)) {
164  invalid:
165  talloc_free(pool);
166  rcode = RLM_MODULE_INVALID;
167  goto finish;
168  }
169 
170  fr_cursor_init(&vlms, &vlm_head);
171 
172  /*
173  * Convert all the maps into list modifications,
174  * which are guaranteed to succeed.
175  */
176  for (map = fr_cursor_head(&maps);
177  map;
178  map = fr_cursor_next(&maps)) {
179  if (map_to_list_mod(pool, &vlm, request, map, NULL, NULL) < 0) goto invalid;
180  fr_cursor_insert(&vlms, vlm);
181  }
182 
183  if (!vlm_head) {
184  RDEBUG2("Nothing to update");
185  talloc_free(pool);
186  rcode = RLM_MODULE_NOOP;
187  goto finish;
188  }
189 
190  /*
191  * Apply the list of modifications
192  */
193  for (vlm = fr_cursor_head(&vlms);
194  vlm;
195  vlm = fr_cursor_next(&vlms)) {
196  int ret;
197 
198  ret = map_list_mod_apply(request, vlm); /* SHOULD NOT FAIL */
199  if (!fr_cond_assert(ret == 0)) {
200  talloc_free(pool);
201  rcode = RLM_MODULE_FAIL;
202  goto finish;
203  }
204  }
205 
206  talloc_free(pool);
207  }
208 
209 finish:
210  /* free json object */
211  if (cookie->jobj) {
212  json_object_put(cookie->jobj);
213  cookie->jobj = NULL;
214  }
215 
216  /* release handle */
217  if (handle) fr_pool_connection_release(inst->pool, request, handle);
218 
219  /* return */
220  RETURN_MODULE_RCODE(rcode);
221 }

223 /** Write accounting data to Couchbase documents
224  *
225  * Handle accounting requests and store the associated data into JSON documents
226  * in couchbase mapping attribute names to JSON element names per the module configuration.
227  *
228  * When an existing document already exists for the same accounting section the new attributes
229  * will be merged with the currently existing data. When conflicts arrise the new attribute
230  * value will replace or be added to the existing value.
231  *
232  * @param[out] p_result Result of calling the module.
233  * @param mctx module calling context.
234  * @param request The accounting request object.
235  */
236 static unlang_action_t mod_accounting(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
237 {
238  rlm_couchbase_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_couchbase_t); /* our module instance */
239  rlm_couchbase_handle_t *handle = NULL; /* connection pool handle */
240  rlm_rcode_t rcode = RLM_MODULE_OK; /* return code */
241  fr_pair_t *vp; /* radius value pair linked list */
242  char buffer[MAX_KEY_SIZE];
243  char const *dockey; /* our document key */
244  char document[MAX_VALUE_SIZE]; /* our document body */
245  char element[MAX_KEY_SIZE]; /* mapped radius attribute to element name */
246  int status = 0; /* account status type */
247  int docfound = 0; /* document found toggle */
248  lcb_error_t cb_error = LCB_SUCCESS; /* couchbase error holder */
249  ssize_t slen;
250  fr_cursor_t cursor;
251 
252 
253  /* assert packet as not null */
254  fr_assert(request->packet != NULL);
255 
256  /* sanity check */
257  if ((vp = fr_pair_find_by_da(&request->request_pairs, attr_acct_status_type)) == NULL) {
258  /* log debug */
259  RDEBUG2("could not find status type in packet");
260  /* return */
261  RETURN_MODULE_NOOP;
262  }
263 
264  /* set status */
265  status = vp->vp_uint32;
266 
267  /* acknowledge the request but take no action */
268  if (status == FR_STATUS_ACCOUNTING_ON || status == FR_STATUS_ACCOUNTING_OFF) {
269  /* log debug */
270  RDEBUG2("handling accounting on/off request without action");
271  /* return */
272  RETURN_MODULE_OK;
273  }
274 
275  /* get handle */
276  handle = fr_pool_connection_get(inst->pool, request);
277 
278  /* check handle */
279  if (!handle) RETURN_MODULE_FAIL;
280 
281  /* set couchbase instance */
282  lcb_t cb_inst = handle->handle;
283 
284  /* set cookie */
285  cookie_t *cookie = handle->cookie;
286 
287  /* attempt to build document key */
288  slen = tmpl_expand(&dockey, buffer, sizeof(buffer), request, inst->acct_key, NULL, NULL);
289  if (slen < 0) {
290  rcode = RLM_MODULE_FAIL;
291  goto finish;
292  }
293  if ((dockey == buffer) && is_truncated((size_t)slen, sizeof(buffer))) {
294  REDEBUG("Key too long, expected < " STRINGIFY(sizeof(buffer)) " bytes, got %zi bytes", slen);
295  rcode = RLM_MODULE_FAIL;
296  /* return */
297  goto finish;
298  }
299 
300  /* attempt to fetch document */
301  cb_error = couchbase_get_key(cb_inst, cookie, dockey);
302 
303  /* check error and object */
304  if (cb_error != LCB_SUCCESS || cookie->jerr != json_tokener_success || !cookie->jobj) {
305  /* log error */
306  RERROR("failed to execute get request or parse returned json object");
307  /* free and reset json object */
308  if (cookie->jobj) {
309  json_object_put(cookie->jobj);
310  cookie->jobj = NULL;
311  }
312  /* check cookie json object */
313  } else if (cookie->jobj) {
314  /* set doc found */
315  docfound = 1;
316  /* debugging */
317  RDEBUG3("parsed json body from couchbase: %s", json_object_to_json_string(cookie->jobj));
318  }
319 
320  /* start json document if needed */
321  if (docfound != 1) {
322  /* debugging */
323  RDEBUG2("no existing document found - creating new json document");
324  /* create new json object */
325  cookie->jobj = json_object_new_object();
326  /* set 'docType' element for new document */
327  json_object_object_add(cookie->jobj, "docType", json_object_new_string(inst->doctype));
328  /* default startTimestamp and stopTimestamp to null values */
329  json_object_object_add(cookie->jobj, "startTimestamp", NULL);
330  json_object_object_add(cookie->jobj, "stopTimestamp", NULL);
331  }
332 
333  /* status specific replacements for start/stop time */
334  switch (status) {
335  case FR_STATUS_START:
336  /* add start time */
337  if ((vp = fr_pair_find_by_da(&request->request_pairs, attr_acct_status_type)) != NULL) {
338  /* add to json object */
339  json_object_object_add(cookie->jobj, "startTimestamp",
340  mod_value_pair_to_json_object(request, vp));
341  }
342  break;
343 
344  case FR_STATUS_STOP:
345  /* add stop time */
346  if ((vp = fr_pair_find_by_da(&request->request_pairs, attr_event_timestamp)) != NULL) {
347  /* add to json object */
348  json_object_object_add(cookie->jobj, "stopTimestamp",
349  mod_value_pair_to_json_object(request, vp));
350  }
351  /* check start timestamp and adjust if needed */
352  mod_ensure_start_timestamp(cookie->jobj, request->request_pairs);
353  break;
354 
355  case FR_STATUS_ALIVE:
356  /* check start timestamp and adjust if needed */
357  mod_ensure_start_timestamp(cookie->jobj, request->request_pairs);
358  break;
359 
360  default:
361  /* don't doing anything */
362  rcode = RLM_MODULE_NOOP;
363  /* return */
364  goto finish;
365  }
366 
367  /* loop through pairs and add to json document */
368  fr_cursor_init(&cursor, &request->request_pairs);
369 
370  for (vp = fr_cursor_head(&cursor);
371  vp;
372  vp = fr_cursor_next(&cursor)) {
373  /* map attribute to element */
374  if (mod_attribute_to_element(vp->da->name, inst->map, &element) == 0) {
375  /* debug */
376  RDEBUG3("mapped attribute %s => %s", vp->da->name, element);
377  /* add to json object with mapped name */
378  json_object_object_add(cookie->jobj, element, mod_value_pair_to_json_object(request, vp));
379  }
380  }
381 
382  /* copy json string to document and check size */
383  if (strlcpy(document, json_object_to_json_string(cookie->jobj), sizeof(document)) >= sizeof(document)) {
384  /* this isn't good */
385  RERROR("could not write json document - insufficient buffer space");
386  /* set return */
387  rcode = RLM_MODULE_FAIL;
388  /* return */
389  goto finish;
390  }
391 
392  /* debugging */
393  RDEBUG3("setting '%s' => '%s'", dockey, document);
394 
395  /* store document/key in couchbase */
396  cb_error = couchbase_set_key(cb_inst, dockey, document, inst->expire);
397 
398  /* check return */
399  if (cb_error != LCB_SUCCESS) {
400  RERROR("failed to store document (%s): %s (0x%x)", dockey, lcb_strerror(NULL, cb_error), cb_error);
401  }
402 
403 finish:
404  /* free and reset json object */
405  if (cookie->jobj) {
406  json_object_put(cookie->jobj);
407  cookie->jobj = NULL;
408  }
409 
410  /* release our connection handle */
411  if (handle) {
412  fr_pool_connection_release(inst->pool, request, handle);
413  }
414 
415  /* return */
416  RETURN_MODULE_RCODE(rcode);
417 }

458 /** Initialize the rlm_couchbase module
459  *
460  * Intialize the module and create the initial Couchbase connection pool.
461  *
462  * @param conf The module configuration.
463  * @param instance The module instance.
464  * @return
465  * - 0 on success.
466  * - -1 on failure.
467  */
468 static int mod_instantiate(void *instance, CONF_SECTION *conf)
469 {
470  rlm_couchbase_t *inst = instance; /* our module instance */
471 
472  {
473  char *server, *p;
474  size_t len, i;
475  bool sep = false;
476 
477  len = talloc_array_length(inst->server_raw) - 1;
478  server = p = talloc_array(inst, char, len + 1);
479  for (i = 0; i < len; i++) {
480  switch (inst->server_raw[i]) {
481  case '\t':
482  case ' ':
483  case ',':
484  /* Consume multiple separators occurring in sequence */
485  if (sep == true) continue;
486 
487  sep = true;
488  *p++ = ';';
489  break;
490 
491  default:
492  sep = false;
493  *p++ = inst->server_raw[i];
494  break;
495  }
496  }
497 
498  *p = '\0';
499  inst->server = server;
500  }
501 
502  /* setup item map */
503  if (mod_build_attribute_element_map(conf, inst) != 0) {
504  /* fail */
505  return -1;
506  }
507 
508  /* setup libcouchbase extra options */
509  if (mod_build_api_opts(conf, inst) != 0) {
510  /* fail */
511  return -1;
512  }
513 
514  /* initiate connection pool */
515  inst->pool = module_connection_pool_init(conf, inst, mod_conn_create, mod_conn_alive, NULL, NULL, NULL);
516 
517  /* check connection pool */
518  if (!inst->pool) {
519  ERROR("failed to initiate connection pool");
520  /* fail */
521  return -1;
522  }
523 
524  /* load clients if requested */
525  if (inst->read_clients) {
526  CONF_SECTION *cs, *map, *tmpl; /* conf section */
527 
528  /* attempt to find client section */
529  cs = cf_section_find(conf, "client", NULL);
530  if (!cs) {
531  ERROR("failed to find client section while loading clients");
532  /* fail */
533  return -1;
534  }
535 
536  /* attempt to find attribute subsection */
537  map = cf_section_find(cs, "attribute", NULL);
538  if (!map) {
539  ERROR("failed to find attribute subsection while loading clients");
540  /* fail */
541  return -1;
542  }
543 
544  tmpl = cf_section_find(cs, "template", NULL);
545 
546  /* debugging */
547  DEBUG("preparing to load client documents");
548 
549  /* attempt to load clients */
550  if (mod_load_client_documents(inst, tmpl, map) != 0) {
551  /* fail */
552  return -1;
553  }
554  }
555 
556  /* return okay */
557  return 0;
558 }
