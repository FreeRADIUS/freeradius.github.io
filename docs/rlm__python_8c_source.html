236 {

238 
239  PyErr_Fetch(&p_type, &p_value, &p_traceback);
240  PyErr_NormalizeException(&p_type, &p_value, &p_traceback);
241  if (!p_type || !p_value) goto failed;
242 
243  if (((p_str_1 = PyObject_Str(p_type)) == NULL) || ((p_str_2 = PyObject_Str(p_value)) == NULL)) goto failed;
244 
245  ROPTIONAL(RERROR, ERROR, "%s (%s)", PyUnicode_AsUTF8(p_str_1), PyUnicode_AsUTF8(p_str_2));
246 
247  if (p_traceback != Py_None) {
248  PyTracebackObject *ptb = (PyTracebackObject*)p_traceback;
249  size_t fnum = 0;
250 
251  for(; ptb != NULL; ptb = ptb->tb_next, fnum++) {
252  PyFrameObject *cur_frame = ptb->tb_frame;
253 
254  ROPTIONAL(RERROR, ERROR, "[%ld] %s:%d at %s()",
255  fnum,
256  PyUnicode_AsUTF8(cur_frame->f_code->co_filename),
257  PyFrame_GetLineNumber(cur_frame),
258  PyUnicode_AsUTF8(cur_frame->f_code->co_name)
259  );
260  }
261  }
262 
263 failed:
264  Py_XDECREF(p_str_1);
265  Py_XDECREF(p_str_2);
266  Py_XDECREF(p_type);
267  Py_XDECREF(p_value);
268  Py_XDECREF(p_traceback);

271 static void mod_vptuple(TALLOC_CTX *ctx, rlm_python_t const *inst, request_t *request,
272  fr_pair_t **vps, PyObject *p_value, char const *funcname, char const *list_name)
273 {
274  int i;
275  Py_ssize_t tuple_len;
276  tmpl_t *dst;
277  fr_pair_t *vp;
278  request_t *current = request;
279 
280  /*
281  * If the Python function gave us None for the tuple,
282  * then just return.
283  */
284  if (p_value == Py_None) return;
285 
286  if (!PyTuple_CheckExact(p_value)) {
287  ERROR("%s - non-tuple passed to %s", funcname, list_name);
288  return;
289  }
290  /* Get the tuple tuple_len. */
291  tuple_len = PyTuple_GET_SIZE(p_value);
292  for (i = 0; i < tuple_len; i++) {
293  PyObject *p_tuple_element = PyTuple_GET_ITEM(p_value, i);
294  PyObject *p_str_1;
295  PyObject *p_str_2;
296  PyObject *p_op;
297  Py_ssize_t pair_len;
298  char const *s1;
299  char const *s2;
300  fr_token_t op = T_OP_EQ;
301 
302  if (!PyTuple_CheckExact(p_tuple_element)) {
303  ERROR("%s - Tuple element %d of %s is not a tuple", funcname, i, list_name);
304  continue;
305  }
306  /* Check if it's a pair */
307 
308  pair_len = PyTuple_GET_SIZE(p_tuple_element);
309  if ((pair_len < 2) || (pair_len > 3)) {
310  ERROR("%s - Tuple element %d of %s is a tuple of size %zu. Must be 2 or 3",
311  funcname, i, list_name, pair_len);
312  continue;
313  }
314 
315  p_str_1 = PyTuple_GET_ITEM(p_tuple_element, 0);
316  p_str_2 = PyTuple_GET_ITEM(p_tuple_element, pair_len - 1);
317 
318  if ((!PyUnicode_CheckExact(p_str_1)) || (!PyUnicode_CheckExact(p_str_2))) {
319  ERROR("%s - Tuple element %d of %s must be as (str, str)",
320  funcname, i, list_name);
321  continue;
322  }
323  s1 = PyUnicode_AsUTF8(p_str_1);
324  s2 = PyUnicode_AsUTF8(p_str_2);
325 
326  if (pair_len == 3) {
327  p_op = PyTuple_GET_ITEM(p_tuple_element, 1);
328  if (PyUnicode_CheckExact(p_op)) {
329  if (!(op = fr_table_value_by_str(fr_tokens_table, PyUnicode_AsUTF8(p_op), 0))) {
330  ERROR("%s - Invalid operator %s.%s %s %s, falling back to '='",
331  funcname, list_name, s1, PyUnicode_AsUTF8(p_op), s2);
332  op = T_OP_EQ;
333  }
334  } else if (PyNumber_Check(p_op)) {
335  long py_op;
336 
337  py_op = PyLong_AsLong(p_op);
338  if (!fr_table_str_by_value(fr_tokens_table, py_op, NULL)) {
339  ERROR("%s - Invalid operator %s.%s %i %s, falling back to '='",
340  funcname, list_name, s1, op, s2);
341  op = T_OP_EQ;
342  } else {
343  op = (fr_token_t)py_op;
344  }
345  } else {
346  ERROR("%s - Invalid operator type for %s.%s ? %s, using default '='",
347  funcname, list_name, s1, s2);
348  }
349  }
350 
351  if (tmpl_afrom_attr_str(ctx, NULL, &dst, s1,
352  &(tmpl_rules_t){
353  .dict_def = request->dict,
354  .list_def = PAIR_LIST_REPLY
355  }) <= 0) {
356  ERROR("%s - Failed to find attribute %s.%s", funcname, list_name, s1);
357  continue;
358  }
359 
360  if (radius_request(&current, tmpl_request(dst)) < 0) {
361  ERROR("%s - Attribute name %s.%s refers to outer request but not in a tunnel, skipping...",
362  funcname, list_name, s1);
363  talloc_free(dst);
364  continue;
365  }
366 
367  MEM(vp = fr_pair_afrom_da(ctx, tmpl_da(dst)));
368  talloc_free(dst);
369 
370  vp->op = op;
371  if (fr_pair_value_from_str(vp, s2, -1, '\0', false) < 0) {
372  DEBUG("%s - Failed: '%s.%s' %s '%s'", funcname, list_name, s1,
373  fr_table_str_by_value(fr_tokens_table, op, "="), s2);
374  } else {
375  DEBUG("%s - '%s.%s' %s '%s'", funcname, list_name, s1,
376  fr_table_str_by_value(fr_tokens_table, op, "="), s2);
377  }
378 
379  radius_pairmove(current, vps, vp, false);
380  }
381 }

397  switch (vp->vp_type) {

399  value = PyUnicode_FromStringAndSize(vp->vp_strvalue, vp->vp_length);

401

403  value = PyBytes_FromStringAndSize((char const *)vp->vp_octets, vp->vp_length);
404  break;
405 
406  case FR_TYPE_BOOL:
407  value = PyBool_FromLong(vp->vp_bool);
408  break;
409

411  value = PyLong_FromUnsignedLong(vp->vp_uint8);
412  break;
413

415  value = PyLong_FromUnsignedLong(vp->vp_uint16);
416  break;
417 
418  case FR_TYPE_UINT32:
419  value = PyLong_FromUnsignedLong(vp->vp_uint32);
420  break;
421 
422  case FR_TYPE_UINT64:
423  value = PyLong_FromUnsignedLongLong(vp->vp_uint64);
424  break;
425 
426  case FR_TYPE_INT8:
427  value = PyLong_FromLong(vp->vp_int8);
428  break;
429 
430  case FR_TYPE_INT16:
431  value = PyLong_FromLong(vp->vp_int16);
432  break;
433 
434  case FR_TYPE_INT32:
435  value = PyLong_FromLong(vp->vp_int32);
436  break;
437 
438  case FR_TYPE_INT64:
439  value = PyLong_FromLongLong(vp->vp_int64);
440  break;
441 
442  case FR_TYPE_FLOAT32:
443  value = PyFloat_FromDouble((double) vp->vp_float32);
444  break;
445 
446  case FR_TYPE_FLOAT64:
447  value = PyFloat_FromDouble(vp->vp_float64);
448  break;
449 
450  case FR_TYPE_SIZE:
451  value = PyLong_FromSize_t(vp->vp_size);
452  break;
453 
454  case FR_TYPE_TIME_DELTA:
455  case FR_TYPE_IPV4_ADDR:
456  case FR_TYPE_DATE:
457  case FR_TYPE_IFID:
458  case FR_TYPE_IPV6_ADDR:
459  case FR_TYPE_IPV6_PREFIX:
460  case FR_TYPE_ETHERNET:
461  case FR_TYPE_IPV4_PREFIX:
462  {
463  ssize_t slen;
464  char buffer[256];
465 
466  slen = fr_pair_print_value_quoted(&FR_SBUFF_OUT(buffer, sizeof(buffer)), vp, T_BARE_WORD);
467  if (slen < 0) {
468  error:
469  ROPTIONAL(REDEBUG, ERROR, "Failed marshalling %pP to Python value", vp);
470  python_error_log(inst, request);
471  Py_XDECREF(attribute);
472  return -1;
473  }
474  value = PyUnicode_FromStringAndSize(buffer, (size_t)slen);
475  }
476  break;
477 
478  case FR_TYPE_NON_VALUES:
479  fr_assert(0);
480  return -1;
481  }
482 
483  if (value == NULL) goto error;
484 
485  PyTuple_SET_ITEM(pp, 0, attribute);
486  PyTuple_SET_ITEM(pp, 1, value);
487 
488  return 0;
489 }

491 static unlang_action_t do_python_single(rlm_rcode_t *p_result,
492  rlm_python_t const *inst, request_t *request, PyObject *p_func, char const *funcname)
493 {
494  fr_cursor_t cursor;
495  fr_pair_t *vp;
496  PyObject *p_ret = NULL;
497  PyObject *p_arg = NULL;
498  int tuple_len;
499  rlm_rcode_t rcode = RLM_MODULE_OK;
500 
501  /*
502  * We will pass a tuple containing (name, value) tuples
503  * We can safely use the Python function to build up a
504  * tuple, since the tuple is not used elsewhere.
505  *
506  * Determine the size of our tuple by walking through the packet.
507  * If request is NULL, pass None.
508  */
509  tuple_len = 0;
510  if (request != NULL) {
511  for (vp = fr_cursor_init(&cursor, &request->request_pairs);
512  vp;
513  vp = fr_cursor_next(&cursor)) tuple_len++;
514  }
515 
516  if (tuple_len == 0) {
517  Py_INCREF(Py_None);
518  p_arg = Py_None;
519  } else {
520  int i = 0;
521  if ((p_arg = PyTuple_New(tuple_len)) == NULL) {
522  rcode = RLM_MODULE_FAIL;
523  goto finish;
524  }
525 
526  for (vp = fr_cursor_init(&cursor, &request->request_pairs);
527  vp;
528  vp = fr_cursor_next(&cursor), i++) {
529  PyObject *pp;
530 
531  /* The inside tuple has two only: */
532  if ((pp = PyTuple_New(2)) == NULL) {
533  rcode = RLM_MODULE_FAIL;
534  goto finish;
535  }
536 
537  if (mod_populate_vptuple(inst, request, pp, vp) == 0) {
538  /* Put the tuple inside the container */
539  PyTuple_SET_ITEM(p_arg, i, pp);
540  } else {
541  Py_INCREF(Py_None);
542  PyTuple_SET_ITEM(p_arg, i, Py_None);
543  Py_DECREF(pp);
544  }
545  }
546  }
547 
548  /* Call Python function. */
549  p_ret = PyObject_CallFunctionObjArgs(p_func, p_arg, NULL);
550  if (!p_ret) {
551  python_error_log(inst, request); /* Needs valid thread with GIL */
552  rcode = RLM_MODULE_FAIL;
553  goto finish;
554  }
555 
556  if (!request) {
557  // check return code at module instantiation time
558  if (PyNumber_Check(p_ret)) rcode = PyLong_AsLong(p_ret);
559  goto finish;
560  }
561 
562  /*
563  * The function returns either:
564  * 1. (returnvalue, replyTuple, configTuple), where
565  * - returnvalue is one of the constants RLM_*
566  * - replyTuple and configTuple are tuples of string
567  * tuples of size 2
568  *
569  * 2. the function return value alone
570  *
571  * 3. None - default return value is set
572  *
573  * xxx This code is messy!
574  */
575  if (PyTuple_CheckExact(p_ret)) {
576  PyObject *p_tuple_int;
577 
578  if (PyTuple_GET_SIZE(p_ret) != 3) {
579  ERROR("%s - Tuple must be (return, replyTuple, configTuple)", funcname);
580  rcode = RLM_MODULE_FAIL;
581  goto finish;
582  }
583 
584  p_tuple_int = PyTuple_GET_ITEM(p_ret, 0);
585  if (!PyNumber_Check(p_tuple_int)) {
586  ERROR("%s - First tuple element not an integer", funcname);
587  rcode = RLM_MODULE_FAIL;
588  goto finish;
589  }
590  /* Now have the return value */
591  rcode = PyLong_AsLong(p_tuple_int);
592  /* Reply item tuple */
593  mod_vptuple(request->reply, inst, request, &request->reply_pairs,
594  PyTuple_GET_ITEM(p_ret, 1), funcname, "reply");
595  /* Config item tuple */
596  mod_vptuple(request, inst, request, &request->control_pairs,
597  PyTuple_GET_ITEM(p_ret, 2), funcname, "config");
598 
599  } else if (PyNumber_Check(p_ret)) {
600  /* Just an integer */
601  rcode = PyLong_AsLong(p_ret);
602 
603  } else if (p_ret == Py_None) {
604  /* returned 'None', return value defaults to "OK, continue." */
605  rcode = RLM_MODULE_OK;
606  } else {
607  /* Not tuple or None */
608  ERROR("%s - Function did not return a tuple or None", funcname);
609  rcode = RLM_MODULE_FAIL;
610  goto finish;
611  }
612 
613 finish:
614  if (rcode == RLM_MODULE_FAIL) python_error_log(inst, request);
615  Py_XDECREF(p_arg);
616  Py_XDECREF(p_ret);
617 
618  RETURN_MODULE_RCODE(rcode);
619 }

621 /** Thread safe call to a python function
622  *
623  * Will swap in thread state specific to module/thread.
624  */
625 static unlang_action_t do_python(rlm_rcode_t *p_result, rlm_python_t const *inst, rlm_python_thread_t *this_thread,
626  request_t *request, PyObject *p_func, char const *funcname)
627 {
628  rlm_rcode_t rcode;
629 
630  /*
631  * It's a NOOP if the function wasn't defined
632  */
633  if (!p_func) RETURN_MODULE_NOOP;
634 
635  RDEBUG3("Using thread state %p/%p", inst, this_thread->state);
636 
637  PyEval_RestoreThread(this_thread->state); /* Swap in our local thread state */
638  do_python_single(&rcode, inst, request, p_func, funcname);
639  (void)fr_cond_assert(PyEval_SaveThread() == this_thread->state);
640 
641  RETURN_MODULE_RCODE(rcode);

708 /*
709  * Parse a configuration section, and populate a dict.
710  * This function is recursively called (allows to have nested dicts.)
711  */
712 static int python_parse_config(rlm_python_t *inst, CONF_SECTION *cs, int lvl, PyObject *dict)
713 {
714  int indent_section = (lvl * 4);
715  int indent_item = (lvl + 1) * 4;
716  int ret = 0;
717  CONF_ITEM *ci = NULL;
718 
719  if (!cs || !dict) return -1;
720 
721  DEBUG("%*s%s {", indent_section, " ", cf_section_name1(cs));
722 
723  while ((ci = cf_item_next(cs, ci))) {
724  /*
725  * This is a section.
726  * Create a new dict, store it in current dict,
727  * Then recursively call python_parse_config with this section and the new dict.
728  */
729  if (cf_item_is_section(ci)) {
730  CONF_SECTION *sub_cs = cf_item_to_section(ci);
731  char const *key = cf_section_name1(sub_cs); /* dict key */
732  PyObject *sub_dict, *p_key;
733 
734  p_key = PyUnicode_FromString(key);
735  if (!p_key) {
736  ERROR("Failed converting config key \"%s\" to python string", key);
737  return -1;
738  }
739 
740  if (PyDict_Contains(dict, p_key)) {
741  WARN("Ignoring duplicate config section '%s'", key);
742  continue;
743  }
744 
745  MEM(sub_dict = PyDict_New());
746  (void)PyDict_SetItem(dict, p_key, sub_dict);
747 
748  ret = python_parse_config(inst, sub_cs, lvl + 1, sub_dict);
749  if (ret < 0) break;
750  } else if (cf_item_is_pair(ci)) {
751  CONF_PAIR *cp = cf_item_to_pair(ci);
752  char const *key = cf_pair_attr(cp); /* dict key */
753  char const *value = cf_pair_value(cp); /* dict value */
754  PyObject *p_key, *p_value;
755 
756  if (!value) {
757  WARN("Skipping \"%s\" as it has no value", key);
758  continue;
759  }
760 
761  p_key = PyUnicode_FromString(key);
762  p_value = PyUnicode_FromString(value);
763  if (!p_key) {
764  ERROR("Failed converting config key \"%s\" to python string", key);
765  return -1;
766  }
767  if (!p_value) {
768  ERROR("Failed converting config value \"%s\" to python string", value);
769  return -1;
770  }
771 
772  /*
773  * This is an item.
774  * Store item attr / value in current dict.
775  */
776  if (PyDict_Contains(dict, p_key)) {
777  WARN("Ignoring duplicate config item '%s'", key);
778  continue;
779  }
780 
781  (void)PyDict_SetItem(dict, p_key, p_value);
782 
783  DEBUG("%*s%s = \"%s\"", indent_item, " ", key, value);
784  }
785  }
786 
787  DEBUG("%*s}", indent_section, " ");
788 
789  return ret;
790 }
