226

228  if (!ms) {
229  fprintf(stderr, "Failed creating message set\n");

234  * Create the KQ and associated sockets.
235  */
236  kq_master = kqueue();
237  fr_assert(kq_master >= 0);
238 
239  aq_master = fr_atomic_queue_alloc(ctx, max_control_plane);
240  fr_assert(aq_master != NULL);
241 
242  control_master = fr_control_create(ctx, kq_master, aq_master, 1024);
243  fr_assert(control_master != NULL);
244 
245  sockfd = fr_socket_server_udp(&my_ipaddr, &my_port, NULL, true);
246  if (sockfd < 0) {
247  fr_perror("radius_test: Failed creating socket");
248  fr_exit_now(EXIT_FAILURE);
249  }
250 
251  if (fr_socket_bind(sockfd, &my_ipaddr, &my_port, NULL) < 0) {
252  fr_perror("radius_test: Failed binding to socket");
253  fr_exit_now(EXIT_FAILURE);
254  }
255 
256  /*
257  * Set up the KQ filter for reading.
258  */
259  EV_SET(&events[0], sockfd, EVFILT_READ, EV_ADD | EV_ENABLE, 0, 0, NULL);
260  if (kevent(kq_master, events, 1, NULL, 0, NULL) < 0) {
261  fr_perror("Failed setting KQ for EVFILT_READ");
262  fr_exit_now(EXIT_FAILURE);
263  }
264 
265  /*
266  * Create the worker threads.
267  */
268  (void) pthread_attr_init(&pthread_attr);
269  (void) pthread_attr_setdetachstate(&pthread_attr, PTHREAD_CREATE_JOINABLE);
270 
271  for (i = 0; i < num_workers; i++) {
272  workers[i].id = i;
273  (void) pthread_create(&workers[i].pthread_id, &pthread_attr, worker_thread, &workers[i]);
274  }
275 
276  MPRINT1("Master created %d workers.\n", num_workers);
277 
278  /*
279  * Busy loop because that's fine for the test
280  */
281  num_outstanding = 0;
282  while (num_outstanding < num_workers) {
283  for (i = 0; i < num_workers; i++) {
284  if (!workers[i].worker) continue;
285  if (workers[i].ch != NULL) continue;
286 
287  /*
288  * Create the channel and signal the
289  * worker that it is open
290  */
291  MPRINT1("Master creating channel to worker %d.\n", num_workers);
292  workers[i].ch = fr_worker_channel_create(workers[i].worker, ctx, control_master);
293  fr_assert(workers[i].ch != NULL);
294 
295  (void) fr_channel_master_ctx_add(workers[i].ch, &workers[i]);
296 
297  num_outstanding++;
298  }
299  }
300 
301  MPRINT1("Master created all channels.\n");
302 
303  which_worker = 0;
304  running = true;
305 
306  while (running) {
307  bool control_plane_signal;
308  fr_time_t now;
309  fr_channel_data_t *cd, *reply;
310 
311  MPRINT1("Master waiting on events.\n");
312 
313  num_events = kevent(kq_master, NULL, 0, events, MAX_KEVENTS, NULL);
314  MPRINT1("Master kevent returned %d\n", num_events);
315 
316  if (num_events < 0) {
317  if (errno == EINTR) continue;
318 
319  fprintf(stderr, "Failed waiting for kevent: %s\n", fr_syserror(errno));
320  fr_exit_now(EXIT_FAILURE);
321  }
322 
323  if (num_events == 0) continue;
324 
325  control_plane_signal = false;
326 
327  /*
328  * Service the events.
329  *
330  * @todo this should NOT take a channel pointer
331  */
332  for (i = 0; i < num_events; i++) {
333  uint8_t *packet, *attr, *end;
334  size_t total_len;
335  ssize_t data_size;
336  fr_radius_packet_ctx_t *packet_ctx;
337 
338  if (events[i].filter == EVFILT_USER) {
339  (void) fr_channel_service_kevent(workers[0].ch, control_master, &events[i]);
340  control_plane_signal = true;
341  break;
342  }
343 
344  fr_assert(events[i].filter == EVFILT_READ);
345 
346  cd = (fr_channel_data_t *) fr_message_reserve(ms, 4096);
347  fr_assert(cd != NULL);
348 
349  packet_ctx = talloc(ctx, fr_radius_packet_ctx_t);
350  fr_assert(packet_ctx != NULL);
351  packet_ctx->salen = sizeof(packet_ctx->src);
352 
353  cd->priority = 0;
354  cd->packet_ctx = packet_ctx;
355  cd->listen = &listen;
356 
357  data_size = recvfrom(sockfd, cd->m.data, cd->m.rb_size, 0,
358  (struct sockaddr *) &packet_ctx->src, &packet_ctx->salen);
359  MPRINT1("Master got packet size %zd\n", data_size);
360  if (data_size <= 20) {
361  MPRINT1("Master ignoring packet (data length %zd)\n", data_size);
362 
363  discard:
364  fr_message_done(&cd->m); /* yeah, re-use it for the next packet... */
365  continue;
366  }
367 
368  /*
369  * Verify the packet before doing anything more with it.
370  */
371  packet = cd->m.data;
372  if (packet[0] != FR_CODE_ACCESS_REQUEST) {
373  MPRINT1("Master ignoring packet code %u\n", packet[0]);
374  goto discard;
375  }
376 
377  total_len = (packet[2] << 8) | packet[3];
378  if (total_len < 20) {
379  MPRINT1("Master ignoring packet (header length %zu)\n", total_len);
380  goto discard;
381  }
382  if (total_len > (size_t) data_size) {
383  MPRINT1("Master ignoring truncated packet (read %zd, says %zu)\n",
384  data_size, total_len);
385  goto discard;
386  }
387 
388  attr = packet + 20;
389  end = packet + data_size;
390  while (attr < end) {
391  if ((end - attr) < 2) goto discard;
392  if (attr[0] == 0) goto discard;
393  if (attr[1] < 2) goto discard;
394  if ((attr + attr[1]) > end) goto discard;
395 
396  attr += attr[1];
397  }
398 
399  (void) fr_message_alloc(ms, &cd->m, total_len);
400 
401  MPRINT1("Master sending packet size %zd to worker %d\n", cd->m.data_size, which_worker);
402  cd->m.when = fr_time();
403 
404  packet_ctx->id = packet[1];
405  memcpy(packet_ctx->vector, packet + 4, 16);
406 
407  rcode = fr_channel_send_request(workers[which_worker].ch, cd, &reply);
408  if (rcode < 0) {
409  fprintf(stderr, "Failed sending request: %s\n", fr_syserror(errno));
410  fr_exit_now(EXIT_FAILURE);
411  }
412  which_worker++;
413  if (which_worker >= num_workers) which_worker = 0;
414 
415  fr_assert(rcode == 0);
416  if (reply) send_reply(sockfd, reply);
417  }
418 
419  if (!control_plane_signal) continue;
420 
421  now = fr_time();
422 
423  MPRINT1("Master servicing control-plane\n");
424 
425  while (true) {
426  uint32_t id;
427  size_t data_size;
428  char data[256];
429 
430  data_size = fr_control_message_pop(aq_master, &id, data, sizeof(data));
431  if (!data_size) break;
432 
433  fr_assert(id == FR_CONTROL_ID_CHANNEL);
434 
435  ce = fr_channel_service_message(now, &ch, data, data_size);
436  MPRINT1("Master got channel event %d\n", ce);
437 
438  switch (ce) {
439  case FR_CHANNEL_DATA_READY_REQUESTOR:
440  MPRINT1("Master got data ready signal\n");
441 
442  reply = fr_channel_recv_reply(ch);
443  if (!reply) {
444  MPRINT1("Master SIGNAL WITH NO DATA!\n");
445  continue;
446  }
447 
448  do {
449  send_reply(sockfd, reply);
450  } while ((reply = fr_channel_recv_reply(ch)) != NULL);
451  break;
452 
453  case FR_CHANNEL_CLOSE:
454  sw = fr_channel_master_ctx_get(ch);
455  fr_assert(sw != NULL);
456 
457  MPRINT1("Master received close ack signal for worker %d\n", sw->id);
458 
459  (void) pthread_kill(sw->pthread_id, SIGTERM);
460  running = false;
461  break;
462 
463  case FR_CHANNEL_NOOP:
464  break;
465 
466  default:
467  fprintf(stderr, "Master got unexpected CE %d\n", ce);
468 
469  /*
470  * Not written yet!
471  */
472  fr_assert(0 == 1);
473  break;
474  } /* switch over signal returned */
475  } /* drain the control plane */
476  } /* loop until told to exit */
477 
478  MPRINT1("Master exiting.\n");
479 
480  fr_time_t last_checked = fr_time();
481 
482  /*
483  * Busy-wait for the workers to exit;
484  */
485  do {
486  fr_time_t now = fr_time();
487 
488  num_outstanding = num_workers;
489 
490  for (i = 0; i < num_workers; i++) {
491  if (!workers[i].worker) num_outstanding--;
492  }
493 
494  if ((now - last_checked) > (NSEC / 10)) {
495  MPRINT1("still num_outstanding %d\n", num_outstanding);
496  }
497 
498  } while (num_outstanding > 0);
499 
500  /*
501  * Force all messages to be garbage collected
502  */
503  MPRINT2("GC\n");
504  fr_message_set_gc(ms);
505 
506  if (debug_lvl > 1) fr_message_set_debug(ms, stdout);
507 
508  /*
509  * After the garbage collection, all messages marked "done" MUST also be marked "free".
510  */
511  rcode = fr_message_set_messages_used(ms);
512  MPRINT2("Master messages used = %d\n", rcode);
513  fr_assert(rcode == 0);
514  close(sockfd);
515 }
