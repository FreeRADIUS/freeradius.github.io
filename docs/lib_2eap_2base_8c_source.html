118 fr_pair_t *eap_packet_to_vp(fr_radius_packet_t *packet, eap_packet_raw_t const *eap)
119 {
120  int total, size;
121  uint8_t const *ptr;
122  fr_pair_t *head = NULL;
123  fr_pair_t *vp;
124  fr_cursor_t out;
125 
126  total = eap->length[0] * 256 + eap->length[1];
127 
128  if (total == 0) {
129  DEBUG("Asked to encode empty EAP-Message!");
130  return NULL;
131  }
132 
133  ptr = (uint8_t const *) eap;
134 
135  fr_cursor_init(&out, &head);
136  do {
137  size = total;
138  if (size > 253) size = 253;
139 
140  MEM(vp = fr_pair_afrom_da(packet, attr_eap_message));
141  fr_pair_value_memdup(vp, ptr, size, false);
142 
143  fr_cursor_append(&out, vp);
144 
145  ptr += size;
146  total -= size;
147  } while (total > 0);
148 
149  return head;
150 }

152 /** Basic EAP packet verifications & validations
153  *
154  * @param[in] eap_packet_p to validate.
155  * @return
156  * - true the packet is valid.
157  * - false the packet is invalid.
158  */
159 static bool eap_is_valid(eap_packet_raw_t **eap_packet_p)
160 {
161  uint16_t len;
162  size_t packet_len;
163  eap_packet_raw_t *eap_packet = *eap_packet_p;
164 
165  /*
166  * These length checks are also done by eap_packet_from_vp(),
167  * but that's OK. The static analysis tools aren't smart
168  * enough to figure that out.
169  */
170  packet_len = talloc_array_length((uint8_t *) eap_packet);
171  if (packet_len <= EAP_HEADER_LEN) {
172  fr_strerror_printf("Invalid EAP data lenth %zd <= 4", packet_len);
173  return false;
174  }
175 
176  memcpy(&len, eap_packet->length, sizeof(uint16_t));
177  len = ntohs(len);
178 
179  if ((len <= EAP_HEADER_LEN) || (len > packet_len)) {
180  fr_strerror_printf("Invalid EAP length field. Expected value in range %u-%zu, was %u bytes",
181  EAP_HEADER_LEN, packet_len, len);
182  return false;
183  }
184 
185  /*
186  * High level EAP packet checks
187  */
188  switch (eap_packet->code) {
189  case FR_EAP_CODE_RESPONSE:
190  case FR_EAP_CODE_REQUEST:
191  break;
192 
193  default:
194  fr_strerror_printf("Invalid EAP code %d: Ignoring the packet", eap_packet->code);
195  return false;
196  }
197 
198  if ((eap_packet->data[0] == 0) ||
199  (eap_packet->data[0] >= FR_EAP_METHOD_MAX)) {
200  /*
201  * Handle expanded types by smashing them to
202  * normal types.
203  */
204  if (eap_packet->data[0] == FR_EAP_EXPANDED_TYPE) {
205  uint8_t *p, *q;
206 
207  if (len <= (EAP_HEADER_LEN + 1 + 3 + 4)) {
208  fr_strerror_printf("Expanded EAP type is too short: ignoring the packet");
209  return false;
210  }
211 
212  if ((eap_packet->data[1] != 0) ||
213  (eap_packet->data[2] != 0) ||
214  (eap_packet->data[3] != 0)) {
215  fr_strerror_printf("Expanded EAP type has unknown Vendor-ID: ignoring the packet");
216  return false;
217  }
218 
219  if ((eap_packet->data[4] != 0) ||
220  (eap_packet->data[5] != 0) ||
221  (eap_packet->data[6] != 0)) {
222  fr_strerror_printf("Expanded EAP type has unknown Vendor-Type: ignoring the packet");
223  return false;
224  }
225 
226  if ((eap_packet->data[7] == 0) ||
227  (eap_packet->data[7] >= FR_EAP_METHOD_MAX)) {
228  fr_strerror_printf("Unsupported Expanded EAP type %s (%u): ignoring the packet",
229  eap_type2name(eap_packet->data[7]), eap_packet->data[7]);
230  return false;
231  }
232 
233  if (eap_packet->data[7] == FR_EAP_METHOD_NAK) {
234  fr_strerror_printf("Unsupported Expanded EAP-NAK: ignoring the packet");
235  return false;
236  }
237 
238  /*
239  * Re-write the EAP packet to NOT have the expanded type.
240  */
241  q = (uint8_t *) eap_packet;
242  memmove(q + EAP_HEADER_LEN, q + EAP_HEADER_LEN + 7, len - 7 - EAP_HEADER_LEN);
243 
244  p = talloc_realloc(talloc_parent(eap_packet), eap_packet, uint8_t, len - 7);
245  if (!p) {
246  fr_strerror_printf("Unsupported EAP type %s (%u): ignoring the packet",
247  eap_type2name(eap_packet->data[0]), eap_packet->data[0]);
248  return false;
249  }
250 
251  len -= 7;
252  p[2] = (len >> 8) & 0xff;
253  p[3] = len & 0xff;
254 
255  *eap_packet_p = (eap_packet_raw_t *)p;
256 
257  return true;
258  }
259 
260  fr_strerror_printf("Unsupported EAP type %s (%u): ignoring the packet",
261  eap_type2name(eap_packet->data[0]), eap_packet->data[0]);
262  return false;
263  }
264 
265  /* we don't expect notification, but we send it */
266  if (eap_packet->data[0] == FR_EAP_METHOD_NOTIFICATION) {
267  fr_strerror_printf("Got NOTIFICATION, Ignoring the packet");
268  return false;
269  }
270 
271  return true;
272 }

274 /*
275  * Handles multiple EAP-Message attrs
276  * ie concatenates all to get the complete EAP packet.
277  *
278  * NOTE: Sometimes Framed-MTU might contain the length of EAP-Message,
279  * refer fragmentation in rfc2869.
280  */
281 eap_packet_raw_t *eap_packet_from_vp(TALLOC_CTX *ctx, fr_pair_t *vps)
282 {
283  fr_pair_t *vp;
284  eap_packet_raw_t *eap_packet;
285  unsigned char *ptr;
286  uint16_t len;
287  int total_len;
288  fr_cursor_t cursor;
289 
290  /*
291  * Get only EAP-Message attribute list
292  */
293  vp = fr_cursor_iter_by_da_init(&cursor, &vps, attr_eap_message);
294  if (!vp) {
295  fr_strerror_printf("EAP-Message not found");
296  return NULL;
297  }
298 
299  /*
300  * Sanity check the length before doing anything.
301  */
302  if (vp->vp_length < 4) {
303  fr_strerror_printf("EAP packet is too short");
304  return NULL;
305  }
306 
307  /*
308  * Get the Actual length from the EAP packet
309  * First EAP-Message contains the EAP packet header
310  */
311  memcpy(&len, vp->vp_strvalue + 2, sizeof(len));
312  len = ntohs(len);
313 
314  /*
315  * Take out even more weird things.
316  */
317  if (len < 4) {
318  fr_strerror_printf("EAP packet has invalid length (less than 4 bytes)");
319  return NULL;
320  }
321 
322  /*
323  * Sanity check the length, BEFORE allocating memory.
324  */
325  total_len = 0;
326  for (vp = fr_cursor_head(&cursor);
327  vp;
328  vp = fr_cursor_next(&cursor)) {
329  total_len += vp->vp_length;
330 
331  if (total_len > len) {
332  fr_strerror_printf("Malformed EAP packet. Length in packet header %i, "
333  "does not match actual length %i", len, total_len);
334  return NULL;
335  }
336  }
337 
338  /*
339  * If the length is SMALLER, die, too.
340  */
341  if (total_len < len) {
342  fr_strerror_printf("Malformed EAP packet. Length in packet header does not "
343  "match actual length");
344  return NULL;
345  }
346 
347  /*
348  * Now that we know the lengths are OK, allocate memory.
349  */
350  eap_packet = (eap_packet_raw_t *) talloc_zero_array(ctx, uint8_t, len);
351  if (!eap_packet) return NULL;
352 
353  /*
354  * Copy the data from EAP-Message's over to our EAP packet.
355  */
356  ptr = (unsigned char *)eap_packet;
357 
358  /* RADIUS ensures order of attrs, so just concatenate all */
359  for (vp = fr_cursor_head(&cursor);
360  vp;
361  vp = fr_cursor_next(&cursor)) {
362  memcpy(ptr, vp->vp_strvalue, vp->vp_length);
363  ptr += vp->vp_length;
364  }
365 
366  if (!eap_is_valid(&eap_packet)) {
367  talloc_free(eap_packet);
368  return NULL;
369  }
370 
371  return eap_packet;
372 }

389 /** Run a subrequest through a virtual server, managing the eap_session_t of the child
390  *
391  * If eap_session_t has a child, inject that into the request.
392  *
393  * If after the request has run, the child eap_session_t is no longer present,
394  * we assume it has been freed, and fixup the parent eap_session_t.
395  *
396  * If the eap_session_t pointer changes, this is considered a fatal error.
397  *
398  * @param[in] request the current (real) request.
399  * @param[in] eap_session representing the outer eap method.
400  * @param[in] virtual_server The default virtual server to send the request to.
401  * @return the rcode of the last executed section in the virtual server.
402  */
403 rlm_rcode_t eap_virtual_server(request_t *request, eap_session_t *eap_session, char const *virtual_server)
404 {
405  eap_session_t *eap_session_inner;
406  rlm_rcode_t rcode;
407  fr_pair_t *vp;
408 
409  vp = fr_pair_find_by_da(&request->control_pairs, attr_virtual_server);
410  request->server_cs = vp ? virtual_server_find(vp->vp_strvalue) : virtual_server_find(virtual_server);
411 
412  if (request->server_cs) {
413  RDEBUG2("Running request through virtual server \"%s\"", cf_section_name2(request->server_cs));
414  } else {
415  RDEBUG2("Running request in virtual server");
416  }
417 
418  /*
419  * Add a previously recorded inner eap_session_t back
420  * to the request. This in theory allows infinite
421  * nesting, but this is probably limited somewhere.
422  */
423  if (eap_session->child) {
424  RDEBUG4("Adding eap_session_t %p to child request", eap_session->child);
425  request_data_talloc_add(request, NULL, REQUEST_DATA_EAP_SESSION,
426  eap_session_t, eap_session->child, false, false, false);
427  }
428 
429  rad_virtual_server(&rcode, request);
430  eap_session_inner = request_data_get(request, NULL, REQUEST_DATA_EAP_SESSION);
431  if (eap_session_inner) {
432  /*
433  * We assume if the inner eap session has changed
434  * then the old one has been freed.
435  */
436  if (!eap_session->child || (eap_session->child != eap_session_inner)) {
437  RDEBUG4("Binding lifetime of child eap_session %p to parent eap_session %p",
438  eap_session_inner, eap_session);
439  talloc_link_ctx(eap_session, eap_session_inner);
440  eap_session->child = eap_session_inner;
441  } else {
442  RDEBUG4("Got eap_session_t %p back unmolested", eap_session->child);
443  }
444  /*
445  * Assume the inner server freed the
446  * eap_session_t and remove our reference to it.
447  *
448  * If it didn't actually free the child (due to error)
449  * the call to talloc_link_ctx (above) ensures it will
450  * be freed when the parent is.
451  */
452  } else if (eap_session->child) {
453  RDEBUG4("Inner server freed eap_session %p", eap_session->child);
454  eap_session->child = NULL;
455  }
456 
457  return rcode;
458 }
