47 char const *rlm_krb5_error(rlm_krb5_t const *inst, krb5_context context, krb5_error_code code)
48 {
49  char const *msg;
50  char *buffer;
51 
52  if (!fr_cond_assert(inst)) return NULL;
53 
54  buffer = krb5_error_buffer;
55  if (!buffer) {
56  buffer = talloc_array(NULL, char, KRB5_STRERROR_BUFSIZE);
57  if (!buffer) {
58  ERROR("Failed allocating memory for krb5 error buffer");
59  return NULL;
60  }
61 
62  fr_thread_local_set_destructor(krb5_error_buffer, _krb5_logging_free, buffer);
63  }
64 
65  msg = krb5_get_error_message(context, code);
66  if (msg) {
67  strlcpy(buffer, msg, KRB5_STRERROR_BUFSIZE);
68 # ifdef HAVE_KRB5_FREE_ERROR_MESSAGE
69  krb5_free_error_message(context, msg);
70 # elif defined(HAVE_KRB5_FREE_ERROR_STRING)
71  {
72  char *free;
73 
74  memcpy(&free, &msg, sizeof(free));
75  krb5_free_error_string(context, free);
76  }
77 # else
78 # error "No way to free error strings, missing krb5_free_error_message() and krb5_free_error_string()"
79 # endif
80  } else {
81  strlcpy(buffer, "Unknown error", KRB5_STRERROR_BUFSIZE);
82  }
83 
84  return buffer;
85 }

95 static int _mod_conn_free(rlm_krb5_handle_t *conn) {
96  krb5_free_context(conn->context);

98  if (conn->keytab) krb5_kt_close(conn->context, conn->keytab);

107 /** Create and return a new connection
108  *
109  * libkrb5(s) can talk to the KDC over TCP. Were assuming something sane is implemented
110  * by libkrb5 and that it does connection caching associated with contexts, so it's
111  * worth using a connection pool to preserve connections when workers die.
112  */
113 void *krb5_mod_conn_create(TALLOC_CTX *ctx, void *instance, UNUSED fr_time_delta_t timeout)
114 {
115  rlm_krb5_t const *inst = talloc_get_type_abort_const(instance, rlm_krb5_t);
116  rlm_krb5_handle_t *conn;
117  krb5_error_code ret;
118 
119  MEM(conn = talloc_zero(ctx, rlm_krb5_handle_t));
120  ret = krb5_init_context(&conn->context);
121  if (ret) {
122  ERROR("Context initialisation failed: %s", rlm_krb5_error(inst, NULL, ret));
123 
124  return NULL;
125  }
126  talloc_set_destructor(conn, _mod_conn_free);
127 
128  ret = inst->keytabname ?
129  krb5_kt_resolve(conn->context, inst->keytabname, &conn->keytab) :
130  krb5_kt_default(conn->context, &conn->keytab);
131  if (ret) {
132  ERROR("Resolving keytab failed: %s", rlm_krb5_error(inst, conn->context, ret));
133 
134  goto cleanup;
135  }
136 
137 #ifdef HEIMDAL_KRB5
138  ret = krb5_cc_new_unique(conn->context, "MEMORY", NULL, &conn->ccache);
139  if (ret) {
140  ERROR("Credential cache creation failed: %s", rlm_krb5_error(inst, conn->context, ret));
141 
142  return NULL;
143  }
144 
145  krb5_verify_opt_init(&conn->options);
146  krb5_verify_opt_set_ccache(&conn->options, conn->ccache);
147 
148  krb5_verify_opt_set_keytab(&conn->options, conn->keytab);
149  krb5_verify_opt_set_secure(&conn->options, true);
150 
151  if (inst->service) krb5_verify_opt_set_service(&conn->options, inst->service);
152 #else
153  krb5_verify_init_creds_opt_set_ap_req_nofail(inst->vic_options, true);
154 #endif
155  return conn;
156 
157 cleanup:
158  talloc_free(conn);
159  return NULL;
160 }
