62  * Callback sent to libunbound for xlat functions. Simply links the
63  * new ub_result via a pointer that has been allocated from the heap.
64  * This pointer has been pre-initialized to a magic value.

66 static void link_ubres(void *my_arg, int err, struct ub_result *result)
67 {
68  struct ub_result **ubres = (struct ub_result **)my_arg;
69 
70  /*
71  * Note that while result will be NULL on error, we are explicit
72  * here because that is actually a behavior that is suboptimal
73  * and only documented in the examples. It could change.
74  */
75  if (err) {
76  ERROR("%s", ub_strerror(err));
77  *ubres = NULL;
78  } else {
79  *ubres = result;
80  }

83 /*
84  * Convert labels as found in a DNS result to a NULL terminated string.
85  *
86  * Result is written to memory pointed to by "out" but no result will
87  * be written unless it and its terminating NULL character fit in "left"
88  * bytes. Returns the number of bytes written excluding the terminating
89  * NULL, or -1 if nothing was written because it would not fit or due
90  * to a violation in the labels format.
91  */
92 static int rrlabels_tostr(char *out, char *rr, size_t left)
93 {
94  int offset = 0;
95 
96  /*
97  * TODO: verify that unbound results (will) always use this label
98  * format, and review the specs on this label format for nuances.
99  */
100 
101  if (!left) {
102  return -1;
103  }
104  if (left > 253) {
105  left = 253; /* DNS length limit */
106  }
107  /* As a whole this should be "NULL terminated" by the 0-length label */
108  if (strnlen(rr, left) > left - 1) {
109  return -1;
110  }
111 
112  /* It will fit, but does it it look well formed? */
113  while (1) {
114  size_t count;
115 
116  count = *((unsigned char *)(rr + offset));
117  if (!count) break;
118 
119  offset++;
120  if (count > 63 || strlen(rr + offset) < count) {
121  return -1;
122  }
123  offset += count;
124  }
125 
126  /* Data is valid and fits. Copy it. */
127  offset = 0;
128  while (1) {
129  int count;
130 
131  count = *((unsigned char *)(rr));
132  if (!count) break;
133 
134  if (offset) {
135  *(out + offset) = '.';
136  offset++;
137  }
138 
139  rr++;
140  memcpy(out + offset, rr, count);
141  rr += count;
142  offset += count;
143  }
144 
145  *(out + offset) = '\0';
146  return offset;
147 }

149 static int ub_common_wait(rlm_unbound_t const *inst, request_t *request,
150  char const *name, struct ub_result **ub, int async_id)
151 {
152  useconds_t iv, waited;
153 
154  iv = inst->timeout > 64 ? 64000 : inst->timeout * 1000;
155  ub_process(inst->ub);
156 
157  for (waited = 0; (void const *)*ub == (void const *)inst; waited += iv, iv *= 2) {
158 
159  if (waited + iv > (useconds_t)inst->timeout * 1000) {
160  usleep(inst->timeout * 1000 - waited);
161  ub_process(inst->ub);
162  break;
163  }
164 
165  usleep(iv);
166 
167  /* Check if already handled by event loop */
168  if ((void const *)*ub != (void const *)inst) {
169  break;
170  }
171 
172  /* In case we are running single threaded */
173  ub_process(inst->ub);
174  }
175 
176  if ((void const *)*ub == (void const *)inst) {
177  int res;
178 
179  REDEBUG2("%s - DNS took too long", name);
180 
181  res = ub_cancel(inst->ub, async_id);
182  if (res) {
183  REDEBUG("%s - ub_cancel: %s", name, ub_strerror(res));
184  }
185  return -1;
186  }
187 
188  return 0;
189 }

215 static ssize_t xlat_a(TALLOC_CTX *ctx, char **out, size_t outlen,
216  void const *mod_inst, UNUSED void const *xlat_inst,
217  request_t *request, char const *fmt)
218 {
219  rlm_unbound_t const *inst = mod_inst;
220  struct ub_result **ubres;
221  int async_id;
222  char *fmt2; /* For const warnings. Keep till new libunbound ships. */
223 
224  /* This has to be on the heap, because threads. */
225  ubres = talloc(inst, struct ub_result *);
226 
227  /* Used and thus impossible value from heap to designate incomplete */
228  memcpy(ubres, &mod_inst, sizeof(*ubres));
229 
230  fmt2 = talloc_typed_strdup(ctx, fmt);
231  ub_resolve_async(inst->ub, fmt2, 1, 1, ubres, link_ubres, &async_id);
232  talloc_free(fmt2);
233 
234  if (ub_common_wait(inst, request, inst->xlat_a_name, ubres, async_id)) {
235  goto error0;
236  }
237 
238  if (*ubres) {
239  if (ub_common_fail(request, inst->xlat_a_name, *ubres)) {
240  goto error1;
241  }
242 
243  if (!inet_ntop(AF_INET, (*ubres)->data[0], *out, outlen)) {
244  goto error1;
245  };
246 
247  ub_resolve_free(*ubres);
248  talloc_free(ubres);
249  return strlen(*out);
250  }
251 
252  RWDEBUG("%s - No result", inst->xlat_a_name);
253 
254  error1:
255  ub_resolve_free(*ubres); /* Handles NULL gracefully */
256 
257  error0:
258  talloc_free(ubres);
259  return -1;
260 }

382 static int mod_instantiate(void *instance, CONF_SECTION *conf)
383 {
384  rlm_unbound_t *inst = instance;
385  int res;
386  char k[64]; /* To silence const warns until newer unbound in distros */
387 
388  /*
389  * @todo - move this to the thread-instantiate function
390  */
391  inst->ub = ub_ctx_create();
392  if (!inst->ub) {
393  cf_log_err(conf, "ub_ctx_create failed");
394  return -1;
395  }
396 
397  /*
398  * Note unbound threads WILL happen with -s option, if it matters.
399  * We cannot tell from here whether that option is in effect.
400  */
401  res = ub_ctx_async(inst->ub, 1);
402  if (res) goto error;
403 
404  /* Now load the config file, which can override gleaned settings. */
405  {
406  char *file;
407 
408  memcpy(&file, &inst->filename, sizeof(file));
409  res = ub_ctx_config(inst->ub, file);
410  if (res) goto error;
411  }
412 
413  if (unbound_log_init(inst, &inst->u_log, inst->ub) < 0) goto error;
414 
415  /*
416  * Now we need to finalize the context.
417  *
418  * There's no clean API to just finalize the context made public
419  * in libunbound. But we can trick it by trying to delete data
420  * which as it happens fails quickly and quietly even though the
421  * data did not exist.
422  */
423  strcpy(k, "notar33lsite.foo123.nottld A 127.0.0.1");
424  ub_ctx_data_remove(inst->ub, k);
425  return 0;
426 
427  error:
428  cf_log_err(conf, "%s", ub_strerror(res));
429 
430  return -1;
431 }

433 static int mod_bootstrap(void *instance, CONF_SECTION *conf)
434 {
435  rlm_unbound_t *inst = instance;
436 
437  inst->name = cf_section_name2(conf);
438  if (!inst->name) inst->name = cf_section_name1(conf);
439 
440  if (inst->timeout > 10000) {
441  cf_log_err(conf, "timeout must be 0 to 10000");
442  return -1;
443  }
444 
445  MEM(inst->xlat_a_name = talloc_typed_asprintf(inst, "%s-a", inst->name));
446  MEM(inst->xlat_aaaa_name = talloc_typed_asprintf(inst, "%s-aaaa", inst->name));
447  MEM(inst->xlat_ptr_name = talloc_typed_asprintf(inst, "%s-ptr", inst->name));
448 
449  if (xlat_register_legacy(inst, inst->xlat_a_name, xlat_a, NULL, NULL, 0, XLAT_DEFAULT_BUF_LEN) ||
450  xlat_register_legacy(inst, inst->xlat_aaaa_name, xlat_aaaa, NULL, NULL, 0, XLAT_DEFAULT_BUF_LEN) ||
451  xlat_register_legacy(inst, inst->xlat_ptr_name, xlat_ptr, NULL, NULL, 0, XLAT_DEFAULT_BUF_LEN)) {
452  cf_log_err(conf, "Failed registering xlats");
453  return -1;
454  }
455 
456  return 0;
457 }
