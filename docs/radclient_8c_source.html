170 {

172 
173  prev = request->prev;
174  next = request->next;
175 
176  if (prev) {
177  assert(request_head != request);
178  prev->next = next;
179  } else if (request_head) {
180  assert(request_head == request);
181  request_head = next;
182  }
183 
184  if (next) {
185  assert(rc_request_tail != request);
186  next->prev = prev;
187  } else if (rc_request_tail) {
188  assert(rc_request_tail == request);
189  rc_request_tail = prev;
190  }
191 
192  return 0;
193 }

195 static int mschapv1_encode(fr_radius_packet_t *packet, fr_pair_t **request,
196  char const *password)
197 {
198  unsigned int i;
199  uint8_t *p;
200  fr_pair_t *challenge, *reply;
201  uint8_t nthash[16];
202 
203  fr_pair_delete_by_da(&packet->vps, attr_ms_chap_challenge);
204  fr_pair_delete_by_da(&packet->vps, attr_ms_chap_response);
205 
206  MEM(challenge = fr_pair_afrom_da(packet, attr_ms_chap_challenge));
207 
208  fr_pair_add(request, challenge);
209  challenge->vp_length = 8;
210  challenge->vp_octets = p = talloc_array(challenge, uint8_t, challenge->vp_length);
211  for (i = 0; i < challenge->vp_length; i++) {
212  p[i] = fr_rand();
213  }
214 
215  MEM(reply = fr_pair_afrom_da(packet, attr_ms_chap_response));
216  fr_pair_add(request, reply);
217  reply->vp_length = 50;
218  reply->vp_octets = p = talloc_array(reply, uint8_t, reply->vp_length);
219  memset(p, 0, reply->vp_length);
220 
221  p[1] = 0x01; /* NT hash */
222 
223  if (mschap_nt_password_hash(nthash, password) < 0) return 0;
224 
225  smbdes_mschap(nthash, challenge->vp_octets, p + 26);
226  return 1;
227 }

243 static void radclient_get_port(FR_CODE type, uint16_t *port)
244 {
245  switch (type) {
246  default:
247  case FR_CODE_ACCESS_REQUEST:
248  case FR_CODE_ACCESS_CHALLENGE:
249  case FR_CODE_STATUS_SERVER:
250  if (*port == 0) *port = getport("radius");
251  if (*port == 0) *port = FR_AUTH_UDP_PORT;
252  return;
253 
254  case FR_CODE_ACCOUNTING_REQUEST:
255  if (*port == 0) *port = getport("radacct");
256  if (*port == 0) *port = FR_ACCT_UDP_PORT;
257  return;
258 
259  case FR_CODE_DISCONNECT_REQUEST:
260  if (*port == 0) *port = FR_POD_UDP_PORT;
261  return;
262 
263  case FR_CODE_COA_REQUEST:
264  if (*port == 0) *port = FR_COA_UDP_PORT;
265  return;
266 
267  case FR_CODE_UNDEFINED:
268  if (*port == 0) *port = 0;
269  return;
270  }
271 }

273 /*
274  * Resolve a port to a request type
275  */
276 static FR_CODE radclient_get_code(uint16_t port)
277 {
278  /*
279  * getport returns 0 if the service doesn't exist
280  * so we need to return early, to avoid incorrect
281  * codes.
282  */
283  if (port == 0) return FR_CODE_UNDEFINED;
284 
285  if ((port == getport("radius")) || (port == FR_AUTH_UDP_PORT) || (port == FR_AUTH_UDP_PORT_ALT)) {
286  return FR_CODE_ACCESS_REQUEST;
287  }
288  if ((port == getport("radacct")) || (port == FR_ACCT_UDP_PORT) || (port == FR_ACCT_UDP_PORT_ALT)) {
289  return FR_CODE_ACCOUNTING_REQUEST;
290  }
291  if (port == FR_COA_UDP_PORT) return FR_CODE_COA_REQUEST;
292 
293  return FR_CODE_UNDEFINED;
294 }

297 static bool already_hex(fr_pair_t *vp)
298 {
299  size_t i;
300 
301  if (!vp || (vp->vp_type != FR_TYPE_OCTETS)) return true;
302 
303  /*
304  * If it's 17 octets, it *might* be already encoded.
305  * Or, it might just be a 17-character password (maybe UTF-8)
306  * Check it for non-printable characters. The odds of ALL
307  * of the characters being 32..255 is (1-7/8)^17, or (1/8)^17,
308  * or 1/(2^51), which is pretty much zero.
309  */
310  for (i = 0; i < vp->vp_length; i++) {
311  if (vp->vp_octets[i] < 32) {
312  return true;
313  }
314  }
315 
316  return false;
317 }

319 /*
320  * Initialize a radclient data structure and add it to
321  * the global linked list.
322  */
323 static int radclient_init(TALLOC_CTX *ctx, rc_file_pair_t *files)
324 {
325  FILE *packets, *filters = NULL;
326 
327  fr_cursor_t cursor;
328  fr_pair_t *vp;
329  rc_request_t *request;
330  bool packets_done = false;
331  uint64_t num = 0;
332 
333  assert(files->packets != NULL);
334 
335  /*
336  * Determine where to read the VP's from.
337  */
338  if (strcmp(files->packets, "-") != 0) {
339  packets = fopen(files->packets, "r");
340  if (!packets) {
341  ERROR("Error opening %s: %s", files->packets, fr_syserror(errno));
342  return -1;
343  }
344 
345  /*
346  * Read in the pairs representing the expected response.
347  */
348  if (files->filters) {
349  filters = fopen(files->filters, "r");
350  if (!filters) {
351  ERROR("Error opening %s: %s", files->filters, fr_syserror(errno));
352  fclose(packets);
353  return -1;
354  }
355  }
356  } else {
357  packets = stdin;
358  }
359 
360  /*
361  * Loop until the file is done.
362  */
363  do {
364  /*
365  * Allocate it.
366  */
367  request = talloc_zero(ctx, rc_request_t);
368  if (!request) {
369  ERROR("Out of memory");
370  goto error;
371  }
372 
373  request->packet = fr_radius_alloc(request, true);
374  if (!request->packet) {
375  ERROR("Out of memory");
376  goto error;
377  }
378 
379  request->packet->socket.inet.src_ipaddr = client_ipaddr;
380  request->packet->socket.inet.src_port = client_port;
381  request->packet->socket.inet.dst_ipaddr = server_ipaddr;
382  request->packet->socket.inet.dst_port = server_port;
383  request->packet->socket.proto = ipproto;
384 
385  request->files = files;
386  request->packet->id = last_used_id;
387  request->num = num++;
388 
389  /*
390  * Read the request VP's.
391  */
392  if (fr_pair_list_afrom_file(request->packet, dict_radius,
393  &request->request_pairs, packets, &packets_done) < 0) {
394  char const *input;
395 
396  if ((files->packets[0] == '-') && (files->packets[1] == '\0')) {
397  input = "stdin";
398  } else {
399  input = files->packets;
400  }
401 
402  REDEBUG("Error parsing \"%s\"", input);
403  goto error;
404  }
405 
406  /*
407  * Skip empty entries
408  */
409  if (!request->request_pairs) {
410  WARN("Skipping \"%s\": No Attributes", files->packets);
411  talloc_free(request);
412  continue;
413  }
414 
415  /*
416  * Read in filter VP's.
417  */
418  if (filters) {
419  bool filters_done;
420 
421  if (fr_pair_list_afrom_file(request, dict_radius,
422  &request->filter, filters, &filters_done) < 0) {
423  REDEBUG("Error parsing \"%s\"", files->filters);
424  goto error;
425  }
426 
427  if (filters_done && !packets_done) {
428  REDEBUG("Differing number of packets/filters in %s:%s "
429  "(too many requests))", files->packets, files->filters);
430  goto error;
431  }
432 
433  if (!filters_done && packets_done) {
434  REDEBUG("Differing number of packets/filters in %s:%s "
435  "(too many filters))", files->packets, files->filters);
436  goto error;
437  }
438 
439  /*
440  * xlat expansions aren't supported here
441  */
442  for (vp = fr_cursor_init(&cursor, &request->filter);
443  vp;
444  vp = fr_cursor_next(&cursor)) {
445  again:
446  /*
447  * Xlat expansions are not supported. Convert xlat to value box (if possible).
448  */
449  if (vp->type == VT_XLAT) {
450  fr_type_t type = vp->da->type;
451  if (fr_value_box_from_str(vp, &vp->data, &type, NULL, vp->xlat, -1, '\0', false) < 0) {
452  fr_perror("radclient");
453  goto error;
454  }
455  vp->type = VT_DATA;
456  }
457 
458  if (vp->da == attr_packet_type) {
459  vp = fr_cursor_remove(&cursor); /* so we don't break the filter */
460  request->filter_code = vp->vp_uint32;
461  talloc_free(vp);
462  vp = fr_cursor_current(&cursor);
463  if (!vp) break;
464  goto again;
465  }
466  }
467 
468  /*
469  * This allows efficient list comparisons later
470  */
471  fr_pair_list_sort(&request->filter, fr_pair_cmp_by_da);
472  }
473 
474  /*
475  * Process special attributes
476  */
477  for (vp = fr_cursor_init(&cursor, &request->request_pairs);
478  vp;
479  vp = fr_cursor_next(&cursor)) {
480  /*
481  * Xlat expansions are not supported. Convert xlat to value box (if possible).
482  */
483  if (vp->type == VT_XLAT) {
484  fr_type_t type = vp->da->type;
485  if (fr_value_box_from_str(vp, &vp->data, &type, NULL, vp->xlat, -1, '\0', false) < 0) {
486  fr_perror("radclient");
487  goto error;
488  }
489  vp->type = VT_DATA;
490  }
491 
492  /*
493  * Allow it to set the packet type in
494  * the attributes read from the file.
495  */
496  if (vp->da == attr_packet_type) {
497  request->packet->code = vp->vp_uint32;
498  } else if (vp->da == attr_packet_dst_port) {
499  request->packet->socket.inet.dst_port = vp->vp_uint16;
500  } else if ((vp->da == attr_packet_dst_ip_address) ||
501  (vp->da == attr_packet_dst_ipv6_address)) {
502  memcpy(&request->packet->socket.inet.dst_ipaddr, &vp->vp_ip, sizeof(request->packet->socket.inet.dst_ipaddr));
503  } else if (vp->da == attr_packet_src_port) {
504  if (vp->vp_uint16 < 1024) {
505  ERROR("Invalid value '%u' for Packet-Src-Port", vp->vp_uint16);
506  goto error;
507  }
508  request->packet->socket.inet.src_port = vp->vp_uint16;
509  } else if ((vp->da == attr_packet_src_ip_address) ||
510  (vp->da == attr_packet_src_ipv6_address)) {
511  memcpy(&request->packet->socket.inet.src_ipaddr, &vp->vp_ip, sizeof(request->packet->socket.inet.src_ipaddr));
512  } else if (vp->da == attr_request_authenticator) {
513  if (vp->vp_length > sizeof(request->packet->vector)) {
514  memcpy(request->packet->vector, vp->vp_octets, sizeof(request->packet->vector));
515  } else {
516  memset(request->packet->vector, 0, sizeof(request->packet->vector));
517  memcpy(request->packet->vector, vp->vp_octets, vp->vp_length);
518  }
519  } else if (vp->da == attr_cleartext_password) {
520  request->password = vp;
521  /*
522  * Keep a copy of the the password attribute.
523  */
524  } else if (vp->da == attr_chap_password) {
525  /*
526  * If it's already hex, do nothing.
527  */
528  if ((vp->vp_length == 17) && (already_hex(vp))) continue;
529 
530  /*
531  * CHAP-Password is octets, so it may not be zero terminated.
532  */
533  MEM(pair_update_request(&request->password, attr_cleartext_password) >= 0);
534  fr_pair_value_bstrndup(request->password, vp->vp_strvalue, vp->vp_length, true);
535  } else if ((vp->da == attr_user_password) ||
536  (vp->da == attr_ms_chap_password)) {
537  MEM(pair_update_request(&request->password, attr_cleartext_password) >= 0);
538  fr_pair_value_bstrndup(request->password, vp->vp_strvalue, vp->vp_length, true);
539  } else if (vp->da == attr_radclient_test_name) {
540  request->name = vp->vp_strvalue;
541  }
542  } /* loop over the VP's we read in */
543 
544  /*
545  * Use the default set on the command line
546  */
547  if (request->packet->code == FR_CODE_UNDEFINED) request->packet->code = packet_code;
548 
549  /*
550  * Default to the filename
551  */
552  if (!request->name) request->name = request->files->packets;
553 
554  /*
555  * Automatically set the response code from the request code
556  * (if one wasn't already set).
557  */
558  if (request->filter_code == FR_CODE_UNDEFINED) {
559  switch (request->packet->code) {
560  case FR_CODE_ACCESS_REQUEST:
561  request->filter_code = FR_CODE_ACCESS_ACCEPT;
562  break;
563 
564  case FR_CODE_ACCOUNTING_REQUEST:
565  request->filter_code = FR_CODE_ACCOUNTING_RESPONSE;
566  break;
567 
568  case FR_CODE_COA_REQUEST:
569  request->filter_code = FR_CODE_COA_ACK;
570  break;
571 
572  case FR_CODE_DISCONNECT_REQUEST:
573  request->filter_code = FR_CODE_DISCONNECT_ACK;
574  break;
575 
576  case FR_CODE_STATUS_SERVER:
577  switch (radclient_get_code(request->packet->socket.inet.dst_port)) {
578  case FR_CODE_ACCESS_REQUEST:
579  request->filter_code = FR_CODE_ACCESS_ACCEPT;
580  break;
581 
582  case FR_CODE_ACCOUNTING_REQUEST:
583  request->filter_code = FR_CODE_ACCOUNTING_RESPONSE;
584  break;
585 
586  default:
587  request->filter_code = FR_CODE_UNDEFINED;
588  break;
589  }
590  break;
591 
592  case FR_CODE_UNDEFINED:
593  REDEBUG("Packet-Type must be defined,"
594  "or a well known RADIUS port");
595  goto error;
596 
597  default:
598  REDEBUG("Can't determine expected &reply.Packet-Type for Packet-Type %i",
599  request->packet->code);
600  goto error;
601  }
602  /*
603  * Automatically set the request code from the response code
604  * (if one wasn't already set).
605  */
606  } else if (request->packet->code == FR_CODE_UNDEFINED) {
607  switch (request->filter_code) {
608  case FR_CODE_ACCESS_ACCEPT:
609  case FR_CODE_ACCESS_REJECT:
610  request->packet->code = FR_CODE_ACCESS_REQUEST;
611  break;
612 
613  case FR_CODE_ACCOUNTING_RESPONSE:
614  request->packet->code = FR_CODE_ACCOUNTING_REQUEST;
615  break;
616 
617  case FR_CODE_DISCONNECT_ACK:
618  case FR_CODE_DISCONNECT_NAK:
619  request->packet->code = FR_CODE_DISCONNECT_REQUEST;
620  break;
621 
622  case FR_CODE_COA_ACK:
623  case FR_CODE_COA_NAK:
624  request->packet->code = FR_CODE_COA_REQUEST;
625  break;
626 
627  default:
628  REDEBUG("Can't determine expected Packet-Type for &reply.Packet-Type %i",
629  request->filter_code);
630  goto error;
631  }
632  }
633 
634  /*
635  * Automatically set the dst port (if one wasn't already set).
636  */
637  if (request->packet->socket.inet.dst_port == 0) {
638  radclient_get_port(request->packet->code, &request->packet->socket.inet.dst_port);
639  if (request->packet->socket.inet.dst_port == 0) {
640  REDEBUG("Can't determine destination port");
641  goto error;
642  }
643  }
644 
645  /*
646  * Add it to the tail of the list.
647  */
648  if (!request_head) {
649  assert(rc_request_tail == NULL);
650  request_head = request;
651  request->prev = NULL;
652  } else {
653  assert(rc_request_tail->next == NULL);
654  rc_request_tail->next = request;
655  request->prev = rc_request_tail;
656  }
657  rc_request_tail = request;
658  request->next = NULL;
659 
660  /*
661  * Set the destructor so it removes itself from the
662  * request list when freed. We don't set this until
663  * the packet is actually in the list, else we trigger
664  * the asserts in the free callback.
665  */
666  talloc_set_destructor(request, _rc_request_free);
667  } while (!packets_done); /* loop until the file is done. */
668 
669  if (packets != stdin) fclose(packets);
670  if (filters) fclose(filters);
671 
672  /*
673  * And we're done.
674  */
675  return 0;
676 
677 error:
678  talloc_free(request);
679 
680  if (packets != stdin) fclose(packets);
681  if (filters) fclose(filters);
682 
683  return -1;
684 }

765 /*
766  * Send one packet.
767  */
768 static int send_one_packet(rc_request_t *request)
769 {
770  assert(request->done == false);
771 
772  /*
773  * Remember when we have to wake up, to re-send the
774  * request, of we didn't receive a reply.
775  */
776  if ((sleep_time == -1) || (sleep_time > timeout)) sleep_time = timeout;
777 
778  /*
779  * Haven't sent the packet yet. Initialize it.
780  */
781  if (!request->tries || request->packet->id == -1) {
782  bool rcode;
783 
784  assert(request->reply == NULL);
785 
786  /*
787  * Didn't find a free packet ID, we're not done,
788  * we don't sleep, and we stop trying to process
789  * this packet.
790  */
791  retry:
792  request->packet->socket.inet.src_ipaddr.af = server_ipaddr.af;
793  rcode = fr_packet_list_id_alloc(packet_list, ipproto, &request->packet, NULL);
794  if (!rcode) {
795  int mysockfd;
796 
797  if (ipproto == IPPROTO_TCP) {
798  mysockfd = fr_socket_client_tcp(NULL,
799  &request->packet->socket.inet.dst_ipaddr,
800  request->packet->socket.inet.dst_port, false);
801  if (mysockfd < 0) {
802  fr_perror("Error opening socket");
803  return -1;
804  }
805  } else {
806  uint16_t port = 0;
807 
808  mysockfd = fr_socket_server_udp(&client_ipaddr, &port, NULL, true);
809  if (mysockfd < 0) {
810  fr_perror("Error opening socket");
811  return -1;
812  }
813 
814  if (fr_socket_bind(mysockfd, &client_ipaddr, &port, NULL) < 0) {
815  fr_perror("Error binding socket");
816  return -1;
817  }
818  }
819 
820  if (!fr_packet_list_socket_add(packet_list, mysockfd, ipproto,
821  &request->packet->socket.inet.dst_ipaddr,
822  request->packet->socket.inet.dst_port, NULL)) {
823  ERROR("Can't add new socket");
824  fr_exit_now(1);
825  }
826  goto retry;
827  }
828 
829  assert(request->packet->id != -1);
830  assert(request->packet->data == NULL);
831 
832  /*
833  * Update the password, so it can be encrypted with the
834  * new authentication vector.
835  */
836  if (request->password) {
837  fr_pair_t *vp;
838 
839  if ((vp = fr_pair_find_by_da(&request->request_pairs, attr_user_password)) != NULL) {
840  fr_pair_value_strdup(vp, request->password->vp_strvalue);
841 
842  } else if ((vp = fr_pair_find_by_da(&request->request_pairs,
843  attr_chap_password)) != NULL) {
844  uint8_t buffer[17];
845  fr_pair_t *challenge;
846  uint8_t const *vector;
847 
848  /*
849  * Use Chap-Challenge pair if present,
850  * Request Authenticator otherwise.
851  */
852  challenge = fr_pair_find_by_da(&request->request_pairs, attr_chap_challenge);
853  if (challenge && (challenge->vp_length == RADIUS_AUTH_VECTOR_LENGTH)) {
854  vector = challenge->vp_octets;
855  } else {
856  vector = request->packet->vector;
857  }
858 
859  fr_radius_encode_chap_password(buffer,
860  fr_rand() & 0xff, vector,
861  request->password->vp_strvalue,
862  request->password->vp_length);
863  fr_pair_value_memdup(vp, buffer, sizeof(buffer), false);
864 
865  } else if (fr_pair_find_by_da(&request->request_pairs, attr_ms_chap_password) != NULL) {
866  mschapv1_encode(request->packet, &request->request_pairs, request->password->vp_strvalue);
867 
868  } else {
869  DEBUG("WARNING: No password in the request");
870  }
871  }
872 
873  request->timestamp = fr_time();
874  request->tries = 1;
875  request->resend++;
876 
877  } else { /* request->packet->id >= 0 */
878  fr_time_delta_t now = fr_time();
879 
880  /*
881  * FIXME: Accounting packets are never retried!
882  * The Acct-Delay-Time attribute is updated to
883  * reflect the delay, and the packet is re-sent
884  * from scratch!
885  */
886 
887  /*
888  * Not time for a retry, do so.
889  */
890  if ((now - request->timestamp) < timeout) {
891  /*
892  * When we walk over the tree sending
893  * packets, we update the minimum time
894  * required to sleep.
895  */
896  if ((sleep_time == -1) ||
897  (sleep_time > (now - request->timestamp))) {
898  sleep_time = now - request->timestamp;
899  }
900  return 0;
901  }
902 
903  /*
904  * We're not trying later, maybe the packet is done.
905  */
906  if (request->tries == retries) {
907  assert(request->packet->id >= 0);
908 
909  /*
910  * Delete the request from the tree of
911  * outstanding requests.
912  */
913  fr_packet_list_yank(packet_list, request->packet);
914 
915  REDEBUG("No reply from server for ID %d socket %d",
916  request->packet->id, request->packet->socket.fd);
917  deallocate_id(request);
918 
919  /*
920  * Normally we mark it "done" when we've received
921  * the reply, but this is a special case.
922  */
923  if (request->resend == resend_count) {
924  request->done = true;
925  }
926  stats.lost++;
927  return -1;
928  }
929 
930  /*
931  * We are trying later.
932  */
933  request->timestamp = now;
934  request->tries++;
935  }
936 
937  /*
938  * Send the packet.
939  */
940  if (fr_radius_packet_send(request->packet, NULL, secret) < 0) {
941  REDEBUG("Failed to send packet for ID %d", request->packet->id);
942  deallocate_id(request);
943  request->done = true;
944  return -1;
945  }
946 
947  fr_packet_log(&default_log, request->packet, false);
948 
949  return 0;
950 }

952 /*
953  * Receive one packet, maybe.
954  */
955 static int recv_one_packet(fr_time_t wait_time)
956 {
957  fd_set set;
958  fr_time_delta_t our_wait_time;
959  rc_request_t *request;
960  fr_radius_packet_t *reply, **packet_p;
961  volatile int max_fd;
962 
963  /* And wait for reply, timing out as necessary */
964  FD_ZERO(&set);
965 
966  max_fd = fr_packet_list_fd_set(packet_list, &set);
967  if (max_fd < 0) fr_exit_now(1); /* no sockets to listen on! */
968 
969  our_wait_time = (wait_time <= 0) ? 0 : wait_time;
970 
971  /*
972  * No packet was received.
973  */
974  if (select(max_fd, &set, NULL, NULL, &fr_time_delta_to_timeval(our_wait_time)) <= 0) return 0;
975 
976  /*
977  * Look for the packet.
978  */
979  reply = fr_packet_list_recv(packet_list, &set, RADIUS_MAX_ATTRIBUTES, false);
980  if (!reply) {
981  ERROR("Received bad packet");
982  /*
983  * If the packet is bad, we close the socket.
984  * I'm not sure how to do that now, so we just
985  * die...
986  */
987  if (ipproto == IPPROTO_TCP) fr_exit_now(1);
988  return -1; /* bad packet */
989  }
990 
991  /*
992  * We don't use udpfromto. So if we bind to "*", we want
993  * to find replies sent to 192.0.2.4. Therefore, we
994  * force all replies to have the one address we know
995  * about, no matter what real address they were sent to.
996  *
997  * This only works if were not using any of the
998  * Packet-* attributes, or running with 'auto'.
999  */
1000  reply->socket.inet.dst_ipaddr = client_ipaddr;
1001  reply->socket.inet.dst_port = client_port;
1002 
1003  /*
1004  * TCP sockets don't use recvmsg(), and thus don't get
1005  * the source IP/port. However, since they're TCP, we
1006  * know what the source IP/port is, because that's where
1007  * we connected to.
1008  */
1009  if (ipproto == IPPROTO_TCP) {
1010  reply->socket.inet.src_ipaddr = server_ipaddr;
1011  reply->socket.inet.src_port = server_port;
1012  }
1013 
1014  packet_p = fr_packet_list_find_byreply(packet_list, reply);
1015  if (!packet_p) {
1016  ERROR("Received reply to request we did not send. (id=%d socket %d)",
1017  reply->id, reply->socket.fd);
1018  fr_radius_packet_free(&reply);
1019  return -1; /* got reply to packet we didn't send */
1020  }
1021  request = fr_packet2myptr(rc_request_t, packet, packet_p);
1022 
1023  /*
1024  * Fails the signature validation: not a real reply.
1025  * FIXME: Silently drop it and listen for another packet.
1026  */
1027  if (fr_radius_packet_verify(reply, request->packet, secret) < 0) {
1028  REDEBUG("Reply verification failed");
1029  stats.lost++;
1030  goto packet_done; /* shared secret is incorrect */
1031  }
1032 
1033  if (print_filename) {
1034  RDEBUG("%s response code %d", request->files->packets, reply->code);
1035  }
1036 
1037  deallocate_id(request);
1038  request->reply = reply;
1039  reply = NULL;
1040 
1041  /*
1042  * If this fails, we're out of memory.
1043  */
1044  if (fr_radius_packet_decode(request->reply, request->packet, RADIUS_MAX_ATTRIBUTES, false, secret) != 0) {
1045  REDEBUG("Reply decode failed");
1046  stats.lost++;
1047  goto packet_done;
1048  }
1049 
1050  fr_packet_log(&default_log, request->reply, true);
1051 
1052  /*
1053  * Increment counters...
1054  */
1055  switch (request->reply->code) {
1056  case FR_CODE_ACCESS_ACCEPT:
1057  case FR_CODE_ACCOUNTING_RESPONSE:
1058  case FR_CODE_COA_ACK:
1059  case FR_CODE_DISCONNECT_ACK:
1060  stats.accepted++;
1061  break;
1062 
1063  case FR_CODE_ACCESS_CHALLENGE:
1064  break;
1065 
1066  default:
1067  stats.rejected++;
1068  }
1069 
1070  fr_strerror(); /* Clear strerror buffer */
1071 
1072  /*
1073  * If we had an expected response code, check to see if the
1074  * packet matched that.
1075  */
1076  if ((request->filter_code != FR_CODE_UNDEFINED) && (request->reply->code != request->filter_code)) {
1077  if (is_radius_code(request->reply->code)) {
1078  REDEBUG("%s: Expected %s got %s", request->name, fr_packet_codes[request->filter_code],
1079  fr_packet_codes[request->reply->code]);
1080  } else {
1081  REDEBUG("%s: Expected %u got %i", request->name, request->filter_code,
1082  request->reply->code);
1083  }
1084  stats.failed++;
1085  /*
1086  * Check if the contents of the packet matched the filter
1087  */
1088  } else if (!request->filter) {
1089  stats.passed++;
1090  } else {
1091  fr_pair_t const *failed[2];
1092 
1093  fr_pair_list_sort(&request->reply_pairs, fr_pair_cmp_by_da);
1094  if (fr_pair_validate(failed, &request->filter, &request->reply_pairs)) {
1095  RDEBUG("%s: Response passed filter", request->name);
1096  stats.passed++;
1097  } else {
1098  fr_pair_validate_debug(request, failed);
1099  REDEBUG("%s: Response for failed filter", request->name);
1100  stats.failed++;
1101  }
1102  }
1103 
1104  if (request->resend == resend_count) {
1105  request->done = true;
1106  }
1107 
1108 packet_done:
1109  fr_radius_packet_free(&request->reply);
1110  fr_radius_packet_free(&reply); /* may be NULL */
1111 
1112  return 0;
1113 }
