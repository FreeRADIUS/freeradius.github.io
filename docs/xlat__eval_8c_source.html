97 /** Reconstruct the original expansion string from an xlat tree
98  *
99  * @param[in] ctx to allocate result in.
100  * @param[in] node in the tree to start printing.
101  * @return
102  * - The original expansion string on success.
103  * - NULL on error.
104  */
105 static char *xlat_fmt_aprint(TALLOC_CTX *ctx, xlat_exp_t const *node)
106 {
107  switch (node->type) {
108  case XLAT_LITERAL:
109  case XLAT_GROUP:
110  return talloc_asprintf(ctx, "%s", node->fmt);
111 
112  case XLAT_ONE_LETTER:
113  return talloc_asprintf(ctx, "%%%s", node->fmt);
114 
115  case XLAT_ATTRIBUTE:
116  return talloc_asprintf(ctx, "%%{%s}", node->fmt);
117 
118  case XLAT_VIRTUAL:
119  return talloc_asprintf(ctx, "%%{%s}", node->call.func->name);;
120 
121  case XLAT_FUNC:
122  {
123  xlat_exp_t const *child = node->child;
124  char *out, *n_out;
125  TALLOC_CTX *pool;
126 
127  if (!child) return talloc_asprintf(ctx, "%%{%s:}", node->call.func->name);
128 
129  out = talloc_asprintf(ctx, "%%{%s:", node->call.func->name);
130  pool = talloc_pool(NULL, 128); /* Size of a single child (probably ok...) */
131  do {
132  char *child_str;
133 
134  child_str = xlat_fmt_aprint(pool, child);
135  if (child_str) {
136  n_out = talloc_buffer_append_buffer(ctx, out, child_str);
137  if (!n_out) {
138  talloc_free(out);
139  talloc_free(pool);
140  return NULL;
141  }
142  out = n_out;
143  }
144  talloc_free_children(pool); /* Clear pool contents */
145  } while ((child = child->next));
146  talloc_free(pool);
147 
148  n_out = talloc_strdup_append_buffer(out, "}");
149  if (!n_out) {
150  talloc_free(out);
151  return NULL;
152  }
153  return n_out;
154  }
155 
156  case XLAT_ALTERNATE:
157  {
158  char *first, *second, *result;
159 
160  first = xlat_fmt_aprint(NULL, node->child);
161  second = xlat_fmt_aprint(NULL, node->alternate);
162  result = talloc_asprintf(ctx, "%%{%s:-%s}", first, second);
163  talloc_free(first);
164  talloc_free(second);
165 
166  return result;
167  }
168 
169  default:
170  return NULL;
171  }
172 }

213 /** One letter expansions
214  *
215  * @param[in] ctx to allocate boxed value, and buffers in.
216  * @param[out] out Where to write the boxed value.
217  * @param[in] request The current request.
218  * @param[in] letter to expand.
219  * @return
220  * - #XLAT_ACTION_FAIL on memory allocation errors.
221  * - #XLAT_ACTION_DONE if we're done processing this node.
222  *
223  */
224 static xlat_action_t xlat_eval_one_letter(TALLOC_CTX *ctx, fr_cursor_t *out, request_t *request, char letter)
225 {
226 
227  char buffer[64];
228  struct tm ts;
229  time_t now;
230  fr_value_box_t *value;
231 
232  now = fr_time_to_sec(request->packet->timestamp);
233 
234  switch (letter) {
235  case '%':
236  MEM(value = fr_value_box_alloc_null(ctx));
237  if (fr_value_box_strdup(value, value, NULL, "%", false) < 0) return XLAT_ACTION_FAIL;
238  break;
239 
240  /*
241  * RADIUS request values
242  */
243 
244  case 'I': /* Request ID */
245  MEM(value = fr_value_box_alloc(ctx, FR_TYPE_UINT32, NULL, false));
246  value->datum.uint32 = request->packet->id;
247  break;
248 
249  case 'n': /* Request number */
250  MEM(value = fr_value_box_alloc(ctx, FR_TYPE_UINT64, NULL, false));
251  value->datum.uint64 = request->number;
252  break;
253 
254  case 's': /* First request in this sequence */
255  MEM(value = fr_value_box_alloc(ctx, FR_TYPE_UINT64, NULL, false));
256  value->datum.uint64 = request->seq_start;
257  break;
258 
259  /*
260  * Current time
261  */
262 
263  case 'c': /* Current epoch time seconds */
264  MEM(value = fr_value_box_alloc(ctx, FR_TYPE_UINT64, NULL, false));
265  value->datum.uint64 = (uint64_t)fr_time_to_sec(fr_time());
266  break;
267 
268  case 'C': /* Current epoch time microsecond component */
269  MEM(value = fr_value_box_alloc(ctx, FR_TYPE_UINT64, NULL, false));
270  value->datum.uint64 = (uint64_t)fr_time_to_usec(fr_time()) % 1000000;
271  break;
272 
273  /*
274  * Time of the current request
275  */
276 
277  case 'd': /* Request day */
278  if (!localtime_r(&now, &ts)) {
279  error:
280  REDEBUG("Failed converting packet timestamp to localtime: %s", fr_syserror(errno));
281  return XLAT_ACTION_FAIL;
282  }
283 
284  MEM(value = fr_value_box_alloc(ctx, FR_TYPE_UINT8, NULL, false));
285  value->datum.uint8 = ts.tm_mday;
286  break;
287 
288  case 'D': /* Request date */
289  if (!localtime_r(&now, &ts)) goto error;
290 
291  strftime(buffer, sizeof(buffer), "%Y%m%d", &ts);
292 
293  MEM(value = fr_value_box_alloc_null(ctx));
294  if (fr_value_box_strdup(value, value, NULL, buffer, false) < 0) goto error;
295  break;
296 
297  case 'e': /* Request second */
298  if (!localtime_r(&now, &ts)) goto error;
299 
300  MEM(value = fr_value_box_alloc(ctx, FR_TYPE_UINT8, NULL, false));
301  value->datum.uint8 = ts.tm_sec;
302  break;
303 
304  case 'G': /* Request minute */
305  if (!localtime_r(&now, &ts)) goto error;
306 
307  MEM(value = fr_value_box_alloc(ctx, FR_TYPE_UINT8, NULL, false));
308  value->datum.uint8 = ts.tm_min;
309  break;
310 
311  case 'H': /* Request hour */
312  if (!localtime_r(&now, &ts)) goto error;
313 
314  MEM(value = fr_value_box_alloc(ctx, FR_TYPE_UINT8, NULL, false));
315  value->datum.uint8 = ts.tm_hour;
316  break;
317 
318  case 'l': /* Request timestamp as a 32-bit integer */
319  MEM(value = fr_value_box_alloc(ctx, FR_TYPE_UINT32, NULL, false));
320  value->datum.uint32 = (uint32_t ) now;
321  break;
322 
323  case 'm': /* Request month */
324  if (!localtime_r(&now, &ts)) goto error;
325 
326  MEM(value = fr_value_box_alloc(ctx, FR_TYPE_UINT8, NULL, false));
327  value->datum.uint8 = ts.tm_mon + 1;
328  break;
329 
330  case 'M': /* Request time microsecond component */
331  MEM(value = fr_value_box_alloc(ctx, FR_TYPE_UINT32, NULL, false));
332  value->datum.uint32 = fr_time_to_msec(request->packet->timestamp) % 1000;
333  break;
334 
335  case 'S': /* Request timestamp in SQL format */
336  if (!localtime_r(&now, &ts)) goto error;
337 
338  strftime(buffer, sizeof(buffer), "%Y-%m-%d %H:%M:%S", &ts);
339 
340  MEM(value = fr_value_box_alloc_null(ctx));
341  if (fr_value_box_strdup(value, value, NULL, buffer, false) < 0) goto error;
342  break;
343 
344  case 't': /* Request timestamp in CTIME format */
345  {
346  char *p;
347 
348  CTIME_R(&now, buffer, sizeof(buffer));
349  p = strchr(buffer, '\n');
350  if (p) *p = '\0';
351 
352  MEM(value = fr_value_box_alloc_null(ctx));
353  if (fr_value_box_strdup(value, value, NULL, buffer, false) < 0) goto error;
354  }
355  break;
356 
357  case 'T': /* Request timestamp in ISO format */
358  {
359  int len = 0;
360 
361  if (!gmtime_r(&now, &ts)) goto error;
362 
363  if (!(len = strftime(buffer, sizeof(buffer) - 1, "%Y-%m-%dT%H:%M:%S", &ts))) {
364  REDEBUG("Failed converting packet timestamp to gmtime: Buffer full");
365  return XLAT_ACTION_FAIL;
366  }
367  strcat(buffer, ".");
368  len++;
369  snprintf(buffer + len, sizeof(buffer) - len, "%03i",
370  (int) fr_time_to_msec(request->packet->timestamp) % 1000);
371 
372  MEM(value = fr_value_box_alloc_null(ctx));
373  if (fr_value_box_strdup(value, value, NULL, buffer, false) < 0) goto error;
374  }
375  break;
376 
377  case 'Y': /* Request year */
378  if (!localtime_r(&now, &ts)) goto error;
379 
380  MEM(value = fr_value_box_alloc(ctx, FR_TYPE_UINT16, NULL, false));
381 
382  value->datum.int16 = ts.tm_year + 1900;
383  break;
384 
385  default:
386  fr_assert(0);
387  return XLAT_ACTION_FAIL;
388  }
389 
390  fr_cursor_append(out, value);
391  fr_cursor_next(out); /* Advance to our first value */
392 
393  return XLAT_ACTION_DONE;
394 }

396 /** Gets the value of a virtual attribute
397  *
398  * These attribute *may* be overloaded by the user using real attribute.
399  *
400  * @todo There should be a virtual attribute registry.
401  *
402  * @param[in] ctx to allocate boxed value, and buffers in.
403  * @param[out] out Where to write the boxed value.
404  * @param[in] request The current request.
405  * @param[in] vpt Representing the attribute.
406  * @return
407  * - #XLAT_ACTION_FAIL on memory allocation errors.
408  * - #XLAT_ACTION_DONE if we're done processing this node.
409  */
410 static xlat_action_t xlat_eval_pair_virtual(TALLOC_CTX *ctx, fr_cursor_t *out, request_t *request, tmpl_t const *vpt)
411 {
412  fr_radius_packet_t *packet = NULL;
413  fr_value_box_t *value;
414 
415  /*
416  * Virtual attributes always have a count of 1
417  */
418  if (tmpl_num(vpt) == NUM_COUNT) {
419  MEM(value = fr_value_box_alloc(ctx, FR_TYPE_UINT32, NULL, false));
420  value->datum.uint32 = 1;
421  goto done;
422  }
423 
424  /*
425  * Some non-packet expansions
426  */
427  if (tmpl_da(vpt) == attr_client_shortname) {
428  if (!request->client || !request->client->shortname) return XLAT_ACTION_DONE;
429 
430  MEM(value = fr_value_box_alloc_null(ctx));
431  if (fr_value_box_bstrdup_buffer(ctx, value, tmpl_da(vpt), request->client->shortname, false) < 0) {
432  error:
433  talloc_free(value);
434  return XLAT_ACTION_FAIL;
435  }
436  goto done;
437  }
438 
439  if (tmpl_da(vpt) == attr_request_processing_stage) {
440  if (!request->component) return XLAT_ACTION_DONE;
441 
442  MEM(value = fr_value_box_alloc_null(ctx));
443  if (fr_value_box_strdup(ctx, value, tmpl_da(vpt), request->component, false) < 0) goto error;
444  goto done;
445  }
446 
447  if (tmpl_da(vpt) == attr_virtual_server) {
448  if (!request->server_cs) return XLAT_ACTION_DONE;
449 
450  MEM(value = fr_value_box_alloc_null(ctx));
451  if (fr_value_box_bstrdup_buffer(ctx, value, tmpl_da(vpt),
452  cf_section_name2(request->server_cs), false) < 0) goto error;
453  goto done;
454  }
455 
456  if (tmpl_da(vpt) == attr_module_return_code) {
457  MEM(value = fr_value_box_alloc(ctx, tmpl_da(vpt)->type, tmpl_da(vpt), false));
458  value->datum.int32 = request->rcode;
459  goto done;
460  }
461 
462  /*
463  * All of the attributes must now refer to a packet.
464  * If there's no packet, we can't print any attribute
465  * referencing it.
466  */
467  packet = radius_packet(request, tmpl_list(vpt));
468  if (!packet) return XLAT_ACTION_DONE;
469 
470  if (tmpl_da(vpt) == attr_packet_type) {
471  if (!packet || !packet->code) return XLAT_ACTION_DONE;
472 
473  MEM(value = fr_value_box_alloc(ctx, tmpl_da(vpt)->type, NULL, false));
474  value->enumv = tmpl_da(vpt);
475  value->datum.int32 = packet->code;
476 
477  /*
478  * Virtual attributes which require a temporary fr_pair_t
479  * to be allocated. We can't use stack allocated memory
480  * because of the talloc checks sprinkled throughout the
481  * various VP functions.
482  */
483  } else if (tmpl_da(vpt) == attr_packet_authentication_vector) {
484  MEM(value = fr_value_box_alloc_null(ctx));
485  fr_value_box_memdup(ctx, value, tmpl_da(vpt), packet->vector, sizeof(packet->vector), true);
486 
487  } else if (tmpl_da(vpt) == attr_client_ip_address) {
488  if (request->client) {
489  MEM(value = fr_value_box_alloc_null(ctx));
490  fr_value_box_ipaddr(value, NULL, &request->client->ipaddr, false); /* Enum might not match type */
491  goto done;
492  }
493  goto src_ip_address;
494 
495  } else if (tmpl_da(vpt) == attr_packet_src_ip_address) {
496  src_ip_address:
497  if (!fr_socket_is_inet(packet->socket.proto) ||
498  (packet->socket.inet.src_ipaddr.af != AF_INET)) return XLAT_ACTION_DONE;
499 
500  MEM(value = fr_value_box_alloc_null(ctx));
501  fr_value_box_ipaddr(value, tmpl_da(vpt), &packet->socket.inet.src_ipaddr, true);
502 
503  } else if (tmpl_da(vpt) == attr_packet_dst_ip_address) {
504  if (!fr_socket_is_inet(packet->socket.proto) ||
505  (packet->socket.inet.dst_ipaddr.af != AF_INET)) return XLAT_ACTION_DONE;
506 
507  MEM(value = fr_value_box_alloc_null(ctx));
508  fr_value_box_ipaddr(value, tmpl_da(vpt), &packet->socket.inet.dst_ipaddr, true);
509 
510  } else if (tmpl_da(vpt) == attr_packet_src_ipv6_address) {
511  if (!fr_socket_is_inet(packet->socket.proto) ||
512  (packet->socket.inet.src_ipaddr.af != AF_INET6)) return XLAT_ACTION_DONE;
513 
514  MEM(value = fr_value_box_alloc_null(ctx));
515  fr_value_box_ipaddr(value, tmpl_da(vpt), &packet->socket.inet.src_ipaddr, true);
516 
517  } else if (tmpl_da(vpt) == attr_packet_dst_ipv6_address) {
518  if (!fr_socket_is_inet(packet->socket.proto) ||
519  (packet->socket.inet.dst_ipaddr.af != AF_INET6)) return XLAT_ACTION_DONE;
520 
521  MEM(value = fr_value_box_alloc_null(ctx));
522  fr_value_box_ipaddr(value, tmpl_da(vpt), &packet->socket.inet.dst_ipaddr, true);
523 
524  } else if (tmpl_da(vpt) == attr_packet_src_port) {
525  if (!fr_socket_is_inet(packet->socket.proto)) return XLAT_ACTION_DONE;
526 
527  MEM(value = fr_value_box_alloc(ctx, tmpl_da(vpt)->type, NULL, true));
528  value->datum.uint16 = packet->socket.inet.src_port;
529 
530  } else if (tmpl_da(vpt) == attr_packet_dst_port) {
531  if (!fr_socket_is_inet(packet->socket.proto)) return XLAT_ACTION_DONE;
532 
533  MEM(value = fr_value_box_alloc(ctx, tmpl_da(vpt)->type, NULL, true));
534  value->datum.uint16 = packet->socket.inet.dst_port;
535 
536  } else {
537  RERROR("Attribute \"%s\" incorrectly marked as virtual", tmpl_da(vpt)->name);
538  return XLAT_ACTION_FAIL;
539  }
540 
541 done:
542  fr_cursor_append(out, value);
543  fr_cursor_next(out); /* Advance to our first value */
544 
545  return XLAT_ACTION_DONE;
546 }

549 /** Gets the value of a real or virtual attribute
550  *
551  * @param[in] ctx to allocate boxed value, and buffers in.
552  * @param[out] out Where to write the boxed value.
553  * @param[in] request The current request.
554  * @param[in] vpt Representing the attribute.
555  * @return
556  * - #XLAT_ACTION_FAIL we failed getting a value for the attribute.
557  * - #XLAT_ACTION_DONE we
558  */

560 {

566

568 
569  /*
570  * See if we're dealing with an attribute in the request
571  *
572  * This allows users to manipulate virtual attributes as if
573  * they were real ones.
574  */
575  vp = tmpl_cursor_init(NULL, NULL, &cc, &cursor, request, vpt);
576 
577  /*
578  * We didn't find the VP in a list, check to see if it's
579  * virtual.
580  */
581  if (!vp) {
582  if (tmpl_is_attr(vpt) && tmpl_da(vpt)->flags.virtual) return xlat_eval_pair_virtual(ctx, out, request, vpt);
583 
584  /*
585  * Zero count.
586  */
587  if (tmpl_num(vpt) == NUM_COUNT) {
588  value = fr_value_box_alloc(ctx, FR_TYPE_UINT32, NULL, false);
589  if (!value) {
590  oom:
591  fr_strerror_printf("Out of memory");
592  tmpl_cursor_clear(&cc);
593  return XLAT_ACTION_FAIL;
594  }
595  value->datum.int32 = 0;
596  fr_cursor_append(out, value);
597  fr_cursor_next(out); /* Advance to our first value */
598 
599  return XLAT_ACTION_DONE;
600  }
601 
602  return XLAT_ACTION_DONE;
603  }
604 
605 
606  switch (tmpl_num(vpt)) {
607  /*
608  * Return a count of the VPs.
609  */
610  case NUM_COUNT:
611  {
612  uint32_t count = 0;
613 
614  for (vp = fr_cursor_current(&cursor);
615  vp;
616  vp = fr_cursor_next(&cursor)) count++;
617 
618  value = fr_value_box_alloc(ctx, FR_TYPE_UINT32, NULL, false);
619  value->datum.uint32 = count;
620  fr_cursor_append(out, value);
621  fr_cursor_next(out); /* Advance to our first value */
622  tmpl_cursor_clear(&cc);
623 
624  return XLAT_ACTION_DONE;
625  }
626 
627  /*
628  * Output multiple #value_box_t, one per attribute.
629  */
630  case NUM_ALL:
631  if (!fr_cursor_current(&cursor)) return XLAT_ACTION_DONE;
632 
633  /*
634  * Loop over all matching #fr_value_pair
635  * shallow copying buffers.
636  */
637  for (vp = fr_cursor_current(&cursor); /* Initialised above to the first matching attribute */
638  vp;
639  vp = fr_cursor_next(&cursor)) {
640  value = fr_value_box_alloc(ctx, vp->data.type, vp->da, vp->data.tainted);
641  fr_value_box_copy(value, value, &vp->data);
642  fr_cursor_append(out, value);
643  }
644  fr_cursor_next(out); /* Advance to our first value */
645  tmpl_cursor_clear(&cc);
646 
647  return XLAT_ACTION_DONE;
648 
649  default:
650  /*
651  * The cursor was set to the correct
652  * position above by tmpl_cursor_init.
653  */
654  vp = fr_cursor_current(&cursor); /* NULLness checked above */
655  value = fr_value_box_alloc(ctx, vp->data.type, vp->da, vp->data.tainted);
656  if (!value) goto oom;
657 
658  fr_value_box_copy(value, value, &vp->data); /* Also dups taint */
659  fr_cursor_append(out, value);
660  fr_cursor_next(out); /* Advance to our first value */
661  tmpl_cursor_clear(&cc);
662 
663  return XLAT_ACTION_DONE;
664  }

734 /** Process the result of a previous nested expansion
735  *
736  * @param[in] ctx to allocate value boxes in.
737  * @param[out] out a list of #fr_value_box_t to append to.
738  * @param[out] child to evaluate. If a child needs to be evaluated
739  * by the caller, we return XLAT_ACTION_PUSH_CHILD
740  * and place the child to be evaluated here.
741  * Once evaluation is complete, the caller
742  * should call us with the same #xlat_exp_t and the
743  * result of the nested evaluation in result.
744  * @param[in,out] alternate Whether we processed, or have previously processed
745  * the alternate.
746  * @param[in] request the current request.
747  * @param[in,out] in xlat node to evaluate. Advanced as we process
748  * additional #xlat_exp_t.
749  * @param[in] result of a previous nested evaluation.
750  */
751 xlat_action_t xlat_frame_eval_repeat(TALLOC_CTX *ctx, fr_cursor_t *out,
752  xlat_exp_t const **child, bool *alternate,
753  request_t *request, xlat_exp_t const **in,
754  fr_value_box_t **result)
755 {
756  xlat_exp_t const *node = *in;
757 
758  fr_cursor_tail(out); /* Needed for reentrant behaviour and debugging */
759 
760  switch (node->type) {
761  case XLAT_FUNC:
762  switch (node->call.func->type) {
763  case XLAT_FUNC_LEGACY:
764  {
765  fr_value_box_t *value;
766  char *str = NULL;
767  char *result_str = NULL;
768  ssize_t slen;
769 
770  if (*result) {
771  (void) talloc_list_get_type_abort(*result, fr_value_box_t);
772  result_str = fr_value_box_list_aprint(NULL, *result, NULL, NULL);
773  if (!result_str) return XLAT_ACTION_FAIL;
774  } else {
775  result_str = talloc_typed_strdup(NULL, "");
776  }
777 
778  MEM(value = fr_value_box_alloc_null(ctx));
779  if (node->call.func->buf_len > 0) {
780  fr_value_box_bstr_alloc(value, &str, value, NULL, node->call.func->buf_len, false);
781  }
782 
783  XLAT_DEBUG("** [%i] %s(func) - %%{%s:%pV}", unlang_interpret_stack_depth(request), __FUNCTION__,
784  node->fmt,
785  fr_box_strvalue_len(result_str, talloc_array_length(result_str) - 1));
786 
787  slen = node->call.func->func.sync(value, &str, node->call.func->buf_len,
788  node->call.func->mod_inst, NULL, request, result_str);
789  xlat_debug_log_expansion(request, *in, *result);
790  if (slen < 0) {
791  talloc_free(value);
792  talloc_free(result_str);
793  return XLAT_ACTION_FAIL;
794  }
795  if (slen == 0) { /* Zero length result */
796  talloc_free(result_str);
797  talloc_free(value);
798  break;
799  }
800  (void)talloc_get_type_abort(str, char); /* Check output buffer is sane */
801 
802  /*
803  * Shrink the buffer
804  */
805  if (node->call.func->buf_len > 0) {
806  if (slen > 0) fr_value_box_bstr_realloc(value, &str, value, slen);
807  } else {
808  fr_value_box_bstrdup_buffer_shallow(NULL, value, NULL, str, false);
809  }
810 
811  fr_cursor_append(out, value); /* Append the result of the expansion */
812  talloc_free(result_str);
813  xlat_debug_log_result(request, value);
814  }
815  break;
816 
817  case XLAT_FUNC_NORMAL:
818  {
819  xlat_action_t xa;
820  xlat_thread_inst_t *thread_inst;
821  fr_value_box_t *result_copy = NULL;
822 
823  thread_inst = xlat_thread_instance_find(node);
824 
825  XLAT_DEBUG("** [%i] %s(func-async) - %%{%s:%pM}", unlang_interpret_stack_depth(request), __FUNCTION__,

827 
828  /*
829  * Need to copy the input list in case
830  * the async function mucks with it.
831  */
832  if (RDEBUG_ENABLED2) fr_value_box_list_acopy(NULL, &result_copy, *result);
833 
834  if (*result) (void) talloc_list_get_type_abort(*result, fr_value_box_t);
835  xa = node->call.func->func.async(ctx, out, request, node->call.inst->data, thread_inst->data, result);
836  if (*result) (void) talloc_list_get_type_abort(*result, fr_value_box_t);
837 
838  if (RDEBUG_ENABLED2) {
839  xlat_debug_log_expansion(request, *in, result_copy);
840  talloc_list_free(&result_copy);
841  }
842  switch (xa) {
843  case XLAT_ACTION_FAIL:
844  return xa;
845 
846  case XLAT_ACTION_PUSH_CHILD:
847  RDEBUG2(" -- CHILD");
848  return xa;
849 
850  case XLAT_ACTION_YIELD:
851  RDEBUG2(" -- YIELD");
852  return xa;
853 
854  case XLAT_ACTION_DONE: /* Process the result */
855  fr_cursor_next(out);
856  xlat_debug_log_result(request, fr_cursor_current(out));
857  break;
858  }
859  break;
860  }
861  }
862  break;
863 
864  case XLAT_ALTERNATE:
865  {
866  fr_cursor_t from;
867 
868  fr_assert(alternate);
869 
870  /*
871  * No result from the first child, try the alternate
872  */
873  if (!*result) {
874  /*
875  * Already tried the alternate
876  */
877  if (*alternate) {
878  XLAT_DEBUG("** [%i] %s(alt-second) - string empty, null expansion, continuing...",
879  unlang_interpret_stack_depth(request), __FUNCTION__);
880  *alternate = false; /* Reset */
881 
882  xlat_debug_log_expansion(request, *in, NULL);
883  xlat_debug_log_result(request, NULL); /* Record the fact it's NULL */
884  break;
885  }
886 
887  XLAT_DEBUG("** [%i] %s(alt-first) - string empty, evaluating alternate: %s",
888  unlang_interpret_stack_depth(request), __FUNCTION__, (*in)->alternate->fmt);
889  *child = (*in)->alternate;
890  *alternate = true;
891 
892  return XLAT_ACTION_PUSH_CHILD;
893  }
894 
895  *alternate = false; /* Reset */
896 
897  xlat_debug_log_expansion(request, *in, NULL);
898  xlat_debug_log_result(request, *result);
899 
900  (void) talloc_list_get_type_abort(*result, fr_value_box_t);
901  fr_cursor_init(&from, result);
902  fr_cursor_merge(out, &from);
903  fr_assert(!*result);
904  }
905  break;
906 
907  case XLAT_GROUP:
908  {
909  fr_value_box_t *value;
910  fr_cursor_t from;
911 
912  XLAT_DEBUG("** [%i] %s(child) - continuing %%{%s ...}", unlang_interpret_stack_depth(request), __FUNCTION__,
913  node->fmt);
914 
915  fr_assert(*result != NULL);
916  (void) talloc_list_get_type_abort(*result, fr_value_box_t);
917 
918  /*
919  * If the input is a series of xlat expansions,
920  * then the outout is a series of value boxes.
921  * However, the caller wants only one output box,
922  * so we have to stringify the result.
923  */
924  if ((*in)->child->next) {
925  char *str;
926 
927  MEM(value = fr_value_box_alloc_null(ctx));
928 
929  if (*result) {
930  (void) talloc_list_get_type_abort(*result, fr_value_box_t);
931  str = fr_value_box_list_aprint(value, *result, NULL, NULL);
932  if (!str) return XLAT_ACTION_FAIL;
933  } else {
934  str = talloc_typed_strdup(value, "");
935  }
936 
937  fr_value_box_strdup_shallow(value, NULL, str, fr_value_box_list_tainted(*result));
938  talloc_free(*result);
939  *result = value;
940  }
941 
942  MEM(value = fr_value_box_alloc(ctx, FR_TYPE_GROUP, NULL, false));
943  value->vb_group = *result;
944  *result = NULL;
945  xlat_debug_log_expansion(request, *in, NULL);
946  xlat_debug_log_result(request, value);
947 
948  fr_cursor_init(&from, &value);
949  fr_cursor_merge(out, &from);
950  fr_assert(!value);
951  }
952  break;
953 
954  default:
955  fr_assert(0);
956  return XLAT_ACTION_FAIL;
957  }
958 
959  /*
960  * It's easier if we get xlat_frame_eval to continue evaluating the frame.
961  */
962  *in = (*in)->next; /* advance */

966 /** Converts xlat nodes to value boxes
967  *
968  * Evaluates a single level of expansions.
969  *
970  * @param[in] ctx to allocate value boxes in.
971  * @param[out] out a list of #fr_value_box_t to append to.
972  * @param[out] child to evaluate. If a child needs to be evaluated
973  * by the caller, we return XLAT_ACTION_PUSH_CHILD
974  * and place the child to be evaluated here.
975  * Once evaluation is complete, the caller
976  * should call us with the same #xlat_exp_t and the
977  * result of the nested evaluation in result.
978  * @param[in] request the current request.
979  * @param[in,out] in xlat node to evaluate. Advanced as we process
980  * additional #xlat_exp_t.
981  * @return
982  * - XLAT_ACTION_PUSH_CHILD if we need to evaluate a deeper level of nested.
983  * child will be filled with the node that needs to be evaluated.
984  * call #xlat_frame_eval_repeat on this node, once there are results
985  * from the nested expansion.
986  * - XLAT_ACTION_YIELD a resumption frame was pushed onto the stack by an
987  * xlat function and we need to wait for the request to be resumed
988  * before continuing.
989  * - XLAT_ACTION_DONE we're done, pop the frame.
990  * - XLAT_ACTION_FAIL an xlat module failed.
991  */
992 xlat_action_t xlat_frame_eval(TALLOC_CTX *ctx, fr_cursor_t *out, xlat_exp_t const **child,
993  request_t *request, xlat_exp_t const **in)
994 {
995  xlat_exp_t const *node = *in;
996  xlat_action_t xa = XLAT_ACTION_DONE;
997  fr_value_box_t *value;
998 
999  *child = NULL;
1000 
1001  if (!node) return XLAT_ACTION_DONE;
1002 
1003  XLAT_DEBUG("** [%i] %s >> entered", unlang_interpret_stack_depth(request), __FUNCTION__);
1004 
1005  for (node = *in; node; node = (*in)->next) {
1006  *in = node; /* Update node in our caller */
1007  fr_cursor_tail(out); /* Needed for debugging */
1008 
1009  switch (node->type) {
1010  case XLAT_LITERAL:
1011  XLAT_DEBUG("** [%i] %s(literal) - %s", unlang_interpret_stack_depth(request), __FUNCTION__, node->fmt);
1012 
1013  /*
1014  * Empty literals are only allowed if
1015  * they're the only node in the expansion.
1016  *
1017  * If they're found anywhere else the xlat
1018  * parser has an error.
1019  */
1020  fr_assert(((node == *in) && !node->next) || (talloc_array_length(node->fmt) > 1));
1021 
1022  /*
1023  * We unfortunately need to dup the buffer
1024  * because references aren't threadsafe.
1025  */
1026  MEM(value = fr_value_box_alloc_null(ctx));
1027  fr_value_box_bstrdup_buffer(value, value, NULL, node->fmt, false);
1028  fr_cursor_append(out, value);
1029  continue;
1030 
1031  case XLAT_ONE_LETTER:
1032  XLAT_DEBUG("** [%i] %s(one-letter) - %%%s", unlang_interpret_stack_depth(request), __FUNCTION__,
1033  node->fmt);
1034 
1035  xlat_debug_log_expansion(request, node, NULL);
1036  if (xlat_eval_one_letter(ctx, out, request, node->fmt[0]) == XLAT_ACTION_FAIL) {
1037  fail:
1038  fr_cursor_free_list(out); /* Only frees what we've added during this call */
1039  xa = XLAT_ACTION_FAIL;
1040  goto finish;
1041  }
1042  xlat_debug_log_result(request, fr_cursor_current(out));
1043  continue;
1044 
1045  case XLAT_ATTRIBUTE:
1046  XLAT_DEBUG("** [%i] %s(attribute) - %%{%s}", unlang_interpret_stack_depth(request), __FUNCTION__,
1047  node->fmt);
1048 
1049  xlat_debug_log_expansion(request, node, NULL);
1050  if (xlat_eval_pair_real(ctx, out, request, node->attr) == XLAT_ACTION_FAIL) goto fail;
1051  xlat_debug_log_result(request, fr_cursor_current(out));
1052  continue;
1053 
1054  case XLAT_VIRTUAL:
1055  {
1056  char *str = NULL;
1057  ssize_t slen;
1058 
1059  XLAT_DEBUG("** [%i] %s(virtual) - %%{%s}", unlang_interpret_stack_depth(request), __FUNCTION__,
1060  node->fmt);
1061 
1062  xlat_debug_log_expansion(request, node, NULL);
1063  MEM(value = fr_value_box_alloc_null(ctx));
1064  slen = node->call.func->func.sync(value, &str, node->call.func->buf_len, node->call.func->mod_inst,
1065  NULL, request, NULL);
1066  if (slen < 0) {
1067  talloc_free(value);
1068  goto fail;
1069  }
1070  if (slen == 0) continue;
1071 
1072  fr_value_box_bstrdup_buffer_shallow(NULL, value, NULL, str, false);
1073  fr_cursor_append(out, value);
1074  fr_cursor_next(out);
1075 
1076  xlat_debug_log_result(request, fr_cursor_current(out));
1077  }
1078  continue;
1079 
1080  case XLAT_FUNC:
1081  {
1082  fr_value_box_t *result = NULL;
1083 
1084  XLAT_DEBUG("** [%i] %s(func) - %%{%s:...}", unlang_interpret_stack_depth(request), __FUNCTION__,
1085  node->fmt);
1086 
1087  /*
1088  * Hand back the child node to the caller
1089  * for evaluation.
1090  */
1091  if (node->child) {
1092  *child = node->child;
1093  xa = XLAT_ACTION_PUSH_CHILD;
1094  goto finish;
1095  }
1096 
1097  /*
1098  * If there's no children we can just
1099  * call the function directly.
1100  */
1101  xa = xlat_frame_eval_repeat(ctx, out, child, NULL, request, in, &result);
1102  if (xa != XLAT_ACTION_DONE || (!*in)) goto finish;
1103  }
1104  continue;
1105 
1106 #ifdef HAVE_REGEX
1107  case XLAT_REGEX:
1108  {
1109  char *str = NULL;
1110 
1111  XLAT_DEBUG("** [%i] %s(regex) - %%{%s}", unlang_interpret_stack_depth(request), __FUNCTION__,
1112  node->fmt);
1113  MEM(value = fr_value_box_alloc_null(ctx));
1114  if (regex_request_to_sub(ctx, &str, request, node->regex_index) < 0) {
1115  talloc_free(value);
1116  continue;
1117  }
1118  fr_value_box_bstrdup_buffer_shallow(NULL, value, NULL, str, false);
1119  fr_cursor_append(out, value);
1120  }
1121  continue;
1122 #endif
1123 
1124  case XLAT_ALTERNATE:
1125  XLAT_DEBUG("** [%i] %s(alternate) - %%{%%{%s}:-%%{%s}}", unlang_interpret_stack_depth(request),
1126  __FUNCTION__, node->child->fmt, node->alternate->fmt);
1127  fr_assert(node->child != NULL);
1128  fr_assert(node->alternate != NULL);
1129 
1130  *child = node->child;
1131  xa = XLAT_ACTION_PUSH_CHILD;
1132  goto finish;
1133 
1134  case XLAT_GROUP:
1135  XLAT_DEBUG("** [%i] %s(child) - %%{%s ...}", unlang_interpret_stack_depth(request), __FUNCTION__,
1136  node->fmt);
1137 
1138  /*
1139  * Hand back the child node to the caller
1140  * for evaluation.
1141  */
1142  *child = node->child;
1143  xa = XLAT_ACTION_PUSH_CHILD;
1144  goto finish;
1145 
1146  /*
1147  * Should have been fixed up during pass2
1148  */
1149  case XLAT_INVALID:
1150  case XLAT_FUNC_UNRESOLVED:
1151  case XLAT_VIRTUAL_UNRESOLVED:
1152  fr_assert(0);
1153  return XLAT_ACTION_FAIL;
1154  }
1155 
1156  }
1157 
1158 finish:
1159  XLAT_DEBUG("** [%i] %s << %s", unlang_interpret_stack_depth(request),
1160  __FUNCTION__, fr_table_str_by_value(xlat_action_table, xa, "<INVALID>"));
1161 
1162  return xa;
1163 }
