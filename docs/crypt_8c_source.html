43 /** Performs a crypt password check in an thread-safe way.
44  *
45  * @param password The user's plaintext password.
46  * @param reference_crypt The 'known good' crypt the password
47  * is being compared to.
48  * @return
49  * - 0 crypt output matched reference crypt.
50  * - 1 crypt output did not match reference crypt.
51  * - -1 crypt failed.
52  */
53 int fr_crypt_check(char const *password, char const *reference_crypt)
54 {
55  char *crypt_out;
56  int cmp = 0;
57 
58 #ifdef HAVE_CRYPT_R
59  struct crypt_data crypt_data = { .initialized = 0 };
60 
61  crypt_out = crypt_r(password, reference_crypt, &crypt_data);
62  if (crypt_out) cmp = strcmp(reference_crypt, crypt_out);
63 #else
64  /*
65  * Ensure we're thread-safe, as crypt() isn't.
66  */
67  pthread_mutex_lock(&fr_crypt_mutex);
68  crypt_out = crypt(password, reference_crypt);
69 
70  /*
71  * Got something, check it within the lock. This is
72  * faster than copying it to a local buffer, and the
73  * time spent within the lock is critical.
74  */
75  if (crypt_out) cmp = strcmp(reference_crypt, crypt_out);
76  pthread_mutex_unlock(&fr_crypt_mutex);
77 #endif
78 
79  /*
80  * Error.
81  */
82  if (!crypt_out) return -1;
83 
84  /*
85  * OK, return OK.
86  */
87  if (cmp == 0) return 0;
88 
89  /*
90  * Comparison failed.
91  */
92  return 1;
93 }
