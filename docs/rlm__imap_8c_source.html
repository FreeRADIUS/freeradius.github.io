70 /*
71  * Called when the IMAP server responds
72  * It checks if the response was CURLE_OK
73  * If it wasn't we returns REJECT, if it was we returns OK
74 */
75 static unlang_action_t CC_HINT(nonnull) mod_authenticate_resume(rlm_rcode_t *p_result, module_ctx_t const *mctx,
76  request_t *request, void *rctx)
77 {
78  rlm_imap_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_imap_t);
79  fr_curl_io_request_t *randle = rctx;
80  fr_curl_tls_t const *tls;
81  long curl_out;
82  long curl_out_valid;
83 
84  tls = &inst->tls;
85 
86  curl_out_valid = curl_easy_getinfo(randle->candle, CURLINFO_SSL_VERIFYRESULT, &curl_out);
87  if (curl_out_valid == CURLE_OK){
88  RDEBUG2("server certificate %s verified", curl_out ? "was" : "not");
89  } else {
90  RDEBUG2("server certificate result not found");
91  }
92 
93  if (randle->result != CURLE_OK) {
94  talloc_free(randle);
95  RETURN_MODULE_REJECT;
96  }
97 
98  if (tls->extract_cert_attrs) fr_curl_response_certinfo(request, randle);
99 
100  talloc_free(randle);
101  RETURN_MODULE_OK;
102 }

104 /*
105  * Checks that there is a User-Name and User-Password field in the request
106  * Checks that User-Password is not Blank
107  * Sets the: username, password
108  * website URI
109  * timeout information
110  * and TLS information
111  *
112  * Then it queues the request and yeilds until a response is given
113  * When it responds, mod_authenticate_resume is called.
114  */
115 static unlang_action_t CC_HINT(nonnull(1,2)) mod_authenticate(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
116 {
117  rlm_imap_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_imap_t);
118  rlm_imap_thread_t *t = talloc_get_type_abort(mctx->thread, rlm_imap_thread_t);
119 
120  fr_pair_t const *username;
121  fr_pair_t const *password;
122  fr_curl_io_request_t *randle;
123 
124  randle = fr_curl_io_request_alloc(request);
125  if (!randle){
126  error:
127  RETURN_MODULE_FAIL;
128  }
129 
130  username = fr_pair_find_by_da(&request->request_pairs, attr_user_name);
131  password = fr_pair_find_by_da(&request->request_pairs, attr_user_password);
132 
133  if (!username) {
134  REDEBUG("Attribute \"User-Name\" is required for authentication");
135  RETURN_MODULE_INVALID;
136  }
137 
138  if (!password) {
139  RDEBUG2("Attribute \"User-Password\" is required for authentication");
140  RETURN_MODULE_INVALID;
141  }
142 
143  if (password->vp_length == 0) {
144  RDEBUG2("\"User-Password\" must not be empty");
145  RETURN_MODULE_INVALID;
146  }
147 
148  FR_CURL_REQUEST_SET_OPTION(CURLOPT_USERNAME, username->vp_strvalue);
149  FR_CURL_REQUEST_SET_OPTION(CURLOPT_PASSWORD, password->vp_strvalue);
150 
151  FR_CURL_REQUEST_SET_OPTION(CURLOPT_DEFAULT_PROTOCOL, "imap");
152  FR_CURL_REQUEST_SET_OPTION(CURLOPT_URL, inst->uri);
153  FR_CURL_REQUEST_SET_OPTION(CURLOPT_CONNECTTIMEOUT_MS, fr_time_delta_to_msec(inst->timeout));
154  FR_CURL_REQUEST_SET_OPTION(CURLOPT_TIMEOUT_MS, fr_time_delta_to_msec(inst->timeout));
155 
156  FR_CURL_REQUEST_SET_OPTION(CURLOPT_VERBOSE, 1L);
157 
158  if (fr_curl_easy_tls_init(randle, &inst->tls) != 0) RETURN_MODULE_INVALID;
159 
160  if (fr_curl_io_request_enqueue(t->mhandle, request, randle)) RETURN_MODULE_INVALID;
161 
162  return unlang_module_yield(request, mod_authenticate_resume, NULL, randle);
163 }
