124 /*
125  * Do authentication, by letting EAP-TLS do most of the work.
126  */
127 static unlang_action_t CC_HINT(nonnull) mod_process(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
128 {
129  rlm_eap_peap_t *inst = talloc_get_type(mctx->instance, rlm_eap_peap_t);
130 
131  rlm_rcode_t rcode;
132  eap_tls_status_t status;
133 
134  eap_session_t *eap_session = eap_session_get(request->parent);
135  eap_tls_session_t *eap_tls_session = talloc_get_type_abort(eap_session->opaque, eap_tls_session_t);
136  fr_tls_session_t *tls_session = eap_tls_session->tls_session;
137  peap_tunnel_t *peap = NULL;
138 
139  if (tls_session->opaque) {
140  peap = talloc_get_type_abort(tls_session->opaque, peap_tunnel_t);
141  /*
142  * Session resumption requires the storage of data, so
143  * allocate it if it doesn't already exist.
144  */
145  } else {
146  peap = tls_session->opaque = peap_alloc(tls_session, inst);
147  }
148 
149  status = eap_tls_process(request, eap_session);
150  if ((status == EAP_TLS_INVALID) || (status == EAP_TLS_FAIL)) {
151  REDEBUG("[eap-tls process] = %s", fr_table_str_by_value(eap_tls_status_table, status, "<INVALID>"));
152  } else {
153  RDEBUG2("[eap-tls process] = %s", fr_table_str_by_value(eap_tls_status_table, status, "<INVALID>"));
154  }
155 
156  switch (status) {
157  /*
158  * EAP-TLS handshake was successful, tell the
159  * client to keep talking.
160  *
161  * If this was EAP-TLS, we would just return
162  * an EAP-TLS-Success packet here.
163  */
164  case EAP_TLS_ESTABLISHED:
165  peap->status = PEAP_STATUS_TUNNEL_ESTABLISHED;
166  break;
167 
168  /*
169  * The TLS code is still working on the TLS
170  * exchange, and it's a valid TLS request.
171  * do nothing.
172  */
173  case EAP_TLS_HANDLED:
174  /*
175  * FIXME: If the SSL session is established, grab the state
176  * and EAP id from the inner tunnel, and update it with
177  * the expected EAP id!
178  */
179  RETURN_MODULE_HANDLED;
180 
181  /*
182  * Handshake is done, proceed with decoding tunneled
183  * data.
184  */
185  case EAP_TLS_RECORD_RECV_COMPLETE:
186  break;
187 
188  /*
189  * Anything else: fail.
190  */
191  default:
192  RETURN_MODULE_FAIL;
193  }
194 
195  /*
196  * Session is established, proceed with decoding
197  * tunneled data.
198  */
199  RDEBUG2("Session established. Decoding tunneled data");
200 
201  /*
202  * We may need PEAP data associated with the session, so
203  * allocate it here, if it wasn't already alloacted.
204  */
205  if (!tls_session->opaque) tls_session->opaque = peap_alloc(tls_session, inst);
206 
207  /*
208  * Process the PEAP portion of the request.
209  */
210  eap_peap_process(&rcode, request, eap_session, tls_session);
211  switch (rcode) {
212  case RLM_MODULE_REJECT:
213  eap_tls_fail(request, eap_session);
214  break;
215 
216  case RLM_MODULE_HANDLED:
217  eap_tls_request(request, eap_session);
218  break;
219 
220  case RLM_MODULE_OK:
221  {
222  static char keying_prf_label[] = "client EAP encryption";
223 
224  /*
225  * Success: Automatically return MPPE keys.
226  */
227  if (eap_tls_success(request, eap_session,
228  keying_prf_label, sizeof(keying_prf_label) - 1,
229  NULL, 0) < 0) return 0;
230  }
231  break;
232 
233  /*
234  * No response packet, MUST be proxying it.
235  * The main EAP module will take care of discovering
236  * that the request now has a "proxy" packet, and
237  * will proxy it, rather than returning an EAP packet.
238  */
239  case RLM_MODULE_UPDATED:
240  break;
241 
242  default:
243  eap_tls_fail(request, eap_session);
244  break;
245  }
246 
247  RETURN_MODULE_RCODE(rcode);
248 }

251  * Send an initial eap-tls request to the peer, using the libeap functions.
252  */
253 static unlang_action_t mod_session_init(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
254 {
255  rlm_eap_peap_t *inst = talloc_get_type_abort(mctx->instance, rlm_eap_peap_t);
256  eap_session_t *eap_session = eap_session_get(request->parent);
257  eap_tls_session_t *eap_tls_session;
258 
259  fr_pair_t *vp;
260  bool client_cert;
261 
262  eap_session->tls = true;
263 
264  /*
265  * EAP-TLS-Require-Client-Cert attribute will override
266  * the require_client_cert configuration option.
267  */
268  vp = fr_pair_find_by_da(&request->control_pairs, attr_eap_tls_require_client_cert);
269  if (vp) {
270  client_cert = vp->vp_uint32 ? true : false;
271  } else {
272  client_cert = inst->req_client_cert;
273  }
274 
275  eap_session->opaque = eap_tls_session = eap_tls_session_init(request, eap_session, inst->tls_conf, client_cert);
276  if (!eap_tls_session) RETURN_MODULE_FAIL;
277 
278  /*
279  * As it is a poorly designed protocol, PEAP uses
280  * bits in the TLS header to indicate PEAP
281  * version numbers. For now, we only support
282  * PEAP version 0, so it doesn't matter too much.
283  * However, if we support later versions of PEAP,
284  * we will need this flag to indicate which
285  * version we're currently dealing with.
286  */
287  eap_tls_session->base_flags = 0x00;
288 
289  /*
290  * PEAP version 0 requires 'include_length = no',
291  * so rather than hoping the user figures it out,
292  * we force it here.
293  */
294  eap_tls_session->include_length = false;
295 
296  /*
297  * TLS session initialization is over. Now handle TLS
298  * related handshaking or application data.
299  */
300  if (eap_tls_start(request, eap_session) < 0) {
301  talloc_free(eap_tls_session);
302  RETURN_MODULE_FAIL;
303  }
304 
305  eap_session->process = mod_process;
306 
307  RETURN_MODULE_HANDLED;
308 }

313 static int mod_instantiate(void *instance, CONF_SECTION *cs)
314 {
315  rlm_eap_peap_t *inst = talloc_get_type_abort(instance, rlm_eap_peap_t);

318  cf_log_err_by_child(cs, "virtual_server", "Unknown virtual server '%s'", inst->virtual_server);
319  return -1;
320  }
321 
322  if (inst->soh_virtual_server) {
323  if (!virtual_server_find(inst->soh_virtual_server)) {
324  cf_log_err_by_child(cs, "soh_virtual_server", "Unknown virtual server '%s'", inst->virtual_server);
325  return -1;
326  }
327  }
328 
329  /*
330  * Read tls configuration, either from group given by 'tls'
331  * option, or from the eap-tls configuration.
332  */
333  inst->tls_conf = eap_tls_conf_parse(cs, "tls");
334  if (!inst->tls_conf) {
335  ERROR("Failed initializing SSL context");
336  return -1;
337  }
338 
339  return 0;
340 }
