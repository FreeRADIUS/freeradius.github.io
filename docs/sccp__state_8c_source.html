70 void app_forward_sccp(struct ss7_application *app, struct msgb *_msg, int sls)
71 {
72  int rc;
73  struct sccp_parse_result result;
74  struct msc_connection *msc;
75  struct mtp_link_set *set;
76 
77  struct msgb *msg;
78 
79  set = app->route_src.set;
80  msc = app->route_dst.msc;
81 
82  if (app->forward_only) {
83  send_direct(msc, _msg);
84  return;
85  }
86 
87  rc = bss_patch_filter_msg(app, _msg, &result, BSS_DIR_MSC);
88  if (rc == BSS_FILTER_RESET) {
89  LOGP(DMSC, LOGL_NOTICE, "Filtering BSS Reset from the BSC\n");
90  msc_mgcp_reset(msc);
91  send_reset_ack(set, sls);
92  return;
93  }
94 
95  /* special responder */
96  if (msc->msc_link_down) {
97  if (rc == BSS_FILTER_RESET_ACK && app->reset_count > 0) {
98  LOGP(DMSC, LOGL_ERROR, "Received reset ack for closing.\n");
99  app_clear_connections(app);
100  app_resources_released(app);
101  return;
102  }
103 
104  if (rc != 0 && rc != BSS_FILTER_RLSD && rc != BSS_FILTER_RLC) {
105  LOGP(DMSC, LOGL_ERROR, "Ignoring unparsable msg during closedown.\n");
106  return;
107  }
108 
109  return handle_local_sccp(set, _msg, &result, sls);
110  }
111 
112  /* update the connection state */
113  update_con_state(app, rc, &result, _msg, 0, sls);
114 
115  if (rc == BSS_FILTER_CLEAR_COMPL) {
116  send_local_rlsd(set, &result);
117  } else if (rc == BSS_FILTER_RLC || rc == BSS_FILTER_RLSD) {
118  LOGP(DMSC, LOGL_DEBUG, "Not forwarding RLC/RLSD to the MSC.\n");
119  return;
120  }
121 
122  /* now send it out */
123  bsc_ussd_handle_out_msg(msc, &result, _msg);
124 
125  msg = msgb_alloc_headroom(4096, 128, "SCCP to MSC");
126  if (!msg) {
127  LOGP(DMSC, LOGL_ERROR, "Failed to alloc MSC msg.\n");
128  return;
129  }
130 
131  bss_rewrite_header_for_msc(rc, msg, _msg, &result);
132  msc_send_direct(msc, msg);
133 }

138 static void handle_local_sccp(struct mtp_link_set *set, struct msgb *inpt, struct sccp_parse_result *result, int sls)
139 {
140  /* Handle msg with a reject */
141  if (inpt->l2h[0] == SCCP_MSG_TYPE_CR) {
142  struct sccp_connection_request *cr;
143  struct msgb *msg;
144 
145  LOGP(DINP, LOGL_NOTICE, "Handling CR localy.\n");
146  cr = (struct sccp_connection_request *) inpt->l2h;
147  msg = create_sccp_refuse(&cr->source_local_reference);
148  if (msg) {
149  mtp_link_set_submit_sccp_data(set, sls, msg->l2h, msgb_l2len(msg));
150  msgb_free(msg);
151  }
152  return;
153  } else if (inpt->l2h[0] == SCCP_MSG_TYPE_DT1 && result->data_len >= 3) {
154  struct active_sccp_con *con;
155  struct sccp_data_form1 *form1;
156  struct msgb *msg;
157 
158  if (inpt->l3h[0] == 0 && inpt->l3h[2] == BSS_MAP_MSG_CLEAR_COMPLETE) {
159  LOGP(DINP, LOGL_DEBUG, "Received Clear Complete. Sending Release.\n");
160 
161  form1 = (struct sccp_data_form1 *) inpt->l2h;
162 
163  llist_for_each_entry(con, &set->app->sccp_connections, entry) {
164  if (memcmp(&form1->destination_local_reference,
165  &con->dst_ref, sizeof(con->dst_ref)) == 0) {
166  LOGP(DINP, LOGL_DEBUG, "Sending a release request now.\n");
167  msg = create_sccp_rlsd(&con->dst_ref, &con->src_ref);
168  if (msg) {
169  mtp_link_set_submit_sccp_data(set, con->sls, msg->l2h, msgb_l2len(msg));
170  msgb_free(msg);
171  }
172  return;
173  }
174  }
175 
176  LOGP(DINP, LOGL_ERROR, "Could not find connection for the Clear Command.\n");
177  }
178  } else if (inpt->l2h[0] == SCCP_MSG_TYPE_UDT && result->data_len >= 3) {
179  if (inpt->l3h[0] == 0 && inpt->l3h[2] == BSS_MAP_MSG_RESET_ACKNOWLEDGE) {
180  LOGP(DINP, LOGL_NOTICE, "Reset ACK. Connecting to the MSC again.\n");
181  app_resources_released(set->app);
182  return;
183  }
184  }
185 
186 
187  /* Update the state, maybe the connection was released? */
188  update_con_state(set->app, 0, result, inpt, 0, sls);
189  if (llist_empty(&set->app->sccp_connections))
190  app_resources_released(set->app);
191  return;
192 }

238 /*
239  * We have lost the connection to the MSC. This is tough. We
240  * can not just bring down the MTP link as this will disable
241  * the BTS radio. We will have to do the following:
242  *
243  * 1.) Bring down all open SCCP connections. As this will close
244  * all radio resources
245  * 2.) Bring down all MGCP endpoints
246  * 3.) Clear the connection data.
247  *
248  * To make things worse we need to buffer the BSC messages... atfer
249  * everything has been sent we will try to connect to the MSC again.
250  *
251  * We will have to veriy that all connections are closed properly..
252  * this means we need to parse response message. In the case the
253  * MTP link is going down while we are sending. We will simply
254  * reconnect to the MSC.
255  *
256  * This could be called for the relay type and the cellmgr type, in case
257  * of the relay type the list of connections should be empty so we can
258  * avoid branching out.
259  */
260 void release_bsc_resources(struct msc_connection *fw)
261 {
262  struct ss7_application *app;
263  struct mtp_link_set *set;
264  struct active_sccp_con *tmp;
265  struct active_sccp_con *con;
266 
267  if (!fw->app) {
268  LOGP(DINP, LOGL_ERROR, "No app assigned to the MSC connection %d/%s\n",
269  fw->nr, fw->name);
270  return;
271  }
272 
273  app = fw->app;
274  set = app->route_src.set;
275  osmo_timer_del(&app->reset_timeout);
276 
277  /* 2. clear the MGCP endpoints */
278  msc_mgcp_reset(fw);
279 
280  /* 1. send BSSMAP Cleanup.. if we have any connection */
281  llist_for_each_entry_safe(con, tmp, &app->sccp_connections, entry) {
282  if (!con->has_dst_ref) {
283  free_con(con);
284  continue;
285  }
286 
287  struct msgb *msg = create_clear_command(&con->src_ref);
288  if (!msg)
289  continue;
290 
291  /* wait for the clear commands */
292  mtp_link_set_submit_sccp_data(set, con->sls, msg->l2h, msgb_l2len(msg));
293  msgb_free(msg);
294  }
295 
296  if (llist_empty(&app->sccp_connections)) {
297  app_resources_released(app);
298  } else {
299  /* Send a reset in 20 seconds if we fail to bring everything down */
300  app->reset_timeout.cb = bsc_reset_timeout;
301  app->reset_timeout.data = app;
302  app->reset_count = 0;
303  osmo_timer_schedule(&app->reset_timeout, 10, 0);
304  }
305 }

324 static void handle_rlsd(struct ss7_application *app, struct sccp_connection_released *rlsd, int from_msc)
325 {
326  struct active_sccp_con *con;
327  struct msc_connection *msc = app->route_dst.msc;
328  struct mtp_link_set *set = app->route_src.set;
329 
330  if (from_msc) {
331  /* search for a connection, reverse src/dest for MSC */
332  con = find_con_by_src_dest_ref(app, &rlsd->destination_local_reference,
333  &rlsd->source_local_reference);
334  if (con) {
335  LOGP(DINP, LOGL_DEBUG, "RLSD conn still alive: local: 0x%x remote: 0x%x\n",
336  sccp_src_ref_to_int(&con->src_ref),
337  sccp_src_ref_to_int(&con->dst_ref));
338  con->released_from_msc = 1;
339  } else {
340  /* send RLC */
341  LOGP(DINP, LOGL_DEBUG, "Sending RLC for MSC: src: 0x%x dst: 0x%x\n",
342  sccp_src_ref_to_int(&rlsd->destination_local_reference),
343  sccp_src_ref_to_int(&rlsd->source_local_reference));
344  msc_send_rlc(msc, &rlsd->destination_local_reference,
345  &rlsd->source_local_reference);
346  }
347  } else {
348  unsigned int sls = -1;
349  con = find_con_by_src_dest_ref(app, &rlsd->source_local_reference,
350  &rlsd->destination_local_reference);
351  if (con) {
352  LOGP(DINP, LOGL_DEBUG, "Timeout on BSC. Sending RLC. src: 0x%x\n",
353  sccp_src_ref_to_int(&rlsd->source_local_reference));
354 
355  if (con->released_from_msc)
356  msc_send_rlc(msc, &con->src_ref, &con->dst_ref);
357  sls = con->sls;
358  free_con(con);
359  } else {
360  LOGP(DINP, LOGL_ERROR, "Timeout on BSC for unknown connection. src: 0x%x\n",
361  sccp_src_ref_to_int(&rlsd->source_local_reference));
362  }
363 
364  /* now send a rlc back to the BSC */
365  send_rlc_to_bsc(set, sls, &rlsd->destination_local_reference, &rlsd->source_local_reference);
366  }
367 }

369 /*
370  * Update connection state and also send message.....
371  *
372  * RLSD from MSC:
373  * 1.) We don't find the entry in this case we will send a
374  * forged RLC to the MSC and we are done.
375  * 2.) We find an entry in this we will need to register that
376  * we need to send a RLC and we are done for now.
377  * RLSD from BSC:
378  * 1.) This is an error we are ignoring for now.
379  * RLC from BSC:
380  * 1.) We are destroying the connection, we might send a RLC to
381  * the MSC if we are waiting for one.
382  */
383 void update_con_state(struct ss7_application *app, int rc, struct sccp_parse_result *res, struct msgb *msg, int from_msc, int sls)
384 {
385  struct active_sccp_con *con;
386  struct sccp_connection_request *cr;
387  struct sccp_connection_confirm *cc;
388  struct sccp_connection_release_complete *rlc;
389  struct sccp_connection_refused *cref;
390  struct msc_connection *msc;
391 
392  /* was the header okay? */
393  if (rc < 0)
394  return;
395 
396  msc = app->route_dst.msc;
397 
398  /* the header was size checked */
399  switch (msg->l2h[0]) {
400  case SCCP_MSG_TYPE_CR:
401  if (from_msc) {
402  LOGP(DMSC, LOGL_ERROR, "CR from MSC is not handled.\n");
403  return;
404  }
405 
406  cr = (struct sccp_connection_request *) msg->l2h;
407  con = find_con_by_src_ref(app, &cr->source_local_reference);
408  if (con) {
409  LOGP(DINP, LOGL_ERROR, "Duplicate SRC reference for: 0x%x. Reusing\n",
410  sccp_src_ref_to_int(&con->src_ref));
411  free_con(con);
412  }
413 
414  con = talloc_zero(NULL, struct active_sccp_con);
415  if (!con) {
416  LOGP(DINP, LOGL_ERROR, "Failed to allocate\n");
417  return;
418  }
419 
420  con->src_ref = cr->source_local_reference;
421  con->sls = sls;
422  con->app = app;
423  llist_add_tail(&con->entry, &app->sccp_connections);
424  LOGP(DINP, LOGL_DEBUG, "Adding CR: local ref: 0x%x\n", sccp_src_ref_to_int(&con->src_ref));
425  break;
426  case SCCP_MSG_TYPE_CC:
427  if (!from_msc) {
428  LOGP(DINP, LOGL_ERROR, "CC from BSC is not handled.\n");
429  return;
430  }
431 
432  cc = (struct sccp_connection_confirm *) msg->l2h;
433  con = find_con_by_src_ref(app, &cc->destination_local_reference);
434  if (con) {
435  con->dst_ref = cc->source_local_reference;
436  con->has_dst_ref = 1;
437  LOGP(DINP, LOGL_DEBUG, "Updating CC: local: 0x%x remote: 0x%x\n",
438  sccp_src_ref_to_int(&con->src_ref), sccp_src_ref_to_int(&con->dst_ref));
439  return;
440  }
441 
442  LOGP(DINP, LOGL_ERROR, "CCed connection can not be found: 0x%x\n",
443  sccp_src_ref_to_int(&cc->destination_local_reference));
444  break;
445  case SCCP_MSG_TYPE_CREF:
446  if (!from_msc) {
447  LOGP(DINP, LOGL_ERROR, "CREF from BSC is not handled.\n");
448  return;
449  }
450 
451  cref = (struct sccp_connection_refused *) msg->l2h;
452  con = find_con_by_src_ref(app, &cref->destination_local_reference);
453  if (con) {
454  LOGP(DINP, LOGL_DEBUG, "Releasing local: 0x%x\n", sccp_src_ref_to_int(&con->src_ref));
455  free_con(con);
456  return;
457  }
458 
459  LOGP(DINP, LOGL_ERROR, "CREF from BSC is not handled.\n");
460  break;
461  case SCCP_MSG_TYPE_RLSD:
462  handle_rlsd(app, (struct sccp_connection_released *) msg->l2h, from_msc);
463  break;
464  case SCCP_MSG_TYPE_RLC:
465  if (from_msc) {
466  LOGP(DINP, LOGL_ERROR, "RLC from MSC is wrong.\n");
467  return;
468  }
469 
470  rlc = (struct sccp_connection_release_complete *) msg->l2h;
471  con = find_con_by_src_dest_ref(app, &rlc->source_local_reference,
472  &rlc->destination_local_reference);
473  if (con) {
474  LOGP(DINP, LOGL_DEBUG, "Releasing local: 0x%x\n", sccp_src_ref_to_int(&con->src_ref));
475  if (con->released_from_msc)
476  msc_send_rlc(msc, &con->src_ref, &con->dst_ref);
477  free_con(con);
478  return;
479  }
480 
481  LOGP(DINP, LOGL_ERROR, "RLC can not be found. 0x%x 0x%x\n",
482  sccp_src_ref_to_int(&rlc->source_local_reference),
483  sccp_src_ref_to_int(&rlc->destination_local_reference));
484  break;
485  }
486 }

489 {
490  struct msgb *rlsd;
491  struct active_sccp_con *con = (struct active_sccp_con *) data;
492  struct mtp_link_set *set;
493 
494  /* try again in three seconds */
495  con->rlc_timeout.data = con;

514  mtp_link_set_submit_sccp_data(set, con->sls, rlsd->l2h, msgb_l2len(rlsd));

516 }

542 void msc_dispatch_sccp(struct msc_connection *msc, struct msgb *msg)
543 {
544  struct mtp_link_set *set;
545 
546  if (!msc->app) {
547  LOGP(DINP, LOGL_ERROR, "The MSC Connection %d/%s has no app assigned.\n",
548  msc->nr, msc->name);
549  return;
550  }
551 
552 
553  set = msc->app->route_src.set;
554 
555  /* we can not forward it right now */
556  if (msc->app->forward_only) {
557  if (!set->sccp_up)
558  return;
559  mtp_link_set_submit_sccp_data(set, -1,
560  msg->l2h, msgb_l2len(msg));
561  } else {
562  struct sccp_parse_result result;
563  int rc;
564 
565  rc = bss_patch_filter_msg(msc->app, msg, &result, BSS_DIR_BSC);
566 
567  if (rc == BSS_FILTER_RESET_ACK) {
568  LOGP(DMSC, LOGL_NOTICE, "Filtering reset ack from the MSC\n");
569  } else if (rc == BSS_FILTER_RLSD) {
570  LOGP(DMSC, LOGL_DEBUG, "Filtering RLSD from the MSC\n");
571  update_con_state(msc->app, rc, &result, msg, 1, 0);
572  } else if (rc == BSS_FILTER_RLC) {
573  /* if we receive this we have forwarded a RLSD to the network */
