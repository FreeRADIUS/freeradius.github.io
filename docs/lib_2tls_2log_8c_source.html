39 /** Print errors in the TLS thread local error stack
40  *
41  * Drains the thread local OpenSSL error queue, and prints out errors.
42  *
43  * @param[in] request The current request (may be NULL).
44  * @param[in] msg Error message describing the operation being attempted.
45  * @param[in] ap Arguments for msg.
46  * @return the number of errors drained from the stack.
47  */
48 static int fr_tls_log_error_va(request_t *request, char const *msg, va_list ap)
49 {
50  unsigned long error;
51  char *p;
52  int in_stack = 0;
53  char buffer[256];
54 
55  int line;
56  char const *file;
57  char const *data;
58  int flags = 0;
59 
60  /*
61  * Pop the first error, so ERR_peek_error()
62  * can be used to determine if there are
63  * multiple errors.
64  */
65  error = ERR_get_error_line_data(&file, &line, &data, &flags);
66  if (!(flags & ERR_TXT_STRING)) data = NULL;
67 
68  if (msg) {
69  p = talloc_vasprintf(request, msg, ap);
70 
71  /*
72  * Single line mode (there's only one error)
73  */
74  if (error && !ERR_peek_error()) {
75  ERR_error_string_n(error, buffer, sizeof(buffer));
76 
77  /* Extra verbose */
78  if ((request && RDEBUG_ENABLED3) || DEBUG_ENABLED3) {
79  ROPTIONAL(REDEBUG, ERROR, "%s: %s[%i]:%s%c%s", p, file, line, buffer,
80  data ? ':' : '\0', data ? data : "");
81  } else {
82  ROPTIONAL(REDEBUG, ERROR, "%s: %s%c%s", p, buffer,
83  data ? ':' : '\0', data ? data : "");
84  }
85 
86  talloc_free(p);
87 
88  return 1;
89  }
90 
91  /*
92  * Print the error we were given, irrespective
93  * of whether there were any OpenSSL errors.
94  */
95  ROPTIONAL(RERROR, ERROR, "%s", p);
96  talloc_free(p);
97  }
98 
99  /*
100  * Stack mode (there are multiple errors)
101  */
102  if (!error) return 0;
103  do {
104  if (!(flags & ERR_TXT_STRING)) data = NULL;
105 
106  ERR_error_string_n(error, buffer, sizeof(buffer));
107  /* Extra verbose */
108  if ((request && RDEBUG_ENABLED3) || DEBUG_ENABLED3) {
109  ROPTIONAL(REDEBUG, ERROR, "%s[%i]:%s%c%s", file, line, buffer,
110  data ? ':' : '\0', data ? data : "");
111  } else {
112  ROPTIONAL(REDEBUG, ERROR, "%s%c%s", buffer,
113  data ? ':' : '\0', data ? data : "");
114  }
115  in_stack++;
116  } while ((error = ERR_get_error_line_data(&file, &line, &data, &flags)));
117 
118  return in_stack;
119 }

152 /** Print errors in the TLS thread local error stack
153  *
154  * Drains the thread local OpenSSL error queue, and prints out the first error
155  * storing it in libfreeradius's error buffer.

161 static int tls_strerror_vasprintf(char const *msg, va_list ap)
162 {
163  unsigned long error;
164  char *p = NULL;
165  int drained = 0;
166  char buffer[256];
167 
168  int line;
169  char const *file;
170  char const *data;
171  int flags = 0;
172 
173  /*
174  * Pop the first error, so ERR_peek_error()
175  * can be used to determine if there are
176  * multiple errors.
177  */
178  error = ERR_get_error_line_data(&file, &line, &data, &flags);
179  if (!(flags & ERR_TXT_STRING)) data = NULL;
180 
181  if (msg) {
182  /*
183  * Print the error we were passed, and
184  * OpenSSL's error.
185  */
186  p = talloc_vasprintf(NULL, msg, ap);
187  if (error) {
188  ERR_error_string_n(error, buffer, sizeof(buffer));
189  fr_strerror_printf("%s: %s%c%s", p, buffer, data ? ':' : '\0', data ? data : "");
190  talloc_free(p);
191  drained++;
192  /*
193  * Print the error we were given, irrespective
194  * of whether there were any OpenSSL errors.
195  */
196  } else {
197  fr_strerror_printf("%s", p);
198  talloc_free(p);
199  }
200  } else if (error) {
201  ERR_error_string_n(error, buffer, sizeof(buffer));
202  fr_strerror_printf("%s%c%s", buffer, data ? ':' : '\0', data ? data : "");
203  drained++;
204  } else {
205  return 0;
206  }
207 
208  while ((error = ERR_get_error_line_data(&file, &line, &data, &flags))) {
209  if (!(flags & ERR_TXT_STRING)) data = NULL;
210 
211  ERR_error_string_n(error, buffer, sizeof(buffer));
212  fr_strerror_printf_push("%s%c%s", buffer, data ? ':' : '\0', data ? data : "");
213  drained++;
214  }
215 
216  return drained;
217 }

277 /** Print errors raised by OpenSSL I/O functions
278  *
279  * Drains the thread local OpenSSL error queue, and prints out errors
280  * based on the SSL handle and the return code of the I/O function.
281  *
282  * OpenSSL lists I/O functions to be:
283  * - SSL_connect
284  * - SSL_accept
285  * - SSL_do_handshake
286  * - SSL_read
287  * - SSL_peek
288  * - SSL_write
289  *
290  * @param request The current request (may be NULL).
291  * @param session The current tls_session.
292  * @param ret from the I/O operation.
293  * @param msg Error message describing the operation being attempted.
294  * @param ... Arguments for msg.
295  * @return
296  * - 0 TLS session may still be viable.
297  * - -1 TLS session cannot continue.
298  */
299 int fr_tls_log_io_error(request_t *request, fr_tls_session_t *session, int ret, char const *msg, ...)
300 {
301  int error;
302  va_list ap;
303 
304  if (ERR_peek_error()) {
305  va_start(ap, msg);
306  fr_tls_log_error_va(request, msg, ap);
307  va_end(ap);
308  }
309 
310  error = SSL_get_error(session->ssl, ret);
311  switch (error) {
312  /*
313  * These seem to be harmless and already "dealt
314  * with" by our non-blocking environment. NB:
315  * "ZERO_RETURN" is the clean "error"
316  * indicating a successfully closed SSL
317  * tunnel. We let this happen because our IO
318  * loop should not appear to have broken on
319  * this condition - and outside the IO loop, the
320  * "shutdown" state is checked.
321  *
322  * Don't print anything if we ignore the error.
323  */
324  case SSL_ERROR_NONE:
325  case SSL_ERROR_WANT_READ:
326  case SSL_ERROR_WANT_WRITE:
327  case SSL_ERROR_WANT_X509_LOOKUP:
328  case SSL_ERROR_ZERO_RETURN:
329  break;
330 
331  /*
332  * These seem to be indications of a genuine
333  * error that should result in the SSL tunnel
334  * being regarded as "dead".
335  */
336  case SSL_ERROR_SYSCALL:
337  ROPTIONAL(REDEBUG, ERROR, "System call (I/O) error (%i)", ret);
338  return -1;
339 
340  case SSL_ERROR_SSL:
341  ROPTIONAL(REDEBUG, ERROR, "TLS protocol error (%i)", ret);
342  return -1;
343 
344  /*
345  * For any other errors that (a) exist, and (b)
346  * crop up - we need to interpret what to do with
347  * them - so "politely inform" the caller that
348  * the code needs updating here.
349  */
350  default:
351  ROPTIONAL(REDEBUG, ERROR, "TLS session error %i (%i)", error, ret);
352  return -1;
353  }
354 
355  return 0;
356 }
