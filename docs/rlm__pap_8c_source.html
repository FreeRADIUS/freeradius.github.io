129  if (fr_pair_find_by_da(&request->control_pairs, attr_auth_type) != NULL) {

137  RETURN_MODULE_NOOP;
138  }
139 
140  if (!inst->auth_type) {
141  WARN("No 'authenticate %s {...}' section or 'Auth-Type = %s' set. Cannot setup PAP authentication.",
142  inst->name, inst->name);
143  RETURN_MODULE_NOOP;
144  }
145 
146  if (!module_section_type_set(request, attr_auth_type, inst->auth_type)) RETURN_MODULE_NOOP;
147

392 /** Validates Crypt::PBKDF2 LDAP format strings
393  *
394  * @param[out] p_result The result of comparing the pbkdf2 hash with the password.
395  * @param[in] request The current request.
396  * @param[in] str Raw PBKDF2 string.
397  * @param[in] len Length of string.
398  * @param[in] hash_names Table containing valid hash names.
399  * @param[in] hash_names_len How long the table is.
400  * @param[in] scheme_sep Separation character between the scheme and the next component.
401  * @param[in] iter_sep Separation character between the iterations and the next component.
402  * @param[in] salt_sep Separation character between the salt and the next component.
403  * @param[in] iter_is_base64 Whether the iterations is are encoded as base64.
404  * @param[in] password to validate.
405  * @return
406  * - RLM_MODULE_REJECT
407  * - RLM_MODULE_OK
408  */
409 static inline CC_HINT(nonnull) unlang_action_t pap_auth_pbkdf2_parse(rlm_rcode_t *p_result,
410  request_t *request, const uint8_t *str, size_t len,
411  fr_table_num_sorted_t const hash_names[], size_t hash_names_len,
412  char scheme_sep, char iter_sep, char salt_sep,
413  bool iter_is_base64, fr_pair_t const *password)
414 {
415  rlm_rcode_t rcode = RLM_MODULE_INVALID;
416 
417  uint8_t const *p, *q, *end;
418  ssize_t slen;
419 
420  EVP_MD const *evp_md;
421  int digest_type;
422  size_t digest_len;
423 
424  uint32_t iterations;
425 
426  uint8_t *salt = NULL;
427  size_t salt_len;
428  uint8_t hash[EVP_MAX_MD_SIZE];
429  uint8_t digest[EVP_MAX_MD_SIZE];
430 
431  RDEBUG2("Comparing with \"known-good\" PBKDF2-Password");
432 
433  if (len <= 1) {
434  REDEBUG("PBKDF2-Password is too short");
435  goto finish;
436  }
437 
438  /*
439  * Parse PBKDF string = {hash_algorithm}<scheme_sep><iterations><iter_sep>b64(<salt>)<salt_sep>b64(<hash>)
440  */
441  p = str;
442  end = p + len;
443 
444  q = memchr(p, scheme_sep, end - p);
445  if (!q) {
446  REDEBUG("PBKDF2-Password has no component separators");
447  goto finish;
448  }
449 
450  digest_type = fr_table_value_by_substr(hash_names, (char const *)p, q - p, -1);
451  switch (digest_type) {
452  case FR_SSHA1_PASSWORD:
453  evp_md = EVP_sha1();
454  digest_len = SHA1_DIGEST_LENGTH;
455  break;
456 
457  case FR_SSHA2_224_PASSWORD:
458  evp_md = EVP_sha224();
459  digest_len = SHA224_DIGEST_LENGTH;
460  break;
461 
462  case FR_SSHA2_256_PASSWORD:
463  evp_md = EVP_sha256();
464  digest_len = SHA256_DIGEST_LENGTH;
465  break;
466 
467  case FR_SSHA2_384_PASSWORD:
468  evp_md = EVP_sha384();
469  digest_len = SHA384_DIGEST_LENGTH;
470  break;
471 
472  case FR_SSHA2_512_PASSWORD:
473  evp_md = EVP_sha512();
474  digest_len = SHA512_DIGEST_LENGTH;
475  break;
476 
477 # if OPENSSL_VERSION_NUMBER >= 0x10101000L
478  case FR_SSHA3_224_PASSWORD:
479  evp_md = EVP_sha3_224();
480  digest_len = SHA224_DIGEST_LENGTH;
481  break;
482 
483  case FR_SSHA3_256_PASSWORD:
484  evp_md = EVP_sha3_256();
485  digest_len = SHA256_DIGEST_LENGTH;
486  break;
487 
488  case FR_SSHA3_384_PASSWORD:
489  evp_md = EVP_sha3_384();
490  digest_len = SHA384_DIGEST_LENGTH;
