33 static unlang_action_t unlang_caller(rlm_rcode_t *p_result, request_t *request)
34 {
35  unlang_stack_t *stack = request->stack;
36  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
37  unlang_t *instruction = frame->instruction;
38 
39  unlang_group_t *g;
40  unlang_caller_t *gext;
41 
42  g = unlang_generic_to_group(instruction);
43  gext = unlang_group_to_caller(g);
44 
45  fr_assert(g->num_children > 0); /* otherwise the compilation is broken */
46 
47  /*
48  * No parent, or the dictionaries don't match. Ignore it.
49  */
50  if (!request->parent || (request->parent->dict != gext->dict)) {
51  RDEBUG2("...");
52  return UNLANG_ACTION_EXECUTE_NEXT;
53  }
54 
55  /*
56  * The dictionary matches. Go recurse into its' children.
57  */
58  return unlang_group(p_result, request);
59 }
