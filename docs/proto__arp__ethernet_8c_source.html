64 static ssize_t mod_read(fr_listen_t *li, UNUSED void **packet_ctx, fr_time_t *recv_time_p, uint8_t *buffer, size_t buffer_len, size_t *leftover, UNUSED uint32_t *priority, UNUSED bool *is_dup)
65 {
66  proto_arp_ethernet_thread_t *thread = talloc_get_type_abort(li->thread_instance, proto_arp_ethernet_thread_t);
67  int ret;
68  uint8_t const *data;
69  struct pcap_pkthdr *header;
70  uint8_t const *p, *end;
71  ssize_t len;
72 
73  *leftover = 0; /* always for message oriented protocols */
74 
75  ret = pcap_next_ex(thread->pcap->handle, &header, &data);
76  if (ret == 0) return 0;
77  if (ret < 0) {
78  DEBUG("Failed getting next PCAP packet");
79  return 0;
80  }
81 
82  p = data;
83  end = data + header->caplen;
84 
85  len = fr_pcap_link_layer_offset(data, header->caplen, thread->pcap->link_layer);
86  if (len < 0) {
87  DEBUG("Failed determining link layer header offset");
88  return 0;
89  }
90  p += len;
91 
92  if ((end - p) < FR_ARP_PACKET_SIZE) {
93  DEBUG("Packet is too small (%d) to be ARP", (int) (end - p));
94  return 0;
95  }
96 
97  /*
98  * Shouldn't happen.
99  */
100  if (buffer_len < FR_ARP_PACKET_SIZE) {
101  return 0;
102  }
103 
104  memcpy(buffer, p, FR_ARP_PACKET_SIZE);
105 
106  // @todo - talloc packet_ctx which is the ethernet header, so we know what kind of VLAN, etc. to encode?
107 
108  *recv_time_p = fr_time();
109  return FR_ARP_PACKET_SIZE;
110 }

113 static ssize_t mod_write(fr_listen_t *li, UNUSED void *packet_ctx, UNUSED fr_time_t request_time,
114  uint8_t *buffer, size_t buffer_len, UNUSED size_t written)
115 {
116  proto_arp_ethernet_thread_t *thread = talloc_get_type_abort(li->thread_instance, proto_arp_ethernet_thread_t);
117 
118  int ret;
119  uint8_t arp_packet[64] = { 0 };
120  ethernet_header_t *eth_hdr;
121  fr_arp_packet_t *arp;
122  /* Pointer to the current position in the frame */
123  uint8_t *end = arp_packet;
124 
125  /*
126  * Don't write anything.
127  */
128  if (buffer_len == 1) return buffer_len;
129 
130  /* fill in Ethernet layer (L2) */
131  eth_hdr = (ethernet_header_t *)arp_packet;
132  eth_hdr->ether_type = htons(ETH_TYPE_ARP);
133  end += ETHER_ADDR_LEN + ETHER_ADDR_LEN + sizeof(eth_hdr->ether_type);
134 
135  /*
136  * Just copy what FreeRADIUS has encoded for us.
137  */
138  arp = (fr_arp_packet_t *) end;
139  memcpy(arp, buffer, buffer_len);
140 
141  /*
142  * Set our MAC address as the ethernet source.
143  *
144  * Set the destination MAC as the target address from
145  * ARP.
146  */
147  memcpy(eth_hdr->src_addr, thread->pcap->ether_addr, ETHER_ADDR_LEN);
148  memcpy(eth_hdr->dst_addr, arp->tha, ETHER_ADDR_LEN);
149 
150  /*
151  * If we fail injecting the reply, just ignore it.
152  * Returning <0 means "close the socket", which is likely
153  * not what we want.
154  */
155  ret = pcap_inject(thread->pcap->handle, arp_packet, (end - arp_packet + buffer_len));
156  if (ret < 0) {
157  fr_strerror_printf("Error sending packet with pcap: %d, %s", ret, pcap_geterr(thread->pcap->handle));
158  return 0;
159  }
160 
161  /*
162  * @todo - mirror src/protocols/dhcpv4/pcap.c for ARP send / receive.
163  * We will need that functionality for rlm_arp, too.
164  */
165 
166  return FR_ARP_PACKET_SIZE;
167 }

172 static int mod_open(fr_listen_t *li)
173 {
174  proto_arp_ethernet_t const *inst = talloc_get_type_abort_const(li->app_io_instance, proto_arp_ethernet_t);
175  proto_arp_ethernet_thread_t *thread = talloc_get_type_abort(li->thread_instance, proto_arp_ethernet_thread_t);
176 
177  char const *filter;
178  char *our_filter = NULL;
179 
180  thread->pcap = fr_pcap_init(thread, inst->interface, PCAP_INTERFACE_IN);
181  if (!thread->pcap) {
182  PERROR("Failed initializing pcap handle.");
183  return -1;
184  }
185 
186  if (fr_pcap_open(thread->pcap) < 0) {
187  PERROR("Failed opening interface %s", inst->interface);
188  return -1;
189  }
190 
191  /*
192  * Ensure that we only get ARP, and an optional additional filter.
193  */
194  if (!inst->filter) {
195  filter = "arp";
196  } else {
197  MEM(filter = our_filter = talloc_asprintf(li, "arp and %s", inst->filter));
198  }
199 
200  if (fr_pcap_apply_filter(thread->pcap, filter) < 0) {
201  PERROR("Failed applying pcap filter '%s'", filter);
202  talloc_free(our_filter);
203  return -1;
204  }
205  talloc_free(our_filter);
206 
207  li->fd = thread->pcap->fd;
208 
209  fr_assert(cf_parent(inst->cs) != NULL); /* listen { ... } */
210 
211  thread->name = talloc_asprintf(thread, "arp on interface %s", inst->interface);
212  return 0;
213 }
