18  *
19  * The weird "reverse" function is based on an idea from
20  * "Split-Ordered Lists - Lock-free Resizable Hash Tables", with
21  * modifications so that they're not lock-free. :(
22  *
23  * However, the split-order idea allows a fast & easy splitting of the
24  * hash bucket chain when the hash table is resized. Without it, we'd
25  * have to check & update the pointers for every node in the buck chain,

148 /*
149  * Reverse a key.
150  */
151 static uint32_t reverse(uint32_t key)
152 {
153  /*
154  * Cast to uint32_t is required because the
155  * default type of of the expression is an
156  * int and ubsan correctly complains that
157  * the result of 0xff << 24 won't fit in a
158  * signed 32bit integer.
159  */
160  return (((uint32_t)reversed_byte[key & 0xff] << 24) |
161  ((uint32_t)reversed_byte[(key >> 8) & 0xff] << 16) |
162  ((uint32_t)reversed_byte[(key >> 16) & 0xff] << 8) |
163  ((uint32_t)reversed_byte[(key >> 24) & 0xff]));
164 }
165 
166 /*
167  * Take the parent by discarding the highest bit that is set.
168  */
169 static uint32_t parent_of(uint32_t key)
170 {
171  if (key > 0x00ffffff)
172  return (key & 0x00ffffff) | (parent_byte[key >> 24] << 24);
173 
174  if (key > 0x0000ffff)
175  return (key & 0x0000ffff) | (parent_byte[key >> 16] << 16);
176 
177  if (key > 0x000000ff)
178  return (key & 0x000000ff) | (parent_byte[key >> 8] << 8);
179 
180  return parent_byte[key];
181 }

184 static fr_hash_entry_t *list_find(fr_hash_table_t *ht,
185  fr_hash_entry_t *head, uint32_t reversed, void const *data)
186 {
187  fr_hash_entry_t *cur;
188 
189  for (cur = head; cur != &ht->null; cur = cur->next) {
190  if (cur->reversed == reversed) {
191  if (ht->cmp) {
192  int cmp = ht->cmp(data, cur->data);
193  if (cmp > 0) break;
194  if (cmp < 0) continue;
195  }
196  return cur;
197  }
198  if (cur->reversed > reversed) break;
199  }
200 
201  return NULL;
202 }

205 /*
206  * Inserts a new entry into the list, in order.
207  */
208 static int list_insert(fr_hash_table_t *ht,
209  fr_hash_entry_t **head, fr_hash_entry_t *node)
210 {
211  fr_hash_entry_t **last, *cur;
212 
213  last = head;
214 
215  for (cur = *head; cur != &ht->null; cur = cur->next) {
216  if (cur->reversed > node->reversed) break;
217  last = &(cur->next);
218 
219  if (cur->reversed == node->reversed) {
220  if (ht->cmp) {
221  int cmp = ht->cmp(node->data, cur->data);
222  if (cmp > 0) break;
223  if (cmp < 0) continue;
224  }
225  return 0;
226  }
227  }
228 
229  node->next = *last;
230  *last = node;
231 
232  return 1;
233 }

256 static int _fr_hash_table_free(fr_hash_table_t *ht)
257 {
258  int i;
259  fr_hash_entry_t *node, *next;
260 
261  /*
262  * Walk over the buckets, freeing them all.
263  */
264  for (i = 0; i < ht->num_buckets; i++) {
265  if (ht->buckets[i]) for (node = ht->buckets[i];
266  node != &ht->null;
267  node = next) {
268  next = node->next;
269  if (!node->data) continue; /* dummy entry */
270

272  }
273  }
274  talloc_free(ht->buckets);
275 
276  return 0;

279 /*
280  * Create the table.
281  *
282  * Memory usage in bytes is (20/3) * number of entries.
283  */
284 fr_hash_table_t *fr_hash_table_create(TALLOC_CTX *ctx,
285  fr_hash_table_hash_t hash_func,
286  fr_hash_table_cmp_t cmp_func,
287  fr_hash_table_free_t free_func)
288 {
289  fr_hash_table_t *ht;
290 
291  if (!hash_func) return NULL;
292 
293  ht = talloc_zero(NULL, fr_hash_table_t);
294  if (!ht) return NULL;
295  talloc_set_destructor(ht, _fr_hash_table_free);
296  talloc_link_ctx(ctx, ht);
297 
298  ht->free = free_func;
299  ht->hash = hash_func;
300  ht->cmp = cmp_func;
301  ht->num_buckets = FR_HASH_NUM_BUCKETS;
302  ht->mask = ht->num_buckets - 1;
303 
304  /*
305  * Have a default load factor of 2.5. In practice this
306  * means that the average load will hit 3 before the
307  * table grows.
308  */
309  ht->next_grow = (ht->num_buckets << 1) + (ht->num_buckets >> 1);
310 
311  ht->buckets = talloc_zero_array(NULL, fr_hash_entry_t *, ht->num_buckets);
312  if (!ht->buckets) {
313  talloc_free(ht);
314  return NULL;
315  }
316 
317  ht->null.reversed = ~0;
318  ht->null.key = ~0;
319  ht->null.next = &ht->null;
320  ht->buckets[0] = &ht->null;
321 
322  return ht;
323 }

326 /*
327  * If the current bucket is uninitialized, initialize it
328  * by recursively copying information from the parent.
329  *
330  * We may have a situation where entry E is a parent to 2 other
331  * entries E' and E". If we split E into E and E', then the
332  * nodes meant for E" end up in E or E', either of which is
333  * wrong. To solve that problem, we walk down the whole chain,
334  * inserting the elements into the correct place.
335  */
336 static void fr_hash_table_fixup(fr_hash_table_t *ht, uint32_t entry)
337 {
338  uint32_t parent_entry;
339  fr_hash_entry_t **last, *cur;
340  uint32_t this;
341 
342  parent_entry = parent_of(entry);
343 
344  /* parent_entry == entry if and only if entry == 0 */
345 
346  if (!ht->buckets[parent_entry]) {
347  fr_hash_table_fixup(ht, parent_entry);
348  }
349 
350  /*
351  * Keep walking down cur, trying to find entries that
352  * don't belong here any more. There may be multiple
353  * ones, so we can't have a naive algorithm...
354  */
355  last = &ht->buckets[parent_entry];
356  this = parent_entry;
357 
358  for (cur = *last; cur != &ht->null; cur = cur->next) {
359  uint32_t real_entry;
360 
361  real_entry = cur->key & ht->mask;
362  if (real_entry != this) { /* ht->buckets[real_entry] == NULL */
363  *last = &ht->null;
364  ht->buckets[real_entry] = cur;
365  this = real_entry;
366  }
367 
368  last = &(cur->next);
369  }
370 
371  /*
372  * We may NOT have initialized this bucket, so do it now.
373  */
374  if (!ht->buckets[entry]) ht->buckets[entry] = &ht->null;
375 }

383 /*
384  * Grow the hash table.
385  */
386 static void fr_hash_table_grow(fr_hash_table_t *ht)
387 {
388  fr_hash_entry_t **buckets;
389 
390  buckets = talloc_zero_array(NULL, fr_hash_entry_t *, GROW_FACTOR * ht->num_buckets);
391  if (!buckets) return;
392 
393  memcpy(buckets, ht->buckets, sizeof(*buckets) * ht->num_buckets);
394  talloc_free(ht->buckets); /* Free the old buckets */
395 
396  ht->buckets = buckets;
397  ht->num_buckets *= GROW_FACTOR;
398  ht->next_grow *= GROW_FACTOR;
399  ht->mask = ht->num_buckets - 1;
400 #ifdef TESTING
401  grow = 1;
402  fprintf(stderr, "GROW TO %d\n", ht->num_buckets);
403 #endif
404 }

407 /*
408  * Insert data.
409  */
410 int fr_hash_table_insert(fr_hash_table_t *ht, void const *data)
411 {
412  uint32_t key;
413  uint32_t entry;
414  uint32_t reversed;
415  fr_hash_entry_t *node;
416 
417  if (!ht || !data) return 0;
418 
419  key = ht->hash(data);
420  entry = key & ht->mask;
421  reversed = reverse(key);
422 
423  if (!ht->buckets[entry]) fr_hash_table_fixup(ht, entry);
424 
425  /*
426  * If we try to do our own memory allocation here, the
427  * speedup is only ~15% or so, which isn't worth it.
428  */
429  node = talloc_zero(NULL, fr_hash_entry_t);
430  if (!node) return 0;
431 
432  node->next = &ht->null;
433  node->reversed = reversed;
434  node->key = key;
435  memcpy(&node->data, &data, sizeof(node->data));
436 
437  /* already in the table, can't insert it */
438  if (!list_insert(ht, &ht->buckets[entry], node)) {
439  talloc_free(node);
440  return 0;
441  }
442 
443  /*
444  * Check the load factor, and grow the table if
445  * necessary.
446  */
447  ht->num_elements++;
448  if (ht->num_elements >= ht->next_grow) fr_hash_table_grow(ht);
449 
450  return 1;
451 }

530 {

532  uint32_t entry;
533  uint32_t reversed;
534  void *old;
535  fr_hash_entry_t *node;
536

541  reversed = reverse(key);
542 
543  if (!ht->buckets[entry]) fr_hash_table_fixup(ht, entry);

617 /*
618  * Walk over the nodes, allowing deletes & inserts to happen.
619  */
620 int fr_hash_table_walk(fr_hash_table_t *ht,
621  fr_hash_table_walk_t callback,
622  void *context)
623 {
624  int i, ret;
625 
626  if (!ht || !callback) return 0;
627 
628  for (i = ht->num_buckets - 1; i >= 0; i--) {
629  fr_hash_entry_t *node, *next;
630 
631  /*
632  * Ensure that the current bucket is filled.
633  */
634  if (!ht->buckets[i]) fr_hash_table_fixup(ht, i);
635 
636  for (node = ht->buckets[i]; node != &ht->null; node = next) {
637  next = node->next;
638 
639  ret = callback(context, node->data);
640 
641  if (ret != 0) return ret;
642  }
643  }
644 
645  return 0;
646 }

648 /** Iterate over entries in a hash table
649  *
650  * @note If the hash table is modified the iterator should be considered invalidated.
651  *
652  * @param[in] ht to iterate over.
653  * @param[in] iter Pointer to an iterator struct, used to maintain
654  * state between calls.
655  * @return
656  * - User data.
657  * - NULL if at the end of the list.
658  */
659 void *fr_hash_table_iter_next(fr_hash_table_t *ht, fr_hash_iter_t *iter)
660 {
661  fr_hash_entry_t *node;
662 
663  if (unlikely(!ht)) return NULL;
664 
665  /*
666  * Return the next element in the bucket
667  */
668  if (iter->node != &ht->null) {
669  node = iter->node;
670  iter->node = node->next;
671 
672  return node->data;
673  }
674 
675  /*
676  * We might have to go through multiple empty
677  * buckets to find one that contains something
678  * we should return
679  */
680  while ((--iter->bucket) >= 0) {
681  if (!ht->buckets[iter->bucket]) fr_hash_table_fixup(ht, iter->bucket);
682 
683  node = ht->buckets[iter->bucket];
684  if (node == &ht->null) continue; /* This bucket was empty too... */
685 
686  iter->node = node->next; /* Store the next one to examine */
687  return node->data;
688  }
689 
690  return NULL;
691 }

805 /*
806  * A fast hash function. For details, see:
807  *
808  * http://www.isthe.com/chongo/tech/comp/fnv/
809  *
810  * Which also includes public domain source. We've re-written
811  * it here for our purposes.
812  */
813 uint32_t fr_hash(void const *data, size_t size)
814 {
815  uint8_t const *p = data;
816  uint8_t const *q = p + size;
817  uint32_t hash = FNV_MAGIC_INIT;
818 
819  /*
820  * FNV-1 hash each octet in the buffer
821  */
822  while (p != q) {
823  /*
824  * XOR the 8-bit quantity into the bottom of
825  * the hash.
826  */
827  hash ^= (uint32_t) (*p++);
828 
829  /*
830  * Multiple by 32-bit magic FNV prime, mod 2^32
831  */
832  hash *= FNV_MAGIC_PRIME;
833 #if 0
834  /*
835  * Potential optimization.
836  */
837  hash += (hash<<1) + (hash<<4) + (hash<<7) + (hash<<8) + (hash<<24);
838 #endif
839  }
840 
841  return hash;
842 }

845  * Continue hashing data.
846  */
847 uint32_t fr_hash_update(void const *data, size_t size, uint32_t hash)
848 {
849  uint8_t const *p = data;
850  uint8_t const *q;
851 
852  if (size == 0) return hash; /* Avoid ubsan issues with access NULL pointer */
853 
854  q = p + size;
855  while (p < q) {
856  hash *= FNV_MAGIC_PRIME;
857  hash ^= (uint32_t) (*p++);
858  }
859 
860  return hash;
861 }
