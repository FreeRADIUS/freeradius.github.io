29 static struct msgb *isup_status_alloc(int cic, int msg_type, uint8_t *extra, int range, int val)
30 {
31  struct isup_msg_hdr *hdr;
32  struct msgb *msg;
33  int bits, len;
34  uint8_t *data;
35 
36  msg = msgb_alloc_headroom(4096, 128, "ISUP Simple MSG");
37  if (!msg) {
38  LOGP(DISUP, LOGL_ERROR, "Allocation of status message failed.\n");
39  return NULL;
40  }
41 
42  msg->l2h = msgb_put(msg, sizeof(*hdr));
43 
44  /* write the ISUP header */
45  hdr = (struct isup_msg_hdr *) msg->l2h;
46  hdr->cic = cic;
47  hdr->msg_type = msg_type;
48 
49  if (extra)
50  msgb_v_put(msg, *extra);
51 
52  /*
53  * place the pointers here.
54  * 1.) place the variable start after us
55  * 2.) place the length
56  */
57  msgb_v_put(msg, 1);
58 
59  bits = range + 1;
60  len = (bits / 8) + 1;
61  msgb_v_put(msg, len + 1);
62  msgb_v_put(msg, range);
63 
64  data = msgb_put(msg, len);
65 
66  /* set the status bits to val... FIXME this sets the extra bits too */
67  memset(data, val, len);
68 
69  return msg;
70 }

95 int isup_parse_status(const uint8_t *data, uint8_t in_length)
96 {
97  uint8_t ptr;
98 
99  if (in_length < 3) {
100  LOGP(DISUP, LOGL_ERROR, "This needs three bytes.\n");
101  return -1;
102  }
103 
104  ptr = data[0];
105  if (1 + ptr > in_length) {
106  LOGP(DISUP, LOGL_ERROR, "Pointing outside the packet.\n");
107  return -1;
108  }
109 
110  if (1 + ptr + 1 > in_length) {
111  LOGP(DISUP, LOGL_ERROR, "No space for the data.\n");
112  return -1;
113  }
114 
115  return data[0 + ptr + 1];
116 }

218 int mtp_link_set_isup(struct mtp_link_set *set, struct msgb *msg, int sls)
219 {
220  int rc = -1;
221  int payload_size;
222  struct isup_msg_hdr *hdr;
223 
224  if (msgb_l3len(msg) < sizeof(*hdr)) {
225  LOGP(DISUP, LOGL_ERROR, "ISUP header is too short.\n");
226  return -1;
227  }
228 
229  if (set->pass_all_isup) {
230  mtp_link_set_forward_isup(set, msg, sls);
231  return 0;
232  }
233 
234  hdr = (struct isup_msg_hdr *) msg->l3h;
235  payload_size = msgb_l3len(msg) - sizeof(*hdr);
236 
237  switch (hdr->msg_type) {
238  case ISUP_MSG_GRS:
239  rc = handle_circuit_reset_grs(set, sls, hdr->cic, hdr->data, payload_size);
240  break;
241  case ISUP_MSG_CGB:
242  rc = handle_circuit_reset_cgb(set, sls, hdr->cic, hdr->data, payload_size);
243  if (rc == 0)
244  rc = send_cgu(set, sls, hdr->cic, 28);
245  break;
246  case ISUP_MSG_CGU:
247  rc = handle_cgu(set, sls, hdr->cic, hdr->data, payload_size);
248  break;
249  case ISUP_MSG_CGUA:
250  LOGP(DISUP, LOGL_NOTICE, "CIC %d is now unblocked on linkset %d/%s.\n",
251  hdr->cic, set->nr, set->name);
252  break;
253  case ISUP_MSG_RSC:
254  rc = handle_simple_resp(set, sls, hdr->cic, ISUP_MSG_RLC);
255  break;
256  default:
257  mtp_link_set_forward_isup(set, msg, sls);
258  rc = 0;
259  break;
260  }
261 
262  return rc;
263 }
