248  ( (tptr)[0] == (bptr)[0] && \

251 /*
252  * Read a word from a buffer and advance pointer.
253  * This function knows about escapes and quotes.
254  *
255  * At end-of-line, buf[0] is set to '\0'.
256  * Returns 0 or special token value.
257  */
258 static fr_token_t getthing(char const **ptr, char *buf, int buflen, bool tok,
259  fr_table_num_ordered_t const *tokenlist, size_t tokenlist_len, bool unescape)
260 {
261  char *s;
262  char const *p;
263  char quote;
264  unsigned int x;
265  size_t i;
266  fr_token_t token;
267 
268  buf[0] = '\0';
269 
270  /* Skip whitespace */
271  p = *ptr;
272 
273  fr_skip_whitespace(p);
274 
275  if (!*p) {
276  *ptr = p;
277  return T_EOL;
278  }
279 
280  /*
281  * Might be a 1 or 2 character token.
282  */
283  if (tok) {
284  for (i = 0; i < tokenlist_len; i++) {
285  if (TOKEN_MATCH(p, tokenlist[i].name.str)) {
286  strcpy(buf, tokenlist[i].name.str);
287  p += tokenlist[i].name.len;
288 
289  token = tokenlist[i].value;
290  goto done;
291  }
292  }
293  }
294 
295  /* Read word. */
296  quote = '\0';
297  switch (*p) {
298  default:
299  token = T_BARE_WORD;
300  break;
301 
302  case '\'':
303  token = T_SINGLE_QUOTED_STRING;
304  break;
305 
306  case '"':
307  token = T_DOUBLE_QUOTED_STRING;
308  break;
309 
310  case '`':
311  token = T_BACK_QUOTED_STRING;
312  break;
313  }
314 
315  if (token != T_BARE_WORD) {
316  quote = *p;
317  p++;
318  }
319  s = buf;
320 
321  while (*p && buflen-- > 1) {
322  /*
323  * We're looking for strings. Stop on spaces, or
324  * (if given a token list), on a token, or on a
325  * comma.
326  */
327  if (!quote) {
328  if (isspace((int) *p)) break;
329 
330 
331  if (tok) {
332  for (i = 0; i < tokenlist_len; i++) {
333  if (TOKEN_MATCH(p, tokenlist[i].name.str)) {
334  *s++ = 0;
335  goto done;
336  }
337  }
338  }
339  if (*p == ',') break;
340 
341  /*
342  * Copy the character over.
343  */
344  *s++ = *p++;
345  continue;
346  } /* else there was a quotation character */
347 
348  /*
349  * Un-escaped quote character. We're done.
350  */
351  if (*p == quote) {
352  p++;
353  *s++ = 0;
354  goto done;
355  }
356 
357  /*
358  * Everything but backslash gets copied over.
359  */
360  if (*p != '\\') {
361  *s++ = *p++;
362  continue;
363  }
364 
365  /*
366  * There's nothing after the backslash, it's an error.
367  */
368  if (!p[1]) {
369  fr_strerror_printf("Unterminated string");
370  return T_INVALID;
371  }
372 
373  if (unescape) {
374  p++;
375 
376  switch (*p) {
377  case 'r':
378  *s++ = '\r';
379  break;
380  case 'n':
381  *s++ = '\n';
382  break;
383  case 't':
384  *s++ = '\t';
385  break;
386 
387  default:
388  if (*p >= '0' && *p <= '9' &&
389  sscanf(p, "%3o", &x) == 1) {
390  *s++ = x;
391  p += 2;
392  } else
393  *s++ = *p;
394  break;
395  }
396  p++;
397 
398  } else {
399  /*
400  * Convert backslash-quote to quote, but
401  * leave everything else alone.
402  */
403  if (p[1] == quote) { /* convert '\'' --> ' */
404  p++;
405  } else {
406  if (buflen < 2) {
407  fr_strerror_printf("Truncated input");
408  return T_INVALID;
409  }
410 
411  *(s++) = *(p++);
412  }
413  *(s++) = *(p++);
414  }
415  }
416 
417  *s++ = 0;
418 
419  if (quote) {
420  fr_strerror_printf("Unterminated string");
421  return T_INVALID;
422  }
423 
424 done:
425  /* Skip whitespace again. */
426  fr_skip_whitespace(p);
427 
428  *ptr = p;
429 
430  return token;
431 }

471 {

495 ssize_t fr_skip_string(char const *start, char const *end)
496 {
497  char const *p = start;
498  char quote;
499 
500  quote = *(p++);
501 
502  while (p < end) {
503  if (*p == quote) {
504  p++;
505  return p - start;
506  }
507 
508  if (*p == '\\') {
509  if (((p + 1) >= end) || !p[1]) {
510  break;
511  }
512 
513  p += 2;
514  continue;
515  }
516 
517  p++;
518  }
519 
520  /*
521  * Unexpected end of string.
522  */
523  fr_strerror_printf("Unexpected end of string");
524  return -(p - start);
