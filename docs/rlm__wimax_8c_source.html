113 /*
114  * Find the named user in this modules database. Create the set
115  * of attribute-value pairs to check and reply with for this user
116  * from the database. The authentication code only needs to check
117  * the password, the rest is done here.
118  */
119 static unlang_action_t CC_HINT(nonnull) mod_authorize(rlm_rcode_t *p_result, UNUSED module_ctx_t const *mctx, request_t *request)
120 {
121  fr_pair_t *vp;
122 
123  /*
124  * Fix Calling-Station-Id. Damn you, WiMAX!
125  */
126  vp = fr_pair_find_by_da(&request->request_pairs, attr_calling_station_id);
127  if (vp && (vp->vp_length == 6)) {
128  int i;
129  char *p;
130  uint8_t buffer[6];
131 
132  memcpy(buffer, vp->vp_strvalue, 6);
133 
134  MEM(fr_pair_value_bstr_realloc(vp, &p, (5 * 3) + 2) == 0);
135 
136  /*
137  * RFC 3580 Section 3.20 says this is the preferred
138  * format. Everyone *SANE* is using this format,
139  * so we fix it here.
140  */
141  for (i = 0; i < 6; i++) {
142  fr_bin2hex(&FR_SBUFF_OUT(&p[i * 3], 2 + 1), &FR_DBUFF_TMP(&buffer[i], 1), SIZE_MAX);
143  p[(i * 3) + 2] = '-';
144  }
145 
146  DEBUG2("Fixing WiMAX binary Calling-Station-Id to %pV", &vp->data);
147  RETURN_MODULE_OK;
148  }
149 
150  RETURN_MODULE_NOOP;
151 }

161 /*
162  * Generate the keys after the user has been authenticated.
163  */
164 static unlang_action_t CC_HINT(nonnull) mod_post_auth(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
165 {
166  rlm_wimax_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_wimax_t);
167  fr_pair_t *msk, *emsk, *vp;
168  fr_pair_t *mn_nai, *ip, *fa_rk;
169  HMAC_CTX *hmac;
170  unsigned int rk1_len, rk2_len, rk_len;
171  uint32_t mip_spi;
172  uint8_t usage_data[24];
173  uint8_t mip_rk_1[EVP_MAX_MD_SIZE], mip_rk_2[EVP_MAX_MD_SIZE];
174  uint8_t mip_rk[2 * EVP_MAX_MD_SIZE];
175 
176  msk = fr_pair_find_by_da(&request->reply_pairs, attr_eap_msk);
177  emsk = fr_pair_find_by_da(&request->reply_pairs, attr_eap_emsk);
178  if (!msk || !emsk) {
179  REDEBUG2("No EAP-MSK or EAP-EMSK. Cannot create WiMAX keys");
180  RETURN_MODULE_NOOP;
181  }
182 
183  /*
184  * If we delete the MS-MPPE-*-Key attributes, then add in
185  * the WiMAX-MSK so that the client has a key available.
186  */
187  if (inst->delete_mppe_keys) {
188  pair_delete_reply(attr_ms_mppe_send_key);
189  pair_delete_reply(attr_ms_mppe_recv_key);
190 
191  MEM(pair_update_reply(&vp, attr_wimax_msk) >= 0);
192  fr_pair_value_memdup(vp, msk->vp_octets, msk->vp_length, false);
193  }
194 
195  /*
196  * Initialize usage data.
197  */
198  memcpy(usage_data, "miprk@wimaxforum.org", 21); /* with trailing \0 */
199  usage_data[21] = 0x02;
200  usage_data[22] = 0x00;
201  usage_data[23] = 0x01;
202 
203  /*
204  * MIP-RK-1 = HMAC-SSHA256(EMSK, usage-data | 0x01)
205  */
206  hmac = HMAC_CTX_new();
207  HMAC_Init_ex(hmac, emsk->vp_octets, emsk->vp_length, EVP_sha256(), NULL);
208 
209  HMAC_Update(hmac, &usage_data[0], sizeof(usage_data));
210  HMAC_Final(hmac, &mip_rk_1[0], &rk1_len);
211 
212  /*
213  * MIP-RK-2 = HMAC-SSHA256(EMSK, MIP-RK-1 | usage-data | 0x01)
214  */
215  HMAC_Init_ex(hmac, emsk->vp_octets, emsk->vp_length, EVP_sha256(), NULL);
216 
217  HMAC_Update(hmac, (uint8_t const *) &mip_rk_1, rk1_len);
218  HMAC_Update(hmac, &usage_data[0], sizeof(usage_data));
219  HMAC_Final(hmac, &mip_rk_2[0], &rk2_len);
220 
221  memcpy(mip_rk, mip_rk_1, rk1_len);
222  memcpy(mip_rk + rk1_len, mip_rk_2, rk2_len);
223  rk_len = rk1_len + rk2_len;
224 
225  /*
226  * MIP-SPI = HMAC-SSHA256(MIP-RK, "SPI CMIP PMIP");
227  */
228  HMAC_Init_ex(hmac, mip_rk, rk_len, EVP_sha256(), NULL);
229 
230  HMAC_Update(hmac, (uint8_t const *) "SPI CMIP PMIP", 12);
231  HMAC_Final(hmac, &mip_rk_1[0], &rk1_len);
232 
233  /*
234  * Take the 4 most significant octets.
235  * If less than 256, add 256.
236  */
237  mip_spi = ((mip_rk_1[0] << 24) | (mip_rk_1[1] << 16) |
238  (mip_rk_1[2] << 8) | mip_rk_1[3]);
239  if (mip_spi < 256) mip_spi += 256;
240 
241  REDEBUG2("MIP-RK = 0x%pH", fr_box_octets(mip_rk, rk_len));
242  REDEBUG2("MIP-SPI = %08x", ntohl(mip_spi));
243 
244  /*
245  * FIXME: Perform SPI collision prevention
246  */
247 
248  /*
249  * Calculate mobility keys
250  */
251  mn_nai = fr_pair_find_by_da(&request->request_pairs, attr_wimax_mn_nai);
252  if (!mn_nai) mn_nai = fr_pair_find_by_da(&request->reply_pairs, attr_wimax_mn_nai);
253  if (!mn_nai) {
254  RWDEBUG("%s was not found in the request or in the reply", attr_wimax_mn_nai->name);
255  RWDEBUG("We cannot calculate MN-HA keys");
256  }
257 
258  /*
259  * WiMAX-IP-Technology
260  */
261  vp = NULL;
262  if (mn_nai) vp = fr_pair_find_by_da(&request->reply_pairs, attr_wimax_ip_technology);
263  if (!vp) {
264  RWDEBUG("%s not found in reply", attr_wimax_ip_technology->name);
265  RWDEBUG("Not calculating MN-HA keys");
266  }
267 
268  if (vp) switch (vp->vp_uint32) {
269  case 2: /* PMIP4 */
270  /*
271  * Look for WiMAX-hHA-IP-MIP4
272  */
273  ip = fr_pair_find_by_da(&request->reply_pairs, attr_wimax_hha_ip_mip4);
274  if (!ip) {
275  RWDEBUG("%s not found. Cannot calculate MN-HA-PMIP4 key", attr_wimax_hha_ip_mip4->name);
276  break;
277  }
278 
279  /*
280  * MN-HA-PMIP4 =
281  * H(MIP-RK, "PMIP4 MN HA" | HA-IPv4 | MN-NAI);
282  */
283  HMAC_Init_ex(hmac, mip_rk, rk_len, EVP_sha1(), NULL);
284 
285  HMAC_Update(hmac, (uint8_t const *) "PMIP4 MN HA", 11);
286  HMAC_Update(hmac, (uint8_t const *) &ip->vp_ipv4addr, 4);
287  HMAC_Update(hmac, (uint8_t const *) &mn_nai->vp_strvalue, mn_nai->vp_length);
288  HMAC_Final(hmac, &mip_rk_1[0], &rk1_len);
289 
290  /*
291  * Put MN-HA-PMIP4 into WiMAX-MN-hHA-MIP4-Key
292  */
293  MEM(pair_update_reply(&vp, attr_wimax_mn_hha_mip4_key) >= 0);
294  fr_pair_value_memdup(vp, &mip_rk_1[0], rk1_len, false);
295 
296  /*
297  * Put MN-HA-PMIP4-SPI into WiMAX-MN-hHA-MIP4-SPI
298  */
299  MEM(pair_update_reply(&vp, attr_wimax_mn_hha_mip4_spi) >= 0);
300  vp->vp_uint32 = mip_spi + 1;
301  break;
302 
303  case 3: /* CMIP4 */
304  /*
305  * Look for WiMAX-hHA-IP-MIP4
306  */
307  ip = fr_pair_find_by_da(&request->reply_pairs, attr_wimax_hha_ip_mip4);
308  if (!ip) {
309  RWDEBUG("%s not found. Cannot calculate MN-HA-CMIP4 key", attr_wimax_hha_ip_mip4->name);
310  break;
311  }
312 
313  /*
314  * MN-HA-CMIP4 =
315  * H(MIP-RK, "CMIP4 MN HA" | HA-IPv4 | MN-NAI);
316  */
317  HMAC_Init_ex(hmac, mip_rk, rk_len, EVP_sha1(), NULL);
318 
319  HMAC_Update(hmac, (uint8_t const *) "CMIP4 MN HA", 11);
320  HMAC_Update(hmac, (uint8_t const *) &ip->vp_ipv4addr, 4);
321  HMAC_Update(hmac, (uint8_t const *) &mn_nai->vp_strvalue, mn_nai->vp_length);
322  HMAC_Final(hmac, &mip_rk_1[0], &rk1_len);
323 
324  /*
325  * Put MN-HA-CMIP4 into WiMAX-MN-hHA-MIP4-Key
326  */
327  MEM(pair_update_reply(&vp, attr_wimax_mn_hha_mip4_key) >= 0);
328  fr_pair_value_memdup(vp, &mip_rk_1[0], rk1_len, false);
329 
330  /*
331  * Put MN-HA-CMIP4-SPI into WiMAX-MN-hHA-MIP4-SPI
332  */
333  MEM(pair_update_reply(&vp, attr_wimax_mn_hha_mip4_spi) >= 0);
334  vp->vp_uint32 = mip_spi;
335  break;
336 
337  case 4: /* CMIP6 */
338  /*
339  * Look for WiMAX-hHA-IP-MIP6
340  */
341  ip = fr_pair_find_by_da(&request->reply_pairs, attr_wimax_hha_ip_mip6);
342  if (!ip) {
343  RWDEBUG("%s not found. Cannot calculate MN-HA-CMIP6 key", attr_wimax_hha_ip_mip6->name);
344  break;
345  }
346 
347  /*
348  * MN-HA-CMIP6 =
349  * H(MIP-RK, "CMIP6 MN HA" | HA-IPv6 | MN-NAI);
350  */
351  HMAC_Init_ex(hmac, mip_rk, rk_len, EVP_sha1(), NULL);
352 
353  HMAC_Update(hmac, (uint8_t const *) "CMIP6 MN HA", 11);
354  HMAC_Update(hmac, (uint8_t const *) &ip->vp_ipv6addr, 16);
355  HMAC_Update(hmac, (uint8_t const *) &mn_nai->vp_strvalue, mn_nai->vp_length);
356  HMAC_Final(hmac, &mip_rk_1[0], &rk1_len);
357 
358  /*
359  * Put MN-HA-CMIP6 into WiMAX-MN-hHA-MIP6-Key
360  */
361  MEM(pair_update_reply(&vp, attr_wimax_mn_hha_mip6_key) >= 0);
362  fr_pair_value_memdup(vp, &mip_rk_1[0], rk1_len, false);
363 
364  /*
365  * Put MN-HA-CMIP6-SPI into WiMAX-MN-hHA-MIP6-SPI
366  */
367  MEM(pair_update_reply(&vp, attr_wimax_mn_hha_mip6_spi) >= 0);
368  vp->vp_uint32 = mip_spi + 2;
369  break;
370 
371  default:
372  break; /* do nothing */
373  }
374 
375  /*
376  * Generate FA-RK, if requested.
377  *
378  * FA-RK= H(MIP-RK, "FA-RK")
379  */
380  fa_rk = fr_pair_find_by_da(&request->reply_pairs, attr_wimax_fa_rk_key);
381  if (fa_rk && (fa_rk->vp_length <= 1)) {
382  HMAC_Init_ex(hmac, mip_rk, rk_len, EVP_sha1(), NULL);
383 
384  HMAC_Update(hmac, (uint8_t const *) "FA-RK", 5);
385 
386  HMAC_Final(hmac, &mip_rk_1[0], &rk1_len);
387 
388  fr_pair_value_memdup(fa_rk, &mip_rk_1[0], rk1_len, false);
389  }
390 
391  /*
392  * Create FA-RK-SPI, which is really SPI-CMIP4, which is
393  * really MIP-SPI. Clear? Of course. This is WiMAX.
394  */
395  if (fa_rk) {
396  MEM(pair_update_reply(&vp, attr_wimax_fa_rk_spi) >= 0);
397  vp->vp_uint32 = mip_spi;
398  }
399 
400  /*
401  * Give additional information about requests && responses
402  *
403  * WiMAX-RRQ-MN-HA-SPI
404  */
405  vp = fr_pair_find_by_da(&request->request_pairs, attr_wimax_rrq_mn_ha_spi);
406  if (vp) {
407  REDEBUG2("Client requested MN-HA key: Should use SPI to look up key from storage");
408  if (!mn_nai) {
409  RWDEBUG("MN-NAI was not found!");
410  }
411 
412  /*
413  * WiMAX-RRQ-HA-IP
414  */
415  if (!fr_pair_find_by_da(&request->request_pairs, attr_wimax_rrq_ha_ip)) {
416  RWDEBUG("HA-IP was not found!");
417  }
418 
419  /*
420  * WiMAX-HA-RK-Key-Requested
421  */
422  vp = fr_pair_find_by_da(&request->request_pairs, attr_wimax_ha_rk_key_requested);
423  if (vp && (vp->vp_uint32 == 1)) {
424  REDEBUG2("Client requested HA-RK: Should use IP to look it up from storage");
425  }
426  }
427 
428  /*
429  * Wipe the context of all sensitive information.
430  */
431  HMAC_CTX_free(hmac);
432 
433  RETURN_MODULE_UPDATED;
434 }
