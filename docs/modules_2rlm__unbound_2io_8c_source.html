49 /** Wrapper around event handle for our event loop
50  *
51  * This stores libunbound specific information for an event in our event loop.
52  *
53  * Lifetime should be bound to the event base.
54  */
55 typedef struct {
56  struct ub_event base; //!< Unbound event base, which we populate with
57  ///< callback functions for adding events for FDs
58  ///< setting timers etc...
59  ///< MUST BE LISTED FIRST.
60 
61  unbound_io_event_base_t *ev_b; //!< Event base this handle was created for.
62 
63  fr_event_timer_t const *timer; //!< Stores the pointer to the enabled timer for
64  ///< this event handled. libunbound uses a single
65  ///< handle for managing related FD events and
66  ///< timers, which is weird, but ok...
67 
68  short events; //!< The events this event handle should receive
69  ///< when activated.
70 
71  int fd; //!< File descriptor this event handle relates to.
72 
73  unbound_cb_t cb; //!< The callback we need to call when a specified
74  ///< event happens.
75 
76  void *uctx; //!< This is the argument libunbound wants passed to
77  ///< the callback when it's called. It usually
78  ///< contains libunbound's internal connection handled.
79  ///< We don't have any visibility, it just remains
80  ///< an opaque blob to us.
81 
82  bool active; //!< Whether this event is considered active.
83 } unbound_io_event_t;

211 /** Unbound FD errored
212  *
213  * libunbound doesn't request errors, so tell it a timeout occurred
214  *
215  * Because we don't have the separation between the IO event loop
216  * and the event loop processing results, we call ub_process
217  * immediately after calling the IO callback.
218  */
219 static void _unbound_io_service_errored(UNUSED fr_event_list_t *el,
220  int fd, UNUSED int flags, int fd_errno, void *uctx)
221 {
222  unbound_io_event_t *ev = talloc_get_type_abort(uctx, unbound_io_event_t);
223 
224  fr_assert(ev->active); /* must be active */
225 
226  DEBUG4("unbound event %p - FD %i errored: %s", ev, fd, fr_syserror(fd_errno));
227 
228  /*
229  * Delete the timer as we're telling libunbound
230  * that it fired. This is imperfect but unbound
231  * doesn't have a callback for receiving errors.
232  */
233  if (fr_event_timer_delete(&ev->timer) < 0) {
234  PERROR("ubound event %p - Failed disarming timeout", ev);
235  }
236 
237  ev->cb(-1, UB_EV_TIMEOUT, ev->uctx); /* Call libunbound - pretend this is a timeout */
238 }

241 /** Activate FD events and set a timer for a timeout
242  *
243  */
244 static int _unbound_io_event_activate(struct ub_event *ub_ev, struct timeval *tv)
245 {
246  unbound_io_event_t *ev = talloc_get_type_abort(ub_ev, unbound_io_event_t);
247 
248  fr_assert(!ev->active); /* must not be active */
249 
250  /*
251  * File descriptor event
252  */
253  if ((ev->events & UB_EV_READ) && (ev->events & UB_EV_WRITE)) {
254  fr_assert(ev->fd >= 0); /* File descriptor must valid */
255 
256  DEBUG4("unbound event %p - Registered for read+write events on FD %i", ev, ev->fd);
257 
258  if (fr_event_fd_insert(ev, ev->ev_b->el, ev->fd,
259  _unbound_io_service_readable,
260  _unbound_io_service_writable,
261  _unbound_io_service_errored,
262  ev) < 0) {
263  PERROR("unbound event %p - Registration failed for read+write+error events on FD %i",
264  ev, ev->fd);
265 
266  return -1;
267  }
268  } else if (ev->events & UB_EV_READ) {
269  fr_assert(ev->fd >= 0); /* File descriptor must valid */
270 
271  DEBUG4("unbound event %p - Registered for read+error events on FD %i", ev, ev->fd);
272 
273  if (fr_event_fd_insert(ev, ev->ev_b->el, ev->fd,
274  _unbound_io_service_readable,
275  NULL,
276  _unbound_io_service_errored,
277  ev) < 0) {
278  PERROR("unbound event %p - Registration failed for read+error events on FD %i",
279  ev, ev->fd);
280 
281  return -1;
282  }
283  } else if (ev->events & UB_EV_WRITE) {
284  fr_assert(ev->fd >= 0); /* File descriptor must valid */
285 
286  DEBUG4("unbound event %p - Registered for write+error events on FD %i", ev, ev->fd);
287 
288  if (fr_event_fd_insert(ev, ev->ev_b->el, ev->fd,
289  NULL,
290  _unbound_io_service_writable,
291  _unbound_io_service_errored,
292  ev) < 0) {
293  PERROR("unbound event %p - Registration failed for write+error events on FD %i",
294  ev, ev->fd);
295 
296  return -1;
297  }
298  }
299 
300  /*
301  * Add a timeout event
302  */
303  if (ev->events & UB_EV_TIMEOUT) {
304  fr_time_delta_t timeout = fr_time_delta_from_timeval(tv);
305 
306  DEBUG4("unbound event %p - Timeout in %pV seconds", ev, fr_box_time_delta(timeout));
307 
308  if (fr_event_timer_in(ev, ev->ev_b->el, &ev->timer,
309  timeout, _unbound_io_service_timer_expired, ev) < 0) {
310  PERROR("unbound event %p - Failed adding timeout", ev);
311 
312  if (ev->events & (UB_EV_READ | UB_EV_WRITE)) {
313  fr_event_fd_delete(ev->ev_b->el, ev->fd, FR_EVENT_FILTER_IO);
314  }
315 
316  return -1;
317  }
318  }
319 
320  ev->active = true; /* Event is now active! */
321 
322  return 0;
323 }

325 /* Deactivate FD events and disarm the timeout
326  *
327  */
328 static int _unbound_io_event_deactivate(struct ub_event *ub_ev)
329 {
330  unbound_io_event_t *ev = talloc_get_type_abort(ub_ev, unbound_io_event_t);
331  int ret = 0;
332 
333  if (!ev->active) return 0; /* Allow this to be called multiple times */
334 
335  if (ev->events & (UB_EV_READ | UB_EV_WRITE)) {
336  DEBUG4("unbound event %p - De-registering FD %i", ev, ev->fd);
337 
338  if (fr_event_fd_delete(ev->ev_b->el, ev->fd, FR_EVENT_FILTER_IO) < 0) {
339  PERROR("unbound event %p - De-registration failed for FD %i", ev, ev->fd);
340 
341  ret = -1;
342  }
343  }
344 
345  if (ev->events & UB_EV_TIMEOUT) {
346  DEBUG4("unbound event %p - Disarming timeout", ev);
347 
348  if (ev->timer && (fr_event_timer_delete(&ev->timer) < 0)) {
349  PERROR("ubound event %p - Failed disarming timeout", ev);
350 
351  ret = -1;
352  }
353  }
354 
355  ev->active = false; /* Event is now inactive and can be modified */
356 
357  return ret;
358 }

360 /** Modify an existing timeout
361  *
362  */
363 static int _unbound_io_timer_modify(struct ub_event *ub_ev, UNUSED struct ub_event_base *ev_b,
364  void (*cb)(int, short, void*),
365  void *uctx, struct timeval *tv)
366 {
367  unbound_io_event_t *ev = talloc_get_type_abort(ub_ev, unbound_io_event_t);
368  int ret = 0;
369  fr_time_delta_t timeout;
370 
371  fr_assert(ev->events & UB_EV_TIMEOUT);
372 
373  if (ev->cb != cb) {
374  DEBUG4("unbound event %p - New callback %p (old callback was %p)",
375  ev, cb, ev->cb);
376  ev->cb = cb;
377  }
378  if (ev->uctx != uctx) {
379  DEBUG4("unbound event %p - New uctx %p (old uctx was %p)",
380  ev, uctx, ev->uctx);
381  ev->uctx = uctx;
382  }
383  if (ev->timer && (fr_event_timer_delete(&ev->timer) < 0)) {
384  PERROR("ubound event %p - Failed disarming timeout", ev);
385 
386  ret = -1; /* Continue ? */
387  }
388 
389  timeout = fr_time_delta_from_timeval(tv);
390 
391  DEBUG4("unbound event %p - Timeout in %pV seconds", ev, fr_box_time_delta(timeout));
392 
393  if (fr_event_timer_in(ev, ev->ev_b->el, &ev->timer,
394  timeout, _unbound_io_service_timer_expired, ev) < 0) {
395  PERROR("unbound event %p - Failed adding timeout", ev);
396 
397  ret = -1;
398  }
399 
400  return ret;
401 }

423 /** Returns a new libunbound event handle
424  *
425  * This handle is used by libunbound to interface with the worker's event loop

427 static struct ub_event *_unbound_io_event_new(struct ub_event_base* base, int fd, short flags,
428  void (*cb)(int, short, void*), void *uctx)
429 {
430  unbound_io_event_base_t *ev_b = talloc_get_type_abort(base, unbound_io_event_base_t);
431  unbound_io_event_t *ev;
432 
433  static struct ub_event_vmt vmt = {
434  .add_bits = _unbound_io_event_flags_add,
435  .del_bits = _unbound_io_event_flags_del,
436  .set_fd = _unbound_io_event_fd_set,
437  .free = _unbound_io_event_free,
438  .add = _unbound_io_event_activate,
439  .del = _unbound_io_event_deactivate,
440  .add_timer = _unbound_io_timer_modify,
441  .del_timer = _unbound_io_timer_deactivate
442  };
443 
444  MEM(ev = talloc_zero(ev_b, unbound_io_event_t));
445  ev->base.magic = UB_EVENT_MAGIC; /* Magic value libunbound requires */
446  ev->base.vmt = &vmt; /* Callbacks for adding/removing timers/fd events */
447  ev->ev_b = ev_b; /* Our event base (containing the el ) */
448  ev->events = flags; /* When this event should fire */
449  ev->fd = fd; /* File descriptor to register events for */
450  ev->cb = cb; /* Callback to execute on event */
451  ev->uctx = uctx; /* Lib unbound's arg to pass to the cb */
452  ev->active = false; /* Event is not currently active */
453 
454  DEBUG4("unbound event %p - Allocated - Events %i, FD %i, callback %p, uctx %p", ev, flags, fd, cb, uctx);
455 
456  return (struct ub_event *)ev;
457 }

466 /** Alloc a new event base, and unbound ctx initialised from that event base
467  *
468  * The ub_ctx is configured to use the el specified.
469  *
470  * When the thread ctx is freed, unbound_io_free should be called to gracefully
471  * free the ub_ctx, and then the event base structure it depends on.
472  *
473  * @param[in] ctx Talloc ctx to allocate even base in.
474  * @param[out] ev_b_out Event base. Free with talloc_free.
475  * @param[in] el To use to run the unbound event loop.
476  * @return
477  * - 0 on success.
478  * - -1 on failure.
479  */
480 int unbound_io_init(TALLOC_CTX *ctx, unbound_io_event_base_t **ev_b_out, fr_event_list_t *el)
481 {
482  unbound_io_event_base_t *ev_b;
483 
484  static struct ub_event_base_vmt vmt = {
485  .new_event = _unbound_io_event_new
486  };
487 
488  /*
489  * Should be manually freed *AFTER* t->ub
490  * is freed. So must be parented from the NULL
491  * ctx.
492  */
493  MEM(ev_b = talloc_zero(ctx, unbound_io_event_base_t));
494  ev_b->base.magic = UB_EVENT_MAGIC;
495  ev_b->base.vmt = &vmt;
496  ev_b->el = el;
497 
498  /*
499  * Create the main ub_ctx using our event base
500  * which specifies how libunbound integrates
501  * with our event loop.
502  */
503  ev_b->ub = ub_ctx_create_ub_event((struct ub_event_base *)ev_b);
504  if (!ev_b->ub) {
505  ERROR("Failed creating ub_ctx");
506  TALLOC_FREE(ev_b);
507  return -1;
508  }
509  talloc_set_destructor(ev_b, _event_base_free);
510 
511  *ev_b_out = ev_b;
512 
513  return 0;
514 }
