56 /** Open a raw socket to read/write packets from/to
57  *
58  * @param[out] link_layer A sockaddr_ll struct to populate. Must be passed to other raw
59  * functions.
60  * @param[in] ifindex of the interface we're binding to.
61  * @return
62  * - >= 0 a file descriptor to read/write packets on.
63  * - <0 an error ocurred.
64  */
65 int fr_dhcpv4_raw_socket_open(struct sockaddr_ll *link_layer, int ifindex)
66 {
67  int fd;
68 
69  /*
70  * PF_PACKET - packet interface on device level.
71  * using a raw socket allows packet data to be unchanged by the device driver.
72  */
73  fd = socket(PF_PACKET, SOCK_RAW, htons(ETH_P_ALL));
74  if (fd < 0) {
75  fr_strerror_printf("Cannot open socket: %s", fr_syserror(errno));
76  return fd;
77  }
78 
79  /* Set link layer parameters */
80  memset(link_layer, 0, sizeof(struct sockaddr_ll));
81 
82  link_layer->sll_family = AF_PACKET;
83  link_layer->sll_protocol = htons(ETH_P_ALL);
84  link_layer->sll_ifindex = ifindex;
85  link_layer->sll_hatype = ARPHRD_ETHER;
86  link_layer->sll_pkttype = PACKET_OTHERHOST;
87  link_layer->sll_halen = 6;
88 
89  if (bind(fd, (struct sockaddr *)link_layer, sizeof(struct sockaddr_ll)) < 0) {
90  close(fd);
91  fr_strerror_printf("Cannot bind raw socket: %s", fr_syserror(errno));
92  return -1;
93  }
94 
95  return fd;
96 }

98 /** Create the requisite L2/L3 headers, and write a DHCPv4 packet to a raw socket
99  *
100  * @param[in] sockfd to write to.
101  * @param[in] link_layer information, as returned by fr_dhcpv4_raw_socket_open.
102  * @param[in] packet to write.
103  * @return
104  * - 0 on success.
105  * - -1 on failure.
106  */
107 int fr_dhcpv4_raw_packet_send(int sockfd, struct sockaddr_ll *link_layer, fr_radius_packet_t *packet)
108 {
109  uint8_t dhcp_packet[1518] = { 0 };
110  ethernet_header_t *eth_hdr = (ethernet_header_t *)dhcp_packet;
111  ip_header_t *ip_hdr = (ip_header_t *)(dhcp_packet + ETH_HDR_SIZE);
112  udp_header_t *udp_hdr = (udp_header_t *) (dhcp_packet + ETH_HDR_SIZE + IP_HDR_SIZE);
113  dhcp_packet_t *dhcp = (dhcp_packet_t *)(dhcp_packet + ETH_HDR_SIZE + IP_HDR_SIZE + UDP_HDR_SIZE);
114 
115  uint16_t l4_len = (UDP_HDR_SIZE + packet->data_len);
116  fr_pair_t *vp;
117 
118  /* set ethernet source address to our MAC address (DHCP-Client-Hardware-Address). */
119  uint8_t dhmac[ETH_ADDR_LEN] = { 0 };
120  if ((vp = fr_pair_find_by_da(&packet->vps, attr_dhcp_client_hardware_address))) {
121  if (vp->vp_type == FR_TYPE_ETHERNET) memcpy(dhmac, vp->vp_ether, sizeof(vp->vp_ether));
122  }
123 
124  /* fill in Ethernet layer (L2) */
125  memcpy(eth_hdr->dst_addr, eth_bcast, ETH_ADDR_LEN);
126  memcpy(eth_hdr->src_addr, dhmac, ETH_ADDR_LEN);
127  eth_hdr->ether_type = htons(ETH_TYPE_IP);
128 
129  /* fill in IP layer (L3) */
130  ip_hdr->ip_vhl = IP_VHL(4, 5);
131  ip_hdr->ip_tos = 0;
132  ip_hdr->ip_len = htons(IP_HDR_SIZE + UDP_HDR_SIZE + packet->data_len);
133  ip_hdr->ip_id = 0;
134  ip_hdr->ip_off = 0;
135  ip_hdr->ip_ttl = 64;
136  ip_hdr->ip_p = 17;
137  ip_hdr->ip_sum = 0; /* Filled later */
138 
139  /* saddr: Packet-Src-IP-Address (default: 0.0.0.0). */
140  ip_hdr->ip_src.s_addr = packet->socket.inet.src_ipaddr.addr.v4.s_addr;
141 
142  /* daddr: packet destination IP addr (should be 255.255.255.255 for broadcast). */
143  ip_hdr->ip_dst.s_addr = packet->socket.inet.dst_ipaddr.addr.v4.s_addr;
144 
145  /* IP header checksum */
146  ip_hdr->ip_sum = fr_ip_header_checksum((uint8_t const *)ip_hdr, 5);
147 
148  udp_hdr->src = htons(packet->socket.inet.src_port);
149  udp_hdr->dst = htons(packet->socket.inet.dst_port);
150 
151  udp_hdr->len = htons(l4_len);
152  udp_hdr->checksum = 0; /* UDP checksum will be done after dhcp header */
153 
154  /* DHCP layer (L7) */
155 
156  /* just copy what FreeRADIUS has encoded for us. */
157  memcpy(dhcp, packet->data, packet->data_len);
158 
159  /* UDP checksum is done here */
160  udp_hdr->checksum = fr_udp_checksum((uint8_t const *)(dhcp_packet + ETH_HDR_SIZE + IP_HDR_SIZE),
161  ntohs(udp_hdr->len), udp_hdr->checksum,
162  packet->socket.inet.src_ipaddr.addr.v4, packet->socket.inet.dst_ipaddr.addr.v4);
163 
164  return sendto(sockfd, dhcp_packet, (ETH_HDR_SIZE + IP_HDR_SIZE + UDP_HDR_SIZE + packet->data_len),
165  0, (struct sockaddr *) link_layer, sizeof(struct sockaddr_ll));
166 }

168 /*
169  * For a client, receive a DHCP packet from a raw packet
170  * socket. Make sure it matches the ongoing request.
171  *
172  * FIXME: split this into two, recv_raw_packet, and verify(packet, original)
173  */
174 fr_radius_packet_t *fr_dhcv4_raw_packet_recv(int sockfd, struct sockaddr_ll *link_layer, fr_radius_packet_t *request)
175 {
176  fr_pair_t *vp;
177  fr_radius_packet_t *packet;
178  uint8_t const *code;
179  uint32_t magic, xid;
180  ssize_t data_len;
181 
182  uint8_t *raw_packet;
183  ethernet_header_t *eth_hdr;
184  ip_header_t *ip_hdr;
185  udp_header_t *udp_hdr;
186  dhcp_packet_t *dhcp_hdr;
187  uint16_t udp_src_port;
188  uint16_t udp_dst_port;
189  size_t dhcp_data_len;
190  socklen_t sock_len;
191 
192  packet = fr_radius_alloc(NULL, false);
193  if (!packet) {
194  fr_strerror_printf("Failed allocating packet");
195  return NULL;
196  }
197 
198  raw_packet = talloc_zero_array(packet, uint8_t, MAX_PACKET_SIZE);
199  if (!raw_packet) {
200  fr_strerror_printf("Out of memory");
201  fr_radius_packet_free(&packet);
202  return NULL;
203  }
204 
205  packet->socket.fd = sockfd;
206 
207  /* a packet was received (but maybe it is not for us) */
208  sock_len = sizeof(struct sockaddr_ll);
209  data_len = recvfrom(sockfd, raw_packet, MAX_PACKET_SIZE, 0, (struct sockaddr *)link_layer, &sock_len);
210 
211  uint8_t data_offset = ETH_HDR_SIZE + IP_HDR_SIZE + UDP_HDR_SIZE; /* DHCP data datas after Ethernet, IP, UDP */
212 
213  if (data_len <= data_offset) DISCARD_RP("Payload (%d) smaller than required for layers 2+3+4", (int)data_len);
214 
215  /* map raw packet to packet header of the different layers (Ethernet, IP, UDP) */
216  eth_hdr = (ethernet_header_t *)raw_packet;
217 
218  /*
219  * Check Ethernet layer data (L2)
220  */
221  if (ntohs(eth_hdr->ether_type) != ETH_TYPE_IP) DISCARD_RP("Ethernet type (%d) != IP",
222  ntohs(eth_hdr->ether_type));
223 
224  /*
225  * If Ethernet destination is not broadcast (ff:ff:ff:ff:ff:ff)
226  * Check if it matches the source HW address used (DHCP-Client-Hardware-Address = 267)
227  */
228  if ((memcmp(&eth_bcast, &eth_hdr->dst_addr, ETH_ADDR_LEN) != 0) &&
229  (vp = fr_pair_find_by_da(&request->vps, attr_dhcp_client_hardware_address)) &&
230  ((vp->vp_type == FR_TYPE_ETHERNET) && (memcmp(vp->vp_ether, &eth_hdr->dst_addr, ETH_ADDR_LEN) != 0))) {
231 
232  /* No match. */
233  DISCARD_RP("Ethernet destination (%pV) is not broadcast and doesn't match request source (%pV)",
234  fr_box_ether(eth_hdr->dst_addr), &vp->data);
235  }
236 
237  /*
238  * Ethernet is OK. Now look at IP.
239  */
240  ip_hdr = (ip_header_t *)(raw_packet + ETH_HDR_SIZE);
241 
242  /*
243  * Check IPv4 layer data (L3)
244  */
245  if (ip_hdr->ip_p != IPPROTO_UDP) DISCARD_RP("IP protocol (%d) != UDP", ip_hdr->ip_p);
246 
247  /*
248  * note: checking the destination IP address is not
249  * useful (it would be the offered IP address - which we
250  * don't know beforehand, or the broadcast address).
251  */
252 
253  /*
254  * Now check UDP.
255  */
256  udp_hdr = (udp_header_t *)(raw_packet + ETH_HDR_SIZE + IP_HDR_SIZE);
257 
258  /*
259  * Check UDP layer data (L4)
260  */
261  udp_src_port = ntohs(udp_hdr->src);
262  udp_dst_port = ntohs(udp_hdr->dst);
263 
264  /*
265  * Check DHCP layer data
266  */
267  dhcp_data_len = data_len - data_offset;
268 
269  if (dhcp_data_len < MIN_PACKET_SIZE) DISCARD_RP("DHCP packet is too small (%zu < %i)",
270  dhcp_data_len, MIN_PACKET_SIZE);
271  if (dhcp_data_len > MAX_PACKET_SIZE) DISCARD_RP("DHCP packet is too large (%zu > %i)",
272  dhcp_data_len, MAX_PACKET_SIZE);
273 
274  dhcp_hdr = (dhcp_packet_t *)(raw_packet + ETH_HDR_SIZE + IP_HDR_SIZE + UDP_HDR_SIZE);
275 
276  if (dhcp_hdr->htype != 1) DISCARD_RP("DHCP hardware type (%d) != Ethernet (1)", dhcp_hdr->htype);
277  if (dhcp_hdr->hlen != 6) DISCARD_RP("DHCP hardware address length (%d) != 6", dhcp_hdr->hlen);
278 
279  magic = ntohl(dhcp_hdr->option_format);
280 
281  if (magic != DHCP_OPTION_MAGIC_NUMBER) DISCARD_RP("DHCP magic cookie (0x%04x) != DHCP (0x%04x)",
282  magic, DHCP_OPTION_MAGIC_NUMBER);
283 
284  /*
285  * Reply transaction id must match value from request.
286  */
287  xid = ntohl(dhcp_hdr->xid);
288  if (xid != (uint32_t)request->id) DISCARD_RP("DHCP transaction ID (0x%04x) != xid from request (0x%04x)",
289  xid, request->id)
290 
291  /* all checks ok! this is a DHCP reply we're interested in. */
292  packet->data_len = dhcp_data_len;
293  packet->data = talloc_memdup(packet, raw_packet + data_offset, dhcp_data_len);
294  TALLOC_FREE(raw_packet);
295  packet->id = xid;
296 
297  code = fr_dhcpv4_packet_get_option((dhcp_packet_t const *) packet->data,
298  packet->data_len, attr_dhcp_message_type);
299  if (!code) {
300  fr_strerror_printf("No message-type option was found in the packet");
301  fr_radius_packet_free(&packet);
302  return NULL;
303  }
304 
305  if ((code[1] < 1) || (code[2] == 0) || (code[2] > 8)) {
306  fr_strerror_printf("Unknown value for message-type option");
307  fr_radius_packet_free(&packet);
308  return NULL;
309  }
310 
311  packet->code = code[2];
312 
313  /*
314  * Create a unique vector from the MAC address and the
315  * DHCP opcode. This is a hack for the RADIUS
316  * infrastructure in the rest of the server.
317  *
318  * Note: packet->data[2] == 6, which is smaller than
319  * sizeof(packet->vector)
320  *
321  * FIXME: Look for client-identifier in packet,
322  * and use that, too?
323  */
324  memset(packet->vector, 0, sizeof(packet->vector));
325  memcpy(packet->vector, packet->data + 28, packet->data[2]);
326  packet->vector[packet->data[2]] = packet->code & 0xff;
327 
328  packet->socket.inet.src_port = udp_src_port;
329  packet->socket.inet.dst_port = udp_dst_port;
330 
331  packet->socket.inet.src_ipaddr.af = AF_INET;
332  packet->socket.inet.src_ipaddr.addr.v4.s_addr = ip_hdr->ip_src.s_addr;
333  packet->socket.inet.dst_ipaddr.af = AF_INET;
334  packet->socket.inet.dst_ipaddr.addr.v4.s_addr = ip_hdr->ip_dst.s_addr;
335 
336  return packet;
337 }
