48 /** Validation function for ipaddr conf_file types
49  *
50  */
51 static inline int CC_HINT(nonnull) fr_item_validate_ipaddr(CONF_SECTION *cs, char const *name,
52  fr_type_t type, char const *value,
53  fr_ipaddr_t *ipaddr)
54 {
55  char ipbuf[128];
56 
57  if (strcmp(value, "*") == 0) {
58  cf_log_debug(cs, "%.*s%s = *", PAIR_SPACE(cs), parse_spaces, name);
59  } else if (strspn(value, ".0123456789abdefABCDEF:%[]/") == strlen(value)) {
60  cf_log_debug(cs, "%.*s%s = %s", PAIR_SPACE(cs), parse_spaces, name, value);
61  } else {
62  cf_log_debug(cs, "%.*s%s = %s IPv%s address [%s]", PAIR_SPACE(cs), parse_spaces, name, value,
63  (ipaddr->af == AF_INET ? "4" : " 6"), fr_inet_ntoh(ipaddr, ipbuf, sizeof(ipbuf)));
64  }
65 
66  switch (type) {
67  case FR_TYPE_IPV4_ADDR:
68  case FR_TYPE_IPV6_ADDR:
69  case FR_TYPE_COMBO_IP_ADDR:
70  switch (ipaddr->af) {
71  case AF_INET:
72  if (ipaddr->prefix != 32) {
73  ERROR("Invalid IPv4 mask length \"/%i\". Only \"/32\" permitted for non-prefix types",
74  ipaddr->prefix);
75 
76  return -1;
77  }
78  break;
79 
80  case AF_INET6:
81  if (ipaddr->prefix != 128) {
82  ERROR("Invalid IPv6 mask length \"/%i\". Only \"/128\" permitted for non-prefix types",
83  ipaddr->prefix);
84 
85  return -1;
86  }
87  break;
88 
89  default:
90  return -1;
91  }
92  return 0;
93 
94  case FR_TYPE_IPV4_PREFIX:
95  case FR_TYPE_IPV6_PREFIX:
96  case FR_TYPE_COMBO_IP_PREFIX:
97  return 0;
98 
99  default:
100  fr_assert(0);
101  return -1;
102  }
103 }

105 static void cf_pair_debug(CONF_SECTION const *cs, CONF_PAIR const *cp, bool secret)
106 {
107  if (secret && (fr_debug_lvl < L_DBG_LVL_3)) {
108  cf_log_debug(cs, "%.*s%s = <<< secret >>>", PAIR_SPACE(cs), parse_spaces, cp->attr);
109 
110  } else if (cp->rhs_quote == T_BARE_WORD) {
111  cf_log_debug(cs, "%.*s%s = %s", PAIR_SPACE(cs), parse_spaces, cp->attr, cp->value);
112 
113  } else {
114  /*
115  * Print the strings with the correct quotation character and escaping.
116  */
117  char *tmp = fr_asprint(NULL, cp->value, talloc_array_length(cp->value) - 1, cp->rhs_quote);
118  char quote;
119 
120  switch (cp->rhs_quote) {
121  default:
122  case T_DOUBLE_QUOTED_STRING:
123  quote = '"';
124  break;
125 
126  case T_SINGLE_QUOTED_STRING:
127  quote = '\'';
128  break;
129 
130  case T_BACK_QUOTED_STRING:
131  quote = '`';
132  break;
133  }
134 
135  cf_log_debug(cs, "%.*s%s = %c%s%c", PAIR_SPACE(cs), parse_spaces, cp->attr, quote, tmp, quote);
136  talloc_free(tmp);
137  }

140 /** Parses a #CONF_PAIR into a C data type
141  *
142  * @copybrief cf_pair_value
143  * @see cf_pair_value
144  *
145  * @param[in] ctx to allocate any dynamic buffers in.
146  * @param[out] out Where to write the parsed value.
147  * @param[in] base address of the structure out points into.
148  * May be NULL in the case of manual parsing.
149  * @param[in] ci to parse.
150  * @param[in] rule to parse to. May contain flags.
151  * @return
152  * - 0 on success.
153  * - -1 on failure.
154  */
155 int cf_pair_parse_value(TALLOC_CTX *ctx, void *out, UNUSED void *base, CONF_ITEM *ci, CONF_PARSER const *rule)
156 {
157  int rcode = 0;
158  bool attribute, required, secret, file_input, cant_be_empty, tmpl, file_exists, nonblock;
159 
160  ssize_t slen;
161 
162  int type = rule->type;
163  CONF_PAIR *cp = cf_item_to_pair(ci);
164  CONF_SECTION *cs = cf_item_to_section(cf_parent(ci));
165 
166  attribute = (type & FR_TYPE_ATTRIBUTE);
167  required = (type & FR_TYPE_REQUIRED);
168  secret = (type & FR_TYPE_SECRET);
169  file_input = (type == FR_TYPE_FILE_INPUT); /* check, not and */
170  file_exists = (type == FR_TYPE_FILE_EXISTS); /* check, not and */
171  cant_be_empty = (type & FR_TYPE_NOT_EMPTY);
172  tmpl = (type & FR_TYPE_TMPL);
173  nonblock = (type & FR_TYPE_NON_BLOCKING);
174 
175  fr_assert(cp);
176  fr_assert(!(type & FR_TYPE_ATTRIBUTE) || tmpl); /* Attribute flag only valid for templates */
177  fr_assert((type & FR_TYPE_ON_READ) == 0);
178 
179  if (required) cant_be_empty = true; /* May want to review this in the future... */
180 
181  type = FR_BASE_TYPE(type); /* normal types are small */
182 
183  /*
184  * Everything except templates must have a base type.
185  */
186  if (!type && !tmpl) {
187  cf_log_err(cp, "Configuration pair \"%s\" must have a data type", cp->attr);
188  return -1;
189  }
190 
191  /*
192  * Catch crazy errors.
193  */
194  if (!cp->value) {
195  cf_log_err(cp, "Configuration pair \"%s\" must have a value", cp->attr);
196  return -1;
197  }
198 
199  /*
200  * Check for zero length strings
201  */
202  if ((cp->value[0] == '\0') && cant_be_empty) {
203  cf_log_err(cp, "Configuration pair \"%s\" must not be empty (zero length)", cp->attr);
204  if (!required) cf_log_err(cp, "Comment item to silence this message");
205  rcode = -1;
206 
207  error:
208  return rcode;
209  }
210 
211  if (tmpl) {
212  tmpl_t *vpt;
213  static tmpl_rules_t rules = {
214  .allow_unknown = true,
215  .allow_unresolved = true,
216  .allow_foreign = true
217  };
218  fr_type_t cast = FR_TYPE_INVALID;
219  fr_sbuff_t sbuff = FR_SBUFF_IN(cp->value, strlen(cp->value));
220 
221  if (!cp->printed) cf_pair_debug(cs, cp, secret);
222 
223  /*
224  * Parse the cast operator for barewords
225  */
226  if (cp->rhs_quote == T_BARE_WORD) {
227  slen = tmpl_cast_from_substr(&cast, &sbuff);
228  if (slen < 0) {
229  char *spaces, *text;
230  tmpl_error:
231  fr_canonicalize_error(ctx, &spaces, &text, slen, cp->value);
232 
233  cf_log_err(cp, "Failed parsing attribute reference:");
234  cf_log_err(cp, "%s", text);
235  cf_log_perr(cp, "%s^", spaces);
236 
237  talloc_free(spaces);
238  talloc_free(text);
239  goto error;
240  }
241  fr_sbuff_adv_past_whitespace(&sbuff, SIZE_MAX);
242  } else if (attribute) {
243  cf_log_err(cp, "Invalid quoting. Unquoted attribute reference is required");
244  goto error;
245  }
246 
247  slen = tmpl_afrom_substr(cp, &vpt, &sbuff, cp->rhs_quote,
248  tmpl_parse_rules_unquoted[cp->rhs_quote],
249  &rules);
250  if (slen < 0) goto tmpl_error;
251 
252  if (attribute && (!tmpl_is_attr(vpt) && !tmpl_is_attr_unresolved(vpt))) {
253  cf_log_err(cp, "Expected attr got %s",
254  fr_table_str_by_value(tmpl_type_table, vpt->type, "???"));
255  return -1;
256  }
257 
258  tmpl_cast_set(vpt, cast);
259 
260  /*
261  * Non-blocking xlat's
262  */
263  if (nonblock) {
264  if (vpt->type == TMPL_TYPE_EXEC) {
265  cf_log_err(cp, "The '%s' configuration item MUST NOT be used with a blocking operation.", cp->attr);
266  return -1;
267  }
268 
269  if ((vpt->type == TMPL_TYPE_XLAT) &&
270  xlat_async_required(tmpl_xlat(vpt))) {
271  cf_log_err(cp, "The '%s' configuration item MUST NOT be used with a blocking expansion.", cp->attr);
272  return -1;
273  }
274  }
275 
276  *(tmpl_t **)out = vpt;
277 
278  goto finish;
279  }
280 
281  switch (type) {
282  case FR_TYPE_BOOL:
283  /*
284  * Allow yes/no, true/false, and on/off
285  */
286  if ((strcasecmp(cp->value, "yes") == 0) ||
287  (strcasecmp(cp->value, "true") == 0) ||
288  (strcasecmp(cp->value, "on") == 0)) {
289  *(bool *)out = true;
290  } else if ((strcasecmp(cp->value, "no") == 0) ||
291  (strcasecmp(cp->value, "false") == 0) ||
292  (strcasecmp(cp->value, "off") == 0)) {
293  *(bool *)out = false;
294  } else {
295  cf_log_err(cs, "Invalid value \"%s\" for boolean variable %s",
296  cp->value, cp->attr);
297  rcode = -1;
298  goto error;
299  }
300  if (!cp->printed) cf_log_debug(cs, "%.*s%s = %s", PAIR_SPACE(cs), parse_spaces, cp->attr, cp->value);
301  break;
302 
303  case FR_TYPE_UINT32:
304  {
305  unsigned long v = strtoul(cp->value, 0, 0);
306 
307  /*
308  * Restrict integer values to 0-INT32_MAX, this means
309  * it will always be safe to cast them to a signed type
310  * for comparisons, and imposes the same range limit as
311  * before we switched to using an unsigned type to
312  * represent config item integers.
313  */
314  if (v > INT32_MAX) {
315  cf_log_err(cs, "Invalid value \"%s\" for variable %s, must be between 0-%u", cp->value,
316  cp->attr, INT32_MAX);
317  rcode = -1;
318  goto error;
319  }
320 
321  *(uint32_t *)out = v;
322  if (!cp->printed) cf_log_debug(cs, "%.*s%s = %u", PAIR_SPACE(cs), parse_spaces, cp->attr, *(uint32_t *)out);
323  }
324  break;
325 
326  case FR_TYPE_UINT8:
327  {
328  unsigned long v = strtoul(cp->value, 0, 0);
329 
330  if (v > UINT8_MAX) {
331  cf_log_err(cs, "Invalid value \"%s\" for variable %s, must be between 0-%u", cp->value,
332  cp->attr, UINT8_MAX);
333  rcode = -1;
334  goto error;
335  }
336  *(uint8_t *)out = (uint8_t) v;
337  if (!cp->printed) cf_log_debug(cs, "%.*s%s = %u", PAIR_SPACE(cs), parse_spaces, cp->attr, *(uint8_t *)out);
338  }
339  break;
340 
341  case FR_TYPE_UINT16:
342  {
343  unsigned long v = strtoul(cp->value, 0, 0);
344 
345  if (v > UINT16_MAX) {
346  cf_log_err(cs, "Invalid value \"%s\" for variable %s, must be between 0-%u", cp->value,
347  cp->attr, UINT16_MAX);
348  rcode = -1;
349  goto error;
350  }
351  *(uint16_t *)out = (uint16_t) v;
352  if (!cp->printed) cf_log_debug(cs, "%.*s%s = %u", PAIR_SPACE(cs), parse_spaces, cp->attr, *(uint16_t *)out);
353  }
354  break;
355 
356  case FR_TYPE_UINT64:
357  *(uint64_t *)out = strtoull(cp->value, NULL, 0);
358  if (!cp->printed) cf_log_debug(cs, "%.*s%s = %" PRIu64, PAIR_SPACE(cs), parse_spaces, cp->attr, *(uint64_t *)out);
359  break;
360 
361  case FR_TYPE_SIZE:
362  {
363  if (fr_size_from_str((size_t *)out, cp->value) < 0) {
364  cf_log_perr(cs, "Invalid value \"%s\" for variable %s", cp->value, cp->attr);
365  rcode = -1;
366  goto error;
367  }
368  if (!cp->printed) cf_log_debug(cs, "%.*s%s = %zu", PAIR_SPACE(cs), parse_spaces, cp->attr, *(size_t *)out);
369  break;
370  }
371 
372  case FR_TYPE_INT32:
373  *(int32_t *)out = strtol(cp->value, NULL, 10);
374  if (!cp->printed) cf_log_debug(cs, "%.*s%s = %d", PAIR_SPACE(cs), parse_spaces, cp->attr, *(int32_t *)out);
375  break;
376 
377  case FR_TYPE_STRING:
378  {
379  char **str = out;
380 
381  if (!cp->printed) cf_pair_debug(cs, cp, secret);
382 
383  /*
384  * If there's out AND it's an input file, check
385  * that we can read it. This check allows errors
386  * to be caught as early as possible, during
387  * server startup.
388  */
389  if (file_input && !cf_file_check(cs, cp->value, true)) {
390  rcode = -1;
391  goto error;
392  }
393 
394  if (file_exists && !cf_file_check(cs, cp->value, false)) {
395  rcode = -1;
396  goto error;
397  }
398 
399  /*
400  * Free any existing buffers
401  */
402  talloc_free(*str);
403  *str = talloc_typed_strdup(cs, cp->value);
404  }
405  break;
406 
407  case FR_TYPE_IPV4_ADDR:
408  case FR_TYPE_IPV4_PREFIX:
409  {
410  fr_ipaddr_t *ipaddr = out;
411 
412  if (fr_inet_pton4(ipaddr, cp->value, -1, true, false, true) < 0) {
413  cf_log_perr(cp, "Failed parsing config item");
414  rcode = -1;
415  goto error;
416  }
417  /* Also prints the IP to the log */
418  if (fr_item_validate_ipaddr(cs, cp->attr, type, cp->value, ipaddr) < 0) {
419  rcode = -1;
420  goto error;
421  }
422  }
423  break;
424 
425  case FR_TYPE_IPV6_ADDR:
426  case FR_TYPE_IPV6_PREFIX:
427  {
428  fr_ipaddr_t *ipaddr = out;
429 
430  if (fr_inet_pton6(ipaddr, cp->value, -1, true, false, true) < 0) {
431  cf_log_perr(cp, "Failed parsing config item");
432  rcode = -1;
433  goto error;
434  }
435  /* Also prints the IP to the log */
436  if (fr_item_validate_ipaddr(cs, cp->attr, type, cp->value, ipaddr) < 0) {
437  rcode = -1;
438  goto error;
439  }
440  }
441  break;
442 
443  case FR_TYPE_COMBO_IP_ADDR:
444  case FR_TYPE_COMBO_IP_PREFIX:
445  {
446  fr_ipaddr_t *ipaddr = out;
447  int af = AF_UNSPEC;
448  fr_type_t our_type = FR_TYPE_INVALID;
449  fr_sbuff_t sbuff = FR_SBUFF_IN(cp->value, strlen(cp->value));
450 
451  slen = tmpl_cast_from_substr(&our_type, &sbuff);
452  if (slen < 0) {
453  cf_log_perr(cp, "Failed parsing config item");
454  rcode = -1;
455  goto error;
456  }
457 
458  if (slen > 0) {
459  if (type == FR_TYPE_COMBO_IP_ADDR) {
460  switch (our_type) {
461  case FR_TYPE_IPV4_ADDR:
462  af = AF_INET;
463  break;
464 
465  case FR_TYPE_IPV6_ADDR:
466  af = AF_INET6;
467  break;
468 
469  default:
470  cf_log_perr(cp, "Invalid cast, expecting 'ipv4addr' or 'ipv6addr'");
471  rcode = -1;
472  goto error;
473  }
474  } else if (type == FR_TYPE_COMBO_IP_PREFIX) {
475  switch (our_type) {
476  case FR_TYPE_IPV4_PREFIX:
477  af = AF_INET;
478  break;
479 
480  case FR_TYPE_IPV6_PREFIX:
481  af = AF_INET6;
482  break;
483 
484  default:
485  cf_log_perr(cp, "Invalid cast, expecting 'ipv4prefix' or 'ipv6prefix'");
486  rcode = -1;
487  goto error;
488  }
489  }
490  }
491 
492  if (fr_inet_pton(ipaddr, cp->value, -1, af, true, true) < 0) {
493  cf_log_perr(cp, "Failed parsing config item");
494  rcode = -1;
495  goto error;
496  }
497  /* Also prints the IP to the log */
498  if (fr_item_validate_ipaddr(cs, cp->attr, type, cp->value, ipaddr) < 0) {
499  rcode = -1;
500  goto error;
501  }
502  }
503  break;
504 
505  case FR_TYPE_TIME_DELTA:
506  {
507  fr_time_delta_t delta;
508 
509  if (fr_time_delta_from_str(&delta, cp->value, FR_TIME_RES_SEC) < 0) {
510  cf_log_perr(cp, "Failed parsing config item");
511  rcode = -1;
512  goto error;
513  }
514 
515  if (!cp->printed) {
516  char *p;
517  fr_value_box_aprint(NULL, &p, fr_box_time_delta(delta), NULL);
518  cf_log_debug(cs, "%.*s%s = %s", PAIR_SPACE(cs), parse_spaces, cp->attr, p);
519  talloc_free(p);
520  }
521 
522  memcpy(out, &delta, sizeof(delta));
523  }
524  break;
525 
526  case FR_TYPE_FLOAT32:
527  {
528  float num;
529 
530  if (sscanf(cp->value, "%f", &num) != 1) {
531  cf_log_err(cp, "Failed parsing floating point number");
532  rcode = -1;
533  goto error;
534  }
535  if (!cp->printed) cf_log_debug(cs, "%.*s%s = %f", PAIR_SPACE(cs), parse_spaces, cp->attr,
536  (double) num);
537  memcpy(out, &num, sizeof(num));
538  }
539  break;
540 
541  case FR_TYPE_FLOAT64:
542  {
543  double num;
544 
545  if (sscanf(cp->value, "%lf", &num) != 1) {
546  cf_log_err(cp, "Failed parsing floating point number");
547  rcode = -1;
548  goto error;
549  }
550  if (!cp->printed) cf_log_debug(cs, "%.*s%s = %f", PAIR_SPACE(cs), parse_spaces, cp->attr, num);
551  memcpy(out, &num, sizeof(num));
552  }
553  break;
554 
555  default:
556  /*
557  * If we get here, it's a sanity check error.
558  * It's not an error parsing the configuration
559  * file.
560  */
561  fr_assert(type > FR_TYPE_INVALID);
562  fr_assert(type < FR_TYPE_MAX);
563 
564  cf_log_err(cp, "type '%s' (%i) is not supported in the configuration files",
565  fr_table_str_by_value(fr_value_box_type_table, type, "?Unknown?"), type);
566  rcode = -1;
567  goto error;
568  }
569 
570 finish:
571  cp->parsed = true;
572  cp->printed = true;
573 
574  return rcode;
575 }

577 /** Allocate a pair using the dflt value and quotation
578  *
579  * The pair created by this function should fed to #cf_pair_parse for parsing.
580  *
581  * @param[out] out Where to write the CONF_PAIR we created with the default value.
582  * @param[in] cs to parent the CONF_PAIR from.
583  * @param[in] name of the CONF_PAIR to create.
584  * @param[in] type of conf item being parsed (determines default quoting).
585  * @param[in] dflt value to assign the CONF_PAIR.
586  * @param[in] dflt_quote surrounding the CONF_PAIR.
587  * @return
588  * - 0 on success.
589  * - -1 on failure.
590  */
591 static int cf_pair_default(CONF_PAIR **out, CONF_SECTION *cs, char const *name,
592  int type, char const *dflt, fr_token_t dflt_quote)
593 {
594  int lineno = 0;
595  char const *expanded;
596  CONF_PAIR *cp;
597  char buffer[8192];
598 
599  fr_assert(dflt);
600 
601  type = FR_BASE_TYPE(type);
602 
603  /*
604  * Defaults may need their values expanding
605  */
606  expanded = cf_expand_variables("<internal>", lineno, cs, buffer, sizeof(buffer), dflt, -1, NULL);
607  if (!expanded) {
608  cf_log_err(cs, "Failed expanding variable %s", name);
609  return -1;
610  }
611 
612  /*
613  * If no default quote was set, determine it from the type
614  */
615  if (dflt_quote == T_INVALID) {
616  switch (type) {
617  case FR_TYPE_STRING:
618  dflt_quote = T_DOUBLE_QUOTED_STRING;
619  break;
620 
621  case FR_TYPE_FILE_INPUT:
622  case FR_TYPE_FILE_OUTPUT:
623  dflt_quote = T_DOUBLE_QUOTED_STRING;
624  break;
625 
626  default:
627  dflt_quote = T_BARE_WORD;
628  break;
629  }
630  }
631 
632  cp = cf_pair_alloc(cs, name, expanded, T_OP_EQ, T_BARE_WORD, dflt_quote);
633  if (!cp) return -1;
634 
635  cp->parsed = true;
636 
637  /*
638  * Set the rcode to indicate we used a default value
639  */
640  *out = cp;
641 
642  return 1;
643 }

645 /** Parses a #CONF_PAIR into a C data type, with a default value.
646  *
647  * @param[in] ctx To allocate arrays and values in.
648  * @param[out] out Where to write the result.
649  * @param[in] base address of the structure out points into.
650  * May be NULL in the case of manual parsing.
651  * @param[in] cs to search for matching #CONF_PAIR in.
652  * @param[in] rule to parse #CONF_PAIR with.
653  * @return
654  * - 1 if default value was used, or if there was no CONF_PAIR or dflt.
655  * - 0 on success.
656  * - -1 on error.
657  * - -2 if deprecated.
658  */
659 static int CC_HINT(nonnull(4,5)) cf_pair_parse_internal(TALLOC_CTX *ctx, void *out, void *base,
660  CONF_SECTION *cs, CONF_PARSER const *rule)
661 {
662  bool multi, required, deprecated;
663  size_t count = 0;
664  CONF_PAIR *cp, *dflt_cp = NULL;
665 
666  unsigned int type = rule->type;
667  char const *dflt = rule->dflt;
668  fr_token_t dflt_quote = rule->quote;
669 
670  fr_assert(!(type & FR_TYPE_TMPL) || !dflt || (dflt_quote != T_INVALID)); /* We ALWAYS need a quoting type for templates */
671 
672  multi = (type & FR_TYPE_MULTI);
673  required = (type & FR_TYPE_REQUIRED);
674  deprecated = (type & FR_TYPE_DEPRECATED);
675 
676  /*
677  * If the item is multi-valued we allocate an array
678  * to hold the multiple values.
679  */
680  if (multi) {
681  CONF_PAIR *first;
682  void **array;
683  size_t i;
684 
685  /*
686  * Don't re-parse things which have already been parsed.
687  */
688  first = cf_pair_find(cs, rule->name);
689  if (first && first->parsed) return 0;
690 
691  /*
692  * Easier than re-allocing
693  */
694  for (cp = first;
695  cp;
696  cp = cf_pair_find_next(cs, cp, rule->name)) count++;
697 
698  /*
699  * Multivalued, but there's no value, create a
700  * default pair.
701  */
702  if (!count) {
703  if (deprecated) return 0;
704  if (!dflt) {
705  if (required) {
706  need_value:
707  cf_log_err(cs, "Configuration item \"%s\" must have a value", rule->name);
708  return -1;
709  }
710  return 1;
711  }
712 
713  if (cf_pair_default(&dflt_cp, cs, rule->name, type, dflt, dflt_quote) < 0) return -1;
714  cp = dflt_cp;
715  count = 1; /* Need one to hold the default */
716  } else {
717  cp = first; /* reset */
718  }
719 
720  if (deprecated) {
721  deprecated:
722  cf_log_err(cp, "Configuration pair \"%s\" is deprecated", cp->attr);
723  return -2;
724  }
725 
726  array = NULL;
727 
728  /*
729  * Functions don't necessarily *need* to write
730  * anywhere, so their data pointer can be NULL.
731  */
732  if (!out) {
733  if (!rule->func) {
734  no_out:
735  cf_log_err(cs, "Rule doesn't specify output destination");
736  return -1;
737  }
738  }
739  /*
740  * Tmpl is outside normal range
741  */
742  else if (type & FR_TYPE_TMPL) {
743  array = (void **)talloc_zero_array(ctx, tmpl_t *, count);
744  /*
745  * Allocate an array of values.
746  *
747  * We don't NULL terminate. Consumer must use
748  * talloc_array_length().
749  */
750  } else switch (FR_BASE_TYPE(type)) {
751  case FR_TYPE_BOOL:
752  array = (void **)talloc_zero_array(ctx, bool, count);
753  break;
754 
755  case FR_TYPE_UINT32:
756  array = (void **)talloc_zero_array(ctx, uint32_t, count);
757  break;
758 
759  case FR_TYPE_UINT16:
760  array = (void **)talloc_zero_array(ctx, uint16_t, count);
761  break;
762 
763  case FR_TYPE_UINT64:
764  array = (void **)talloc_zero_array(ctx, uint64_t, count);
765  break;
766 
767  case FR_TYPE_INT32:
768  array = (void **)talloc_zero_array(ctx, int32_t, count);
769  break;
770 
771  case FR_TYPE_STRING:
772  array = (void **)talloc_zero_array(ctx, char *, count);
773  break;
774 
775  case FR_TYPE_IPV4_ADDR:
776  case FR_TYPE_IPV4_PREFIX:
777  case FR_TYPE_IPV6_ADDR:
778  case FR_TYPE_IPV6_PREFIX:
779  case FR_TYPE_COMBO_IP_ADDR:
780  case FR_TYPE_COMBO_IP_PREFIX:
781  array = (void **)talloc_zero_array(ctx, fr_ipaddr_t, count);
782  break;
783 
784  case FR_TYPE_TIME_DELTA:
785  array = (void **)talloc_zero_array(ctx, fr_time_delta_t, count);
786  break;
787 
788  case FR_TYPE_VOID:
789  fr_assert(rule->func);
790  array = (void **)talloc_zero_array(ctx, void *, count);
791  break;
792 
793  default:
794  cf_log_err(cp, "Unsupported type %i (%i)", type, FR_BASE_TYPE(type));
795  fr_assert_fail(NULL);
796  return -1; /* Unsupported type */
797  }
798 
799  if (!array) return -1;
800 
801  for (i = 0; i < count; i++, cp = cf_pair_find_next(cs, cp, rule->name)) {
802  int ret;
803  cf_parse_t func;
804  void *entry;
805  TALLOC_CTX *value_ctx = array ? array : ctx;
806 
807  /*
808  * Figure out where to write the output
809  */
810  if (!array) {
811  entry = NULL;
812  } else if ((FR_BASE_TYPE(type) == FR_TYPE_VOID) || (type & FR_TYPE_TMPL)) {
813  entry = &array[i];
814  } else {
815  entry = ((uint8_t *) array) + (i * fr_value_box_field_sizes[FR_BASE_TYPE(type)]);
816  }
817 
818  /*
819  * Switch between custom parsing function
820  * and the standard value parsing function.
821  */
822  if (rule->func) {
823  cf_log_debug(cs, "%.*s%s = %s", PAIR_SPACE(cs), parse_spaces,
824  cp->attr, cp->value);
825  func = rule->func;
826  } else {
827  if (!entry) goto no_out;
828  func = cf_pair_parse_value;
829  }
830 
831  ret = func(value_ctx, entry, base, cf_pair_to_item(cp), rule);
832  if (ret < 0) {
833  talloc_free(array);
834  talloc_free(dflt_cp);
835  return -1;
836  }
837  cp->parsed = true;
838  }
839 
840  if (out) *(void **)out = array;
841  /*
842  * Single valued config item gets written to
843  * the data pointer directly.
844  */
845  } else {
846  CONF_PAIR *next;
847  int ret;
848  cf_parse_t func = cf_pair_parse_value;
849 
850  cp = cf_pair_find(cs, rule->name);
851  if (!cp) {
852  if (deprecated) return 0;
853  if (!dflt) {
854  if (required) goto need_value;
855  return 1;
856  }
857 
858  if (cf_pair_default(&dflt_cp, cs, rule->name, type, dflt, dflt_quote) < 0) return -1;
859  cp = dflt_cp;
860 
861  } else if (cp->parsed) {
862  /*
863  * Don't re-parse things which have already been parsed.
864  */
865  return 0;
866  }
867 
868  next = cf_pair_find_next(cs, cp, rule->name);
869  if (next) {
870  cf_log_err(&(next->item), "Invalid duplicate configuration item '%s'", rule->name);
871  return -1;
872  }
873 
874  if (deprecated) goto deprecated;
875 
876  if (rule->func) {
877  cf_log_debug(cs, "%.*s%s = %s", PAIR_SPACE(cs), parse_spaces, cp->attr, cp->value);
878  cp->printed = true;
879  func = rule->func;
880  }
881 
882  ret = func(ctx, out, base, cf_pair_to_item(cp), rule);
883  if (ret < 0) {
884  talloc_free(dflt_cp);
885  return -1;
886  }
887  cp->parsed = true;
888  }
889 
890  /*
891  * If we created a default cp and succeeded
892  * in parsing the dflt value, add the new
893  * cp to the enclosing section.
894  */
895  if (dflt_cp) {
896  cf_item_add(cs, &(dflt_cp->item));
897  return 1;
898  }
899 
900  return 0;
901 }

1060  return 0;
1061  }

1064  *(char **) rule->data = NULL;
1065  } else if (base) {
1066  *(char **) (((char *)base) + rule->offset) = NULL;
1067  } else {
1068  return 0;
1069  }
1070 
1071  return 0;
1072 }

1102 /** Parse a subsection
1103  *
1104  * @note Turns out using nested structures (instead of pointers) for subsections, was actually
1105  * a pretty bad design decision, and will need to be fixed at some future point.
1106  * For now we have a horrible hack where only multi-subsections get an array of structures
1107  * of the appropriate size.
1108  *
1109  * @param[in] ctx to allocate any additional structures under.
1110  * @param[out] out pointer to a struct/pointer to fill with data.
1111  * @param[in] base address of the structure out points into.
1112  * May be NULL in the case of manual parsing.
1113  * @param[in] cs to parse.
1114  * @param[in] rule to parse the subcs with.
1115  * @return
1116  * - 0 on success.
1117  * - -1 on general error.
1118  * - -2 if a deprecated #CONF_ITEM was found.
1119  */
1120 static int cf_subsection_parse(TALLOC_CTX *ctx, void *out, void *base, CONF_SECTION *cs, CONF_PARSER const *rule)
1121 {
1122  CONF_SECTION *subcs = NULL;
1123  int count = 0, i = 0, ret;
1124 
1125  fr_type_t type = rule->type;
1126  size_t subcs_size = rule->subcs_size;
1127  CONF_PARSER const *rules = rule->subcs;
1128 
1129  uint8_t **array = NULL;
1130 
1131  fr_assert(type & FR_TYPE_SUBSECTION);
1132 
1133  subcs = cf_section_find(cs, rule->name, rule->ident2);
1134  if (!subcs) return 0;
1135 
1136  /*
1137  * Handle the single subsection case (which is simple)
1138  */
1139  if (!(type & FR_TYPE_MULTI)) {
1140  uint8_t *buff = NULL;
1141 
1142  if (DEBUG_ENABLED4) cf_log_debug(cs, "Evaluating rules for %s section. Output %p",
1143  cf_section_name1(subcs), out);
1144 
1145  /*
1146  * Add any rules, so the func can just call cf_section_parse
1147  * if it wants to continue after doing its stuff.
1148  */
1149  if (cf_section_rules_push(subcs, rules) < 0) return -1;
1150  if (rule->func) return rule->func(ctx, out, base, cf_section_to_item(subcs), rule);
1151 
1152  /*
1153  * FIXME: We shouldn't allow nested structures like this.
1154  * Each subsection struct should be allocated separately so
1155  * we have a clean talloc hierarchy.
1156  */
1157  if (!subcs_size) return cf_section_parse(ctx, out, subcs);
1158 
1159  if (out) {
1160  MEM(buff = talloc_zero_array(ctx, uint8_t, subcs_size));
1161  if (rule->subcs_type) talloc_set_name_const(buff, rule->subcs_type);
1162  }
1163 
1164  ret = cf_section_parse(buff, buff, subcs);
1165  if (ret < 0) {
1166  talloc_free(buff);
1167  return ret;
1168  }
1169 
1170  if (out) *((uint8_t **)out) = buff;
1171 
1172  return 0;
1173  }
1174 
1175  fr_assert(subcs_size);
1176 
1177  /*
1178  * Handle the multi subsection case (which is harder)
1179  */
1180  subcs = NULL;
1181  while ((subcs = cf_section_find_next(cs, subcs, rule->name, rule->ident2))) count++;
1182 
1183  /*
1184  * Allocate an array to hold the subsections
1185  */
1186  if (out) {
1187  MEM(array = talloc_zero_array(ctx, uint8_t *, count));
1188  if (rule->subcs_type) talloc_set_name(array, "%s *", rule->subcs_type);
1189  }
1190  /*
1191  * Start parsing...
1192  *
1193  * Note, we allocate each subsection structure individually
1194  * so that they can be used as talloc contexts and we can
1195  * keep the talloc hierarchy clean.
1196  */
1197  subcs = NULL;
1198  while ((subcs = cf_section_find_next(cs, subcs, rule->name, rule->ident2))) {
1199  uint8_t *buff = NULL;
1200 
1201  if (DEBUG_ENABLED4) cf_log_debug(cs, "Evaluating rules for %s[%i] section. Output %p",
1202  cf_section_name1(subcs),
1203  i, out);
1204 
1205  if (array) {
1206  MEM(buff = talloc_zero_array(array, uint8_t, subcs_size));
1207  if (rule->subcs_type) talloc_set_name_const(buff, rule->subcs_type);
1208  array[i++] = buff;
1209  }
1210 
1211  /*
1212  * Add any rules, so the func can just call cf_section_parse
1213  * if it wants to continue after doing its stuff.
1214  */
1215  if (cf_section_rules_push(subcs, rules) < 0) {
1216  talloc_free(array);
1217  return -1;
1218  }
1219  if (rule->func) {
1220  ret = rule->func(ctx, buff, base, cf_section_to_item(subcs), rule);
1221  if (ret < 0) {
1222  talloc_free(array);
1223  return ret;
1224  }
1225  continue;
1226  }
1227 
1228  ret = cf_section_parse(buff, buff, subcs);
1229  if (ret < 0) {
1230  talloc_free(array);
1231  return ret;
1232  }
1233  }
1234 
1235  if (out) *((uint8_t ***)out) = array;
1236 
1237  return 0;
1238 }

1243  * Usually the same as base, unless base is a nested struct.
1244  * @param[out] base pointer to a struct to fill with data.
1245  * @param[in] cs to parse.
1246  * @return
1247  * - 0 on success.
1248  * - -1 on general error.
1249  * - -2 if a deprecated #CONF_ITEM was found.
1250  */
1251 int cf_section_parse(TALLOC_CTX *ctx, void *base, CONF_SECTION *cs)
1252 {
1253  int ret = 0;
1254  void *data = NULL;
1255  CONF_DATA const *rule_cd = NULL;
