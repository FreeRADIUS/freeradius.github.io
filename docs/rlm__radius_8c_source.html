160 /** Set which types of packets we can parse
161  *
162  * @param[in] ctx to allocate data in (instance of rlm_radius).
163  * @param[out] out Where to write the parsed data.
164  * @param[in] parent Base structure address.
165  * @param[in] ci #CONF_PAIR specifying the name of the type module.
166  * @param[in] rule unused.
167  * @return
168  * - 0 on success.
169  * - -1 on failure.
170  */
171 static int type_parse(UNUSED TALLOC_CTX *ctx, void *out, UNUSED void *parent,
172  CONF_ITEM *ci, UNUSED CONF_PARSER const *rule)
173 {
174  char const *type_str = cf_pair_value(cf_item_to_pair(ci));
175  CONF_SECTION *cs = cf_item_to_section(cf_parent(ci));
176  fr_dict_enum_t const *type_enum;
177  uint32_t code;
178 
179  /*
180  * Must be the RADIUS module
181  */
182  fr_assert(cs && (strcmp(cf_section_name1(cs), "radius") == 0));
183 
184  /*
185  * Allow the process module to be specified by
186  * packet type.
187  */
188  type_enum = fr_dict_enum_by_name(attr_packet_type, type_str, -1);
189  if (!type_enum) {
190  invalid_code:
191  cf_log_err(ci, "Unknown or invalid RADIUS packet type '%s'", type_str);
192  return -1;
193  }
194 
195  code = type_enum->value->vb_uint32;
196 
197  /*
198  * Status-Server packets cannot be proxied.
199  */
200  if (code == FR_CODE_STATUS_SERVER) {
201  cf_log_err(ci, "Invalid setting of 'type = Status-Server'. Status-Server packets cannot be proxied.");
202  return -1;
203  }
204 
205  if (!code ||
206  (code >= FR_RADIUS_MAX_PACKET_CODE) ||
207  (!type_interval_config[code].name)) goto invalid_code;
208 
209  /*
210  * If we're doing async proxying, push the timers for the
211  * various packet types.
212  */
213  cf_section_rule_push(cs, &type_interval_config[code]);
214 
215  memcpy(out, &code, sizeof(code));
216 
217  return 0;
218 }

254 /** Allow for Status-Server ping checks
255  *
256  * @param[in] ctx to allocate data in (instance of proto_radius).
257  * @param[out] out Where to write our parsed data.
258  * @param[in] parent Base structure address.

265 static int status_check_type_parse(UNUSED TALLOC_CTX *ctx, void *out, UNUSED void *parent,
266  CONF_ITEM *ci, UNUSED CONF_PARSER const *rule)
267 {
268  char const *type_str = cf_pair_value(cf_item_to_pair(ci));
269  CONF_SECTION *cs = cf_item_to_section(cf_parent(ci));
270  fr_dict_enum_t const *type_enum;
271  uint32_t code;
272 
273  /*
274  * Allow the process module to be specified by
275  * packet type.
276  */
277  type_enum = fr_dict_enum_by_name(attr_packet_type, type_str, -1);
278  if (!type_enum) {
279  invalid_code:
280  cf_log_err(ci, "Unknown or invalid RADIUS packet type '%s'", type_str);
281  return -1;
282  }
283 
284  code = type_enum->value->vb_uint32;
285 
286  /*
287  * Cheat, and re-use the "type" array for allowed packet
288  * types.
289  */
290  if (!code ||
291  (code >= FR_RADIUS_MAX_PACKET_CODE) ||
292  (!type_interval_config[code].name)) goto invalid_code;
293 
294  /*
295  * Add irt / mrt / mrd / mrc parsing, in the parent
296  * configuration section.
297  */
298  cf_section_rule_push(cf_item_to_section(cf_parent(cs)), &type_interval_config[code]);
299 
300  memcpy(out, &code, sizeof(code));
301 
302  /*
303  * Nothing more to do here, so we stop.
304  */
305  if (code == FR_CODE_STATUS_SERVER) return 0;
306 
307  cf_section_rule_push(cs, status_check_update_config);
308 
309  return 0;
310 }

312 /** Allow the admin to set packet contents for Status-Server ping checks
313  *
314  * @param[in] ctx to allocate data in (instance of proto_radius).
315  * @param[out] out Where to write our parsed data
316  * @param[in] parent Base structure address.
317  * @param[in] ci #CONF_SECTION specifying the things to update
318  * @param[in] rule unused.
319  * @return
320  * - 0 on success.
321  * - -1 on failure.
322  */
323 static int status_check_update_parse(TALLOC_CTX *ctx, void *out, UNUSED void *parent,
324  CONF_ITEM *ci, UNUSED CONF_PARSER const *rule)
325 {
326  int rcode;
327  CONF_SECTION *cs;
328  char const *name2;
329  map_t *head = NULL;
330 
331  fr_assert(cf_item_is_section(ci));
332 
333  cs = cf_item_to_section(ci);
334  name2 = cf_section_name2(cs);
335  if (!name2 || (strcmp(name2, "request") != 0)) {
336  cf_log_err(cs, "You must specify 'request' as the destination list");
337  return -1;
338  }
339 
340  /*
341  * Compile the "update" section.
342  */
343  {
344  tmpl_rules_t parse_rules = {
345  .allow_foreign = true /* Because we don't know where we'll be called */
346  };
347 
348  rcode = map_afrom_cs(ctx, &head, cs, &parse_rules, &parse_rules, unlang_fixup_update, NULL, 128);
349  if (rcode < 0) return -1; /* message already printed */
350  if (!head) {
351  cf_log_err(cs, "'update' sections cannot be empty");
352  return -1;
353  }
354  }
355 
356  /*
357  * Rely on "bootstrap" to do sanity checks between 'type
358  * = Access-Request', and 'update' containing passwords.
359  */
360  memcpy(out, &head, sizeof(head));
361 
362  return 0;
363 }

395 /** Do any RADIUS-layer fixups for proxying.
396  *
397  */
398 static void radius_fixups(rlm_radius_t const *inst, request_t *request)
399 {
400  fr_pair_t *vp;
401 
402  /*
403  * Check for proxy loops.
404  */
405  if (RDEBUG_ENABLED) {
406  fr_cursor_t cursor;
407 
408  for (vp = fr_cursor_iter_by_da_init(&cursor, &request->request_pairs, attr_proxy_state);
409  vp;
410  vp = fr_cursor_next(&cursor)) {
411  if (vp->vp_length != 4) continue;
412 
413  if (memcmp(&inst->proxy_state, vp->vp_octets, 4) == 0) {
414  RWARN("Possible proxy loop - please check server configuration.");
415  break;
416  }
417  }
418  }
419 
420  if (request->packet->code != FR_CODE_ACCESS_REQUEST) return;
421 
422  if (fr_pair_find_by_da(&request->request_pairs, attr_chap_password) &&
423  !fr_pair_find_by_da(&request->request_pairs, attr_chap_challenge)) {
424  MEM(pair_add_request(&vp, attr_chap_challenge) >= 0);
425  fr_pair_value_memdup(vp, request->packet->vector, sizeof(request->packet->vector), true);
426  }
427 }

430 /** Send packets outbound.
431  *
432  */
433 static unlang_action_t CC_HINT(nonnull) mod_process(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
434 {
435  rlm_radius_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_radius_t);
436  rlm_radius_thread_t *t = talloc_get_type_abort(mctx->thread, rlm_radius_thread_t);
437  rlm_rcode_t rcode;
438 
439  void *rctx = NULL;
440 
441  if (!request->packet->code) {
442  REDEBUG("You MUST specify a packet code");
443  RETURN_MODULE_FAIL;
444  }
445 
446  /*
447  * Reserve Status-Server for ourselves, for link-specific
448  * signaling.
449  */
450  if (request->packet->code == FR_CODE_STATUS_SERVER) {
451  REDEBUG("Cannot proxy Status-Server packets");
452  RETURN_MODULE_FAIL;
453  }
454 
455  if ((request->packet->code >= FR_RADIUS_MAX_PACKET_CODE) ||
456  !inst->retry[request->packet->code].irt) { /* can't be zero */
457  REDEBUG("Invalid packet code %d", request->packet->code);
458  RETURN_MODULE_FAIL;
459  }
460 
461  if (!inst->allowed[request->packet->code]) {
462  REDEBUG("Packet code %s is disallowed by the configuration",
463  fr_packet_codes[request->packet->code]);
464  RETURN_MODULE_FAIL;
465  }
466 
467  if (request->client->dynamic && !request->client->active) {
468  REDEBUG("Cannot proxy packets which define dynamic clients");
469  RETURN_MODULE_FAIL;
470  }
471 
472  /*
473  * Do any necessary RADIUS level fixups
474  * - check Proxy-State
475  * - do CHAP-Challenge fixups
476  */
477  radius_fixups(inst, request);
478 
479  /*
480  * Push the request and it's data to the IO submodule.
481  *
482  * This may return YIELD, for "please yield", or it may
483  * return another code which indicates what happened to
484  * the request...b
485  */
486  inst->io->enqueue(&rcode, &rctx, inst->io_instance, t->io_thread, request);
487  if (rcode != RLM_MODULE_YIELD) {
488  fr_assert(rctx == NULL);
489  RETURN_MODULE_RCODE(rcode);
490  }
491 
492  return unlang_module_yield(request, mod_radius_resume, mod_radius_signal, rctx);
493 }

568 /** Bootstrap the module
569  *
570  * Bootstrap I/O and type submodules.
571  *
572  * @param[in] instance Ctx data for this module
573  * @param[in] conf our configuration section parsed to give us instance.
574  * @return
575  * - 0 on success.
576  * - -1 on failure.
577  */
578 static int mod_bootstrap(void *instance, CONF_SECTION *conf)
579 {
580  size_t i, num_types;
581  rlm_radius_t *inst = talloc_get_type_abort(instance, rlm_radius_t);
582 
583  inst->name = cf_section_name2(conf);
584  if (!inst->name) inst->name = cf_section_name1(conf);
585 
586  /*
587  * These limits are specific to RADIUS, and cannot be over-ridden
588  */
589  FR_INTEGER_BOUND_CHECK("trunk.per_connection_max", inst->trunk_conf.max_req_per_conn, >=, 2);
590  FR_INTEGER_BOUND_CHECK("trunk.per_connection_max", inst->trunk_conf.max_req_per_conn, <=, 255);
591  FR_INTEGER_BOUND_CHECK("trunk.per_connection_target", inst->trunk_conf.target_req_per_conn, <=, inst->trunk_conf.max_req_per_conn / 2);
592 
593  FR_TIME_DELTA_BOUND_CHECK("zombie_period", inst->zombie_period, >=, fr_time_delta_from_sec(1));
594  FR_TIME_DELTA_BOUND_CHECK("zombie_period", inst->zombie_period, <=, fr_time_delta_from_sec(120));
595 
596  if (!inst->status_check) {
597  FR_TIME_DELTA_BOUND_CHECK("revive_interval", inst->revive_interval, >=, fr_time_delta_from_sec(10));
598  FR_TIME_DELTA_BOUND_CHECK("revive_interval", inst->revive_interval, <=, fr_time_delta_from_sec(3600));
599  }
600 
601  num_types = talloc_array_length(inst->types);
602  fr_assert(num_types > 0);
603 
604  /*
605  * Allow for O(1) lookup later...
606  */
607  for (i = 0; i < num_types; i++) {
608  uint32_t code;
609 
610  code = inst->types[i];
611  fr_assert(code > 0);
612  fr_assert(code < FR_RADIUS_MAX_PACKET_CODE);
613 
614  inst->allowed[code] = true;
615  }
616 
617  fr_assert(inst->status_check < FR_RADIUS_MAX_PACKET_CODE);
618 
619  /*
620  * If we're replicating, we don't care if the other end
621  * is alive.
622  */
623  if (inst->replicate && inst->status_check) {
624  cf_log_warn(conf, "Ignoring 'status_check = %s' due to 'replicate = true'",
625  fr_packet_codes[inst->status_check]);
626  inst->status_check = 0;
627  }
628 
629 
630  /*
631  * If we have status checks, then do some sanity checks.
632  * Status-Server is always allowed. Otherwise, the
633  * status checks have to match one of the allowed
634  * packets.
635  */
636  if (inst->status_check) {
637  if (inst->status_check == FR_CODE_STATUS_SERVER) {
638  inst->allowed[inst->status_check] = true;
639 
640  } else if (!inst->allowed[inst->status_check]) {
641  cf_log_err(conf, "Using 'status_check = %s' requires also 'type = %s'",
642  fr_packet_codes[inst->status_check], fr_packet_codes[inst->status_check]);
643  return -1;
644  }
645 
646  /*
647  * @todo - check the contents of the "update"
648  * section, to be sure that (e.g.) Access-Request
649  * contains User-Name, etc.
650  */
651  }
652 
653  /*
654  * Don't sanity check the async timers if we're doing
655  * synchronous proxying.
656  */
657  if (inst->synchronous) goto setup_io_submodule;
658 
659  /*
660  * Set limits on retransmission timers
661  */
662  if (inst->allowed[FR_CODE_ACCESS_REQUEST]) {
663  FR_TIME_DELTA_BOUND_CHECK("Access-Request.initial_rtx_time", inst->retry[FR_CODE_ACCESS_REQUEST].irt, >=, fr_time_delta_from_sec(1));
664  FR_TIME_DELTA_BOUND_CHECK("Access-Request.max_rtx_time", inst->retry[FR_CODE_ACCESS_REQUEST].mrt, >=, fr_time_delta_from_sec(5));
665  FR_INTEGER_BOUND_CHECK("Access-Request.max_rtx_count", inst->retry[FR_CODE_ACCESS_REQUEST].mrc, >=, 1);
666  FR_TIME_DELTA_BOUND_CHECK("Access-Request.max_rtx_duration", inst->retry[FR_CODE_ACCESS_REQUEST].mrd, >=, fr_time_delta_from_sec(5));
667 
668  FR_TIME_DELTA_BOUND_CHECK("Access-Request.initial_rtx_time", inst->retry[FR_CODE_ACCESS_REQUEST].irt, <=, fr_time_delta_from_sec(3));
669  FR_TIME_DELTA_BOUND_CHECK("Access-Request.max_rtx_time", inst->retry[FR_CODE_ACCESS_REQUEST].mrt, <=, fr_time_delta_from_sec(30));
670  FR_INTEGER_BOUND_CHECK("Access-Request.max_rtx_count", inst->retry[FR_CODE_ACCESS_REQUEST].mrc, <=, 10);
671  FR_TIME_DELTA_BOUND_CHECK("Access-Request.max_rtx_duration", inst->retry[FR_CODE_ACCESS_REQUEST].mrd, <=, fr_time_delta_from_sec(30));
672  }
673 
674  /*
675  * Note that RFC 5080 allows for Accounting-Request to
676  * have mrt=mrc=mrd = 0, which means "retransmit
677  * forever". We allow that, with the restriction that
678  * the server core will automatically free the request at
679  * max_request_time.
680  */
681  if (inst->allowed[FR_CODE_ACCOUNTING_REQUEST]) {
682  FR_TIME_DELTA_BOUND_CHECK("Accounting-Request.initial_rtx_time", inst->retry[FR_CODE_ACCOUNTING_REQUEST].irt, >=, fr_time_delta_from_sec(1));
683 #if 0
684  FR_TIME_DELTA_BOUND_CHECK("Accounting-Request.max_rtx_time", inst->retry[FR_CODE_ACCOUNTING_REQUEST].mrt, >=, fr_time_delta_from_sec(5));
685  FR_INTEGER_BOUND_CHECK("Accounting-Request.max_rtx_count", inst->retry[FR_CODE_ACCOUNTING_REQUEST].mrc, >=, 0);
686  FR_TIME_DELTA_BOUND_CHECK("Accounting-Request.max_rtx_duration", inst->retry[FR_CODE_ACCOUNTING_REQUEST].mrd, >=, fr_time_delta_from_sec(0));
687 #endif
688 
689  FR_TIME_DELTA_BOUND_CHECK("Accounting-Request.initial_rtx_time", inst->retry[FR_CODE_ACCOUNTING_REQUEST].irt, <=, fr_time_delta_from_sec(3));
690  FR_TIME_DELTA_BOUND_CHECK("Accounting-Request.max_rtx_time", inst->retry[FR_CODE_ACCOUNTING_REQUEST].mrt, <=, fr_time_delta_from_sec(30));
691  FR_INTEGER_BOUND_CHECK("Accounting-Request.max_rtx_count", inst->retry[FR_CODE_ACCOUNTING_REQUEST].mrc, <=, 10);
692  FR_TIME_DELTA_BOUND_CHECK("Accounting-Request.max_rtx_duration", inst->retry[FR_CODE_ACCOUNTING_REQUEST].mrd, <=, fr_time_delta_from_sec(30));
693  }
694 
695  /*
696  * Status-Server
697  */
698  if (inst->allowed[FR_CODE_STATUS_SERVER]) {
699  FR_TIME_DELTA_BOUND_CHECK("Status-Server.initial_rtx_time", inst->retry[FR_CODE_STATUS_SERVER].irt, >=, fr_time_delta_from_sec(1));
700  FR_TIME_DELTA_BOUND_CHECK("Status-Server.max_rtx_time", inst->retry[FR_CODE_STATUS_SERVER].mrt, >=, fr_time_delta_from_sec(5));

717  FR_TIME_DELTA_BOUND_CHECK("CoA-Request.max_rtx_duration", inst->retry[FR_CODE_COA_REQUEST].mrd, >=, fr_time_delta_from_sec(5));

719  FR_TIME_DELTA_BOUND_CHECK("CoA-Request.initial_rtx_time", inst->retry[FR_CODE_COA_REQUEST].irt, <=, fr_time_delta_from_sec(3));
720  FR_TIME_DELTA_BOUND_CHECK("CoA-Request.max_rtx_time", inst->retry[FR_CODE_COA_REQUEST].mrt, <=, fr_time_delta_from_sec(60));
721  FR_INTEGER_BOUND_CHECK("CoA-Request.max_rtx_count", inst->retry[FR_CODE_COA_REQUEST].mrc, <=, 10);

723  }

727  */
728  if (inst->allowed[FR_CODE_DISCONNECT_REQUEST]) {
729  FR_TIME_DELTA_BOUND_CHECK("Disconnect-Request.initial_rtx_time", inst->retry[FR_CODE_DISCONNECT_REQUEST].irt, >=, fr_time_delta_from_sec(1));

737  FR_TIME_DELTA_BOUND_CHECK("Disconnect-Request.max_rtx_duration", inst->retry[FR_CODE_DISCONNECT_REQUEST].mrd, <=, fr_time_delta_from_sec(30));
