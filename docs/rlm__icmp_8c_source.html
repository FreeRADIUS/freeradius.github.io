95 /*
96  * Calculate the ICMP portion of the checksum
97  */
98 static uint16_t icmp_checksum(uint8_t *data, size_t data_len, uint16_t checksum)
99 {
100  uint8_t *p, *end;
101  uint64_t sum;
102 
103  sum = checksum;
104  data_len &= ~((size_t) 1); /* ensure it's always 16-bit aligned */
105 
106  p = data;
107  end = data + data_len;
108  while (p < end) {
109  sum += fr_net_to_uint16(p); /* type / code */
110  p += 2;
111  }
112 
113  while (sum >> 16) sum = (sum & 0xffff) + (sum >> 16);
114 
115  return ((uint16_t) ~sum);
116 }

172 /** Xlat to delay the request
173  *
174  * Example (ping 192.0.2.1):
175 @verbatim
176 "%{icmp:192.0.2.1}"
177 @endverbatim
178  *
179  * @ingroup xlat_functions
180  */
181 static xlat_action_t xlat_icmp(TALLOC_CTX *ctx, UNUSED fr_cursor_t *out,
182  request_t *request, void const *xlat_inst, void *xlat_thread_inst,
183  fr_value_box_t **in)
184 {
185  void *instance;
186  rlm_icmp_t const *inst;
187  xlat_icmp_thread_inst_t *thread = talloc_get_type_abort(xlat_thread_inst, xlat_icmp_thread_inst_t);
188  rlm_icmp_echo_t *echo;
189  icmp_header_t icmp;
190  uint16_t checksum;
191  ssize_t rcode;
192  socklen_t salen;
193  struct sockaddr_storage dst;
194 
195  memcpy(&instance, xlat_inst, sizeof(instance)); /* Stupid const issues */
196 
197  inst = talloc_get_type_abort(instance, rlm_icmp_t);
198 
199  /*
200  * If there's no input, do nothing.
201  */
202  if (!*in) return XLAT_ACTION_FAIL;
203 
204  if (fr_value_box_list_concat(ctx, *in, in, FR_TYPE_STRING, true) < 0) {
205  RPEDEBUG("Failed concatenating input");
206  return XLAT_ACTION_FAIL;
207  }
208 
209  if (fr_value_box_cast_in_place(ctx, *in, thread->t->ipaddr_type, NULL) < 0) {
210  RPEDEBUG("Failed casting result to IP address");
211  return XLAT_ACTION_FAIL;
212  }
213 
214  MEM(echo = talloc_zero(ctx, rlm_icmp_echo_t));
215  echo->ip = *in;
216  echo->request = request;
217  echo->counter = thread->t->counter++;
218 
219  /*
220  * Add the IP to the local tracking heap, so that the IO
221  * functions can find it.
222  *
223  * This insert will never fail, because of the unique
224  * counter above.
225  */
226  if (!rbtree_insert(thread->t->tree, echo)) {
227  RPEDEBUG("Failed inserting IP into tracking table");
228  talloc_free(echo);
229  return XLAT_ACTION_FAIL;
230  }
231 
232  if (unlang_xlat_event_timeout_add(request, _xlat_icmp_timeout, echo, fr_time() + inst->timeout) < 0) {
233  RPEDEBUG("Failed adding timeout");
234  (void) rbtree_deletebydata(thread->t->tree, echo);
235  talloc_free(echo);
236  return XLAT_ACTION_FAIL;
237  }
238 
239  RDEBUG("Sending ICMP request to %pV (counter=%d)", echo->ip, echo->counter);
240 
241  icmp = (icmp_header_t) {
242  .type = thread->t->request_type,
243  .ident = thread->t->ident,
244  .data = thread->t->data,
245  .counter = echo->counter
246  };
247 
248  (void) fr_ipaddr_to_sockaddr(&dst, &salen, &echo->ip->vb_ip, 0);
249 
250  /*
251  * Calculate the checksum
252  */
253  checksum = 0;
254 
255  /*
256  * Start off with the IPv6 pseudo-header checksum
257  */
258  if (thread->t->ipaddr_type == FR_TYPE_IPV6_ADDR) {
259  checksum = fr_ip6_pesudo_header_checksum(&thread->t->inst->src_ipaddr.addr.v6, &echo->ip->vb_ip.addr.v6,
260  sizeof(ip_header6_t) + sizeof(icmp), IPPROTO_ICMPV6);
261  }
262 
263  /*
264  * Followed by checksumming the actual ICMP packet.
265  */
266  icmp.checksum = htons(icmp_checksum((uint8_t *) &icmp, sizeof(icmp), checksum));
267 
268  rcode = sendto(thread->t->fd, &icmp, sizeof(icmp), 0, (struct sockaddr *) &dst, salen);
269  if (rcode < 0) {
270  REDEBUG("Failed sending ICMP request to %pV: %s", echo->ip, fr_syserror(errno));
271  (void) rbtree_deletebydata(thread->t->tree, echo);
272  talloc_free(echo);
273  return XLAT_ACTION_FAIL;
274  }
275 
276  if ((size_t) rcode < sizeof(icmp)) {
277  REDEBUG("Failed sending entire ICMP packet");
278  (void) rbtree_deletebydata(thread->t->tree, echo);
279  talloc_free(echo);
280  return XLAT_ACTION_FAIL;
281  }
282 
283  return unlang_xlat_yield(request, xlat_icmp_resume, xlat_icmp_cancel, echo);
284 }

286 static int echo_cmp(void const *one, void const *two)
287 {
288  rlm_icmp_echo_t const *a = one;
289  rlm_icmp_echo_t const *b = two;
290 
291  /*
292  * No need to check IP, because "counter" is unique for each packet.
293  */
294  return (a->counter < b->counter) - (a->counter > b->counter);
295 }

297 static void mod_icmp_read(UNUSED fr_event_list_t *el, UNUSED int sockfd, UNUSED int flags, void *ctx)
298 {
299  rlm_icmp_thread_t *t = talloc_get_type_abort(ctx, rlm_icmp_thread_t);
300  rlm_icmp_t *inst = t->inst;
301  ssize_t len;
302  icmp_header_t *icmp;
303  rlm_icmp_echo_t my_echo, *echo;
304  uint64_t buffer[256];
305 
306  len = read(t->fd, (char *) buffer, sizeof(buffer));
307  if (len <= 0) return;
308 
309  HEXDUMP4((uint8_t const *)buffer, len, "received icmp packet ");
310 
311  /*
312  * Ignore packets if we haven't sent any requests.
313  */
314  if (rbtree_num_elements(t->tree) == 0) {
315  return;
316  }
317 
318  // buffer is actually the IP header + the ICMP packet
319  if (t->ipaddr_type == FR_TYPE_IPV4_ADDR) {
320  ip_header_t *ip = (ip_header_t *) buffer;
321 
322  if (IP_V(ip) != 4) {
323  return;
324  }
325 
326  if ((IP_HL(ip) + sizeof(*icmp)) > sizeof(buffer)) {
327  return;
328  }
329 
330  icmp = (icmp_header_t *) (((uint8_t *) buffer) + IP_HL(ip));
331  } else if (t->ipaddr_type == FR_TYPE_IPV6_ADDR) {
332  /*
333  * Outgoing packets automatically have an IPv6 header prepended to them
334  * (based on the destination address). ICMPv6 pseudo header checksum field
335  * (icmp6_cksum) will be filled automatically by the kernel. Incoming packets
336  * are received without the IPv6 header nor IPv6 extension headers.
337  *
338  * Note that this behavior is opposite from IPv4
339  * raw sockets and ICMPv4 sockets.
340  *
341  * Therefore, we don't have ip6 headers here. Only the icmp6 packet.
342  */
343  icmp = (icmp_header_t *) (((uint8_t *) buffer));
344  } else {
345  /*
346  * No idea. Ignore it.
347  */
348  return;
349  }
350 
351  /*
352  * Ignore packets which aren't an echo reply, or which
353  * weren't for us. This is done *before* looking packets
354  * up in the rbtree, as these checks ensure that the
355  * packet is for this specific thread.
356  */
357  if ((icmp->type != t->reply_type) ||
358  (icmp->ident != t->ident) || (icmp->data != t->data)) {
359  return;
360  }
361 
362  /*
363  * Look up the packet by the fields which determine *our* ICMP packets.
364  */
365  my_echo.counter = icmp->counter;
366  echo = rbtree_finddata(t->tree, &my_echo);
367  if (!echo) {
368  DEBUG("Can't find packet counter=%d in tree", icmp->counter);
369  return;
370  }
371 
372  (void) rbtree_deletebydata(t->tree, echo);
373 
374  /*
375  * We have a reply!
376  */
377  echo->replied = true;
378  unlang_interpret_resumable(echo->request);
379 }

422 /** Instantiate thread data for the submodule.
423  *
424  */
425 static int mod_thread_instantiate(UNUSED CONF_SECTION const *cs, void *instance, fr_event_list_t *el, void *thread)
426 {
427  int fd, af, proto;
428  rlm_icmp_t *inst = talloc_get_type_abort(instance, rlm_icmp_t);
429  rlm_icmp_thread_t *t = talloc_get_type_abort(thread, rlm_icmp_thread_t);
430  fr_ipaddr_t ipaddr, *src;
431 
432  MEM(t->tree = rbtree_alloc(t, echo_cmp, NULL, RBTREE_FLAG_NONE));
433  t->inst = inst;
434  t->el = el;
435 
436  /*
437  * Since these fields are random numbers, we don't care
438  * about network / host byte order. No one other than us
439  * will be interpreting these fields. As such, we can
440  * just treat them as host byte order.
441  *
442  * The only side effect of this choice is that this code
443  * will use (e.g.) 0xabcd for the ident, and Wireshark,
444  * tcpdump, etc. may show the ident as 0xcdab. That's
445  * fine.
446  */
447  t->data = fr_rand();
448  t->ident = fr_rand();
449 
450  af = inst->src_ipaddr.af;
451 
452  switch (af) {
453  default:
454  fr_strerror_printf("Unsupported address family");
455  return -1;
456 
457  case AF_UNSPEC:
458  case AF_INET:
459  af = AF_INET;
460  proto = IPPROTO_ICMP;
461  t->request_type = ICMP_ECHOREQUEST;
462  t->reply_type = ICMP_ECHOREPLY;
463  t->ipaddr_type = FR_TYPE_IPV4_ADDR;
464  break;
465 
466  case AF_INET6:
467  af = AF_INET6;
468  proto = IPPROTO_ICMPV6;
469  t->request_type = ICMPV6_ECHOREQUEST;
470  t->reply_type = ICMPV6_ECHOREPLY;
471  t->ipaddr_type = FR_TYPE_IPV6_ADDR;
472  break;
473  }
474 
475  /*
476  * Try and open with SOCK_DGRAM.
477  * If we get permission denied, fall back to SOCK_RAW.
478  * For some reason with docker, even if we have all
479  * the capabilities opening a SOCK_DGRAM/IPPROTO_ICMP
480  * socket fails.
481  *
482  * We don't appear to need to specify the IP header
483  * and the xlat works fine. Very strange.
484  */
485  fd = socket(af, SOCK_DGRAM, proto);
486  if (fd < 0) fd = socket(af, SOCK_RAW, proto);
487  if (fd < 0) {
488  fr_strerror_printf("Failed opening socket (%s, %s): %s",
489  fr_table_str_by_value(fr_net_af_table, af, "<INVALID>"),
490  fr_table_str_by_value(fr_net_ip_proto_table, proto, "<INVALID>"),
491  fr_syserror(errno));
492  return -1;
493  }
494 
495 #ifndef FD_CLOEXEC
496 #define FD_CLOEXEC (0)
497 #endif
498 
499  (void) fcntl(fd, F_SETFL, O_NONBLOCK | FD_CLOEXEC);
500 
501  if (inst->src_ipaddr.af != AF_UNSPEC) {
502  ipaddr = inst->src_ipaddr;
503  src = &ipaddr;
504  } else {
505  src = NULL;
506  }
507 
508  /*
509  * Only bind if we have a src and interface.
510  */
511  if (src && inst->interface && (fr_socket_bind(fd, src, NULL, inst->interface) < 0)) {
512  close(fd);
513  return -1;
514  }
515 
516  /*
517  * We assume that the outbound socket is always writable.
518  * If not, too bad. Packets will get lost.
519  */
520  if (fr_event_fd_insert(t, el, fd,
521  mod_icmp_read,
522  NULL,
523  mod_icmp_error,
524  t) < 0) {
525  fr_strerror_printf_push("Failed adding socket to event loop");
526  close(fd);
527  return -1;
528  }
529  t->fd = fd;
530 
531  return 0;
532 }
