329  *

803 /** Synchronously load cookie data
804  *
805  * FIXME: This should not be synchronous, but integrating it into the event loop
806  * before the server has started processing requests makes my head hurt.
807  *

809  * @param[out] cookie Where to write the cookie we loaded.
810  * @param[in] listen structure encapsulating the LDAP
811  * @param[in] config of the sync we're loading the cookie for.
812  * @return
813  * - -1 on failure.
814  * - 0 on success.
815  * - 1 no cookie returned.
816  */
817 static int proto_ldap_cookie_load(TALLOC_CTX *ctx, uint8_t **cookie, rad_listen_t *listen, sync_config_t const *config)
818 {
819  proto_ldap_inst_t *inst = talloc_get_type_abort(listen->data, proto_ldap_inst_t);
820  request_t *request;
821  CONF_SECTION *unlang;
822  int ret = 0;

825 
826  request = proto_ldap_request_setup(listen, inst, 0);
827  if (!request) return -1;
828 
829  proto_ldap_attributes_add(request, config);
830  request->packet->code = LDAP_SYNC_CODE_COOKIE_STORE;
831 
832  unlang = cf_section_find(request->server_cs, "load", "Cookie");
833  if (!unlang) {
834  RDEBUG2("Ignoring %s operation. Add \"load Cookie {}\" to virtual-server \"%s\""
835  " to handle", fr_table_str_by_value(ldap_sync_code_table, request->packet->code, "<INVALID>"),
836  cf_section_name2(request->server_cs));
837  }
838 
839  *cookie = NULL;
840 
841  rcode = unlang_interpret_synchronous(request, unlang, RLM_MODULE_NOOP, true);
842  switch (rcode) {

844  case RLM_MODULE_UPDATED:
845  {
846  fr_pair_t *vp;
847

851  "sync configuration will be returned");

853  goto finish;
854  }
855 
856  /*
857  * So the request pool doesn't hang around indefinitely.
858  */
859  MEM(*cookie = talloc_memdup(ctx, vp->vp_octets, vp->vp_length));
860  ret = 0;
861  }
862  break;
863 
864  case RLM_MODULE_NOOP:
865  if (config->allow_refresh) RDEBUG2("Section returned \"noop\". All entries matching sync "
866  "configuration will be returned");
867  ret = 1;
868  break;
869 
870  default:
871  RERROR("Section must return \"ok\", \"updated\", or \"noop\" for listener instantiation to succeed");
872  ret = -1;
873  break;
874  }

877  talloc_free(request);

881 /** De-multiplex incoming LDAP Messages
882  *
883  * This callback should be called when the LDAP socket is readable. It drains
884  * any outstanding LDAPMessages using sync_demux. sync_demux in-turn calls one
885  * of the following callbacks:
886  * - _proto_ldap_entry We received a notification an entry
887  * was added, deleted, or modified.
888  * - _proto_ldap_cookie_store We received a new cookie value.
889  * - _proto_ldap_present The server wants to perform a refresh present
890  * phase (which we don't allow).
891  * - _proto_ldap_refresh_required The server wants us to download all content
892  * specified by our sync/search.
893  *
894  * These callbacks may result in requests being enqueued or syncs restarted.
895  *
896  * @note We do not currently make enqueuing the cookie requests dependent on all
897  * previous LDAP entries being processed, so we may miss updates in some
898  * circumstances. This needs to be fixed, but is waiting on v4.0.0
899  * re-architecture.
900  *
901  * @param[in] listen encapsulating the libldap socket.
902  * @return
903  * - 1 on success.
904  * - 0 on failure.
905  */
906 static int proto_ldap_socket_recv(rad_listen_t *listen)
907 {
908  proto_ldap_inst_t *inst = talloc_get_type_abort(listen->data, proto_ldap_inst_t);
909  int sync_id;
910  void *ctx;
911  sync_config_t const *config;
912 
913  /*
914  * Demultiplex drains any outstanding messages from the socket,
915  * and calls the _proto_ldap_entry() callback above to create
916  * the request..
917  *
918  * Multiple requests may be created from one call to sync_demux.
919  */
920  switch (sync_demux(&sync_id, inst->conn)) {
921  default:
922  return 1;
923 
924  case -1:
925  PERROR("Sync failed - will retry in %pV seconds", fr_box_time_delta(inst->sync_retry_interval));
926 
927  config = sync_state_config_get(inst->conn, sync_id);
928  sync_state_destroy(inst->conn, sync_id); /* Destroy the old state */
929 
930  /*
931  * Schedule sync reinit, but don't perform it immediately.
932  */
933  memcpy(&ctx, &config, sizeof(ctx));
934  if (fr_event_timer_in(inst, inst->el, &inst->sync_retry_ev,
935  inst->sync_retry_interval,
936  proto_ldap_sync_reinit, ctx) < 0) {
937  FATAL("Failed inserting event: %s", fr_strerror());
938  }
939  return 1;
940 
941  case -2:
942  PERROR("Connection failed - will retry in %pV seconds", fr_box_time_delta(inst->conn_retry_interval));
943 
944  /*
945  * Schedule conn reinit, but don't perform it immediately
946  */
947  memcpy(&ctx, &config, sizeof(ctx));
948  if (fr_event_timer_in(inst, inst->el, &inst->conn_retry_ev,
949  inst->conn_retry_interval,
950  proto_ldap_connection_init, listen) < 0) {
951  FATAL("Failed inserting event: %s", fr_strerror());
952  }
953 
954  return 0;
955  }
956 }

1156  */

1158  if (map_cs && map_afrom_cs(inst, &inst->sync_config[i]->entry_map, map_cs,
