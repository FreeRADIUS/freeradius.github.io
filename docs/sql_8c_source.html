66 void *sql_mod_conn_create(TALLOC_CTX *ctx, void *instance, fr_time_delta_t timeout)
67 {
68  int rcode;
69  rlm_sql_t *inst = instance;
70  rlm_sql_handle_t *handle;
71 
72  /*
73  * Connections cannot be alloced from the inst or
74  * pool contexts due to threading issues.
75  */
76  handle = talloc_zero(ctx, rlm_sql_handle_t);
77  if (!handle) return NULL;
78 
79  handle->log_ctx = talloc_pool(handle, 2048);
80  if (!handle->log_ctx) {
81  talloc_free(handle);
82  return NULL;
83  }
84 
85  /*
86  * Handle requires a pointer to the SQL inst so the
87  * destructor has access to the module configuration.
88  */
89  handle->inst = inst;
90 
91  rcode = (inst->driver->sql_socket_init)(handle, inst->config, timeout);
92  if (rcode != 0) {
93  fail:
94  /*
95  * Destroy any half opened connections.
96  */
97  talloc_free(handle);
98  return NULL;
99  }
100 
101  if (inst->config->connect_query) {
102  if (rlm_sql_select_query(inst, NULL, &handle, inst->config->connect_query) != RLM_SQL_OK) goto fail;
103  (inst->driver->sql_finish_select_query)(handle, inst->config);
104  }
105 
106  return handle;
107 }

116 int sql_fr_pair_list_afrom_str(TALLOC_CTX *ctx, request_t *request, fr_cursor_t *cursor, rlm_sql_row_t row)
117 {
118  fr_pair_t *vp;
119  char const *ptr, *value;
120  char buf[FR_MAX_STRING_LEN];
121  char do_xlat = 0;
122  fr_token_t token, op = T_EOL;
123 
124  /*
125  * Verify the 'Attribute' field
126  */
127  if (!row[2] || row[2][0] == '\0') {
128  REDEBUG("Attribute field is empty or NULL, skipping the entire row");
129  return -1;
130  }
131 
132  /*
133  * Verify the 'op' field
134  */
135  if (row[4] != NULL && row[4][0] != '\0') {
136  ptr = row[4];
137  op = gettoken(&ptr, buf, sizeof(buf), false);
138  if (!fr_assignment_op[op] && !fr_equality_op[op]) {
139  REDEBUG("Invalid op \"%s\" for attribute %s", row[4], row[2]);
140  return -1;
141  }
142 
143  } else {
144  /*
145  * Complain about empty or invalid 'op' field
146  */
147  op = T_OP_CMP_EQ;
148  REDEBUG("The op field for attribute '%s = %s' is NULL, or non-existent.", row[2], row[3]);
149  REDEBUG("You MUST FIX THIS if you want the configuration to behave as you expect");
150  }
151 
152  /*
153  * The 'Value' field may be empty or NULL
154  */
155  if (!row[3]) {
156  REDEBUG("Value field is empty or NULL, skipping the entire row");
157  return -1;
158  }
159 
160  value = row[3];
161 
162  /*
163  * If we have a new-style quoted string, where the
164  * *entire* string is quoted, do xlat's.
165  */
166  if (row[3] != NULL &&
167  ((row[3][0] == '\'') || (row[3][0] == '`') || (row[3][0] == '"')) &&
168  (row[3][0] == row[3][strlen(row[3])-1])) {
169 
170  token = gettoken(&value, buf, sizeof(buf), false);
171  switch (token) {
172  /*
173  * Mark the pair to be allocated later.
174  */
175  case T_BACK_QUOTED_STRING:
176  do_xlat = 1;
177  FALL_THROUGH;
178 
179  /*
180  * Take the unquoted string.
181  */
182  case T_SINGLE_QUOTED_STRING:
183  case T_DOUBLE_QUOTED_STRING:
184  value = buf;
185  break;
186 
187  /*
188  * Keep the original string.
189  */
190  default:
191  value = row[3];
192  break;
193  }
194  }
195 
196  /*
197  * Create the pair
198  */
199  vp = fr_pair_make(ctx, request->dict, NULL, row[2], NULL, op);
200  if (!vp) {
201  RPEDEBUG("Failed to create the pair");
202  return -1;
203  }
204 
205  if (do_xlat) {
206  if (fr_pair_mark_xlat(vp, value) < 0) {
207  RPEDEBUG("Error marking pair for xlat");
208 
209  talloc_free(vp);
210  return -1;
211  }
212  } else {
213  if (fr_pair_value_from_str(vp, value, -1, '\0', true) < 0) {
214  RPEDEBUG("Error parsing value");
215 
216  talloc_free(vp);
217  return -1;
218  }
219  }
220 
221  /*
222  * Add the pair into the packet
223  */
224  fr_cursor_append(cursor, vp);
225  return 0;
226 }

230  * Calls the driver's sql_fetch_row logging any errors. On success, will

242 {

244 
245  if (!*handle || !(*handle)->conn) return RLM_SQL_ERROR;

247  /*

261

320 /** Call the driver's sql_query method, reconnecting if necessary.
321  *
322  * @note Caller must call ``(inst->driver->sql_finish_query)(handle, inst->config);``
323  * after they're done with the result.
324  *
325  * @param handle to query the database with. *handle should not be NULL, as this indicates
326  * previous reconnection attempt has failed.
327  * @param request Current request.
328  * @param inst #rlm_sql_t instance data.
329  * @param query to execute. Should not be zero length.
330  * @return
331  * - #RLM_SQL_OK on success.
332  * - #RLM_SQL_RECONNECT if a new handle is required (also sets *handle = NULL).
333  * - #RLM_SQL_QUERY_INVALID, #RLM_SQL_ERROR on invalid query or connection error.
334  * - #RLM_SQL_ALT_QUERY on constraints violation.
335  */
336 sql_rcode_t rlm_sql_query(rlm_sql_t const *inst, request_t *request, rlm_sql_handle_t **handle, char const *query)
337 {
338  int ret = RLM_SQL_ERROR;
339  int i, count;
340 
341  /* Caller should check they have a valid handle */
342  fr_assert(*handle);
343 
344  /* There's no query to run, return an error */
345  if (query[0] == '\0') {
346  if (request) REDEBUG("Zero length query");
347  return RLM_SQL_QUERY_INVALID;
348  }
349 
350  /*
351  * inst->pool may be NULL is this function is called by sql_mod_conn_create.
352  */
353  count = inst->pool ? fr_pool_state(inst->pool)->num : 0;
354 
355  /*
356  * Here we try with each of the existing connections, then try to create
357  * a new connection, then give up.
358  */
359  for (i = 0; i < (count + 1); i++) {
360  ROPTIONAL(RDEBUG2, DEBUG2, "Executing query: %s", query);
361 
362  ret = (inst->driver->sql_query)(*handle, inst->config, query);
363  switch (ret) {
364  case RLM_SQL_OK:
365  break;
366 
367  /*
368  * Run through all available sockets until we exhaust all existing
369  * sockets in the pool and fail to establish a *new* connection.
370  */
371  case RLM_SQL_RECONNECT:
372  *handle = fr_pool_connection_reconnect(inst->pool, request, *handle);
373  /* Reconnection failed */
374  if (!*handle) return RLM_SQL_RECONNECT;
375  /* Reconnection succeeded, try again with the new handle */
376  continue;
377 
378  /*
379  * These are bad and should make rlm_sql return invalid
380  */
381  case RLM_SQL_QUERY_INVALID:
382  rlm_sql_print_error(inst, request, *handle, false);
383  (inst->driver->sql_finish_query)(*handle, inst->config);
384  break;
385 
386  /*
387  * Server or client errors.
388  *
389  * If the driver claims to be able to distinguish between
390  * duplicate row errors and other errors, and we hit a
391  * general error treat it as a failure.
392  *
393  * Otherwise rewrite it to RLM_SQL_ALT_QUERY.
394  */
395  case RLM_SQL_ERROR:
396  if (inst->driver->flags & RLM_SQL_RCODE_FLAGS_ALT_QUERY) {
397  rlm_sql_print_error(inst, request, *handle, false);
398  (inst->driver->sql_finish_query)(*handle, inst->config);
399  break;
400  }
401  ret = RLM_SQL_ALT_QUERY;
402  FALL_THROUGH;
403 
404  /*
405  * Driver suggested using an alternative query
406  */
407  case RLM_SQL_ALT_QUERY:
408  rlm_sql_print_error(inst, request, *handle, true);
409  (inst->driver->sql_finish_query)(*handle, inst->config);
410  break;
411 
412  }
413 
414  return ret;
415  }
416 
417  ROPTIONAL(RERROR, ERROR, "Hit reconnection limit");
418 
419  return RLM_SQL_ERROR;
420 }
