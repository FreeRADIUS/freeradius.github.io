105 /** Generate a reply using the attributes we received in the request
106  *
107  * This should be called before the user has the opportunity to mangle
108  * the request so that we get the original values.
109  */
110 static inline CC_HINT(always_inline) void dhcpv6_reply_initialise(request_t *request)
111 {
112  fr_cursor_t cursor;
113  fr_pair_t *vp;
114 
115  switch (request->packet->code) {
116  case FR_DHCPV6_RELAY_FORWARD:
117  for (vp = fr_cursor_init(&cursor, &request->request_pairs);
118  vp;
119  vp = fr_cursor_next(&cursor)) {
120  if (vp->da == attr_hop_count) {
121  fr_pair_add(&request->reply_pairs, fr_pair_copy(request->reply, vp));
122  continue;
123  }
124  if (vp->da == attr_relay_link_address) {
125  fr_pair_add(&request->reply_pairs, fr_pair_copy(request->reply, vp));
126  continue;
127  }
128  if (vp->da == attr_relay_peer_address) {
129  fr_pair_add(&request->reply_pairs, fr_pair_copy(request->reply, vp));
130  continue;
131  }
132  if (vp->da == attr_interface_id) {
133  fr_pair_add(&request->reply_pairs, fr_pair_copy(request->reply, vp));
134  continue;
135  }
136  }
137  break;
138 
139  default:
140  fr_pair_list_copy_by_da(request->reply, &request->reply_pairs, &request->request_pairs, attr_transaction_id, 1);
141  fr_pair_list_copy_by_ancestor(request->reply, &request->reply_pairs, &request->request_pairs, attr_client_id, 0);
142  break;
143  }
144 }

146 static unlang_action_t mod_process(rlm_rcode_t *p_result, UNUSED module_ctx_t const *mctx, request_t *request)
147 {
148  rlm_rcode_t rcode;
149  CONF_SECTION *unlang;
150  fr_dict_enum_t const *dv;
151  fr_pair_t *vp;
152 
153  static int reply_ok[] = {
154  [0] = FR_DHCPV6_DO_NOT_RESPOND,
155  [FR_DHCPV6_SOLICIT] = FR_DHCPV6_ADVERTISE,
156  [FR_DHCPV6_REQUEST] = FR_DHCPV6_CONFIRM,
157  [FR_DHCPV6_RENEW] = FR_DHCPV6_REPLY,
158  [FR_DHCPV6_REBIND] = FR_DHCPV6_REPLY,
159  [FR_DHCPV6_RELEASE] = FR_DHCPV6_REPLY,
160  [FR_DHCPV6_DECLINE] = FR_DHCPV6_REPLY,
161  [FR_DHCPV6_INFORMATION_REQUEST] = FR_DHCPV6_REPLY,
162  [FR_DHCPV6_RELAY_FORWARD] = FR_DHCPV6_RELAY_REPLY
163  };
164 
165  static int reply_fail[] = {
166  [0] = FR_DHCPV6_DO_NOT_RESPOND,
167  [FR_DHCPV6_SOLICIT] = FR_DHCPV6_DO_NOT_RESPOND,
168  [FR_DHCPV6_REQUEST] = FR_DHCPV6_DO_NOT_RESPOND,
169  [FR_DHCPV6_RENEW] = FR_DHCPV6_DO_NOT_RESPOND,
170  [FR_DHCPV6_REBIND] = FR_DHCPV6_DO_NOT_RESPOND,
171  [FR_DHCPV6_RELEASE] = FR_DHCPV6_DO_NOT_RESPOND,
172  [FR_DHCPV6_DECLINE] = FR_DHCPV6_DO_NOT_RESPOND,
173  [FR_DHCPV6_INFORMATION_REQUEST] = FR_DHCPV6_DO_NOT_RESPOND,
174  [FR_DHCPV6_RELAY_FORWARD] = FR_DHCPV6_DO_NOT_RESPOND
175  };
176 
177 #define REPLY_OK(_code, _default) ((_code < NUM_ELEMENTS(reply_ok)) ? reply_ok[_code] : _default)
178 #define REPLY_FAIL(_code, _default) ((_code < NUM_ELEMENTS(reply_fail)) ? reply_fail[_code] : _default)
179 
180  REQUEST_VERIFY(request);
181  fr_assert(request->packet->code > 0);
182  fr_assert(request->packet->code < NUM_ELEMENTS(reply_ok));
183 
184  switch (request->request_state) {
185  case REQUEST_INIT:
186  {
187  dhcpv6_packet_debug(request, request->packet, true);
188 
189  dhcpv6_reply_initialise(request);
190 
191  request->component = "dhcpv6";
192 
193  unlang = cf_section_find(request->server_cs, "recv", fr_dhcpv6_packet_types[request->packet->code]);
194  if (!unlang) {
195  RWDEBUG("Failed to find 'recv %s' section", fr_dhcpv6_packet_types[request->packet->code]);
196  request->reply->code = FR_DHCPV6_DO_NOT_RESPOND;
197  goto send_reply;
198  }
199 
200  RDEBUG("Running 'recv %s' from file %s", cf_section_name2(unlang), cf_filename(unlang));
201  if (unlang_interpret_push_section(request, unlang, RLM_MODULE_NOOP, UNLANG_TOP_FRAME) < 0) {
202  RETURN_MODULE_FAIL;
203  }
204 
205  request->request_state = REQUEST_RECV;
206  }
207  FALL_THROUGH;
208 
209  case REQUEST_RECV:
210  rcode = unlang_interpret(request);
211 
212  if (request->master_state == REQUEST_STOP_PROCESSING) {
213  *p_result = RLM_MODULE_HANDLED;
214  return UNLANG_ACTION_STOP_PROCESSING;
215  }
216 
217  if (rcode == RLM_MODULE_YIELD) return UNLANG_ACTION_YIELD;
218 
219  /*
220  * Allow the admin to explicitly set the reply
221  * type.
222  */
223  vp = fr_pair_find_by_da(&request->reply_pairs, attr_packet_type);
224  if (vp) {
225  request->reply->code = vp->vp_uint32;
226  } else switch (rcode) {
227  case RLM_MODULE_NOOP:
228  case RLM_MODULE_OK:
229  case RLM_MODULE_UPDATED:
230  request->reply->code = REPLY_OK(request->packet->code, FR_DHCPV6_DO_NOT_RESPOND);
231  break;
232 
233  default:
234  case RLM_MODULE_REJECT:
235  case RLM_MODULE_FAIL:
236  request->reply->code = REPLY_FAIL(request->packet->code, FR_DHCPV6_DO_NOT_RESPOND);
237  break;
238 
239  case RLM_MODULE_HANDLED:
240  if (!request->reply->code) request->reply->code = FR_DHCPV6_DO_NOT_RESPOND;
241  break;
242  }
243 
244  /*
245  * DHCP-Release / Decline doesn't send a reply, and doesn't run "send DHCP-Do-Not-Respond"
246  */
247  if (!request->reply->code) {
248  RETURN_MODULE_HANDLED;
249  }
250 
251  dv = fr_dict_enum_by_value(attr_packet_type, fr_box_uint32(request->reply->code));
252  unlang = NULL;
253  if (dv) unlang = cf_section_find(request->server_cs, "send", dv->name);
254 
255  if (!unlang) goto send_reply;
256 
257  rerun_nak:
258  RDEBUG("Running 'send %s' from file %s", cf_section_name2(unlang), cf_filename(unlang));
259  if (unlang_interpret_push_section(request, unlang, RLM_MODULE_NOOP, UNLANG_TOP_FRAME) < 0) {
260  RETURN_MODULE_FAIL;
261  }
262 
263  request->request_state = REQUEST_SEND;
264  FALL_THROUGH;
265 
266  case REQUEST_SEND:
267  rcode = unlang_interpret(request);
268 
269  if (request->master_state == REQUEST_STOP_PROCESSING) {
270  *p_result = RLM_MODULE_HANDLED;
271  return UNLANG_ACTION_STOP_PROCESSING;
272  }
273 
274  if (rcode == RLM_MODULE_YIELD) return UNLANG_ACTION_YIELD;
275 
276  switch (rcode) {
277  case RLM_MODULE_NOOP:
278  case RLM_MODULE_OK:
279  case RLM_MODULE_UPDATED:
280  case RLM_MODULE_HANDLED:
281  /* reply is already set */
282  break;
283 
284  default:
285  /*
286  * If we over-ride an ACK with a NAK, run
287  * the NAK section.
288  */
289  if (request->reply->code != FR_DHCPV6_DO_NOT_RESPOND) {
290  dv = fr_dict_enum_by_value(attr_packet_type, fr_box_uint32(request->reply->code));
291  RWDEBUG("Failed running 'send %s', trying 'send Do-Not-Respond'", dv->name);
292 
293  request->reply->code = FR_DHCPV6_DO_NOT_RESPOND;
294 
295  dv = fr_dict_enum_by_value(attr_packet_type, fr_box_uint32(request->reply->code));
296  unlang = NULL;
297  if (!dv) goto send_reply;
298 
299  unlang = cf_section_find(request->server_cs, "send", dv->name);
300  if (unlang) goto rerun_nak;
301 
302  RWDEBUG("Not running 'send %s' section as it does not exist", dv->name);
303  }
304  break;
305  }
306 
307  send_reply:
308  {
309  fr_pair_t *reply_packet_type;
310 
311  /*
312  * Add reply->packet-type in case we're
313  * being called via the `call {}` keyword.
314  */
315  MEM(fr_pair_update_by_da(request->reply, &reply_packet_type,
316  &request->reply->vps, attr_packet_type) >= 0);
317  reply_packet_type->vp_uint32 = request->reply->code;
318 
319  /*
320  * Check for "do not respond".
321  */
322  if (request->reply->code == FR_DHCPV6_DO_NOT_RESPOND) {
323  RDEBUG("Not sending reply to client");
324  RETURN_MODULE_HANDLED;
325  }
326 
327  if (RDEBUG_ENABLED) dhcpv6_packet_debug(request, request->reply, false);
328  }
329  break;
330 
331  default:
332  RETURN_MODULE_FAIL;
333  }
334 
335  RETURN_MODULE_OK;
336 }
