98 /*
99  * We have an array of workers, so we can index the workers in
100  * O(1) time. remove the heap of "workers ordered by CPU time"
101  * when we send a packet to a worker, just update the predicted
102  * CPU time in place. when we receive a reply from a worker,
103  * just update the predicted CPU time in place.
104  *
105  * when we need to choose a worker, pick 2 at random, and then
106  * choose the one with the lowe cpu time. For background, see
107  * "Power of Two-Choices" and

109  * https://www.eecs.harvard.edu/~michaelm/postscripts/tpds2001.pdf

428 /** Callback which handles a message being received on the network side.
429  *
430  * @param[in] ctx the network
431  * @param[in] ch the channel that the message is on.
432  * @param[in] cd the message (if any) to start with
433  */
434 static void fr_network_recv_reply(void *ctx, fr_channel_t *ch, fr_channel_data_t *cd)
435 {
436  fr_network_t *nr = ctx;
437  fr_network_worker_t *worker;
438 
439  cd->channel.ch = ch;
440 
441  /*
442  * Update stats for the worker.
443  */
444  worker = fr_channel_requestor_uctx_get(ch);
445  worker->stats.out++;
446  worker->cpu_time = cd->reply.cpu_time;
447  if (!worker->predicted) {
448  worker->predicted = cd->reply.processing_time;
449  } else {
450  worker->predicted = RTT(worker->predicted, cd->reply.processing_time);
451  }
452 
453  /*
454  * Unblock the worker.
455  */
456  if (worker->blocked) {
457  worker->blocked = false;
458  nr->num_blocked--;
459  fr_network_unsuspend(nr);
460  }
461 
462  /*
463  * Ensure that heap insert works.
464  */
465  cd->channel.heap_id = -1;
466  if (fr_heap_insert(nr->replies, cd) < 0) {
467  fr_message_done(&cd->m);
468  fr_assert(0 == 1);
469  }
470 }

472 /** Handle a network control message callback for a channel
473  *
474  * This is called from the event loop when we get a notification
475  * from the event signalling pipe.
476  *
477  * @param[in] ctx the network
478  * @param[in] data the message
479  * @param[in] data_size size of the data
480  * @param[in] now the current time
481  */
482 static void fr_network_channel_callback(void *ctx, void const *data, size_t data_size, fr_time_t now)
483 {
484  fr_channel_event_t ce;
485  fr_channel_t *ch;
486  fr_network_t *nr = ctx;
487 
488  ce = fr_channel_service_message(now, &ch, data, data_size);
489  DEBUG3("Channel %s",
490  fr_table_str_by_value(channel_signals, ce, "<INVALID>"));
491  switch (ce) {
492  case FR_CHANNEL_ERROR:
493  return;
494 
495  case FR_CHANNEL_EMPTY:
496  return;
497 
498  case FR_CHANNEL_NOOP:
499  break;
500 
501  case FR_CHANNEL_DATA_READY_REQUESTOR:
502  fr_assert(ch != NULL);
503  while (fr_channel_recv_reply(ch));
504  break;
505 
506  case FR_CHANNEL_DATA_READY_RESPONDER:
507  fr_assert(0 == 1);
508  break;
509 
510  case FR_CHANNEL_OPEN:
511  fr_assert(0 == 1);
512  break;
513 
514  case FR_CHANNEL_CLOSE:
515  {
516  fr_network_worker_t *w = talloc_get_type_abort(fr_channel_requestor_uctx_get(ch),
517  fr_network_worker_t);
518  int i;
519 
520  /*
521  * Remove this worker from the array
522  */
523  for (i = 0; i < nr->num_workers; i++) {
524  DEBUG3("Worker acked our close request");
525  if (nr->workers[i] == w) {
526  nr->workers[i] = NULL;
527 
528  if (i == (nr->num_workers - 1)) break;
529 
530  /*
531  * Close the hole...
532  */
533  memcpy(&nr->workers[i], &nr->workers[i + 1], ((nr->num_workers - i) - 1));
534  break;
535  }
536  }
537  nr->num_workers--;
538  }
539  break;
540  }
541 }

543 /** Send a message on the "best" channel.
544  *
545  * @param nr the network
546  * @param cd the message we've received
547  */
548 static int fr_network_send_request(fr_network_t *nr, fr_channel_data_t *cd)
549 {
550  fr_network_worker_t *worker;
551 
552  (void) talloc_get_type_abort(nr, fr_network_t);
553 
554 retry:
555  if (nr->num_workers == 1) {
556  worker = nr->workers[0];
557  if (worker->blocked) {
558  RATE_LIMIT_GLOBAL(ERROR, "Failed sending packet to worker - "
559  "In single-threaded mode and worker is blocked");
560  drop:
561  worker->stats.dropped++;
562  done:
563  fr_message_done(&cd->m);
564  return -1;
565  }
566 
567  } else if (nr->num_blocked == 0) {
568  uint32_t one, two;
569 
570  one = fr_rand() % nr->num_workers;
571  do {
572  two = fr_rand() % nr->num_workers;
573  } while (two == one);
574 
575  if (nr->workers[one]->cpu_time < nr->workers[two]->cpu_time) {
576  worker = nr->workers[one];
577  } else {
578  worker = nr->workers[two];
579  }
580  } else {
581  int i;
582  fr_time_t cpu_time = ~((fr_time_t) 0);
583  fr_network_worker_t *found = NULL;
584 
585  /*
586  * Some workers are blocked. Pick an active
587  * worker with low CPU time.
588  */
589  for (i = 0; i < nr->num_workers; i++) {
590  worker = nr->workers[i];
591  if (worker->blocked) continue;
592 
593  if (worker->cpu_time < cpu_time) {
594  found = worker;
595  }
596  }
597 
598  if (!found) {
599  RATE_LIMIT_GLOBAL(PERROR, "Failed sending packet to worker - Couldn't find active worker, "
600  "%u/%u workers are blocked", nr->num_blocked, nr->num_workers);
601  goto done;
602  }
603 
604  worker = found;
605  }
606 
607  (void) talloc_get_type_abort(worker, fr_network_worker_t);
608 
609  /*
610  * Too many outstanding packets for this worker. Drop
611  * the request.
612  *
613  * @todo - pick another worker? Or maybe keep a
614  * local/temporary set of blacklisted workers.
615  */
616  fr_assert(worker->stats.in >= worker->stats.out);
617  if (nr->config.max_outstanding &&
618  ((worker->stats.in - worker->stats.out) >= nr->config.max_outstanding)) {
619  RATE_LIMIT_GLOBAL(PERROR, "max_outstanding reached - dropping packet");
620  goto drop;
621  }
622 
623  /*
624  * Send the message to the channel. If we fail, drop the
625  * packet. The only reason for failure is that the
626  * worker isn't servicing it's input queue. When that
627  * happens, we have no idea what to do, and the whole
628  * thing falls over.
629  */
630  if (fr_channel_send_request(worker->channel, cd) < 0) {
631  worker->stats.dropped++;
632  worker->blocked = true;
633  nr->num_blocked++;
634 
635  RATE_LIMIT_GLOBAL(PERROR, "Failed sending packet to worker - %u/%u workers are blocked",
636  nr->num_blocked, nr->num_workers);
637 
638  if (nr->num_blocked == nr->num_workers) {
639  fr_network_suspend(nr);
640  fr_message_done(&cd->m);
641  return -1;
642  }
643  goto retry;
644  }
645 
646  worker->stats.in++;
647 
648  /*
649  * We're projecting that the worker will use more CPU
650  * time to process this request. The CPU time will be
651  * updated with a more accurate number when we receive a
652  * reply from this channel.
653  */
654  worker->cpu_time += worker->predicted;
655 
656  return 0;
657 }

690  * @param[in] el the event list.
691  * @param[in] sockfd the socket which is ready to read.
692  * @param[in] flags from kevent.
693  * @param[in] ctx the network socket context.
694  */
695 static void fr_network_read(UNUSED fr_event_list_t *el, int sockfd, UNUSED int flags, void *ctx)
696 {
697  int num_messages = 0;
698  fr_network_socket_t *s = ctx;
699  fr_network_t *nr = s->nr;
700  ssize_t data_size;
701  fr_channel_data_t *cd, *next;
702 #ifndef NDEBUG
703  fr_time_t now;
704 #endif
705 
706  if (!fr_cond_assert_msg(s->listen->fd == sockfd, "Expected listen->fd (%u) to be equal event fd (%u)",
707  s->listen->fd, sockfd)) return;
708 
709  DEBUG3("Reading data from FD %u", sockfd);
710 
711  if (!s->cd) {
712  cd = (fr_channel_data_t *) fr_message_reserve(s->ms, s->listen->default_message_size);
713  if (!cd) {
714  ERROR("Failed allocating message size %zd! - Closing socket",
715  s->listen->default_message_size);
716  fr_network_socket_dead(nr, s);
717  return;
718  }
719  } else {
720  cd = s->cd;
721  }
722 
723  fr_assert(cd->m.data != NULL);
724 
725 next_message:
726  /*
727  * Poll this socket, but not too often. We have to go
728  * service other sockets, too.
729  */
730  if (num_messages > 16) {
731  s->cd = cd;
732  return;
733  }
734 
735  cd->request.is_dup = false;
736  cd->priority = PRIORITY_NORMAL;
737 
738  /*
739  * Read data from the network.
740  *
741  * Return of 0 means "no data", which is fine for UDP.
742  * For TCP, if an underlying read() on the TCP socket
743  * returns 0, (which signals that the FD is no longer
744  * usable) this function should return -1, so that the
745  * network side knows that it needs to close the
746  * connection.
747  */
748  data_size = s->listen->app_io->read(s->listen, &cd->packet_ctx, &cd->request.recv_time,
749  cd->m.data, cd->m.rb_size, &s->leftover, &cd->priority, &cd->request.is_dup);
750  if (data_size == 0) {
751  /*
752  * Cache the message for later. This is
753  * important for stream sockets, which can do
754  * partial reads into the current buffer. We
755  * need to be able to give the same buffer back
756  * to the stream socket for subsequent reads.
757  *
758  * Since we have a message set for each
759  * fr_io_socket_t, no "head of line"
760  * blocking issues can happen for stream sockets.
761  */
762  s->cd = cd;
763  return;
764  }
765 
766  /*
767  * Error: close the connection, and remove the fr_listen_t
768  */
769  if (data_size < 0) {
770 // fr_log(nr->log, L_DBG_ERR, "error from transport read on socket %d", sockfd);
771  fr_network_socket_dead(nr, s);
772  return;
773  }
774  s->cd = NULL;
775 
776  DEBUG3("Read %zd byte(s) from FD %u", data_size, sockfd);
777  nr->stats.in++;
778  s->stats.in++;
779 
780  /*
781  * Initialize the rest of the fields of the channel data.
782  *
783  * We always use "now" as the time of the message, as the
784  * packet MAY be a duplicate packet magically resurrected
785  * from the past. i.e. If the read routines are doing
786  * dedup, then they notice that the packet is a
787  * duplicate. In that case, they send over a copy of the
788  * packet, BUT with the original timestamp. This
789  * information tells the worker that the packet is a
790  * duplicate.
791  */
792  cd->m.when = fr_time();
793 #ifndef NDEBUG
794  now = cd->m.when;
795 #endif
796  cd->listen = s->listen;
797 
798  /*
799  * Nothing in the buffer yet. Allocate room for one
800  * packet.
801  */
802  if ((cd->m.data_size == 0) && (!s->leftover)) {
803 
804  (void) fr_message_alloc(s->ms, &cd->m, data_size);
805  next = NULL;
806 
807  } else {
808  /*
809  * There are leftover bytes in the buffer, feed
810  * them to the next round of reading.
811  */
812  next = (fr_channel_data_t *) fr_message_alloc_reserve(s->ms, &cd->m, data_size, s->leftover,
813  s->listen->default_message_size);
814  if (!next) {
815  PERROR("Failed reserving partial packet.");
816  // @todo - probably close the socket...
817  fr_assert(0 == 1);
818  }
819  }
820 
821  /*
822  * Ensure this hasn't been somehow corrupted during
823  * ring buffer allocation.
824  */
825  fr_assert(cd->m.when == now);
826 
827  if (fr_network_send_request(nr, cd) < 0) {
828  fr_message_done(&cd->m);
829  nr->stats.dropped++;
830  s->stats.dropped++;
831 
832  } else {
833  /*
834  * One more packet sent to a worker.
835  */
836  s->outstanding++;
837  }
838 
839  /*
840  * If there is a next message, go read it from the buffer.
841  *
842  * @todo - note that this calls read(), even if the
843  * app_io has paused the reader. We likely want to be
844  * able to check that, too. We might just remove this
845  * "goto"...
846  */
847  if (next) {
848  cd = next;
849  num_messages++;
850  goto next_message;
851  }
852 }

900 /** Write packets to the network.
901  *
902  * @param el the event list
903  * @param sockfd the socket which is ready to write
904  * @param flags returned by kevent.
905  * @param ctx the network socket context.
906  */
907 static void fr_network_write(UNUSED fr_event_list_t *el, UNUSED int sockfd, UNUSED int flags, void *ctx)
908 {
909  fr_network_socket_t *s = ctx;
910  fr_listen_t *li = s->listen;
911  fr_network_t *nr = s->nr;
912  fr_channel_data_t *cd;
913 
914  (void) talloc_get_type_abort(nr, fr_network_t);
915 
916  /*
917  * Start with the currently pending message, and then
918  * work through the priority heap.
919  */
920  if (s->pending) {
921  cd = s->pending;
922  s->pending = NULL;
923 
924  } else {
925  cd = fr_heap_pop(s->waiting);
926  }
927 
928  while (cd != NULL) {
929  int rcode;
930 
931  fr_assert(li == cd->listen);
932  rcode = li->app_io->write(li, cd->packet_ctx,
933  cd->reply.request_time,
934  cd->m.data, cd->m.data_size, s->written);
935 
936  /*
937  * As a special case, allow write() to return
938  * "0", which means "close the socket".
939  */
940  if (rcode == 0) goto dead;
941 
942  /*
943  * Or we have a write error.
944  */
945  if (rcode < 0) {
946  /*
947  * Stop processing the heap, and set the
948  * pending message to the current one.
949  */
950  if (errno == EWOULDBLOCK) {
951  save_pending:
952  fr_assert(!s->pending);
953 
954  if (cd->m.status != FR_MESSAGE_LOCALIZED) {
955  fr_message_t *lm;
956 
957  lm = fr_message_localize(s, &cd->m, sizeof(*cd));
958  fr_message_done(&cd->m);
959 
960  if (!lm) {
961  ERROR("Failed saving pending packet");
962  goto dead;
963  }
964 
965  cd = (fr_channel_data_t *) lm;
966  }
967 
968  if (!s->blocked) {
969  if (fr_event_fd_insert(nr, nr->el, s->listen->fd,
970  fr_network_read,
971  fr_network_write,
972  fr_network_error,
973  s) < 0) {
974  PERROR("Failed adding write callback to event loop");
975  goto dead;
976  }
977 
978  s->blocked = true;
979  }
980 
981  s->pending = cd;
982  return;
983  }
984 
985  /*
986  * As a special hack, check for something
987  * that will never be returned from a
988  * real write() routine. Which then
989  * signals to us that we have to close
990  * the socket, but NOT complain about it.
991  */
992  if (errno == ECONNREFUSED) goto dead;
993 
994  PERROR("Failed writing to socket %s", s->listen->name);
995  if (li->app_io->error) li->app_io->error(li);
996 
997  dead:
998  fr_message_done(&cd->m);
999  fr_network_socket_dead(nr, s);
1000  return;
1001  }
1002 
1003  /*
1004  * If we've done a partial write, localize the message and continue.
1005  */
1006  if ((size_t) rcode < cd->m.data_size) {
1007  s->written = rcode;
1008  goto save_pending;
1009  }
1010 
1011  s->written = 0;
1012 
1013  /*
1014  * Reset for the next message.
1015  */
1016  fr_message_done(&cd->m);
1017  nr->stats.out++;
1018  s->stats.out++;
1019 
1020  /*
1021  * Grab the net entry.
1022  */
1023  cd = fr_heap_pop(s->waiting);
1024  }
1025 
1026  /*
1027  * We've successfully written all of the packets. Remove
1028  * the write callback.
1029  */
1030  if (fr_event_fd_insert(nr, nr->el, s->listen->fd,
1031  fr_network_read,
1032  NULL,
1033  fr_network_error,
1034  s) < 0) {
1035  PERROR("Failed removing write callback from event loop");
1036  fr_network_socket_dead(nr, s);
1037  }
1038 
1039  s->blocked = false;
1040 }

1080  * @param[in] now the current time

1086  fr_app_io_t const *app_io;
1087  size_t size;
1088  int num_messages;

1322 /** Handle replies after all FD and timer events have been serviced
1323  *
1324  * @param el the event loop
1325  * @param now the current time (mostly)
1326  * @param uctx the fr_network_t
1327  */
1328 static void fr_network_post_event(UNUSED fr_event_list_t *el, UNUSED fr_time_t now, void *uctx)
1329 {
1330  fr_channel_data_t *cd;
1331  fr_network_t *nr = talloc_get_type_abort(uctx, fr_network_t);
1332 
1333  /*
1334  * Pull the replies off of our global heap, and try to
1335  * push them to the individual sockets.
1336  */
1337  while ((cd = fr_heap_pop(nr->replies)) != NULL) {
1338  fr_listen_t *li;
1339  fr_network_socket_t *s;
1340 
1341  li = cd->listen;
1342 
1343  /*
1344  * @todo - cache this somewhere so we don't need
1345  * to do an rbtree lookup for every packet.
1346  */
1347  s = rbtree_finddata(nr->sockets, &(fr_network_socket_t){ .listen = li });
1348 
1349  /*
1350  * This shouldn't happen, but be safe...
1351  */
1352  if (!s) {
1353  fr_message_done(&cd->m);
1354  continue;
1355  }
1356 
1357  if (cd->m.status != FR_MESSAGE_LOCALIZED) {
1358  fr_assert(s->outstanding > 0);
1359  s->outstanding--;
1360  }
1361 
1362  /*
1363  * Just mark the message done, and skip it.
1364  */
1365  if (s->dead) {
1366  fr_message_done(&cd->m);
1367 
1368  /*
1369  * No more packets, it's safe to delete
