126 static int ipaccess_a_fd_cb(struct osmo_fd *bfd)
127 {
128  int error;
129  struct ipaccess_head *hh;
130  struct msc_connection *fw;
131  struct msgb *msg;
132 
133  fw = bfd->data;
134  msg = ipaccess_read_msg(bfd, &error);
135 
136  if (!msg) {
137  if (error == 0)
138  fprintf(stderr, "The connection to the MSC was lost, exiting\n");
139  else
140  fprintf(stderr, "Error in the IPA stream.\n");
141 
142  msc_close_connection(fw);
143  return -1;
144  }
145 
146  LOGP(DMSC, LOGL_DEBUG, "From MSC: %s proto: %d\n", osmo_hexdump(msg->data, msg->len), msg->l2h[0]);
147 
148  /* handle base message handling */
149  hh = (struct ipaccess_head *) msg->data;
150  ipaccess_rcvmsg_base(msg, bfd);
151 
152  /* initialize the networking. This includes sending a GSM08.08 message */
153  if (hh->proto == IPAC_PROTO_IPACCESS) {
154  if (fw->first_contact) {
155  LOGP(DMSC, LOGL_NOTICE, "Connected to MSC. Sending reset.\n");
156  osmo_timer_del(&fw->msc_timeout);
157  fw->first_contact = 0;
158  fw->msc_link_down = 0;
159  ss7_application_msc_up(fw->app);
160  msc_send_reset(fw);
161  }
162  if (msg->l2h[0] == IPAC_MSGT_ID_GET && fw->token) {
163  msc_send_id_response(fw);
164  } else if (msg->l2h[0] == IPAC_MSGT_PONG) {
165  osmo_timer_del(&fw->pong_timeout);
166  } else if (msg->l2h[0] == IPAC_MSGT_ID_RESP) {
167  msc_handle_id_response(fw, msg);
168  }
169 
170  msgb_free(msg);
171  return 0;
172  }
173 
174  if (fw->mode == MSC_MODE_SERVER && !fw->auth) {
175  LOGP(DMSC, LOGL_ERROR,
176  "Ignoring non ipa message for unauth user.\n");
177  msgb_free(msg);
178  return -1;
179  }
180 
181  if (hh->proto == IPAC_PROTO_SCCP) {
182  msc_dispatch_sccp(fw, msg);
183  } else if (hh->proto == NAT_MUX) {
184  abort();
185  /*
186  msg = mgcp_patch(fw->app, msg);
187  mgcp_forward(&fw->mgcp_agent, msg->l2h, msgb_l2len(msg));
188  */
189  } else {
190  LOGP(DMSC, LOGL_ERROR, "Unknown IPA proto 0x%x\n", hh->proto);
191  }
192 
193  msgb_free(msg);
194  return 0;
195 }

210 static int msc_connection_connect(struct osmo_fd *fd, unsigned int what)
211 {
212  int rc;
213  int val;
214  socklen_t len = sizeof(val);
215  struct msc_connection *fw;
216 
217  fw = fd->data;
218 
219  if (fd != &fw->msc_connection.bfd) {
220  LOGP(DMSC, LOGL_ERROR, "This is only working with the MSC connection.\n");
221  return -1;
222  }
223 
224  if ((what & BSC_FD_WRITE) == 0)
225  return -1;
226 
227  /* check the socket state */
228  rc = getsockopt(fd->fd, SOL_SOCKET, SO_ERROR, &val, &len);
229  if (rc != 0) {
230  LOGP(DMSC, LOGL_ERROR, "getsockopt for the MSC socket failed.\n");
231  goto error;
232  }
233  if (val != 0) {
234  LOGP(DMSC, LOGL_ERROR, "Not connected to the MSC.\n");
235  goto error;
236  }
237 
238 
239  /* go to full operation */
240  fd->cb = osmo_wqueue_bfd_cb;
241  fd->when = BSC_FD_READ;
242  if (!llist_empty(&fw->msc_connection.msg_queue))
243  fd->when |= BSC_FD_WRITE;
244  return 0;
245 
246 error:
247  msc_close_connection(fw);
248  return -1;
249 }

251 static int setnonblocking(struct osmo_fd *fd)
252 {
253  int flags;
254 
255  flags = fcntl(fd->fd, F_GETFL);
256  if (flags < 0) {
257  perror("fcntl get failed");
258  close(fd->fd);
259  fd->fd = -1;
260  return -1;
261  }
262 
263  flags |= O_NONBLOCK;
264  flags = fcntl(fd->fd, F_SETFL, flags);
265  if (flags < 0) {
266  perror("fcntl get failed");
267  close(fd->fd);
268  fd->fd = -1;
269  return -1;
270  }
271 
272  return 0;

275 static int connect_to_msc(struct osmo_fd *fd, const char *ip, int port, int tos)
276 {
277  struct sockaddr_in sin;
278  int on = 1, ret;
279 
280  LOGP(DMSC, LOGL_NOTICE, "Attempting to connect MSC at %s:%d\n", ip, port);
281 
282  fd->fd = socket(AF_INET, SOCK_STREAM, IPPROTO_TCP);
283 
284  if (fd->fd < 0) {
285  perror("Creating TCP socket failed");
286  return fd->fd;
287  }
288 
289  /* make it non blocking */
290  if (setnonblocking(fd) != 0)
291  return -1;
292 
293  memset(&sin, 0, sizeof(sin));
294  sin.sin_family = AF_INET;
295  sin.sin_port = htons(port);
296  inet_aton(ip, &sin.sin_addr);
297 
298  setsockopt(fd->fd, SOL_SOCKET, SO_REUSEADDR, &on, sizeof(on));
299  ret = setsockopt(fd->fd, IPPROTO_TCP, TCP_NODELAY, &on, sizeof(on));
300  if (ret != 0)
301  LOGP(DMSC, LOGL_ERROR, "Failed to set TCP_NODELAY: %s\n", strerror(errno));
302  ret = setsockopt(fd->fd, IPPROTO_IP, IP_TOS, &tos, sizeof(tos));
303  if (ret != 0)
304  LOGP(DMSC, LOGL_ERROR, "Failed to set IP_TOS: %s\n", strerror(errno));
305 
306  ret = connect(fd->fd, (struct sockaddr *) &sin, sizeof(sin));
307 
308  if (ret == -1 && errno == EINPROGRESS) {
309  LOGP(DMSC, LOGL_ERROR, "MSC Connection in progress\n");
310  fd->when = BSC_FD_WRITE;
311  fd->cb = msc_connection_connect;
312  } else if (ret < 0) {
313  perror("Connection failed");
314  close(fd->fd);
315  fd->fd = -1;
316  return ret;
317  } else {
318  fd->when = BSC_FD_READ;
319  fd->cb = osmo_wqueue_bfd_cb;
320  }
321 
322  ret = osmo_fd_register(fd);
323  if (ret < 0) {
324  perror("Registering the fd failed");
325  close(fd->fd);
326  fd->fd = -1;
327  return ret;
328  }
329 
330  return ret;
331 }

578 static void msc_handle_id_response(struct msc_connection *msc, struct msgb *msg)
579 {
580  unsigned int len;
581  const char *token;
582 
583  /* only for the server */
584  if (msc->mode != MSC_MODE_SERVER) {
585  LOGP(DMSC, LOGL_ERROR, "Unexpected ID response for client.\n");
586  return;
587  }
588 
589  if (!msc->token) {
590  LOGP(DMSC, LOGL_ERROR, "No token defined. Giving up.\n");
591  goto clean;
592  }
593 
594  if (msgb_l2len(msg) < 4) {
595  LOGP(DMSC, LOGL_ERROR, "Too short message...%u\n",
596  msgb_l2len(msg));
597  goto clean;
598  }
599 
600  /* in lack of ipaccess_idtag_parse we have a very basic method */
601  if (msg->l2h[3] != IPAC_IDTAG_UNITNAME) {
602  LOGP(DMSC, LOGL_ERROR, "Expected unitname tag got %d\n",
603  msg->l2h[3]);
604  goto clean;
605  }
606 
607  token = (const char *) &msg->l2h[4];
608  len = msgb_l2len(msg) - 4;
609 
610  if (len != strlen(msc->token)) {
611  LOGP(DMSC, LOGL_ERROR, "Wrong length %u vs. %zu\n", len, strlen(msc->token));
612  goto clean;
613  }
614 
615  if (memcmp(msc->token, token, len) != 0) {
616  LOGP(DMSC, LOGL_ERROR, "Token has the wrong size.\n");
617  goto clean;
618  }
619 
620  LOGP(DMSC, LOGL_NOTICE, "Authenticated the connection.\n");
621  msc->auth = 1;
622  ss7_application_msc_up(msc->app);
623  return;
624 clean:
625  msc_close_connection(msc);
626 }

628 static int msc_conn_accept(struct osmo_fd *bsc_fd, unsigned int what)
629 {
630  struct sockaddr_in addr;
631  socklen_t len = sizeof(addr);
632  struct msc_connection *msc = bsc_fd->data;
633  int ret;
634 
635  LOGP(DMSC, LOGL_NOTICE, "Going to accept a connection.\n");
636 
637  ret = accept(bsc_fd->fd, (struct sockaddr *) &addr, &len);
638  if (ret < 0) {
639  LOGP(DMSC, LOGL_ERROR, "Accept failed with fd(%d) errno(%d)\n",
640  ret, errno);
641  return -1;
642  }
643 
644  /*
645  * Close the previous/current connection.
646  * TODO: switch only once we know it is a valid connection
647  */
648  msc_close_connection(msc);
649 
650  /* re-set the internal state */
651  msc->auth = 0;
652 
653  /* adopt the connection */
654  msc->msc_connection.bfd.fd = ret;
655  msc->msc_connection.bfd.when = BSC_FD_READ;
656  ret = osmo_fd_register(&msc->msc_connection.bfd);
657  if (ret < 0) {
658  LOGP(DMSC, LOGL_ERROR, "Failed to register fd.\n");
659  close(msc->msc_connection.bfd.fd);
660  msc->msc_connection.bfd.fd = -1;
661  return -1;
662  }
