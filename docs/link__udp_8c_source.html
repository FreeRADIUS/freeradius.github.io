51 static int udp_write_cb(struct osmo_fd *fd, struct msgb *msg)
52 {
53  struct mtp_udp_data *data;
54  struct mtp_udp_link *link;
55  int rc;
56 
57  data = fd->data;
58  link = find_link(data, msg->cb[0]);
59  if (!link) {
60  LOGP(DINP, LOGL_ERROR, "Failed to find link with %lu\n", msg->cb[0]);
61  return -1;
62  }
63 
64  LOGP(DINP, LOGL_DEBUG, "Sending MSU: %s\n", osmo_hexdump(msg->data, msg->len));
65  mtp_handle_pcap(link->base, NET_OUT, msg->l2h, msgb_l2len(msg));
66 
67  /* the assumption is we have connected the socket to the remote */
68  rc = sendto(fd->fd, msg->data, msg->len, 0,
69  (struct sockaddr *) &link->remote, sizeof(link->remote));
70  if (rc != msg->len) {
71  LOGP(DINP, LOGL_ERROR, "Failed to write msg to socket: %d\n", rc);
72  return -1;
73  }
74 
75  return 0;
76 }

78 static int udp_read_cb(struct osmo_fd *fd)
79 {
80  struct mtp_udp_data *data;
81  struct mtp_udp_link *ulnk;
82  struct mtp_link *link;
83  struct udp_data_hdr *hdr;
84  struct msgb *msg;
85  int rc;
86  unsigned int length;
87 
88  msg = msgb_alloc_headroom(4096, 128, "UDP datagram");
89  if (!msg) {
90  LOGP(DINP, LOGL_ERROR, "Failed to allocate memory.\n");
91  return -1;
92  }
93 
94 
95  data = (struct mtp_udp_data *) fd->data;
96  rc = read(fd->fd, msg->data, 2096);
97  if (rc < sizeof(*hdr)) {
98  LOGP(DINP, LOGL_ERROR, "Failed to read at least size of the header: %d\n", rc);
99  rc = -1;
100  goto exit;
101  }
102 
103  hdr = (struct udp_data_hdr *) msgb_put(msg, sizeof(*hdr));
104  ulnk = find_link(data, ntohs(hdr->data_link_index));
105 
106  if (!ulnk) {
107  LOGP(DINP, LOGL_ERROR, "No link registered for %d\n",
108  ntohs(hdr->data_link_index));
109  goto exit;
110  }
111 
112  link = ulnk->base;
113  if (link->blocked) {
114  LOGP(DINP, LOGL_ERROR, "The link is blocked.\n");
115  rc = 0;
116  goto exit;
117  }
118 
119  if (hdr->data_type == UDP_DATA_RETR_COMPL || hdr->data_type == UDP_DATA_RETR_IMPOS) {
120  LOGP(DINP, LOGL_ERROR, "Link retrieval done on %d/%s of %d/%s.\n",
121  link->nr, link->name, link->set->nr, link->set->name);
122  mtp_link_failure(link);
123  goto exit;
124  } else if (hdr->data_type == UDP_DATA_LINK_UP) {
125  LOGP(DINP, LOGL_NOTICE, "Link of %d/%s of %d/%s is up.\n",
126  link->nr, link->name, link->set->nr, link->set->name);
127  mtp_link_up(link);
128  goto exit;
129  } else if (hdr->data_type == UDP_DATA_LINK_DOWN) {
130  LOGP(DINP, LOGL_NOTICE, "Link of %d/%s of %d/%s is down.\n",
131  link->nr, link->name, link->set->nr, link->set->name);
132  mtp_link_failure(link);
133  goto exit;
134  } else if (hdr->data_type > UDP_DATA_MSU_PRIO_3) {
135  LOGP(DINP, LOGL_ERROR, "Link failue on %d/%s of %d/%s.\n",
136  link->nr, link->name, link->set->nr, link->set->name);
137  mtp_link_failure(link);
138  goto exit;
139  }
140 
141  /* throw away data as the link is down */
142  if (link->set->available == 0) {
143  LOGP(DINP, LOGL_ERROR, "Link %d/%s of %d/%s is down. Not forwarding.\n",
144  link->nr, link->name, link->set->nr, link->set->name);
145  rc = 0;
146  goto exit;
147  }
148 
149  length = ntohl(hdr->data_length);
150  if (length + sizeof(*hdr) > (unsigned int) rc) {
151  LOGP(DINP, LOGL_ERROR,
152  "The MSU payload does not fit: %u + %zu > %d on link %d/%s of %d/%s.\n",
153  length, sizeof(*hdr), rc,
154  link->nr, link->name, link->set->nr, link->set->name);
155  rc = 0;
156  rc = -1;
157  goto exit;
158  }
159 
160  msg->l2h = msgb_put(msg, length);
161 
162  LOGP(DINP, LOGL_DEBUG, "MSU data on link %d/%s of %d/%s data %s.\n",
163  link->nr, link->name, link->set->nr, link->set->name,
164  osmo_hexdump(msg->data, msg->len));
165  mtp_handle_pcap(link, NET_IN, msg->l2h, msgb_l2len(msg));
166  mtp_link_set_data(link, msg);
167 
168 exit:
169  msgb_free(msg);
170  return rc;
171 }

201 static int udp_link_write(struct mtp_link *link, struct msgb *msg)
202 {
203  struct mtp_udp_link *ulnk;
204  struct udp_data_hdr *hdr;
205 
206  ulnk = (struct mtp_udp_link *) link->data;
207 
208  hdr = (struct udp_data_hdr *) msgb_push(msg, sizeof(*hdr));
209  hdr->format_type = UDP_FORMAT_SIMPLE_UDP;
210  hdr->data_type = UDP_DATA_MSU_PRIO_0;
211  hdr->data_link_index = htons(ulnk->link_index);
212  hdr->user_context = 0;
213  hdr->data_length = htonl(msgb_l2len(msg));
214 
215  msg->cb[0] = ulnk->link_index;
216 
217  if (osmo_wqueue_enqueue(&ulnk->data->write_queue, msg) != 0) {
218  LOGP(DINP, LOGL_ERROR, "Failed to enqueue msg on link %d/%s of %d/%s.\n",
219  link->nr, link->name, link->set->nr, link->set->name);
220  rate_ctr_inc(&link->ctrg->ctr[MTP_LNK_DRP]);
221  rate_ctr_inc(&link->set->ctrg->ctr[MTP_LSET_TOTA_DRP_MSG]);
222  msgb_free(msg);
223  return -1;
224  }
225 
226  return 0;
227 }

261 int link_global_bind(struct mtp_udp_data *data, int src_port)
262 {
263  struct sockaddr_in addr;
264  int fd;
265  int on;
266 
267  data->write_queue.bfd.fd = fd = socket(AF_INET, SOCK_DGRAM, 0);
268  if (fd < 0) {
269  LOGP(DINP, LOGL_ERROR, "Failed to create UDP socket.\n");
270  return -1;
271  }
272 
273  on = 1;
274  setsockopt(fd, SOL_SOCKET, SO_REUSEADDR, &on, sizeof(on));
275 
276  memset(&addr, 0, sizeof(addr));
277  addr.sin_family = AF_INET;
278  addr.sin_port = htons(src_port);
279  addr.sin_addr.s_addr = INADDR_ANY;
280 
281  if (bind(fd, (struct sockaddr *) &addr, sizeof(addr)) < 0) {
282  perror("Failed to bind UDP socket");
283  close(fd);
284  return -1;
285  }
286 
287  /* now connect the socket to the remote */
288  if (osmo_fd_register(&data->write_queue.bfd) != 0) {
289  LOGP(DINP, LOGL_ERROR, "Failed to register BFD.\n");
290  close(fd);
291  return -1;
292  }
293 
294  data->snmp_poll.data = data;
295  data->snmp_poll.cb = snmp_poll;
296  snmp_poll(data);
297 
298  return 0;
299 }

301 void snmp_mtp_callback(struct snmp_mtp_session *session,
302  int area, int res, int link_id)
303 {
304  struct mtp_udp_link *ulink;
305  struct mtp_link *link;
306 
307  ulink = session->data;
308  if (!ulink) {
309  LOGP(DINP, LOGL_ERROR, "Failed to find link_id %d\n", link_id);
310  return;
311  }
312 
313  link = ulink->base;
314 
315  if (res == SNMP_STATUS_TIMEOUT && !link->blocked) {
316  LOGP(DINP, LOGL_ERROR,
317  "Failed to restart link %d/%s of linkset %d/%s\n",
318  link->nr, link->name, link->set->nr, link->set->name);
319  udp_link_reset(link);
320  return;
321  }
322 
323  switch (area) {
324  case SNMP_LINK_UP:
325  break;
326  case SNMP_LINK_DOWN:
327  mtp_link_down(link);
328 
329  /*
330  * restart the link in 90 seconds...
331  * to force a timeout on the BSC
332  */
333  if (!link->blocked) {
334  link->link_activate.cb = do_start;
335  link->link_activate.data = ulink;
336  osmo_timer_schedule(&link->link_activate, ulink->reset_timeout, 0);
337  LOGP(DINP, LOGL_NOTICE,
338  "Will bring up link %d/%s of linkset %d/%s in %d seconds.\n",
339  link->nr, link->name,
340  link->set->nr, link->set->name,
341  ulink->reset_timeout);
342  }
343  break;
344  default:
345  LOGP(DINP, LOGL_ERROR,
346  "Unknown event %d on %d/%s of linkset %d/%s.\n",
347  area, link->nr, link->name, link->set->nr, link->set->name);
348  }
