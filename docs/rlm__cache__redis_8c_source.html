124 /** Locate a cache entry in redis
125  *
126  * @copydetails cache_entry_find_t
127  */
128 static cache_status_t cache_entry_find(rlm_cache_entry_t **out,
129  UNUSED rlm_cache_config_t const *config, void *instance,
130  request_t *request, UNUSED void *handle, uint8_t const *key, size_t key_len)
131 {
132  rlm_cache_redis_t *driver = instance;
133  size_t i;
134 
135  fr_redis_cluster_state_t state;
136  fr_redis_conn_t *conn;
137  fr_redis_rcode_t status;
138  redisReply *reply = NULL;
139  int s_ret;
140 
141  map_t *head = NULL, **last = &head;
142 #ifdef HAVE_TALLOC_ZERO_POOLED_OBJECT
143  size_t pool_size = 0;
144 #endif
145  rlm_cache_entry_t *c;
146 
147  for (s_ret = fr_redis_cluster_state_init(&state, &conn, driver->cluster, request, key, key_len, false);
148  s_ret == REDIS_RCODE_TRY_AGAIN; /* Continue */
149  s_ret = fr_redis_cluster_state_next(&state, &conn, driver->cluster, request, status, &reply)) {
150  /*
151  * Grab all the data for this hash, should return an array
152  * of alternating keys/values which we then convert into maps.
153  */
154  RDEBUG3("LRANGE %pV 0 -1", fr_box_strvalue_len((char const *)key, key_len));
155  reply = redisCommand(conn->handle, "LRANGE %b 0 -1", key, key_len);
156  status = fr_redis_command_status(conn, reply);
157  }
158  if (s_ret != REDIS_RCODE_SUCCESS) {
159  RERROR("Failed retrieving entry for key \"%pV\"", fr_box_strvalue_len((char const *)key, key_len));
160 
161  error:
162  fr_redis_reply_free(&reply);
163  return CACHE_ERROR;
164  }
165 
166  if (!fr_cond_assert(reply)) goto error;
167 
168  if (reply->type != REDIS_REPLY_ARRAY) {
169  REDEBUG("Bad result type, expected array, got %s",
170  fr_table_str_by_value(redis_reply_types, reply->type, "<UNKNOWN>"));
171  goto error;
172  }
173 
174  RDEBUG3("Entry contains %zu elements", reply->elements);
175 
176  if (reply->elements == 0) {
177  fr_redis_reply_free(&reply);
178  return CACHE_MISS;
179  }
180 
181  if (reply->elements % 3) {
182  REDEBUG("Invalid number of reply elements (%zu). "
183  "Reply must contain triplets of keys operators and values",
184  reply->elements);
185  goto error;
186  }
187 
188 #ifdef HAVE_TALLOC_ZERO_POOLED_OBJECT
189  /*
190  * We can get a pretty good idea of the required size of the pool
191  */
192  for (i = 0; i < reply->elements; i += 3) {
193  pool_size += sizeof(map_t) + (sizeof(tmpl_t) * 2);
194  if (reply->element[i]->type == REDIS_REPLY_STRING) pool_size += reply->element[i]->len + 1;
195  }
196 
197  /*
198  * reply->elements gives us the number of chunks, as the maps are triplets, and there
199  * are three chunks per map
200  */
201 
202  c = talloc_zero_pooled_object(NULL, rlm_cache_entry_t, reply->elements, pool_size);
203 #else
204  c = talloc_zero(NULL, rlm_cache_entry_t);
205 #endif
206  /*
207  * Convert the key/value pairs back into maps
208  */
209  for (i = 0; i < reply->elements; i += 3) {
210  if (fr_redis_reply_to_map(c, last, request,
211  reply->element[i], reply->element[i + 1], reply->element[i + 2]) < 0) {
212  talloc_free(c);
213  fr_redis_reply_free(&reply);
214  return CACHE_ERROR;
215  }
216  last = &(*last)->next;
217  }
218  fr_redis_reply_free(&reply);
219 
220  /*
221  * Pull out the cache created date
222  */
223  if (tmpl_da(head->lhs) == attr_cache_created) {
224  map_t *map;
225 
226  c->created = tmpl_value(head->rhs)->vb_date;
227 
228  map = head;
229  head = head->next;
230  talloc_free(map);
231  }
232 
233  /*
234  * Pull out the cache expires date
235  */
236  if (tmpl_da(head->lhs) == attr_cache_expires) {
237  map_t *map;
238 
239  c->expires = tmpl_value(head->rhs)->vb_date;
240 
241  map = head;
242  head = head->next;
243  talloc_free(map);
244  }
245 
246  c->key = talloc_memdup(c, key, key_len);
247  c->key_len = key_len;
248  c->maps = head;
249  *out = c;
250 
251  return CACHE_OK;
252 }

255 /** Insert a new entry into the data store
256  *
257  * @copydetails cache_entry_insert_t
258  */
259 static cache_status_t cache_entry_insert(UNUSED rlm_cache_config_t const *config, void *instance,
260  request_t *request, UNUSED void *handle, const rlm_cache_entry_t *c)
261 {
262  rlm_cache_redis_t *driver = instance;
263  TALLOC_CTX *pool;
264 
265  map_t *map;
266 
267  fr_redis_conn_t *conn;
268  fr_redis_cluster_state_t state;
269  fr_redis_rcode_t status;
270  redisReply *reply = NULL;
271  int s_ret;
272 
273  static char const command[] = "RPUSH";
274  char const **argv;
275  size_t *argv_len;
276  char const **argv_p;
277  size_t *argv_len_p;
278 
279  unsigned int pipelined = 0; /* How many commands pending in the pipeline */
280  redisReply *replies[5]; /* Should have the same number of elements as pipelined commands */
281  size_t reply_cnt = 0, i;
282 
283  int cnt;
284 
285  tmpl_t expires_value;
286  map_t expires = {
287  .op = T_OP_SET,
288  .lhs = driver->expires_attr,
289  .rhs = &expires_value,
290  };
291 
292  tmpl_t created_value;
293  map_t created = {
294  .op = T_OP_SET,
295  .lhs = driver->created_attr,
296  .rhs = &created_value,
297  .next = &expires
298  };
299 
300  /*
301  * Encode the entry created date
302  */
303  tmpl_init_shallow(&created_value, TMPL_TYPE_DATA, T_BARE_WORD, "<TEMP>", 6);
304  fr_value_box_init(&created_value.data.literal, FR_TYPE_DATE, NULL, true);
305  tmpl_value(&created_value)->vb_date = c->created;
306 
307  /*
308  * Encode the entry expiry time
309  *
310  * Although Redis objects expire on their own, we still need this
311  * to ignore entries that were created before the last epoch.
312  */
313  tmpl_init_shallow(&expires_value, TMPL_TYPE_DATA, T_BARE_WORD, "<TEMP>", 6);
314  fr_value_box_init(&expires_value.data.literal, FR_TYPE_DATE, NULL, true);
315  tmpl_value(&expires_value)->vb_date = c->expires;
316  expires.next = c->maps; /* Head of the list */
317 
318  for (cnt = 0, map = &created; map; cnt++, map = map->next);
319 
320  /*
321  * The majority of serialized entries should be under 1k.
322  *
323  * @todo We should really calculate this using some sort of moving average.
324  */
325  pool = talloc_pool(request, 1024);
326  if (!pool) return CACHE_ERROR;
327 
328  argv_p = argv = talloc_array(pool, char const *, (cnt * 3) + 2); /* pair = 3 + cmd + key */
329  argv_len_p = argv_len = talloc_array(pool, size_t, (cnt * 3) + 2); /* pair = 3 + cmd + key */
330 
331  *argv_p++ = command;
332  *argv_len_p++ = sizeof(command) - 1;
333 
334  *argv_p++ = (char const *)c->key;
335  *argv_len_p++ = c->key_len;
336 
337  /*
338  * Add the maps to the command string in reverse order
339  */
340  for (map = &created; map; map = map->next) {
341  if (fr_redis_tuple_from_map(pool, argv_p, argv_len_p, map) < 0) {
342  REDEBUG("Failed encoding map as Redis K/V pair");
343  talloc_free(pool);
344  return CACHE_ERROR;
345  }
346  argv_p += 3;
347  argv_len_p += 3;
348  }
349 
350  RDEBUG3("Pipelining commands");
351 
352  for (s_ret = fr_redis_cluster_state_init(&state, &conn, driver->cluster, request, c->key, c->key_len, false);
353  s_ret == REDIS_RCODE_TRY_AGAIN; /* Continue */
354  s_ret = fr_redis_cluster_state_next(&state, &conn, driver->cluster, request, status, &reply)) {
355  /*
356  * Start the transaction, as we need to set an expiry time too.
357  */
358  if (c->expires > 0) {
359  RDEBUG3("MULTI");
360  if (redisAppendCommand(conn->handle, "MULTI") != REDIS_OK) {
361  append_error:
362  REXDENT();
363  RERROR("Failed appending Redis command to output buffer: %s", conn->handle->errstr);
364  talloc_free(pool);
365  return CACHE_ERROR;
366  }
367  pipelined++;
368  }
369 
370  RDEBUG3("DEL \"%pV\"", fr_box_strvalue_len((char const *)c->key, c->key_len));
371 
372  if (redisAppendCommand(conn->handle, "DEL %b", c->key, c->key_len) != REDIS_OK) goto append_error;
373  pipelined++;
374 
375  if (RDEBUG_ENABLED3) {
376  RDEBUG3("argv command");
377  RINDENT();
378  for (i = 0; i < talloc_array_length(argv); i++) {
379  RDEBUG3("%pV", fr_box_strvalue_len(argv[i], argv_len[i]));
380  }
381  REXDENT();
382  }
383  redisAppendCommandArgv(conn->handle, talloc_array_length(argv), argv, argv_len);
384  pipelined++;
385 
386  /*
387  * Set the expiry time and close out the transaction.
388  */
389  if (c->expires > 0) {
390  RDEBUG3("EXPIREAT \"%pV\" %" PRIu64,
391  fr_box_strvalue_len((char const *)c->key, c->key_len),
392  fr_unix_time_to_sec(c->expires));
393  if (redisAppendCommand(conn->handle, "EXPIREAT %b %" PRIu64, c->key,
394  c->key_len,
395  fr_unix_time_to_sec(c->expires)) != REDIS_OK) goto append_error;
396  pipelined++;
397  RDEBUG3("EXEC");
398  if (redisAppendCommand(conn->handle, "EXEC") != REDIS_OK) goto append_error;
399  pipelined++;
400  }
401 
402  reply_cnt = fr_redis_pipeline_result(&pipelined, &status,
403  replies, NUM_ELEMENTS(replies),
404  conn);
405  reply = replies[0];
406  }
407  talloc_free(pool);
408 
409  if (s_ret != REDIS_RCODE_SUCCESS) {
410  RPERROR("Failed inserting entry");
411  return CACHE_ERROR;
412  }
413 
414  RDEBUG3("Command results");
415  RINDENT();
416  if (RDEBUG_ENABLED3) for (i = 0; i < reply_cnt; i++) fr_redis_reply_print(L_DBG_LVL_3, replies[i], request, i);
417  fr_redis_pipeline_free(replies, reply_cnt);
418  REXDENT();
419 
420  return CACHE_OK;
421 }

433  fr_redis_rcode_t status;

436  cache_status_t cache_status;
437 
438  for (s_ret = fr_redis_cluster_state_init(&state, &conn, driver->cluster, request, key, key_len, false);

461  fr_table_str_by_value(redis_reply_types, reply->type, "<UNKNOWN>"));

464  return CACHE_ERROR;
465 }
