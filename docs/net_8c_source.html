54 /** Check UDP header is valid
55  *
56  * @param data Pointer to the start of the UDP header
57  * @param remaining bits of received packet
58  * @param ip pointer to IP header structure
59  * @return
60  * - 1 if checksum is incorrect.
61  * - 0 if UDP payload length and checksum are correct
62  * - -1 on validation error.
63  */
64  int fr_udp_header_check(uint8_t const *data, uint16_t remaining, ip_header_t const * ip)
65  {
66  int ret = 0;
67  udp_header_t const *udp;
68 
69  /*
70  * UDP header validation.
71  */
72  udp = (udp_header_t const *)data;
73  uint16_t udp_len;
74  ssize_t diff;
75  uint16_t expected;
76 
77  udp_len = ntohs(udp->len);
78  diff = udp_len - remaining;
79  /* Truncated data */
80  if (diff > 0) {
81  fr_strerror_printf("packet too small by %zi bytes, UDP header + Payload should be %hu bytes",
82  diff, udp_len);
83  return -1;
84  }
85  /* Trailing data */
86  else if (diff < 0) {
87  fr_strerror_printf("Packet too big by %zi bytes, UDP header + Payload should be %hu bytes",
88  diff * -1, udp_len);
89  return -1;
90  }
91 
92  expected = fr_udp_checksum((uint8_t const *) udp, ntohs(udp->len), udp->checksum,
93  ip->ip_src, ip->ip_dst);
94  if (udp->checksum != expected) {
95  fr_strerror_printf("UDP checksum invalid, packet: 0x%04hx calculated: 0x%04hx",
96  ntohs(udp->checksum), ntohs(expected));
97  /* Not a fatal error */
98  ret = 1;
99  }
100 
101  return ret;
102  }

104 /** Calculate UDP checksum
105  *
106  * Zero out UDP checksum in UDP header before calling #fr_udp_checksum to get 'expected' checksum.
107  *
108  * @param data Pointer to the start of the UDP header
109  * @param len value of udp length field in host byte order. Must be validated to make
110  * sure it won't overrun data buffer.
111  * @param checksum current checksum, leave as 0 to just enable validation.
112  * @param src_addr in network byte order.
113  * @param dst_addr in network byte order.
114  * @return
115  * - 0 if the checksum is correct.
116  * - !0 if checksum is incorrect.
117  */
118 uint16_t fr_udp_checksum(uint8_t const *data, uint16_t len, uint16_t checksum,
119  struct in_addr const src_addr, struct in_addr const dst_addr)
120 {
121  uint64_t sum = 0; /* using 64bits avoids overflow check */
122  uint16_t const *p = (uint16_t const *)data;
123 
124  uint16_t const *ip_src = (void const *) &src_addr.s_addr;
125  uint16_t const *ip_dst = (void const *) &dst_addr.s_addr;
126  uint16_t i;
127 
128  sum += *(ip_src++);
129  sum += *ip_src;
130  sum += *(ip_dst++);
131  sum += *ip_dst;
132 
133  sum += htons(IPPROTO_UDP);
134  sum += htons(len);
135 
136  for (i = len; i > 1; i -= 2) sum += *p++;
137  if (i) sum += (0xff & *(uint8_t const *)p) << 8;
138 
139  sum -= checksum;
140 
141  while (sum >> 16) sum = (sum & 0xffff) + (sum >> 16);
142 
143  return ((uint16_t) ~sum);
144 }

146 /** Calculate IP header checksum.
147  *
148  * Zero out IP header checksum in IP header before calling fr_ip_header_checksum to get 'expected' checksum.
149  *
150  * @param data Pointer to the start of the IP header
151  * @param ihl value of ip header length field (number of 32 bit words)
152  */
153 uint16_t fr_ip_header_checksum(uint8_t const *data, uint8_t ihl)
154 {
155  uint64_t sum = 0;
156  uint16_t const *p = (uint16_t const *)data;
157 
158  uint8_t nwords = (ihl << 1); /* number of 16-bit words */
159 
160  for (sum = 0; nwords > 0; nwords--) {
161  sum += *p++;
162  }
163  sum = (sum >> 16) + (sum & 0xffff);
164  sum += (sum >> 16);
165  return ((uint16_t) ~sum);
166 }

168 uint16_t fr_ip6_pesudo_header_checksum(struct in6_addr *src, struct in6_addr *dst, uint16_t ip_len, uint8_t ip_next)
169 {
170  uint64_t sum = 0;
171  ip_pseudo_header6_t ip6; /* Keep correct alignment for the pointer */
172  uint8_t const *p = (uint8_t const *) &ip6;
173  int8_t nwords = sizeof(ip6) >> 1; /* number of 16-bit words */
174 
175  memcpy(&ip6.ip_src, src, sizeof(ip6.ip_src));
176  memcpy(&ip6.ip_dst, dst, sizeof(ip6.ip_dst));
177  ip6.ip_len = ip_len;
178  ip6.ip_next = ip_next;
179 
180  for (sum = 0; nwords > 0; nwords--) {
181  uint16_t word;
182  memcpy(&word, p, sizeof(word)); /* Can't use a uint16_t * as GCC flags this for unaligned access */
183  sum += word;
184  p += sizeof(word);
185  }
186  sum = (sum >> 16) + (sum & 0xffff);
187  sum += (sum >> 16);
188  return ((uint16_t) ~sum);
189 }
