180 /*
181  * Attach the module.
182  */
183 static int mod_instantiate(void *instance, CONF_SECTION *cs)
184 {
185  rlm_eap_fast_t *inst = talloc_get_type_abort(instance, rlm_eap_fast_t);
186 
187  if (!virtual_server_find(inst->virtual_server)) {
188  cf_log_err_by_child(cs, "virtual_server", "Unknown virtual server '%s'", inst->virtual_server);
189  return -1;
190  }
191 
192  inst->default_provisioning_method = eap_name2type(inst->default_provisioning_method_name);
193  if (!inst->default_provisioning_method) {
194  cf_log_err_by_child(cs, "default_provisioning_eap_type", "Unknown EAP type %s",
195  inst->default_provisioning_method_name);
196  return -1;
197  }
198 
199  /*
200  * Read tls configuration, either from group given by 'tls'
201  * option, or from the eap-tls configuration.
202  */
203  inst->tls_conf = eap_tls_conf_parse(cs, "tls");
204 
205  if (!inst->tls_conf) {
206  cf_log_err_by_child(cs, "tls", "Failed initializing SSL context");
207  return -1;
208  }
209 
210  if (talloc_array_length(inst->pac_opaque_key) - 1 != 32) {
211  cf_log_err_by_child(cs, "pac_opaque_key", "Must be 32 bytes long");
212  return -1;
213  }
214 
215  /*
216  * Allow anything for the TLS version, we try to forcibly
217  * disable TLSv1.2 later.
218  */
219  if (inst->tls_conf->tls_min_version > (float) 1.1) {
220  cf_log_err_by_child(cs, "tls_min_version", "require tls_min_version <= 1.1");
221  return -1;
222  }
223 
224  if (!inst->pac_lifetime) {
225  cf_log_err_by_child(cs, "pac_lifetime", "must be non-zero");
226  return -1;
227  }
228 
229  fr_assert(PAC_A_ID_LENGTH == MD5_DIGEST_LENGTH);
230 
231  fr_md5_calc(inst->a_id, (uint8_t const *)inst->authority_identity,
232  talloc_array_length(inst->authority_identity) - 1);
233 
234  return 0;
235 }

259 static void eap_fast_session_ticket(fr_tls_session_t *tls_session, const SSL *s,
260  uint8_t *secret, int *secret_len)
261 {
262  eap_fast_tunnel_t *t = talloc_get_type_abort(tls_session->opaque, eap_fast_tunnel_t);
263  uint8_t seed[2 * SSL3_RANDOM_SIZE];
264 
265  fr_assert(t->pac.key);
266 
267  SSL_get_server_random(s, seed, SSL3_RANDOM_SIZE);
268  SSL_get_client_random(s, &seed[SSL3_RANDOM_SIZE], SSL3_RANDOM_SIZE);
269 
270  T_PRF(t->pac.key, PAC_KEY_LENGTH, "PAC to master secret label hash",
271  seed, sizeof(seed), secret, SSL_MAX_MASTER_KEY_LENGTH);
272  *secret_len = SSL_MAX_MASTER_KEY_LENGTH;
273 }

309 /*
310  * hints from hostap:src/crypto/tls_openssl.c:fr_tls_session_ticket_ext_cb()
311  *
312  * N.B. we actually always tell OpenSSL we have digested the ticket so that
313  * it does not cause a fail loop and enables us to update the PAC easily
314  *
315  */
316 static int _session_ticket(SSL *s, uint8_t const *data, int len, void *arg)
317 {
318  fr_tls_session_t *tls_session = talloc_get_type_abort(arg, fr_tls_session_t);
319  request_t *request = talloc_get_type_abort(SSL_get_ex_data(s, FR_TLS_EX_INDEX_REQUEST), request_t);
320  eap_fast_tunnel_t *t;
321  fr_pair_t *fast_vps = NULL, *vp;
322  fr_cursor_t cursor;
323  char const *errmsg;
324  int dlen, plen;
325  uint16_t length;
326  eap_fast_attr_pac_opaque_t const *opaque = (eap_fast_attr_pac_opaque_t const *) data;
327  eap_fast_attr_pac_opaque_t opaque_plaintext;
328 
329  if (!tls_session) return 0;
330 
331  t = talloc_get_type_abort(tls_session->opaque, eap_fast_tunnel_t);
332 
333  RDEBUG2("PAC provided via ClientHello SessionTicket extension");
334  RHEXDUMP3(data, len, "PAC-Opaque");
335 
336  if ((ntohs(opaque->hdr.type) & EAP_FAST_TLV_TYPE) != attr_eap_fast_pac_opaque_tlv->attr) {
337  errmsg = "PAC is not of type Opaque";
338 error:
339  RERROR("%s, sending alert to client", errmsg);
340  if (fr_tls_session_alert(request, tls_session, SSL3_AL_FATAL, SSL_AD_BAD_CERTIFICATE)) {
341  RERROR("too many alerts");
342  return 0;
343  }
344  if (t->pac.key) talloc_free(t->pac.key);
345 
346  memset(&t->pac, 0, sizeof(t->pac));
347  if (fast_vps) fr_pair_list_free(&fast_vps);
348  return 1;
349  }
350 
351  /*
352  * we would like to use the length of the SessionTicket
353  * but Cisco hates everyone and sends a zero padding payload
354  * so we have to use the length in the PAC-Opaque header
355  */
356  length = ntohs(opaque->hdr.length);
357  if (len - sizeof(opaque->hdr) < length) {
358  errmsg = "PAC has bad length in header";
359  goto error;
360  }
361 
362  if (length < PAC_A_ID_LENGTH + EVP_MAX_IV_LENGTH + EVP_GCM_TLS_TAG_LEN + 1) {
363  errmsg = "PAC file too short";
364  goto error;
365  }
366 
367  if (memcmp(opaque->aad, t->a_id, PAC_A_ID_LENGTH)) {
368  errmsg = "PAC has incorrect A_ID";
369  goto error;
370  }
371 
372  dlen = length - sizeof(opaque->aad) - sizeof(opaque->iv) - sizeof(opaque->tag);
373  plen = eap_fast_decrypt(opaque->data, dlen, opaque->aad, PAC_A_ID_LENGTH,
374  (uint8_t const *) opaque->tag, t->pac_opaque_key, opaque->iv,
375  (uint8_t *)&opaque_plaintext);
376  if (plen == -1) {
377  errmsg = "PAC failed to decrypt";
378  goto error;
379  }
380 
381  RHEXDUMP3((uint8_t const *)&opaque_plaintext, plen, "PAC-Opaque plaintext data section");
382 
383  fr_cursor_init(&cursor, &fast_vps);
384  if (eap_fast_decode_pair(tls_session, &cursor, attr_eap_fast_pac_opaque_tlv, (uint8_t *)&opaque_plaintext, plen, NULL) < 0) {
385  errmsg = fr_strerror();
386  goto error;
387  }
388 
389  for (vp = fr_cursor_head(&cursor);
390  vp;
391  vp = fr_cursor_next(&cursor)) {
392  if (vp->da == attr_eap_fast_pac_info_pac_type) {
393  fr_assert(t->pac.type == 0);
394  t->pac.type = vp->vp_uint32;
395  } else if (vp->da == attr_eap_fast_pac_info_pac_lifetime) {
396  fr_assert(t->pac.expires == 0);
397  t->pac.expires = vp->vp_uint32;
398  t->pac.expired = (vp->vp_uint32 <= time(NULL));
399  /*
400  * Not sure if this is the correct attr
401  * The original enum didn't match a specific TLV nesting level
402  */
403  } else if (vp->da == attr_eap_fast_pac_key) {
404  fr_assert(t->pac.key == NULL);
405  fr_assert(vp->vp_length == PAC_KEY_LENGTH);
406  t->pac.key = talloc_array(t, uint8_t, PAC_KEY_LENGTH);

447 /*
448  * Do authentication, by letting EAP-TLS do most of the work.
449  */
450 static unlang_action_t mod_process(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
451 {
452  rlm_eap_fast_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_eap_fast_t);
453  eap_tls_status_t status;
454 
455  eap_session_t *eap_session = eap_session_get(request->parent);
456  eap_tls_session_t *eap_tls_session = talloc_get_type_abort(eap_session->opaque, eap_tls_session_t);
457  fr_tls_session_t *tls_session = eap_tls_session->tls_session;
458 
459  /*
460  * We need FAST data associated with the session, so
461  * allocate it here, if it wasn't already alloacted.
462  */
463  if (!tls_session->opaque) tls_session->opaque = eap_fast_alloc(tls_session, inst);
464 
465  /*
466  * Process TLS layer until done.
467  */
468  status = eap_tls_process(request, eap_session);
469  if ((status == EAP_TLS_INVALID) || (status == EAP_TLS_FAIL)) {
470  REDEBUG("[eap-tls process] = %s", fr_table_str_by_value(eap_tls_status_table, status, "<INVALID>"));
471  } else {
472  RDEBUG2("[eap-tls process] = %s", fr_table_str_by_value(eap_tls_status_table, status, "<INVALID>"));
473  }
474 
475  switch (status) {
476  /*
477  * EAP-TLS handshake was successful, tell the
478  * client to keep talking.
479  *
480  * If this was EAP-TLS, we would just return
481  * an EAP-TLS-Success packet here.
482  */
483  case EAP_TLS_ESTABLISHED:
484  fr_tls_session_send(request, tls_session);
485  fr_assert(tls_session->opaque != NULL);
486  break;
487 
488  /*
489  * The TLS code is still working on the TLS
490  * exchange, and it's a valid TLS request.
491  * do nothing.
492  */
493  case EAP_TLS_HANDLED:
494  RETURN_MODULE_HANDLED;
495 
496  /*
497  * Handshake is done, proceed with decoding tunneled
498  * data.
499  */
500  case EAP_TLS_RECORD_RECV_COMPLETE:
501  break;
502 
503  /*
504  * Anything else: fail.
505  */
506  default:
507  RETURN_MODULE_FAIL;
508  }
509 
510  /*
511  * Session is established, proceed with decoding
512  * tunneled data.
513  */
514  RDEBUG2("Session established. Proceeding to decode tunneled attributes");
515 
516  /*
517  * Process the FAST portion of the request.
518  */
519  switch (eap_fast_process(request, eap_session, tls_session)) {
520  case FR_CODE_ACCESS_REJECT:
521  eap_tls_fail(request, eap_session);
522  RETURN_MODULE_FAIL;
523 
524  /*
525  * Access-Challenge, continue tunneled conversation.
526  */
527  case FR_CODE_ACCESS_CHALLENGE:
528  fr_tls_session_send(request, tls_session);
529  eap_tls_request(request, eap_session);
530  RETURN_MODULE_HANDLED;
531 
532  /*
533  * Success: Automatically return MPPE keys.
534  */
535  case FR_CODE_ACCESS_ACCEPT:
536  if (eap_tls_success(request, eap_session, NULL, 0, NULL, 0) < 0) RETURN_MODULE_FAIL;
537  RETURN_MODULE_OK;
538 
539  /*
540  * No response packet, MUST be proxying it.
541  * The main EAP module will take care of discovering
542  * that the request now has a "proxy" packet, and
543  * will proxy it, rather than returning an EAP packet.
544  */
545  case FR_CODE_STATUS_CLIENT:
546  RETURN_MODULE_OK;
547 
548  default:
549  break;
550  }
551 
552  /*
553  * Something we don't understand: Reject it.
554  */
555  eap_tls_fail(request, eap_session);
556  RETURN_MODULE_FAIL;
557 }

559 /*
560  * Send an initial eap-tls request to the peer, using the libeap functions.
561  */
562 static unlang_action_t mod_session_init(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
563 {
564  rlm_eap_fast_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_eap_fast_t);
565  eap_session_t *eap_session = eap_session_get(request->parent);
566  eap_tls_session_t *eap_tls_session;
567  fr_tls_session_t *tls_session;
568 
569  fr_pair_t *vp;
570  bool client_cert;
571 
572  eap_session->tls = true;
573 
574  /*
575  * EAP-TLS-Require-Client-Cert attribute will override
576  * the require_client_cert configuration option.
577  */
578  vp = fr_pair_find_by_da(&request->control_pairs, attr_eap_tls_require_client_cert);
579  if (vp) {
580  client_cert = vp->vp_uint32 ? true : false;
581  } else {
582  client_cert = inst->req_client_cert;
583  }
584 
585  eap_session->opaque = eap_tls_session = eap_tls_session_init(request, eap_session, inst->tls_conf, client_cert);
586  if (!eap_tls_session) RETURN_MODULE_FAIL;
587 
588  tls_session = eap_tls_session->tls_session;
589 
590  if (inst->cipher_list) {
591  RDEBUG2("Over-riding main cipher list with '%s'", inst->cipher_list);
592 
593  if (!SSL_set_cipher_list(tls_session->ssl, inst->cipher_list)) {
594  REDEBUG("Failed over-riding cipher list to '%s'. EAP-FAST will likely not work",
595  inst->cipher_list);
596  }
597  }
598 
599 #ifdef SSL_OP_NO_TLSv1_2
600  /*
601  * Forcibly disable TLSv1.2
602  */
603  SSL_set_options(tls_session->ssl, SSL_OP_NO_TLSv1_2);
604 #endif
605 
606  /*
607  * Push TLV of authority_identity into tls_record
608  * call eap_tls_compose() with args
609  *
610  * RFC 4851 section 4.1.1
611  * N.B. mandatory/reserved flags are not applicable here
612  */
613  eap_fast_tlv_append(tls_session, attr_eap_fast_pac_info_a_id, false, PAC_A_ID_LENGTH, inst->a_id);
614 
615  /*
616  * TLS session initialization is over. Now handle TLS
617  * related handshaking or application data.
618  */
619  if (eap_tls_compose(request, eap_session, EAP_TLS_START_SEND,
620  SET_START(eap_tls_session->base_flags) | EAP_FAST_VERSION,
621  &tls_session->clean_in, tls_session->clean_in.used,
622  tls_session->clean_in.used) < 0) {
623  talloc_free(tls_session);
624  RETURN_MODULE_FAIL;
625  }
626 
627  tls_session->record_init(&tls_session->clean_in);
628  eap_session->process = mod_process;
629 
630  if (!SSL_set_session_ticket_ext_cb(tls_session->ssl, _session_ticket, tls_session)) {
631  RERROR("Failed setting SSL session ticket callback");
632  RETURN_MODULE_FAIL;
633  }
634 
635  RETURN_MODULE_HANDLED;
636 }
