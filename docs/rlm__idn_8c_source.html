40 /*
41  * The primary use case for this module is DNS-safe encoding of realms
42  * appearing in requests for a DDDS scheme. Some notes on that usage
43  * scenario:
44  *
45  * RFC2865 5.1 User-Name may be one of:
46  *
47  * 1) UTF-8 text: in which case this conversion is needed
48  *
49  * 2) realm part of an NAI: in which case this conversion should do nothing
50  * since only ASCII digits, ASCII alphas, ASCII dots, and ASCII hyphens
51  * are allowed.
52  *
53  * 3) "A name in ASN.1 form used in Public Key authentication systems.":
54  * I count four things in that phrase that are rather ... vague.
55  * However, most X.509 docs yell at you to IDNA internationalized
56  * domain names to IA5String, so if it is coming from inside an X.509
57  * certificate IDNA should be idempotent in the encode direction.
58  *
59  * Except for that last loophole, which we will leave up to the user
60  * to sort out, we should be safe in processing the realm as UTF-8.
61  */

90 /** Convert domain name to ASCII punycode
91  *
92 @verbatim
93 %{idn:<domain>}
94 @endverbatim
95  *
96  * @ingroup xlat_functions
97  */
98 static ssize_t xlat_idna(UNUSED TALLOC_CTX *ctx, char **out, size_t outlen,
99  void const *mod_inst, UNUSED void const *xlat_inst,
100  request_t *request, char const *fmt)
101 {
102  rlm_idn_t const *inst = mod_inst;
103  char *idna = NULL;
104  int res;
105  size_t len;
106  int flags = 0;
107 
108  if (inst->use_std3_ascii_rules) {
109  flags |= IDNA_USE_STD3_ASCII_RULES;
110  }
111  if (inst->allow_unassigned) {
112  flags |= IDNA_ALLOW_UNASSIGNED;
113  }
114 
115  res = idna_to_ascii_8z(fmt, &idna, flags);
116  if (res) {
117  if (idna) {
118  free (idna); /* Docs unclear, be safe. */
119  }
120 
121  REDEBUG("%s", idna_strerror(res));
122  return -1;
123  }
124 
125  len = strlen(idna);
126 
127  /* 253 is max DNS length */
128  if (!((len < (outlen - 1)) && (len <= 253))) {
129  /* Never provide a truncated result, as it may be queried. */
130  REDEBUG("Conversion was truncated");
131 
132  free(idna);
133  return -1;
134 
135  }
136 
137  strlcpy(*out, idna, outlen);
138  free(idna);
139 
140  return len;
141 }
