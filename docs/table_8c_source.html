118 /** Convert a string to a value using a lexicographically sorted table
119  *
120  * @param[in] table to search in.
121  * @param[in] table_len The number of elements in the table.
122  * @param[in] element_size Size of elements in the table.
123  * @param[in] name to resolve to a value.
124  * @return
125  * - value of matching entry.
126  * - NULL if no matching entries.
127  */
128 static void const *table_sorted_value_by_str(void const *table, size_t table_len, size_t element_size,
129  char const *name)
130 {
131  ssize_t start = 0;
132  ssize_t end = table_len - 1;
133  ssize_t mid;
134 
135  int ret;
136 
137  if (!name) return NULL;
138 
139  while (start <= end) {
140  void const *offset;
141 
142  mid = start + ((end - start) / 2); /* Avoid overflow */
143 
144  offset = TABLE_IDX(table, mid, element_size);
145  ret = strcasecmp(name, ELEM_STR(offset));
146  if (ret == 0) return offset;
147  if (ret < 0) {
148  end = mid - 1;
149  } else {
150  start = mid + 1;
151  }
152  }
153 
154  return NULL;
155 }

293 /** Find the longest string match using a lexicographically sorted table
294  *
295  * Performs a binary search in the specified table, returning the longest
296  * offsetent which is a prefix of name.
297  *
298  * i.e. given name of "food", and table of f, foo, of - foo would be returned.
299  *
300  * @note The table *MUST* be sorted lexicographically, else the result may be incorrect.
301  *
302  * @param[out] match_len How much of the input string matched.
303  * @param[in] table to search in.
304  * @param[in] table_len The number of elements in the table.
305  * @param[in] element_size Size of elements in the table.
306  * @param[in] name to locate.
307  * @param[in] name_len the maximum amount of name that should be matched.
308  * @return
309  * - num value of matching entry.
310  * - NULL if no matching entries.
311  */
312 static void const *table_sorted_value_by_longest_prefix(size_t *match_len,
313  void const *table, size_t table_len, size_t element_size,
314  char const *name, ssize_t name_len)
315 {
316  ssize_t start = 0;
317  ssize_t end = table_len - 1;
318  ssize_t mid;
319 
320  int ret;
321  void const *found = NULL;
322 
323  if (!name) return NULL;
324  if (name_len < 0) name_len = strlen(name);
325 
326  while (start <= end) {
327  void const *offset;
328  char const *elem;
329  size_t tlen;
330 
331  mid = start + ((end - start) / 2); /* Avoid overflow */
332 
333  offset = TABLE_IDX(table, mid, element_size);
334  elem = ELEM_STR(offset);
335  tlen = ELEM_LEN(offset);
336 
337  ret = strncasecmp(name, elem, tlen < (size_t)name_len ? tlen : (size_t)name_len);
338  if (ret == 0) {
339  /*
340  * Exact match
341  */
342  if (tlen == (size_t)name_len) {
343  if (match_len) *match_len = tlen;
344  return offset;
345  }
346 
347  /*
348  * Partial match.
349  * Name we're searching for is longer.
350  * This might be the longest prefix,
351  * so record it.
352  */
353  if (tlen < (size_t)name_len) {
354  found = offset;
355  if (match_len) *match_len = tlen;
356  ret = 1;
357  } else {
358  ret = -1;
359  }
360  }
361 
362  if (ret < 0) {
363  end = mid - 1;
364  } else {
365  start = mid + 1;
366  }
367  }
368 
369  if (!found && match_len) *match_len = 0;
370 
371  return found;
372 }

379 /** Find the longest string match using an arbitrarily ordered table
380  *
381  * i.e. given name of "food", and table of f, foo, of - foo would be returned.
382  *
383  * @param[out] match_len How much of the input string matched.
384  * @param[in] table to search in.
385  * @param[in] table_len The number of elements in the table.
386  * @param[in] element_size Size of elements in the table.
387  * @param[in] name to locate.
388  * @param[in] name_len the maximum amount of name that should be matched.
389  * @return
390  * - num value of matching entry.
391  * - def if no matching entries.
392  */
393 static void const *table_ordered_value_by_longest_prefix(size_t *match_len,
394  void const *table, size_t table_len, size_t element_size,
395  char const *name, ssize_t name_len)
396 {
397  size_t i;
398  size_t found_len = 0;
399  void const *found = NULL;
400 
401  if (!name) return NULL;
402  if (name_len < 0) name_len = strlen(name);
403 
404  for (i = 0; i < table_len; i++) {
405  void const *offset;
406  size_t j;
407 
408  offset = TABLE_IDX(table, i, element_size);
409 
410  for (j = 0; (j < (size_t)name_len) && (name[j] == (ELEM_STR(offset))[j]); j++);
411 
412  /*
413  * If we didn't get to the end of the
414  * table string, then continue.
415  */
416  if ((ELEM_STR(offset))[j] != '\0') continue;
417 
418  /*
419  * Exact match
420  */
421  if (j == (size_t)name_len) {
422  if (match_len) *match_len = name_len;
423  return offset;
424  }
425 
426  /*
427  * Partial match.
428  * Name we're searching for is longer.
429  * This might be the longest prefix,
430  * so record it.
431  */
432  if (j > found_len) {
433  found_len = j;
434  found = offset;
435  }
436  }
437 
438  if (match_len) *match_len = found_len;
439 
440  return found;
441 }
