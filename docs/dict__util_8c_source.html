138 /** Apply a simple (case insensitive) hashing function to the name of an attribute, vendor or protocol
139  *
140  * @param[in] name of the attribute, vendor or protocol.
141  * @param[in] len length of the input string.
142  *
143  * @return the hashed derived from the name.
144  */
145 static uint32_t dict_hash_name(char const *name, size_t len)
146 {
147  uint32_t hash = FNV_MAGIC_INIT;
148 
149  char const *p = name, *q = name + len;
150 
151  while (p < q) {
152  int c = *(unsigned char const *)p;
153  if (isalpha(c)) c = tolower(c);
154 
155  hash *= FNV_MAGIC_PRIME;
156  hash ^= (uint32_t)(c & 0xff);
157  p++;
158  }
159 
160  return hash;
161 }

348 /** Set a dictionary attribute's name
349  *
350  * @note This function can only be used _before_ the attribute is inserted into the dictionary.
351  *
352  * @param[in] da_p to set name for.
353  * @param[in] name to set. If NULL a name will be automatically generated.
354  */
355 static inline CC_HINT(always_inline) int dict_attr_name_set(fr_dict_attr_t **da_p, char const *name)
356 {
357  char buffer[FR_DICT_ATTR_MAX_NAME_LEN + 1];
358  size_t name_len;
359  char *name_start, *name_end;
360  fr_dict_attr_t *da = *da_p;
361 
362  /*
363  * Generate a name if none is specified
364  */
365  if (!name) {
366  fr_sbuff_t unknown_name = FR_SBUFF_OUT(buffer, sizeof(buffer));
367 
368  fr_sbuff_in_strcpy_literal(&unknown_name, "Attr-");
369  if (da->parent) {
370  if (fr_dict_print_attr_oid(&unknown_name, NULL, da->parent) > 0) {
371  fr_sbuff_in_char(&unknown_name, '.');
372  }
373  }
374  fr_sbuff_in_sprintf(&unknown_name, "%u", da->attr);
375 
376  name = buffer;
377  name_len = fr_sbuff_used(&unknown_name);
378  } else {
379  name_len = strlen(name);
380  }
381 
382  /*
383  * Grow the structure to hold the name
384  *
385  * We add the name as an extension because it makes
386  * the code less complex, and means the name value
387  * is copied automatically when if the fr_dict_attr_t
388  * is copied.
389  *
390  * We do still need to fixup the da->name pointer
391  * though.
392  */
393  name_start = dict_attr_ext_alloc_size(da_p, FR_DICT_ATTR_EXT_NAME, name_len + 1);
394  if (!name_start) return -1;
395 
396  name_end = name_start + name_len;
397 
398  memcpy(name_start, name, name_len);
399  *name_end = '\0';
400 
401  (*da_p)->name = name_start;
402  (*da_p)->name_len = name_len;
403 
404  return 0;
405 }

509 /** Initialise fields in a dictionary attribute structure
510  *
511  * @note This function can only be used _before_ the attribute is inserted into the dictionary.
512  *
513  * @param[in] da_p to initialise.
514  * @param[in] parent of the attribute, if none, should be
515  * the dictionary root.
516  * @param[in] name of attribute. Pass NULL for auto-generated name.
517  * @param[in] attr number.
518  * @param[in] type of the attribute.
519  * @param[in] flags to assign.
520  */
521 int dict_attr_init(fr_dict_attr_t **da_p,
522  fr_dict_attr_t const *parent,
523  char const *name, int attr,
524  fr_type_t type, fr_dict_attr_flags_t const *flags)
525 {
526  **da_p = (fr_dict_attr_t) {
527  .attr = attr,
528  .type = type,
529  .flags = *flags,
530  .parent = parent,
531  };
532 
533  /*
534  * Record the parent
535  */
536  if (parent) {
537  (*da_p)->dict = parent->dict;
538  (*da_p)->depth = parent->depth + 1;
539 
540  /*
541  * Point to the vendor definition. Since ~90% of
542  * attributes are VSAs, caching this pointer will help.
543  */
544  if (parent->type == FR_TYPE_VENDOR) {
545  if (dict_attr_vendor_set(da_p, parent) < 0) return -1;
546  } else {
547  dict_attr_ext_copy(da_p, parent, FR_DICT_ATTR_EXT_VENDOR); /* Noop if no vendor extension */
548  }
549  } else {
550  (*da_p)->depth = 0;
551  }
552 
553  /*
554  * Cache the da_stack so we don't need
555  * to generate it at runtime.
556  */
557  dict_attr_da_stack_set(da_p);
558 
559  /*
560  * Structural types can have children
561  * so add the extension for them.
562  */
563  switch (type) {
564  case FR_TYPE_STRUCTURAL:
565  structural:
566  if (type == FR_TYPE_GROUP) {
567  if (dict_attr_ref_init(da_p) < 0) return -1;
568  break;
569  }
570 
571  if (type == FR_TYPE_TLV) {
572  if (dict_attr_ref_init(da_p) < 0) return -1;
573  }
574 
575  if (dict_attr_children_init(da_p) < 0) return -1;
576  break;
577 
578  /*
579  * Keying types *sigh*
580  */
581  case FR_TYPE_UINT8: /* Hopefully temporary until unions are done properly */
582  case FR_TYPE_UINT16: /* Same here */
583  if (dict_attr_enumv_init(da_p) < 0) return -1;
584  goto structural;
585 
586  /*
587  * Leaf types
588  */
589  default:
590  if (dict_attr_enumv_init(da_p) < 0) return -1;
591  break;
592  }
593 
594  /*
595  * Name is a separate talloc chunk. We allocate
596  * it last because we cache the pointer value.
597  */
598  if (dict_attr_name_set(da_p, name) < 0) return -1;
599 
600  DA_VERIFY(*da_p);
601 
602  return 0;
603 }

807 /** See if a #fr_dict_attr_t can have children
808  *
809  * The check for children is complicated by the need for "int" types
810  * to have children, when they are `key` fields in a `struct`. This
811  * situation occurs when a struct has multiple sub-structures, which
812  * are selected based on a `key` field.
813  *
814  * There is no other place for the sub-structures to go. In the
815  * future, we may extend the functionality of the `key` field, by
816  * allowing non-integer data types. That would require storing keys
817  * as #fr_dict_enum_t, and then placing the child (i.e. sub)
818  * structures there. But that would involve adding children to
819  * enums, which is currently not supported.
820  *
821  * @param da the dictionary attribute to check.
822  */
823 bool dict_attr_can_have_children(fr_dict_attr_t const *da)
824 {
825  switch (da->type) {
826  case FR_TYPE_TLV:
827  case FR_TYPE_VENDOR:
828  case FR_TYPE_VSA:
829  case FR_TYPE_STRUCT:
830  return true;
831 
832  case FR_TYPE_UINT8:
833  case FR_TYPE_UINT16:
834  case FR_TYPE_UINT32:
835  /*
836  * Children are allowed here, but ONLY if this
837  * attribute is a key field.
838  */
839  if (da->parent && (da->parent->type == FR_TYPE_STRUCT) && da_is_key_field(da)) return true;
840  break;
841 
842  default:
843  break;
844  }
845 
846  return false;
847 }

849 /** Add a child to a parent.
850  *
851  * @param[in] parent we're adding a child to.
852  * @param[in] child to add to parent.
853  * @return
854  * - 0 on success.
855  * - -1 on failure (memory allocation error).
856  */
857 int dict_attr_child_add(fr_dict_attr_t *parent, fr_dict_attr_t *child)
858 {
859  fr_dict_attr_t const * const *bin;
860  fr_dict_attr_t **this;
861  fr_dict_attr_t const **children;
862 
863  /*
864  * Setup fields in the child
865  */
866  fr_assert(child->parent == parent);
867 
868  DA_VERIFY(child);
869 
870  if (da_has_ref(parent)) {
871  fr_strerror_printf("Cannot add children to attribute '%s' which has 'ref=%s'",
872  parent->name, fr_dict_attr_ref(parent)->name);
873  return false;
874  }
875 
876  if (!dict_attr_can_have_children(parent)) {
877  fr_strerror_printf("Cannot add children to attribute '%s' of type %s",
878  parent->name, fr_table_str_by_value(fr_value_box_type_table, parent->type, "?Unknown?"));
879  return false;
880  }
881 
882  /*
883  * We only allocate the pointer array *if* the parent has children.
884  */
885  children = dict_attr_children(parent);
886  if (!children) {
887  children = talloc_zero_array(parent, fr_dict_attr_t const *, UINT8_MAX + 1);
888  if (!children) {
889  fr_strerror_printf("Out of memory");
890  return -1;
891  }
892  if (dict_attr_children_set(parent, children) < 0) return -1;
893  }
894 
895  /*
896  * Treat the array as a hash of 255 bins, with attributes
897  * sorted into bins using num % 255.
898  *
899  * Although the various protocols may define numbers higher than 255:
900  *
901  * RADIUS/DHCPv4 - 1-255
902  * Diameter/Internal - 1-4294967295
903  * DHCPv6 - 1-65535
904  *
905  * In reality very few will ever use attribute numbers > 500, so for
906  * the majority of lookups we get O(1) performance.
907  *
908  * Attributes are inserted into the bin in order of their attribute
909  * numbers to allow slightly more efficient lookups.
910  */
911  bin = &children[child->attr & 0xff];
912  for (;;) {
913  bool child_is_struct = false;
914  bool bin_is_struct = false;
915 
916  if (!*bin) break;
917 
918  /*
919  * Workaround for vendors that overload the RFC space.
920  * Structural attributes always take priority.
921  */
922  switch (child->type) {
923  case FR_TYPE_STRUCTURAL:
924  child_is_struct = true;
925  break;
926 
927  default:
928  break;
929  }
930 
931  switch ((*bin)->type) {
932  case FR_TYPE_STRUCTURAL:
933  bin_is_struct = true;
934  break;
935 
936  default:
937  break;
938  }
939 
940  if (child_is_struct && !bin_is_struct) break;
941  else if (fr_dict_vendor_num_by_da(child) <= fr_dict_vendor_num_by_da(*bin)) break; /* Prioritise RFC attributes */
942  else if (child->attr <= (*bin)->attr) break;
943 
944  bin = &(*bin)->next;
945  }
946 
947  memcpy(&this, &bin, sizeof(this));
948  child->next = *this;
949  *this = child;
950 
951  return 0;
952 }

1510 /** Get the leaf attribute of an OID string
1511  *
1512  * @note On error, vendor will be set (if present), parent will be the
1513  * maximum depth we managed to resolve to, and attr will be the child
1514  * we failed to resolve.
1515  *
1516  * @param[in] dict of protocol context we're operating in.
1517  * If NULL the internal dictionary will be used.
1518  * @param[out] attr Number we parsed.
1519  * @param[in,out] parent attribute (or root of dictionary).
1520  * Will be updated to the parent directly beneath the leaf.
1521  * @param[in] oid string to parse.
1522  * @return
1523  * - > 0 on success (number of bytes parsed).
1524  * - <= 0 on parse error (negative offset of parse error).
1525  */
1526 ssize_t fr_dict_attr_by_oid(fr_dict_t const *dict, fr_dict_attr_t const **parent, unsigned int *attr, char const *oid)
1527 {
1528  char const *p = oid;
1529  unsigned int num = 0;
1530  ssize_t slen;
1531 
1532  if (!*parent) return -1;
1533 
1534  /*
1535  * It's a partial OID. Grab it, and skip to the next bit.
1536  */
1537  if (p[0] == '.') {
1538  p++;
1539  }
1540 
1541  *attr = 0;
1542 
1543  if (fr_dict_oid_component(&num, &p) < 0) return oid - p;
1544 
1545  /*
1546  * Record progress even if we error out.
1547  *
1548  * Don't change this, you will break things.
1549  */
1550  *attr = num;
1551 
1552  switch ((*parent)->type) {
1553  case FR_TYPE_STRUCTURAL:
1554  break;
1555 
1556  default:
1557  fr_strerror_printf("Attribute %s (%i) is not a TLV, so cannot contain a child attribute. "
1558  "Error at sub OID \"%s\"", (*parent)->name, (*parent)->attr, oid);
1559  return 0; /* We parsed nothing */
1560  }
1561 
1562  /*
1563  * If it's not a vendor type, it must be between 0..8*type_size
1564  *
1565  * @fixme: find the TLV parent, and check it's size
1566  */
1567  if (((*parent)->type != FR_TYPE_VENDOR) && ((*parent)->type != FR_TYPE_VSA) && !(*parent)->flags.is_root &&
1568  (num > UINT8_MAX)) {
1569  fr_strerror_printf("TLV attributes must be between 0..255 inclusive");
1570  return 0;
1571  }
1572 
1573  switch (p[0]) {
1574  /*
1575  * We've not hit the leaf yet, so the attribute must be
1576  * defined already.
1577  */
1578  case '.':
1579  {
1580  fr_dict_attr_t const *child;
1581  p++;
1582 
1583  child = dict_attr_child_by_num(*parent, num);
1584  if (!child) {
1585  fr_strerror_printf("Unknown attribute '%i' in OID string \"%s\" for parent %s",
1586  num, oid, (*parent)->name);
1587  return 0; /* We parsed nothing */
1588  }
1589 
1590  /*
1591  * Record progress even if we error out.
1592  *
1593  * Don't change this, you will break things.
1594  */
1595  *parent = child;
1596 
1597  slen = fr_dict_attr_by_oid(dict, parent, attr, p);
1598  if (slen <= 0) return slen - (p - oid);
1599  return slen + (p - oid);
1600  }
1601 
1602  /*
1603  * Hit the leaf, this is the attribute we need to define.
1604  */
1605  case '\0':
1606  *attr = num;
1607  return p - oid;
1608 
1609  default:
1610  fr_strerror_printf("Malformed OID string, got trailing garbage '%s'", p);
1611  return oid - p;
1612  }
1613 }
