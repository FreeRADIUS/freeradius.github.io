45 static ssize_t date_convert_string(request_t *request, char **out, size_t outlen,
46  const char *str, const char *fmt)
47 {
48  struct tm tminfo;
49  time_t date = 0;
50 
51  if (strptime(str, fmt, &tminfo) == NULL) {
52  REDEBUG("Failed to parse time string \"%s\" as format '%s'", str, fmt);
53  return -1;
54  }
55 
56  date = mktime(&tminfo);
57  if (date < 0) {
58  REDEBUG("Failed converting parsed time into unix time");
59  return -1;
60  }
61 
62  return snprintf(*out, outlen, "%" PRIu64, (uint64_t) date);
63 }

65 static ssize_t date_encode_strftime(char **out, size_t outlen, rlm_date_t const *inst,
66  request_t *request, time_t date)
67 {
68  struct tm tminfo;
69 
70  if (inst->utc) {
71  if (gmtime_r(&date, &tminfo) == NULL) {
72  REDEBUG("Failed converting time string to gmtime: %s", fr_syserror(errno));
73  return -1;
74  }
75  } else {
76  if (localtime_r(&date, &tminfo) == NULL) {
77  REDEBUG("Failed converting time string to localtime: %s", fr_syserror(errno));
78  return -1;
79  }
80  }
81 
82  return strftime(*out, outlen, inst->fmt, &tminfo);
83 }

86 /** Get or convert time and date
87  *
88  * Using the format in the module instance configuration, get
89  * various timestamps, or convert strings to date format.
90  *
91  * When the request arrived:
92 @verbatim
93 %{date:request}
94 @endverbatim
95  *
96  * Now:
97 @verbatim
98 %{date:now}
99 @endverbatim
100  *
101  * Examples (Tmp-Integer-0 = 1506101100):
102 @verbatim
103 update request {
104  &Tmp-String-0 := "%{date:&Tmp-Integer-0}" ("Fri 22 Sep 18:25:00 BST 2017")
105  &Tmp-Integer-1 := "%{date:&Tmp-String-0}" (1506101100)
106 }
107 @endverbatim
108  *
109  * @ingroup xlat_functions
110  */
111 static ssize_t xlat_date_convert(UNUSED TALLOC_CTX *ctx, char **out, size_t outlen,
112  void const *mod_inst, UNUSED void const *xlat_inst,
113  request_t *request, char const *fmt)
114 {
115  rlm_date_t const *inst = mod_inst;
116  struct tm tminfo;
117  fr_pair_t *vp;
118 
119  memset(&tminfo, 0, sizeof(tminfo));
120 
121  if (strcmp(fmt, "request") == 0) {
122  return date_encode_strftime(out, outlen, inst, request,
123  fr_time_to_sec(request->packet->timestamp));
124  }
125 
126  if (strcmp(fmt, "now") == 0) {
127  return date_encode_strftime(out, outlen, inst, request, fr_time_to_sec(fr_time()));
128  }
129 
130  if ((xlat_fmt_get_vp(&vp, request, fmt) < 0) || !vp) return 0;
131 
132  switch (vp->vp_type) {
133  /*
134  * These are 'to' types, i.e. we'll convert the integers
135  * to a time structure, and then output it in the specified
136  * format as a string.
137  */
138  case FR_TYPE_DATE:
139  return date_encode_strftime(out, outlen, inst, request, fr_time_to_sec(vp->vp_date));
140 
141  case FR_TYPE_UINT32:
142  return date_encode_strftime(out, outlen, inst, request, (time_t) vp->vp_uint32);
143 
144 
145  case FR_TYPE_UINT64:
146  return date_encode_strftime(out, outlen, inst, request, (time_t) vp->vp_uint64);
147 
148  /*
149  * These are 'from' types, i.e. we'll convert the input string
150  * into a time structure, and then output it as an integer
151  * unix timestamp.
152  */
153  case FR_TYPE_STRING:
154  return date_convert_string(request, out, outlen, vp->vp_strvalue, inst->fmt);
155 
156  default:
157  REDEBUG("Can't convert type %s into date", fr_table_str_by_value(fr_value_box_type_table, vp->da->type, "<INVALID>"));
158  }
159 
160  return -1;
161 }
