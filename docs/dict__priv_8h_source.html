39 /** Set the internal dictionary if none was provided
40  *
41  * @param _dict Dict pointer to check/set.

43  */
44 #define INTERNAL_IF_NULL(_dict, _ret) \
45  do { \
46  if (!(_dict)) { \
47  _dict = dict_gctx ? dict_gctx->internal : NULL; \
48  if (unlikely(!(_dict))) { \
49  fr_strerror_printf("No dictionaries available for attribute resolution"); \

51  } \
52  } \
53  } while (0)

55 /** Vendors and attribute names

57  * It's very likely that the same vendors will operate in multiple
58  * protocol spaces, but number their attributes differently, so we need
59  * per protocol dictionaries.
60  *
61  * There would also be conflicts for DHCP(v6)/RADIUS attributes etc...
62  */
