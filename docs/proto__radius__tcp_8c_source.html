104 static ssize_t mod_read(fr_listen_t *li, UNUSED void **packet_ctx, fr_time_t *recv_time_p, uint8_t *buffer, size_t buffer_len, size_t *leftover, UNUSED uint32_t *priority, UNUSED bool *is_dup)
105 {
106  proto_radius_tcp_t const *inst = talloc_get_type_abort_const(li->app_io_instance, proto_radius_tcp_t);
107  proto_radius_tcp_thread_t *thread = talloc_get_type_abort(li->thread_instance, proto_radius_tcp_thread_t);
108  ssize_t data_size;
109  size_t packet_len, in_buffer;
110  decode_fail_t reason;
111 
112  /*
113  * Read data into the buffer.
114  */
115  data_size = read(thread->sockfd, buffer + *leftover, buffer_len - *leftover);
116  if (data_size < 0) {
117  PDEBUG2("proto_radius_tcp got read error %zd", data_size);
118  return data_size;
119  }
120 
121  /*
122  * Note that we return ERROR for all bad packets, as
123  * there's no point in reading RADIUS packets from a TCP
124  * connection which isn't sending us RADIUS packets.
125  */
126 
127  /*
128  * TCP read of zero means the socket is dead.
129  */
130  if (!data_size) {
131  DEBUG2("proto_radius_tcp - other side closed the socket.");
132  return -1;
133  }
134 
135  /*
136  * We MUST always start with a known RADIUS packet.
137  */
138  if ((buffer[0] == 0) || (buffer[0] > FR_RADIUS_MAX_PACKET_CODE)) {
139  DEBUG("proto_radius_tcp got invalid packet code %d", buffer[0]);
140  thread->stats.total_unknown_types++;
141  return -1;
142  }
143 
144  in_buffer = data_size + *leftover;
145 
146  /*
147  * Not enough for one packet. Tell the caller that we need to read more.
148  */
149  if (in_buffer < 20) {
150  *leftover = in_buffer;
151  return 0;
152  }
153 
154  /*
155  * Figure out how large the RADIUS packet is.
156  */
157  packet_len = (buffer[2] << 8) | buffer[3];
158 
159  /*
160  * We don't have a complete RADIUS packet. Tell the
161  * caller that we need to read more.
162  */
163  if (in_buffer < packet_len) {
164  *leftover = in_buffer;
165  return 0;
166  }
167 
168  /*
169  * We've read more than one packet. Tell the caller that
170  * there's more data available, and return only one packet.
171  */
172  if (in_buffer > packet_len) {
173  *leftover = in_buffer - packet_len;
174  }
175 
176  /*
177  * If it's not a RADIUS packet, ignore it.
178  */
179  if (!fr_radius_ok(buffer, &packet_len, inst->max_attributes, false, &reason)) {
180  /*
181  * @todo - check for F5 load balancer packets. <sigh>
182  */
183  DEBUG2("proto_radius_tcp got a packet which isn't RADIUS");
184  thread->stats.total_malformed_requests++;
185  return -1;
186  }
187 
188  *recv_time_p = fr_time();
189  thread->stats.total_requests++;
190 
191  /*
192  * proto_radius sets the priority
193  */
194 
195  /*
196  * Print out what we received.
197  */
198  DEBUG2("proto_radius_tcp - Received %s ID %d length %d %s",
199  fr_packet_codes[buffer[0]], buffer[1],
200  (int) packet_len, thread->name);
201 
202  return packet_len;
203 }

206 static ssize_t mod_write(fr_listen_t *li, void *packet_ctx, UNUSED fr_time_t request_time,
207  uint8_t *buffer, size_t buffer_len, size_t written)
208 {
209  proto_radius_tcp_thread_t *thread = talloc_get_type_abort(li->thread_instance, proto_radius_tcp_thread_t);
210  fr_io_track_t *track = talloc_get_type_abort(packet_ctx, fr_io_track_t);
211  ssize_t data_size;
212 
213  /*
214  * @todo - share a stats interface with the parent? or
215  * put the stats in the listener, so that proto_radius
216  * can update them, too.. <sigh>
217  */
218  if (!written) thread->stats.total_responses++;
219 
220  /*
221  * This handles the race condition where we get a DUP,
222  * but the original packet replies before we're run.
223  * i.e. this packet isn't marked DUP, so we have to
224  * discover it's a dup later...
225  *
226  * As such, if there's already a reply, then we ignore
227  * the encoded reply (which is probably going to be a
228  * NAK), and instead just ignore the DUP and don't reply.
229  */
230  if (track->reply_len) {
231  return buffer_len;
232  }
233 
234  /*
235  * We only write RADIUS packets.
236  */
237  fr_assert(buffer_len >= 20);
238  fr_assert(written < buffer_len);
239 
240  /*
241  * Only write replies if they're RADIUS packets.
242  * sometimes we want to NOT send a reply...
243  */
244  data_size = write(thread->sockfd, buffer + written, buffer_len - written);
245 
246  /*
247  * This socket is dead. That's an error...
248  */
249  if (data_size <= 0) return data_size;
250 
251  /*
252  * Root through the reply to determine any
253  * connection-level negotiation data, but only the first
254  * time the packet is being written.
255  */
256  if ((written == 0) && (track->packet[0] == FR_CODE_STATUS_SERVER)) {
257 // status_check_reply(inst, buffer, buffer_len);
258  }
259 
260  /*
261  * Add in previously written data to the response.
262  */
263  return data_size + written;
264 }

296 
297  fr_assert(!thread->connection);
298 
299  li->fd = sockfd = fr_socket_server_tcp(&inst->ipaddr, &port, inst->port_name, true);
300  if (sockfd < 0) {
301  PERROR("Failed opening TCP socket");
302  error:
303  return -1;
304  }
305 
306  if (fr_socket_bind(sockfd, &inst->ipaddr, &port, inst->interface) < 0) {
307  close(sockfd);
308  PERROR("Failed binding socket");
309  goto error;
310  }
311 
312  if (listen(sockfd, 8) < 0) {
313  close(sockfd);
314  PERROR("Failed listening on socket");
315  goto error;
316  }
317 
318  thread->sockfd = sockfd;
319 
320  fr_assert((cf_parent(inst->cs) != NULL) && (cf_parent(cf_parent(inst->cs)) != NULL)); /* listen { ... } */
321 
322  thread->name = fr_app_io_socket_name(thread, &proto_radius_tcp,
323  NULL, 0,
324  &inst->ipaddr, inst->port,

348 static int mod_compare(void const *instance, UNUSED void *thread_instance, UNUSED RADCLIENT *client,
349  void const *one, void const *two)
350 {
351  int ret;
352  proto_radius_tcp_t const *inst = talloc_get_type_abort_const(instance, proto_radius_tcp_t);
353 
354  uint8_t const *a = one;
355  uint8_t const *b = two;
356 
357  /*
358  * Do a better job of deduping input packet.
359  */
360  if (inst->dedup_authenticator) {
361  ret = memcmp(a + 4, b + 4, RADIUS_AUTH_VECTOR_LENGTH);
362  if (ret != 0) return ret;
363  }
364 
365  /*
366  * The tree is ordered by IDs, which are (hopefully)
367  * pseudo-randomly distributed.
368  */
369  ret = (a[1] < b[1]) - (a[1] > b[1]);
370  if (ret != 0) return ret;
371 
372  /*
373  * Then ordered by code, which is usally the same.
374  */
375  return (a[0] < b[0]) - (a[0] > b[0]);
376 }

387 static int mod_bootstrap(void *instance, CONF_SECTION *cs)
388 {
389  proto_radius_tcp_t *inst = talloc_get_type_abort(instance, proto_radius_tcp_t);
390  size_t i, num;
391 
392  inst->cs = cs;
393 
394  /*
395  * Complain if no "ipaddr" is set.
396  */
397  if (inst->ipaddr.af == AF_UNSPEC) {
398  cf_log_err(cs, "No 'ipaddr' was specified in the 'tcp' section");
399  return -1;
400  }
401 
402  if (inst->recv_buff_is_set) {
403  FR_INTEGER_BOUND_CHECK("recv_buff", inst->recv_buff, >=, 32);
404  FR_INTEGER_BOUND_CHECK("recv_buff", inst->recv_buff, <=, INT_MAX);
405  }
406 
407  FR_INTEGER_BOUND_CHECK("max_packet_size", inst->max_packet_size, >=, 20);
408  FR_INTEGER_BOUND_CHECK("max_packet_size", inst->max_packet_size, <=, 65536);
409 
410  if (!inst->port) {
411  struct servent *s;
412 
413  if (!inst->port_name) {
414  cf_log_err(cs, "No 'port' was specified in the 'tcp' section");
415  return -1;
416  }
417 
418  s = getservbyname(inst->port_name, "tcp");
419  if (!s) {
420  cf_log_err(cs, "Unknown value for 'port_name = %s", inst->port_name);
421  return -1;
422  }
423 
424  inst->port = ntohl(s->s_port);
425  }
426 
427  /*
428  * Parse and create the trie for dynamic clients, even if
429  * there's no dynamic clients.
430  *
431  * @todo - we could use this for source IP filtering?
432  * e.g. allow clients from a /16, but not from a /24
433  * within that /16.
434  */
435  num = talloc_array_length(inst->allow);
436  if (!num) {
437  if (inst->dynamic_clients) {
438  cf_log_err(cs, "The 'allow' subsection MUST contain at least one 'network' entry when 'dynamic_clients = true'.");
439  return -1;
440  }
441  } else {
442  MEM(inst->trie = fr_trie_alloc(inst));
443 
444  for (i = 0; i < num; i++) {
445  fr_ipaddr_t *network;
446 
447  /*
448  * Can't add v4 networks to a v6 socket, or vice versa.
449  */
450  if (inst->allow[i].af != inst->ipaddr.af) {
451  cf_log_err(cs, "Address family in entry %zd - 'allow = %pV' does not match 'ipaddr'",
452  i + 1, fr_box_ipaddr(inst->allow[i]));
453  return -1;
454  }
455 
456  /*
457  * Duplicates are bad.
458  */
459  network = fr_trie_match(inst->trie,
460  &inst->allow[i].addr, inst->allow[i].prefix);
461  if (network) {
462  cf_log_err(cs, "Cannot add duplicate entry 'allow = %pV'",
463  fr_box_ipaddr(inst->allow[i]));
464  return -1;
465  }
466 
467  /*
468  * Look for overlapping entries.
469  * i.e. the networks MUST be disjoint.
470  *
471  * Note that this catches 192.168.1/24
472  * followed by 192.168/16, but NOT the
473  * other way around. The best fix is
474  * likely to add a flag to
475  * fr_trie_alloc() saying "we can only
476  * have terminal fr_trie_user_t nodes"
477  */
478  network = fr_trie_lookup(inst->trie,
479  &inst->allow[i].addr, inst->allow[i].prefix);
480  if (network && (network->prefix <= inst->allow[i].prefix)) {
481  cf_log_err(cs, "Cannot add overlapping entry 'allow = %pV'",
482  fr_box_ipaddr(inst->allow[i]));
483  cf_log_err(cs, "Entry is completely enclosed inside of a previously defined network");
484  return -1;
485  }
486 
487  /*
488  * Insert the network into the trie.
489  * Lookups will return the fr_ipaddr_t of
490  * the network.
491  */
492  if (fr_trie_insert(inst->trie,
493  &inst->allow[i].addr, inst->allow[i].prefix,
494  &inst->allow[i]) < 0) {
495  cf_log_err(cs, "Failed adding 'allow = %pV' to tracking table",
496  fr_box_ipaddr(inst->allow[i]));
497  return -1;
498  }
499  }
500 
501  /*
502  * And now check denied networks.
503  */
504  num = talloc_array_length(inst->deny);
505  if (!num) return 0;
506 
507  /*
508  * Since the default is to deny, you can only add
509  * a "deny" inside of a previous "allow".
510  */
511  for (i = 0; i < num; i++) {
512  fr_ipaddr_t *network;
513 
514  /*
515  * Can't add v4 networks to a v6 socket, or vice versa.
516  */
517  if (inst->deny[i].af != inst->ipaddr.af) {
518  cf_log_err(cs, "Address family in entry %zd - 'deny = %pV' does not match 'ipaddr'",
519  i + 1, fr_box_ipaddr(inst->deny[i]));
520  return -1;
521  }
522 
523  /*
524  * Duplicates are bad.
525  */
526  network = fr_trie_match(inst->trie,
527  &inst->deny[i].addr, inst->deny[i].prefix);
528  if (network) {
529  cf_log_err(cs, "Cannot add duplicate entry 'deny = %pV'", fr_box_ipaddr(inst->deny[i]));
530  return -1;
531  }
532 
533  /*
534  * A "deny" can only be within a previous "allow".
535  */
536  network = fr_trie_lookup(inst->trie,
537  &inst->deny[i].addr, inst->deny[i].prefix);
538  if (!network) {
539  cf_log_err(cs, "The network in entry %zd - 'deny = %pV' is not contained "
540  "within a previous 'allow'", i + 1, fr_box_ipaddr(inst->deny[i]));
541  return -1;
542  }
543 
544  /*
545  * We hack the AF in "deny" rules. If
546  * the lookup gets AF_UNSPEC, then we're
547  * adding a "deny" inside of a "deny".
548  */
549  if (network->af != inst->ipaddr.af) {
550  cf_log_err(cs, "The network in entry %zd - 'deny = %pV' overlaps with "
551  "another 'deny' rule", i + 1, fr_box_ipaddr(inst->deny[i]));
552  return -1;
553  }
554 
555  /*
556  * Insert the network into the trie.
557  * Lookups will return the fr_ipaddr_t of
558  * the network.
559  */
560  if (fr_trie_insert(inst->trie,
561  &inst->deny[i].addr, inst->deny[i].prefix,
562  &inst->deny[i]) < 0) {
563  cf_log_err(cs, "Failed adding 'deny = %pV' to tracking table",
564  fr_box_ipaddr(inst->deny[i]));
565  return -1;
566  }
567 
568  /*
569  * Hack it to make it a deny rule.
570  */
571  inst->deny[i].af = AF_UNSPEC;
572  }
573  }
574 
575  return 0;
576 }
