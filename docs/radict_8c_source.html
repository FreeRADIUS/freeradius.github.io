62 static int load_dicts(char const *dict_dir)
63 {
64  DIR *dir;
65  struct dirent *dp;
66 
67  INFO("Reading directory %s", dict_dir);
68 
69  dir = opendir(dict_dir);
70  if (!dir) {
71  fr_strerror_printf("Failed opening \"%s\": %s", dict_dir, fr_syserror(errno));
72  return -1;
73  }
74 
75  while ((dp = readdir(dir)) != NULL) {
76  struct stat stat_buff;
77  char *file_str;
78 
79  if (dp->d_name[0] == '.') continue;
80 
81  /*
82  * Skip the internal FreeRADIUS dictionary.
83  */
84  if (strcmp(dp->d_name, "freeradius") == 0) continue;
85 
86  file_str = talloc_asprintf(NULL, "%s/%s", dict_dir, dp->d_name);
87 
88  if (stat(file_str, &stat_buff) == -1) {
89  fr_strerror_printf("Failed stating file \"%s\": %s", file_str, fr_syserror(errno));
90  error:
91  closedir(dir);
92  talloc_free(file_str);
93  return -1;
94  }
95 
96  /*
97  * Only process directories
98  */
99  if ((stat_buff.st_mode & S_IFMT) == S_IFDIR) {
100  char *dict_file;
101  struct stat dict_stat_buff;
102  int ret;
103 
104  dict_file = talloc_asprintf(NULL, "%s/dictionary", file_str);
105  ret = stat(dict_file, &dict_stat_buff);
106  talloc_free(dict_file);
107 
108  /*
109  * If the directory contains a dictionary file,
110  * load it as a dictionary.
111  */
112  if (ret == 0) {
113  if (dict_end >= (dicts + (NUM_ELEMENTS(dicts)))) {
114  fr_strerror_printf("Reached maximum number of dictionaries");
115  goto error;
116  }
117 
118  INFO("Loading dictionary: %s/dictionary", file_str);
119  if (fr_dict_protocol_afrom_file(dict_end, dp->d_name, NULL) < 0) {
120  goto error;
121  }
122  dict_end++;
123  }
124 
125  /*
126  * For now, don't do sub-protocols.
127  */
128  }
129  talloc_free(file_str);
130  }
131  closedir(dir);
132 
133  return 0;
134 }

150 static void _fr_dict_export(fr_dict_t const *dict, uint64_t *count, uintptr_t *low, uintptr_t *high, fr_dict_attr_t const *da, unsigned int lvl)
151 {
152  unsigned int i;
153  size_t len;
154  fr_dict_attr_t const *p;
155  char flags[256];
156  fr_dict_attr_t const **children;
157 
158  fr_dict_snprint_flags(&FR_SBUFF_OUT(flags, sizeof(flags)), dict, da->type, &da->flags);
159 
160  /*
161  * Root attributes are allocated outside of the pool
162  * so it's not helpful to include them in the calculation.
163  */
164  if (!da->flags.is_root) {
165  if (low && ((uintptr_t)da < *low)) {
166  *low = (uintptr_t)da;
167  }
168  if (high && ((uintptr_t)da > *high)) {
169  *high = (uintptr_t)da;
170  }
171 
172  da_print_info_td(fr_dict_by_da(da), da);
173  }
174 
175  if (count) (*count)++;
176 
177  /*
178  * Todo - Should be fixed to use attribute walking API
179  */
180  children = dict_attr_children(da);
181  len = talloc_array_length(children);
182  for (i = 0; i < len; i++) {
183  for (p = children[i]; p; p = p->next) {
184  _fr_dict_export(dict, count, low, high, p, lvl + 1);
185  }
186  }
187 }

189 static void fr_dict_export(uint64_t *count, uintptr_t *low, uintptr_t *high, fr_dict_t *dict)
190 {
191  if (count) *count = 0;
192  if (low) *low = UINTPTR_MAX;
193  if (high) *high = 0;
194 
195  _fr_dict_export(dict, count, low, high, fr_dict_root(dict), 0);
196 }

198 int main(int argc, char *argv[])
199 {
200  char const *dict_dir = DICTDIR;
201  char c;
202  int ret = 0;
203  bool found = false;
204  bool export = false;
205 
206  TALLOC_CTX *autofree;
207 
208  /*
209  * Must be called first, so the handler is called last
210  */
211  fr_thread_local_atexit_setup();
212 
213  autofree = talloc_autofree_context();
214 
215 #ifndef NDEBUG
216  if (fr_fault_setup(autofree, getenv("PANIC_ACTION"), argv[0]) < 0) {
217  fr_perror("radict");
218  fr_exit(EXIT_FAILURE);
219  }
220 #endif
221 
222  talloc_set_log_stderr();
223 
224  fr_debug_lvl = 1;
225 
226  while ((c = getopt(argc, argv, "ED:xh")) != -1) switch (c) {
227  case 'E':
228  export = true;
229  break;
230 
231  case 'D':
232  dict_dir = optarg;
233  break;
234 
235  case 'x':
236  fr_log_fp = stdout;
237  fr_debug_lvl++;
238  break;
239 
240  case 'h':
241  default:
242  usage();
243  found = true;
244  goto finish;
245  }
246  argc -= optind;
247  argv += optind;
248 
249  /*
250  * Mismatch between the binary and the libraries it depends on
251  */
252  if (fr_check_lib_magic(RADIUSD_MAGIC_NUMBER) < 0) {
253  fr_perror("radict");
254  ret = 1;
255  goto finish;
256  }
257 
258  if (!fr_dict_global_ctx_init(autofree, dict_dir)) {
259  fr_perror("radict");
260  ret = 1;
261  goto finish;
262  }
263 
264  INFO("Loading dictionary: %s/%s", dict_dir, FR_DICTIONARY_FILE);
265 
266  if (fr_dict_internal_afrom_file(dict_end++, FR_DICTIONARY_INTERNAL_DIR) < 0) {
267  fr_perror("radict");
268  ret = 1;
269  goto finish;
270  }
271 
272  if (load_dicts(dict_dir) < 0) {
273  fr_perror("radict");
274  ret = 1;
275  goto finish;
276  }
277 
278  if (dict_end == dicts) {
279  fr_perror("radict: No dictionaries loaded");
280  ret = 1;
281  goto finish;
282  }
283 
284  if (export) {
285  fr_dict_t **dict_p = dicts;
286 
287  do {
288  uint64_t count;
289  uintptr_t high;
290  uintptr_t low;
291 
292  fr_dict_export(&count, &low, &high, *dict_p);
293  DEBUG2("Attribute count %" PRIu64, count);
294  DEBUG2("Memory allocd %zu (bytes)", talloc_total_size(*dict_p));
295  DEBUG2("Memory spread %zu (bytes)", (size_t) (high - low));
296  } while (++dict_p < dict_end);
297  }
298 
299  while (argc-- > 0) {
300  char *attr;
301  fr_dict_attr_t const *da;
302  fr_dict_t **dict_p = dicts;
303 
304  attr = *argv++;
305 
306 
307  /*
308  * Loop through all the dicts. An attribute may
309  * exist in multiple dictionaries.
310  */
311  do {
312  DEBUG2("Looking for \"%s\" in dict \"%s\"", attr, fr_dict_root(*dict_p)->name);
313 
314  da = fr_dict_attr_by_name(*dict_p, attr);
315  if (da) {
316  da_print_info_td(*dict_p, da);
317  found = true;
318  }
319  } while (++dict_p < dict_end);
320  }
321 
322 finish:
323  talloc_free(autofree);
324  return found ? ret : 64;
325 }
