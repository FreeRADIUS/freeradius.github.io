79 /** Copy function that allows overlapping memory ranges to be copied
80  *
81  * @param[out] o_start start of output buffer.
82  * @param[in] o_end end of the output buffer.
83  * @param[in] i_start start of the input buffer.
84  * @param[in] i_end end of data to copy.
85  * @return
86  * - >0 the number of bytes copied.
87  * - 0 invalid args.
88  * - <0 the number of bytes we'd need to complete the copy.
89  */
90 static inline CC_HINT(always_inline) ssize_t safecpy(char *o_start, char *o_end,
91  char const *i_start, char const *i_end)
92 {
93  ssize_t diff;
94  size_t i_len = i_end - i_start;
95 
96  if (unlikely((o_end < o_start) || (i_end < i_start))) return 0; /* sanity check */
97 
98  diff = (o_end - o_start) - (i_len);
99  if (diff < 0) return diff;
100 
101  if ((i_start > o_end) || (i_end < o_start)) { /* no-overlap */
102  memcpy(o_start, i_start, i_len);
103  } else { /* overlap */
104  memmove(o_start, i_start, i_len);
105  }
106 
107  return (i_len);
108 }

110 /** Update all markers and pointers in the set of sbuffs to point to new_buff
111  *
112  * This function should be used if the underlying buffer is realloced.
113  *
114  * @param[in] sbuff to update.
115  * @param[in] new_buff to assign to to sbuff.
116  * @param[in] new_len Length of the new buffer.
117  */
118 void fr_sbuff_update(fr_sbuff_t *sbuff, char *new_buff, size_t new_len)
119 {
120  fr_sbuff_t *sbuff_i;
121  char *old_buff; /* Current buff */
122 
123 #define update_ptr(_old_buff, _new_buff, _new_len, _field) \
124  _field = (size_t)((_field) - (_old_buff)) < _new_len ? \
125  (_new_buff) + ((_field) - (_old_buff)) : \
126  (_new_buff) + (_new_len)
127 
128  old_buff = sbuff->buff;
129 
130  /*
131  * Update pointers to point to positions
132  * in new buffer based on their relative
133  * offsets in the old buffer.
134  */
135  for (sbuff_i = sbuff; sbuff_i; sbuff_i = sbuff_i->parent) {
136  fr_sbuff_marker_t *m_i;
137 
138  sbuff_i->buff = new_buff;
139  update_ptr(old_buff, new_buff, new_len, sbuff_i->start);
140  sbuff_i->end = sbuff_i->buff + new_len;
141  *(sbuff_i->end) = '\0'; /* Re-terminate */
142  update_ptr(old_buff, new_buff, new_len, sbuff_i->p);
143 
144  for (m_i = sbuff_i->m; m_i; m_i = m_i->next) update_ptr(old_buff, new_buff, new_len, m_i->p);
145  }
146 #undef update_ptr
147 }

149 /** Shift the contents of the sbuff, returning the number of bytes we managed to shift
150  *
151  * @param[in] sbuff to shift.
152  * @param[in] shift the contents of the buffer this many bytes
153  * towards the start of the buffer.
154  * @return
155  * - 0 the shift failed due to constraining pointers.
156  * - >0 the number of bytes we managed to shift pointers
157  * in the sbuff. memmove should be used to move the
158  * existing contents of the buffer, and fill the free
159  * space at the end of the buffer with additional data.
160  */
161 size_t fr_sbuff_shift(fr_sbuff_t *sbuff, size_t shift)
162 {
163  fr_sbuff_t *sbuff_i;
164  char *buff; /* Current start */
165  size_t max_shift = shift;
166  bool reterminate = false;
167 
168  CHECK_SBUFF_INIT(sbuff);
169 
170 #define update_ptr(_buff, _shift, _field) _field = ((_field) - (_shift)) <= (_buff) ? (_buff) : ((_field) - (_shift))
171 #define update_max_shift(_buff, _max_shift, _field) if (((_buff) + (_max_shift)) > (_field)) _max_shift -= (((_buff) + (_max_shift)) - (_field))
172 
173  buff = sbuff->buff;
174 
175  /*
176  * If the sbuff is already \0 terminated
177  * and we're not working on a const buffer
178  * then assume we need to re-terminate
179  * later.
180  */
181  reterminate = (*sbuff->p == '\0') && !sbuff->is_const;
182 
183  /*
184  * Determine the maximum shift amount.
185  * Shifts are constrained by the position
186  * of pointers into the buffer.
187  */
188  for (sbuff_i = sbuff; sbuff_i; sbuff_i = sbuff_i->parent) {
189  fr_sbuff_marker_t *m_i;
190 
191  update_max_shift(buff, max_shift, sbuff_i->p);
192  if (!max_shift) return 0;
193 
194  for (m_i = sbuff_i->m; m_i; m_i = m_i->next) {
195  update_max_shift(buff, max_shift, m_i->p);
196  if (!max_shift) return 0;
197  }
198  }
199 
200  for (sbuff_i = sbuff; sbuff_i; sbuff_i = sbuff_i->parent) {
201  fr_sbuff_marker_t *m_i;
202  char *start = sbuff_i->start;
203 
204  /*
205  * Current position shifts, but stays the same
206  * relative to content.
207  */
208  update_ptr(buff, max_shift, sbuff_i->start);
209  update_ptr(buff, max_shift, sbuff_i->p);
210  update_ptr(buff, max_shift, sbuff_i->end);
211  sbuff_i->shifted += (max_shift - (start - sbuff_i->start));
212 
213  for (m_i = sbuff_i->m; m_i; m_i = m_i->next) update_ptr(buff, max_shift, m_i->p);
214  }
215 
216  if (reterminate) *sbuff->p = '\0';
217 
218 #undef update_ptr
219 #undef update_max_shift
220 
221  return max_shift;
222 }

265 /** Reallocate the current buffer
266  *
267  * @param[in] sbuff to be extended.
268  * @param[in] extension How many additional bytes should be allocated
269  * in the buffer.
270  * @return
271  * - 0 the extension operation failed.
272  * - >0 the number of bytes the buffer was extended by.
273  */
274 size_t fr_sbuff_extend_talloc(fr_sbuff_t *sbuff, size_t extension)
275 {
276  fr_sbuff_uctx_talloc_t *tctx = sbuff->uctx;
277  size_t clen, nlen, elen = extension;
278  char *new_buff;
279 
280  CHECK_SBUFF_INIT(sbuff);
281 
282  clen = sbuff->buff ? talloc_array_length(sbuff->buff) : 0;
283  /*
284  * If the current buffer size + the extension
285  * is less than init, extend the buffer to init.
286  *
287  * This can happen if the buffer has been
288  * trimmed, and then additional data is added.
289  */
290  if ((clen + elen) < tctx->init) {
291  elen = (tctx->init - clen) + 1; /* add \0 */
292  /*
293  * Double the buffer size if it's more than the
294  * requested amount.
295  */
296  } else if (elen < clen) {
297  elen = clen - 1; /* Don't double alloc \0 */
298  }
299 
300  /*
301  * Check we don't exceed the maximum buffer
302  * length.
303  */
304  if (tctx->max && ((clen + elen) > tctx->max)) {
305  elen = tctx->max - clen;
306  if (elen == 0) {
307  fr_strerror_printf("Failed extending buffer by %zu bytes to "
308  "%zu bytes, max is %zu bytes",
309  extension, clen + extension, tctx->max);
310  return 0;
311  }
312  elen += 1; /* add \0 */
313  }
314  nlen = clen + elen;
315 
316  new_buff = talloc_realloc(tctx->ctx, sbuff->buff, char, nlen);
317  if (unlikely(!new_buff)) {
318  fr_strerror_printf("Failed extending buffer by %zu bytes to %zu bytes", elen, nlen);
319  return 0;
320  }
321 
322  (void)fr_sbuff_update(sbuff, new_buff, nlen - 1); /* Shouldn't fail as we're extending */
323 
324  return elen;
325 }

411 /** Efficient terminal string search
412  *
413  * Caller should ensure that a buffer extension of needle_len bytes has been requested
414  * before calling this function.
415  *
416  * @param[in] in Sbuff to search in.
417  * @param[in] p Current position (may be ahead of in->p).
418  * @param[in] idx Fastpath index, populated by
419  * fr_sbuff_terminal_idx_init.
420  * @param[in] term terminals to search in.
421  * @return
422  * - true if found.
423  * - false if not.
424  */
425 static inline bool fr_sbuff_terminal_search(fr_sbuff_t *in, char const *p,
426  uint8_t idx[static UINT8_MAX + 1],
427  fr_sbuff_term_t const *term)
428 {
429  uint8_t term_idx;
430 
431  ssize_t start = 0;
432  ssize_t end;
433  ssize_t mid;
434 
435  size_t remaining;
436 
437  if (!term) return false; /* If there's no terminals, we don't need to search */
438 
439  end = term->len - 1;
440  term_idx = idx[(uint8_t)*p]; /* Fast path */
441  if (!term_idx) return false;
442 
443  remaining = fr_sbuff_remaining(in);
444  mid = term_idx - 1; /* Inform the mid point from the index */
445 
446  while (start <= end) {
447  char const *elem;
448  size_t tlen;
449  int ret;
450 
451  elem = term->elem[mid].str;
452  tlen = strlen(elem);
453 
454  ret = strncmp(p, elem, tlen < (size_t)remaining ? tlen : (size_t)remaining);
455  if (ret == 0) {
456  /*
457  * If we have more text than the table element, that's fine
458  */
459  if (remaining >= tlen) return true;
460 
461  /*
462  * If input was shorter than the table element we need to
463  * keep searching.
464  */
465  ret = -1;
466  }
467 
468  if (ret < 0) {
469  end = mid - 1;
470  } else {
471  start = mid + 1;
472  }
473 
474  mid = start + ((end - start) / 2); /* Avoid overflow */
475  }
476 
477  return false;
478 }

762  *
763  * As soon as a disallowed character is found the copy is stopped.

765  *

767  *

769  * @param[in] in Where to copy from. Will copy len bytes from current position in buffer.
770  * @param[in] len How many bytes to copy. If SIZE_MAX the entire buffer will be copied.

772  * @param[in] u_rules for processing unescape sequences.

779  fr_sbuff_unescape_rules_t const *u_rules)

782  bool do_escape = false; /* Track state across extensions */
783  fr_sbuff_marker_t o_s;
784  fr_sbuff_marker_t c_s;
785  fr_sbuff_marker_t end;
786 
787  uint8_t idx[UINT8_MAX + 1]; /* Fast path index */
788  size_t needle_len = 1;
789 
790  fr_sbuff_extend_status_t status = FR_SBUFF_EXTENDABLE; /* Tracks if we can extend */
791 
792  /*
793  * If we don't need to do unescaping
794  * call a more suitable function.
795  */
796  if (!u_rules || (u_rules->chr == '\0')) return fr_sbuff_out_bstrncpy_until(out, in, len, tt, u_rules);
797 
798  CHECK_SBUFF_INIT(in);
799 
800  our_in = FR_SBUFF_NO_ADVANCE(in);
801 
802  /*
803  * Chunk tracking...
804  */
805  fr_sbuff_marker(&c_s, &our_in);
806  fr_sbuff_marker(&end, &our_in);
807  fr_sbuff_marker_update_end(&end, len);
808 
809  fr_sbuff_marker(&o_s, out);
810 
811  /*
812  * Initialise the fastpath index and
813  * figure out the longest needle.
814  */
815  fr_sbuff_terminal_idx_init(&needle_len, idx, tt);
816 
817  /*
818  * ...while we have remaining data
819  */
820  while (fr_sbuff_extend_lowat(&status, &our_in, needle_len) > 0) {
821  if (fr_sbuff_was_extended(status)) fr_sbuff_marker_update_end(&end, len);
822  if (!fr_sbuff_diff(&our_in, &end)) break; /* Reached the end */
823 
824  if (do_escape) {
825  do_escape = false;
826 
827  /*
828  * Check for \x<hex><hex>
829  */
830  if (u_rules->do_hex && fr_sbuff_is_char(&our_in, 'x')) {
831  uint8_t escape;
832  fr_sbuff_marker_t m;
833 
834  fr_sbuff_marker(&m, &our_in); /* allow for backtrack */
835  fr_sbuff_advance(&our_in, 1); /* skip over the 'x' */
836 
837  if (fr_sbuff_out_uint8_hex(NULL, &escape, &our_in, false) != 2) {
838  fr_sbuff_set(&our_in, &m); /* backtrack */
839  fr_sbuff_marker_release(&m);
840  goto check_subs; /* allow sub for \x */
841  }
842 
843  if (fr_sbuff_in_char(out, escape) <= 0) {
844  fr_sbuff_set(&our_in, &m); /* backtrack */
845  fr_sbuff_marker_release(&m);
846  break;
847  }
848  fr_sbuff_marker_release(&m);
849  fr_sbuff_set(&c_s, &our_in);
850  continue;
851  }
852 
853  /*
854  * Check for <oct><oct><oct>
855  */
856  if (u_rules->do_oct && fr_sbuff_is_digit(&our_in)) {
857  uint8_t escape;
858  fr_sbuff_marker_t m;
859 
860  fr_sbuff_marker(&m, &our_in); /* allow for backtrack */
861 
862  if (fr_sbuff_out_uint8_oct(NULL, &escape, &our_in, false) != 3) {
863  fr_sbuff_set(&our_in, &m); /* backtrack */
864  fr_sbuff_marker_release(&m);
865  goto check_subs; /* allow sub for <oct> */
866  }
867 
868  if (fr_sbuff_in_char(out, escape) <= 0) {
869  fr_sbuff_set(&our_in, &m); /* backtrack */
870  fr_sbuff_marker_release(&m);
871  break;
872  }
873  fr_sbuff_marker_release(&m);
874  fr_sbuff_set(&c_s, &our_in);
875  continue;
876  }
877 
878  check_subs:
879  /*
880  * Not a recognised hex or octal escape sequence
881  * may be a substitution or a sequence that
882  * should be copied to the output buffer.
883  */
884  {
885  uint8_t c = *fr_sbuff_current(&our_in);
886 
887  if (u_rules->subs[c] == '\0') {
888  if (u_rules->skip[c] == true) goto next;
889  goto next_esc;
890  }
891 
892  /*
893  * We already copied everything up
894  * to this point, so we can now
895  * write the substituted char to
896  * the output buffer.
897  */
898  if (fr_sbuff_in_char(out, u_rules->subs[c]) <= 0) break;
899 
900  /*
901  * ...and advance past the entire
902  * escape seq in the input buffer.
903  */
904  fr_sbuff_advance(&our_in, 1);
905  fr_sbuff_set(&c_s, &our_in);
906  continue;
907  }
908  }
909 
910  next_esc:
911  if (*fr_sbuff_current(&our_in) == u_rules->chr) {
912  /*
913  * Copy out any data we got before
914  * we hit the escape char.
915  *
916  * We need to do this before we
917  * can write the escape char to
918  * the output sbuff.
919  */
920  FILL_OR_GOTO_DONE(out, &c_s, fr_sbuff_behind(&c_s));
921 
922  do_escape = true;
923  fr_sbuff_advance(&our_in, 1);
924  continue;
925  }
926 
927  next:
928  if (tt && fr_sbuff_terminal_search(in, fr_sbuff_current(&our_in), idx, tt)) break;
929  fr_sbuff_advance(&our_in, 1);
930  };
931 
932  /*
933  * Copy any remaining data over
934  */
935  FILL_OR_GOTO_DONE(out, &c_s, fr_sbuff_behind(&c_s));
936 
937 done:
938  fr_sbuff_set(in, &c_s); /* Only advance by as much as we copied */
939  *out->p = '\0';
940 
941  return fr_sbuff_marker_release_behind(&o_s);
942 }
