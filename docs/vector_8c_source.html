42 static int vector_opc_from_op(request_t *request, uint8_t const **out, uint8_t opc_buff[MILENAGE_OPC_SIZE],
43  fr_pair_t *list, uint8_t const ki[MILENAGE_KI_SIZE])
44 {
45  fr_pair_t *opc_vp;
46  fr_pair_t *op_vp;
47 
48  opc_vp = fr_pair_find_by_da(&list, attr_sim_opc);
49  if (opc_vp) {
50  if (opc_vp->vp_length != MILENAGE_OPC_SIZE) {
51  REDEBUG("&control.%s has incorrect length, expected %u bytes got %zu bytes",
52  attr_sim_opc->name, MILENAGE_OPC_SIZE, opc_vp->vp_length);
53  return -1;
54  }
55  *out = opc_vp->vp_octets;
56  return 0;
57  }
58 
59  op_vp = fr_pair_find_by_da(&list, attr_sim_op);
60  if (op_vp) {
61  if (op_vp->vp_length != MILENAGE_OP_SIZE) {
62  REDEBUG("&control.%s has incorrect length, expected %u bytes got %zu bytes",
63  attr_sim_op->name, MILENAGE_OP_SIZE, op_vp->vp_length);
64  return -1;
65  }
66  if (milenage_opc_generate(opc_buff, op_vp->vp_octets, ki) < 0) {
67  RPEDEBUG("Deriving OPc failed");
68  return -1;
69  }
70  *out = opc_buff;
71  return 0;
72  }
73 
74  *out = NULL;
75  return 1;
76 }

78 static int vector_gsm_from_ki(request_t *request, fr_pair_t *vps, int idx, fr_aka_sim_keys_t *keys)
79 {
80  fr_pair_t *ki_vp, *version_vp;
81  uint8_t opc_buff[MILENAGE_OPC_SIZE];
82  uint8_t const *opc_p;
83  uint32_t version;
84  int i;
85 
86  /*
87  * Generate a new RAND value, and derive Kc and SRES from Ki
88  */
89  ki_vp = fr_pair_find_by_da(&vps, attr_sim_ki);
90  if (!ki_vp) {
91  RDEBUG3("No &control.%sfound, not generating triplets locally", attr_sim_ki->name);
92  return 1;
93  } else if (ki_vp->vp_length != MILENAGE_KI_SIZE) {
94  REDEBUG("&control.%s has incorrect length, expected 16 bytes got %zu bytes",
95  attr_sim_ki->name, ki_vp->vp_length);
96  return -1;
97  }
98 
99  /*
100  * Check to see if we have a Ki for the IMSI, this allows us to generate the rest
101  * of the triplets.
102  */
103  version_vp = fr_pair_find_by_da(&vps, attr_sim_algo_version);
104  if (!version_vp) {
105  if (vector_opc_from_op(request, &opc_p, opc_buff, vps, ki_vp->vp_octets) < 0) return -1;
106  version = opc_p ? FR_SIM_ALGO_VERSION_VALUE_COMP128_4 : FR_SIM_ALGO_VERSION_VALUE_COMP128_3;
107  /*
108  * Version was explicitly specified, see if we can find the prerequisite
109  * attributes.
110  */
111  } else {
112  version = version_vp->vp_uint32;
113  if (version == FR_SIM_ALGO_VERSION_VALUE_COMP128_4) {
114  if (vector_opc_from_op(request, &opc_p, opc_buff, vps, ki_vp->vp_octets) < 0) return -1;
115  if (!opc_p) {
116  RPEDEBUG2("No &control.%s or &control.%s found, "
117  "can't run Milenage (COMP128-4)", attr_sim_op->name, attr_sim_opc->name);
118  return -1;
119  }
120  }
121  }
122 
123  for (i = 0; i < AKA_SIM_VECTOR_GSM_RAND_SIZE; i += sizeof(uint32_t)) {
124  uint32_t rand = fr_rand();
125  memcpy(&keys->gsm.vector[idx].rand[i], &rand, sizeof(rand));
126  }
127 
128  switch (version) {
129  case FR_SIM_ALGO_VERSION_VALUE_COMP128_1:
130  comp128v1(keys->gsm.vector[idx].sres,
131  keys->gsm.vector[idx].kc,
132  ki_vp->vp_octets,
133  keys->gsm.vector[idx].rand);
134  break;
135 
136  case FR_SIM_ALGO_VERSION_VALUE_COMP128_2:
137  comp128v23(keys->gsm.vector[idx].sres,
138  keys->gsm.vector[idx].kc,
139  ki_vp->vp_octets,
140  keys->gsm.vector[idx].rand, true);
141  break;
142 
143  case FR_SIM_ALGO_VERSION_VALUE_COMP128_3:
144  comp128v23(keys->gsm.vector[idx].sres,
145  keys->gsm.vector[idx].kc,
146  ki_vp->vp_octets,
147  keys->gsm.vector[idx].rand, false);
148  break;
149 
150  case FR_SIM_ALGO_VERSION_VALUE_COMP128_4:
151  if (milenage_gsm_generate(keys->gsm.vector[idx].sres,
152  keys->gsm.vector[idx].kc,
153  opc_p,
154  ki_vp->vp_octets,
155  keys->gsm.vector[idx].rand) < 0) {
156  RPEDEBUG2("Failed deriving GSM triplet");
157  return -1;
158  }
159  break;
160 
161  default:
162  REDEBUG("Unknown/unsupported algorithm %i", version);
163  return -1;
164  }
165 
166  /*
167  * Store for completeness...
168  */
169  memcpy(keys->auc.ki, ki_vp->vp_octets, sizeof(keys->auc.ki));
170  memcpy(keys->auc.opc, opc_p, sizeof(keys->auc.opc));
171  keys->vector_src = AKA_SIM_VECTOR_SRC_KI;
172 
173  return 0;
174 }

176 static int vector_gsm_from_triplets(request_t *request, fr_pair_t *vps,
177  int idx, fr_aka_sim_keys_t *keys)
178 {
179  fr_pair_t *rand = NULL, *sres = NULL, *kc = NULL;
180  fr_cursor_t cursor;
181  int i;
182 
183  for (i = 0, (kc = fr_cursor_iter_by_da_init(&cursor, &vps, attr_eap_aka_sim_kc));
184  (i < idx) && (kc = fr_cursor_next(&cursor));
185  i++);
186  if (!kc) {
187  RDEBUG3("No &control.%s[%i] attribute found, not using GSM triplets",
188  attr_eap_aka_sim_kc->name, idx);
189  return 1;
190  }
191  if (kc->vp_length != AKA_SIM_VECTOR_GSM_KC_SIZE) {
192  REDEBUG("&control.%s[%i] is not " STRINGIFY(AKA_SIM_VECTOR_GSM_KC_SIZE) " bytes, got %zu bytes",
193  attr_eap_aka_sim_kc->name, idx, kc->vp_length);
194  return -1;
195  }
196 
197  for (i = 0, (rand = fr_cursor_iter_by_da_init(&cursor, &vps, attr_eap_aka_sim_rand));
198  (i < idx) && (rand = fr_cursor_next(&cursor));
199  i++);
200  if (!rand) {
201  RDEBUG3("No &control.%s[%i] attribute found, not using GSM triplets",
202  attr_eap_aka_sim_rand->name, idx);
203  return 1;
204  }
205  if (rand->vp_length != AKA_SIM_VECTOR_GSM_RAND_SIZE) {
206  REDEBUG("&control.EAP-SIM-Rand[%i] is not " STRINGIFY(SIM_RAND_SIZE) " bytes, got %zu bytes",
207  idx, rand->vp_length);
208  return -1;
209  }
210 
211  for (i = 0, (sres = fr_cursor_iter_by_da_init(&cursor, &vps, attr_eap_aka_sim_sres));
212  (i < idx) && (sres = fr_cursor_next(&cursor));
213  i++);
214  if (!sres) {
215  RDEBUG3("No &control.%s[%i] attribute found, not using GSM triplets",
216  attr_eap_aka_sim_sres->name, idx);
217  return 1;
218  }
219  if (sres->vp_length != AKA_SIM_VECTOR_GSM_SRES_SIZE) {
220  REDEBUG("&control.%s[%i] is not " STRINGIFY(AKA_SIM_VECTOR_GSM_SRES_SIZE) " bytes, got %zu bytes",
221  attr_eap_aka_sim_sres->name, idx, sres->vp_length);
222  return -1;
223  }
224 
225  memcpy(keys->gsm.vector[idx].kc, kc->vp_strvalue, AKA_SIM_VECTOR_GSM_KC_SIZE);
226  memcpy(keys->gsm.vector[idx].rand, rand->vp_octets, AKA_SIM_VECTOR_GSM_RAND_SIZE);
227  memcpy(keys->gsm.vector[idx].sres, sres->vp_octets, AKA_SIM_VECTOR_GSM_SRES_SIZE);
228  keys->vector_src = AKA_SIM_VECTOR_SRC_TRIPLETS;
229 
230  return 0;
231 }

236 static int vector_gsm_from_quintuplets(request_t *request, fr_pair_t *vps,
237  int idx, fr_aka_sim_keys_t *keys)
238 {
239  fr_cursor_t cursor;
240 
241  fr_pair_t *ck = NULL, *ik = NULL, *rand = NULL, *xres = NULL;
242 
243  int i;
244 
245  /*
246  * Fetch CK
247  */
248  for (i = 0, (ck = fr_cursor_iter_by_da_init(&cursor, &vps, attr_eap_aka_sim_ck));
249  (i < idx) && (ck = fr_cursor_next(&cursor));
250  i++);
251  if (!ck) {
252  RDEBUG3("No &control.%s[%i] attribute found, not using quintuplet derivation",
253  attr_eap_aka_sim_ck->name, idx);
254  return 1;
255  }
256 
257  /*
258  * Fetch IK
259  */
260  for (i = 0, (ik = fr_cursor_iter_by_da_init(&cursor, &vps, attr_eap_aka_sim_ik));
261  (i < idx) && (ik = fr_cursor_next(&cursor));
262  i++);
263  if (!ik) {
264  RDEBUG3("No &control.%s[%i] attribute found, not using quintuplet derivation",
265  attr_eap_aka_sim_ik->name, idx);
266  return 1;
267  }
268 
269  /*
270  * Fetch RAND
271  */
272  for (i = 0, (rand = fr_cursor_iter_by_da_init(&cursor, &vps, attr_eap_aka_sim_rand));
273  (i < idx) && (rand = fr_cursor_next(&cursor));
274  i++);
275  if (!rand) {
276  RDEBUG3("No &control.%s[%i] attribute found, not using quintuplet derivation",
277  attr_eap_aka_sim_rand->name, idx);
278  return 1;
279  }
280 
281  if (rand->vp_length != AKA_SIM_VECTOR_UMTS_RAND_SIZE) {
282  REDEBUG("&control.%s[%i] incorrect length. Expected "
283  STRINGIFY(AKA_SIM_VECTOR_UMTS_RAND_SIZE) " bytes, "
284  "got %zu bytes", attr_eap_aka_sim_rand->name, idx, rand->vp_length);
285  return -1;
286  }
287 
288  /*
289  * Fetch XRES
290  */
291  for (i = 0, (xres = fr_cursor_iter_by_da_init(&cursor, &vps, attr_eap_aka_sim_xres));
292  (i < idx) && (xres = fr_cursor_next(&cursor));
293  i++);
294  if (!xres) {
295  RDEBUG3("No &control.%s[%i] attribute found, not using quintuplet derivation",
296  attr_eap_aka_sim_xres->name, idx);
297  return 1;
298  }
299 
300  memcpy(keys->gsm.vector[idx].rand, rand->vp_octets, AKA_SIM_VECTOR_GSM_RAND_SIZE);
301 
302  milenage_gsm_from_umts(keys->gsm.vector[idx].sres,
303  keys->gsm.vector[idx].kc,
304  ik->vp_octets,
305  ck->vp_octets,
306  xres->vp_octets);
307 
308  keys->vector_src = AKA_SIM_VECTOR_SRC_QUINTUPLETS;
309 
310  return 0;
311 }

313 /** Retrieve GSM triplets from sets of attributes.
314  *
315  * Hunt for a source of SIM triplets
316  *
317  * @param[in] request The current subrequest.
318  * @param[in] vps List to hunt for triplets in.
319  * @param[in] idx To write EAP-SIM triplets to.
320  * @param[in] keys EAP session keys.
321  * @param[in] src Forces triplets to be retrieved from a particular src
322  * and ensures if multiple triplets are being retrieved
323  * that they all come from the same src.
324  * @return
325  * - 1 Vector could not be retrieved from the specified src.
326  * - 0 Vector was retrieved OK and written to the specified index.
327  * - -1 Error retrieving vector from the specified src.
328  */
329 int fr_aka_sim_vector_gsm_from_attrs(request_t *request, fr_pair_t *vps,
330  int idx, fr_aka_sim_keys_t *keys, fr_aka_sim_vector_src_t *src)
331 {
332  int ret;
333 
334  fr_assert(idx >= 0 && idx < 3);
335  fr_assert((keys->vector_type == AKA_SIM_VECTOR_NONE) || (keys->vector_type == AKA_SIM_VECTOR_GSM));
336 
337  switch (*src) {
338  default:
339  case AKA_SIM_VECTOR_SRC_KI:
340  ret = vector_gsm_from_ki(request, vps, idx, keys);
341  if (ret == 0) {
342  *src = AKA_SIM_VECTOR_SRC_KI;
343  break;
344  }
345  if (ret < 0) return -1;
346  if (*src != AKA_SIM_VECTOR_SRC_AUTO) return 1;
347  FALL_THROUGH;
348 
349  case AKA_SIM_VECTOR_SRC_TRIPLETS:
350  ret = vector_gsm_from_triplets(request, vps, idx, keys);
351  if (ret == 0) {
352  *src = AKA_SIM_VECTOR_SRC_TRIPLETS;
353  break;
354  }
355  if (ret < 0) return -1;
356  if (*src != AKA_SIM_VECTOR_SRC_AUTO) return 1;
357  FALL_THROUGH;
358 
359  case AKA_SIM_VECTOR_SRC_QUINTUPLETS:
360  ret = vector_gsm_from_quintuplets(request, vps, idx, keys);
361  if (ret == 0) {
362  *src = AKA_SIM_VECTOR_SRC_QUINTUPLETS;
363  break;
364  }
365  if (ret < 0) return -1;
366  break;
367  }
368 
369  if (ret == 1) {
370  RWDEBUG("Could not find or derive data for GSM vector[%i]", idx);
371  return 1;
372  }
373 
374  if (RDEBUG_ENABLED2) {
375  RDEBUG2("GSM vector[%i]", idx);
376 
377  RINDENT();
378  /*
379  * Don't change colon indent, matches other messages later...
380  */
381  RHEXDUMP_INLINE2(keys->gsm.vector[idx].kc, AKA_SIM_VECTOR_GSM_KC_SIZE,
382  "KC :");
383  RHEXDUMP_INLINE2(keys->gsm.vector[idx].rand, AKA_SIM_VECTOR_GSM_RAND_SIZE,
384  "RAND :");
385  RHEXDUMP_INLINE2(keys->gsm.vector[idx].sres, AKA_SIM_VECTOR_GSM_SRES_SIZE,
386  "SRES :");
387  REXDENT();
388  }
389 
390  keys->vector_type = AKA_SIM_VECTOR_GSM;
391 
392  return 0;
393 }

395 static int vector_umts_from_ki(request_t *request, fr_pair_t *vps, fr_aka_sim_keys_t *keys)
396 {
397  fr_pair_t *ki_vp, *amf_vp, *sqn_vp, *version_vp;
398 
399  size_t ki_size, amf_size;
400  uint32_t version = FR_SIM_ALGO_VERSION_VALUE_MILENAGE;
401  int i;
402 
403  /*
404  * Select the algorithm (default to Milenage)
405  */
406  version_vp = fr_pair_find_by_da(&vps, attr_sim_algo_version);
407  if (version_vp) version = version_vp->vp_uint32;
408 
409  /*
410  * Get expected input sizes
411  */
412  switch (version) {
413  case FR_SIM_ALGO_VERSION_VALUE_MILENAGE:
414  ki_size = MILENAGE_KI_SIZE;
415  amf_size = MILENAGE_AMF_SIZE;
416  break;
417 
418  case FR_SIM_ALGO_VERSION_VALUE_TS_34_108_UMTS:
419  ki_size = TS_34_108_KI_SIZE;
420  amf_size = TS_34_108_AMF_SIZE;
421  break;
422 
423  /*
424  * GSM algos
425  */
426  case FR_SIM_ALGO_VERSION_VALUE_COMP128_1:
427  case FR_SIM_ALGO_VERSION_VALUE_COMP128_2:
428  case FR_SIM_ALGO_VERSION_VALUE_COMP128_3:
429  case FR_SIM_ALGO_VERSION_VALUE_COMP128_4:
430  REDEBUG("COMP128-* algorithms cannot generate UMTS vectors");
431  return -1;
432 
433  default:
434  REDEBUG("Unknown/unsupported algorithm %i", version);
435  return -1;
436  }
437 
438  /*
439  * Find the Ki VP and check its length
440  */
441  ki_vp = fr_pair_find_by_da(&vps, attr_sim_ki);
442  if (!ki_vp) {
443  RDEBUG3("No &control.%s found, not generating quintuplets locally", attr_sim_ki->name);
444  return 1;
445  } else if (ki_vp->vp_length != ki_size) {
446  REDEBUG("&control.%s has incorrect length, expected %zu bytes got %zu bytes",
447  attr_sim_ki->name, ki_size, ki_vp->vp_length);
448  return -1;
449  }
450 
451  /*
452  * Find the Sequence Number VP or default to SQN = 2
453  */
454  sqn_vp = fr_pair_find_by_da(&vps, attr_sim_sqn);
455  keys->sqn = sqn_vp ? sqn_vp->vp_uint64 : 2; /* 2 is the lowest valid SQN on our side */
456 
457  /*
458  * Check if we have an AMF value
459  */
460  amf_vp = fr_pair_find_by_da(&vps, attr_sim_amf);
461  if (amf_vp) {
462  if (amf_vp->vp_length != amf_size) {
463  REDEBUG("&control.%s has incorrect length, expected %zu bytes got %zu bytes",
464  attr_sim_amf->name, amf_size, amf_vp->vp_length);
465  return -1;
466  }
467  }
468 
469  /*
470  * Generate rand
471  */
472  for (i = 0; i < AKA_SIM_VECTOR_UMTS_RAND_SIZE; i += sizeof(uint32_t)) {
473  uint32_t rand = fr_rand();
474  memcpy(&keys->umts.vector.rand[i], &rand, sizeof(rand));
475  }
476 
477  switch (version) {
478  case FR_SIM_ALGO_VERSION_VALUE_MILENAGE:
479  {
480  uint8_t amf_buff[MILENAGE_AMF_SIZE] = { 0x00, 0x00 };
481  uint8_t sqn_buff[MILENAGE_SQN_SIZE];
482  uint8_t opc_buff[MILENAGE_OPC_SIZE];
483  uint8_t const *opc_p;
484 
485  if (vector_opc_from_op(request, &opc_p, opc_buff, vps, ki_vp->vp_octets) < 0) return -1;
486 
487  uint48_to_buff(sqn_buff, keys->sqn);
488  if (amf_vp) memcpy(amf_buff, amf_vp->vp_octets, amf_size);
489 
490  RDEBUG3("Milenage inputs");
491  RINDENT();
492  /*
493  * Don't change colon indent, matches other messages later...
494  */
495  RHEXDUMP_INLINE3(ki_vp->vp_octets, ki_size,
496  "Ki :");
497  RHEXDUMP_INLINE3(opc_p, sizeof(opc_buff),
498  "OPc :");
499  RHEXDUMP_INLINE3(sqn_buff, sizeof(sqn_buff),
500  "SQN :");
501  RHEXDUMP_INLINE3(amf_buff, sizeof(amf_buff),
502  "AMF :");
503  REXDENT();
504 
505  if (milenage_umts_generate(keys->umts.vector.autn,
506  keys->umts.vector.ik,
507  keys->umts.vector.ck,
508  keys->umts.vector.ak,
509  keys->umts.vector.xres,
510  opc_p,
511  amf_buff,
512  ki_vp->vp_octets,
513  keys->sqn,
514  keys->umts.vector.rand) < 0) {
515  RPEDEBUG2("Failed deriving UMTS Quintuplet");
516  return -1;
517  }
518  keys->umts.vector.xres_len = MILENAGE_RES_SIZE;
519 
520  /*
521  * Store the keys we used for possible AUTS
522  * validation later.
523  */
524  memcpy(keys->auc.ki, ki_vp->vp_octets, sizeof(keys->auc.ki));
525  memcpy(keys->auc.opc, opc_p, sizeof(keys->auc.opc));
526  keys->vector_src = AKA_SIM_VECTOR_SRC_KI;
527  }
528  return 0;
529 
530  /*
531  * This is a dummy algorithm and should be used for testing
532  * purposes only. It offers no security and can be trivially
533  * broken and the original Ki retrieved.
534  */
535  case FR_SIM_ALGO_VERSION_VALUE_TS_34_108_UMTS:
536  {
537  uint8_t amf_buff[TS_34_108_AMF_SIZE] = { 0x00, 0x00 };
538  uint8_t sqn_buff[TS_34_108_SQN_SIZE];
539 
540  uint48_to_buff(sqn_buff, keys->sqn);
541 
542  if (amf_vp) memcpy(amf_buff, amf_vp->vp_octets, amf_size);
543 
544  RDEBUG3("TS-34-108-UMTS inputs");
545  RINDENT();
546  /*
547  * Don't change colon indent, matches other messages later...
548  */
549  RHEXDUMP_INLINE3(
550  ki_vp->vp_octets, ki_size,
551  "Ki :");
552  RHEXDUMP_INLINE3(
553  sqn_buff, sizeof(sqn_buff),
554  "SQN :");
555  RHEXDUMP_INLINE3(
556  amf_buff, sizeof(amf_buff),
557  "AMF :");
558  REXDENT();
559 
560  if (ts_34_108_umts_generate(keys->umts.vector.autn,
561  keys->umts.vector.ik,
562  keys->umts.vector.ck,
563  keys->umts.vector.ak,
564  keys->umts.vector.xres,
565  amf_buff,
566  ki_vp->vp_octets,
567  keys->sqn,

744  * @param request The current request.
745  * @param vps List to hunt for triplets in.

747  * @param src Forces quintuplets to be retrieved from a particular src.

758
