42 static int _map_proc_client_get_vp(TALLOC_CTX *ctx, fr_pair_t **out, request_t *request,
43  map_t const *map, void *uctx)
44 {
45  client_get_vp_ctx_t *client = uctx;
46  fr_pair_t *head = NULL, *vp;
47  fr_cursor_t cursor;
48  fr_dict_attr_t const *da;
49  CONF_PAIR const *cp;
50 
51  fr_assert(ctx != NULL);
52 
53  fr_cursor_init(&cursor, &head);
54 
55  /*
56  * FIXME: allow multiple entries.
57  */
58  if (tmpl_is_attr(map->lhs)) {
59  da = tmpl_da(map->lhs);
60  } else {
61  char *attr;
62 
63  if (tmpl_aexpand(ctx, &attr, request, map->lhs, NULL, NULL) <= 0) {
64  RWDEBUG("Failed expanding string");
65  error:
66  fr_pair_list_free(&head);
67  return -1;
68  }
69 
70  da = fr_dict_attr_by_name(request->dict, attr);
71  if (!da) {
72  RWDEBUG("No such attribute '%s'", attr);
73  talloc_free(attr);
74  goto error;
75  }
76 
77  talloc_free(attr);
78  }
79 
80  for (cp = client->cp;
81  cp;
82  cp = cf_pair_find_next(client->cs, cp, client->field)) {
83  char const *value = cf_pair_value(cp);
84 
85  MEM(vp = fr_pair_afrom_da(ctx, da));
86  if (fr_pair_value_from_str(vp, value, talloc_array_length(value) - 1, '\0', false) < 0) {
87  RWDEBUG("Failed parsing value \"%pV\" for attribute %s: %s", fr_box_strvalue(value),
88  tmpl_da(map->lhs)->name, fr_strerror());
89  talloc_free(vp);
90  goto error;
91  }
92 
93  vp->op = map->op;
94  fr_cursor_append(&cursor, vp);
95 
96  if (map->op != T_OP_ADD) break; /* Create multiple attribute for multiple CONF_PAIRs */
97  }
98 
99  *out = head;
100 
101  return 0;
102 }

104 /** Map multiple attributes from a client into the request
105  *
106  * @param[in] mod_inst NULL.
107  * @param[in] proc_inst NULL.
108  * @param[in] request The current request.
109  * @param[in] client_override If NULL, use the current client, else use the client matching
110  * the ip given.
111  * @param[in] maps Head of the map list.
112  * @return
113  * - #RLM_MODULE_NOOP no rows were returned.
114  * - #RLM_MODULE_UPDATED if one or more #fr_pair_t were added to the #request_t.
115  * - #RLM_MODULE_FAIL if an error occurred.
116  */
117 static rlm_rcode_t map_proc_client(UNUSED void *mod_inst, UNUSED void *proc_inst, request_t *request,
118  fr_value_box_t **client_override, map_t const *maps)
119 {
120  rlm_rcode_t rcode = RLM_MODULE_OK;
121  map_t const *map;
122  RADCLIENT *client;
123  client_get_vp_ctx_t uctx;
124 
125  if (*client_override) {
126  fr_ipaddr_t ip;
127  char const *client_str;
128 
129  /*
130  * Concat don't asprint, as this becomes a noop
131  * in the vast majority of cases.
132  */
133  if (fr_value_box_list_concat(request, *client_override, client_override, FR_TYPE_STRING, true) < 0) {
134  REDEBUG("Failed concatenating input data");
135  return RLM_MODULE_FAIL;
136  }
137  client_str = (*client_override)->vb_strvalue;
138 
139  if (fr_inet_pton(&ip, client_str, -1, AF_UNSPEC, false, true) < 0) {
140  REDEBUG("\"%s\" is not a valid IPv4 or IPv6 address", client_str);
141  rcode = RLM_MODULE_FAIL;
142  goto finish;
143  }
144 
145  client = client_find(NULL, &ip, IPPROTO_IP);
146  if (!client) {
147  RDEBUG("No client found with IP \"%s\"", client_str);
148  rcode = RLM_MODULE_NOTFOUND;
149  goto finish;
150  }
151 
152  if (client->cs) {
153  char const *filename;
154  int line;
155 
156  filename = cf_filename(client->cs);
157  line = cf_lineno(client->cs);
158 
159  if (filename) {
160  RDEBUG2("Found client matching \"%s\". Defined in \"%s\" line %i",
161  client_str, filename, line);
162  } else {
163  RDEBUG2("Found client matching \"%s\"", client_str);
164  }
165  }
166  } else {
167  client = request->client;
168  }
169  uctx.cs = client->cs;
170 
171  RINDENT();
172  for (map = maps;
173  map != NULL;
174  map = map->next) {
175  char *field = NULL;
176 
177  if (tmpl_aexpand(request, &field, request, map->rhs, NULL, NULL) < 0) {
178  REDEBUG("Failed expanding RHS at %s", map->lhs->name);
179  rcode = RLM_MODULE_FAIL;
180  talloc_free(field);
181  break;
182  }
183 
184  uctx.cp = cf_pair_find(client->cs, field);
185  if (!uctx.cp) {
186  RDEBUG3("No matching client property \"%s\", skipping...", field);
187  goto next; /* No matching CONF_PAIR found */
188  }
189  uctx.field = field;
190 
191  /*
192  * Pass the raw data to the callback, which will
193  * create the VP and add it to the map.
194  */
195  if (map_to_request(request, map, _map_proc_client_get_vp, &uctx) < 0) {
196  rcode = RLM_MODULE_FAIL;
197  talloc_free(field);
198  break;
199  }
200  rcode = RLM_MODULE_UPDATED;
201 
202  next:
203  talloc_free(field);
204  }
205  REXDENT();
206 
207 finish:
208  return rcode;
209 }

211 /** xlat to get client config data
212  *
213  * Example:
214 @verbatim
215 %{client:[<ipaddr>.]foo}
216 @endverbatim
217  *
218  * @ingroup xlat_functions
219  */
220 static ssize_t xlat_client(TALLOC_CTX *ctx, char **out, UNUSED size_t outlen,
221  UNUSED void const *mod_inst, UNUSED void const *xlat_inst,
222  request_t *request, char const *fmt)
223 {
224  char const *value = NULL;
225  char buffer[INET6_ADDRSTRLEN], *q;
226  char const *p = fmt;
227  fr_ipaddr_t ip;
228  CONF_PAIR *cp;
229  RADCLIENT *client = NULL;
230 
231  *out = NULL;
232 
233  q = strrchr(p, '.');
234  if (q) {
235  strlcpy(buffer, p, (q + 1) - p);
236  if (fr_inet_pton(&ip, buffer, -1, AF_UNSPEC, false, true) < 0) goto request_client;
237 
238  p = q + 1;
239 
240  client = client_find(NULL, &ip, IPPROTO_IP);
241  if (!client) {
242  RDEBUG("No client found with IP \"%s\"", buffer);
243  return 0;
244  }
245  } else {
246  request_client:
247  client = request->client;
248  if (!client) {
249  RERROR("No client associated with this request");
250 
251  return -1;
252  }
253  }
254 
255  cp = cf_pair_find(client->cs, p);
256  if (!cp || !(value = cf_pair_value(cp))) {
257  if (strcmp(fmt, "shortname") == 0 && request->client->shortname) {
258  value = request->client->shortname;
259  }
260  else if (strcmp(fmt, "nas_type") == 0 && request->client->nas_type) {
261  value = request->client->nas_type;
262  }
263  if (!value) return 0;
264  }
265 
266  *out = talloc_typed_strdup(ctx, value);
267  return talloc_array_length(*out) - 1;
268 }

274 static unlang_action_t CC_HINT(nonnull) mod_authorize(rlm_rcode_t *p_result, UNUSED module_ctx_t const *mctx, request_t *request)
275 {
276  size_t length;
277  char const *value;
278  CONF_PAIR *cp;
279  RADCLIENT *c;
280  CONF_SECTION *server_cs;
281  char buffer[2048];
282 
283  /*
284  * Ensure we're only being called from the main thread,
285  * with fake packets.
286  */
287  if ((request->packet->socket.inet.src_port != 0) || (request->request_pairs != NULL) ||
288  (request->parent != NULL)) {
289  REDEBUG("Improper configuration");
290  RETURN_MODULE_NOOP;
291  }
292 
293  if (!request->client || !request->client->cs) {
294  REDEBUG("Unknown client definition");
295  RETURN_MODULE_NOOP;
296  }
297 
298  cp = cf_pair_find(request->client->cs, "directory");
299  if (!cp) {
300  REDEBUG("No directory configuration in the client");
301  RETURN_MODULE_NOOP;
302  }
303 
304  value = cf_pair_value(cp);
305  if (!value) {
306  REDEBUG("No value given for the directory entry in the client");
307  RETURN_MODULE_NOOP;
308  }
309 
310  length = strlen(value);
311  if (length > (sizeof(buffer) - 256)) {
312  REDEBUG("Directory name too long");
313  RETURN_MODULE_NOOP;
314  }
315 
316  memcpy(buffer, value, length + 1);
317  fr_inet_ntoh(&request->packet->socket.inet.src_ipaddr, buffer + length, sizeof(buffer) - length - 1);
318 
319  /*
320  * Read the buffer and generate the client.
321  */
322  if (request->client->server) {
323  server_cs = request->client->server_cs;
324 
325  } else if (request->listener) {
326  server_cs = request->listener->server_cs;
327  } else {
328  RETURN_MODULE_FAIL;
329  }
330 
331  c = client_read(buffer, server_cs, true);
332  if (!c) RETURN_MODULE_FAIL;
333 
334  /*
335  * Replace the client. This is more than a bit of a
336  * hack.
337  */
338  request->client = c;
339 
340  RETURN_MODULE_OK;
341 }
