45 /** A position marker associated with a dbuff
46  *
47  * Markers are used whenever the caller needs to access part
48  * of the underlying buffer other than the start, current
49  * position, or end.
50  *
51  * Markers are needed because if a dbuff is extended, pointers
52  * into the underlying buffer may be invalidated by a realloc
53  * or memmove.

320  *

335  .buff = buff,
336  .start = buff,
337  .p = buff,

339  .extend = fr_dbuff_extend_talloc,
340  .uctx = tctx

404 static inline size_t _fr_dbuff_extend_lowat(fr_dbuff_extend_status_t *status, fr_dbuff_t *in,
405  size_t remaining, size_t lowat)
406 {
407  size_t extended;
408 
409  if (status && !fr_dbuff_is_extendable(*status)) {
410  not_extendable:
411  if (status) *status = FR_DBUFF_NOT_EXTENDABLE;
412  return remaining;
413  }
414 
415  if (remaining >= lowat) {
416  if (status) *status = FR_DBUFF_EXTENDABLE;
417  return remaining;
418  }
419 
420  if (!in->extend || !(extended = in->extend(in, lowat - remaining))) goto not_extendable;
421 
422  if (status) *status = FR_DBUFF_EXTENDABLE_EXTENDED;
423 
424  return remaining + extended;
425 }

912  ssize_t diff;

921  memcpy(o_start, i_start, i_len);
922  } else { /* overlap */
923  memmove(o_start, i_start, i_len);
924  }

927 }
