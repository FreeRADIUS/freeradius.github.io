43 /** Returns a friendly identifier for the public key type of a certificate
44  *
45  * @param[in] cert The X509 cert to return the type of.
46  * @return the type string.
47  */
48 char const *fr_tls_utils_x509_pkey_type(X509 *cert)
49 {
50  EVP_PKEY *pkey;
51  int pkey_type;
52  char const *type_str;
53 
54  if (!cert) return NULL;
55 
56  pkey = X509_get_pubkey(cert);
57  if (!pkey) return NULL;
58 
59  pkey_type = EVP_PKEY_type(EVP_PKEY_id(pkey));
60  type_str = fr_table_str_by_value(pkey_types, pkey_type, OBJ_nid2sn(pkey_type));
61  EVP_PKEY_free(pkey);
62 
63  return type_str;
64 }

66 /** Returns the OpenSSL keyblock size
67  *
68  * @copyright (c) 2002-2016, Jouni Malinen (j@w1.fi) and contributors
69  * All Rights Reserved.
70  *
71  * These programs are licensed under the BSD license (the one with
72  * advertisement clause removed).
73  *
74  * this function shamelessly stolen from from
75  * hostap:src/crypto/tls_openssl.c:openssl_get_keyblock_size()
76  *
77  * @param[in] request The current request.
78  * @param[in] ssl The current SSL session.
79  * @return
80  * - -1 problem with the session.
81  * - >=0 length of the block.
82  */
83 int fr_tls_utils_keyblock_size_get(request_t *request, SSL *ssl)
84 {
85  const EVP_CIPHER *c;
86  const EVP_MD *h;
87 #if OPENSSL_VERSION_NUMBER < 0x10100000L
88  int md_size;
89 
90  if (ssl->enc_read_ctx == NULL || ssl->enc_read_ctx->cipher == NULL || ssl->read_hash == NULL)
91  return -1;
92 
93  c = ssl->enc_read_ctx->cipher;
94  h = EVP_MD_CTX_md(ssl->read_hash);
95  if (h)
96  md_size = EVP_MD_size(h);
97  else if (ssl->s3)
98  md_size = ssl->s3->tmp.new_mac_secret_size;
99  else
100  return -1;
101 
102  RDEBUG2("OpenSSL: keyblock size: key_len=%d MD_size=%d "
103  "IV_len=%d", EVP_CIPHER_key_length(c), md_size,
104  EVP_CIPHER_iv_length(c));
105  return 2 * (EVP_CIPHER_key_length(c) +
106  md_size +
107  EVP_CIPHER_iv_length(c));
108 #else
109  const SSL_CIPHER *ssl_cipher;
110  int cipher, digest;
111 
112  ssl_cipher = SSL_get_current_cipher(ssl);
113  if (!ssl_cipher)
114  return -1;
115  cipher = SSL_CIPHER_get_cipher_nid(ssl_cipher);
116  digest = SSL_CIPHER_get_digest_nid(ssl_cipher);
117  RDEBUG2("OpenSSL: cipher nid %d digest nid %d", cipher, digest);
118  if (cipher < 0 || digest < 0)
119  return -1;
120  c = EVP_get_cipherbynid(cipher);
121  h = EVP_get_digestbynid(digest);
122  if (!c || !h)
123  return -1;
124 
125  RDEBUG2("OpenSSL: keyblock size: key_len=%d MD_size=%d IV_len=%d",
126  EVP_CIPHER_key_length(c), EVP_MD_size(h),
127  EVP_CIPHER_iv_length(c));
128  return 2 * (EVP_CIPHER_key_length(c) + EVP_MD_size(h) +
129  EVP_CIPHER_iv_length(c));
130 #endif
131 }

133 /** Convert OpenSSL's ASN1_TIME to an epoch time
134  *
135  * @param[out] out Where to write the time_t.
136  * @param[in] asn1 The ASN1_TIME to convert.
137  * @return
138  * - 0 success.
139  * - -1 on failure.
140  */
141 int fr_tls_utils_asn1time_to_epoch(time_t *out, ASN1_TIME const *asn1)
142 {
143  struct tm t;
144  char const *p = (char const *)asn1->data, *end = p + strlen(p);
145 
146  memset(&t, 0, sizeof(t));
147 
148  if (asn1->type == V_ASN1_UTCTIME) {/* two digit year */
149  if ((end - p) < 2) {
150  fr_strerror_printf("ASN1 date string too short, expected 2 additional bytes, got %zu bytes",
151  end - p);
152  return -1;
153  }
154 
155  t.tm_year = (*(p++) - '0') * 10;
156  t.tm_year += (*(p++) - '0');
157  if (t.tm_year < 70) t.tm_year += 100;
158  } else if (asn1->type == V_ASN1_GENERALIZEDTIME) {/* four digit year */
159  if ((end - p) < 4) {
160  fr_strerror_printf("ASN1 string too short, expected 4 additional bytes, got %zu bytes",
161  end - p);
162  return -1;
163  }
164 
165  t.tm_year = (*(p++) - '0') * 1000;
166  t.tm_year += (*(p++) - '0') * 100;
167  t.tm_year += (*(p++) - '0') * 10;
168  t.tm_year += (*(p++) - '0');
169  t.tm_year -= 1900;
170  }
171 
172  if ((end - p) < 4) {
173  fr_strerror_printf("ASN1 string too short, expected 10 additional bytes, got %zu bytes",
174  end - p);
175  return -1;
176  }
177 
178  t.tm_mon = (*(p++) - '0') * 10;
179  t.tm_mon += (*(p++) - '0') - 1; // -1 since January is 0 not 1.
180  t.tm_mday = (*(p++) - '0') * 10;
181  t.tm_mday += (*(p++) - '0');
182 
183  if ((end - p) < 2) goto done;
184  t.tm_hour = (*(p++) - '0') * 10;
185  t.tm_hour += (*(p++) - '0');
186 
187  if ((end - p) < 2) goto done;
188  t.tm_min = (*(p++) - '0') * 10;
189  t.tm_min += (*(p++) - '0');
190 
191  if ((end - p) < 2) goto done;
192  t.tm_sec = (*(p++) - '0') * 10;
193  t.tm_sec += (*(p++) - '0');
194 
195  /* ASN1_TIME is UTC, so get the UTC time */
196 done:
197  *out = timegm(&t);
198 
199  return 0;
200 }
