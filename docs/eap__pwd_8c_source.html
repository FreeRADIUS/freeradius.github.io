55 /* a counter-based KDF based on NIST SP800-108 */
56 static void eap_pwd_kdf(uint8_t *key, int keylen, char const *label,
57  int label_len, uint8_t *result, int result_bit_len)
58 {
59  HMAC_CTX *hmac_ctx;
60  uint8_t digest[SHA256_DIGEST_LENGTH];
61  uint16_t i, ctr, L;
62  int result_byte_len, len = 0;
63  unsigned int mdlen = SHA256_DIGEST_LENGTH;
64  uint8_t mask = 0xff;
65 
66  MEM(hmac_ctx = HMAC_CTX_new());
67  result_byte_len = (result_bit_len + 7) / 8;
68 
69  ctr = 0;
70  L = htons(result_bit_len);
71  while (len < result_byte_len) {
72  ctr++; i = htons(ctr);
73 
74  HMAC_Init_ex(hmac_ctx, key, keylen, EVP_sha256(), NULL);
75  if (ctr > 1) HMAC_Update(hmac_ctx, digest, mdlen);
76  HMAC_Update(hmac_ctx, (uint8_t *) &i, sizeof(uint16_t));
77  HMAC_Update(hmac_ctx, (uint8_t const *)label, label_len);
78  HMAC_Update(hmac_ctx, (uint8_t *) &L, sizeof(uint16_t));
79  HMAC_Final(hmac_ctx, digest, &mdlen);
80  if ((len + (int) mdlen) > result_byte_len) {
81  memcpy(result + len, digest, result_byte_len - len);
82  } else {
83  memcpy(result + len, digest, mdlen);
84  }
85  len += mdlen;
86  HMAC_CTX_reset(hmac_ctx);
87  }
88 
89  /* since we're expanding to a bit length, mask off the excess */
90  if (result_bit_len % 8) {
91  mask <<= (8 - (result_bit_len % 8));
92  result[result_byte_len - 1] &= mask;
93  }
94 
95  HMAC_CTX_free(hmac_ctx);
96 }

110 static int legendre(BIGNUM *a, BIGNUM *p, BN_CTX *bnctx)
111 {
112  int symbol;
113  unsigned int mask;
114  BIGNUM *res, *pm1over2;
115 
116  pm1over2 = consttime_BN();
117  res = consttime_BN();
118 
119  if (!BN_sub(pm1over2, p, BN_value_one()) ||
120  !BN_rshift1(pm1over2, pm1over2) ||
121  !BN_mod_exp_mont_consttime(res, a, pm1over2, p, bnctx, NULL)) {
122  BN_free(pm1over2);
123  BN_free(res);
124  return -2;
125  }
126 
127  symbol = -1;
128  mask = const_time_eq(BN_is_word(res, 1), 1);
129  symbol = const_time_select_int(mask, 1, symbol);
130  mask = const_time_eq(BN_is_zero(res), 1);
131  symbol = const_time_select_int(mask, -1, symbol);
132 
133  BN_free(pm1over2);
134  BN_free(res);
135 
136  return symbol;

170 static int is_quadratic_residue(BIGNUM *val, BIGNUM *p, BIGNUM *qr, BIGNUM *qnr, BN_CTX *bnctx)
171 {
172  int offset, check, ret = 0;
173  BIGNUM *r = NULL, *pm1 = NULL, *res = NULL, *qr_or_qnr = NULL;
174  unsigned int mask;
175  unsigned char *qr_bin = NULL, *qnr_bin = NULL, *qr_or_qnr_bin = NULL;
176 
177  if (((r = consttime_BN()) == NULL) ||
178  ((res = consttime_BN()) == NULL) ||
179  ((qr_or_qnr = consttime_BN()) == NULL) ||
180  ((pm1 = consttime_BN()) == NULL)) {
181  ret = -2;
182  goto fail;
183  }
184 
185  if (((qr_bin = (unsigned char *)malloc(BN_num_bytes(p))) == NULL) ||
186  ((qnr_bin = (unsigned char *)malloc(BN_num_bytes(p))) == NULL) ||
187  ((qr_or_qnr_bin = (unsigned char *)malloc(BN_num_bytes(p))) == NULL)) {
188  ret = -2;
189  goto fail;
190  }
191 
192  /*
193  * we select binary in constant time so make them binary
194  */
195  memset(qr_bin, 0, BN_num_bytes(p));
196  memset(qnr_bin, 0, BN_num_bytes(p));
197  memset(qr_or_qnr_bin, 0, BN_num_bytes(p));
198 
199  offset = BN_num_bytes(p) - BN_num_bytes(qr);
200  BN_bn2bin(qr, qr_bin + offset);
201 
202  offset = BN_num_bytes(p) - BN_num_bytes(qnr);
203  BN_bn2bin(qnr, qnr_bin + offset);
204 
205  /*
206  * r = (random() mod p-1) + 1
207  */
208  BN_sub(pm1, p, BN_value_one());
209  BN_rand_range(r, pm1);
210  BN_add(r, r, BN_value_one());
211 
212  BN_copy(res, val);
213 
214  /*
215  * res = val * r * r which ensures res != val but has same quadratic residocity
216  */
217  BN_mod_mul(res, res, r, p, bnctx);
218  BN_mod_mul(res, res, r, p, bnctx);
219 
220  /*
221  * if r is even (mask is -1) then multiply by qnr and our check is qnr
222  * otherwise multiply by qr and our check is qr
223  */
224  mask = const_time_is_zero(BN_is_odd(r));
225  const_time_select_bin(mask, qnr_bin, qr_bin, BN_num_bytes(p), qr_or_qnr_bin);
226  BN_bin2bn(qr_or_qnr_bin, BN_num_bytes(p), qr_or_qnr);
227  BN_mod_mul(res, res, qr_or_qnr, p, bnctx);
228  check = const_time_select_int(mask, -1, 1);
229 
230  if ((ret = legendre(res, p, bnctx)) == -2) {
231  ret = -1; /* just say no it's not */
232  goto fail;
233  }
234  mask = const_time_eq(ret, check);
235  ret = const_time_select_int(mask, 1, 0);
236 
237 fail:
238  if (qr_bin != NULL) free(qr_bin);
239  if (qnr_bin != NULL) free(qnr_bin);
240  if (qr_or_qnr_bin != NULL) free(qr_or_qnr_bin);
241  BN_free(r);
242  BN_free(res);
243  BN_free(qr_or_qnr);
244  BN_free(pm1);
245 
246  return ret;
247 }

249 int compute_password_element (request_t *request, pwd_session_t *session, uint16_t grp_num,
250  char const *password, int password_len,
251  char const *id_server, int id_server_len,
252  char const *id_peer, int id_peer_len,
253  uint32_t *token, BN_CTX *bnctx)
254 {
255  BIGNUM *x_candidate = NULL, *rnd = NULL, *y_sqrd = NULL, *qr = NULL, *qnr = NULL;
256  HMAC_CTX *ctx = NULL;
257  uint8_t pwe_digest[SHA256_DIGEST_LENGTH], *prfbuf = NULL, *xbuf = NULL, *pm1buf = NULL, ctr;
258  int nid, is_odd, primebitlen, primebytelen, ret = 0, found = 0, mask;
259  int save, i, rbits, qr_or_qnr, save_is_odd = 0, cmp;
260  unsigned int skip;
261 
262  ctx = HMAC_CTX_new();
263  if (ctx == NULL) {
264  DEBUG("failed allocating HMAC context");
265  goto fail;
266  }
267 
268  switch (grp_num) { /* from IANA registry for IKE D-H groups */
269  case 19:
270  nid = NID_X9_62_prime256v1;
271  break;
272 
273  case 20:
274  nid = NID_secp384r1;
275  break;
276 
277  case 21:
278  nid = NID_secp521r1;
279  break;
280 
281  case 25:
282  nid = NID_X9_62_prime192v1;
283  break;
284 
285  case 26:
286  nid = NID_secp224r1;
287  break;
288 
289  default:
290  DEBUG("unknown group %d", grp_num);
291  goto fail;
292  }
293 
294  session->pwe = NULL;
295  session->order = NULL;
296  session->prime = NULL;
297 
298  if ((session->group = EC_GROUP_new_by_curve_name(nid)) == NULL) {
299  DEBUG("unable to create EC_GROUP");
300  goto fail;
301  }
302 
303  if (((rnd = consttime_BN()) == NULL) ||
304  ((session->pwe = EC_POINT_new(session->group)) == NULL) ||
305  ((session->order = consttime_BN()) == NULL) ||
306  ((session->prime = consttime_BN()) == NULL) ||
307  ((qr = consttime_BN()) == NULL) ||
308  ((qnr = consttime_BN()) == NULL) ||
309  ((x_candidate = consttime_BN()) == NULL) ||
310  ((y_sqrd = consttime_BN()) == NULL)) {
311  DEBUG("unable to create bignums");
312  goto fail;
313  }
314 
315  if (!EC_GROUP_get_curve_GFp(session->group, session->prime, NULL, NULL, NULL)) {
316  DEBUG("unable to get prime for GFp curve");
317  goto fail;
318  }
319 
320  if (!EC_GROUP_get_order(session->group, session->order, NULL)) {
321  DEBUG("unable to get order for curve");
322  goto fail;
323  }
324 
325  primebitlen = BN_num_bits(session->prime);
326  primebytelen = BN_num_bytes(session->prime);
327  if ((prfbuf = talloc_zero_array(session, uint8_t, primebytelen)) == NULL) {
328  DEBUG("unable to alloc space for prf buffer");
329  goto fail;
330  }
331  if ((xbuf = talloc_zero_array(request, uint8_t, primebytelen)) == NULL) {
332  DEBUG("unable to alloc space for x buffer");
333  goto fail;
334  }
335  if ((pm1buf = talloc_zero_array(request, uint8_t, primebytelen)) == NULL) {
336  DEBUG("unable to alloc space for pm1 buffer");
337  goto fail;
338  }
339 
340  /*
341  * derive random quadradic residue and quadratic non-residue
342  */
343  do {
344  BN_rand_range(qr, session->prime);
345  } while (legendre(qr, session->prime, bnctx) != 1);
346 
347  do {
348  BN_rand_range(qnr, session->prime);
349  } while (legendre(qnr, session->prime, bnctx) != -1);
350 
351  if (!BN_sub(rnd, session->prime, BN_value_one())) {
352  goto fail;
353  }
354  BN_bn2bin(rnd, pm1buf);
355 
356  save_is_odd = 0;
357  found = 0;
358  memset(xbuf, 0, primebytelen);
359  ctr = 0;
360  while (ctr < 40) {
361  ctr++;
362 
363  /*
364  * compute counter-mode password value and stretch to prime
365  * pwd-seed = H(token | peer-id | server-id | password |
366  * counter)
367  */
368  HMAC_Init_ex(ctx, allzero, SHA256_DIGEST_LENGTH, EVP_sha256(),NULL);
369  HMAC_Update(ctx, (uint8_t *)token, sizeof(*token));
370  HMAC_Update(ctx, (uint8_t const *)id_peer, id_peer_len);
371  HMAC_Update(ctx, (uint8_t const *)id_server, id_server_len);
372  HMAC_Update(ctx, (uint8_t const *)password, password_len);
373  HMAC_Update(ctx, (uint8_t *)&ctr, sizeof(ctr));
374  pwd_hmac_final(ctx, pwe_digest);
375 
376  BN_bin2bn(pwe_digest, SHA256_DIGEST_LENGTH, rnd);
377  eap_pwd_kdf(pwe_digest, SHA256_DIGEST_LENGTH, "EAP-pwd Hunting And Pecking",
378  strlen("EAP-pwd Hunting And Pecking"), prfbuf, primebitlen);
379 
380  /*
381  * eap_pwd_kdf() returns a string of bits 0..primebitlen but
382  * BN_bin2bn will treat that string of bits as a big endian
383  * number. If the primebitlen is not an even multiple of 8
384  * then excessive bits-- those _after_ primebitlen-- so now
385  * we have to shift right the amount we masked off.
386  */
387  if (primebitlen % 8) {
388  rbits = 8 - (primebitlen % 8);
389  for (i = primebytelen - 1; i > 0; i--) {
390  prfbuf[i] = (prfbuf[i - 1] << (8 - rbits)) | (prfbuf[i] >> rbits);
391  }
392  prfbuf[0] >>= rbits;
393  }
394  BN_bin2bn(prfbuf, primebytelen, x_candidate);
395 
396  /*
397  * it would've been better if the spec reduced the candidate
398  * modulo the prime but it didn't. So if the candidate >= prime
399  * we need to skip it but still run through the operations below
400  */
401  cmp = const_time_memcmp(pm1buf, prfbuf, primebytelen);
402  skip = const_time_fill_msb((unsigned int)cmp);
403 
404  /*
405  * need to unambiguously identify the solution, if there is
406  * one..
407  */
408  is_odd = BN_is_odd(rnd) ? 1 : 0;
409 
410  /*
411  * check whether x^3 + a*x + b is a quadratic residue
412  *
413  * save the first quadratic residue we find in the loop but do
414  * it in constant time.
415  */
416  do_equation(session->group, y_sqrd, x_candidate, bnctx);
417  qr_or_qnr = is_quadratic_residue(y_sqrd, session->prime, qr, qnr, bnctx);
418 
419  /*
420  * if the candidate >= prime then we want to skip it
421  */
422  qr_or_qnr = const_time_select(skip, 0, qr_or_qnr);
423 
424  /*
425  * if we haven't found PWE yet (found = 0) then mask will be true,
426  * if we have found PWE then mask will be false
427  */
428  mask = const_time_select(found, 0, -1);
429 
430  /*
431  * save will be 1 if we want to save this value-- i.e. we haven't
432  * found PWE yet and this is a quadratic residue-- and 0 otherwise
433  */
434  save = const_time_select(mask, qr_or_qnr, 0);
435 
436  /*
437  * mask will be true (-1) if we want to save this and false (0)
438  * otherwise
439  */
440  mask = const_time_eq(save, 1);
441 
442  const_time_select_bin(mask, prfbuf, xbuf, primebytelen, xbuf);
443  save_is_odd = const_time_select(mask, is_odd, save_is_odd);
444  found = const_time_select(mask, -1, found);
445  }
446 
447  /*
448  * now we can savely construct PWE
449  */
450  BN_bin2bn(xbuf, primebytelen, x_candidate);
451  if (!EC_POINT_set_compressed_coordinates_GFp(session->group, session->pwe,
452  x_candidate, save_is_odd, NULL)) {
453  goto fail;
454  }
455 
456  session->group_num = grp_num;
457  if (0) {
458  fail: /* DON'T free session, it's in handler->opaque */
459  ret = -1;
460  }
461 
462  /* cleanliness and order.... */
463  BN_clear_free(x_candidate);
464  BN_clear_free(y_sqrd);
465  BN_clear_free(qr);
466  BN_clear_free(qnr);
467  BN_clear_free(rnd);
468 
469  if (prfbuf) talloc_free(prfbuf);
470  if (xbuf) talloc_free(xbuf);
471  if (pm1buf) talloc_free(pm1buf);
472 
473  HMAC_CTX_free(ctx);
474 
475  return ret;
476 }

518 int process_peer_commit(request_t *request, pwd_session_t *session, uint8_t *in, size_t in_len, BN_CTX *bn_ctx)
519 {

521  size_t data_len;
522  BIGNUM *x = NULL, *y = NULL, *cofactor = NULL;
523  EC_POINT *K = NULL, *point = NULL;
524  int ret = 1;
525 
526  MEM(session->peer_scalar = BN_new());
527  MEM(session->k = BN_new());
528  MEM(session->peer_element = EC_POINT_new(session->group));
529  MEM(point = EC_POINT_new(session->group));
530  MEM(K = EC_POINT_new(session->group));
531 
532  MEM(cofactor = BN_new());
533  MEM(x = BN_new());
534  MEM(y = BN_new());
535 
536  if (!EC_GROUP_get_cofactor(session->group, cofactor, NULL)) {
537  REDEBUG("Unable to get group co-factor");
538  goto finish;
539  }
540 
541  /* element, x then y, followed by scalar */
542  ptr = (uint8_t *)in;
543  data_len = BN_num_bytes(session->prime);
544 
545  /*
546  * Did the peer send enough data?
547  */
548  if (in_len < (2 * data_len + BN_num_bytes(session->order))) {
549  REDEBUG("Invalid commit packet");
550  goto finish;
551  }
552 
553  BN_bin2bn(ptr, data_len, x);
554  ptr += data_len;
555  BN_bin2bn(ptr, data_len, y);
556  ptr += data_len;
557 
558  data_len = BN_num_bytes(session->order);
559  BN_bin2bn(ptr, data_len, session->peer_scalar);
560 
561  /* validate received scalar */
562  if (BN_is_zero(session->peer_scalar) ||
563  BN_is_one(session->peer_scalar) ||
564  BN_cmp(session->peer_scalar, session->order) >= 0) {
565  REDEBUG("Peer's scalar is not within the allowed range");
566  goto finish;
567  }
568 
569  if (!EC_POINT_set_affine_coordinates_GFp(session->group, session->peer_element, x, y, bn_ctx)) {
570  REDEBUG("Unable to get coordinates of peer's element");
571  goto finish;
572  }
573 
574  /* validate received element */
575  if (!EC_POINT_is_on_curve(session->group, session->peer_element, bn_ctx) ||
576  EC_POINT_is_at_infinity(session->group, session->peer_element)) {
577  REDEBUG("Peer's element is not a point on the elliptic curve");
578  goto finish;
579  }
580 
581  /* check to ensure peer's element is not in a small sub-group */
582  if (BN_cmp(cofactor, BN_value_one())) {
583  if (!EC_POINT_mul(session->group, point, NULL, session->peer_element, cofactor, NULL)) {
584  REDEBUG("Unable to multiply element by co-factor");
585  goto finish;
586  }
587

589  REDEBUG("Peer's element is in small sub-group");
590  goto finish;
591  }
592  }
593

595  if (BN_cmp(session->peer_scalar, session->my_scalar) == 0 ||
596  EC_POINT_cmp(session->group, session->peer_element, session->my_element, bn_ctx) == 0) {
597  REDEBUG("Reflection attack detected");
598  goto finish;

600

602  if ((!EC_POINT_mul(session->group, K, NULL, session->pwe, session->peer_scalar, bn_ctx)) ||
603  (!EC_POINT_add(session->group, K, K, session->peer_element, bn_ctx)) ||

605  REDEBUG("Unable to compute shared key, k");
606  goto finish;

644 int compute_server_confirm(request_t *request, pwd_session_t *session, uint8_t *out, BN_CTX *bn_ctx)
645 {
646  BIGNUM *x = NULL, *y = NULL;
647  HMAC_CTX *hmac_ctx = NULL;
648  uint8_t *cruft = NULL;
649  int offset, req = -1;
650 
651  /*
652  * Each component of the cruft will be at most as big as the prime
653  */
654  MEM(cruft = talloc_zero_array(session, uint8_t, BN_num_bytes(session->prime)));
655  MEM(x = BN_new());
656  MEM(y = BN_new());
657 
658  /*
659  * commit is H(k | server_element | server_scalar | peer_element |
660  * peer_scalar | ciphersuite)
661  */
662  MEM(hmac_ctx = HMAC_CTX_new());
663  HMAC_Init_ex(hmac_ctx, allzero, SHA256_DIGEST_LENGTH, EVP_sha256(), NULL);
664 
665  /*
666  * Zero the memory each time because this is mod prime math and some
667  * value may start with a few zeros and the previous one did not.
668  *
669  * First is k
670  */
671  offset = BN_num_bytes(session->prime) - BN_num_bytes(session->k);
672  BN_bn2bin(session->k, cruft + offset);
673  HMAC_Update(hmac_ctx, cruft, BN_num_bytes(session->prime));
674 
675  /*
676  * next is server element: x, y
677  */
678  if (!EC_POINT_get_affine_coordinates_GFp(session->group, session->my_element, x, y, bn_ctx)) {
679  REDEBUG("Unable to get coordinates of server element");
680  goto finish;
681  }
682  memset(cruft, 0, BN_num_bytes(session->prime));
683  offset = BN_num_bytes(session->prime) - BN_num_bytes(x);
684  BN_bn2bin(x, cruft + offset);
685  HMAC_Update(hmac_ctx, cruft, BN_num_bytes(session->prime));
686 
687  memset(cruft, 0, BN_num_bytes(session->prime));
688  offset = BN_num_bytes(session->prime) - BN_num_bytes(y);
689  BN_bn2bin(y, cruft + offset);
690  HMAC_Update(hmac_ctx, cruft, BN_num_bytes(session->prime));
691 
692  /*
693  * and server scalar
694  */
695  memset(cruft, 0, BN_num_bytes(session->prime));
696  offset = BN_num_bytes(session->order) - BN_num_bytes(session->my_scalar);
697  BN_bn2bin(session->my_scalar, cruft + offset);
698  HMAC_Update(hmac_ctx, cruft, BN_num_bytes(session->order));
699 
700  /*
701  * next is peer element: x, y
702  */
703  if (!EC_POINT_get_affine_coordinates_GFp(session->group, session->peer_element, x, y, bn_ctx)) {
704  REDEBUG("Unable to get coordinates of peer's element");
705  goto finish;
706  }
707 
708  memset(cruft, 0, BN_num_bytes(session->prime));
709  offset = BN_num_bytes(session->prime) - BN_num_bytes(x);
710  BN_bn2bin(x, cruft + offset);
711  HMAC_Update(hmac_ctx, cruft, BN_num_bytes(session->prime));
712 
713  memset(cruft, 0, BN_num_bytes(session->prime));
714  offset = BN_num_bytes(session->prime) - BN_num_bytes(y);
715  BN_bn2bin(y, cruft + offset);
716  HMAC_Update(hmac_ctx, cruft, BN_num_bytes(session->prime));
717 
718  /*
719  * and peer scalar
720  */
721  memset(cruft, 0, BN_num_bytes(session->prime));
722  offset = BN_num_bytes(session->order) - BN_num_bytes(session->peer_scalar);
723  BN_bn2bin(session->peer_scalar, cruft + offset);
724  HMAC_Update(hmac_ctx, cruft, BN_num_bytes(session->order));
725 
726  /*
727  * finally, ciphersuite
728  */
729  HMAC_Update(hmac_ctx, (uint8_t *)&session->ciphersuite, sizeof(session->ciphersuite));
730 
731  pwd_hmac_final(hmac_ctx, out);
732 
733  req = 0;
734 
735 finish:
736  HMAC_CTX_free(hmac_ctx);
737  talloc_free(cruft);
738  BN_free(x);
739  BN_free(y);
740 
741  return req;

843 int compute_keys(UNUSED request_t *request, pwd_session_t *session, uint8_t *peer_confirm, uint8_t *msk, uint8_t *emsk)
844 {
845  HMAC_CTX *hmac_ctx;
846  uint8_t mk[SHA256_DIGEST_LENGTH], *cruft;
847  uint8_t session_id[SHA256_DIGEST_LENGTH + 1];
848  uint8_t msk_emsk[128]; /* 64 each */
849  int offset;
850 
851  MEM(cruft = talloc_array(session, uint8_t, BN_num_bytes(session->prime)));
852  MEM(hmac_ctx = HMAC_CTX_new());
853 
854  /*
855  * first compute the session-id = TypeCode | H(ciphersuite | scal_p |
856  * scal_s)
857  */
858  session_id[0] = FR_EAP_METHOD_PWD;
859  HMAC_Init_ex(hmac_ctx, allzero, SHA256_DIGEST_LENGTH, EVP_sha256(), NULL);
860  HMAC_Update(hmac_ctx, (uint8_t *)&session->ciphersuite, sizeof(session->ciphersuite));
861  offset = BN_num_bytes(session->order) - BN_num_bytes(session->peer_scalar);
862  memset(cruft, 0, BN_num_bytes(session->prime));
863  BN_bn2bin(session->peer_scalar, cruft + offset);
864  HMAC_Update(hmac_ctx, cruft, BN_num_bytes(session->order));
865  offset = BN_num_bytes(session->order) - BN_num_bytes(session->my_scalar);
866  memset(cruft, 0, BN_num_bytes(session->prime));
867  BN_bn2bin(session->my_scalar, cruft + offset);
868  HMAC_Update(hmac_ctx, cruft, BN_num_bytes(session->order));
869  pwd_hmac_final(hmac_ctx, (uint8_t *)&session_id[1]);
870 
871  /* then compute MK = H(k | commit-peer | commit-server) */
872  HMAC_Init_ex(hmac_ctx, allzero, SHA256_DIGEST_LENGTH, EVP_sha256(), NULL);
873 
874  memset(cruft, 0, BN_num_bytes(session->prime));
875  offset = BN_num_bytes(session->prime) - BN_num_bytes(session->k);
876  BN_bn2bin(session->k, cruft + offset);
877  HMAC_Update(hmac_ctx, cruft, BN_num_bytes(session->prime));
878 
879  HMAC_Update(hmac_ctx, peer_confirm, SHA256_DIGEST_LENGTH);
880 
881  HMAC_Update(hmac_ctx, session->my_confirm, SHA256_DIGEST_LENGTH);
882 
883  pwd_hmac_final(hmac_ctx, mk);
884 
885  /* stretch the mk with the session-id to get MSK | EMSK */
886  eap_pwd_kdf(mk, SHA256_DIGEST_LENGTH, (char const *)session_id,
887  SHA256_DIGEST_LENGTH + 1, msk_emsk, 1024); /* it's bits, ((64 + 64) * 8) */
888 
889  memcpy(msk, msk_emsk, 64);
890  memcpy(emsk, msk_emsk + 64, 64);
891 
892  HMAC_CTX_free(hmac_ctx);
893  talloc_free(cruft);
894  return 0;
