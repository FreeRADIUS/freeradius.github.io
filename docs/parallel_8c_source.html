32 /** When the chld is done, tell the parent that we've exited.
33  *
34  */
35 static unlang_action_t unlang_parallel_child_done(UNUSED rlm_rcode_t *p_result, UNUSED int *priority, request_t *request, void *uctx)
36 {
37  unlang_parallel_child_t *child = uctx;
38 
39  /*
40  * If we have a parent, then we're running synchronously
41  * with it. Tell the parent that we've exited, and it
42  * can continue.
43  *
44  * Otherwise we're a detached child, and we don't tell
45  * the parent anything. Because we have that kind of
46  * relationship.
47  *
48  * Note that we call unlang_interpret_resumable() here
49  * because unlang_parallel_process() calls
50  * unlang_interpret(), and NOT child->async->process.
51  */
52  if (request->parent) {
53  child->state = CHILD_EXITED;
54  unlang_interpret_resumable(request->parent);
55  }
56 
57  /*
58  * Don't change frame->result, it's the result of the child.
59  */
60 
61  return UNLANG_ACTION_CALCULATE_RESULT;
62 }

65 /** Run one or more sub-sections from the parallel section.
66  *
67  */
68 static unlang_action_t unlang_parallel_process(rlm_rcode_t *p_result, request_t *request)
69 {
70  unlang_stack_t *stack = request->stack;
71  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
72  unlang_parallel_state_t *state = talloc_get_type_abort(frame->state, unlang_parallel_state_t);
73 
74  int i, priority;
75  rlm_rcode_t result;
76  unlang_parallel_child_state_t child_state = CHILD_DONE; /* hope that we're done */
77  request_t *child;
78 
79  /*
80  * If the children should be created detached, we return
81  * "noop". This function then creates the children,
82  * detaches them, and returns.
83  */
84  if (state->detach) {
85  state->priority = 0;
86  state->result = RLM_MODULE_NOOP;
87  }
88 
89  /*
90  * Loop over all the children.
91  *
92  * We always service the parallel section from top to
93  * bottom, and we always service all of it.
94  */
95  for (i = 0; i < state->num_children; i++) {
96  switch (state->children[i].state) {
97  /*
98  * Create the child and then run it.
99  */
100  case CHILD_INIT:
101  RDEBUG3("parallel child %d is INIT", i);
102  fr_assert(state->children[i].instruction != NULL);
103  child = unlang_io_subrequest_alloc(request,
104  request->dict, state->detach);
105  child->packet->code = request->packet->code;
106 
107  if (state->clone) {
108  /*
109  * Note that we do NOT copy the
110  * Session-State list! That
111  * contains state information for
112  * the parent.
113  */
114  if ((fr_pair_list_copy(child->packet,
115  &child->request_pairs,
116  &request->request_pairs) < 0) ||
117  (fr_pair_list_copy(child->reply,
118  &child->reply_pairs,
119  &request->reply_pairs) < 0) ||
120  (fr_pair_list_copy(child,
121  &child->control,
122  &request->control_pairs) < 0)) {
123  REDEBUG("failed copying lists to clone");
124  for (i = 0; i < state->num_children; i++) TALLOC_FREE(state->children[i].child);
125 
126  *p_result = RLM_MODULE_FAIL;
127  return UNLANG_ACTION_CALCULATE_RESULT;
128  }
129  }
130 
131  /*
132  * Push a top frame, followed by a frame
133  * which signals us that the child is
134  * done, followed by the instruction to
135  * run in the child.
136  */
137  if ((unlang_interpret_push(child, NULL, RLM_MODULE_NOOP,
138  UNLANG_NEXT_STOP, UNLANG_TOP_FRAME) < 0) ||
139  (unlang_interpret_push_function(child, unlang_parallel_child_done, NULL, &state->children[i]) < 0) ||
140  (unlang_interpret_push(child,
141  state->children[i].instruction, RLM_MODULE_FAIL,
142  UNLANG_NEXT_STOP, UNLANG_SUB_FRAME) < 0)) {
143  *p_result = RLM_MODULE_FAIL;
144  return UNLANG_ACTION_STOP_PROCESSING;
145  }
146 
147  /*
148  * It is often useful to create detached
149  * children in parallel.
150  */
151  if (state->detach) {
152  state->children[i].state = CHILD_DONE;
153  state->children[i].instruction = NULL;
154 
155  fr_assert(request->backlog != NULL);
156 
157  /*
158  * Detach the child, and insert
159  * it into the backlog.
160  */
161  if (unlang_detached_child_init(child) < 0) {
162  talloc_free(child);
163  child_state = CHILD_DONE;
164  state->result = RLM_MODULE_FAIL;
165  break;
166  }
167  continue;
168  }
169 
170  state->children[i].child = child;
171  state->children[i].state = CHILD_RUNNABLE;
172 
173  FALL_THROUGH;
174 
175  /*
176  * Run this entry.
177  */
178  case CHILD_RUNNABLE:
179  runnable:
180  RDEBUG2("parallel - running entry %d/%d", i + 1, state->num_children);
181 
182  /*
183  * Note that we do NOT call child->async-process()
184  *
185  * Doing that will end up calling
186  * unlang_parallel_child_done(), and all
187  * kinds of bad things happen. We may
188  * want to fix that in the future.
189  */
190  result = unlang_interpret(state->children[i].child);
191  if (result == RLM_MODULE_YIELD) {
192  state->children[i].state = CHILD_YIELDED;
193  child_state = CHILD_YIELDED;
194  continue;
195  }
196 
197  RDEBUG3("parallel child %s returns %s", state->children[i].child->name,
198  fr_table_str_by_value(mod_rcode_table, result, "<invalid>"));
199 
200  fr_assert(result < NUM_ELEMENTS(state->children[i].instruction->actions));
201 
202  /*
203  * Re-run all of the logic from interpret.c
204  *
205  * @todo - Make this a common function?
206  */
207 
208  /*
209  * Remember this before we delete the
210  * reference to 'instruction'.
211  */
212  priority = state->children[i].instruction->actions[result];
213 
214  /*
215  * Clean up the state entry.
216  */
217  state->children[i].state = CHILD_DONE;
218  TALLOC_FREE(state->children[i].child);
219  state->children[i].instruction = NULL;
220 
221  /*
222  * return is "stop processing the
223  * parallel section".
224  */
225  if (priority == MOD_ACTION_RETURN) {
226  RDEBUG2("child %d/%d says 'return' - skipping the remaining children",
227  i, state->num_children);
228 
229  /*
230  * Fall through to processing the
231  * priorities and return codes.
232  */
233  i = state->num_children;
234  priority = 0;
235  child_state = CHILD_DONE;
236  }
237 
238  /*
239  * Reject is just reject.
240  */
241  if (priority == MOD_ACTION_REJECT) {
242  priority = 0;
243  result = RLM_MODULE_REJECT;
244  }
245 
246  /*
247  * Do priority over-ride.
248  */
249  if (priority > state->priority) {
250  state->result = result;
251  state->priority = priority;
252 
253  RDEBUG4("** [%i] %s - over-riding result from higher priority to (%s %d)",
254  stack->depth, __FUNCTION__,
255  fr_table_str_by_value(mod_rcode_table, result, "<invalid>"),
256  priority);
257  }
258 
259  /*
260  * Another child has yielded, so we
261  * remember the yield instead of the fact
262  * that we're done.
263  */
264  if (child_state == CHILD_YIELDED) continue;
265 
266  fr_assert(child_state == CHILD_DONE);
267  break;
268 
269  /*
270  * Not ready to run.
271  */
272  case CHILD_YIELDED:
273  fr_assert(state->children[i].child != NULL);
274 
275  if (state->children[i].child->runnable_id == -2) { /* see unlang_interpret_resumable() */
276  (void) fr_heap_extract(state->children[i].child->backlog,
277  state->children[i].child);
278  goto runnable;
279  }
280 
281  fr_assert(state->children[i].instruction != NULL);
282  RDEBUG3("parallel child %s is already YIELDED", state->children[i].child->name);
283  child_state = CHILD_YIELDED;
284  continue;
285 
286  case CHILD_EXITED:
287  RDEBUG3("parallel child %d has already EXITED", i);
288  state->children[i].state = CHILD_DONE;
289  state->children[i].child = NULL; // someone else freed this somewhere
290  state->children[i].instruction = NULL;
291  FALL_THROUGH;
292 
293  /*
294  * Don't need to call this any more.
295  */
296  case CHILD_DONE:
297  RDEBUG3("parallel child %d is already DONE", i);
298  fr_assert(state->children[i].child == NULL);
299  fr_assert(state->children[i].instruction == NULL);
300  continue;
301 
302  }
303  }
304 
305  /*
306  * Yield if necessary.
307  */
308  if (child_state == CHILD_YIELDED) {
309  return UNLANG_ACTION_YIELD;
310  }
311 
312  fr_assert(child_state == CHILD_DONE);
313 
314  /*
315  * Clean up all of the child requests, because once we
316  * return, no one can access their data any more.
317  */
318  for (i = 0; i < state->num_children; i++) {
319  switch (state->children[i].state) {
320  case CHILD_RUNNABLE:
321  fr_assert(state->children[i].child->backlog == NULL);
322  fr_assert(state->children[i].child->runnable_id < 0);
323 
324  /*
325  * Un-detached children are never in the
326  * runnable queue.
327  */
328  FALL_THROUGH;
329 
330  case CHILD_YIELDED:
331  REQUEST_VERIFY(state->children[i].child);
332  fr_assert(state->children[i].child->runnable_id < 0);
333 
334  /*
335  * Signal the child that it's going to be
336  * stopped. This tells any child modules
337  * to clean up timers, etc.
338  */
339  unlang_interpret_signal(state->children[i].child, FR_SIGNAL_CANCEL);
340  TALLOC_FREE(state->children[i].child);
341  FALL_THROUGH;
342 
343  default:
344  state->children[i].state = CHILD_DONE;
345  state->children[i].child = NULL;
346  state->children[i].instruction = NULL;
347  break;
348  }
349  }
350 
351  *p_result = state->result;
352  return UNLANG_ACTION_CALCULATE_RESULT;
353 }

356 /** Send a signal from parent request to all of it's children
357  *
358  */
359 static void unlang_parallel_signal(request_t *request, fr_state_signal_t action)
360 {
361  unlang_stack_t *stack = request->stack;
362  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
363  unlang_parallel_state_t *state = talloc_get_type_abort(frame->state, unlang_parallel_state_t);
364  int i;
365 
366  /*
367  * Signal all of the children, if they exist.
368  */
369  for (i = 0; i < state->num_children; i++) {
370  switch (state->children[i].state) {
371  case CHILD_INIT:
372  case CHILD_EXITED:
373  case CHILD_DONE:
374  break;
375 
376  case CHILD_RUNNABLE:
377  case CHILD_YIELDED:
378  fr_assert(state->children[i].child != NULL);
379  unlang_interpret_signal(state->children[i].child, action);
380  break;
381  }
382  }

385 static unlang_action_t unlang_parallel(rlm_rcode_t *p_result, request_t *request)
386 {
387  unlang_stack_t *stack = request->stack;
388  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
389  unlang_t *instruction = frame->instruction;
390 
391  unlang_group_t *g;
392  unlang_parallel_t *gext;
393  unlang_parallel_state_t *state;
394 
395  int i;
396 
397  g = unlang_generic_to_group(instruction);
398  if (!g->num_children) {
399  *p_result = RLM_MODULE_NOOP;
400  return UNLANG_ACTION_CALCULATE_RESULT;
401  }
402 
403  gext = unlang_group_to_parallel(g);
404 
405  /*
406  * Allocate an array for the children.
407  */
408  frame->state = state = talloc_zero_size(request,
409  sizeof(unlang_parallel_state_t) +
410  sizeof(state->children[0]) *
411  g->num_children);
412  if (!state) {
413  *p_result = RLM_MODULE_FAIL;
414  return UNLANG_ACTION_CALCULATE_RESULT;
415  };
416 
417  (void) talloc_set_type(state, unlang_parallel_state_t);
418  state->result = RLM_MODULE_FAIL;
419  state->priority = -1; /* as-yet unset */
420  state->detach = gext->detach;
421  state->clone = gext->clone;
422  state->num_children = g->num_children;
423 
424  /*
425  * Initialize all of the children.
426  */
427  for (i = 0, instruction = g->children; instruction != NULL; i++, instruction = instruction->next) {
428  state->children[i].state = CHILD_INIT;
429  state->children[i].instruction = instruction;
430  }
431 
432  frame->process = unlang_parallel_process;
433  return unlang_parallel_process(p_result, request);
434 }
