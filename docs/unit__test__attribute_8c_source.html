276  *
277  */
278 static int poisoned_buffer_allocate(TALLOC_CTX *ctx, uint8_t **buff, size_t size)
279 {
280  uint8_t *our_buff = *buff;
281 
282  if (our_buff) {
283  /*
284  * If it's already the correct length
285  * don't bother re-allocing the buffer,
286  * just memset it to zero.
287  */
288  if ((size + BUFF_POISON_START + BUFF_POISON_END) == talloc_array_length(our_buff)) {
289  memset(our_buff + BUFF_POISON_START, 0, size);
290  return 0;
291  }
292 
293  talloc_free(our_buff); /* Destructor de-poisons */
294  *buff = NULL;
295  }
296 
297  our_buff = talloc_array(ctx, uint8_t, size + BUFF_POISON_START + BUFF_POISON_END);
298  if (!our_buff) return -1;
299 
300 #ifdef HAVE_SANITIZER_LSAN_INTERFACE_H
301  talloc_set_destructor(our_buff, _free_buffer);
302 
303  /*
304  * Poison regions before and after the buffer
305  */
306  ASAN_POISON_MEMORY_REGION(our_buff, BUFF_POISON_START);
307  ASAN_POISON_MEMORY_REGION(our_buff + BUFF_POISON_START + size, BUFF_POISON_END);
308 #endif
309 
310  *buff = our_buff;
311 
312  return 0;
313 }

402 static ssize_t encode_data_string(char *buffer, uint8_t *output, size_t outlen)
403 {

406 
407  p = buffer + 1;
408

410  if (*p == '"') {
411  return slen;
412  }
413 
414  if (*p != '\\') {
415  *(output++) = *(p++);
416  outlen--;
417  slen++;
418  continue;
419  }
420 
421  switch (p[1]) {
422  default:
423  *(output++) = p[1];
424  break;
425 
426  case 'n':
427  *(output++) = '\n';
428  break;
429 
430  case 'r':
431  *(output++) = '\r';
432  break;
433 
434  case 't':
435  *(output++) = '\t';
436  break;
437  }
438 
439  outlen--;
440  slen++;
441  }
442 
443  ERROR("String is not terminated");

448 {
449  int depth = 0;
450  ssize_t slen;

452 
453  for (p = buffer; *p != '\0'; p++) {
454  if (*p == '{') depth++;
455  if (*p == '}') {
456  depth--;
457  if (depth == 0) break;
458  }
459  }
460

462  ERROR("No trailing '}' in string starting with \"%s\"", buffer);
463  return 0;
464  }
465 
466  *endptr = p + 1;

469  p = buffer + 1;

471 
472  slen = encode_tlv(p, output, outlen);

478 static ssize_t hex_to_bin(uint8_t *out, size_t outlen, char *in, size_t inlen)
479 {
480  char *p = in;
481  char *end = in + inlen;
482  uint8_t *out_p = out, *out_end = out_p + outlen;
483 
484  while (p < end) {
485  char *c1, *c2;
486 
487  if (out_p >= out_end) {
488  fr_strerror_printf("Would overflow output buffer");
489  return -(p - in);
490  }
491 
492  fr_skip_whitespace(p);
493 
494  if (!*p) break;
495 
496  c1 = memchr(hextab, tolower((int) *p++), sizeof(hextab));
497  if (!c1) {
498  bad_input:
499  fr_strerror_printf("Invalid hex data starting at \"%s\"", p);
500  return -(p - in);
501  }
502 
503  c2 = memchr(hextab, tolower((int)*p++), sizeof(hextab));
504  if (!c2) goto bad_input;
505 
506  *out_p++ = ((c1 - hextab) << 4) + (c2 - hextab);
507  }
508 
509  return out_p - out;
510 }

513 static ssize_t encode_data(char *p, uint8_t *output, size_t outlen)
514 {
515  ssize_t slen;
516 
517  if (!isspace((int) *p)) {
518  ERROR("Invalid character following attribute definition");
519  return 0;
520  }
521 
522  fr_skip_whitespace(p);
523 
524  if (*p == '{') {
525  size_t sublen;
526  char *q;
527 
528  slen = 0;
529 
530  do {
531  fr_skip_whitespace(p);
532  if (!*p) {
533  if (slen == 0) {
534  ERROR("No data");
535  return 0;
536  }
537 
538  break;
539  }
540 
541  sublen = encode_data_tlv(p, &q, output, outlen);
542  if (sublen <= 0) return 0;
543 
544  slen += sublen;
545  output += sublen;
546  outlen -= sublen;
547  p = q;
548  } while (*q);
549 
550  return slen;
551  }
552 
553  if (*p == '"') {
554  slen = encode_data_string(p, output, outlen);
555  return slen;
556  }
557 
558  slen = hex_to_bin(output, outlen, p, strlen(p));
559  if (slen <= 0) {
560  fr_strerror_printf_push("Empty hex string");
561  return slen;
562  }
563 
564  return slen;
