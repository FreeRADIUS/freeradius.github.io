54 /** Allocate a 'generic' #vp_list_mod_t
55  *
56  * This covers most cases, where we need to allocate a #vp_list_mod_t with a single
57  * modification map, with an attribute ref LHS, and a boxed value RHS.
58  *
59  * @param[in] ctx to allocate #vp_list_mod_t in.
60  * @param[in] original The map from the update section.
61  * @param[in] mutated The original map but with a altered dst (LHS).
62  * If the LHS of the original map was not expanded, this should be
63  * the same as original.
64  * @return
65  * - A new vlm structure on success.
66  * - NULL on failure.
67  */
68 static inline vp_list_mod_t *list_mod_generic_afrom_map(TALLOC_CTX *ctx,
69  map_t const *original, map_t const *mutated)
70 {
71  vp_list_mod_t *n;
72 
73  n = list_mod_alloc(ctx);
74  if (!n) return NULL;
75 
76  n->map = original;
77 
78  n->mod = map_alloc(n);
79  if (!n->mod) return NULL;
80  n->mod->lhs = mutated->lhs;
81  n->mod->op = mutated->op;
82  n->mod->rhs = tmpl_alloc(n->mod, TMPL_TYPE_DATA, T_BARE_WORD, NULL, 0);
83  if (!n->mod->rhs) {
84  talloc_free(n);
85  return NULL;
86  }
87 
88  return n;
89 }

91 /** Allocate a 'delete' #vp_list_mod_t
92  *
93  * This will cause the dst (LHS) to be deleted when applied. This is intended to be
94  * used where the RHS expansion is NULL, and we're doing a := assignment, so need to
95  * delete the LHS.
96  *
97  * @param[in] ctx to allocate #vp_list_mod_t in.
98  * @param[in] original The map from the update section.
99  * @param[in] mutated The original map but with a altered dst (LHS).
100  * If the LHS of the original map was not expanded, this should be
101  * the same as original.
102  *
103  * @return
104  * - A new vlm structure on success.
105  * - NULL on failure.
106  */
107 static inline vp_list_mod_t *list_mod_delete_afrom_map(TALLOC_CTX *ctx,
108  map_t const *original, map_t const *mutated)
109 {
110  vp_list_mod_t *n;
111 
112  n = list_mod_alloc(ctx);
113  if (!n) return NULL;
114 
115  n->map = original;
116 
117  n->mod = map_alloc(n);
118  if (!n->mod) return NULL;
119 
120  n->mod->lhs = mutated->lhs;
121  n->mod->op = T_OP_CMP_FALSE; /* Means delete the LHS */
122  n->mod->rhs = tmpl_alloc(n->mod, TMPL_TYPE_NULL, T_BARE_WORD, NULL, 0);
123  if (!n->mod->rhs) {
124  talloc_free(n);
125  return NULL;
126  }
127 
128  return n;
129 }

131 /** Allocate an 'empty_string' #vp_list_mod_t
132  *
133  * This shallow copies the mutated map, but sets the RHS to be an empty string.
134  *
135  * @param[in] ctx to allocate #vp_list_mod_t in.
136  * @param[in] original The map from the update section.
137  * @param[in] mutated The original map but with a altered dst (LHS).
138  * If the LHS of the original map was not expanded, this should be
139  * the same as original.
140  *
141  * @return
142  * - A new vlm structure on success.
143  * - NULL on failure.
144  */
145 static inline vp_list_mod_t *list_mod_empty_string_afrom_map(TALLOC_CTX *ctx,
146  map_t const *original, map_t const *mutated)
147 {
148  vp_list_mod_t *n;
149  fr_value_box_t empty_string = {
150  .type = FR_TYPE_STRING,
151  .datum = {
152  .strvalue = "",
153  },
154  .length = 0,
155  };
156 
157  n = list_mod_alloc(ctx);
158  if (!n) return NULL;
159 
160  n->map = original;
161 
162  n->mod = map_alloc(n);
163  if (!n->mod) return NULL;
164 
165  n->mod->lhs = mutated->lhs;
166  n->mod->op = mutated->op;
167  n->mod->rhs = tmpl_alloc(n->mod, TMPL_TYPE_DATA, T_DOUBLE_QUOTED_STRING, NULL, -1);
168  if (!n->mod->rhs) {
169  talloc_free(n);
170  return NULL;
171  }
172 
173  /*
174  * For consistent behaviour we don't try and guess
175  * what value we should assign, we try and cast a
176  * zero length string to the specified type and
177  * see what happens...
178  */
179  if (fr_value_box_cast(n->mod->rhs, tmpl_value(n->mod->rhs),
180  mutated->cast ? mutated->cast : tmpl_da(mutated->lhs)->type,
181  tmpl_da(mutated->lhs), &empty_string) < 0) {
182  talloc_free(n);
183  return NULL;
184  }
185 
186  return n;
187 }

189 /** Check that the destination list is currently value
190  *
191  * @param[in] request to resolve in the list in.
192  * @param[in] map to check
193  * @param[in] src_dst a lhs or rhs tmpl to check.
194  * @return
195  * - true if destination list is OK.
196  * - false if destination list is invalid.
197  */
198 static inline fr_pair_t **map_check_src_or_dst(request_t *request, map_t const *map, tmpl_t const *src_dst)
199 {
200  request_t *context = request;
201  fr_pair_t **list;
202  request_ref_t request_ref;
203  pair_list_t list_ref;
204 
205  request_ref = tmpl_request(src_dst);
206  if (radius_request(&context, request_ref) < 0) {
207  REDEBUG("Mapping \"%.*s\" -> \"%.*s\" cannot be performed due to invalid request reference \"%s\"",
208  (int)map->rhs->len, map->rhs->name, (int)map->lhs->len, map->lhs->name,
209  fr_table_str_by_value(request_ref_table, request_ref, "<INVALID>"));
210  return NULL;
211  }
212 
213  list_ref = tmpl_list(src_dst);
214  list = radius_list(context, list_ref);
215  if (!list) {
216  REDEBUG("Mapping \"%.*s\" -> \"%.*s\" cannot be performed due to to invalid list qualifier \"%s\"",
217  (int)map->rhs->len, map->rhs->name, (int)map->lhs->len, map->lhs->name,
218  fr_table_str_by_value(pair_list_table, list_ref, "<INVALID>"));
219  return NULL;
220  }
221 
222  return list;
223 }

225 /** Evaluate a map creating a new map with #TMPL_TYPE_ATTR LHS and #TMPL_TYPE_DATA RHS
226  *
227  * This function creates maps for consumption by map_to_request.
228  *
229  * @param[in,out] ctx to allocate modification maps in.
230  * @param[out] out Where to write the #fr_pair_t (s), which may be NULL if not found
231  * @param[in] request The current request.
232  * @param[in] original the map. The LHS (dst) has to be #TMPL_TYPE_ATTR or #TMPL_TYPE_LIST.
233  * @param[in] lhs_result of previous stack based rhs evaluation.
234  * Must be provided for rhs types:
235  * - TMPL_TYPE_XLAT
236  * - TMPL_TYPE_EXEC (in future)
237  * @param[in] rhs_result of previous stack based rhs evaluation.
238  * Must be provided for rhs types:
239  * - TMPL_TYPE_XLAT
240  * - TMPL_TYPE_EXEC (in future)
241  * Once this function returns result will be invalidated even
242  * if this function errors.
243  * @return
244  * - 0 on success.
245  * - -1 on failure.
246  */
247 int map_to_list_mod(TALLOC_CTX *ctx, vp_list_mod_t **out,
248  request_t *request, map_t const *original,
249  fr_value_box_t **lhs_result, fr_value_box_t **rhs_result)
250 {
251  vp_list_mod_t *n = NULL;
252  map_t map_tmp;
253  map_t const *mutated = original;
254 
255  fr_cursor_t values;
256  fr_value_box_t *head = NULL;
257 
258  TALLOC_CTX *tmp_ctx = NULL;
259 
260  MAP_VERIFY(original);
261 
262  if (!fr_cond_assert(original->lhs != NULL)) return -1;
263  if (!fr_cond_assert(original->rhs != NULL)) return -1;
264 
265  fr_assert(tmpl_is_list(original->lhs) ||
266  tmpl_is_attr(original->lhs) ||
267  tmpl_is_xlat(original->lhs));
268 
269  *out = NULL;
270 
271  /*
272  * Preprocessing of the LHS of the map.
273  */
274  switch (original->lhs->type) {
275  /*
276  * Already in the correct form.
277  */
278  case TMPL_TYPE_LIST:
279  case TMPL_TYPE_ATTR:
280  break;
281 
282  /*
283  * Everything else gets expanded, then re-parsed as an attribute reference.
284  *
285  * This allows the syntax like:
286  * - "Attr-%{number}" := "value"
287  */
288  case TMPL_TYPE_EXEC:
289  case TMPL_TYPE_XLAT:
290  {
291  ssize_t slen;
292 
293  /*
294  * Get our own mutable copy of the original so we can
295  * dynamically expand the LHS.
296  */
297  memcpy(&map_tmp, original, sizeof(map_tmp));
298  mutated = &map_tmp;
299 
300  tmp_ctx = talloc_new(NULL);
301 
302  fr_assert(lhs_result && *lhs_result);
303 
304  /*
305  * This should always be a noop, but included
306  * here for robustness.
307  */
308  if (fr_value_box_list_concat(*lhs_result, *lhs_result, lhs_result, FR_TYPE_STRING, true) < 0) {
309  RPEDEBUG("Left side expansion failed");
310  TALLOC_FREE(*lhs_result);
311  goto error;
312  }
313 
314  slen = tmpl_afrom_attr_str(tmp_ctx, NULL, &map_tmp.lhs, (*lhs_result)->vb_strvalue,
315  &(tmpl_rules_t){
316  .dict_def = request->dict,
317  .prefix = TMPL_ATTR_REF_PREFIX_NO
318  });
319  if (slen <= 0) {
320  RPEDEBUG("Left side expansion result \"%s\" is not an attribute reference",
321  (*lhs_result)->vb_strvalue);
322  TALLOC_FREE(*lhs_result);
323  goto error;
324  }
325  fr_assert(tmpl_is_attr(mutated->lhs) || tmpl_is_list(mutated->lhs));
326  }
327  break;
328 
329  default:
330  fr_assert(0);
331  break;
332  }
333 
334  /*
335  * Special case for !*, we don't need to parse RHS as this is a unary operator.
336  */
337  if (mutated->op == T_OP_CMP_FALSE) {
338  n = list_mod_alloc(ctx);
339  if (!n) goto error;
340 
341  n->map = original;
342  n->mod = map_alloc(n); /* Need to duplicate input map, so next pointer is NULL */
343  n->mod->lhs = mutated->lhs;
344  n->mod->op = mutated->op;
345  n->mod->rhs = mutated->rhs;
346  goto finish;
347  }
348 
349  /*
350  * List to list copy.
351  */
352  if (tmpl_is_list(mutated->lhs) && tmpl_is_list(mutated->rhs)) {
353  fr_cursor_t to;
354  fr_cursor_t from;
355  fr_pair_t **list = NULL;
356  fr_pair_t *vp = NULL;
357 
358  /*
359  * Check source list
360  */
361  list = map_check_src_or_dst(request, mutated, mutated->rhs);
362  if (!list) goto error;
363 
364  vp = fr_cursor_init(&from, list);
365  /*
366  * No attributes found on LHS.
367  */
368  if (!vp) {
369  /*
370  * Special case for := if RHS was NULL.
371  * Should delete all LHS attributes.
372  */
373  if (mutated->op == T_OP_SET) n = list_mod_delete_afrom_map(ctx, original, mutated);
374  goto finish;
375  }
376 
377  n = list_mod_alloc(ctx);
378  n->map = original;
379  fr_cursor_init(&to, &n->mod);
380 
381  /*
382  * Iterate over all attributes in that list
383  */
384  do {
385  map_t *n_mod;
386 
387  n_mod = map_alloc(n);
388  if (!n_mod) goto error;
389 
390  n_mod->op = mutated->op;
391 
392  /*
393  * For the LHS we need to create a reference to
394  * the attribute, with the same destination list
395  * as the current LHS map.
396  */
397  n_mod->lhs = tmpl_alloc(n, TMPL_TYPE_ATTR, T_BARE_WORD, mutated->lhs->name, mutated->lhs->len);
398  if (!n_mod->lhs) goto error;
399 
400  if (tmpl_attr_copy(n_mod->lhs, mutated->lhs) < 0) goto error;
401 
402  tmpl_attr_set_leaf_da(n_mod->lhs, vp->da);
403 
404  /*
405  * For the RHS we copy the value of the attribute
406  * we just found, creating data (literal) tmpl.
407  */
408  n_mod->rhs = tmpl_alloc(n_mod, TMPL_TYPE_DATA,
409  vp->data.type == FR_TYPE_STRING ? T_DOUBLE_QUOTED_STRING : T_BARE_WORD,
410  NULL, 0);
411  if (!n_mod->rhs) goto error;
412 
413  /*
414  * Have to do a full copy, as the attribute we're
415  * getting the buffer value from may be freed
416  * before this map is applied.
417  */
418  if (fr_value_box_copy(n_mod->rhs, tmpl_value(n_mod->rhs), &vp->data) < 0) goto error;
419  fr_cursor_append(&to, n_mod);
420 
421  MAP_VERIFY(n_mod);
422  } while ((vp = fr_cursor_next(&from)));
423 
424  goto finish;
425  }
426 
427  /*
428  * Unparsed. These are easy because they
429  * can only have a single value.
430  */
431  if (tmpl_is_unresolved(mutated->rhs)) {
432  fr_type_t type = tmpl_da(mutated->lhs)->type;
433 
434  fr_assert(tmpl_is_attr(mutated->lhs));
435  fr_assert(tmpl_da(mutated->lhs)); /* We need to know which attribute to create */
436 
437  n = list_mod_generic_afrom_map(ctx, original, mutated);
438  if (!n) goto error;
439 
440  fr_cursor_init(&values, &head);
441 
442  if (fr_value_box_from_str(n->mod, tmpl_value(n->mod->rhs), &type,
443  tmpl_da(mutated->lhs),
444  mutated->rhs->name, mutated->rhs->len, mutated->rhs->quote, false)) {
445  RPEDEBUG("Assigning value to \"%s\" failed", tmpl_da(mutated->lhs)->name);
446  goto error;
447  }
448  goto finish;
449  }
450 
451  /*
452  * Check destination list
453  */
454  if (!map_check_src_or_dst(request, mutated, mutated->lhs)) goto error;
455 
456  (void)fr_cursor_init(&values, &head);
457 
458  switch (mutated->rhs->type) {
459  case TMPL_TYPE_XLAT:
460  {
461  fr_assert(tmpl_xlat(mutated->rhs) != NULL);
462  fr_cursor_t from;
463  fr_value_box_t *vb, *n_vb;
464 
465  assign_values:
466  fr_assert(tmpl_is_attr(mutated->lhs));
467  fr_assert(tmpl_da(mutated->lhs)); /* We need to know which attribute to create */
468 
469  /*
470  * Empty value - Try and cast an empty string
471  * to the destination type, and see what
472  * happens. This is only for XLATs and in future
473  * EXECs.
474  */
475  if (!rhs_result || !*rhs_result) {
476  n = list_mod_empty_string_afrom_map(ctx, original, mutated);
477  if (!n) {
478  RPEDEBUG("Assigning value to \"%s\" failed", tmpl_da(mutated->lhs)->name);
479  xlat_error:
480  fr_cursor_head(&values);
481  fr_cursor_free_list(&values);
482  goto error;
483  }
484  goto finish;
485  }
486 
487  /*
488  * Non-Empty value
489  */
490  n = list_mod_generic_afrom_map(ctx, original, mutated);
491  if (!n) goto error;
492 
493  (void)fr_cursor_init(&from, rhs_result);
494  while ((vb = fr_cursor_remove(&from))) {
495  if (vb->type != tmpl_da(mutated->lhs)->type) {
496  n_vb = fr_value_box_alloc_null(n->mod->rhs);
497  if (!n_vb) {
498  fr_cursor_head(&from);
499  fr_cursor_free_list(&from);
500  goto xlat_error;
501  }
502 
503  if (fr_value_box_cast(n_vb, n_vb,
504  mutated->cast ? mutated->cast : tmpl_da(mutated->lhs)->type,
505  tmpl_da(mutated->lhs), vb) < 0) {
506  RPEDEBUG("Assigning value to \"%s\" failed", tmpl_da(mutated->lhs)->name);
507 
508  fr_cursor_head(&from);
509  fr_cursor_free_list(&from);
510  goto xlat_error;
511  }
512  talloc_free(vb);
513  } else {
514  n_vb = talloc_steal(n, vb); /* Should already be in ctx of n's parent */
515  }
516  fr_cursor_append(&values, n_vb);
517  }
518  }
519  break;
520 
521  case TMPL_TYPE_ATTR:
522  {
523  fr_cursor_t from;
524  tmpl_cursor_ctx_t cc_attr;
525  fr_pair_t *vp;
526  fr_value_box_t *n_vb;
527  int err;
528 
529  fr_assert(!rhs_result || !*rhs_result);
530  fr_assert((tmpl_is_attr(mutated->lhs) && tmpl_da(mutated->lhs)) ||
531  (tmpl_is_list(mutated->lhs) && !tmpl_da(mutated->lhs)));
532 
533  /*
534  * Check source list
535  */
536  if (!map_check_src_or_dst(request, mutated, mutated->rhs)) goto error;
537 
538  /*
539  * Check we have pairs to copy *before*
540  * doing any expensive allocations.
541  */
542  vp = tmpl_cursor_init(&err, request, &cc_attr, &from, request, mutated->rhs);
543  if (!vp) switch (err) {
544  default:
545  break;
546 
547  case -1: /* No input pairs */
548  RDEBUG3("No matching pairs found for \"%s\"", tmpl_da(mutated->rhs)->name);
549  /*
550  * Special case for := if RHS had no attributes
551  * we should delete all LHS attributes.
552  */
553  if (mutated->op == T_OP_SET) n = list_mod_delete_afrom_map(ctx, original, mutated);
554  tmpl_cursor_clear(&cc_attr);
555  goto finish;
556 
557  case -2: /* No matching list */
558  case -3: /* No request context */
559  case -4: /* memory allocation error */
560  RPEDEBUG("Failed resolving attribute source");
561  tmpl_cursor_clear(&cc_attr);
562  goto error;
563  }
564 
565  n = list_mod_generic_afrom_map(ctx, original, mutated);
566  if (!n) {
567  tmpl_cursor_clear(&cc_attr);
568  goto error;
569  }
570 
571  vp = fr_cursor_current(&from);
572  fr_assert(vp); /* Should have errored out */
573  do {
574  n_vb = fr_value_box_alloc_null(n->mod->rhs);
575  if (!n_vb) {
576  attr_error:
577  fr_cursor_head(&values);
578  fr_cursor_free_list(&values);
579  tmpl_cursor_clear(&cc_attr);
580  goto error;
581  }
582 
583  if (vp->data.type != tmpl_da(mutated->lhs)->type) {
584  if (fr_value_box_cast(n_vb, n_vb,
585  mutated->cast ? mutated->cast : tmpl_da(mutated->lhs)->type,
586  tmpl_da(mutated->lhs), &vp->data) < 0) {
587  RPEDEBUG("Assigning value to \"%s\" failed", tmpl_da(mutated->lhs)->name);
588 
589  goto attr_error;
590  }
591  } else {
592  fr_value_box_copy(n_vb, n_vb, &vp->data);
593  }
594  fr_cursor_append(&values, n_vb);
595  } while ((vp = fr_cursor_next(&from)));
596 
597  tmpl_cursor_clear(&cc_attr);
598  }
599  break;
600 
601  case TMPL_TYPE_DATA:
602  {
603  fr_cursor_t from;
604  fr_value_box_t *vb, *vb_head, *n_vb;
605 
606  fr_assert(!rhs_result || !*rhs_result);
607  fr_assert(tmpl_da(mutated->lhs));
608  fr_assert(tmpl_is_attr(mutated->lhs));
609 
610  n = list_mod_generic_afrom_map(ctx, original, mutated);
611  if (!n) goto error;
612 
613  vb_head = tmpl_value(mutated->rhs);
614 
615  for (vb = fr_cursor_init(&from, &vb_head);
616  vb;
617  vb = fr_cursor_next(&from)) {
618  n_vb = fr_value_box_alloc_null(n->mod->rhs);
619  if (!n_vb) {
620  data_error:
621  fr_cursor_head(&values);
622  fr_cursor_free_list(&values);
623  goto error;
624  }
625  /*
626  * This should be optimised away by the map
627  * parser, but in case we're applying runtime
628  * maps we still need to check if we need to
629  * cast.
630  */
631  if (tmpl_da(mutated->lhs)->type != tmpl_value_type(mutated->rhs)) {
632  if (fr_value_box_cast(n_vb, n_vb,
633  mutated->cast ? mutated->cast : tmpl_da(mutated->lhs)->type,
634  tmpl_da(mutated->lhs), vb) < 0) {
635  RPEDEBUG("Assigning value to \"%s\" failed", tmpl_da(mutated->lhs)->name);
636  goto data_error;
637  }
638  } else {
639  /*
640  * We need to do a full copy, as shallow
641  * copy would increase the reference count
642  * on the static/global buffers and possibly
643  * lead to threading issues.
644  */
645  if (fr_value_box_copy(n_vb, n_vb, vb) < 0) goto data_error;
646  }

993  case T_OP_CMP_FALSE:

995  goto finish;

1000  goto finish;
1001 
1002  /*

1004  * of attribute in to is > n?
1005  */
1006  case T_OP_EQ:
1007  {
1008  bool exists = false;
1009  fr_cursor_t from, to, to_insert;
1010  fr_pair_t *vp_from, *vp, *vp_to = NULL, *vp_to_insert = NULL;
1011 
1012  vp_from = map_list_mod_to_vps(parent, vlm);
1013  if (!vp_from) goto finish;
1014

1016  fr_cursor_init(&to_insert, &vp_to_insert);
1017  fr_cursor_init(&to, vp_list);

1024  }

1041  fr_pair_t *vp_from;
1042

1121

1214  case T_OP_GE:
1215  case T_OP_GT:
1216  case T_OP_LE:
1217  case T_OP_LT:

1220

1222  * Instance specific[n] filter

1225  fr_value_box_t *vb = tmpl_value(mod->rhs);

1227

1230  } while ((vb = vb->next));

1251  }
1252  } while ((found = fr_cursor_current(&list)));
1253  }
1254  goto finish;
1255 
1256  default:
1257  fr_assert(0); /* Should have been caught be the caller */
1258  rcode = -1;
1259  goto finish;
1260  }
1261 
1262 finish:
1263  tmpl_cursor_clear(&cc);
