39 /** Iterate over a talloced array of elements
40  *
41 @verbatim
42 talloc_foreach(vpt_m, vpt) {
43  tmpl_debug(vpt);
44 }
45 @endverbatim
46  *
47  * There seems to be a limitation in for loop initialiser arguments where they all
48  * must be the same type, though we can control the number of layers of pointer
49  * indirection on a per variable basis.
50  *
51  * We declare _p to be a pointer of the specified _type, and initialise it to the
52  * start of the array. We declare _end to be a pointer of the specified type and
53  * initialise it to point to the end of the array using talloc_array_length().
54  *
55  * _iter is only updated in the condition to avoid de-referencing invalid memory.
56  *
57  * @param[in] _array to iterate over. May contain zero elements.
58  * @param[in] _iter Name of iteration variable.
59  * Will be declared in the scope of the loop.
60  */
61 #define talloc_foreach(_array, _iter) \
62  for (__typeof__(_array[0]) _iter, *_p = (void *)(_array), *_end = (void *)((_array) + talloc_array_length(_array)); \
63  (_p < _end) && (_iter = *((void **)(_p))); \
64  _p = (__typeof__(_p))((__typeof__(_array))_p) + 1)

197 static inline void _talloc_list_free(void **head, size_t offset)
198 {
199  void *v = *head, *n;
200 
201  while (v) {
202  n = *((void **)(((uint8_t *)(v)) + offset));
203  talloc_free(v);
204  v = n;
205  }
206  *head = NULL;
207 }

209 /** Verify a list of talloced structures are the correct type and are still valid

214 #ifndef TALLOC_GET_TYPE_ABORT_NOOP

216 static inline void *_talloc_list_get_type_abort(void *head, size_t offset, char const *type, char const *location)
217 {
218  void *v = head, *n;
219 
220  if (!v) _talloc_get_type_abort(v, type, location); /* Behave like the normal talloc_get_type_abort function */
221 
222  while (v) {
223  n = *((void **)(((uint8_t *)(v)) + offset));
224  _talloc_get_type_abort(v, type, location);
225  v = n;
226  }
227 
228  return head;
229 }
