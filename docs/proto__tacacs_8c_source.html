88 /** Wrapper around dl_instance which translates the packet-type into a submodule name
89  *
90  * If we found a Packet-Type = Authentication-Start CONF_PAIR for example, here's we'd load
91  * the proto_tacacs_auth module.
92  *
93  * @param[in] ctx to allocate data in (instance of proto_radius).
94  * @param[out] out Where to write a dl_module_inst_t containing the module handle and instance.
95  * @param[in] parent Base structure address.
96  * @param[in] ci #CONF_PAIR specifying the name of the type module.
97  * @param[in] rule unused.
98  * @return
99  * - 0 on success.
100  * - -1 on failure.
101  */
102 static int type_parse(TALLOC_CTX *ctx, void *out, void *parent, CONF_ITEM *ci, UNUSED CONF_PARSER const *rule)
103 {
104  static char const *type_lib_table[FR_PACKET_TYPE_MAX] = {
105  [FR_PACKET_TYPE_VALUE_AUTHENTICATION_START] = "auth",
106  [FR_PACKET_TYPE_VALUE_AUTHENTICATION_CONTINUE] = "auth",
107  [FR_PACKET_TYPE_VALUE_AUTHORIZATION_REQUEST] = "autz",
108  [FR_PACKET_TYPE_VALUE_ACCOUNTING_REQUEST] = "acct",
109  };
110  int code;
111  proto_tacacs_t *inst = talloc_get_type_abort(parent, proto_tacacs_t);
112 
113  code = fr_app_process_type_parse(ctx, out, ci, attr_packet_type,
114  type_lib_table, NUM_ELEMENTS(type_lib_table), "proto_tacacs");
115  if (code < 0) return -1;
116 
117  /*
118  * Set the allowed codes so that the IO functions can
119  * filter them as necessary.
120  */
121  inst->code_allowed[code] = true;
122  return 0;
123 }

125 /** Wrapper around dl_instance
126  *
127  * @param[in] ctx to allocate data in (instance of proto_radius).
128  * @param[out] out Where to write a dl_module_inst_t containing the module handle and instance.
129  * @param[in] parent Base structure address.
130  * @param[in] ci #CONF_PAIR specifying the name of the type module.
131  * @param[in] rule unused.
132  * @return
133  * - 0 on success.
134  * - -1 on failure.
135  */
136 static int transport_parse(TALLOC_CTX *ctx, void *out, UNUSED void *parent, CONF_ITEM *ci, UNUSED CONF_PARSER const *rule)
137 {
138  char const *name = cf_pair_value(cf_item_to_pair(ci));
139  dl_module_inst_t *parent_inst;
140  proto_tacacs_t *inst;
141  CONF_SECTION *listen_cs = cf_item_to_section(cf_parent(ci));
142  CONF_SECTION *transport_cs;
143 
144  transport_cs = cf_section_find(listen_cs, name, NULL);
145 
146  parent_inst = cf_data_value(cf_data_find(listen_cs, dl_module_inst_t, "proto_tacacs"));
147  fr_assert(parent_inst);
148 
149  /*
150  * Set the allowed codes so that we can compile them as
151  * necessary.
152  */
153  inst = talloc_get_type_abort(parent_inst->data, proto_tacacs_t);
154  inst->io.transport = name;
155 
156  /*
157  * Allocate an empty section if one doesn't exist
158  * this is so defaults get parsed.
159  */
160  if (!transport_cs) {
161  transport_cs = cf_section_alloc(listen_cs, listen_cs, name, NULL);
162  cf_section_add(listen_cs, transport_cs);
163  inst->io.app_io_conf = transport_cs;
164  }
165 
166  return dl_module_instance(ctx, out, transport_cs, parent_inst, name, DL_MODULE_TYPE_SUBMODULE);
167 }

171  */
172 static int mod_decode(void const *instance, request_t *request, uint8_t *const data, size_t data_len)
173 {
174  proto_tacacs_t const *inst = talloc_get_type_abort_const(instance, proto_tacacs_t);
175  fr_io_track_t const *track = talloc_get_type_abort_const(request->async->packet_ctx, fr_io_track_t);
176  fr_io_address_t const *address = track->address;
177  RADCLIENT const *client;
178  fr_tacacs_packet_t const *pkt = (fr_tacacs_packet_t const *)data;
179  fr_cursor_t cursor;
180 
181  RHEXDUMP3(data, data_len, "proto_tacacs decode packet");
182 
183  /*
184  * Set the request dictionary so that we can do
185  * generic->protocol attribute conversions as
186  * the request runs through the server.
187  */
188  request->dict = dict_tacacs;
189 
190  client = address->radclient;
191 
192  /*
193  * Decode the header, etc.
194  *
195  * The "type = ..." loader ensures that we only get request packets
196  */
197  switch (pkt->hdr.type) {
198  case FR_TAC_PLUS_AUTHEN:
199  if (packet_is_authen_start_request(pkt)) {
200  request->packet->code = FR_PACKET_TYPE_VALUE_AUTHENTICATION_START;
201  } else {
202  request->packet->code = FR_PACKET_TYPE_VALUE_AUTHENTICATION_CONTINUE;
203  }
204  break;
205 
206  case FR_TAC_PLUS_AUTHOR:
207  request->packet->code = FR_PACKET_TYPE_VALUE_AUTHORIZATION_REQUEST;
208  break;
209 
210  case FR_TAC_PLUS_ACCT:
211  request->packet->code = FR_PACKET_TYPE_VALUE_ACCOUNTING_REQUEST;
212  break;
213 
214  default:
215  fr_assert(0);
216  return -1;
217  }
218 
219  request->packet->id = data[2]; // seq_no
220  request->reply->id = data[2]; // seq_no
221  memcpy(request->packet->vector, &pkt->hdr.session_id, sizeof(pkt->hdr.session_id));
222 
223  request->packet->data = talloc_memdup(request->packet, data, data_len);
224  request->packet->data_len = data_len;
225 
226  /*
227  * Note that we don't set a limit on max_attributes here.
228  * That MUST be set and checked in the underlying
229  * transport, via a call to ???
230  */
231  fr_cursor_init(&cursor, &request->request_pairs);
232  if (fr_tacacs_decode(request->packet, request->packet->data, request->packet->data_len,
233  NULL, client->secret, talloc_array_length(client->secret) - 1,
234  &cursor) < 0) {
235  RPEDEBUG("Failed decoding packet");
236  return -1;
237  }
238 
239  /*
240  * Set the rest of the fields.
241  */
242  memcpy(&request->client, &client, sizeof(client)); /* const issues */
243 
244  request->packet->socket = address->socket;
245  fr_socket_addr_swap(&request->reply->socket, &address->socket);
246 
247  request->config = main_config;
248  REQUEST_VERIFY(request);
249 
250  /*
251  * If we're defining a dynamic client, this packet is
252  * fake. We don't have a secret, so we mash all of the
253  * encrypted attributes to sane (i.e. non-hurtful)
254  * values.
255  */
256  if (!client->active) {
257  fr_pair_t *vp;
258 
259  fr_assert(client->dynamic);
260 
261  for (vp = fr_cursor_init(&cursor, &request->request_pairs);
262  vp != NULL;
263  vp = fr_cursor_next(&cursor)) {
264  if (!vp->da->flags.subtype) {
265  switch (vp->da->type) {
266  default:
267  break;
268 
269  case FR_TYPE_UINT32:
270  vp->vp_uint32 = 0;
271  break;
272 
273  case FR_TYPE_IPV4_ADDR:
274  vp->vp_ipv4addr = INADDR_ANY;
275  break;
276 
277  case FR_TYPE_OCTETS:
278  fr_pair_value_memdup(vp, (uint8_t const *) "", 1, true);
279  break;
280 
281  case FR_TYPE_STRING:
282  fr_pair_value_strdup(vp, "");
283  break;
284  }
285  }
286  }
287  }
288 
289  if (RDEBUG_ENABLED) {
290  fr_pair_t *vp;
291 
292  RDEBUG("Received %s ID %i from %pV:%i to %pV:%i length %zu via socket %s",
293  fr_tacacs_packet_codes[request->packet->code],
294  request->packet->id,
295  fr_box_ipaddr(request->packet->socket.inet.src_ipaddr),
296  request->packet->socket.inet.src_port,
297  fr_box_ipaddr(request->packet->socket.inet.dst_ipaddr),
298  request->packet->socket.inet.dst_port,
299  request->packet->data_len,
300  request->async->listen->name);
301 
302  log_request_pair_list(L_DBG_LVL_1, request, request->request_pairs, "");
303 
304  /*
305  * Maybe the shared secret is wrong?
306  */
307  if (client->active &&
308  ((pkt->hdr.flags & FR_TACACS_FLAGS_VALUE_UNENCRYPTED) == 0) &&
309  RDEBUG_ENABLED2 &&
310  ((vp = fr_pair_find_by_da(&request->request_pairs, attr_tacacs_user_name)) != NULL) &&
311  (fr_utf8_str((uint8_t const *) vp->vp_strvalue, vp->vp_length) < 0)) {
312  RWDEBUG("Unprintable characters in the %s. "
313  "Double-check the shared secret on the server "
314  "and the TACACS+ Client!", attr_tacacs_user_name->name);
315  }
316  }
317 
318  if (!inst->io.app_io->decode) return 0;
319 
320  /*
321  * Let the app_io do anything it needs to do.
322  */
323  return inst->io.app_io->decode(inst->io.app_io_instance, request, data, data_len);
324 }

326 static ssize_t mod_encode(void const *instance, request_t *request, uint8_t *buffer, size_t buffer_len)
327 {
328  proto_tacacs_t const *inst = talloc_get_type_abort_const(instance, proto_tacacs_t);
329  fr_io_track_t const *track = talloc_get_type_abort_const(request->async->packet_ctx, fr_io_track_t);
330  fr_io_address_t const *address = track->address;
331  ssize_t data_len;
332  RADCLIENT const *client;
333 
334  /*
335  * The packet timed out. Tell the network side that the packet is dead.
336  */
337  if (buffer_len == 1) {
338  *buffer = true;
339  return 1;
340  }
341 
342  /*
343  * "Do not respond"
344  */
345  if (request->reply->code == FR_PACKET_TYPE_VALUE_DO_NOT_RESPOND) {
346  *buffer = false;
347  return 1;
348  }
349  fr_assert(request->reply->code != 0);
350  fr_assert(request->reply->code < FR_PACKET_TYPE_MAX);
351 
352  client = address->radclient;
353  fr_assert(client);
354 
355  /*
356  * Dynamic client stuff
357  */
358  if (client->dynamic && !client->active) {
359  RADCLIENT *new_client;
360 
361  fr_assert(buffer_len >= sizeof(client));
362 
363  /*
364  * Allocate the client. If that fails, send back a NAK.
365  *
366  * @todo - deal with NUMA zones? Or just deal with this
367  * client being in different memory.
368  *
369  * Maybe we should create a CONF_SECTION from the client,
370  * and pass *that* back to mod_write(), which can then
371  * parse it to create the actual client....
372  */
373  new_client = client_afrom_request(NULL, request);
374  if (!new_client) {
375  PERROR("Failed creating new client");
376  buffer[0] = true;
377  return 1;
378  }
379 
380  memcpy(buffer, &new_client, sizeof(new_client));
381  return sizeof(new_client);
382  }
383 
384  /*
385  * If the app_io encodes the packet, then we don't need
386  * to do that.
387  */
388  if (inst->io.app_io->encode) {
389  data_len = inst->io.app_io->encode(inst->io.app_io_instance, request, buffer, buffer_len);
390  if (data_len > 0) return data_len;
391  }
392 
393  data_len = fr_tacacs_encode(&FR_DBUFF_TMP(buffer, buffer_len), request->packet->data,
394  client->secret, talloc_array_length(client->secret) - 1,
395  request->reply_pairs);
396  if (data_len < 0) {
397  RPEDEBUG("Failed encoding TACACS+ reply");
398  return -1;
399  }
400 
401  if (RDEBUG_ENABLED) {
402  RDEBUG("Sending %s ID %i from %pV:%i to %pV:%i length %zu via socket %s",
403  fr_tacacs_packet_codes[request->reply->code],
404  request->reply->id,
405  fr_box_ipaddr(request->reply->socket.inet.src_ipaddr),
406  request->reply->socket.inet.src_port,
407  fr_box_ipaddr(request->reply->socket.inet.dst_ipaddr),
408  request->reply->socket.inet.dst_port,
409  data_len,
410  request->async->listen->name);
411 
412  log_request_pair_list(L_DBG_LVL_1, request, request->reply_pairs, "");
413  }
414 
415  RHEXDUMP3(buffer, data_len, "proto_tacacs encode packet");
416 
417  return data_len;
418 }

420 static void mod_entry_point_set(void const *instance, request_t *request)
421 {
422  proto_tacacs_t const *inst = talloc_get_type_abort_const(instance, proto_tacacs_t);
423  dl_module_inst_t *type_submodule;
424  fr_io_track_t *track = request->async->packet_ctx;
425 
426  fr_assert(request->packet->code != 0);
427  fr_assert(request->packet->code < FR_PACKET_TYPE_MAX);
428 
429  request->server_cs = inst->io.server_cs;
430 
431  /*
432  * 'track' can be NULL when there's no network listener.
433  */
434  if (inst->io.app_io && (track->dynamic == request->async->recv_time)) {
435  fr_app_worker_t const *app_process;
436 
437  app_process = (fr_app_worker_t const *) inst->io.dynamic_submodule->module->common;
438 
439  request->async->process = app_process->entry_point;
440  request->async->process_inst = inst->io.dynamic_submodule;
441  track->dynamic = 0;
442  return;
443  }
444 
445  type_submodule = inst->type_submodule_by_code[request->packet->code];
446  if (!type_submodule) {
447  REDEBUG("The server is not configured to accept 'type = %s'", fr_tacacs_packet_codes[request->packet->code]);
448  return;
449  }
450 
451  request->async->process = ((fr_app_worker_t const *)type_submodule->module->common)->entry_point;
452  request->async->process_inst = type_submodule->data;
453 }

551 /** Bootstrap the application
552  *
553  * Bootstrap I/O and type submodules.
554  *
555  * @param[in] instance Ctx data for this application.
556  * @param[in] conf Listen section parsed to give us instance.
557  * @return
558  * - 0 on success.

561 static int mod_bootstrap(void *instance, CONF_SECTION *conf)
562 {
563  proto_tacacs_t *inst = talloc_get_type_abort(instance, proto_tacacs_t);
564 
565  /*
566  * Ensure that the server CONF_SECTION is always set.
567  */
568  inst->io.server_cs = cf_item_to_section(cf_parent(conf));
569 
570  fr_assert(dict_tacacs != NULL);
571 
572  /*
573  * Bootstrap the app_process modules.
574  */
575  if (fr_app_process_bootstrap(inst->io.server_cs, inst->type_submodule, conf) < 0) return -1;
576 
577  /*
578  * No IO module, it's an empty listener.
579  */
580  if (!inst->io.submodule) return 0;
581 
582  /*
583  * These timers are usually protocol specific.
584  */
585  FR_TIME_DELTA_BOUND_CHECK("idle_timeout", inst->io.idle_timeout, >=, fr_time_delta_from_sec(1));
