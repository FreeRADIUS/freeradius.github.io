32 /** Callback for OpenLDAP to rebind and chase referrals
33  *
34  * Called by OpenLDAP when it receives a referral and has to rebind.
35  *
36  * @param handle to rebind.
37  * @param url to bind to.
38  * @param request that triggered the rebind.
39  * @param msgid that triggered the rebind.
40  * @param ctx fr_ldap configuration.
41  */
42 static int fr_ldap_rebind(LDAP *handle, LDAP_CONST char *url,
43  UNUSED ber_tag_t request, UNUSED ber_int_t msgid, void *ctx)
44 {
45  fr_ldap_rcode_t status;
46  fr_ldap_connection_t *conn = talloc_get_type_abort(ctx, fr_ldap_connection_t);
47  fr_ldap_config_t const *handle_config = conn->config;
48 
49  char const *admin_identity = NULL;
50  char const *admin_password = NULL;
51 
52  int ldap_errno;
53 
54  conn->referred = true;
55  conn->rebound = true; /* not really, but oh well... */
56  fr_assert(handle == conn->handle);
57 
58  DEBUG("Rebinding to URL %s", url);
59 
60 # ifdef HAVE_LDAP_URL_PARSE
61  /*
62  * Use bindname and x-bindpw extensions to get the bind credentials
63  * SASL mech is inherited from the module that defined the connection
64  * pool.
65  */
66  if (handle_config->use_referral_credentials) {
67  LDAPURLDesc *ldap_url;
68  int ret;
69  char **ext;
70 
71  ret = ldap_url_parse(url, &ldap_url);
72  if (ret != LDAP_SUCCESS) {
73  ERROR("Failed parsing LDAP URL \"%s\": %s", url, ldap_err2string(ret));
74  return -1;
75  }
76 
77  /*
78  * If there are no extensions, OpenLDAP doesn't
79  * bother allocating an array.
80  */
81  for (ext = ldap_url->lud_exts; ext && *ext; ext++) {
82  char const *p;
83  bool critical = false;
84 
85  p = *ext;
86 
87  if (*p == '!') {
88  critical = true;
89  p++;
90  }
91 
92  /*
93  * LDAP Parse URL unescapes the extensions for us
94  */
95  switch (fr_table_value_by_substr(fr_ldap_supported_extensions, p, -1, LDAP_EXT_UNSUPPORTED)) {
96  case LDAP_EXT_BINDNAME:
97  p = strchr(p, '=');
98  if (!p) {
99  bad_ext:
100  ERROR("Failed parsing extension \"%s\": "
101  "No attribute/value delimiter '='", *ext);
102  ldap_free_urldesc(ldap_url);
103  return LDAP_OTHER;
104  }
105  admin_identity = p + 1;
106  break;
107 
108  case LDAP_EXT_BINDPW:
109  p = strchr(p, '=');
110  if (!p) goto bad_ext;
111  admin_password = p + 1;
112  break;
113 
114  default:
115  if (critical) {
116  ERROR("Failed parsing critical extension \"%s\": "
117  "Not supported by FreeRADIUS", *ext);
118  ldap_free_urldesc(ldap_url);
119  return LDAP_OTHER;
120  }
121  DEBUG2("Skipping unsupported extension \"%s\"", *ext);
122  continue;
123  }
124  }
125  ldap_free_urldesc(ldap_url);
126  } else
127 # endif
128  {
129  admin_identity = handle_config->admin_identity;
130  admin_password = handle_config->admin_password;
131  }
132 
133  status = fr_ldap_bind(NULL, &conn, admin_identity, admin_password,
134  &conn->config->admin_sasl, 0, NULL, NULL);
135  if (status != LDAP_PROC_SUCCESS) {
136  ldap_get_option(handle, LDAP_OPT_ERROR_NUMBER, &ldap_errno);
137 
138  return ldap_errno;
139  }
140 
141  return LDAP_SUCCESS;
142 }
143 #endif

145 /** Allocate and configure a new connection
146  *
147  * Configures both our ldap handle, and libldap's handle.
148  *
149  * This can be used by async code and async code as no attempt is made to connect
150  * to the LDAP server. An attempt will only be made if ldap_start_tls* or ldap_bind*
151  * functions are called.
152  *
153  * If called on an #fr_ldap_connection_t which has already been initialised, will
154  * clear any memory allocated to the connection, unbind the ldap handle, and reinitialise
155  * everything.
156  *
157  * @param[in] c to configure.
158  * @param[in] config to apply.
159  * @return
160  * - 0 on success.
161  * - -1 on error.
162  */
163 int fr_ldap_connection_configure(fr_ldap_connection_t *c, fr_ldap_config_t const *config)
164 {
165  LDAP *handle = NULL;
166  int ldap_errno, ldap_version;
167 
168  fr_assert(config->server);
169 
170 #ifdef HAVE_LDAP_INITIALIZE
171  ldap_errno = ldap_initialize(&handle, config->server);
172  if (ldap_errno != LDAP_SUCCESS) {
173  ERROR("ldap_initialize failed: %s", ldap_err2string(ldap_errno));
174  error:
175  return -1;
176  }
177 #else
178  handle = ldap_init(config->server, config->port);
179  if (!handle) {
180  ERROR("ldap_init failed");
181  error:
182  return -1;
183  }
184 #endif
185 
186  DEBUG3("New libldap handle %p", handle);
187 
188  c->config = config;
189  c->handle = handle;
190  c->rebound = false;
191  c->referred = false;
192 
193  /*
194  * We now have a connection structure, but no actual connection.
195  *
196  * Set a bunch of LDAP options, using common code.
197  */
198 #define do_ldap_option(_option, _name, _value) \
199  if (ldap_set_option(c->handle, _option, _value) != LDAP_OPT_SUCCESS) { \
200  ldap_get_option(c->handle, LDAP_OPT_ERROR_NUMBER, &ldap_errno); \
201  ERROR("Failed setting connection option %s: %s", _name, \
202  (ldap_errno != LDAP_SUCCESS) ? ldap_err2string(ldap_errno) : "Unknown error"); \
203  goto error;\
204  }
205 
206 #define maybe_ldap_option(_option, _name, _value) \
207  if (_value) do_ldap_option(_option, _name, _value)
208 
209  /*
210  * Leave "dereference" unset to use the OpenLDAP default.
211  */
212  if (config->dereference_str) do_ldap_option(LDAP_OPT_DEREF, "dereference", &(config->dereference));
213 
214  /*
215  * Leave "chase_referrals" unset to use the OpenLDAP default.
216  */
217  if (!config->chase_referrals_unset) {
218  if (config->chase_referrals) {
219  do_ldap_option(LDAP_OPT_REFERRALS, "chase_referrals", LDAP_OPT_ON);
220 
221  if (config->rebind == true) {
222 #if LDAP_SET_REBIND_PROC_ARGS == 3
223  ldap_set_rebind_proc(c->handle, fr_ldap_rebind, c);
224 #endif
225  }
226  } else {
227  do_ldap_option(LDAP_OPT_REFERRALS, "chase_referrals", LDAP_OPT_OFF);
228  }
229  }
230 
231 #ifdef LDAP_OPT_NETWORK_TIMEOUT
232  /*
233  * A value of zero results in an handle configuration failure.
234  *
235  * When most people specify zero they mean infinite.
236  *
237  * libldap requires tv_sec to be -1 to mean that.
238  */
239  do_ldap_option(LDAP_OPT_NETWORK_TIMEOUT, "net_timeout",
240  (config->net_timeout ? &fr_time_delta_to_timeval(config->net_timeout) :

263 
264 #ifdef HAVE_LDAP_START_TLS_S
265  /*
266  * Set all of the TLS options
267  */

279 # ifdef LDAP_OPT_X_TLS_NEVER

285  /*

288  */
289 # ifdef LDAP_OPT_X_TLS_NEWCTX

296 
297  if (config->sasl_secprops) do_ldap_option(LDAP_OPT_X_SASL_SECPROPS, "sasl_secprops", config->sasl_secprops);

299  if (config->start_tls) {
300  if (config->port == 636) {
301  WARN("Told to Start TLS on LDAPS port this will probably fail, please correct the "

304  }

306 
307  return 0;

321 /** Close and delete a connection
322  *
323  * Unbinds the LDAP connection, informing the server and freeing any memory, then releases the memory used by the
324  * connection handle.
325  *
326  * @param[in] c to destroy.
327  * @return always indicates success.
328  */
329 static int _ldap_connection_free(fr_ldap_connection_t *c)
330 {
331  talloc_free_children(c); /* Force inverted free order */
332 
333  fr_ldap_control_clear(c);
334 
335  if (!c->handle) return 0; /* Don't need to do anything else if we don't yet have a handle */
336 
337 #ifdef HAVE_LDAP_UNBIND_EXT_S
338  LDAPControl *our_serverctrls[LDAP_MAX_CONTROLS];
339  LDAPControl *our_clientctrls[LDAP_MAX_CONTROLS];
340 
341  fr_ldap_control_merge(our_serverctrls, our_clientctrls,
342  NUM_ELEMENTS(our_serverctrls),
343  NUM_ELEMENTS(our_clientctrls),
344  c, NULL, NULL);
345 
346  DEBUG3("Closing libldap handle %p", c->handle);
347  ldap_unbind_ext(c->handle, our_serverctrls, our_clientctrls); /* Same code as ldap_unbind_ext_s */
348 #else
349  DEBUG3("Closing libldap handle %p", c->handle);
350  ldap_unbind(c->handle); /* Same code as ldap_unbind_s */
351 #endif
352  c->handle = NULL;
353 
354  return 0;
355 }

382 /** (Re-)Initialises the libldap side of the connection handle
383  *
384  * The first ldap state transition is either:
385  *
386  * init -> start tls
387  * or
388  * init -> bind
389  *
390  * Either way libldap will try an open the connection so when fr_ldap_state_next
391  * returns we should have the file descriptor to pass back.
392  *
393  * The complete order of operations is:
394  *
395  * - Initialise the libldap handle with fr_ldap_connection_configure (calls ldap_init)
396  * - Initiate the connection with fr_ldap_state_next, which either binds or calls start_tls.
397  * - Either operation calls ldap_send_server_request.
398  * - Which calls ldap_new_connection.
399  * - Which calls ldap_int_open_connection.
400  * - Which calls ldap_connect_to_(host|path) and adds socket buffers, and possibly
401  * calls ldap_int_tls_start (for ldaps://).
402  * - When ldap_new_connection returns, because LDAP_OPT_CONNECT_ASYNC
403  * is set to LDAP_OPT_ON, lc->lconn_status is set to LDAP_CONNST_CONNECTING.
404  * - ldap_send_server_request checks for lconn_stats == LDAP_CONNST_CONNECTING,
405  * and calls ldap_int_poll, which checks the fd for error conditions
406  * and immediately returns due to the network timeout value.
407  * - If the socket is not yet connected:
408  * - As network timeout on the LDAP handle is 0, ld->ld_errno is set to
409  * LDAP_X_CONNECTING. ldap_send_server_request returns -1.
410  * - bind or start_tls errors with LDAP_X_CONNECTING without sending the request.
411  * - We install a write I/O handler, and wait to be called again, then we retry the
412  * operation.
413  * - else
414  * - the bind or start_tls operation succeeds, our ldap state machine advances,
415  * the connection callback is called and our socket state machine transitions to
416  * connected.
417  * - Continue running the state machine
418  *
419  * @param[out] h Underlying file descriptor from libldap handle.
420  * @param[in] conn Being initialised.
421  * @param[in] uctx Our LDAP connection handle (a #fr_ldap_connection_t).
422  * @return
423  * - FR_CONNECTION_STATE_CONNECTING on success.
424  * - FR_CONNECTION_STATE_FAILED on failure.
425  */
426 static fr_connection_state_t _ldap_connection_init(void **h, fr_connection_t *conn, void *uctx)
427 {
428  fr_ldap_config_t const *config = talloc_get_type_abort(uctx, fr_ldap_config_t);
429  fr_ldap_connection_t *c;
430  fr_ldap_state_t state;
431 
432  c = fr_ldap_connection_alloc(conn);
433 
434  /*
435  * Configure/allocate the libldap handle
436  */
437  if (fr_ldap_connection_configure(c, config) < 0) {
438  error:
439  talloc_free(c);
440  return FR_CONNECTION_STATE_FAILED;
441  }
442 
443  /* Don't block */
444  if (ldap_set_option(c->handle, LDAP_OPT_CONNECT_ASYNC, LDAP_OPT_ON) != LDAP_OPT_SUCCESS) goto error;
445  fr_ldap_connection_timeout_set(c, 0); /* Forces LDAP_X_CONNECTING */
446 
447  state = fr_ldap_state_next(c);
448  if (state == FR_LDAP_STATE_ERROR) goto error;
449 
450  *h = c; /* Set the handle */
451 
452  return FR_CONNECTION_STATE_CONNECTING;
453 }
