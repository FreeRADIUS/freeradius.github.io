52  *
53  * @param[out] out An output buffer of 17 bytes (id + digest).
54  * @param[in] id CHAP ID, a random ID for request/response matching.

59 void fr_radius_encode_chap_password(uint8_t out[static 1 + RADIUS_CHAP_CHALLENGE_LENGTH],
60  uint8_t id, uint8_t const vector[static RADIUS_AUTH_VECTOR_LENGTH],
61  char const *password, size_t password_len)
62 {
63  fr_md5_ctx_t *md5_ctx;
64 
65  md5_ctx = fr_md5_ctx_alloc(true);
66 
67  /*

79 /** "encrypt" a password RADIUS style
80  *
81  * Input and output buffers can be identical if in-place encryption is needed.
82  */
83 static ssize_t encode_password(fr_dbuff_t *dbuff, uint8_t const *input, size_t inlen,
84  char const *secret, uint8_t const *vector)
85 {
86  fr_md5_ctx_t *md5_ctx, *md5_ctx_old;
87  uint8_t digest[RADIUS_AUTH_VECTOR_LENGTH];
88  uint8_t passwd[RADIUS_MAX_PASS_LENGTH];
89  size_t i, n;
90  size_t len;
91 
92  /*
93  * If the length is zero, round it up.
94  */
95  len = inlen;
96 
97  if (len > RADIUS_MAX_PASS_LENGTH) len = RADIUS_MAX_PASS_LENGTH;
98 
99  memcpy(passwd, input, len);
100  if (len < sizeof(passwd)) memset(passwd + len, 0, sizeof(passwd) - len);
101 
102  if (len == 0) len = AUTH_PASS_LEN;
103  else if ((len & 0x0f) != 0) {
104  len += 0x0f;
105  len &= ~0x0f;
106  }
107 
108  md5_ctx = fr_md5_ctx_alloc(false);
109  md5_ctx_old = fr_md5_ctx_alloc(true);
110 
111  fr_md5_update(md5_ctx, (uint8_t const *) secret, talloc_array_length(secret) - 1);
112  fr_md5_ctx_copy(md5_ctx_old, md5_ctx);
113 
114  /*
115  * Do first pass.
116  */
117  fr_md5_update(md5_ctx, vector, AUTH_PASS_LEN);
118 
119  for (n = 0; n < len; n += AUTH_PASS_LEN) {
120  if (n > 0) {
121  fr_md5_ctx_copy(md5_ctx, md5_ctx_old);
122  fr_md5_update(md5_ctx, passwd + n - AUTH_PASS_LEN, AUTH_PASS_LEN);
123  }
124 
125  fr_md5_final(digest, md5_ctx);
126  for (i = 0; i < AUTH_PASS_LEN; i++) passwd[i + n] ^= digest[i];
127  }
128 
129  fr_md5_ctx_free(&md5_ctx);
130  fr_md5_ctx_free(&md5_ctx_old);
131 
132  return fr_dbuff_in_memcpy(dbuff, passwd, len);
133 }

136 static ssize_t encode_tunnel_password(fr_dbuff_t *dbuff, uint8_t const *in, size_t inlen, void *encoder_ctx)
137 {
138  fr_md5_ctx_t *md5_ctx, *md5_ctx_old;
139  uint8_t digest[RADIUS_AUTH_VECTOR_LENGTH];
140  uint8_t tpasswd[RADIUS_MAX_STRING_LENGTH];
141  size_t i, n;
142  size_t encrypted_len;
143  fr_radius_ctx_t *packet_ctx = encoder_ctx;
144  uint32_t r;
145  size_t len;
146  ssize_t slen;
147  fr_dbuff_t work_dbuff = FR_DBUFF_MAX_NO_ADVANCE(dbuff, RADIUS_MAX_STRING_LENGTH);
148 
149  /*
150  * Limit the maximum size of the in password. 2 bytes
151  * are taken up by the salt, and one by the encoded
152  * "length" field. Note that if we have a tag, the
153  * "outlen" will be 252 octets, not 253 octets.
154  */
155  if (inlen > (RADIUS_MAX_STRING_LENGTH - 3)) inlen = (RADIUS_MAX_STRING_LENGTH - 3);
156 
157  /*
158  * If we still overflow the output, let the caller know
159  * how many bytes would have been needed.
160  */
161  FR_DBUFF_SET_RETURN(&work_dbuff, inlen + 3);
162  fr_dbuff_set_to_start(&work_dbuff);
163 
164  /*
165  * Length of the encrypted data is the clear-text
166  * password length plus one byte which encodes the length
167  * of the password. We round up to the nearest encoding
168  * block. Note that this can result in the encoding
169  * length being more than 253 octets.
170  */
171  encrypted_len = ROUND_UP(inlen + 1, 16);
172 
173  /*
174  * We need 2 octets for the salt, followed by the actual
175  * encrypted data. By now we know the password, salt, and
176  * length will fit; we are willing to have a short final
177  * block.
178  */
179  slen = fr_dbuff_set(&work_dbuff, encrypted_len + 2);
180  if (slen < 0) encrypted_len -= -slen;
181  fr_dbuff_set_to_start(&work_dbuff);
182 
183  len = encrypted_len + 2; /* account for the salt */
184 
185  /*
186  * Copy the password over, and fill the remainder with random data.
187  */
188  memcpy(tpasswd + 3, in, inlen);
189 
190  for (i = 3 + inlen; i < (size_t)len; i++) {
191  tpasswd[i] = fr_fast_rand(&packet_ctx->rand_ctx);
192  }
193 
194  /*
195  * Generate salt. The RFCs say:
196  *
197  * The high bit of salt[0] must be set, each salt in a
198  * packet should be unique, and they should be random
199  *
200  * So, we set the high bit, add in a counter, and then
201  * add in some PRNG data. should be OK..
202  */
203  r = fr_fast_rand(&packet_ctx->rand_ctx);
204  tpasswd[0] = (0x80 | (((packet_ctx->salt_offset++) & 0x07) << 4) | ((r >> 8) & 0x0f));
205  tpasswd[1] = r & 0xff;
206  tpasswd[2] = inlen; /* length of the password string */
207 
208  md5_ctx = fr_md5_ctx_alloc(false);
209  md5_ctx_old = fr_md5_ctx_alloc(true);
210 
211  fr_md5_update(md5_ctx, (uint8_t const *) packet_ctx->secret, talloc_array_length(packet_ctx->secret) - 1);
212  fr_md5_ctx_copy(md5_ctx_old, md5_ctx);
213 
214  fr_md5_update(md5_ctx, packet_ctx->vector, RADIUS_AUTH_VECTOR_LENGTH);
215  fr_md5_update(md5_ctx, &tpasswd[0], 2);
216 
217  for (n = 0; n < encrypted_len; n += AUTH_PASS_LEN) {
218  size_t block_len;
219 
220  if (n > 0) {
221  fr_md5_ctx_copy(md5_ctx, md5_ctx_old);
222  fr_md5_update(md5_ctx, tpasswd + 2 + n - AUTH_PASS_LEN, AUTH_PASS_LEN);
223  }
224  fr_md5_final(digest, md5_ctx);
225 
226  block_len = encrypted_len - n;
227  if (block_len > AUTH_PASS_LEN) block_len = AUTH_PASS_LEN;
228 
229  for (i = 0; i < block_len; i++) tpasswd[i + 2 + n] ^= digest[i];
230  }
231 
232  fr_md5_ctx_free(&md5_ctx);
233  fr_md5_ctx_free(&md5_ctx_old);
234 
235  FR_DBUFF_IN_MEMCPY_RETURN(&work_dbuff, tpasswd, len);
236 
237  return fr_dbuff_set(dbuff, &work_dbuff);
238 }

240 static ssize_t encode_tlv_hdr_internal(fr_dbuff_t *dbuff,
241  fr_da_stack_t *da_stack, unsigned int depth,
242  fr_cursor_t *cursor, void *encoder_ctx)
243 {
244  ssize_t slen;
245  fr_pair_t const *vp = fr_cursor_current(cursor);
246  fr_dict_attr_t const *da = da_stack->da[depth];
247  fr_dbuff_t work_dbuff = FR_DBUFF_MAX_NO_ADVANCE(dbuff, 253);
248 
249  for (;;) {
250  FR_PROTO_STACK_PRINT(da_stack, depth);
251 
252  /*
253  * This attribute carries sub-TLVs. The sub-TLVs
254  * can only carry a total of 253 bytes of data.
255  */
256 
257  /*
258  * Determine the nested type and call the appropriate encoder
259  */
260  if (da_stack->da[depth + 1]->type == FR_TYPE_TLV) {
261  slen = encode_tlv_hdr(&work_dbuff, da_stack, depth + 1, cursor, encoder_ctx);
262  } else {
263  slen = encode_rfc_hdr_internal(&work_dbuff, da_stack, depth + 1, cursor, encoder_ctx);
264  }
265 
266  if (slen <= 0) return slen;
267 
268  /*
269  * If nothing updated the attribute, stop
270  */
271  if (!fr_cursor_current(cursor) || (vp == fr_cursor_current(cursor))) break;
272 
273  /*
274  * We can encode multiple sub TLVs, if after
275  * rebuilding the TLV Stack, the attribute
276  * at this depth is the same.
277  */
278  if ((da != da_stack->da[depth]) || (da_stack->depth < da->depth)) break;
279  vp = fr_cursor_current(cursor);
280  }
281 
282  return fr_dbuff_set(dbuff, &work_dbuff);
283 }

285 static ssize_t encode_tlv_hdr(fr_dbuff_t *dbuff,
286  fr_da_stack_t *da_stack, unsigned int depth,
287  fr_cursor_t *cursor, void *encoder_ctx)
288 {
289  ssize_t slen;
290  fr_dbuff_marker_t len_m;
291  fr_dbuff_t work_dbuff = FR_DBUFF_NO_ADVANCE(dbuff);
292 
293  VP_VERIFY(fr_cursor_current(cursor));
294  FR_PROTO_STACK_PRINT(da_stack, depth);
295 
296  if (da_stack->da[depth]->type != FR_TYPE_TLV) {
297  fr_strerror_printf("%s: Expected type \"tlv\" got \"%s\"", __FUNCTION__,
298  fr_table_str_by_value(fr_value_box_type_table, da_stack->da[depth]->type, "?Unknown?"));
299  return PAIR_ENCODE_FATAL_ERROR;
300  }
301 
302  if (!da_stack->da[depth + 1]) {
303  fr_strerror_printf("%s: Can't encode empty TLV", __FUNCTION__);
304  return PAIR_ENCODE_SKIPPED;
305  }
306 
307  /*
308  * Encode the first level of TLVs
309  */
310  FR_DBUFF_IN_RETURN(&work_dbuff, (uint8_t)da_stack->da[depth]->attr);
311  fr_dbuff_marker(&len_m, &work_dbuff); /* Mark the start of the length field */
312  FR_DBUFF_ADVANCE_RETURN(&work_dbuff, 1); /* One byte for the length */
313 
314  slen = encode_tlv_hdr_internal(&FR_DBUFF_MAX(&work_dbuff, 253), da_stack, depth, cursor, encoder_ctx);
315  if (slen <= 0) return slen;
316 
317  fr_dbuff_in(&len_m, (uint8_t)(slen + 2));
318 
319  return fr_dbuff_set(dbuff, &work_dbuff);
320 }

349 /** Encodes the data portion of an attribute
350  *
351  * @return
352  * > 0, Length of the data portion.
353  * = 0, we could not encode anything, skip this attribute (and don't encode the header)
354  * unless it's one of a list of exceptions.
355  * < 0, How many additional bytes we'd need as a negative integer.
356  * PAIR_ENCODE_FATAL_ERROR - Abort encoding the packet.
357  * PAIR_ENCODE_SKIPPED - Unencodable value
358  */
359 static ssize_t encode_value(fr_dbuff_t *dbuff,
360  fr_da_stack_t *da_stack, unsigned int depth,
361  fr_cursor_t *cursor, void *encoder_ctx)
362 {
363  ssize_t slen;
364  size_t len;
365  fr_pair_t const *vp = fr_cursor_current(cursor);
366  fr_dict_attr_t const *da = da_stack->da[depth];
367  fr_radius_ctx_t *packet_ctx = encoder_ctx;
368  fr_dbuff_t work_dbuff = FR_DBUFF_NO_ADVANCE(dbuff);
369  fr_dbuff_t value_dbuff;
370  fr_dbuff_marker_t value_start;
371  fr_dbuff_marker_t start;
372  bool encrypted = false;
373 
374  fr_dbuff_marker(&start, &work_dbuff);
375 
376  VP_VERIFY(vp);
377  FR_PROTO_STACK_PRINT(da_stack, depth);
378 
379  /*
380  * Catch errors early on.
381  */
382  if (flag_encrypted(&vp->da->flags) && !packet_ctx) {
383  fr_strerror_printf("Asked to encrypt attribute, but no packet context provided");
384  return PAIR_ENCODE_FATAL_ERROR;
385  }
386 
387  /*
388  * It's a little weird to consider a TLV as a value,
389  * but it seems to work OK.
390  */
391  if (da->type == FR_TYPE_TLV) return encode_tlv_hdr(dbuff, da_stack, depth, cursor, encoder_ctx);
392 
393  /*
394  * This has special requirements.
395  */
396  if (da->type == FR_TYPE_STRUCT) {
397  slen = fr_struct_to_network_dbuff(&work_dbuff, da_stack, depth, cursor, encoder_ctx, encode_value);
398  if (slen <= 0) return slen;
399 
400  vp = fr_cursor_current(cursor);
401  fr_proto_da_stack_build(da_stack, vp ? vp->da : NULL);
402 
403  /*
404  * Encode any TLV, attributes which are part of this structure.
405  *
406  * The fr_struct_to_network() function can't do
407  * this work, as it's not protocol aware, and
408  * doesn't have the da_stack or encoder_ctx.
409  *
410  * Note that we call the "internal" encode
411  * function, as we don't want the encapsulating
412  * TLV to be encoded here. It's number is just
413  * the field number in the struct.
414  */
415  while (vp && (da_stack->da[depth] == da) && (da_stack->depth >= da->depth)) {
416  slen = encode_tlv_hdr_internal(&work_dbuff, da_stack, depth + 1, cursor, encoder_ctx);
417  if (slen < 0) return slen;
418 
419  vp = fr_cursor_current(cursor);
420  fr_proto_da_stack_build(da_stack, vp ? vp->da : NULL);
421  }
422 
423  return fr_dbuff_set(dbuff, &work_dbuff);
424  }
425 
426  /*
427  * If it's not a TLV, it should be a value type RFC
428  * attribute make sure that it is.
429  */
430  if (da_stack->da[depth + 1] != NULL) {
431  fr_strerror_printf("%s: Encoding value but not at top of stack", __FUNCTION__);
432  return PAIR_ENCODE_FATAL_ERROR;
433  }
434 
435  if (vp->da != da) {
436  fr_strerror_printf("%s: Top of stack does not match vp->da", __FUNCTION__);
437  return PAIR_ENCODE_FATAL_ERROR;
438  }
439 
440  switch (da->type) {
441  case FR_TYPE_STRUCTURAL:
442  fr_strerror_printf("%s: Called with structural type %s", __FUNCTION__,
443  fr_table_str_by_value(fr_value_box_type_table, da_stack->da[depth]->type, "?Unknown?"));
444  return PAIR_ENCODE_FATAL_ERROR;
445 
446  default:
447  break;
448  }
449 
450  /*
451  * Write tag byte
452  *
453  * The Tag field is one octet in length and is intended to provide a
454  * means of grouping attributes in the same packet which refer to the
455  * same tunnel. If the value of the Tag field is greater than 0x00
456  * and less than or equal to 0x1F, it SHOULD be interpreted as
457  * indicating which tunnel (of several alternatives) this attribute
458  * pertains. If the Tag field is greater than 0x1F, it SHOULD be
459  * interpreted as the first byte of the following String field.
460  *
461  * If the first byte of the string value looks like a
462  * tag, then we always encode a tag byte, even one that
463  * is zero.
464  */
465  if ((vp->da->type == FR_TYPE_STRING) && flag_has_tag(&vp->da->flags)) {
466  if (packet_ctx->tag) {
467  FR_DBUFF_IN_RETURN(&work_dbuff, (uint8_t)packet_ctx->tag);
468  } else if (TAG_VALID(vp->vp_strvalue[0])) {
469  FR_DBUFF_IN_RETURN(&work_dbuff, (uint8_t)0x00);
470  }
471  }
472 
473  /*
474  * Starting here is a value that may require encryption.
475  */
476  value_dbuff = FR_DBUFF_NO_ADVANCE(&work_dbuff);
477  fr_dbuff_marker(&value_start, &value_dbuff);
478 
479  /*
480  * Set up the default sources for the data.
481  */
482  len = fr_radius_attr_len(vp);
483 
484  /*
485  * Invalid value, don't encode.
486  */
487  if (len > RADIUS_MAX_STRING_LENGTH) {
488  fr_strerror_printf("%s length of %zu bytes exceeds maximum value length",
489  vp->da->name, len);
490  return PAIR_ENCODE_SKIPPED;
491  }
492 
493  switch (da->type) {
494  /*
495  * If asked to encode more data than allowed, we
496  * encode only the allowed data.
497  */
498  case FR_TYPE_STRING:
499  if (flag_abinary(&da->flags)) {
500  slen = fr_radius_encode_abinary(vp, fr_dbuff_current(&value_dbuff), fr_dbuff_remaining(&value_dbuff));
501  if (slen <= 0) return slen;
502 
503  FR_DBUFF_ADVANCE_RETURN(&value_dbuff, (size_t) slen);
504  break;
505  }
506  FALL_THROUGH;
507 
508  case FR_TYPE_OCTETS:
509  FR_DBUFF_IN_MEMCPY_RETURN(&value_dbuff, (uint8_t const *)(vp->vp_ptr), len);
510  break;
511 
512  /*
513  * Common encoder might add scope byte
514  */
515  case FR_TYPE_IPV6_ADDR:
516  FR_DBUFF_IN_MEMCPY_RETURN(&value_dbuff, vp->vp_ipv6addr, sizeof(vp->vp_ipv6addr));
517  break;
518 
519  /*
520  * Common encoder doesn't add reserved byte
521  */
522  case FR_TYPE_IPV6_PREFIX:
523  len = vp->vp_ip.prefix >> 3; /* Convert bits to whole bytes */
524  FR_DBUFF_IN_BYTES_RETURN(&value_dbuff, 0x00, vp->vp_ip.prefix);
525  /* Only copy the minimum number of address bytes required */
526  FR_DBUFF_IN_MEMCPY_RETURN(&value_dbuff, (uint8_t const *)vp->vp_ipv6addr, len);
527  break;
528 
529  /*
530  * Common encoder doesn't add reserved byte
531  */
532  case FR_TYPE_IPV4_PREFIX:
533  FR_DBUFF_IN_BYTES_RETURN(&value_dbuff, 0x00, vp->vp_ip.prefix);
534  FR_DBUFF_IN_MEMCPY_RETURN(&value_dbuff, (uint8_t const *)&vp->vp_ipv4addr, sizeof(vp->vp_ipv4addr));
535  break;
536 
537  /*
538  * Simple data types use the common encoder.
539  */
540  default:
541  slen = fr_value_box_to_network_dbuff(NULL, &value_dbuff, &vp->data);
542  if (slen < 0) return PAIR_ENCODE_FATAL_ERROR;
543  break;
544  }
545 
546  /*
547  * No data: don't encode the value. The type and length should still
548  * be written.
549  */
550  if (fr_dbuff_used(&value_dbuff) == 0) {
551  vp = fr_cursor_next(cursor);
552  fr_proto_da_stack_build(da_stack, vp ? vp->da : NULL);
553  return 0;
554  }
555 
556  /*
557  * Encrypt the various password styles
558  *
559  * Attributes with encrypted values MUST be less than
560  * 128 bytes long.
561  */
562  if (flag_encrypted(&da->flags)) switch (vp->da->flags.subtype) {
563  case FLAG_ENCRYPT_USER_PASSWORD:
564  /*
565  * Encode the password in place
566  */
567  slen = encode_password(&work_dbuff, fr_dbuff_current(&value_start), fr_dbuff_used(&value_dbuff),
568  packet_ctx->secret, packet_ctx->vector);
569  if (slen < 0) return slen;
570  encrypted = true;
571  break;
572 
573  case FLAG_TAGGED_TUNNEL_PASSWORD:
574  case FLAG_ENCRYPT_TUNNEL_PASSWORD:
575  /*
576  * Always encode the tag even if it's zero.
577  *
578  * The Tunnel-Password uses 2 salt fields which
579  * MAY have any value. As a result, we always
580  * encode a tag. If we would omit the tag, then
581  * perhaps one of the salt fields could be
582  * mistaken for the tag.
583  */
584  if (flag_has_tag(&vp->da->flags)) fr_dbuff_advance(&work_dbuff, 1);
585

603  * always fits.

608  packet_ctx->secret, packet_ctx->vector);
609  if (slen < 0) return slen;
610  encrypted = true;

648 /** Breaks down large data into pieces, each with a header
649  *
650  * @param dbuff dbuff that has at its end a header followed by too much data
651  * for the header's one-byte length field
652  * @param ptr marker that points at said header
653  * @param hdr_len length of the headers that will be added
654  * @param len number of bytes of data, starting at ptr + ptr[1]
655  * @param flag_offset offset within header of a flag byte whose MSB is set for all
656  * but the last piece
657  * @param vsa_offset if non-zero, the offset of a length field in a (sub?)-header
658  * of size 3 that also needs to be adjusted to include the number
659  * of bytes of data in the piece
660  *
661  * NOTE: the header present on entry may be longer than hdr_len (vide the VSA case in
662  * encode_extended_hdr()), in which case the size of first piece is more tightly
663  * constrained then those following.
664  *
665  * attr_shift() is not like other encoding functions. The caller retrieved the data;
666  * here we're chopping it into pieces that will fit into structures whose headers
667  * have one-byte length fields (that have to include the header length). Markers
668  * associated with a child can't access data before the child's start--but that's
669  * where the data is, so we associate them with dbuff.
670  */
671 static ssize_t attr_shift(fr_dbuff_t *dbuff,
672  fr_dbuff_marker_t *ptr, int hdr_len, ssize_t len,
673  int flag_offset, int vsa_offset)
674 {
675  int check_len = len - fr_dbuff_current(ptr)[1];
676  int total = hdr_len;
677  fr_dbuff_t work_dbuff = FR_DBUFF_NO_ADVANCE(dbuff);
678  fr_dbuff_marker_t hdr, next_hdr, next_data;
679 
680  /*
681  * Pass 1: Check if the addition of the headers
682  * overflows the available freespace. If so, return
683  * what we were capable of encoding.
684  */
685 
686  while (check_len > (255 - hdr_len)) {
687  total += hdr_len;
688  check_len -= (255 - hdr_len);
689  }
690 
691  /*
692  * Note that this results in a number of attributes maybe
693  * being marked as "encoded", but which aren't in the
694  * packet. Oh well. The solution is to fix the
695  * "encode_value" function to take into account the header
696  * lengths.
697  */
698  if (fr_dbuff_advance(&work_dbuff, total) < 0) {
699  return (fr_dbuff_current(ptr) + fr_dbuff_current(ptr)[1]) - fr_dbuff_start(&work_dbuff);
700  }
701 
702  /*
703  * Markers associated with dbuff so we can manipulate data
704  * accumulated there.
705  */
706  fr_dbuff_marker(&hdr, dbuff);
707  fr_dbuff_set(&hdr, fr_dbuff_current(ptr));
708  fr_dbuff_marker(&next_hdr, dbuff);
709  fr_dbuff_marker(&next_data, dbuff);
710 
711  /*
712  * Pass 2: Now that we know there's enough freespace,
713  * re-arrange the data to form a set of valid
714  * RADIUS attributes.
715  */
716  for (;;) {
717  /* Extend current attribute as much as possible. */
718  int sublen = 255 - fr_dbuff_current(&hdr)[1];
719  if (len < sublen) sublen = len;
720  fr_dbuff_current(&hdr)[1] += sublen;
721 
722  /* Adjust the other length field if it exists. */
723  if (vsa_offset) fr_dbuff_current(&hdr)[vsa_offset] += sublen;
724 
725  /* If all data are accounted for, we're done. */
726  len -= sublen;
727  if (len == 0) break;
728 
729  /* This attribute isn't the last, so flag it. */
730  fr_dbuff_current(&hdr)[flag_offset] |= 0x80;
731 
732  /* Make room for another header. */
733  fr_dbuff_set(&next_hdr, fr_dbuff_current(&hdr) + 255);
734  fr_dbuff_set(&next_data, fr_dbuff_current(&next_hdr) + hdr_len);
735  fr_dbuff_move(&next_data, &next_hdr, len);
736 
737  /* Copy current header into new header and advance to it... */
738  fr_dbuff_set(&next_hdr, fr_dbuff_current(&hdr) + 255);
739  fr_dbuff_move(&next_hdr, &hdr, hdr_len);
740  fr_dbuff_advance(&hdr, 255 - hdr_len);
741 
742  /* ...and set its length to that of the header. */
743  fr_dbuff_current(&hdr)[1] = hdr_len;
744  }
745 
746  /* Clear our markers from dbuff's list */
747  fr_dbuff_marker_release(&hdr);
748 
749  return fr_dbuff_set(dbuff, &work_dbuff);
750 }
