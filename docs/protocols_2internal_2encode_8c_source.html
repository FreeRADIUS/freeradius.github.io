43 /** Encode the value of the value pair the cursor currently points at.
44  *
45  * @param dbuff data buffer to place the encoded data in
46  * @param da_stack da stack corresponding to the value pair
47  * @param depth in da_stack
48  * @param cursor cursor whose current value is the one to be encoded
49  * @param encoder_ctx encoder context
50  *
51  * @return either a negative number, indicating an error
52  * or the number of bytes used to encode the value
53  */
54 static ssize_t internal_encode(fr_dbuff_t *dbuff,
55  fr_da_stack_t *da_stack, unsigned int depth,
56  fr_cursor_t *cursor, void *encoder_ctx)
57 {
58  fr_dbuff_t work_dbuff = FR_DBUFF_NO_ADVANCE(dbuff);
59  fr_dbuff_marker_t enc_field, value_field, len_field, value_dest;
60  fr_dict_attr_t const *da = da_stack->da[depth];
61  fr_pair_t *vp = fr_cursor_current(cursor);
62 
63  ssize_t flen, slen;
64 
65  uint8_t buff[sizeof(uint64_t)];
66 
67  FR_PROTO_STACK_PRINT(da_stack, depth);
68 
69  fr_dbuff_marker(&enc_field, &work_dbuff);
70 
71  /*
72  * Zero out first encoding byte
73  */
74  FR_DBUFF_IN_BYTES_RETURN(&work_dbuff, 0x00);
75 
76  switch (da->type) {
77  /*
78  * Only leaf attributes can be tainted
79  */
80  case FR_TYPE_VALUE:
81  if (vp->vp_tainted) fr_dbuff_current(&enc_field)[0] |= FR_INTERNAL_FLAG_TAINTED;
82  break;
83 
84  default:
85  break;
86  }
87 
88  /*
89  * Need to use the second encoding byte
90  */
91  if (da->flags.is_unknown) {
92  fr_dbuff_current(&enc_field)[0] |= FR_INTERNAL_FLAG_EXTENDED;
93  FR_DBUFF_IN_BYTES_RETURN(&work_dbuff, FR_INTERNAL_FLAG_INTERNAL);
94  }
95 
96  /*
97  * Encode the type and write the width of the
98  * integer to the encoding byte.
99  */
100  flen = fr_dbuff_in_uint64v(&work_dbuff, da->attr);
101  if (flen <= 0) return flen;
102  fr_dbuff_current(&enc_field)[0] |= ((flen - 1) << 5);
103 
104  /*
105  * Leave one byte in hopes that the length will fit
106  * and remember where the encoded data starts.
107  */
108  fr_dbuff_marker(&len_field, &work_dbuff);
109  FR_DBUFF_ADVANCE_RETURN(&work_dbuff, 1);
110  fr_dbuff_marker(&value_field, &work_dbuff);
111  fr_dbuff_marker(&value_dest, &work_dbuff);
112 
113  switch (da->type) {
114  case FR_TYPE_VALUE:
115  slen = fr_value_box_to_network(&FR_DBUFF_RESERVE(&work_dbuff, sizeof(uint64_t) - 1), &vp->data);
116  if (slen < 0) return PAIR_ENCODE_FATAL_ERROR;
117  FR_PROTO_HEX_DUMP(fr_dbuff_current(&value_field), slen, "value %s",
118  fr_table_str_by_value(fr_value_box_type_table, vp->vp_type, "<UNKNOWN>"));
119  fr_cursor_next(cursor);
120  break;
121 
122  /*
123  * This is the vendor container.
124  * For RADIUS it'd be something like attr 26.
125  *
126  * Inside the VSA you then have the vendor
127  * which is just encoded as another TLVish
128  * type attribute.
129  *
130  * For small vendor PENs <= 255 this
131  * encoding is 6 bytes, the same as RADIUS.
132  *
133  * For larger vendor PENs it's more bytes
134  * but we really don't care.
135  */
136  case FR_TYPE_VSA:
137  case FR_TYPE_VENDOR:
138  slen = internal_encode(&FR_DBUFF_RESERVE(&work_dbuff, sizeof(uint64_t) - 1), da_stack, depth + 1,
139  cursor, encoder_ctx);
140  if (slen < 0) return slen;
141  break;
142 
143  /*
144  * Children of TLVs are encoded in the context
145  * of the TLV.
146  */
147  case FR_TYPE_TLV:
148  /*
149  * We've done the complete stack.
150  * Hopefully this TLV has some
151  * children to encode...
152  */
153  if (da == vp->da) {
154  fr_cursor_t children;
155  fr_pair_t *child;
156 
157  for (child = fr_cursor_talloc_init(&children, &vp->vp_group, fr_pair_t);
158  child;
159  child = fr_cursor_current(&children)) {
160 
161  FR_PROTO_TRACE("encode ctx changed %s -> %s", da->name, child->da->name);
162 
163  fr_proto_da_stack_build_partial(da_stack, da_stack->da[depth], child->da);
164  FR_PROTO_STACK_PRINT(da_stack, depth);
165 
166  slen = internal_encode(&FR_DBUFF_RESERVE(&work_dbuff, sizeof(uint64_t) - 1),
167  da_stack, depth + 1, &children, encoder_ctx);
168  if (slen < 0) return slen;
169  }
170  fr_cursor_next(cursor);
171  break;
172  }
173 
174  /*
175  * Still encoding intermediary TLVs...
176  */
177  slen = internal_encode(&FR_DBUFF_RESERVE(&work_dbuff, sizeof(uint64_t) - 1), da_stack, depth + 1,
178  cursor, encoder_ctx);
179  if (slen < 0) return slen;
180  break;
181 
182  /*
183  * Each child of a group encodes from the
184  * dictionary root to the leaf da.
185  *
186  * Re-enter the encoder at the start.
187  * We do this, because the child may
188  * have a completely different da_stack.
189  */
190  case FR_TYPE_GROUP:
191  {
192  fr_cursor_t children;
193 
194  for (vp = fr_cursor_talloc_init(&children, &vp->vp_group, fr_pair_t);
195  vp;
196  vp = fr_cursor_current(&children)) {
197  FR_PROTO_TRACE("encode ctx changed %s -> %s", da->name, vp->da->name);
198 
199  slen = fr_internal_encode_pair(&FR_DBUFF_RESERVE(dbuff, sizeof(uint64_t) - 1),
200  &children, encoder_ctx);
201  if (slen < 0) return slen;
202  }
203  fr_cursor_next(cursor);
204  }
205  break;
206 
207  default:
208  fr_strerror_printf("%s: Unexpected attribute type \"%s\"",
209  __FUNCTION__, fr_table_str_by_value(fr_value_box_type_table, da->type, "?Unknown?"));
210  return PAIR_ENCODE_FATAL_ERROR;
211  }
212 
213  /*
214  * Encode the total length, and write the width
215  * of the integer to the encoding byte.
216  *
217  * Already did length checks at the start of
218  * the function.
219  */
220  slen = fr_dbuff_current(&work_dbuff) - fr_dbuff_current(&value_field);
221  flen = (ssize_t) fr_net_from_uint64v(buff, slen);
222 
223  /*
224  * Ugh, it's a long one, need to move the data.
225  * NOTE: fr_dbuff_move() advances its source and destination,
226  * but here, the destination is a marker--advancing it won't
227  * advance work_dbuff, so we advance it ourselves. Then the
228  * return will give the appropriate value.
229  */
230  if (flen > 1) {
231  FR_DBUFF_ADVANCE_RETURN(&work_dbuff, flen - 1);
232  fr_dbuff_advance(&value_dest, flen - 1);
233  fr_dbuff_move(&value_dest, &value_field, slen);
234  }
235 
236  fr_dbuff_move(&len_field, &FR_DBUFF_TMP(buff, flen), flen);
237  fr_dbuff_current(&enc_field)[0] |= ((flen - 1) << 2);
238 
239  FR_PROTO_HEX_DUMP(fr_dbuff_current(&enc_field), fr_dbuff_used(&work_dbuff) - slen, "header");
240 
241  return fr_dbuff_set(dbuff, &work_dbuff);
242 }

244 /** Encode a data structure into an internal attribute
245  *
246  * This will become the main entry point when we switch fully to dbuff.
247  *
248  * @param[in,out] dbuff Where to write encoded data and how much one can write.
249  * @param[in] cursor Specifying attribute to encode.
250  * @param[in] encoder_ctx Additional data such as the shared secret to use.
251  * @return
252  * - >0 The number of bytes written to out.
253  * - 0 Nothing to encode (or attribute skipped).
254  * - <0 an error occurred.
255  */
256 ssize_t fr_internal_encode_pair(fr_dbuff_t *dbuff, fr_cursor_t *cursor, void *encoder_ctx)
257 {
258  fr_pair_t *vp;
259  fr_da_stack_t da_stack;
260 
261  vp = fr_cursor_current(cursor);
262  if (!vp) return 0;
263 
264  fr_proto_da_stack_build(&da_stack, vp->da);
265 
266  return internal_encode(dbuff, &da_stack, 0, cursor, encoder_ctx);
267 }
