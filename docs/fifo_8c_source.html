42 /** Free a fifo and optionally, any data still enqueued
43  *
44  * @param[in] fi to free.
45  * @return 0
46  */
47 static int _fifo_free(fr_fifo_t *fi)
48 {
49  unsigned int i;
50 
51  if (fi->free_node) {
52  for (i = 0 ; i < fi->num; i++) {
53  unsigned int element;
54 
55  element = i + fi->first;
56  if (element > fi->max) {
57  element -= fi->max;
58  }
59 
60  fi->free_node(fi->data[element]);
61  fi->data[element] = NULL;
62  }
63  }
64 
65  memset(fi, 0, sizeof(*fi));
66 
67  return 0;
68 }

70 /** Create a fifo queue
71  *
72  * The first element enqueued will be the first to be dequeued.
73  *
74  * @note The created fifo does not provide any thread synchronisation functionality
75  * such as mutexes. If multiple threads are enqueueing and dequeueing data
76  * the callers must synchronise their access.
77  *
78  * @param[in] ctx to allocate fifo array in.
79  * @param[in] type Talloc type of elements (may be NULL).
80  * @param[in] max The maximum number of elements allowed.
81  * @param[in] free_node Function to use to free node data if the fifo is freed.
82  * @return
83  * - A new fifo queue.
84  * - NULL on error.
85  */
86 fr_fifo_t *_fr_fifo_create(TALLOC_CTX *ctx, char const *type, int max, fr_fifo_free_t free_node)
87 {
88  fr_fifo_t *fi;
89 
90  if ((max < 2) || (max > (1024 * 1024))) return NULL;
91 
92  fi = talloc_zero_size(ctx, (sizeof(*fi) + (sizeof(fi->data[0])*max)));
93  if (!fi) return NULL;
94  talloc_set_type(fi, fr_fifo_t);
95  talloc_set_destructor(fi, _fifo_free);
96 
97  fi->max = max;
98  fi->type = type;
99  fi->free_node = free_node;
100 
101  return fi;
102 }

104 /** Push data onto the fifo
105  *
106  * @param[in] fi FIFO to push data onto.
107  * @param[in] data to push.
108  * @return
109  * - 0 on success.
110  * - -1 on error.
111  */
112 int fr_fifo_push(fr_fifo_t *fi, void *data)
113 {
114  if (!fi || !data) return -1;
115 
116  if (fi->num >= fi->max) return -1;
117 
118 #ifndef TALLOC_GET_TYPE_ABORT_NOOP
119  if (fi->type) _talloc_get_type_abort(data, fi->type, __location__);
120 #endif
121 
122  fi->data[fi->last++] = data;
123  if (fi->last >= fi->max) fi->last = 0;
124  fi->num++;
125 
126  return 0;
127 }
128 
129 /** Pop data off of the fifo
130  *
131  * @param[in] fi FIFO to pop data from.
132  * @return
133  * - The data popped.
134  * - NULL if the queue is empty.
135  */
136 void *fr_fifo_pop(fr_fifo_t *fi)
137 {
138  void *data;
139 
140  if (!fi || (fi->num == 0)) return NULL;
141 
142  data = fi->data[fi->first++];
143 
144  if (fi->first >= fi->max) {
145  fi->first = 0;
146  }
147  fi->num--;
148 
149  return data;
150 }

152 /** Examine the next element that would be popped
153  *
154  * @param[in] fi FIFO to peek at.
155  * @return
156  * - The data at the head of the queue
157  * - NULL if the queue is empty.
158  */
159 void *fr_fifo_peek(fr_fifo_t *fi)
160 {
161  if (!fi || (fi->num == 0)) return NULL;
162 
163  return fi->data[fi->first];
164 }
