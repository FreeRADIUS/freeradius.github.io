63  */
64 static struct msgb *mtp_create_slta(struct mtp_link_set *set, int sls,
65  struct mtp_level_3_mng *in_mng, int l3_len)
66 {
67  struct mtp_level_3_hdr *hdr;

73 
74  hdr = (struct mtp_level_3_hdr *) out->l2h;

79  mng->cmn.h0 = MTP_TST_MSG_GRP;

81  mng->length = l3_len - 2;
82  msgb_put(out, mng->length);

147 /** Create an SCCP SCMG message
148  *
149  * SCMG messages are used to manage SCCP link states.
150  *
151  * @param set to send the message out on.
152  * @param type Sub System Status Test?
153  * @param assn
154  * @param apoc
155  * @param sls
156  */
157 static struct msgb *mtp_sccp_alloc_scmg(struct mtp_link_set *set,
158  int type, int assn, int apoc, int sls)
159 {
160  struct sccp_data_unitdata *udt;
161  struct sccp_con_ctrl_prt_mgt *prt;
162  struct mtp_level_3_hdr *hdr;
163  uint8_t *data;
164 
165 
166  struct msgb *out = mtp_msg_alloc(set);
167 
168  if (!out)
169  return NULL;
170 
171  hdr = (struct mtp_level_3_hdr *) out->l2h;
172  hdr->ser_ind = MTP_SI_MNT_SCCP;
173 
174  /* this appears to be round robin or such.. */
175  hdr->addr = MTP_ADDR(sls % 16, set->dpc, set->sccp_opc);
176 
177  /* generate the UDT message... libsccp does not offer formating yet */
178  udt = (struct sccp_data_unitdata *) msgb_put(out, sizeof(*udt));
179  udt->type = SCCP_MSG_TYPE_UDT;
180  udt->proto_class = SCCP_PROTOCOL_CLASS_0;
181  udt->variable_called = 3;
182  udt->variable_calling = 5;
183  udt->variable_data = 7;
184 
185  /* put the called and calling address. It is LV */
186  data = msgb_put(out, 2 + 1);
187  data[0] = 2;
188  data[1] = 0x42;
189  data[2] = 0x1;
190 
191  data = msgb_put(out, 2 + 1);
192  data[0] = 2;
193  data[1] = 0x42;
194  data[2] = 0x1;
195 
196  data = msgb_put(out, 1);
197  data[0] = sizeof(*prt);
198 
199  prt = (struct sccp_con_ctrl_prt_mgt *) msgb_put(out, sizeof(*prt));
200  prt->sst = type;
201  prt->assn = assn;
202  prt->apoc = apoc;
203  prt->mul_ind = 0;
204 
205  return out;
206 }

280 /** Verify a link
281  *
282  * Here we send a TFA (Transfer Allowed Message) to indicate we're OK to
283  * receive data on this link.
284  */
285 int mtp_link_verified(struct mtp_link *link)
286 {
287  struct mtp_link_set *set = link->set;
288 
289  /* the link set is already up */
290  if (set->linkset_up)
291  return 0;
292 
293  set->linkset_up = 1;
294  if (set->timeout_t18 != 0)
295  osmo_timer_schedule(&set->T18, set->timeout_t18, 0);
296  if (set->timeout_t20 != 0)
297  osmo_timer_schedule(&set->T20, set->timeout_t20, 0);
298 
299  /* More the functionality of a SSP here... */
300  if (set->sccp_opc != set->opc &&
301  send_tfa(link, set->sccp_opc) != 0) {
302  LOGP(DINP, LOGL_ERROR,
303  "Failed to send TFA for OPC %d on linkset %d.\n", set->sccp_opc, set->nr);
304  }
305 
306  if (set->isup_opc != set->opc &&
307  send_tfa(link, set->isup_opc) != 0) {
308  LOGP(DINP, LOGL_ERROR,
309  "Failed to send TFA for OPC %d on linkset %d.\n", set->sccp_opc, set->nr);
310  }
311 
312  if (set->timeout_t18 == 0)
313  linkset_t18_cb(set);
314  if (set->timeout_t20 == 0)
315  linkset_t20_cb(set);
316 
317  return 0;
318 }

320 /** T18 is used on linkset restart when we're an STP (Signalling Transfer Point)
321  *
322  * It's shorter than the T20, and indicates a period where we listen, but don't
323  * rebroadcast TFAs from adjacent connections.
324  *
325  * The time between T18 and T20 is when we rebroadcast link advertisements we've
326  * received, to adjacent SPs.
327  */
328 static void linkset_t18_cb(void *_set)
329 {
330  struct mtp_link_set *set = _set;
331  struct mtp_link *link = set->slc[0];
332 
333  if (!link) {
334  LOGP(DINP, LOGL_ERROR,
335  "Linkset restart but no link available on linkset %d\n", set->nr);
336  osmo_timer_del(&set->T20);
337  set->linkset_up = 0;
338  return;
339  }
340 
341  /* TODO: now send out routing states */
342  LOGP(DINP, LOGL_NOTICE, "The linkset %d has collected routing data.\n", set->nr);
343  set->sccp_up = 1;
344 }

346 /** T20 is used on linkset restart
347  *
348  * T20 is used to time a listening period (on MTP reset), where the SEP (Signalling End Point)
349  * listens for routing advertisements from adjacent nodes.
350  *
351  */
352 static void linkset_t20_cb(void *_set)
353 {
354  struct mtp_link_set *set = _set;
355  struct mtp_link *link = set->slc[0];
356 
357  if (!link) {
358  LOGP(DINP, LOGL_ERROR,
359  "Linkset restart but no link available on linkset %d\n", set->nr);
360  osmo_timer_del(&set->T20);
361  set->linkset_up = 0;
362  return;
363  }
364 
365  /* Send the TRA for all PCs */
366  if (send_tra(link, set->opc) != 0)
367  return;
368 
369  LOGP(DINP, LOGL_NOTICE,
370  "The linkset %d/%s is considered running.\n", set->nr, set->name);
371  return;
372 }

374 /** Process an MTP3 link signalling message that's not destined for a higher layer
375  *
376  */
377 static int mtp_link_sign_msg(struct mtp_link_set *set, struct mtp_level_3_hdr *hdr, int l3_len)
378 {
379  struct mtp_level_3_cmn *cmn;
380  uint16_t *apc;
381 
382  if (hdr->ni != set->ni || l3_len < 1) {
383  LOGP(DINP, LOGL_ERROR, "Unhandled data (ni: %d len: %d)\n",
384  hdr->ni, l3_len);
385  return -1;
386  }
387 
388  cmn = (struct mtp_level_3_cmn *) &hdr->data[0];
389  LOGP(DINP, LOGL_DEBUG, "reg msg: h0: 0x%x h1: 0x%x\n",
390  cmn->h0, cmn->h1);
391 
392  switch (cmn->h0) {
393  case MTP_TRF_RESTR_MSG_GRP:
394  switch (cmn->h1) {
395  case MTP_RESTR_MSG_ALLWED:
396  LOGP(DINP, LOGL_INFO,
397  "Received TRA on linkset %d/%s.\n", set->nr, set->name);
398  /*
399  * TODO: routing should be done on a higher level. This should not
400  * arrive after we expired the timer but we are friendly here and
401  * respond with a TFA and TRA...
402  */
403  osmo_timer_del(&set->T18);
404  osmo_timer_del(&set->T20);
405  linkset_t18_cb(set);
406  linkset_t20_cb(set);
407  return 0;
408  break;
409  }
410  break;
411  case MTP_PROHIBIT_MSG_GRP:
412  switch (cmn->h1) {
413  case MTP_PROHIBIT_MSG_SIG:
414  if (l3_len < 3) {
415  LOGP(DINP, LOGL_ERROR, "TFP is too short on %d/%s.\n", set->nr, set->name);
416  return -1;
417  }
418 
419  apc = (uint16_t *) &hdr->data[1];
420  LOGP(DINP, LOGL_INFO,
421  "TFP for the affected point code %d on %d/%s\n",
422  *apc, set->nr, set->name);
423  return 0;
424  break;
425  }
426  break;
427  }
428 
429  LOGP(DINP, LOGL_ERROR, "Unknown message:%d/%d %s on %d/%s.\n",
430  cmn->h0, cmn->h1, osmo_hexdump(&hdr->data[0], l3_len),
431  set->nr, set->name);
432  return -1;
433 }

435 /** Process an MTP3 link message that's not destined for a higher layer
436  *
437  * This is so we can respond, or process the response to link tests, to prove
438  * that the link is up.
439  *
440  * @param link the message was received on.
441  * @param hdr Pointer to the layer 3 header.
442  * @param l3_len Length of the layer 3 data.
443  */
444 static int mtp_link_regular_msg(struct mtp_link *link, struct mtp_level_3_hdr *hdr, int l3_len)
445 {
446  struct msgb *out;
447  struct mtp_level_3_mng *mng;
448 
449  if (hdr->ni != link->set->ni || l3_len < 1) {
450  LOGP(DINP, LOGL_ERROR, "Unhandled data (ni: %d len: %d)\n",
451  hdr->ni, l3_len);
452  return -1;
453  }
454 
455  if (MTP_READ_DPC(hdr->addr) != link->set->opc) {
456  LOGP(DINP, LOGL_ERROR, "MSG for OPC %d not handled on %d/%s\n",
457  MTP_READ_DPC(hdr->addr), link->set->nr, link->set->name);
458  return -1;
459  }
460 
461  mng = (struct mtp_level_3_mng *) &hdr->data[0];
462  LOGP(DINP, LOGL_DEBUG, "reg msg: h0: 0x%x h1: 0x%x\n",
463  mng->cmn.h0, mng->cmn.h1);
464 
465  switch (mng->cmn.h0) {
466  case MTP_TST_MSG_GRP:
467  switch (mng->cmn.h1) {
468  case MTP_TST_MSG_SLTM:
469  /* simply respond to the request... */
470  out = mtp_create_slta(link->set,
471  MTP_LINK_SLS(hdr->addr),
472  mng, l3_len);
473  if (!out)
474  return -1;
475  mtp_link_submit(link, out);
476  return 0;
477  break;
478  case MTP_TST_MSG_SLTA:
479  /* If this link is proven set it up */
480  if (mtp_link_slta(link, l3_len, mng) == 0)
481  mtp_link_verified(link);
482  break;
483  }
484  break;
485  }
486 
487  return -1;
488 }

490 /** Handler for incoming SCCP data from lower layer (MTP3)
491  *
492  * Will attempt to dispatch SCCP data to a higher level application.
493  */
494 static int mtp_link_sccp_data(struct mtp_link_set *set, struct mtp_level_3_hdr *hdr, struct msgb *msg, int l3_len)
495 {
496  struct msgb *out;
497  struct sccp_con_ctrl_prt_mgt *prt;
498  struct sccp_parse_result sccp;
499  int type;
500 
501  msg->l2h = &hdr->data[0];
502  if (msgb_l2len(msg) != l3_len) {
503  LOGP(DINP, LOGL_ERROR, "Size is wrong after playing with the l2h header.\n");
504  return -1;
505  }
506 
507  if (!set->sccp_up) {
508  LOGP(DINP, LOGL_ERROR, "SCCP traffic is not allowed on %d/%s\n",
509  set->nr, set->name);
510  return -1;
511  }
512 
513  memset(&sccp, 0, sizeof(sccp));
514  if (sccp_parse_header(msg, &sccp) != 0) {
515  LOGP(DINP, LOGL_ERROR, "Failed to parsed SCCP header.\n");
516  return -1;
517  }
518 
519  /*
520  * If it's a Sub System Test message, deal with it locally
521  * instead of forwarding it to the application.
522  */
523  if ((sccp_determine_msg_type(msg) == SCCP_MSG_TYPE_UDT) && (msg->l3h[0] == SCCP_SST)) {
524  if (msgb_l3len(msg) != 5) {
525  LOGP(DINP, LOGL_ERROR, "SCCP UDT msg of unexpected size: %u\n", msgb_l3len(msg));
526  return -1;
527  }
528 
529  prt = (struct sccp_con_ctrl_prt_mgt *) &msg->l3h[0];
530  if (prt->apoc != MTP_MAKE_APOC(set->sccp_opc)) {
531  LOGP(DINP, LOGL_ERROR, "Unknown APOC: %u/%u on %d/%s\n",
532  ntohs(prt->apoc), prt->apoc, set->nr, set->name);
533  type = SCCP_SSP;
534  } else if (!set->supported_ssn[prt->assn]) {
535  LOGP(DINP, LOGL_ERROR, "Unknown affected SSN assn: %u on %d/%s\n",
536  prt->assn, set->nr, set->name);
537  type = SCCP_SSP;
538  } else {
539  type = SCCP_SSA;
540  }
541 
542  out = mtp_sccp_alloc_scmg(set, type, prt->assn, prt->apoc,
543  MTP_LINK_SLS(hdr->addr));
544  if (!out)
545  return -1;
546 
547  mtp_link_submit(set->slc[MTP_LINK_SLS(hdr->addr)], out);
548 
549  return 0;
550  }
551 
552  rate_ctr_inc(&set->ctrg->ctr[MTP_LSET_SCCP_IN_MSG]);
553 
554  /*
555  * Forward the SCCP message to the SCCP application.
556  */
557  set->sccp_data_available_cb(set, msg, MTP_LINK_SLS(hdr->addr));
558 
559  return 0;
560 }

562 /** Receive data coming up from lower level (M2UA/M3UA etc...)
563  *
564  * Dispatch data to another handler depending on what the the
565  * service indicator in the MTP3 packet was.
566  *
567  * @param link message was received on.
568  * @param msg from the lower layer.
569  */
570 int mtp_link_handle_data(struct mtp_link *link, struct msgb *msg)
571 {
572  int rc = -1;
573  struct mtp_level_3_hdr *hdr;
574  int l3_len;
575 
576  if (!msg->l2h || msgb_l2len(msg) < sizeof(*hdr))
577  return -1;
578 
579  if (!link->set->running) {
580  LOGP(DINP, LOGL_ERROR,
581  "Link %d/%s of %d/%s is not running. Call mtp_link_reset first.\n",
582  link->nr, link->name, link->set->nr, link->set->name);
583  return -1;
584  }
585 
586  hdr = (struct mtp_level_3_hdr *) msg->l2h;
587  l3_len = msgb_l2len(msg) - sizeof(*hdr);
588 
589  rate_ctr_inc(&link->ctrg->ctr[MTP_LNK_IN]);
590  rate_ctr_inc(&link->set->ctrg->ctr[MTP_LSET_TOTA_IN_MSG]);
591 
592  switch (hdr->ser_ind) {
593  case MTP_SI_MNT_SNM_MSG:
594  rc = mtp_link_sign_msg(link->set, hdr, l3_len);
595  break;
596 
597  case MTP_SI_MNT_REG_MSG:
598  rc = mtp_link_regular_msg(link, hdr, l3_len);
599  break;
600 
601  case MTP_SI_MNT_SCCP:
602  rc = mtp_link_sccp_data(link->set, hdr, msg, l3_len);
603  break;
604 
605  case MTP_SI_MNT_ISUP:
606  msg->l3h = &hdr->data[0];
607  rate_ctr_inc(&link->set->ctrg->ctr[MTP_LSET_IUSP_IN_MSG]);
608  rc = mtp_link_set_isup(link->set, msg, MTP_LINK_SLS(hdr->addr));
609 
610  break;
611  default:
612  fprintf(stderr, "Unhandled: %u\n", hdr->ser_ind);
613  break;
614  }
615 
616  return rc;
617 }

723 /** Find another link within a link set
724  *
725  * @param set to search for link in.
726  * @param data If we find this, use the link *after* it, or just use any available link
727  * @return next mtp link.

729 static struct mtp_link *find_next_link(struct mtp_link_set *set, struct mtp_link *data)
730 {

732  struct mtp_link *next;
733 
734  if (llist_empty(&set->links))
735  return NULL;
736 
737  if (data == NULL)
738  found = 1;
739 
740  /* try to find the next one */
741  llist_for_each_entry(next, &set->links, entry) {
742  if (found && next->available)
743  return next;
744  if (next == data)
745  found = 1;
746  }
747 
748  /* try to find any one */
749  llist_for_each_entry(next, &set->links, entry)
750  if (next->available)
751  return next;
752 
753  return NULL;
754 }
