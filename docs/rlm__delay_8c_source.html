55 static void _delay_done(UNUSED module_ctx_t const *mctx, request_t *request, void *rctx, fr_time_t fired)
56 {
57  fr_time_t *yielded = talloc_get_type_abort(rctx, fr_time_t);

59  RDEBUG2("Delay done");

65  if (!fr_cond_assert(fired >= *yielded)) REDEBUG("Unexpected resume time");

67  unlang_interpret_resumable(request);

86 static int delay_add(request_t *request, fr_time_t *resume_at, fr_time_t now,
87  fr_time_t delay, bool force_reschedule, bool relative)
88 {
89  /*
90  * Delay is zero (and reschedule is not forced)
91  */
92  if (!force_reschedule && (delay == 0)) return 1;
93 
94  /*
95  * Process the delay relative to the start of packet processing
96  */
97  if (relative) {
98  *resume_at = request->packet->timestamp + delay;
99  } else {
100  *resume_at = now + delay;
101  }
102 
103  /*
104  * If resume_at is in the past (and reschedule is not forced), just return noop
105  */
106  if (!force_reschedule && (*resume_at <= now)) return 1;
107 
108  if (*resume_at > now) {
109  RDEBUG2("Delaying request by ~%pVs", fr_box_time_delta(*resume_at - now));
110  } else {
111  RDEBUG2("Rescheduling request");
112  }
113 
114  return 0;
115 }

144 static unlang_action_t CC_HINT(nonnull) mod_delay(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
145 {
146  rlm_delay_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_delay_t);
147  fr_time_delta_t delay;
148  fr_time_t resume_at, *yielded_at;
149 
150  if (inst->delay) {
151  if (tmpl_aexpand_type(request, &delay, FR_TYPE_TIME_DELTA,
152  request, inst->delay, NULL, NULL) < 0) RETURN_MODULE_FAIL;
153  } else {
154  delay = 0;
155  }
156 
157  /*
158  * Record the time that we yielded the request
159  */
160  MEM(yielded_at = talloc(request, fr_time_t));
161  *yielded_at = fr_time();
162 
163  /*
164  * Setup the delay for this request
165  */
166  if (delay_add(request, &resume_at, *yielded_at, delay,
167  inst->force_reschedule, inst->delay) != 0) {
168  RETURN_MODULE_NOOP;
169  }
170 
171  /*
172  * FIXME - Should print wallclock time
173  */
174  RDEBUG3("Current time %pVs, resume time %pVs",
175  fr_box_time_delta(*yielded_at), fr_box_time_delta(resume_at));
176 
177  if (unlang_module_timeout_add(request, _delay_done, yielded_at, resume_at) < 0) {
178  RPEDEBUG("Adding event failed");
179  RETURN_MODULE_FAIL;
180  }
181 
182  return unlang_module_yield(request, mod_delay_return, mod_delay_cancel, yielded_at);
183 }

185 static xlat_action_t xlat_delay_resume(TALLOC_CTX *ctx, fr_cursor_t *out,
186  request_t *request,
187  UNUSED void const *xlat_inst, UNUSED void *xlat_thread_inst,
188  UNUSED fr_value_box_t **in, void *rctx)
189 {
190  fr_time_t *yielded_at = talloc_get_type_abort(rctx, fr_time_t);
191  fr_time_t delayed;
192  fr_value_box_t *vb;
193 
194  delayed = fr_time() - *yielded_at;
195  talloc_free(yielded_at);
196 
197  MEM(vb = fr_value_box_alloc(ctx, FR_TYPE_TIME_DELTA, NULL, false));
198  vb->vb_time_delta = delayed;
199 
200  RDEBUG3("Request delayed by %pVs", vb);
201 
202  fr_cursor_insert(out, vb);
203 
204  return XLAT_ACTION_DONE;
205 }

215 /** Xlat to delay the request
216  *
217  * Example (delay 2 seconds):
218 @verbatim
219 "%{delay:2}"
220 @endverbatim
221  *
222  * @ingroup xlat_functions
223  */
224 static xlat_action_t xlat_delay(TALLOC_CTX *ctx, UNUSED fr_cursor_t *out,
225  request_t *request, void const *xlat_inst, UNUSED void *xlat_thread_inst,
226  fr_value_box_t **in)
227 {
228  rlm_delay_t const *inst;
229  void *instance;
230  fr_time_delta_t delay;
231  fr_time_t resume_at, *yielded_at;
232 
233  memcpy(&instance, xlat_inst, sizeof(instance)); /* Stupid const issues */
234 
235  inst = talloc_get_type_abort(instance, rlm_delay_t);
236 
237  /*
238  * Record the time that we yielded the request
239  */
240  MEM(yielded_at = talloc(request, fr_time_t));
241  *yielded_at = fr_time();
242 
243  /*
244  * If there's no input delay, just yield and
245  * immediately re-enqueue the request.
246  * This is very useful for testing.
247  */
248  if (!*in) {
249  if (!fr_cond_assert(delay_add(request, &resume_at, *yielded_at, 0, true, true) == 0)) {
250  return XLAT_ACTION_FAIL;
251  }
252  goto yield;
253  }
254 
255  if (fr_value_box_list_concat(ctx, *in, in, FR_TYPE_STRING, true) < 0) {
256  RPEDEBUG("Failed concatenating input");
257  talloc_free(yielded_at);
258  return XLAT_ACTION_FAIL;
259  }
260 
261  if (fr_time_delta_from_str(&delay, (*in)->vb_strvalue, FR_TIME_RES_SEC) < 0) {
262  RPEDEBUG("Failed parsing delay time");
263  talloc_free(yielded_at);
264  return XLAT_ACTION_FAIL;
265  }
266 
267  if (delay_add(request, &resume_at, *yielded_at, delay, inst->force_reschedule, inst->relative) != 0) {
268  RDEBUG2("Not adding delay");
269  talloc_free(yielded_at);
270  return XLAT_ACTION_DONE;
271  }
272 
273 yield:
274  /*
275  * FIXME - Should print wallclock time
276  */
277  RDEBUG3("Current time %pVs, resume time %pVs", fr_box_time_delta(*yielded_at), fr_box_time_delta(resume_at));
278 
279  if (unlang_xlat_event_timeout_add(request, _xlat_delay_done, yielded_at, resume_at) < 0) {
280  RPEDEBUG("Adding event failed");
281  return XLAT_ACTION_FAIL;
282  }
283 
284  return unlang_xlat_yield(request, xlat_delay_resume, xlat_delay_cancel, yielded_at);
285 }
