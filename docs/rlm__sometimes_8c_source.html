48 static int mod_instantiate(void *instance, CONF_SECTION *conf)
49 {
50  rlm_sometimes_t *inst = instance;
51 
52  /*
53  * Convert the rcode string to an int, and get rid of it
54  */
55  inst->rcode = fr_table_value_by_str(rcode_table, inst->rcode_str, RLM_MODULE_UNKNOWN);
56  if (inst->rcode == RLM_MODULE_UNKNOWN) {
57  cf_log_err(conf, "Unknown module return code '%s'", inst->rcode_str);
58  return -1;
59  }
60 
61  if ((inst->percentage < 0) || (inst->percentage > 100)) {
62  cf_log_err(conf, "Invalid value for 'percentage'. It must be 0..100 inclusive");
63  return -1;
64  }
65 
66  return 0;
67 }

69 /*
70  * A lie! It always returns!
71  */
72 static unlang_action_t sometimes_return(rlm_rcode_t *p_result, void const *instance, request_t *request,
73  fr_radius_packet_t *packet, fr_radius_packet_t *reply)
74 {
75  uint32_t hash;
76  rlm_sometimes_t const *inst = talloc_get_type_abort_const(instance, rlm_sometimes_t);
77  fr_pair_t *vp;
78  float value;
79 
80  /*
81  * Set it to NOOP and the module will always do nothing
82  */
83  if (inst->rcode == RLM_MODULE_NOOP) RETURN_MODULE_RCODE(inst->rcode);
84 
85  /*
86  * Hash based on the given key. Usually User-Name.
87  */
88  tmpl_find_vp(&vp, request, inst->key);
89  if (!vp) RETURN_MODULE_NOOP;
90 
91  switch (vp->vp_type) {
92  case FR_TYPE_OCTETS:
93  case FR_TYPE_STRING:
94  hash = fr_hash(vp->data.datum.ptr, vp->vp_length);
95  break;
96 
97  case FR_TYPE_STRUCTURAL:
98  RETURN_MODULE_FAIL;
99 
100  default:
101  hash = fr_hash(&vp->data.datum, fr_value_box_field_sizes[vp->vp_type]);
102  break;
103  }
104 
105  hash &= 0xffff; /* all we need are 2^16 bits of precision */
106  value = hash;
107  value /= (1 << 16);
108  value *= 100;
109 
110  if (value > inst->percentage) RETURN_MODULE_NOOP;
111 
112  /*
113  * If we're returning "handled", then set the packet
114  * code in the reply, so that the server responds.
115  */
116  if ((inst->rcode == RLM_MODULE_HANDLED) && reply) {
117  switch (packet->code) {
118  case FR_CODE_ACCESS_REQUEST:
119  reply->code = FR_CODE_ACCESS_ACCEPT;
120  break;
121 
122  case FR_CODE_ACCOUNTING_REQUEST:
123  reply->code = FR_CODE_ACCOUNTING_RESPONSE;
124  break;
125 
126  case FR_CODE_COA_REQUEST:
127  reply->code = FR_CODE_COA_ACK;
128  break;
129 
130  case FR_CODE_DISCONNECT_REQUEST:
131  reply->code = FR_CODE_DISCONNECT_ACK;
132  break;
133 
134  default:
135  break;
136  }
137  }
138 
139  RETURN_MODULE_RCODE(inst->rcode);
140 }
