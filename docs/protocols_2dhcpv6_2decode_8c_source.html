48 static ssize_t decode_raw(TALLOC_CTX *ctx, fr_cursor_t *cursor, UNUSED fr_dict_t const *dict,
49  fr_dict_attr_t const *parent,
50  uint8_t const *data, size_t const data_len, void *decoder_ctx)
51 {
52  fr_pair_t *vp;
53  fr_dict_attr_t const *da;
54  fr_dhcpv6_decode_ctx_t *packet_ctx = decoder_ctx;
55  ssize_t slen;
56 
57 #ifdef __clang_analyzer__
58  if (!packet_ctx || !packet_ctx->tmp_ctx || !parent->parent) return PAIR_DECODE_FATAL_ERROR;
59 #endif
60 
61  FR_PROTO_HEX_DUMP(data, data_len, "decode_raw");
62 
63  /*
64  * Re-write the attribute to be "raw". It is
65  * therefore of type "octets", and will be
66  * handled below.
67  */
68  da = fr_dict_unknown_afrom_fields(packet_ctx->tmp_ctx, parent->parent,
69  fr_dict_vendor_num_by_da(parent), parent->attr);
70  if (!da) {
71  fr_strerror_printf("%s: Internal sanity check %d", __FUNCTION__, __LINE__);
72  return PAIR_DECODE_OOM;
73  }
74 
75  vp = fr_pair_afrom_da(ctx, da);
76  if (!vp) return PAIR_DECODE_OOM;
77 
78  slen = fr_value_box_from_network(vp, &vp->data, vp->da->type, vp->da, data, data_len, true);
79  if (slen < 0) {
80  talloc_free(vp);
81  fr_dict_unknown_free(&da);
82  return slen;
83  }
84 
85  vp->type = VT_DATA;
86  vp->vp_tainted = true;
87  fr_cursor_append(cursor, vp);
88  return data_len;
89 }

121 static ssize_t decode_value(TALLOC_CTX *ctx, fr_cursor_t *cursor, fr_dict_t const *dict,
122  fr_dict_attr_t const *parent,
123  uint8_t const *data, size_t const data_len, void *decoder_ctx)
124 {
125  ssize_t slen;
126  fr_pair_t *vp;
127  fr_dict_attr_t const *tlv;
128  uint8_t prefix_len;
129 
130  FR_PROTO_HEX_DUMP(data, data_len, "decode_value");
131 
132  switch (parent->type) {
133  /*
134  * Address MAY be shorter than 16 bytes.
135  */
136  case FR_TYPE_IPV6_PREFIX:
137  if ((data_len == 0) || (data_len > 17)) {
138  raw:
139  return decode_raw(ctx, cursor, dict, parent, data, data_len, decoder_ctx);
140 
141  };
142 
143  /*
144  * No address, the prefix length MUST be zero.
145  */
146  if (data_len == 1) {
147  if (data[0] != 0) goto raw;
148 
149  vp = fr_pair_afrom_da(ctx, parent);
150  if (!vp) return PAIR_DECODE_OOM;
151 
152  vp->vp_ip.af = AF_INET6;
153  vp->vp_ip.scope_id = 0;
154  vp->vp_ip.prefix = 0;
155  memset(&vp->vp_ipv6addr, 0, sizeof(vp->vp_ipv6addr));
156  break;
157  }
158 
159  prefix_len = data[0];
160 
161  /*
162  * If we have a /64 prefix but only 7 bytes of
163  * address, that's an error.
164  */
165  if ((prefix_len >> 3) > (data_len - 1)) goto raw;
166 
167  vp = fr_pair_afrom_da(ctx, parent);
168  if (!vp) return PAIR_DECODE_OOM;
169 
170  vp->vp_ip.af = AF_INET6;
171  vp->vp_ip.scope_id = 0;
172  vp->vp_ip.prefix = prefix_len;
173  memset(&vp->vp_ipv6addr, 0, sizeof(vp->vp_ipv6addr));
174  memcpy(&vp->vp_ipv6addr, data + 1, data_len - 1);
175  break;
176 
177  /*
178  * A bool is encoded as an empty option if it's
179  * true. A bool is omitted entirely if it's
180  * false.
181  */
182  case FR_TYPE_BOOL:
183  if (data_len != 0) goto raw;
184  vp = fr_pair_afrom_da(ctx, parent);
185  if (!vp) return PAIR_DECODE_OOM;
186  vp->vp_bool = true;
187  break;
188 
189  /*
190  * A standard 32bit integer, but unlike normal UNIX timestamps
191  * starts from the 1st of January 2000.
192  *
193  * In the encoder we subtract 30 years to any values, so
194  * here we need to add that to the time here.
195  */
196  case FR_TYPE_DATE:
197  vp = fr_pair_afrom_da(ctx, parent);
198  if (!vp) return PAIR_DECODE_OOM;
199 
200  if (fr_value_box_from_network(vp, &vp->data, vp->da->type, vp->da, data, data_len, true) < 0) {
201  talloc_free(vp);
202  goto raw;
203  }
204  vp->vp_date += ((fr_time_t) DHCPV6_DATE_OFFSET) * NSEC;
205  break;
206 
207  case FR_TYPE_STRUCT:
208  slen = fr_struct_from_network(ctx, cursor, parent, data, data_len, &tlv,
209  decode_value_trampoline, decoder_ctx);
210  if (slen < 0) return slen;
211 
212  if (tlv) {
213  fr_strerror_printf("decode children not implemented");
214  return PAIR_DECODE_FATAL_ERROR;
215  }
216 
217  return data_len;
218 
219  case FR_TYPE_GROUP:
220  {
221  fr_cursor_t child_cursor;
222  fr_pair_t *head = NULL;
223 
224  vp = fr_pair_afrom_da(ctx, parent);
225  if (!vp) return PAIR_DECODE_OOM;
226 
227  /*
228  * Child VPs go into the child group, not in the
229  * main parent list. We start decoding
230  * attributes from the dictionary root, not from
231  * this parent. We also don't decode an option
232  * header, as we're just decoding the values
233  * here.
234  */
235  fr_cursor_init(&child_cursor, &head);
236  slen = decode_tlvs(vp, &child_cursor, dict, fr_dict_root(dict_dhcpv6), data, data_len, decoder_ctx, false);
237  if (slen < 0) {
238  talloc_free(vp);
239  goto raw;
240  }
241  vp->vp_group = head;
242  break;
243  }
244 
245  default:
246  vp = fr_pair_afrom_da(ctx, parent);
247  if (!vp) return PAIR_DECODE_OOM;
248 
249  if (fr_value_box_from_network(vp, &vp->data, vp->da->type, vp->da, data, data_len, true) < 0) {
250  talloc_free(vp);
251  goto raw;
252  }
253  break;
254  }
255 
256  vp->type = VT_DATA;
257  vp->vp_tainted = true;
258  fr_cursor_append(cursor, vp);
259  return data_len;
260 }

263 static ssize_t decode_array(TALLOC_CTX *ctx, fr_cursor_t *cursor, fr_dict_t const *dict,
264  fr_dict_attr_t const *parent,
265  uint8_t const *data, size_t const data_len, void *decoder_ctx)
266 {
267  uint8_t const *p = data, *end = p + data_len;
268  ssize_t slen;
269  size_t element_len;
270 
271  FR_PROTO_HEX_DUMP(data, data_len, "decode_array");
272 
273  if (!fr_cond_assert_msg(parent->flags.array,
274  "%s: Internal sanity check failed, attribute \"%s\" does not have array bit set",
275  __FUNCTION__, parent->name)) return PAIR_DECODE_FATAL_ERROR;
276 
277  /*
278  * Fixed-size fields get decoded with a simple decoder.
279  */
280  element_len = fr_dhcpv6_attr_sizes[parent->type][0];
281  if (element_len > 0) {
282  while (p < end) {
283  /*
284  * Not enough room for one more element,
285  * decode the last bit as raw data.
286  */
287  if ((size_t) (end - p) < element_len) {
288  slen = decode_raw(ctx, cursor, dict, parent, p, end - p , decoder_ctx);
289  if (slen < 0) return slen;
290  break;
291  }
292 
293  slen = decode_value(ctx, cursor, dict, parent, p, element_len, decoder_ctx);
294  if (slen < 0) return slen;
295  if (!fr_cond_assert((size_t) slen == element_len)) return -(p - data);
296 
297  p += slen;
298  }
299 
300  /*
301  * We MUST have decoded the entire input. If
302  * not, we ignore the extra bits.
303  */
304  return data_len;
305  }
306 
307  /*
308  * If the data is variable length i.e. strings or octets
309  * there is a length field before each element.
310  *
311  * +-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-...-+-+-+-+-+-+-+
312  * | text-len | String |
313  * +-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-...-+-+-+-+-+-+-+
314  */
315  while (p < end) {
316  if ((end - p) < 2) {
317  raw:
318  slen = decode_raw(ctx, cursor, dict, parent, p, end - p , decoder_ctx);
319  if (slen < 0) return slen;
320  break;
321  }
322 
323  element_len = fr_net_to_uint16(p);
324  if ((p + 2 + element_len) > end) {
325  goto raw;
326  }
327 
328  p += 2;
329  slen = decode_value(ctx, cursor, dict, parent, p, element_len, decoder_ctx);
330  if (slen < 0) return slen;
331  p += slen;
332  }
333 
334  return data_len;
335 }

337 static ssize_t decode_dns_labels(TALLOC_CTX *ctx, fr_cursor_t *cursor, fr_dict_t const *dict,
338  fr_dict_attr_t const *parent,
339  uint8_t const *data, size_t const data_len, void *decoder_ctx)
340 {
341  ssize_t slen;
342  size_t total;
343  fr_pair_t *vp;
344  uint8_t const *next = data;
345 
346  FR_PROTO_HEX_DUMP(data, data_len, "decode_dns_labels");
347 
348  /*
349  * This function handles both single-valued and array
350  * types. It's just easier that way.
351  */
352  if (!parent->flags.array) {
353  slen = fr_dns_label_uncompressed_length(data, data_len, &next);
354  if (slen <= 0) goto raw;
355 
356  /*
357  * If the DNS label doesn't exactly fill the option, it's an error.
358  *
359  * @todo - we may want to remove this check.
360  */
361  if (next != (data + data_len)) goto raw;
362 
363  } else {
364  /*
365  * If any one of the labels are invalid, then treat the
366  * entire set as invalid.
367  */
368  slen = fr_dns_labels_network_verify(data, data_len);
369  if (slen < 0) {
370  raw:
371  return decode_raw(ctx, cursor, dict, parent, data, data_len, decoder_ctx);
372  }
373  }
374 
375  /*
376  * Loop over the input buffer, decoding the labels one by
377  * one.
378  */
379  for (total = 0; total < data_len; total += slen) {
380  vp = fr_pair_afrom_da(ctx, parent);
381  if (!vp) return PAIR_DECODE_OOM;
382 
383  /*
384  * Having verified the input above, this next
385  * function should never fail unless there's a
386  * bug in the code.
387  */
388  slen = fr_dns_label_to_value_box(vp, &vp->data, data, data_len, data + total, true);
389  if (slen <= 0) {
390  talloc_free(vp);
391  goto raw;
392  }
393 
394  vp->type = VT_DATA;
395  fr_cursor_append(cursor, vp);
396  }
397 
398  return data_len;
399 }

405 static ssize_t decode_tlvs(TALLOC_CTX *ctx, fr_cursor_t *cursor, fr_dict_t const *dict,
406  fr_dict_attr_t const *parent,
407  uint8_t const *data, size_t const data_len, void *decoder_ctx, bool do_raw)
408 {
409  uint8_t const *p, *end;
410 
411  FR_PROTO_HEX_DUMP(data, data_len, "decode_tlvs");
412 
413  if (!fr_cond_assert_msg((parent->type == FR_TYPE_TLV || (parent->type == FR_TYPE_VENDOR)),
414  "%s: Internal sanity check failed, attribute \"%s\" is not of type 'tlv'",
415  __FUNCTION__, parent->name)) return PAIR_DECODE_FATAL_ERROR;
416  p = data;
417  end = data + data_len;
418 
419  while (p < end) {
420  ssize_t slen;
421 
422  slen = decode_option(ctx, cursor, dict, parent, p, (end - p), decoder_ctx);
423  if (slen <= 0) {
424  if (!do_raw) return slen;
425 
426  slen = decode_raw(ctx, cursor, dict, parent, p, (end - p), decoder_ctx);
427  if (slen <= 0) return slen;
428  break;
429  }
430 
431  p += slen;
432  }
433 
434  return data_len;
435 }

438 static ssize_t decode_vsa(TALLOC_CTX *ctx, fr_cursor_t *cursor, fr_dict_t const *dict,
439  fr_dict_attr_t const *parent,
440  uint8_t const *data, size_t const data_len, void *decoder_ctx)
441 {
442  uint32_t pen;
443  fr_dict_attr_t const *da;
444  fr_dhcpv6_decode_ctx_t *packet_ctx = decoder_ctx;
445 
446  FR_PROTO_HEX_DUMP(data, data_len, "decode_vsa");
447 
448  if (!fr_cond_assert_msg(parent->type == FR_TYPE_VSA,
449  "%s: Internal sanity check failed, attribute \"%s\" is not of type 'vsa'",
450  __FUNCTION__, parent->name)) return PAIR_DECODE_FATAL_ERROR;
451 
452  /*
453  * Enterprise code plus at least one option header
454  */
455  if (data_len < 8) return decode_raw(ctx, cursor, dict, parent, data, data_len, decoder_ctx);
456 
457  memcpy(&pen, data, sizeof(pen));
458  pen = htonl(pen);
459 
460  /*
461  * Verify that the parent (which should be a VSA)
462  * contains a fake attribute representing the vendor.
463  *
464  * If it doesn't then this vendor is unknown, but we know
465  * vendor attributes have a standard format, so we can
466  * decode the data anyway.
467  */
468  da = fr_dict_attr_child_by_num(parent, pen);
469  if (!da) {
470  fr_dict_attr_t *n;
471 
472  if (fr_dict_unknown_vendor_afrom_num(packet_ctx->tmp_ctx, &n, parent, pen) < 0) {
473  return PAIR_DECODE_OOM;
474  }
475  da = n;
476  }
477 
478  FR_PROTO_TRACE("decode context %s -> %s", parent->name, da->name);
479 
480  return decode_tlvs(ctx, cursor, dict, da, data + 4, data_len - 4, decoder_ctx, true);
481 }

483 static ssize_t decode_option(TALLOC_CTX *ctx, fr_cursor_t *cursor, fr_dict_t const *dict,
484  fr_dict_attr_t const *parent,
485  uint8_t const *data, size_t const data_len, void *decoder_ctx)
486 {
487  unsigned int option;
488  size_t len;
489  ssize_t slen;
490  fr_dict_attr_t const *da;
491  fr_dhcpv6_decode_ctx_t *packet_ctx = decoder_ctx;
492 
493 #ifdef __clang_analyzer__
494  if (!packet_ctx || !packet_ctx->tmp_ctx) return PAIR_DECODE_FATAL_ERROR;
495 #endif
496 
497  /*
498  * Must have at least an option header.
499  */
500  if (data_len < 4) {
501  fr_strerror_printf("%s: Insufficient data", __FUNCTION__);
502  return -(data_len);
503  }
504 
505  option = DHCPV6_GET_OPTION_NUM(data);
506  len = DHCPV6_GET_OPTION_LEN(data);
507  if (len > (data_len - 4)) {
508  fr_strerror_printf("%s: Option overflows input. "
509  "Optional length must be less than %zu bytes, got %zu bytes",
510  __FUNCTION__, data_len - 4, len);
511  return PAIR_DECODE_FATAL_ERROR;
512  }
513 
514  da = fr_dict_attr_child_by_num(parent, option);
515  if (!da) {
516  da = fr_dict_unknown_afrom_fields(packet_ctx->tmp_ctx, parent, 0, option);
517  if (!da) return PAIR_DECODE_FATAL_ERROR;
518  }
519  FR_PROTO_TRACE("decode context changed %s -> %s",da->parent->name, da->name);
520 
521  /*
522  * Relay messages are weird, and contain complete DHCPv6
523  * packets, copied verbatim from the DHCPv6 client.
524  */
525  if (da == attr_relay_message) {
526  fr_pair_t *vp;
527  fr_cursor_t cursor_group;
528 
529  vp = fr_pair_afrom_da(ctx, attr_relay_message);
530  if (!vp) return PAIR_DECODE_FATAL_ERROR;
531 
532  fr_cursor_init(&cursor_group, &vp->vp_group);
533  slen = fr_dhcpv6_decode(vp, data + 4, len, &cursor_group);
534  if (slen < 0) {
535  talloc_free(vp);
536  return slen;
537  }
538 
539  fr_cursor_insert(cursor, vp);
540  } else if ((da->type == FR_TYPE_STRING) && !da->flags.extra && da->flags.subtype) {
541  slen = decode_dns_labels(ctx, cursor, dict, da, data + 4, len, decoder_ctx);
542 
543  } else if (da->flags.array) {
544  slen = decode_array(ctx, cursor, dict, da, data + 4, len, decoder_ctx);
545 
546  } else if (da->type == FR_TYPE_VSA) {
547  slen = decode_vsa(ctx, cursor, dict, da, data + 4, len, decoder_ctx);
548 
549  } else if (da->type == FR_TYPE_TLV) {
550  slen = decode_tlvs(ctx, cursor, dict, da, data + 4, len, decoder_ctx, true);
551 
552  } else {
553  slen = decode_value(ctx, cursor, dict, da, data + 4, len, decoder_ctx);
554  }
555 
556  if (slen < 0) return slen;
557 
558  return len + 4;
559 }

612 static ssize_t fr_dhcpv6_decode_proto(TALLOC_CTX *ctx, fr_pair_list_t *list, uint8_t const *data, size_t data_len, UNUSED void *proto_ctx)
613 {
