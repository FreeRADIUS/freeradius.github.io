225 /*
226  * Do any per-module bootstrapping that is separate to each
227  * configured instance of the module. e.g. set up connections
228  * to external databases, read configuration files, set up
229  * dictionary entries, etc.
230  *
231  * If configuration information is given in the config section
232  * that must be referenced in later calls, store a handle to it
233  * in *instance otherwise put a null pointer there.
234  */
235 static int mod_bootstrap(void *instance, UNUSED CONF_SECTION *conf)
236 {
237  rlm_test_t *inst = instance;

239  if (paircmp_register_by_name("Test-Paircmp", attr_user_name, false,
