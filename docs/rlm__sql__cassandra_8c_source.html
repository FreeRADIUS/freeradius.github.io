270 /** Log callback for libcassandra
271  *
272  * libcassandra seems to use this to log global events in the library, other messages
273  * like query errors are not logged here, and should be retrieved with functions like
274  * cass_future_error_message();
275  *
276  * Messages here do not need to be made available via sql_error.
277  *
278  * @param message Contains the log message and information about its source.
279  * @param data user data (not used).
280  */
281 static void _rlm_sql_cassandra_log(CassLogMessage const *message, UNUSED void *data)
282 {
283  switch (message->severity) {
284  case CASS_LOG_CRITICAL:
285  case CASS_LOG_ERROR:
286  if (DEBUG_ENABLED3) {
287  ERROR("%s[%d] %s: %s",
288  message->file, message->line, message->function, message->message);
289  } else {
290  ERROR("%s", message->message);
291  }
292  return;
293 
294  case CASS_LOG_WARN:
295  if (DEBUG_ENABLED3) {
296  WARN("%s[%d] %s: %s",
297  message->file, message->line, message->function, message->message);
298  } else {
299  WARN("%s", message->message);
300  }
301  return;
302 
303  case CASS_LOG_INFO:
304  case CASS_LOG_DISABLED:
305  case CASS_LOG_LAST_ENTRY:
306  if (DEBUG_ENABLED3) {
307  INFO("%s[%d] %s: %s",
308  message->file, message->line, message->function, message->message);
309  } else {
310  INFO("%s", message->message);
311  }
312  return;
313 
314  case CASS_LOG_DEBUG:
315  case CASS_LOG_TRACE:
316  default:
317  if (DEBUG_ENABLED3) {
318  DEBUG3("%s[%d] %s: %s",
319  message->file, message->line, message->function, message->message);
320  } else {
321  DEBUG2("%s", message->message);
322  }
323  return;
324  }
325 }

378 static sql_rcode_t sql_socket_init(rlm_sql_handle_t *handle, rlm_sql_config_t *config, fr_time_delta_t timeout)
379 {
380  rlm_sql_cassandra_conn_t *conn;
381  rlm_sql_cassandra_t *inst = config->driver;
382 
383  MEM(conn = handle->conn = talloc_zero(handle, rlm_sql_cassandra_conn_t));
384  talloc_set_destructor(conn, _sql_socket_destructor);
385 
386  /*
387  * We do this one inside sql_socket_init, to allow pool.start = 0 to
388  * work as expected (allow the server to start if Cassandra is
389  * unavailable).
390  */
391  if (!inst->done_connect_keyspace) {
392  CassFuture *future;
393  CassError ret;
394 
395  pthread_mutex_lock(&inst->connect_mutex);
396  if (!inst->done_connect_keyspace) {
397  /*
398  * Easier to do this here instead of mod_instantiate
399  * as we don't have a pointer to the pool.
400  */
401  cass_cluster_set_connect_timeout(inst->cluster, fr_time_delta_to_msec(timeout));
402 
403  DEBUG2("Connecting to Cassandra cluster");
404  future = cass_session_connect_keyspace(inst->session, inst->cluster, config->sql_db);
405  ret = cass_future_error_code(future);
406  if (ret != CASS_OK) {
407  const char *msg;
408  size_t msg_len;
409 
410  cass_future_error_message(future, &msg, &msg_len);
411  ERROR("Unable to connect: [%x] %s", (int)ret, msg);
412  cass_future_free(future);
413  pthread_mutex_unlock(&inst->connect_mutex);
414 
415  return RLM_SQL_ERROR;
416  }
417  cass_future_free(future);
418  inst->done_connect_keyspace = true;
419  }
420  pthread_mutex_unlock(&inst->connect_mutex);
421  }
422  conn->log_ctx = talloc_pool(conn, 1024); /* Pre-allocate some memory for log messages */
423 
424  return RLM_SQL_OK;
425 }

427 static sql_rcode_t sql_query(rlm_sql_handle_t *handle, rlm_sql_config_t *config, char const *query)
428 {
429  rlm_sql_cassandra_conn_t *conn = handle->conn;
430  rlm_sql_cassandra_t *conf = config->driver;
431  CassStatement *statement;
432  CassFuture *future;
433  CassError ret;
434 
435  statement = cass_statement_new_n(query, talloc_array_length(query) - 1, 0);
436  if (conf->consistency_str) cass_statement_set_consistency(statement, conf->consistency);
437 
438  future = cass_session_execute(conf->session, statement);
439  cass_statement_free(statement);
440 
441  ret = cass_future_error_code(future);
442  if (ret != CASS_OK) {
443  char const *error;
444  size_t len;
445 
446  cass_future_error_message(future, &error, &len);
447  sql_set_last_error(conn, error, len);
448  cass_future_free(future);
449 
450  switch (ret) {
451  case CASS_ERROR_SERVER_SYNTAX_ERROR:
452  case CASS_ERROR_SERVER_INVALID_QUERY:
453  return RLM_SQL_QUERY_INVALID;
454 
455  default:
456  return RLM_SQL_ERROR;
457  }
458  }
459 
460  conn->result = cass_future_get_result(future);
461  cass_future_free(future);
462 
463  return RLM_SQL_OK;
464 }

481 {
482  rlm_sql_cassandra_conn_t *conn = handle->conn;
483 
484  unsigned int fields, i;
485  char const **names;
486 
487  fields = sql_num_fields(handle, config);
488  if (fields == 0) return RLM_SQL_ERROR;
489 
490  MEM(names = talloc_array(handle, char const *, fields));
491 
492  for (i = 0; i < fields; i++) {
493  const char *col_name;
494  size_t col_name_len;
495 
496  /* Writes out a pointer to a buffer in the result */
497  if (cass_result_column_name(conn->result, i, &col_name, &col_name_len) != CASS_OK) {
498  col_name = "<INVALID>";
499  }
500  names[i] = col_name;
501  }
502 
503  *out = names;
504 
505  return RLM_SQL_OK;
506 }

521  if ((_ret = cass_result_column_name(conn->result, i, &_col_name, &_col_name_len)) != CASS_OK) {\
522  _col_name = "<INVALID>";\
523  }\
524  sql_set_last_error_printf(conn, "Failed to retrieve " _t " data at column %s (%d): %s", \
525  _col_name, i, cass_error_desc(_ret));\
526  TALLOC_FREE(handle->row);\
527  return RLM_SQL_ERROR;\
528 } while(0)
529 
530  if (!conn->result) return RLM_SQL_OK; /* no result */
531 
532  *out = NULL;
533 
534  /*
535  * Start of the result set, initialise the iterator.
536  */
537  if (!conn->iterator) conn->iterator = cass_iterator_from_result(conn->result);
538  if (!conn->iterator) return RLM_SQL_OK; /* no result */
539 
540  if (!cass_iterator_next(conn->iterator)) return RLM_SQL_NO_MORE_ROWS; /* no more rows */
541 
542  cass_row = cass_iterator_get_row(conn->iterator); /* this shouldn't fail ? */
543  fields = sql_num_fields(handle, config); /* get the number of fields... */
544 
545  /*
546  * Free the previous result (also gets called on finish_query)
547  */
548  talloc_free(handle->row);
549  MEM(row = handle->row = talloc_zero_array(handle, char *, fields + 1));
550 
551  for (i = 0; i < fields; i++) {
552  CassValue const *value;
553  CassValueType type;
554 
555  value = cass_row_get_column(cass_row, i);
556 
557  if (cass_value_is_null(value) == cass_true) continue;
558 
559  type = cass_value_type(value);
560  switch (type) {
561  case CASS_VALUE_TYPE_ASCII:
562  case CASS_VALUE_TYPE_TEXT:
563  case CASS_VALUE_TYPE_VARCHAR:
564  {
565  const char *str;
566  size_t len;
567 
568  if (cass_value_get_string(value, &str, &len) != CASS_OK) RLM_CASS_ERR_DATA_RETRIVE("string");
569 
570  MEM(row[i] = talloc_array(row, char, len + 1));
571  memcpy(row[i], str, len);
572  row[i][len] = '\0';
573  }
574  break;
575 
576  case CASS_VALUE_TYPE_BOOLEAN:
577  {
578  cass_bool_t bv;
579 
580  if (cass_value_get_bool(value, &bv) != CASS_OK) RLM_CASS_ERR_DATA_RETRIVE("bool");
581 
582  MEM(row[i] = talloc_zero_array(row, char, 2));
583  row[i][0] = (bv == cass_false) ? '0' : '1';
584  }
585  break;
586 
587  case CASS_VALUE_TYPE_INT:
588  {
589  cass_int32_t i32v;
590 
591  if (cass_value_get_int32(value, &i32v) != CASS_OK) RLM_CASS_ERR_DATA_RETRIVE("int32");
592 
593  MEM(row[i] = talloc_typed_asprintf(row, "%"PRId32, (int32_t)i32v));
594  }
595  break;
596 
597  case CASS_VALUE_TYPE_TIMESTAMP:
598  case CASS_VALUE_TYPE_BIGINT:
599  {
600  cass_int64_t i64v;
601 
602  if (cass_value_get_int64(value, &i64v) != CASS_OK) RLM_CASS_ERR_DATA_RETRIVE("int64");
603 
604  MEM(row[i] = talloc_typed_asprintf(row, "%"PRId64, (int64_t)i64v));
605  }
606  break;
607 
608  case CASS_VALUE_TYPE_UUID:
609  case CASS_VALUE_TYPE_TIMEUUID:
610  {
611  CassUuid uuid;
612 
613  if (cass_value_get_uuid(value, &uuid) != CASS_OK) RLM_CASS_ERR_DATA_RETRIVE("UUID");
614  MEM(row[i] = talloc_array(row, char, CASS_UUID_STRING_LENGTH));
615  cass_uuid_string(uuid, row[i]);
616  }
617  break;
618 
619  default:
620  {
621  const char *col_name;
622  size_t col_name_len;
623 
624  if (cass_result_column_name(conn->result, i, &col_name,
625  &col_name_len) != CASS_OK) col_name = "<INVALID>";
626 
627  sql_set_last_error_printf(conn,
628  "Failed to retrieve data at column %s (%d): Unsupported data type",
629  col_name, i);
630  talloc_free(handle->row);
631  return RLM_SQL_ERROR;
632  }
633  }
634  }
635  *out = row;
636 
637  return RLM_SQL_OK;
638 }
