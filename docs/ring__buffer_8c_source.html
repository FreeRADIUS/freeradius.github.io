48 /** Create a ring buffer.
49  *
50  * The size provided will be rounded up to the next highest power of
51  * 2, if it's not already a power of 2.
52  *
53  * The ring buffer manages how much room is reserved (i.e. available
54  * to write to), and used. The application is responsible for
55  * tracking the start of the reservation, *and* it's write offset
56  * within that reservation.
57  *
58  * @param[in] ctx a talloc context
59  * @param[in] size of the raw ring buffer array to allocate.
60  * @return
61  * - A new ring buffer on success.
62  * - NULL on failure.
63  */
64 fr_ring_buffer_t *fr_ring_buffer_create(TALLOC_CTX *ctx, size_t size)
65 {
66  fr_ring_buffer_t *rb;
67 
68  rb = talloc_zero(ctx, fr_ring_buffer_t);
69  if (!rb) {
70  fail:
71  fr_strerror_printf("Failed allocating memory.");
72  return NULL;
73  }
74 
75  if (size < 1024) size = 1024;
76 
77  if (size > (1 << 30)) {
78  fr_strerror_printf("Ring buffer size must be no more than (1 << 30)");
79  talloc_free(rb);
80  return NULL;
81  }
82 
83  /*
84  * Round up to the nearest power of 2.
85  */
86  size--;
87  size |= size >> 1;
88  size |= size >> 2;
89  size |= size >> 4;
90  size |= size >> 8;
91  size |= size >> 16;
92  size++;
93 
94  rb->buffer = talloc_array(rb, uint8_t, size);
95  if (!rb->buffer) {
96  talloc_free(rb);
97  goto fail;
98  }
99  rb->size = size;
100 
101  return rb;
102 }

105 /** Reserve room in the ring buffer.
106  *
107  * The size does not need to be a power of two. The application is
108  * responsible for doing cache alignment, if required.
109  *
110  * If the reservation fails, the application should create a new ring
111  * buffer, and start reserving data there.
112  *
113  * @param[in] rb a ring buffer
114  * @param[in] size to see if we can reserve
115  * @return
116  * - NULL on error. Which can only be "ring buffer is full".
117  * - pointer to data on success
118  */
119 uint8_t *fr_ring_buffer_reserve(fr_ring_buffer_t *rb, size_t size)
120 {
121  (void) talloc_get_type_abort(rb, fr_ring_buffer_t);
122 
123  if (rb->closed) {
124  fr_strerror_printf("Allocation request after ring buffer is closed");
125  return NULL;
126  }
127 
128  /*
129  * We're writing to the start of the buffer, and there is
130  * already data in it. See if the data fits.
131  *
132  * |***W....S****E....|
133  */
134  if (rb->write_offset < rb->data_start) {
135  if ((rb->write_offset + size) < rb->data_start) {
136  rb->reserved = size;
137  return rb->buffer + rb->write_offset;
138  }
139 
140  fr_strerror_printf("No memory available in ring buffer");
141  return NULL;
142  }
143 
144  fr_assert(rb->write_offset == rb->data_end);
145 
146  /*
147  * Data fits at the end of the ring buffer.
148  *
149  * |....S****WE....|
150  */
151  if ((rb->write_offset + size) <= rb->size) {
152  rb->reserved = size;
153  return rb->buffer + rb->write_offset;
154  }
155 
156  /*
157  * Data fits at the start of the ring buffer, ensure that
158  * we write it there. This also catches the case where
159  * data_start==0.
160  *
161  * |W....S****E....|
162  */
163  if (size < rb->data_start) {
164  rb->write_offset = 0;
165  rb->reserved = size;
166  return rb->buffer;
167  }
168 
169  /*
170  * Not enough room for the new data, fail the allocation.
171  *
172  * |....S****WE....|
173  */
174  fr_strerror_printf("No memory available in ring buffer");
175  return NULL;
176 }

179 /** Mark data as allocated.
180  *
181  * The size does not need to be a power of two. The application is
182  * responsible for doing cache-line alignment, if required.
183  *
184  * The application does NOT need to call fr_ring_buffer_reserve() before
185  * calling this function.
186  *
187  * If the allocation fails, the application should create a new ring
188  * buffer, and start allocating data there.
189  *
190  * @param[in] rb a ring buffer
191  * @param[in] size to mark as "used" at the tail end of the buffer.
192  * @return
193  * - NULL on error. Which can only be "ring buffer is full".
194  * - pointer to data on success
195  */
196 uint8_t *fr_ring_buffer_alloc(fr_ring_buffer_t *rb, size_t size)
197 {
198  uint8_t *p;
199 
200  (void) talloc_get_type_abort(rb, fr_ring_buffer_t);
201 
202  if (rb->closed) {
203 #ifndef NDEBUG
204  fr_strerror_printf("Allocation request after ring buffer is closed");
205 #endif
206  return NULL;
207  }
208 
209  /*
210  * Shrink the "reserved" portion of the buffer by the
211  * allocated size.
212  */
213  if (rb->reserved >= size) {
214  rb->reserved -= size;
215  } else {
216  rb->reserved = 0;
217  }
218 
219  /*
220  * We're writing to the start of the buffer, and there is
221  * already data in it. See if the data fits.
222  *
223  * |***W....S****E....|
224  */
225  if (rb->write_offset < rb->data_start) {
226  if ((rb->write_offset + size) < rb->data_start) {
227  p = rb->buffer + rb->write_offset;
228  rb->write_offset += size;
229  return p;
230  }
231 
232 #ifndef NDEBUG
233  fr_strerror_printf("No memory available in ring buffer");
234 #endif
235  return NULL;
236  }
237 
238  fr_assert(rb->write_offset == rb->data_end);
239 
240  /*
241  * Data fits at the end of the ring buffer.
242  *
243  * |....S****WE....|
244  */
245  if ((rb->write_offset + size) <= rb->size) {
246  p = rb->buffer + rb->write_offset;
247 
248  rb->write_offset += size;
249  rb->data_end = rb->write_offset;
250 
251  /*
252  * Don't update write_offset if it's fallen off
253  * of the end of the buffer. The data_start may
254  * be zero, and we don't want to over-write
255  * that.
256  */
257  return p;
258  }
259 
260  /*
261  * Data fits at the start of the ring buffer, ensure that
262  * we write it there. This also catches the case where
263  * data_start==0.
264  *
265  * |W....S****E....|
266  */
267  if (size < rb->data_start) {
268  rb->write_offset = size;
269 
270  /*
271  * Don't update data_end. It points to the tail
272  * end of the ring buffer.
273  */
274  return rb->buffer;
275  }
276 
277  /*
278  * Not enough room for the new data, fail the allocation.
279  *
280  * |....S****WE....|
281  */
282 #ifndef NDEBUG
283  fr_strerror_printf("No memory available in ring buffer");
284 #endif
285  return NULL;
286 }

289 /** Split an existing reservation into two.
290  *
291  * For protocols like TCP, there may sometimes be a partial packet at
292  * the end of the ring buffer. We would like to pass a *complete*
293  * packet around instead of a partial one. In that case, the partial
294  * packet at the end of the buffer should be copied to a reservation
295  * in a new ring buffer.
296  *
297  * i.e. the application uses fr_ring_buffer_reserve() to reserve 32K
298  * of room. He then reads 32K of data into that buffer. This data
299  * comprises 3 full packets of 10K, and one partial packet of 10K.
300  * The application then calls fr_ring_buffer_alloc() three times, to
301  * consume those packets. (Note that the caller doesn't really need
302  * to do 3 calls to fr_ring_buffer_alloc(). The ring buffer does not
303  * keep track of individual allocations).
304  *
305  * The application then calls fr_ring_buffer_reserve() to reserve
306  * another 32K of room, while leaving 2K of data in the ring buffer.
307  * If that reservation succeeds, great. Everything proceeds as
308  * before. (Note that the application has to remember how much data
309  * was in the ring buffer, and do it's reading there, instead of to
310  * the pointer returned from fr_ring_buffer_reserve()).
311  *
312  * If that call fails, there is 2K of partial data in the buffer
313  * which needs to be moved. The application should allocate a new
314  * ring buffer, and then call this function to move the data to the
315  * new ring buffer. The application then uses the new reservation to
316  * read data.
317  *
318  * @param[in] dst ring buffer where the reservation will be made
319  * @param[in] reserve_size size of the new reservation
320  * @param[in] src ring buffer where the data is sitting.
321  * @param[in] move_size of data to move from the tail of the buffer to the start.
322  * @return
323  * - NULL on error.
324  * - pointer to data on success
325  */
326 uint8_t *fr_ring_buffer_reserve_split(fr_ring_buffer_t *dst, size_t reserve_size,
327  fr_ring_buffer_t *src, size_t move_size)
328 {
329  uint8_t *p;
330 
331  (void) talloc_get_type_abort(src, fr_ring_buffer_t);
332  (void) talloc_get_type_abort(dst, fr_ring_buffer_t);
333 
334  if (dst->closed) {
335  fr_strerror_printf("Allocation request after ring buffer is closed");
336  return NULL;
337  }
338 
339  /*
340  * The application hasn't reserved enough space, so we can't
341  * split the reservation.
342  */
343  if (src->reserved < move_size) {
344  fr_strerror_printf("Cannot move more data than was reserved.");
345  return NULL;
346  }
347 
348  /*
349  * Create a new reservation.
350  */
351  p = fr_ring_buffer_reserve(dst, reserve_size);
352  if (!p) return NULL;
353 
354  /*
355  * Alloc and reserve in the same ring buffer. Maybe
356  * there's no need to memcpy() the data?
357  */
358  if ((src == dst) && (p == (src->buffer + src->write_offset))) {
359  return 0;
360  }
361 
362  /*
363  * Copy the data from the old buffer to the new one.
364  */
365  memcpy(p, src->buffer + src->write_offset, move_size);
366 
367  /*
368  * We now have no data reserved here. All bets are
369  * off...
370  */
371  src->reserved = 0;
372 
373  return p;
374 }

377 /** Mark data as free,
378  *
379  * The size does not need to be a power of two. The application is
380  * responsible for doing cache alignment, if required. The
381  * application is responsible for tracking sizes of packets in the
382  * ring buffer.
383  *
384  * If "unused" bytes are more than what's in the buffer, the used
385  * bytes are reset to zero.
386  *
387  * @param[in] rb a ring buffer
388  * @param[in] size_to_free bytes to mark as "unused" in the buffer.
389  * @return
390  * - <0 on error. Which can only be "ring buffer is full".
391  * - 0 on success
392  */
393 int fr_ring_buffer_free(fr_ring_buffer_t *rb, size_t size_to_free)
394 {
395  size_t block_size;
396 
397  (void) talloc_get_type_abort(rb, fr_ring_buffer_t);
398 
399  /*
400  * Nothing to free, do nothing.
401  */
402  if (!size_to_free) return 0;
403 
404  /*
405  * Freeing data from the middle of the buffer.
406  *
407  * |***W....S****E....|
408  */
409  if (rb->write_offset < rb->data_start) {
410  block_size = rb->data_end - rb->data_start;
411 
412  /*
413  * |***W....S****E....|, free 3
414  *
415  * |***W.......S*E....|
416  */
417  if (size_to_free < block_size) {
418  rb->data_start += size_to_free;
419  return 0;
420  }
421 
422  /*
423  * Free all (or more than) the block.
424  */
425  rb->data_start = 0;
426  rb->data_end = rb->write_offset;
427  size_to_free -= block_size;
428 
429  /*
430  * Free everything left: empty the buffer
431  * entirely. This also handles the case of
432  * size_to_free==0 and write_offset==0.
433  */
434  if (size_to_free == rb->write_offset) {
435  goto empty_buffer;
436  }
437 
438  /*
439  * The buffer has data but we're not freeing
440  * any more of it, return.
441  */
442  if (!size_to_free) return 0;
443  }
444 
445  fr_assert(rb->write_offset == rb->data_end);
446 
447  block_size = rb->data_end - rb->data_start;
448 
449  /*
450  * Freeing too much, return an error.
451  */
452  if (size_to_free > block_size) {
453  fr_strerror_printf("Cannot free more memory than exists.");
454  return -1;
455  }
456 
457  /*
458  * Free some data from the start.
459  */
460  if (size_to_free < block_size) {
461  rb->data_start += size_to_free;
462  return 0;
463  }
464 
465  /*
466  * Free all data in the buffer.
467  */
468 empty_buffer:
469  rb->data_start = 0;
470  rb->data_end = 0;
471  rb->write_offset = 0;
472 
473  /*
474  * If the ring buffer is closed to all allocations, and
475  * it's now empty, we automatically free it.
476  */
477  if (rb->closed) talloc_free(rb);
478 
479  return 0;
480 }
