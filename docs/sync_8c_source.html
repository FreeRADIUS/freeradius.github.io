85 /** Process a cookie element
86  *
87  * @param[out] new_cookie Whether we got a new cookie value.
88  * @param[in] sync message was associated with.
89  * @param[in] ber value possibly containing a cookie tag (will be advanced).
90  * @return
91  * - 0 success, a cookie was parsed successfully.
92  * - -1 parse error.
93  */
94 static int sync_new_cookie(bool *new_cookie, sync_state_t *sync, BerElement *ber)
95 {
96  struct berval cookie;
97  size_t cookie_len;
98  ber_tag_t bv_ret;
99  ber_len_t len;
100 
101  if (new_cookie) *new_cookie = false;
102 
103  /*
104  * Look for the (optional) cookie.
105  */
106  if (ber_peek_tag(ber, &len) != LDAP_TAG_SYNC_COOKIE) return 0;
107 
108  bv_ret = ber_scanf(ber, "m", &cookie);
109  if (bv_ret == LBER_ERROR) {
110  ERROR("Malformed cookie tag");
111  return -1;
112  }
113 
114  /*
115  * "no cookie" can mean either no cookie element,
116  * or a NULL cookie element (as per the RFC).
117  */
118  if (!cookie.bv_val) return 0;
119 
120  if (sync->cookie) {
121  if (talloc_array_length(sync->cookie) == cookie.bv_len) {
122  cookie_len = talloc_array_length(sync->cookie);
123  if (memcmp(sync->cookie, cookie.bv_val, cookie.bv_len)) {
124  WARN("Ignoring new cookie \"%pV\": Identical to old cookie",
125  fr_box_strvalue_len((char const *)sync->cookie, cookie_len));
126  return 0;
127  }
128  }
129  }
130 
131  talloc_free(sync->cookie);
132  sync->cookie = fr_ldap_berval_to_bin(sync, &cookie);
133  cookie_len = talloc_array_length(sync->cookie);
134  DEBUG3("Got new cookie value \"%pV\" (%zu)",
135  fr_box_strvalue_len((char const *)sync->cookie, cookie_len), cookie_len);
136 
137  if (new_cookie) *new_cookie = true;
138 
139  return 0;
140 }

142 /** Handle a LDAP_RES_SEARCH_ENTRY (SearchResultEntry) or LDAP_RES_SEARCH_REFRENCE (SearchResultReference) response
143  *
144  * Upon receipt of a search request containing the syncControl the server provides the initial
145  * content using zero or more SearchResultEntries followed by a SearchResultdone.
146  *
147  * Each SearchResultEntry includes a Sync State control with state set to add, an entryUUID
148  * containing the entry's UUID, and no cookie.
149  *
150  * For refreshAndPersist operations SearchResultEntries are also used after the refresh phase
151  * to inform clients of changes to entries within the scope of the search request.
152  *
153  * The Sync State Control is an LDAP Control where the controlType is the object identifier
154  * 1.3.6.1.4.1.4203.1.9.1.2 and the controlValue, an OCTET STRING.
155  * It contains a BER-encoded syncStateValue.
156  *
157  * syncStateValue ::= SEQUENCE {
158  * state ENUMERATED {
159  * present (0),
160  * add (1),
161  * modify (2),
162  * delete (3)
163  * },
164  * entryUUID syncUUID,
165  * cookie syncCookie OPTIONAL
166  * }
167  *
168  * The Sync State Control is only included in SearchResultEntry and SearchResultReference Messages.
169  *
170  * @param[in] sync message was associated with.
171  * @param[in] msg containing an entry to process.
172  * @param[in] ctrls associated with the msg.
173  * @return
174  * - 0 on success.
175  * - -1 on failure.
176  */
177 static int sync_search_entry_or_reference(sync_state_t *sync, LDAPMessage *msg, LDAPControl **ctrls)
178 {
179  int ret = 0, i;
180  ber_tag_t bv_ret;
181  BerElement *ber = NULL;
182  struct berval entry_uuid = { 0 };
183  sync_states_t state = SYNC_STATE_INVALID;
184  bool new_cookie;
185 
186  fr_assert(sync->conn);
187  fr_assert(sync);
188  fr_assert(msg);
189 
190  if (!ctrls) {
191  missing_control:
192  ERROR("searchResEntry missing syncStateValue control");
193  return -1;
194  }
195 
196  /*
197  * Every SearchResultEntry/Reference must contain a Sync State Control
198  * describing the state of an object/the changes that should
199  * be made to it.
200  *
201  * If the entry doesn't have this control then the LDAP server
202  * is broken.
203  */
204  for (i = 0; ctrls[i] != NULL; i++) {
205  if (strcmp(ctrls[i]->ldctl_oid, LDAP_CONTROL_SYNC_STATE) == 0) break;
206  }
207  if (!ctrls[i]) goto missing_control;
208 
209  /*
210  * Get the value of the control.
211  */
212  ber = ber_init(&ctrls[i]->ldctl_value);
213  if (!ber) {
214  ERROR("Failed allocating ber to handle syncStateValue control");
215 
216  error:
217  if (ber) ber_free(ber, 1);
218 
219  return -1;
220  }
221 
222  bv_ret = ber_scanf(ber, "{em", &state, &entry_uuid);
223  if ((bv_ret == LBER_ERROR) || (entry_uuid.bv_len == 0)) {
224  ERROR("Malformed syncUUID value");
225  goto error;
226  }
227 
228  if (sync_new_cookie(&new_cookie, sync, ber) < 0) goto error;
229 
230  if (ber_scanf(ber, "}") == LBER_ERROR ) {
231  ERROR("Malformed syncStatevalue sequence");
232  goto error;
233  }
234 
235  switch (state) {
236  case SYNC_STATE_PRESENT:
237  switch (sync->phase) {
238 /*
239  case SYNC_PHASE_INIT:
240  sync->phase = SYNC_PHASE_PRESENT;
241  break;
242 */
243 
244  case SYNC_PHASE_PRESENT:
245  break;
246 
247  default:
248  bad_phase:
249  ERROR("Entries with %s state are not allowed during refresh %s phase",
250  fr_table_str_by_value(sync_state_table, state, "<unknown>"),
251  fr_table_str_by_value(sync_phase_table, sync->phase, "<unknown>"));
252  goto error;
253  }
254  break;
255 
256  case SYNC_STATE_DELETE:
257  switch (sync->phase) {
258  case SYNC_PHASE_DELETE:
259  case SYNC_PHASE_DONE:
260  break;
261 
262  default:
263  goto bad_phase;
264  }
265  break;
266 
267  case SYNC_STATE_ADD:
268  case SYNC_STATE_MODIFY:
269  switch (sync->phase) {
270  case SYNC_PHASE_INIT:
271  case SYNC_PHASE_DONE:
272  break;
273 
274  default:
275  goto bad_phase;
276  }
277  break;
278 
279  default:
280  ERROR("Unknown entry state (%i)", state);
281  goto error;
282  }
283 
284  if (DEBUG_ENABLED3) {
285  char *entry_dn;
286  fr_value_box_t uuid_box;
287 
288  entry_dn = ldap_get_dn(sync->conn->handle, msg);
289  fr_ldap_berval_to_value_shallow(&uuid_box, &entry_uuid);
290 
291  DEBUG3("Processing %s (%s), dn \"%s\", entryUUID %pV",
292  fr_table_str_by_value(sync_protocol_op_table, ldap_msgtype(msg), "<unknown>"),
293  fr_table_str_by_value(sync_state_table, state, "<unknown>"),
294  entry_dn ? entry_dn : "<unknown>",
295  &uuid_box);
296 
297  ldap_memfree(entry_dn);
298  }
299 
300  /*
301  * Call the entry callback to notify caller that the object has changed.
302  */
303  if (ldap_msgtype(msg) == LDAP_RES_SEARCH_REFERENCE) {
304  if (sync->config->reference) {
305  ret = sync->config->reference(sync->conn, sync->config, sync->msgid, sync->phase,
306  (uint8_t const *)entry_uuid.bv_val,
307  msg, state, sync->config->user_ctx);
308  }
309  } else {
310  if (sync->config->entry) {
311  ret = sync->config->entry(sync->conn, sync->config, sync->msgid, sync->phase,
312  (uint8_t const *)entry_uuid.bv_val,
313  msg, state, sync->config->user_ctx);
314  }
315  }
316 
317  if ((ret == 0) && new_cookie && sync->config->cookie) {
318  ret = sync->config->cookie(sync->conn, sync->config, sync->msgid, sync->cookie, sync->config->user_ctx);
319  }
320  ber_free(ber, 1);
321 
322  return ret;
323 }

325 /** Handle a LDAP_RES_INTERMEDIATE (SyncInfo) response
326  *
327  * These allow the LDAP server to communicate sync state to clients
328  *
329  * The Sync Info Message is an LDAP Intermediate Response Message [RFC4511] where
330  * responseName is the object identifier 1.3.6.1.4.1.4203.1.9.1.4 and responseValue
331  * contains a BER-encoded syncInfoValue.
332  *
333  * syncInfoValue ::= CHOICE {
334  * newcookie [0] syncCookie,
335  * refreshDelete [1] SEQUENCE {
336  * cookie syncCookie OPTIONAL,
337  * refreshDone BOOLEAN DEFAULT TRUE
338  * },
339  * refreshPresent [2] SEQUENCE {
340  * cookie syncCookie OPTIONAL,
341  * refreshDone BOOLEAN DEFAULT TRUE
342  * },
343  * syncIdSet [3] SEQUENCE {
344  * cookie syncCookie OPTIONAL,
345  * refreshDeletes BOOLEAN DEFAULT FALSE,
346  * syncUUIDs SET OF syncUUID
347  * }
348  * }
349  *
350  * @param[in] sync message was associated with.
351  * @param[in] msg containing an entry to process.
352  * @param[in] ctrls associated with the msg.
353  * @return
354  * - 0 on success.
355  * - -1 on failure.
356  */
357 static int sync_intermediate(sync_state_t *sync, LDAPMessage *msg, UNUSED LDAPControl **ctrls)
358 {
359  int ret, i;
360  char *oid = NULL;
361  struct berval *data = NULL;
362  BerElement *ber = NULL;
363  ber_len_t len;
364  ber_tag_t sync_info_tag;
365  int refresh_deletes = 0;
366  BerVarray sync_uuids = NULL;
367 
368  bool new_cookie;
369  int refresh_done = false;
370 
371  ret = ldap_parse_intermediate(sync->conn->handle, msg, &oid, &data, NULL, 0);
372  if (!fr_cond_assert(ret == LDAP_SUCCESS)) return -1; /* should have been caught earlier */
373 
374  if (!oid || (strcmp(oid, LDAP_SYNC_INFO) != 0)) {
375  WARN("Ignoring intermediateResult with unexpected OID \"%s\"", oid ? oid : "<unknown>");
376  return 0;
377  }
378 
379  ber = ber_init(data);
380  if (ber == NULL) {
381  ERROR("Failed allocating ber to handle syncInfo data");
382 
383  return -1;
384  }
385 
386  sync_info_tag = ber_peek_tag(ber, &len);
387  DEBUG3("Processing syncInfo (%s)", fr_table_str_by_value(sync_info_tag_table, sync_info_tag, "<unknown>"));
388 
389  switch (sync_info_tag) {
390  case LDAP_TAG_SYNC_NEW_COOKIE:
391  if (sync_new_cookie(&new_cookie, sync, ber) < 0) {
392  error:
393  if (sync_uuids) ber_bvarray_free(sync_uuids);
394  if (ber) ber_free(ber, 1);
395  if (data) ber_bvfree(data);
396  if (oid) ldap_memfree(oid);
397 
398  return -1;
399  }
400 
401  if (!new_cookie) {
402  ERROR("Missing cookie value");
403  goto error;
404  }
405  break;
406 
407  case LDAP_TAG_SYNC_REFRESH_PRESENT:
408  switch (sync->phase) {
409  case SYNC_PHASE_INIT:
410  sync->phase = SYNC_PHASE_PRESENT;
411  break;
412 
413  default:
414  ERROR("Invalid refresh phase transition (%s->%s)",
415  fr_table_str_by_value(sync_phase_table, sync->phase, "<unknown>"),
416  fr_table_str_by_value(sync_phase_table, SYNC_PHASE_DELETE, "<unknown>"));
417  goto error;
418  }
419 
420  if (ber_scanf(ber, "{") == LBER_ERROR) {
421  ERROR("Malformed refreshPresent sequence");
422  goto error;
423  }
424 
425  if (sync_new_cookie(&new_cookie, sync, ber) < 0) goto error;
426 
427  if (ber_peek_tag(ber, &len) == LDAP_TAG_REFRESHDONE) {
428  if (ber_scanf(ber, "b", &refresh_done) == LBER_ERROR) {
429  ERROR("Malformed refresh sequence: Missing refreshDone tag value");
430  goto error;
431  }
432  } else {
433  refresh_done = true; /* Default (when absent) is true */
434  }
435 
436  if (ber_scanf(ber, "}") == LBER_ERROR ) {
437  ERROR("Malformed refreshPresent sequence");
438  goto error;
439  }
440 
441  /*
442  * The refreshPresent contains refreshDone, which is always FALSE in the
443  * refreshOnly mode of Sync Operation because it is followed by a delete
444  * phase.
445  */
446  if (!sync->config->persist) {
447  ERROR("Got refreshPresent refreshDone = true in refreshOnly mode (which is invalid)");
448  goto error;
449  }
450 
451  if (refresh_done) sync->phase = SYNC_PHASE_DONE;
452  break;
453 
454  case LDAP_TAG_SYNC_REFRESH_DELETE:
455  switch (sync->phase) {
456  case SYNC_PHASE_INIT:
457  case SYNC_PHASE_PRESENT:
458  case SYNC_PHASE_PRESENT_IDSET:
459  sync->phase = SYNC_PHASE_DELETE;
460  break;
461 
462  default:
463  ERROR("Invalid refresh phase transition (%s->%s)",
464  fr_table_str_by_value(sync_phase_table, sync->phase, "<unknown>"),
465  fr_table_str_by_value(sync_phase_table, SYNC_PHASE_DELETE, "<unknown>"));
466 
467  goto error;
468  }
469 
470  if (ber_scanf(ber, "{") == LBER_ERROR) {
471  ERROR("Malformed refreshPresent sequence");
472  goto error;
473  }
474 
475  if (sync_new_cookie(&new_cookie, sync, ber) < 0) goto error;
476 
477  if (ber_peek_tag(ber, &len) == LDAP_TAG_REFRESHDONE) {
478  if (ber_scanf(ber, "b", &refresh_done) == LBER_ERROR) {
479  ERROR("Malformed refresh sequence: Missing refreshDone tag value");
480  goto error;
481  }
482  } else {
483  refresh_done = true;
484  }
485 
486  if (ber_scanf(ber, "}") == LBER_ERROR ) {
487  ERROR("Malformed refreshPresent sequence");
488  goto error;
489  }
490  if (refresh_done) sync->phase = SYNC_PHASE_DONE;
491  break;
492 
493  case LDAP_TAG_SYNC_ID_SET:
494  if (ber_scanf( ber, "{") == LBER_ERROR) {
495  ERROR("Malformed syncIDSet");
496  goto error;
497  }
498 
499  if (sync_new_cookie(&new_cookie, sync, ber) < 0) goto error;
500 
501  if (ber_peek_tag(ber, &len) == LDAP_TAG_REFRESHDELETES) {
502  if (ber_scanf(ber, "b", &refresh_deletes) == LBER_ERROR) {
503  ERROR("Malformed refresh_deletes tag");
504  goto error;
505  }
506  }
507 
508  if ((ber_scanf(ber, "[W]}", &sync_uuids) == LBER_ERROR) || !sync_uuids) {
509  ERROR("syncIDSet missing set of IDs");
510  goto error;
511  }
512 
513  if (refresh_deletes) {
514  switch (sync->phase) {
515  case SYNC_PHASE_INIT:
516  case SYNC_PHASE_PRESENT:
517  case SYNC_PHASE_PRESENT_IDSET:
518  sync->phase = SYNC_PHASE_DELETE_IDSET;
519  break;
520 
521  default:
522  ERROR("Invalid refresh phase transition (%s->%s)",
523  fr_table_str_by_value(sync_phase_table, sync->phase, "<unknown>"),
524  fr_table_str_by_value(sync_phase_table, SYNC_PHASE_DELETE, "<unknown>"));
525  goto error;
526  }
527  } else {
528  switch (sync->phase) {
529  case SYNC_PHASE_INIT:
530  sync->phase = SYNC_PHASE_PRESENT_IDSET;
531  break;
532 
533  default:
534  ERROR("Invalid refresh phase transition (%s->%s)",
535  fr_table_str_by_value(sync_phase_table, sync->phase, "<unknown>"),
536  fr_table_str_by_value(sync_phase_table, SYNC_PHASE_DELETE, "<unknown>"));
537  goto error;
538  }
539  }
540 
541  for (i = 0; sync_uuids[i].bv_val != NULL; i++) {
542  if (sync_uuids[i].bv_len != SYNC_UUID_LENGTH) {
543  ERROR("Invalid entryUUID length, expected " STRINGIFY(SYNC_UUID_LENGTH) " "
544  "bytes got %zu bytes", sync_uuids[i].bv_len);
545  goto error;
546  }
547 
548  ret = sync->config->entry(sync->conn, sync->config, sync->msgid, sync->phase,
549  (uint8_t const *)sync_uuids[i].bv_val, NULL, SYNC_STATE_DELETE,
550  sync->config->user_ctx);
551  if (!ret) goto error;
552  }
553 
554  ber_bvarray_free(sync_uuids);
555  sync_uuids = NULL;
556  break;
557 
558  default:
559  ERROR("Invalid syncInfo tag %lu", sync_info_tag);
560  goto error;
561 
562  }
563 
564  if (new_cookie && sync->config->cookie) {
565  ret = sync->config->cookie(sync->conn, sync->config, sync->msgid, sync->cookie, sync->config->user_ctx);
566  }
567 
568  if (ber) ber_free(ber, 1);
569  if (oid) ldap_memfree(oid);
570  if (data) ber_bvfree(data);
571 
572  return ret;
573 }

575 /** Handle a LDAP_RES_SEARCH_RESULT (searchResultDone)
576  *
577  * The Sync Done Control is an LDAP Control where the controlType is the object identifier
578  * 1.3.6.1.4.1.4203.1.9.1.3 and the controlValue contains a BER-encoded syncDoneValue.
579  *
580  * syncDoneValue ::= SEQUENCE {
581  * cookie syncCookie OPTIONAL,
582  * refreshDeletes BOOLEAN DEFAULT FALSE
583  * }
584  *
585  * The Sync Done Control is only applicable to the SearchResultDone Message.
586  *
587  * It's essentially used to make the end of the set of searchEntries, and indicate the
588  * possible start of a present of delete phase.
589  *
590  * It's only used for refreshOnly operations to indicate that the server is done returning
591  * search results, with refreshAndPersist an intermediateResult message is used instead.
592  *
593  * @param[in] sync message was associated with.
594  * @param[in] msg containing an entry to process.
595  * @param[in] ctrls associated with the msg.
596  * @return
597  * - 0 on success.
598  * - -1 on failure.
599  */
600 static int sync_search_result(sync_state_t *sync, LDAPMessage *msg, LDAPControl **ctrls)
601 {
602  int ret = 0;
603  int refresh_deletes = 0;
604 
605  int i;
606  BerElement *ber = NULL;
607  ber_len_t len;
608  bool new_cookie;
609 
610  fr_assert(sync->conn);
611  fr_assert(sync);
612  fr_assert(msg);
613 
614  /*
615  * Should not happen with refreshAndPersist
616  */
617  if (sync->config->persist) {
618  ERROR("searchResult is invalid for refreshAndPersist mode");
619  return -1;
620  }
621 
622  /*
623  * Process the syncDoneValue
624  */
625  if (!ctrls) {
626  missing_control:
627  ERROR("searchResult missing syncDoneValue control");
628  return -1;
629  }
630  for (i = 0; ctrls[i] != NULL; i++) {
631  if (strcmp(ctrls[i]->ldctl_oid, LDAP_CONTROL_SYNC_DONE) == 0) break;
632  }
633  if (!ctrls[i]) goto missing_control;
634 
635  ber = ber_init(&ctrls[i]->ldctl_value);
636  if (!ber) {
637  ERROR("Failed allocating ber to handle syncDoneValue control");
638 
639  error:
640  if (ber) ber_free(ber, 1);
641 
642  return -1;
643  }
644 
645  if (ber_scanf( ber, "{" /*"}"*/) == LBER_ERROR) goto error;
646 
647  if (sync_new_cookie(&new_cookie, sync, ber) < 0) goto error;
648 
649  if (ber_peek_tag(ber, &len) == LDAP_TAG_REFRESHDELETES) {
650  if (ber_scanf(ber, "b", &refresh_deletes) == LBER_ERROR) {
651  ERROR("Malformed refresh sequence: Missing refreshDeletes tag value");
652  goto error;
653  }
654  } else {
655  refresh_deletes = 0;
656  }
657 
658  if (ber_scanf( ber, /*"{"*/ "}" ) == LBER_ERROR) {
659  ERROR("Malformed syncDoneValue sequence");
660  goto error;
661  }
662 
663  ber_free(ber, 1);
664

709 /** Function to call when the LDAP handle's FD is readable
710  *
711  * @param[out] sync_id the last sync_id serviced.
712  * @param[in] conn to service.
713  * @return
714  * - 0 on success.
715  * - -1 on sync error.
716  * - -2 on conn error. Requires the handle to be destroyed.
717  */
718 int sync_demux(int *sync_id, fr_ldap_connection_t *conn)
719 {
720  struct timeval poll = { 1, 0 }; /* Poll */
721  LDAPMessage *msg, *head = NULL;
722  int ret = 0;
723  fr_ldap_rcode_t rcode;
724  sync_state_t find = { .msgid = -1 }, *sync = NULL;
725  rbtree_t *tree;
726 
727  tree = talloc_get_type_abort(conn->uctx, rbtree_t);
728 
729  fr_assert(conn);
730 
731  /*
732  * Drain any messages outstanding on this connection
733  */
734  ret = ldap_result(conn->handle, LDAP_RES_ANY, LDAP_MSG_RECEIVED, &poll, &head);
735  switch (ret) {
736  case 0: /* timeout - shouldn't happen */
737  fr_assert(0);
738  return -2;
739 
740  case -1:
741  rcode = fr_ldap_error_check(NULL, conn, NULL, NULL);
742  if (rcode == LDAP_PROC_BAD_CONN) return -2;
743  return -1;
744 
745  default:
746  break;
747  }
748 
749  /*
750  * De-multiplex based on msgid
751  */
752  for (msg = ldap_first_message(conn->handle, head);
753  msg;
754  msg = ldap_next_message(conn->handle, msg)) {
755  int type;
756  int msgid;
757  LDAPControl **ctrls;
758 
759  *sync_id = msgid = ldap_msgid(msg);
760  type = ldap_msgtype(msg);
761 
762  if (msgid == 0) {
763  WARN("Ignoring unsolicited %s message",
764  fr_table_str_by_value(sync_protocol_op_table, type, "<invalid>"));
765  continue;
766  }
767 
768  /*
769  * Only search if we're receiving messages from
770  * a different sync.
771  */
772  if (!sync || (sync->msgid != msgid)) {
773  find.msgid = msgid;
774 
775  sync = rbtree_finddata(tree, &find);
776  if (!sync) {
777  WARN("Ignoring msgid %i, doesn't match any outstanding syncs",
778  find.msgid);
779  continue;
780  }
781  }
782 
783  /*
784  * Check for errors contained within the message
785  * This has to be per message, as multiple syncs
786  * are multiplexed together on one connection.
787  */
788  switch (fr_ldap_error_check(&ctrls, conn, msg, sync->config->base_dn)) {
789  case LDAP_PROC_SUCCESS:
790  break;
791 
792  case LDAP_PROC_REFRESH_REQUIRED:
793  if (!sync->config->refresh_required) return -1;
794 
795  DEBUG2("LDAP Server returned e-syncRefreshRequired");
796  return sync->config->refresh_required(conn, sync->config, sync->msgid,
797  sync->phase, sync->config->user_ctx);
798 
799  /*
800  * Don't think this should happen... but libldap
801  * is wonky sometimes...
802  */
803  case LDAP_PROC_BAD_CONN:
804  if (ctrls) ldap_controls_free(ctrls);
805  PERROR("Connection unusable");
806  return -2;
807 
808  default:
809  sync_error:
810  if (ctrls) ldap_controls_free(ctrls);
811  PERROR("Sync error");
812  return -1;
813  }
814 
815  DEBUG3("Got %s message for sync (msgid %i)",
816  fr_table_str_by_value(sync_protocol_op_table, type, "<invalid>"), sync->msgid);
817 
818  switch (type) {
819  case LDAP_RES_SEARCH_REFERENCE:
820  case LDAP_RES_SEARCH_ENTRY:
821  ret = sync_search_entry_or_reference(sync, msg, ctrls);
822  if (ret < 0) goto sync_error;
823  break;
824 
825  case LDAP_RES_SEARCH_RESULT:
826  ret = sync_search_result(sync, msg, ctrls);
827  if (ret < 0) goto sync_error;
828  break;
829 
830  case LDAP_RES_INTERMEDIATE:
831  ret = sync_intermediate(sync, msg, ctrls);
832  if (ret < 0) goto sync_error;
833  break;
834 
835  default:
836  WARN("Ignoring unexpected message type (%i)", type);
837  break;
838  }
839 
840  ldap_controls_free(ctrls);
841  }
842 
843  ldap_msgfree(head);
844 
845  return 0;
846 }

849  *
850  * Terminates the search informing the remote server that we no longer want to receive results
851  * for this sync. A RFC 4511 abandon request is used to inform the server.
852  *
853  * This allows individual syncs to be stopped without destroying the underlying connection.
854  *
855  * Removes the sync's msgid from the tree of msgids associated with the connection.
856  *
857  * @param[in] sync to abandon.
858  * @return 0
859  */
860 static int _sync_state_free(sync_state_t *sync)
861 {
862 
863  fr_ldap_connection_t *conn = talloc_get_type_abort(sync->conn, fr_ldap_connection_t); /* check for premature free */
864  rbtree_t *tree = talloc_get_type_abort(conn->uctx, rbtree_t);
865  sync_state_t find = { .msgid = sync->msgid };
866 
867  fr_assert(sync->conn->handle);
868 
869  DEBUG3("Abandoning sync");
870 
871  if (!sync->conn->handle) return 0; /* Handled already closed? */
872 
873  /*
874  * Tell the remote server to stop sending results
875  */
876  ldap_abandon_ext(sync->conn->handle, sync->msgid, NULL, NULL);
877  rbtree_deletebydata(tree, &find);
878 
879  return 0;
880 }

943 /** Allocate a sync state structure and issue the search
944  *
945  * The Sync Request Control is an LDAP Control [RFC4511] where the controlType is the object
946  * identifier 1.3.6.1.4.1.4203.1.9.1.1 and the controlValue, an OCTET STRING, contains a
947  * BER-encoded syncRequestValue. The criticality field is either TRUE or FALSE.
948  *
949  * syncRequestValue ::= SEQUENCE {
950  * mode ENUMERATED {
951  * -- 0 unused
952  * refreshOnly (1),
953  * -- 2 reserved
954  * refreshAndPersist (3)
955  * },
956  * cookie syncCookie OPTIONAL,
957  * reloadHint BOOLEAN DEFAULT FALSE
958  * }
959  *
960  * The Sync Request Control is only applicable to the SearchRequest Message.
961  *
962  * The sync structure is parented off the conn. When the sync is no longer needed, or
963  * an error has occurred, it should be freed with talloc_free(), which will result in
964  * an ldap_abandon message to the server to tell it to cancel the search.
965  *
966  * @param[in] conn Connection to issue the search request on.
967  * @param[in] config containing callbacks and search parameters.
968  * @param[in] cookie NULL to perform a complete refresh, else the last
969  * cookie provided
970  * @param[in] reload_hint If true, hint to the server that we need to be sent all
971  * entries in the directory.
972  */
973 int sync_state_init(fr_ldap_connection_t *conn, sync_config_t const *config,
974  uint8_t const *cookie, bool reload_hint)
975 {

977 
978  LDAPControl ctrl = {0}, *ctrls[2] = { &ctrl, NULL };
979  BerElement *ber = NULL;
980  int ret;
981  int mode;
982  sync_state_t *sync;
983  rbtree_t *tree;
984 
985  fr_assert(conn);
986  fr_assert(config);
987 
988  mode = config->persist ? LDAP_SYNC_REFRESH_AND_PERSIST : LDAP_SYNC_REFRESH_ONLY;
989 
990  /*
991  * Allocate or retrieve the tree of outstanding msgids
992  * these are specific to the connection.
993  */
994  if (!conn->uctx) {
995  MEM(tree = rbtree_talloc_alloc(conn, _sync_cmp, sync_state_t, NULL, RBTREE_FLAG_NONE));
996  conn->uctx = tree;
997  } else {
998  tree = talloc_get_type_abort(conn->uctx, rbtree_t);
999  }
1000 
1001  /*
1002  * Allocate the sync request control
1003  */
1004  ber = ber_alloc_t(LBER_USE_DER);
1005  if (!ber) {
1006  ERROR("Failed allocating sync control");
1007  return -1;
1008  }
1009 
1010  /*
1011  * If the connection is freed, all the sync state is also freed
1012  */
1013  MEM(sync = talloc_zero(tree, sync_state_t));
1014  sync->config = config;
1015  sync->phase = SYNC_PHASE_INIT;
1016  sync->conn = conn;
1017  talloc_set_destructor(sync, _sync_state_free);
1018 
1019  /*
1020  * Might not necessarily have a cookie
1021  */
1022  if (cookie) {
1023  char *bv_val;
1024  struct berval bvc;
1025 
1026  memcpy(&bv_val, &cookie, sizeof(bv_val));
1027 
1028  bvc.bv_val = bv_val;
1029  bvc.bv_len = talloc_array_length(cookie);
1030 
1031  ber_printf(ber, "{eOb}",mode, &bvc, reload_hint);
1032  } else {
1033  ber_printf(ber, "{eb}", mode, reload_hint );
1034  }
1035 
1036  ret = ber_flatten2(ber, &ctrl.ldctl_value, 0);
1037  if (ret < 0) {
1038  ERROR("Failed creating sync control");
1039 
1040  ber_free(ber, 1);
1041  talloc_free(sync);
1042 
1043  return -1;
1044  }
1045 
1046  /*
1047  * Mark the control as critical
1048  */
1049  memcpy(&ctrl.ldctl_oid, &sync_ctl_oid, sizeof(ctrl.ldctl_oid));
1050  ctrl.ldctl_iscritical = 1;
1051 
1052  ret = fr_ldap_search_async(&sync->msgid, NULL,
1053  &conn,
1054  config->base_dn, config->scope, config->filter, config->attrs,
1055  ctrls, NULL);
1056  ber_free(ber, 1);
1057  if (ret != LDAP_PROC_SUCCESS) {
1058  talloc_free(sync);
1059  return -1;
1060  }
1061 
1062  if (!rbtree_insert(tree, sync)) {
1063  ERROR("Duplicate sync (msgid %i)", sync->msgid);
1064  return -1;
1065  }
1066  DEBUG3("Sync created with msgid %i", sync->msgid);
1067 
1068  return 0;
