63 /** Set module status or rcode
64  *
65  * Look ma, no locks...
66  *
67  * Example: "%{db_status:fail}"
68  */
69 static xlat_action_t always_xlat(TALLOC_CTX *ctx, fr_cursor_t *out,
70  request_t *request, void const *xlat_inst,
71  UNUSED void *xlat_thread_inst,
72  fr_value_box_t **in)
73 {
74  rlm_always_t const *inst = talloc_get_type_abort_const(*((void const * const *)xlat_inst),rlm_always_t);
75  module_instance_t *mi;
76  char const *status;
77  char const *p;
78  fr_value_box_t *vb;
79 
80  mi = module_by_name(NULL, inst->xlat_name);
81  if (!mi) {
82  RERROR("Can't find the module that registered this xlat: %s", inst->xlat_name);
83  return XLAT_ACTION_FAIL;
84  }
85 
86  /*
87  * Expand to the existing status
88  */
89  p = "alive";
90  if (mi->force) {
91  p = fr_table_str_by_value(rcode_table, mi->code, "<invalid>");
92  }
93 
94  if (!(*in) || (*in)->vb_length == 0) goto done;
95  status = (*in)->vb_strvalue;
96 
97  /*
98  * Set the module status
99  */
100  if (strcmp(status, "alive") == 0) {
101  mi->force = false;
102  } else {
103  int rcode;
104 
105  rcode = fr_table_value_by_str(rcode_table, status, RLM_MODULE_UNKNOWN);
106  if (rcode == RLM_MODULE_UNKNOWN) {
107  RWARN("Unknown status \"%s\"", status);
108  return XLAT_ACTION_FAIL;
109  }
110 
111  mi->code = rcode;
112  mi->force = true;
113 
114  }
115 
116 done:
117 
118  MEM(vb = fr_value_box_alloc_null(ctx));
119  fr_value_box_strdup(vb, vb, NULL, p, false);
120  fr_cursor_append(out, vb);
121 
122  return XLAT_ACTION_DONE;
123 
124 }

126 static int mod_bootstrap(void *instance, CONF_SECTION *conf)
127 {
128  rlm_always_t *inst = instance;
129  xlat_t const *xlat;
130 
131  inst->xlat_name = cf_section_name2(conf);
132  if (!inst->xlat_name) {
133  inst->xlat_name = cf_section_name1(conf);
134  }
135 
136  xlat = xlat_register(inst, inst->xlat_name, always_xlat, false);
137  xlat_async_instantiate_set(xlat, always_xlat_instantiate, rlm_always_t *, NULL, inst);
138 
139  return 0;
140 }

142 static int mod_instantiate(void *instance, CONF_SECTION *conf)
143 {
144  rlm_always_t *inst = instance;
145 
146  /*
147  * Convert the rcode string to an int
148  */
149  inst->rcode = fr_table_value_by_str(rcode_table, inst->rcode_str, RLM_MODULE_UNKNOWN);
150  if (inst->rcode == RLM_MODULE_UNKNOWN) {
151  cf_log_err(conf, "rcode value \"%s\" is invalid", inst->rcode_str);
152  return -1;
153  }
154 
155  return 0;
156 }
