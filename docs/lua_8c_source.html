45 /** Convert fr_pair_ts to Lua values
46  *
47  * Pushes a Lua representation of an attribute value onto the stack.
48  *
49  * @param[in] request The current request.
50  * @param[in] L Lua interpreter.
51  * @param[in] vp to convert.
52  * @return
53  * - 0 on success.
54  * - -1 on failure.
55  */
56 static int fr_lua_marshall(request_t *request, lua_State *L, fr_pair_t const *vp)
57 {
58  if (!vp) return -1;
59 
60 #define IN_RANGE_INTEGER_SIGNED(_x) \
61  do { \
62  if ((((int64_t)(_x)) < PTRDIFF_MIN) || (((int64_t)(_x)) > PTRDIFF_MAX)) { \
63  REDEBUG("Value (%" PRId64 ") cannot be represented as Lua integer. Must be between %td-%td", \
64  (int64_t)(_x), (ptrdiff_t)PTRDIFF_MIN, (ptrdiff_t)PTRDIFF_MAX); \
65  return -1; \
66  } \
67  } while (0)
68 
69 #define IN_RANGE_INTEGER_UNSIGNED(_x) \
70  do { \
71  if (((uint64_t)(_x)) > PTRDIFF_MAX) { \
72  REDEBUG("Value (%" PRIu64 ") cannot be represented as Lua integer. Must be between 0-%td", \
73  (uint64_t)(_x), (ptrdiff_t)PTRDIFF_MAX); \
74  return -1; \
75  } \
76  } while (0)
77 
78 #define IN_RANGE_FLOAT_SIGNED(_x) \
79  do { \
80  if ((((double)(_x)) < DBL_MIN) || (((double)(_x)) > DBL_MAX)) { \
81  REDEBUG("Value (%f) cannot be represented as Lua number. Must be between %f-%f", \
82  (double)(_x), DBL_MIN, DBL_MAX); \
83  return -1; \
84  } \
85  } while (0)
86 
87  switch (vp->vp_type) {
88  case FR_TYPE_ETHERNET:
89  case FR_TYPE_IPV4_ADDR:
90  case FR_TYPE_IPV6_ADDR:
91  case FR_TYPE_IPV4_PREFIX:
92  case FR_TYPE_IPV6_PREFIX:
93  case FR_TYPE_IFID:
94  case FR_TYPE_TIME_DELTA:
95  {
96  char buff[128];
97  ssize_t slen;
98 
99  slen = fr_pair_print_value_quoted(&FR_SBUFF_OUT(buff, sizeof(buff)), vp, T_BARE_WORD);
100  if (slen < 0) {
101  REDEBUG("Cannot convert %s to Lua type, insufficient buffer space",
102  fr_table_str_by_value(fr_value_box_type_table, vp->vp_type, "<INVALID>"));
103  return -1;
104  }
105 
106  lua_pushlstring(L, buff, (size_t)slen);
107  }
108  break;
109 
110  case FR_TYPE_STRING:
111  lua_pushlstring(L, vp->vp_strvalue, vp->vp_length);
112  break;
113 
114  case FR_TYPE_OCTETS:
115  lua_pushlstring(L, (char const *)vp->vp_octets, vp->vp_length); /* lstring variant is embedded NULL safe */
116  break;
117 
118  case FR_TYPE_BOOL:
119  lua_pushinteger(L, (lua_Integer)(vp->vp_bool ? 1 : 0));
120  break;
121 
122  case FR_TYPE_UINT8:
123  lua_pushinteger(L, (lua_Integer)vp->vp_uint8);
124  break;
125 
126  case FR_TYPE_UINT16:
127  lua_pushinteger(L, (lua_Integer)vp->vp_uint16);
128  break;
129 
130  case FR_TYPE_UINT32:
131  lua_pushinteger(L, (lua_Integer)vp->vp_uint32);
132  break;
133 
134  case FR_TYPE_UINT64:
135  IN_RANGE_INTEGER_UNSIGNED(vp->vp_uint64);
136  lua_pushinteger(L, (lua_Integer)vp->vp_uint64);
137  break;
138 
139  case FR_TYPE_INT8:
140  lua_pushinteger(L, (lua_Integer)vp->vp_int8);
141  break;
142 
143  case FR_TYPE_INT16:
144  lua_pushinteger(L, (lua_Integer)vp->vp_int16);
145  break;
146 
147  case FR_TYPE_INT32:
148  lua_pushinteger(L, (lua_Integer)vp->vp_int32);
149  break;
150 
151  case FR_TYPE_INT64:
152  IN_RANGE_INTEGER_SIGNED(vp->vp_int64);
153  lua_pushinteger(L, (lua_Integer)vp->vp_int64);
154  break;
155 
156  case FR_TYPE_DATE:
157  lua_pushinteger(L, (lua_Integer) fr_time_to_sec(vp->vp_date));
158  break;
159 
160  case FR_TYPE_FLOAT32:
161  IN_RANGE_FLOAT_SIGNED(vp->vp_float32);
162  lua_pushnumber(L, (lua_Number)vp->vp_float32);
163  break;
164 
165  case FR_TYPE_FLOAT64:
166  IN_RANGE_FLOAT_SIGNED(vp->vp_float64);
167  lua_pushnumber(L, (lua_Number)vp->vp_float64);
168  break;
169 
170  case FR_TYPE_SIZE:
171  IN_RANGE_INTEGER_UNSIGNED(vp->vp_size);
172  lua_pushinteger(L, (lua_Integer)vp->vp_size);
173  break;
174 
175  case FR_TYPE_NON_VALUES:
176  REDEBUG("Cannot convert %s to Lua type", fr_table_str_by_value(fr_value_box_type_table, vp->vp_type, "<INVALID>"));
177  return -1;
178  }
179  return 0;
180 }

183 /** Convert Lua values to fr_pair_ts
184  *
185  * Convert Lua values back to fr_pair_ts. How the Lua value is converted is dependent
186  * on the type of the DA.
187  *
188  * @param[out] out Where to write a pointer to the new fr_pair_t.
189  * @param[in] inst the current instance.
190  * @param[in] request the current request.
191  * @param[in] L Lua interpreter.
192  * @param[in] da specifying the type of attribute to create.
193  * @return
194  * - 0 on success.
195  * - -1 on failure.
196  */
197 static int fr_lua_unmarshall(fr_pair_t **out,
198  UNUSED rlm_lua_t const *inst, request_t *request, lua_State *L, fr_dict_attr_t const *da)
199 {
200  fr_pair_t *vp;
201 
202  MEM(vp = fr_pair_afrom_da(request, da));
203  switch (lua_type(L, -1)) {
204  case LUA_TNUMBER:
205  {
206  fr_value_box_t vb;
207 
208  /*
209  * lua_tonumber actually returns ptrdiff_t
210  * so we need to check if our input box
211  * type is the same width or greater.
212  */
213  static_assert(SIZEOF_MEMBER(fr_value_box_t, vb_int64) >= sizeof(ptrdiff_t),
214  "fr_value_box_t field smaller than return from lua_tointeger");
215 
216  static_assert(SIZEOF_MEMBER(fr_value_box_t, vb_float64) >= sizeof(double),
217  "fr_value_box_t field smaller than return from lua_tonumber");
218 
219  switch (vp->da->type) {
220  /*
221  * Preserve decimal precision.
222  *
223  * Our FR_TYPE_FLOAT64 is a double, which is apparently
224  * what lua_tonumber returns on most platforms.
225  */
226  case FR_TYPE_FLOAT32:
227  case FR_TYPE_FLOAT64:
228  fr_value_box_init(&vb, FR_TYPE_FLOAT64, NULL, true);
229  vb.vb_float64 = lua_tonumber(L, -1);
230  break;
231 
232  default:
233  fr_value_box_init(&vb, FR_TYPE_INT64, NULL, true);
234  vb.vb_int64 = lua_tointeger(L, -1);
235  break;
236  }
237 
238 
239  if (fr_value_box_cast(vp, &vp->data, vp->da->type, vp->da, &vb) < 0) {
240  RPEDEBUG("Failed unmarshalling Lua number for \"%s\"", vp->da->name);
241  return -1;
242  }
243  }
244  break;
245 
246  case LUA_TSTRING:
247  {
248  fr_value_box_t vb;
249  char const *p;
250  size_t len;
251 
252  p = (char const *)lua_tolstring(L, -1, &len);
253  if (!p) {
254  REDEBUG("Unmarshalling failed, Lua bstring was NULL");
255  return -1;
256  }
257 
258  fr_value_box_bstrndup_shallow(&vb, NULL, p, len, true);
259 
260  if (fr_value_box_cast(vp, &vp->data, vp->da->type, vp->da, &vb) < 0) {
261  RPEDEBUG("Failed unmarshalling Lua string for \"%s\"", vp->da->name);
262  return -1;
263  }
264  }
265  break;
266 
267  case LUA_TLIGHTUSERDATA:
268  case LUA_TUSERDATA:
269  {
270  size_t len;
271  uint8_t *p;
272 
273  len = lua_objlen(L, -1);
274  if (len == 0) {
275  REDEBUG("Unmarshalling failed, can't determine length of user data");
276  return -1;
277  }
278  p = lua_touserdata(L, -1);
279  if (!p) {
280  REDEBUG("Unmarshalling failed, user data was NULL");
281  }
282  fr_pair_value_memdup(vp, p, len, true);
283  }
284  break;
285 
286  default:
287  {
288  int type = lua_type(L, -1);
289  REDEBUG("Unmarshalling failed, unsupported Lua type %s (%i)", lua_typename(L, type), type);
290 
291  return -1;
292  }
293  }
294 
295  *out = vp;
296  return 0;
297 }

299 /** Get an instance of an attribute
300  *
301  * @note Should only be present in the Lua environment as a closure.
302  * @note Takes one upvalue - the fr_dict_attr_t to search for as light user data.
303  * @note Is called as an __index metamethod, so takes the table (can be ignored)
304  * and the field (an integer index value)
305  *
306  * @param[in] L Lua interpreter.
307  * @return
308  * - 0 (no results) on success.
309  * - 1 on success with the fr_pair_t value on the stack.
310  */
311 static int _lua_pair_get(lua_State *L)
312 {
313  request_t *request = fr_lua_util_get_request();
314 
315  fr_cursor_t cursor;
316  fr_dict_attr_t const *da;
317  fr_pair_t *vp = NULL;
318  int index;
319 
320  fr_assert(lua_islightuserdata(L, lua_upvalueindex(1)));
321 
322  da = lua_touserdata(L, lua_upvalueindex(1));
323  fr_assert(da);
324 
325  /*
326  * @fixme Packet list should be light user data too at some point
327  */
328  fr_cursor_iter_by_da_init(&cursor, &request->request_pairs, da);
329 
330  for (index = (int) lua_tointeger(L, -1); index >= 0; index--) {
331  vp = fr_cursor_next(&cursor);
332  if (!vp) return 0;
333  }
334 
335  if (fr_lua_marshall(request, L, vp) < 0) return -1;
336 
337  return 1;
338 }

340 /** Set an instance of an attribute
341  *
342  * @note Should only be present in the Lua environment as a closure.
343  * @note Takes one upvalue - the fr_dict_attr_t to search for as light user data.
344  * @note Is called as an __newindex metamethod, so takes the table (can be ignored),
345  * the field (an integer index value) and the new value.
346  *
347  * @param[in] L Lua interpreter.
348  * @return
349  * - 0 on success.
350  * - -1 on failure.
351  */
352 static int _lua_pair_set(lua_State *L)
353 {
354  rlm_lua_t const *inst = fr_lua_util_get_inst();
355  request_t *request = fr_lua_util_get_request();
356  fr_cursor_t cursor;
357  fr_dict_attr_t const *da;
358  fr_pair_t *vp = NULL, *new;
359  lua_Integer index;
360  bool delete = false;
361 
362 
363  /*
364  * This function should only be called as a closure.
365  * As we control the upvalues, we should assert on errors.
366  */
367  fr_assert(lua_islightuserdata(L, lua_upvalueindex(1)));
368 
369  da = lua_touserdata(L, lua_upvalueindex(1));
370  fr_assert(da);
371 
372  delete = lua_isnil(L, -1);
373 
374  /*
375  * @fixme Packet list should be light user data too at some point
376  */
377  fr_cursor_iter_by_da_init(&cursor, &request->request_pairs, da);
378 
379  for (index = lua_tointeger(L, -2); index >= 0; index--) {
380  vp = fr_cursor_next(&cursor);
381  if (vp) break;
382  }
383 
384  /*
385  * If the value of the Lua stack was nil, we delete the
386  * attribute the cursor is currently positioned at.
387  */
388  if (delete) {
389  fr_cursor_remove(&cursor);
390  return 0;
391  }
392 
393  if (fr_lua_unmarshall(&new, inst, request, L, da) < 0) return -1;
394 
395  /*
396  * If there was already a VP at that index we replace it
397  * else we add a new VP to the list.
398  */
399  if (vp) {
400  fr_cursor_replace(&cursor, new);
401  } else {
402  fr_cursor_append(&cursor, new);
403  }
404 
405  return 0;
406 }

515 /** Initialise and return a new accessor table
516  *
517  *
518  */
519 static int _lua_pair_accessor_init(lua_State *L)
520 {
521  request_t *request = fr_lua_util_get_request();
522  char const *attr;
523  fr_dict_attr_t const *da;
524  fr_dict_attr_t *up;
525 
526  attr = lua_tostring(L, -1);
527  if (!attr) {
528  REDEBUG("Failed retrieving field name \"%s\"", attr);
529  return -1;
530  }
531 
532  da = fr_dict_attr_by_name(request->dict, attr);
533  if (!da) {
534  REDEBUG("Unknown or invalid attribute name \"%s\"", attr);
535  return -1;
536  }
537  memcpy(&up, &da, sizeof(up));
538 
539  /*
540  * Add the pairs method to the main table, this allows
541  * easy iteration over multiple values of the same
542  * attribute.
543  *
544  * for v in request[User-Name].pairs() do
545  */
546  lua_newtable(L);
547  lua_pushlightuserdata(L, request->request_pairs);
548  lua_pushlightuserdata(L, up);
549  lua_pushcclosure(L, _lua_pair_iterator_init, 2);
550  lua_setfield(L, -2, "pairs");
551 
552  /*
553  * Metatable methods for getting and setting
554  */
555  lua_newtable(L);
556  lua_pushlightuserdata(L, up);
557  lua_pushcclosure(L, _lua_pair_get, 1);
558  lua_setfield(L, -2, "__index");
559 
560  lua_pushlightuserdata(L, up);
561  lua_pushcclosure(L, _lua_pair_set, 1);
562  lua_setfield(L, -2, "__newindex");
563 
564  lua_setmetatable(L, -2);
565  lua_settable(L, -3); /* Cache the attribute manipulation object */
566  lua_getfield(L, -1, attr); /* and return it */
567 
568  return 1;
569 }

654 /** Resolve a path string to a field value in Lua
655  *
656  * Parses a string in the format
657  * @verbatim obj0[.obj1][.objN] @endverbatim, adding all tables it traverses
658  * to the stack.
659  *
660  * All paths are assumed to start at a global, so the first field
661  * will be looked up in the global table.
662  *
663  */
664 static int fr_lua_get_field(lua_State *L, request_t *request, char const *field)
665 {
666  char buff[512];
667  char const *p = field, *q;
668 
669  q = strchr(p, '.');
670  if (!q) { /* No field, just global */
671  lua_getglobal(L, p);
672  if (lua_isnil(L, -1)) {
673  does_not_exist:
674  REMARKER(field, p - field, "Field does not exist");
675  return -1;
676  }
677  return 0;
678  }
679 
680  if ((size_t) (q - p) >= sizeof(buff)) {
681  too_long:
682  REDEBUG("Field name too long, expected < %zu, got %zu", q - p, sizeof(buff));
683  return -1;
684  }
685 
686  strlcpy(buff, p, (q - p) + 1);
687  lua_getglobal(L, buff);

693

695  lua_getfield(L, -1, buff);
696  if (lua_isnil(L, -1)) goto does_not_exist;
697  p = q + 1;
698  }
699 
700  lua_getfield(L, -1, p);
701  if (lua_isnil(L, -1)) goto does_not_exist;
702 
703  return 0;

739 unlang_action_t fr_lua_run(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request, char const *funcname)
740 {
741  rlm_lua_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_lua_t);
742  rlm_lua_thread_t *thread = talloc_get_type_abort(mctx->thread, rlm_lua_thread_t);
743  lua_State *L = thread->interpreter;
744  rlm_rcode_t rcode = RLM_MODULE_OK;
745 
746  fr_lua_util_set_inst(inst);
747  fr_lua_util_set_request(request);
748 
749  ROPTIONAL(RDEBUG2, DEBUG2, "Calling %s() in interpreter %p", funcname, L);
750 
751  _lua_fr_request_register(L, request);
752 
753  /*
754  * Get the function were going to be calling
755  */
756  if (fr_lua_get_field(L, request, funcname) < 0) {
757 error:
758  fr_lua_util_set_inst(NULL);
759  fr_lua_util_set_request(NULL);
760 
761  RETURN_MODULE_FAIL;
762  }
763 
764  if (!lua_isfunction(L, -1)) {
765  int type = lua_type(L, -1);
766 
767  ROPTIONAL(RDEBUG2, DEBUG2, "'%s' is not a function, is a %s (%i)", funcname, lua_typename(L, type), type);
768  goto error;
769  }
770 
771  if (lua_pcall(L, 0, 1, 0) != 0) {
772  char const *msg = lua_tostring(L, -1);
773 
774  ROPTIONAL(RDEBUG2, DEBUG2, "Call to %s failed: %s", funcname, msg ? msg : "unknown error");
775  goto error;
776  }
777 
778  /*
779  * functions without rcodeurn or rcodeurning none/nil will be RLM_MODULE_OK
780  */
781  if (!lua_isnoneornil(L, -1)) {
782  /*
783  * e.g: rcodeurn 2, rcodeurn "2", rcodeurn fr.handled, fr.fail, ...
784  */
785  if (lua_isnumber(L, -1)) {
786  rcode = lua_tointeger(L, -1);
787  if (fr_table_str_by_value(rcode_table, rcode, NULL) != NULL) goto done;
788  }
789 
790  /*
791  * e.g: rcodeurn "handled", "ok", "fail", ...
792  */
793  if (lua_isstring(L, -1)) {
794  rcode = fr_table_value_by_str(rcode_table, lua_tostring(L, -1), -1);
795  if ((int)rcode != -1) goto done;
796  }
797 
798  ROPTIONAL(RDEBUG2, DEBUG2, "Lua function %s() rcodeurned invalid rcode \"%s\"", funcname, lua_tostring(L, -1));
799  goto error;
800  }
801 
802 done:
803  fr_lua_util_set_inst(NULL);
804  fr_lua_util_set_request(NULL);
805 
806  RETURN_MODULE_RCODE(rcode);
807 }

871 /** Initialise a new Lua/LuaJIT interpreter
872  *
873  * Creates a new lua_State and verifies all required functions have been loaded correctly.
874  *
875  * @param[in] out Where to write a pointer to the new state.
876  * @param[in] instance Current instance of fr_lua, a talloc marker
877  * context will be inserted into the context of instance
878  * to ensure the interpreter is freed when instance data is freed.
879  * @return 0 on success else -1.
880  */
881 int fr_lua_init(lua_State **out, rlm_lua_t const *instance)
882 {
883  rlm_lua_t const *inst = talloc_get_type_abort_const(instance, rlm_lua_t);
884  lua_State *L;
885 
886  fr_lua_util_set_inst(inst);
887 
888  L = luaL_newstate();
889  if (!L) {
890  ERROR("Failed initialising Lua state");
891  return -1;
892  }
893 
894  luaL_openlibs(L);
895 
896  /*
897  * Load the Lua file into our environment.
898  */
899  if (luaL_loadfile(L, inst->module) != 0) {
900  ERROR("Failed loading file: %s", lua_gettop(L) ? lua_tostring(L, -1) : "Unknown error");
901 
902  error:
903  *out = NULL;
904  fr_lua_util_set_inst(NULL);
905  lua_close(L);
906  return -1;
907  }
908 
909  if (lua_pcall(L, 0, LUA_MULTRET, 0) != 0) {
910  ERROR("Failed executing script: %s", lua_gettop(L) ? lua_tostring(L, -1) : "Unknown error");
911 
912  goto error;
913  }
914 
915  /*
916  * Setup "fr.{}"
917  */
918  fr_lua_util_fr_register(L);
919 
920  /*
921  * Setup "fr.log.{}"
922  */
923  if (inst->jit) {
924  DEBUG4("Initialised new LuaJIT interpreter %p", L);
925  if (fr_lua_util_jit_log_register(inst, L) < 0) goto error;
926  } else {
927  DEBUG4("Initialised new Lua interpreter %p", L);
928  if (fr_lua_util_log_register(inst, L) < 0) goto error;
929  }
930 
931  /*
932  * Setup the "fr.rcode.{}" with all RLM_MODULE_*
933  * e.g: "fr.rcode.reject", "fr.rcode.ok", ...
934  */
935  fr_lua_rcode_register(L, "rcode");
936 
937  /*
938  * Verify all the functions were provided.
939  */
940  if (fr_lua_check_func(inst, L, inst->func_authorize)
941  || fr_lua_check_func(inst, L, inst->func_authenticate)
942  || fr_lua_check_func(inst, L, inst->func_preacct)
943  || fr_lua_check_func(inst, L, inst->func_accounting)
944  || fr_lua_check_func(inst, L, inst->func_post_auth)
945  || fr_lua_check_func(inst, L, inst->func_instantiate)
946  || fr_lua_check_func(inst, L, inst->func_detach)
947  || fr_lua_check_func(inst, L, inst->func_xlat)) {
948  goto error;
949  }
950 
951  *out = L;
952  return 0;
953 }
