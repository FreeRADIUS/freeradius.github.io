127 /** Read big-endian 2-byte unsigned from p
128  *
129  * caller must ensure enough data exists at "p"
130  */
131 uint16_t soh_pull_be_16(uint8_t const *p) {
132  uint16_t r;
133 
134  r = *p++ << 8;
135  r += *p++;
136 
137  return r;
138 }

145  uint32_t r;
146 
147  r = *p++ << 16;

171 /** Parses the MS-SOH type/value (note: NOT type/length/value) data and update the sohvp list
172  *
173  * See section 2.2.4 of MS-SOH. Because there's no "length" field we CANNOT just skip
174  * unknown types; we need to know their length ahead of time. Therefore, we abort
175  * if we find an unknown type. Note that sohvp may still have been modified in the
176  * failure case.
177  *
178  * @param request Current request
179  * @param p binary blob
180  * @param data_len length of blob
181  * @return
182  * - 0 on success.
183  * - -1 on failure.
184  */
185 static int eap_peap_soh_mstlv(request_t *request, uint8_t const *p, unsigned int data_len)
186 {
187  fr_pair_t *vp;
188  uint8_t c;
189  int t;
190 
191  while (data_len > 0) {
192  c = *p++;
193  data_len--;
194 
195  switch (c) {
196  /*
197  * MS-Machine-Inventory-Packet
198  * MS-SOH section 2.2.4.1
199  */
200  case 1:
201  if (data_len < 18) {
202  RDEBUG2("insufficient data for MS-Machine-Inventory-Packet");
203  return 0;
204  }
205  data_len -= 18;
206 
207  MEM(pair_update_request(&vp, attr_soh_ms_machine_os_vendor) >= 0);
208  fr_pair_value_from_str(vp, "Microsoft", -1, '\0', false);
209 
210  MEM(pair_update_request(&vp, attr_soh_ms_machine_os_version) >= 0);
211  vp->vp_uint32 = soh_pull_be_32(p);
212  p += 4;
213 
214  MEM(pair_update_request(&vp, attr_soh_ms_machine_os_release) >= 0);
215  vp->vp_uint32 = soh_pull_be_32(p);
216  p += 4;
217 
218  MEM(pair_update_request(&vp, attr_soh_ms_machine_os_build) >= 0);
219  vp->vp_uint32 = soh_pull_be_32(p);
220  p += 4;
221 
222  MEM(pair_update_request(&vp, attr_soh_ms_machine_sp_version) >= 0);
223  vp->vp_uint32 = soh_pull_be_16(p);
224  p += 2;
225 
226  MEM(pair_update_request(&vp, attr_soh_ms_machine_sp_release) >= 0);
227  vp->vp_uint32 = soh_pull_be_16(p);
228  p += 2;
229 
230  MEM(pair_update_request(&vp, attr_soh_ms_machine_processor) >= 0);
231  vp->vp_uint32 = soh_pull_be_16(p);
232  p += 2;
233  break;
234  /*
235  * MS-Quarantine-State - FIXME: currently unhandled
236  * MS-SOH 2.2.4.1
237  *
238  * 1 byte reserved
239  * 1 byte flags
240  * 8 bytes NT Time field (100-nanosec since 1 Jan 1601)
241  * 2 byte urilen
242  * N bytes uri
243  */
244  case 2:
245 
246  p += 10;
247  t = soh_pull_be_16(p); /* t == uri len */
248  p += 2;
249  p += t;
250  data_len -= 12 + t;
251  break;
252 
253  /*
254  * MS-Packet-Info
255  * MS-SOH 2.2.4.3
256  */
257  case 3:
258 
259  RDEBUG3("SoH MS-Packet-Info %s vers=%i", (*p & 0x10) ? "request" : "response", *p & 0xf);
260  p++;
261  data_len--;
262  break;
263 
264  /*
265  * MS-SystemGenerated-Ids - FIXME: currently unhandled
266  * MS-SOH 2.2.4.4
267  *
268  * 2 byte length
269  * N bytes (3 bytes IANA enterprise# + 1 byte component id#)
270  */
271  case 4:
272 
273  t = soh_pull_be_16(p);
274  p += 2;
275  p += t;
276  data_len -= 2 + t;
277  break;
278 
279  /*
280  * MS-MachineName
281  * MS-SOH 2.2.4.5
282  *
283  * 1 byte namelen
284  * N bytes name
285  */
286  case 5:
287 
288  t = soh_pull_be_16(p);
289  p += 2;
290 
291  MEM(pair_update_request(&vp, attr_soh_ms_machine_name) >= 0);
292  fr_pair_value_bstrndup(vp, (char const *)p, t, true);
293 
294  p += t;
295  data_len -= 2 + t;
296  break;
297 
298  /*
299  * MS-CorrelationId
300  * MS-SOH 2.2.4.6
301  *
302  * 24 bytes opaque binary which we might, in future, have
303  * to echo back to the client in a final SoHR
304  */
305  case 6:
306  MEM(pair_update_request(&vp, attr_soh_ms_correlation_id) >= 0);
307  fr_pair_value_memdup(vp, p, 24, true);
308  p += 24;
309  data_len -= 24;
310  break;
311 
312  /*
313  * MS-Installed-Shvs - FIXME: currently unhandled
314  * MS-SOH 2.2.4.7
315  *
316  * 2 bytes length
317  * N bytes (3 bytes IANA enterprise# + 1 byte component id#)
318  */
319  case 7:
320  t = soh_pull_be_16(p);
321  p += 2;
322  p += t;
323  data_len -= 2 + t;
324  break;
325 
326  /*
327  * MS-Machine-Inventory-Ex
328  * MS-SOH 2.2.4.8
329  *
330  * 4 bytes reserved
331  * 1 byte product type (client=1 domain_controller=2 server=3)
332  */
333  case 8:
334  p += 4;
335  MEM(pair_update_request(&vp, attr_soh_ms_machine_role));
336  vp->vp_uint32 = *p;
337  p++;
338  data_len -= 5;
339  break;
340 
341  default:
342  RDEBUG2("SoH Unknown MS TV %i stopping", c);
343  return 0;
344  }
345  }
346  return 1;
347 }

412 /** Parse the MS-SOH response in data and update sohvp
413  *
414  * Note that sohvp might still have been updated in event of a failure.
415  *
416  * @param request Current request
417  * @param data MS-SOH blob
418  * @param data_len length of MS-SOH blob
419  *
420  * @return
421  * - 0 on success.
422  * - -1 on failure.
423  */
424 int soh_verify(request_t *request, uint8_t const *data, unsigned int data_len) {
425 
426  fr_pair_t *vp;
427  eap_soh hdr;
428  soh_response resp;
429  soh_mode_subheader mode;
430  soh_tlv tlv;
431  int curr_shid =- 1, curr_shid_c =- 1, curr_hc =- 1;
432 
433  fr_assert(request->packet != NULL);
434 
435  hdr.tlv_type = soh_pull_be_16(data); data += 2;
436  hdr.tlv_len = soh_pull_be_16(data); data += 2;
437  hdr.tlv_vendor = soh_pull_be_32(data); data += 4;
438 
439  if (hdr.tlv_type != 7 || hdr.tlv_vendor != 0x137) {
440  RDEBUG2("SoH payload is %i %08x not a ms-vendor packet", hdr.tlv_type, hdr.tlv_vendor);
441  return -1;
442  }
443 
444  hdr.soh_type = soh_pull_be_16(data); data += 2;
445  hdr.soh_len = soh_pull_be_16(data); data += 2;
446  if (hdr.soh_type != 1) {
447  RDEBUG2("SoH tlv %04x is not a response", hdr.soh_type);
448  return -1;
449  }
450 
451  /* FIXME: check for sufficient data */
452  resp.outer_type = soh_pull_be_16(data); data += 2;
453  resp.outer_len = soh_pull_be_16(data); data += 2;
454  resp.vendor = soh_pull_be_32(data); data += 4;
455  resp.inner_type = soh_pull_be_16(data); data += 2;
456  resp.inner_len = soh_pull_be_16(data); data += 2;
457 
458 
459  if ((resp.outer_type != 7) || (resp.vendor != 0x137)) {
460  RDEBUG2("SoH response outer type %i/vendor %08x not recognised", resp.outer_type, resp.vendor);
461  return -1;
462  }
463  switch (resp.inner_type) {
464  case 1:
465  /* no mode sub-header */
466  RDEBUG2("SoH without mode subheader");
467  break;
468 
469  case 2:
470  mode.outer_type = soh_pull_be_16(data); data += 2;
471  mode.outer_len = soh_pull_be_16(data); data += 2;
472  mode.vendor = soh_pull_be_32(data); data += 4;
473  memcpy(mode.corrid, data, 24); data += 24;
474  mode.intent = data[0];
475  mode.content_type = data[1];
476  data += 2;
477 
478  if ((mode.outer_type != 7) || (mode.vendor != 0x137) || (mode.content_type != 0)) {
479  RDEBUG3("SoH mode subheader outer type %i/vendor %08x/content type %i invalid",
480  mode.outer_type, mode.vendor, mode.content_type);
481  return -1;
482  }
483  RDEBUG3("SoH with mode subheader");
484  break;
485 
486  default:
487  RDEBUG2("SoH invalid inner type %i", resp.inner_type);
488  return -1;
489  }
490 
491  /* subtract off the relevant amount of data */
492  if (resp.inner_type==2) {
493  data_len = resp.inner_len - 34;
494  } else {
495  data_len = resp.inner_len;
496  }
497 
498  /* TLV
499  * MS-SOH 2.2.1
500  * See also 2.2.3
501  *
502  * 1 bit mandatory
503  * 1 bit reserved
504  * 14 bits tlv type
505  * 2 bytes tlv length
506  * N bytes payload
507  *
508  */
509  while (data_len >= 4) {
510  tlv.tlv_type = soh_pull_be_16(data); data += 2;
511  tlv.tlv_len = soh_pull_be_16(data); data += 2;
512 
513  data_len -= 4;
514 
515  switch (tlv.tlv_type) {
516  case 2:
517  /* System-Health-Id TLV
518  * MS-SOH 2.2.3.1
519  *
520  * 3 bytes IANA/SMI vendor code
521  * 1 byte component (i.e. within vendor, which SoH component
522  */
523  curr_shid = soh_pull_be_24(data);
524  curr_shid_c = data[3];
525  RDEBUG2("SoH System-Health-ID vendor %08x component=%i", curr_shid, curr_shid_c);
526  break;
527 
528  case 7:
529  /* Vendor-Specific packet
530  * MS-SOH 2.2.3.3
531  *
532  * 4 bytes vendor, supposedly ignored by NAP
533  * N bytes payload; for Microsoft component#0 this is the MS TV stuff
534  */
535  if (curr_shid==0x137 && curr_shid_c==0) {
536  RDEBUG2("SoH MS type-value payload");
537  eap_peap_soh_mstlv(request, data + 4, tlv.tlv_len - 4);
538  } else {
539  RDEBUG2("SoH unhandled vendor-specific TLV %08x/component=%i %i bytes payload",
540  curr_shid, curr_shid_c, tlv.tlv_len);
541  }
542  break;
543 
544  case 8:
545  /* Health-Class
546  * MS-SOH 2.2.3.5.6
547  *
548  * 1 byte integer
549  */
550  RDEBUG2("SoH Health-Class %i", data[0]);
551  curr_hc = data[0];
552  break;
553 
554  case 9:
555  /* Software-Version
556  * MS-SOH 2.2.3.5.7
557  *
558  * 1 byte integer
559  */
560  RDEBUG2("SoH Software-Version %i", data[0]);
561  break;
562 
563  case 11:
564  /* Health-Class status
565  * MS-SOH 2.2.3.5.9
566  *
567  * variable data; for the MS System Health vendor, these are 4-byte
568  * integers which are a really, really dumb format:
569  *
570  * 28 bits ignore
571  * 1 bit - 1==product snoozed
572  * 1 bit - 1==microsoft product
573  * 1 bit - 1==product up-to-date
574  * 1 bit - 1==product enabled
575  */
576  RDEBUG2("SoH Health-Class-Status - current shid=%08x component=%i", curr_shid, curr_shid_c);
577 
578  if (curr_shid == 0x137 && curr_shid_c == 128) {
579  char const *s, *t;
580  uint32_t hcstatus = soh_pull_be_32(data);
581 
582  RDEBUG2("SoH Health-Class-Status microsoft DWORD=%08x", hcstatus);
583 
584  MEM(pair_update_request(&vp, attr_soh_ms_windows_health_status) >= 0);
585  switch (curr_hc) {
586  case 4:
587  /* security updates */
588  s = "security-updates";
589  switch (hcstatus) {
590  case 0xff0005:
591  fr_pair_value_aprintf(vp, "%s ok all-installed", s);
592  break;
593 
594  case 0xff0006:
595  fr_pair_value_aprintf(vp, "%s warn some-missing", s);
596  break;
597 
598  case 0xff0008:
599  fr_pair_value_aprintf(vp, "%s warn never-started", s);
600  break;
601 
602  case 0xc0ff000c:
603  fr_pair_value_aprintf(vp, "%s error no-wsus-srv", s);
604  break;
605 
606  case 0xc0ff000d:
607  fr_pair_value_aprintf(vp, "%s error no-wsus-clid", s);
608  break;
609 
610  case 0xc0ff000e:
611  fr_pair_value_aprintf(vp, "%s warn wsus-disabled", s);
612  break;
613 
614  case 0xc0ff000f:
615  fr_pair_value_aprintf(vp, "%s error comm-failure", s);
616  break;
617 
618  case 0xc0ff0010:
619  fr_pair_value_aprintf(vp, "%s warn needs-reboot", s);
620  break;
621 
622  default:
623  fr_pair_value_aprintf(vp, "%s error %08x", s, hcstatus);
624  break;
625  }
626  break;
627 
628  case 3:
629  /* auto updates */
630  s = "auto-updates";
631  switch (hcstatus) {
632  case 1:
633  fr_pair_value_aprintf(vp, "%s warn disabled", s);
634  break;
635 
636  case 2:
637  fr_pair_value_aprintf(vp, "%s ok action=check-only", s);
638  break;
639 
640  case 3:
641  fr_pair_value_aprintf(vp, "%s ok action=download", s);
642  break;
643 
644  case 4:
645  fr_pair_value_aprintf(vp, "%s ok action=install", s);
646  break;
647 
648  case 5:
649  fr_pair_value_aprintf(vp, "%s warn unconfigured", s);
650  break;
651 
652  case 0xc0ff0003:
653  fr_pair_value_aprintf(vp, "%s warn service-down", s);
654  break;
655 
656  case 0xc0ff0018:
