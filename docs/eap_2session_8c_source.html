31 static int _eap_session_free(eap_session_t *eap_session)
32 {
33  request_t *request = eap_session->request;
34 
35  if (eap_session->identity) {
36  talloc_free(eap_session->identity);
37  eap_session->identity = NULL;
38  }
39 
40 #ifdef WITH_VERIFY_PTR
41  if (eap_session->prev_round) (void)fr_cond_assert(talloc_parent(eap_session->prev_round) == eap_session);
42  if (eap_session->this_round) (void)fr_cond_assert(talloc_parent(eap_session->this_round) == eap_session);
43 #endif
44 
45  /*
46  * Give helpful debug messages if:
47  *
48  * we're debugging TLS sessions, which don't finish,
49  * and which aren't deleted early due to a likely RADIUS
50  * retransmit which nukes our ID, and therefore our state.
51  */
52  if (((request && RDEBUG_ENABLED) || (!request && DEBUG_ENABLED)) &&
53  (eap_session->tls && !eap_session->finished && ((fr_time() - eap_session->updated) > (((fr_time_t) 3) * NSEC)))) {
54  ROPTIONAL(RWDEBUG, WARN, "!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!");
55  ROPTIONAL(RWDEBUG, WARN, "!! EAP session %016" PRIxPTR " did not finish! !!",
56  (uintptr_t)eap_session);
57  ROPTIONAL(RWDEBUG, WARN, "!! See http://wiki.freeradius.org/guide/Certificate_Compatibility !!");
58  ROPTIONAL(RWDEBUG, WARN, "!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!");
59  }
60 
61  ROPTIONAL(RDEBUG4, DEBUG4, "Freeing eap_session_t %p", eap_session);
62 
63  return 0;
64 }

66 /** Allocate a new eap_session_t
67  *
68  * Allocates a new eap_session_t, and inserts it into the REQUEST_DATA_EAP_SESSION index
69  * of the request.
70  *
71  * @note The eap_session_t will remove itself from the #REQUEST_DATA_EAP_SESSION index
72  * if it is freed. This is to simplify management of the request data entry.
73  *
74  * @param[in] request That generated this eap_session_t.
75  * @return
76  * - A new #eap_session_t on success.
77  * - NULL on failure.
78  */
79 static eap_session_t *eap_session_alloc(request_t *request)
80 {
81  eap_session_t *eap_session;
82 
83  eap_session = talloc_zero(NULL, eap_session_t);
84  if (!eap_session) {
85  ERROR("Failed allocating eap_session");
86  return NULL;
87  }
88  eap_session->request = request;
89  eap_session->updated = request->packet->timestamp;
90 
91  talloc_set_destructor(eap_session, _eap_session_free);
92 
93  return eap_session;
94 }

96 /** 'destroy' an EAP session and dissasociate it from the current request
97  *
98  * @note This could be done in the eap_session_t destructor (and was done previously)
99  * but this made the code too hard to follow, and too fragile.
100  *
101  * @see eap_session_continue
102  * @see eap_session_freeze
103  * @see eap_session_thaw
104  *
105  * @param eap_session to destroy (disassociate and free).
106  */
107 void eap_session_destroy(eap_session_t **eap_session)
108 {
109  if (!*eap_session) return;
110 
111  if (!(*eap_session)->request) {
112  TALLOC_FREE(*eap_session);
113  return;
114  }
115 
116 #ifndef NDEBUG
117  {
118  eap_session_t *in_request;
119 
120  in_request = request_data_get((*eap_session)->request, NULL, REQUEST_DATA_EAP_SESSION);
121 
122  /*
123  * Additional sanity check. Either there's no eap_session
124  * associated with the request, or it matches the one we're
125  * about to free.
126  */
127  fr_assert(!in_request || (*eap_session == in_request));
128  }
129 #else
130  (void) request_data_get((*eap_session)->request, NULL, REQUEST_DATA_EAP_SESSION);
131 #endif
132 
133  TALLOC_FREE(*eap_session);
134 }

136 /** Freeze an #eap_session_t so that it can continue later
137  *
138  * Sets the request and pointer to the eap_session to NULL. Primarily here to help track
139  * the lifecycle of an #eap_session_t.
140  *
141  * The actual freezing/thawing and management (ensuring it's available during multiple
142  * rounds of EAP) of the #eap_session_t associated with REQUEST_DATA_EAP_SESSION, is
143  * done by the state API.
144  *
145  * @note must be called before mod_* functions in rlm_eap return.
146  *
147  * @see eap_session_continue
148  * @see eap_session_thaw
149  * @see eap_session_destroy
150  *
151  * @param eap_session to freeze.
152  */
153 void eap_session_freeze(eap_session_t **eap_session)
154 {
155  if (!*eap_session) return;
156 
157  fr_assert((*eap_session)->request);
158  (*eap_session)->request = NULL;
159  *eap_session = NULL;
160 }

162 /** Thaw an eap_session_t so it can be continued
163  *
164  * Retrieve an #eap_session_t from the request data, and set relevant fields. Primarily
165  * here to help track the lifecycle of an #eap_session_t.
166  *
167  * The actual freezing/thawing and management (ensuring it's available during multiple
168  * rounds of EAP) of the #eap_session_t associated with REQUEST_DATA_EAP_SESSION, is
169  * done by the state API.
170  *
171  * @note #eap_session_continue should be used instead if ingesting an #eap_packet_raw_t.
172  *
173  * @see eap_session_continue
174  * @see eap_session_freeze
175  * @see eap_session_destroy
176  *
177  * @param request to retrieve session from.
178  * @return
179  * - The #eap_session_t associated with this request.
180  * MUST be freed with #eap_session_destroy if being disposed of, OR
181  * MUST be re-frozen with #eap_session_freeze if the authentication session will
182  * continue when a future request is received.
183  * - NULL if no #eap_session_t associated with this request.
184  */
185 eap_session_t *eap_session_thaw(request_t *request)
186 {
187  eap_session_t *eap_session;
188 
189  eap_session = request_data_reference(request, NULL, REQUEST_DATA_EAP_SESSION);
190  if (!eap_session) return NULL;
191 
192  if (!fr_cond_assert(eap_session->inst)) return NULL;
193 
194  fr_assert(!eap_session->request); /* If triggered, something didn't freeze the session */
195  eap_session->request = request;
196  eap_session->updated = request->packet->timestamp;
197 
198  return eap_session;
199 }

201 /** Extract the EAP identity from EAP-Identity-Response packets
202  *
203  * @param[in] request The current request.
204  * @param[in] eap_session EAP-Session to associate identity with.
205  * @param[in] eap_packet To extract the identity from.
206  * @return
207  * - The user's EAP-Identity.
208  * - or NULL on error.
209  */
210 static char *eap_identity(request_t *request, eap_session_t *eap_session, eap_packet_raw_t *eap_packet)
211 {
212  uint16_t len;
213 
214  if (!eap_packet ||
215  (eap_packet->code != FR_EAP_CODE_RESPONSE) ||
216  (eap_packet->data[0] != FR_EAP_METHOD_IDENTITY)) return NULL;
217 
218  memcpy(&len, eap_packet->length, sizeof(uint16_t));
219  len = ntohs(len);
220 
221  /*
222  * Note: The minimum length here is 5.
223  * Previous versions of FreeRADIUS limited the length to 6 and
224  * checked for data[0] != \0.
225  *
226  * This was incorrect, and broke encrypted pseudonyms in EAP-SIM/AKA.
227  *
228  * RFC 3748 states - If the Identity is unknown, the
229  * Identity Response field should be zero bytes in length. The
230  * Identity Response field MUST NOT be null terminated. In all
231  * cases, the length of the Type-Data field is derived from the
232  * Length field of the Request/Response packet.
233  *
234  * Code (1) + Identifier (1) + Length (2) + Type (1) = 5.
235  *
236  * The maximum value is not bounded by the RFC. The eap_validation()
237  * function called before eap_identity(), checks that the length
238  * field does not overrun the available data.
239  *
240  * In some EAP methods, the identity may be encrypted, and padded
241  * out to the block size of the encryption method. These identities
242  * may contain nuls, and made be much larger than humanly readable
243  * identiies.
244  *
245  * The identity value *MUST NOT* be artificially limited or truncated
246  * here.
247  */
248  if (len < sizeof(eap_packet_raw_t)) {
249  REDEBUG("EAP-Identity length field too short, expected >= 5, got %u", len);
250  return NULL;
251  }
252 
253  /*
254  * If the length is 5, then a buffer with a length of 1 is
255  * created with a \0 byte.
256  */
257  return talloc_bstrndup(eap_session, (char *)&eap_packet->data[1], len - 5);
258 }

260 /** Ingest an eap_packet into a thawed or newly allocated session
261  *
262  * If eap_packet is an Identity-Response then allocate a new eap_session and fill the identity.
263  *
264  * If eap_packet is not an identity response, retrieve the pre-existing eap_session_t from request
265  * data.
266  *
267  * If no User-Name attribute is present in the request, one will be created from the
268  * Identity-Response received when the eap_session was allocated.
269  *
270  * @see eap_session_freeze
271  * @see eap_session_thaw
272  * @see eap_session_destroy
273  *
274  * @param[in] instance of rlm_eap that created the session.
275  * @param[in] eap_packet_p extracted from the RADIUS Access-Request.
276  * Consumed or freed by this function.
277  * Do not access after calling this function.
278  * Is a **so the packet pointer can be
279  * set to NULL.
280  * @param[in] request The current request.
281  * @return
282  * - A newly allocated eap_session_t, or the one associated with the current request.
283  * MUST be freed with #eap_session_destroy if being disposed of, OR
284  * MUST be re-frozen with #eap_session_freeze if the authentication session will
285  * continue when a future request is received.
286  * - NULL on error.
287  */
288 eap_session_t *eap_session_continue(void const *instance, eap_packet_raw_t **eap_packet_p, request_t *request)
289 {
290  eap_session_t *eap_session = NULL;
291  eap_packet_raw_t *eap_packet;
292  fr_pair_t *user;
293 
294  eap_packet = *eap_packet_p;
295 
296  /*
297  * RFC 3579 - Once EAP has been negotiated, the NAS SHOULD
298  * send an initial EAP-Request message to the authenticating
299  * peer. This will typically be an EAP-Request/Identity,
300  * although it could be an EAP-Request for an authentication
301  * method (Types 4 and greater).
302  *
303  * This means that if there is no State attribute, we should
304  * consider this as the start of a new session.
305  */
306  eap_session = eap_session_thaw(request);
307  if (!eap_session) {
308  eap_session = eap_session_alloc(request);
309  if (!eap_session) {
310  error_round:
311  talloc_free(*eap_packet_p);
312  *eap_packet_p = NULL;
313  return NULL;
314  }
315  eap_session->inst = instance;
316 
317  if (RDEBUG_ENABLED4) {
318  RDEBUG4("New EAP session - eap_session_t %p", eap_session);
319  } else {
320  RDEBUG2("New EAP session started");
321  }
322 
323  /*
324  * All fields in the eap_session are set to zero.
325  */
326  switch (eap_packet->data[0]) {
327  case FR_EAP_METHOD_IDENTITY:
328  eap_session->identity = eap_identity(request, eap_session, eap_packet);
329  if (!eap_session->identity) {
330  REDEBUG("Invalid identity response");
331  goto error_session;
332  }
333 
334  /*
335  * Sometimes we need the hex stream to determine where
336  * random junk is coming from.
337  */
338  RHEXDUMP3((uint8_t *const)eap_session->identity,
339  talloc_array_length(eap_session->identity) - 1,
340  "EAP Identity Response - \"%pV\"",
341  fr_box_strvalue_len(eap_session->identity,
342  talloc_array_length(eap_session->identity) - 1));
343  break;
344 
345  case FR_EAP_METHOD_INVALID:
346  case FR_EAP_METHOD_NOTIFICATION:
347  case FR_EAP_METHOD_NAK:
348  REDEBUG("Initial EAP method %s(%u) invalid",
349  eap_type2name(eap_packet->data[0]), eap_packet->data[0]);
350  goto error_session;
351 
352  /*
353  * Initialise a zero length identity, as we've
354  * not been provided with one at the start of the
355  * EAP method.
356  */
357  default:
358  eap_session->identity = talloc_bstrndup(eap_session, "", 0);
359  break;
360  }
361 
362  /*
363  * If the index is removed by something else
364  * like the state being cleaned up, then we
365  * still want the eap_session to be freed, which
366  * is why we set free_opaque to true.
367  *
368  * We must pass a NULL pointer to associate the
369  * the EAP_SESSION data with, else we'll break
370  * tunneled EAP, where the inner EAP module is
371  * a different instance to the outer one.
372  */
373  request_data_talloc_add(request, NULL, REQUEST_DATA_EAP_SESSION, eap_session_t,
374  eap_session, true, true, true);
375 
376  /*
377  * Continue a previously started EAP-Session
378  */
379  } else {
380  if (RDEBUG_ENABLED4) {
381  RDEBUG4("Continuing EAP session - eap_session_t %p", eap_session);
382  } else {
383  RDEBUG2("Continuing EAP session");
384  }
385 
386  (void) talloc_get_type_abort(eap_session, eap_session_t);
387  eap_session->rounds++;
388  if (eap_session->rounds >= 50) {
389  RERROR("Failing EAP session due to too many round trips");
390  error_session:
391  eap_session_destroy(&eap_session);
392  goto error_round;
393  }
394 
395  /*
396  * Even more paranoia. Without this, some weird
397  * clients could do crazy things.
398  *
399  * It's ok to send EAP sub-type NAK in response
400  * to a request for a particular type, but it's NOT
401  * OK to blindly return data for another type.
402  */
403  if ((eap_packet->data[0] != FR_EAP_METHOD_NAK) &&
404  (eap_packet->data[0] != eap_session->type)) {
405  RERROR("Response appears to match a previous request, but the EAP type is wrong");
406  RERROR("We expected EAP type %s, but received type %s",
407  eap_type2name(eap_session->type),
408  eap_type2name(eap_packet->data[0]));
409  RERROR("Your Supplicant or NAS is probably broken");
410  goto error_round;
411  }
412  }
413 
414  /*
415  * RFC3579 In order to permit non-EAP aware RADIUS proxies to forward the
416  * Access-Request packet, if the NAS initially sends an
417  * EAP-Request/Identity message to the peer, the NAS MUST copy the
418  * contents of the Type-Data field of the EAP-Response/Identity received
419  * from the peer into the User-Name attribute and MUST include the
420  * Type-Data field of the EAP-Response/Identity in the User-Name
421  * attribute in every subsequent Access-Request.
422  */
423  user = fr_pair_find_by_da(&request->request_pairs, attr_user_name);
424  if (!user) {
425  /*
426  * NAS did not set the User-Name
427  * attribute, so we set it here and
428  * prepend it to the beginning of the
429  * request vps so that autz's work
430  * correctly
431  */
432  RDEBUG2("Broken NAS did not set User-Name, setting from EAP Identity");
433  MEM(pair_add_request(&user, attr_user_name) >= 0);
434  fr_pair_value_bstrdup_buffer(user, eap_session->identity, true);
435  /*
436  * The RFC 3579 is pretty unambiguous, the main issue is that the EAP Identity Response
437  * can be significantly longer than 253 bytes (the maximum RADIUS
438  * attribute length), and the RFC is silent about what happens then.
439  *
440  * The behaviour seen in the wild, is that the NAS will use the Mac-Address
441  * of the connecting device as the User-name, and send the Identity in full,
442  * so if the EAP identity is longer than the max RADIUS attribute length
443  * then ignore mismatches.
444  */
445  } else if ((talloc_array_length(eap_session->identity) - 1) <= RADIUS_MAX_STRING_LENGTH) {
446  /*
447  * A little more paranoia. If the NAS
448  * *did* set the User-Name, and it doesn't
449  * match the identity, (i.e. If they
450  * change their User-Name part way through
451  * the EAP transaction), then reject the
452  * request as the NAS is doing something
453  * funny.
454  */
455  if (talloc_memcmp_bstr(eap_session->identity, user->vp_strvalue) != 0) {
456  REDEBUG("Identity from EAP Identity-Response \"%s\" does not match User-Name attribute \"%s\"",
457  eap_session->identity, user->vp_strvalue);
458  goto error_round;
459  }
460  }
461 
462  eap_session->this_round = eap_round_build(eap_session, eap_packet_p);
463  if (!eap_session->this_round) {
464  REDEBUG("Failed allocating memory for round");
465  goto error_session;
466  }
467 
468  return eap_session;
469 }
