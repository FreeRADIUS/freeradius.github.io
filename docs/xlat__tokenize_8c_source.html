213 static inline int xlat_tokenize_alternation(TALLOC_CTX *ctx, xlat_exp_t **head, xlat_flags_t *flags, fr_sbuff_t *in,
214  tmpl_rules_t const *t_rules)
215 {
216  xlat_exp_t *node;
217 
218  XLAT_DEBUG("ALTERNATE <-- %pV", fr_box_strvalue_len(fr_sbuff_current(in), fr_sbuff_remaining(in)));
219 
220  node = xlat_exp_alloc_null(ctx);
221  xlat_exp_set_type(node, XLAT_ALTERNATE);
222  if (xlat_tokenize_expansion(node, &node->child, &node->flags, in, t_rules) < 0) {
223  error:
224  *head = NULL;
225  talloc_free(node);
226  return -1;
227  }
228 
229  if (!fr_sbuff_adv_past_str_literal(in, ":-")) {
230  fr_strerror_printf("Expected ':-' after first expansion");
231  goto error;
232  }
233 
234  /*
235  * Allow the RHS to be empty as a special case.
236  */
237  if (fr_sbuff_next_if_char(in, '}')) {
238  node->alternate = xlat_exp_alloc(node, XLAT_LITERAL, "", 0);
239  xlat_flags_merge(&node->flags, &node->child->flags);
240  *head = node;
241  return 0;
242  }
243 
244  /*
245  * Parse the alternate expansion.
246  */
247  if (xlat_tokenize_literal(node, &node->alternate, &node->flags, in,
248  true, &xlat_rules, t_rules) < 0) goto error;
249 
250  if (!node->alternate) {
251  talloc_free(node);
252  fr_strerror_printf("Empty expansion is invalid");
253  goto error;
254  }
255 
256  if (!fr_sbuff_next_if_char(in, '}')) {
257  fr_strerror_printf("Missing closing brace");
258  goto error;
259  }
260 
261  xlat_flags_merge(flags, &node->flags);
262  *head = node;
263 
264  return 0;
265 }

323 /** Parse an xlat function and its child arguments
324  *
325  * Parses a function call string in the format
326  * @verbatim %{<func>:<arguments} @endverbatim
327  *
328  * @return
329  * - 0 if the string was parsed into a function.
330  * - <0 on parse error.
331  */
332 static inline int xlat_tokenize_function(TALLOC_CTX *ctx, xlat_exp_t **head, xlat_flags_t *flags, fr_sbuff_t *in,
333  tmpl_rules_t const *rules)
334 {
335  xlat_exp_t *node;
336  xlat_t *func;
337  fr_sbuff_marker_t m_s;
338 
339  /*
340  * Special characters, spaces, etc. cannot be
341  * module names.
342  */
343  static bool const func_chars[UINT8_MAX + 1] = {
344  SBUFF_CHAR_CLASS_ALPHA_NUM,
345  ['.'] = true, ['-'] = true, ['_'] = true,
346  };
347 
348  XLAT_DEBUG("FUNC <-- %pV", fr_box_strvalue_len(fr_sbuff_current(in), fr_sbuff_remaining(in)));
349 
350  /*
351  * %{module:args}
352  */
353  fr_sbuff_marker(&m_s, in);
354  fr_sbuff_adv_past_allowed(in, SIZE_MAX, func_chars);
355 
356  if (!fr_sbuff_is_char(in, ':')) {
357  fr_strerror_printf("Can't find function/argument separator");
358  bad_function:
359  *head = NULL;
360  fr_sbuff_set(in, &m_s); /* backtrack */
361  fr_sbuff_marker_release(&m_s);
362  return -1;
363  }
364 
365  func = xlat_func_find(fr_sbuff_current(&m_s), fr_sbuff_behind(&m_s));
366 
367  /*
368  * Allocate a node to hold the function
369  */
370  node = xlat_exp_alloc(ctx, XLAT_FUNC, fr_sbuff_current(&m_s), fr_sbuff_behind(&m_s));
371  if (!func) {
372  if (!rules || !rules->allow_unresolved) {
373  fr_strerror_printf("Unresolved expansion functions are not allowed here");
374  goto bad_function;
375  }
376  xlat_exp_set_type(node, XLAT_FUNC_UNRESOLVED);
377  node->flags.needs_resolving = true; /* Needs resolution during pass2 */
378  } else {
379  node->call.func = func;
380  node->flags.needs_async = func->needs_async;
381  }
382 
383  fr_sbuff_next(in); /* Skip the ':' */
384  XLAT_DEBUG("FUNC-ARGS <-- %s ... %pV",
385  node->fmt, fr_box_strvalue_len(fr_sbuff_current(in), fr_sbuff_remaining(in)));
386 
387  fr_sbuff_marker_release(&m_s);
388 
389  /*
390  * Now parse the child nodes that form the
391  * function's arguments.
392  */
393  if (xlat_tokenize_literal(node, &node->child, &node->flags, in, true, &xlat_rules, rules) < 0) {
394  error:
395  *head = NULL;
396  talloc_free(node);
397  return -1;
398  }
399 
400  if (!fr_sbuff_next_if_char(in, '}')) {
401  fr_strerror_printf("Missing closing brace");
402  goto error;
403  }
404 
405  xlat_flags_merge(flags, &node->flags);
406  *head = node;
407 
408  return 0;
409 }

411 /** Parse an attribute ref or a virtual attribute
412  *
413  */
414 static inline int xlat_tokenize_attribute(TALLOC_CTX *ctx, xlat_exp_t **head, xlat_flags_t *flags, fr_sbuff_t *in,
415  fr_sbuff_parse_rules_t const *p_rules, tmpl_rules_t const *t_rules)
416 {
417  ssize_t slen;
418  tmpl_attr_error_t err;
419  tmpl_t *vpt = NULL;
420  xlat_exp_t *node;
421  xlat_t *func;
422  fr_sbuff_marker_t m_s;
423 
424  XLAT_DEBUG("ATTRIBUTE <-- %pV", fr_box_strvalue_len(fr_sbuff_current(in), fr_sbuff_remaining(in)));
425 
426  /*
427  * We need a local copy as we always allow unknowns.
428  * This is because not all attribute references
429  * reference real attributes in the dictionaries,
430  * and instead are "virtual" attributes like
431  * Foreach-Variable-N.
432  */
433  tmpl_rules_t our_t_rules;
434 
435  if (t_rules) {
436  memcpy(&our_t_rules, t_rules, sizeof(our_t_rules));
437  } else {
438  memset(&our_t_rules, 0, sizeof(our_t_rules));
439  }
440 
441  our_t_rules.allow_unresolved = true; /* So we can check for virtual attributes later */
442  our_t_rules.prefix = TMPL_ATTR_REF_PREFIX_NO; /* Must be NO to stop %{&User-Name} */
443 
444  fr_sbuff_marker(&m_s, in);
445 
446  MEM(node = xlat_exp_alloc_null(ctx));
447  slen = tmpl_afrom_attr_substr(node, &err, &vpt, in, p_rules, &our_t_rules);
448  if (slen <= 0) {
449  fr_sbuff_advance(in, slen * -1);
450 
451  /*
452  * If the parse error occurred before the ':'
453  * then the error is changed to 'Unknown module',
454  * as it was more likely to be a bad module name,
455  * than a request qualifier.
456  */
457  if (err == TMPL_ATTR_ERROR_MISSING_TERMINATOR) fr_sbuff_set(in, &m_s);
458  error:
459  *head = NULL;
460  fr_sbuff_marker_release(&m_s);
461  talloc_free(node);
462  return -1;
463  }
464 
465  /*
466  * Might be a virtual XLAT attribute, which is identical
467  * to a normal function but called without an argument
468  * list.
469  *
470  * We only consider virtual attributes at the top level
471  * but we do allow request references and lists to be
472  * specified so that the virtual attribute can operate
473  * in different contexts (i.e. on the parent request).
474  */
475  if ((tmpl_is_attr_unresolved(vpt) || (tmpl_is_attr(vpt) && tmpl_da(vpt)->flags.virtual)) &&
476  (tmpl_attr_count(vpt) == 1)) {
477  if (tmpl_is_attr(vpt)) {
478  func = xlat_func_find(tmpl_da(vpt)->name, -1);
479  } else {
480  func = xlat_func_find(tmpl_attr_unresolved(vpt), -1);
481  }
482  if (func) {
483  xlat_exp_set_type(node, XLAT_VIRTUAL);
484  xlat_exp_set_name_buffer_shallow(node, vpt->name);
485 
486  XLAT_DEBUG("VIRTUAL <-- %pV",
487  fr_box_strvalue_len(fr_sbuff_current(in), fr_sbuff_remaining(in)));
488  node->call.func = func;
489  node->attr = vpt; /* Store for context */
490  node->flags.needs_async = func->needs_async;
491 
492  if (!fr_sbuff_next_if_char(in, '}')) {
493  fr_strerror_printf("Missing closing brace");
494  goto error;
495  }
496 
497  goto done;
498  }
499 
500  /*
501  * Hack - We need a proper virtual attribute
502  * registry. There seems to have been a
503  * partial conversion at some point which
504  * needs completing.
505  *
506  * This means we fall back to creating the
507  * tmpl at a normal attribute which gets
508  * processed correctly by the eval code
509  * because of the virtual flag in the dictionary.
510  */
511  if (tmpl_is_attr(vpt)) goto do_attr;
512 
513  /*
514  * Not a normal attribute, not a virtual attribute
515  * and we're not allowing unresolved attributes.
516  */
517  if (!t_rules || !t_rules->allow_unresolved) {
518  talloc_free(vpt);
519 
520  fr_strerror_printf("Unresolved attributes not allowed in expansions here");
521  fr_sbuff_set(in, &m_s); /* Error at the start of the attribute */
522  goto error;
523  }
524 
525  /*
526  * Mark the xlat exp as needing pass2 resolution.
527  */
528  xlat_exp_set_type(node, XLAT_VIRTUAL_UNRESOLVED);
529  xlat_exp_set_name_buffer_shallow(node, vpt->name); /* vpt may be freed */
530 
531  node->attr = vpt;
532  node->flags.needs_resolving = true;
533  /*
534  * It's a straight attribute, nothing special
535  */
536  } else {
537  do_attr:
538  xlat_exp_set_type(node, XLAT_ATTRIBUTE);
539  xlat_exp_set_name_buffer_shallow(node, vpt->name);
540 
541  node->attr = vpt;
542  }
543 
544  if (!fr_sbuff_next_if_char(in, '}')) {
545  fr_strerror_printf("Missing closing brace");
546  goto error;
547  }
548 
549 done:
550  xlat_flags_merge(flags, &node->flags);
551  *head = node;
552  fr_sbuff_marker_release(&m_s);
553  return 0;
554 }

556 static int xlat_tokenize_expansion(TALLOC_CTX *ctx, xlat_exp_t **head, xlat_flags_t *flags, fr_sbuff_t *in,
557  tmpl_rules_t const *t_rules)
558 {
559  size_t len;
560  fr_sbuff_marker_t s_m;
561  char hint;
562  int ret;
563  fr_sbuff_term_t hint_tokens = FR_SBUFF_TERMS(
564  L(" "), /* First special token is a ' ' - Likely a syntax error */
565  L(":"), /* First special token is a ':' i.e. '%{func:' */
566  L("["), /* First special token is a '[' i.e. '%{attr[<idx>]}' */
567  L("}") /* First special token is a '}' i.e. '%{<attrref>}' */
568  );
569 
570  fr_sbuff_parse_rules_t attr_p_rules = {
571  .escapes = &xlat_unescape,
572  .terminals = &FR_SBUFF_TERM("}")
573  };
574 
575  XLAT_DEBUG("EXPANSION <-- %pV", fr_box_strvalue_len(fr_sbuff_current(in), fr_sbuff_remaining(in)));
576 
577  /*
578  * %{...}:-bar}
579  */
580  if (fr_sbuff_adv_past_str_literal(in, "%{")) {
581  return xlat_tokenize_alternation(ctx, head, flags, in, t_rules);
582  }
583 
584  /*
585  * :-bar}
586  */
587  if (fr_sbuff_is_str_literal(in, ":-")) {
588  fr_strerror_printf("First item in alternation cannot be empty");
589  return -2;
590  }
591 
592 #ifdef HAVE_REGEX
593  /*
594  * Handle regex's %{<num>} specially.
595  */
596  if (fr_sbuff_is_digit(in)) {
597  ret = xlat_tokenize_regex(ctx, head, flags, in);
598  if (ret <= 0) return ret;
599  }
600 #endif /* HAVE_REGEX */
601 
602  /*
603  * %{Attr-Name}
604  * %{Attr-Name[#]}
605  * %{request.Attr-Name}
606  * %{mod:foo}
607  */
608 
609  /*
610  * Check for empty expressions %{} %{: %{[
611  */
612  fr_sbuff_marker(&s_m, in);
613  len = fr_sbuff_adv_until(in, SIZE_MAX, &hint_tokens, '\0');
614 
615  /*
616  * This means the end of a string not containing any of the other
617  * tokens was reached.
618  *
619  * e.g. '%{myfirstxlat'
620  */
621  if (!fr_sbuff_remaining(in)) {
622  fr_strerror_printf("Missing closing brace");
623  fr_sbuff_marker_release(&s_m);
624  return -1;
625  }
626 
627  hint = *fr_sbuff_current(in);
628 
629  XLAT_DEBUG("EXPANSION HINT TOKEN '%c'", hint);
630  if (len == 0) {
631  switch (hint) {
632  case '}':
633  fr_strerror_printf("Empty expression is invalid");
634  return -1;
635 
636  case ':':
637  fr_strerror_printf("Missing expansion function");
638  return -1;
639 
640  case '[':
641  fr_strerror_printf("Missing attribute name");
642  return -1;
643 
644  default:
645  break;
646  }
647  }
648 
649  /*
650  * Hint token is a ':' it's an xlat function %{<func>:<args}
651  */
652  switch (hint) {
653  case ':':
654  {
655  fr_sbuff_set(in, &s_m); /* backtrack */
656  fr_sbuff_marker_release(&s_m);
657 
658  ret = xlat_tokenize_function(ctx, head, flags, in, t_rules);
659  if (ret <= 0) return ret;
660  }
661  break;
662 
663  /*
664  * Hint token is a:
665  * - '[' - Which is an attribute index, so it must be an attribute.
666  * - '}' - The end of the expansion, which means it was a bareword.
667  */
668  case '.':
669  case '}':
670  case '[':
671  fr_sbuff_set(in, &s_m); /* backtrack */
672  fr_sbuff_marker_release(&s_m);
673 
674  if (xlat_tokenize_attribute(ctx, head, flags, in, &attr_p_rules, t_rules) < 0) return -1;
675  break;
676 
677  /*
678  * Hint token was whitespace
679  *
680  * e.g. '%{my '
681  */
682  default:
683  /*
684  * Box print is so we get \t \n etc..
685  */
686  fr_strerror_printf("Invalid char '%pV' in expression", fr_box_strvalue_len(fr_sbuff_current(in), 1));
687  return -1;
688  }
689 
690  return 0;
691 }

693 /** Parse an xlat literal i.e. a non-expansion or non-function
694  *
695  * When this function is being used outside of an xlat expansion, i.e. on a string
696  * which contains one or more xlat expansions, it uses the terminal grammar and
697  * escaping rules of that string type.
698  *
699  * Which this function is being used inside of an xlat expansion, it uses xlat specific
700  * terminal grammar and escaping rules.
701  *
702  * This allows us to be smart about processing quotes within the expansions themselves.
703  *
704  * @param[in] ctx to allocate nodes in. Note: All nodes will be
705  * allocated in the same ctx. This is to allow
706  * manipulation by xlat instantiation functions
707  * later.
708  * @param[out] head Where to write the first child node.
709  * @param[out] flags where we store flags information for the parent.
710  * @param[in] in sbuff to parse.
711  * @param[in] brace true if we're inside a braced expansion, else false.
712  * @param[in] p_rules that control parsing.
713  * @param[in] t_rules that control attribute reference and xlat function parsing.
714  * @return
715  * - 0 on success.
716  * - -1 on failure.
717  */
718 static int xlat_tokenize_literal(TALLOC_CTX *ctx, xlat_exp_t **head, xlat_flags_t *flags,
719  fr_sbuff_t *in, bool brace,
720  fr_sbuff_parse_rules_t const *p_rules, tmpl_rules_t const *t_rules)
721 {
722  xlat_exp_t *node = NULL;
723  size_t len;
724  fr_sbuff_term_t expansions = FR_SBUFF_TERMS(
725  L("%C"),
726  L("%D"),
727  L("%G"),
728  L("%H"),
729  L("%I"),
730  L("%M"),
731  L("%S"),
732  L("%T"),
733  L("%Y"),
734  L("%c"),
735  L("%d"),
736  L("%e"),
737  L("%l"),
738  L("%m"),
739  L("%n"),
740  L("%s"),
741  L("%t"),
742  L("%v"),
743  L("%{")
744  );
745  fr_sbuff_term_t *tokens;
746  fr_cursor_t cursor;
747  fr_sbuff_unescape_rules_t const *escapes;
748 
749  *head = NULL;
750 
751  fr_cursor_init(&cursor, head);
752  escapes = p_rules ? p_rules->escapes : NULL;
753  tokens = p_rules && p_rules->terminals ?
754  fr_sbuff_terminals_amerge(NULL, p_rules->terminals, &expansions) : &expansions;
755 
756  for (;;) {
757  char *str;
758 
759  /*
760  * pre-allocate the node so we don't have to steal it later.
761  */
762  node = xlat_exp_alloc_null(ctx);
763 
764  /*
765  * Find the next token
766  */
767  len = fr_sbuff_out_aunescape_until(node, &str, in, SIZE_MAX, tokens, escapes);
768 
769  /*
770  * It's a literal, create a literal node...
771  */
772  if (len > 0) {
773  xlat_exp_set_type(node, XLAT_LITERAL);
774  xlat_exp_set_name_buffer_shallow(node, str);
775 
776  XLAT_DEBUG("LITERAL (%s)<-- %pV",
777  escapes ? escapes->name : "(none)",
778  fr_box_strvalue_len(str, talloc_array_length(str) - 1));
779  XLAT_HEXDUMP((uint8_t const *)str, talloc_array_length(str) - 1, " LITERAL ");
780  node->flags.needs_async = false; /* literals are always true */
781  fr_cursor_insert(&cursor, node);
782  node = NULL;
783  }
784 
785  if (fr_sbuff_adv_past_str_literal(in, "%{")) {
786  if (len == 0) TALLOC_FREE(node); /* Free the empty node */
787 
788  if (xlat_tokenize_expansion(ctx, &node, flags, in, t_rules) < 0) {
789  error:
790  talloc_free(node);
791  fr_cursor_head(&cursor);
792  fr_cursor_free_list(&cursor);
793 
794  /*
795  * Free our temporary array of terminals
796  */
797  if (tokens != &expansions) talloc_free(tokens);
798  return -1;
799  }
800  fr_cursor_insert(&cursor, node);
801  node = NULL;
802  continue;
803  }
804 
805  /*
806  * %[a-z] - A one letter expansion
807  */
808  if (fr_sbuff_next_if_char(in, '%') && fr_sbuff_is_alpha(in)) {
809  XLAT_DEBUG("ONE-LETTER <-- %pV",
810  fr_box_strvalue_len(str, talloc_array_length(str) - 1));
811 
812  if (len == 0) {
813  talloc_free_children(node); /* re-use empty nodes */
814  } else {
815  node = xlat_exp_alloc_null(ctx);
816  }
817 
818  fr_sbuff_out_abstrncpy(node, &str, in, 1);
819  xlat_exp_set_type(node, XLAT_ONE_LETTER);
820  xlat_exp_set_name_buffer_shallow(node, str);
821 
822  node->flags.needs_async = false; /* literals are always true */
823  xlat_flags_merge(flags, &node->flags);
824  fr_cursor_insert(&cursor, node);
825  node = NULL;
826  continue;
827  }
828 
829  /*
830  * We were told to look for a brace, but we ran off of
831  * the end of the string before we found one.
832  */
833  if (brace) {
834  if (len == 0) TALLOC_FREE(node); /* Free the empty node */
835 
836  if (!fr_sbuff_is_char(in, '}')) {
837  fr_strerror_printf("Missing closing brace");
838  goto error;
839  }
840  /*
841  * We're parsing the string *containing* the xlat
842  * expansions.
843  */
844  } else {
845  /* If we have an empty node, finish building it and
846  * emit it.
847  *
848  * We're about to return, and it's a useful
849  * indication to the caller that this wasn't a parse
850  * error but just an empty string.
851  */
852  if (len == 0) {
853  /*
854  * This isn't the only node in the sequence
855  * don't emit an empty trailing literal.
856  */
857  if (*head) {
858  talloc_free(node);
859  break;
860  }
861 
862  xlat_exp_set_type(node, XLAT_LITERAL);
863  xlat_exp_set_name_buffer_shallow(node, str);
864 
865  XLAT_DEBUG("LITERAL <-- (empty)");
866  node->flags.needs_async = false; /* literals are always true */
867  xlat_flags_merge(flags, &node->flags);
868  fr_cursor_insert(&cursor, node);
869  }
870  }
871  break;
872  }
873 
874  /*
875  * Free our temporary array of terminals
876  */
877  if (tokens != &expansions) talloc_free(tokens);
878 
879  return 0;
880 }

982 /** Reconstitute an xlat expression from its constituent nodes
983  *
984  * @param[in] out Where to write the output string.
985  * @param[in] head First node to print.
986  * @param[in] e_rules Specifying how to escape literal values.
987  */
988 ssize_t xlat_print(fr_sbuff_t *out, xlat_exp_t const *head, fr_sbuff_escape_rules_t const *e_rules)
989 {
990  ssize_t slen;
991  size_t at_in = fr_sbuff_used_total(out);
992  xlat_exp_t const *node = head;
993 
994  if (!node) return 0;
995 
996  while (node) {
997  switch (node->type) {
998  case XLAT_GROUP:
999  if (node->quote != T_BARE_WORD) FR_SBUFF_IN_CHAR_RETURN(out, fr_token_quote[node->quote]);
1000  xlat_print(out, node->child, fr_value_escape_by_quote[node->quote]);
1001  if (node->quote != T_BARE_WORD) FR_SBUFF_IN_CHAR_RETURN(out, fr_token_quote[node->quote]);
1002  if (node->next) FR_SBUFF_IN_CHAR_RETURN(out, ' '); /* Add ' ' between args */
1003  goto next;
1004 
1005  case XLAT_LITERAL:
1006  FR_SBUFF_IN_ESCAPE_BUFFER_RETURN(out, node->fmt, e_rules);
1007  goto next;
1008 
1009  case XLAT_ONE_LETTER:
1010  FR_SBUFF_IN_CHAR_RETURN(out, '%', node->fmt[0]);
1011  goto next;
1012 
1013  default:
1014  break;
1015  }
1016 
1017  FR_SBUFF_IN_STRCPY_LITERAL_RETURN(out, "%{");
1018  switch (node->type) {
1019  case XLAT_ATTRIBUTE:
1020  slen = tmpl_attr_print(out, node->attr, TMPL_ATTR_REF_PREFIX_NO);
1021  if (slen < 0) {
1022  error:
1023  return slen;
1024  }
1025  break;
1026 #ifdef HAVE_REGEX
1027  case XLAT_REGEX:
1028  FR_SBUFF_IN_SPRINTF_RETURN(out, "%i", node->regex_index);
1029  break;
1030 #endif
1031  case XLAT_VIRTUAL:
1032  FR_SBUFF_IN_BSTRCPY_BUFFER_RETURN(out, node->call.func->name);
1033  break;
1034 
1035  case XLAT_VIRTUAL_UNRESOLVED:
1036  FR_SBUFF_IN_BSTRCPY_BUFFER_RETURN(out, node->fmt);
1037  break;
1038 
1039  case XLAT_FUNC:
1040  FR_SBUFF_IN_BSTRCPY_BUFFER_RETURN(out, node->call.func->name);
1041  FR_SBUFF_IN_CHAR_RETURN(out, ':');
1042 
1043  if (node->child) {
1044  slen = xlat_print(out, node->child, &xlat_escape);
1045  if (slen < 0) goto error;
1046  }
1047  break;
1048 
1049  case XLAT_FUNC_UNRESOLVED:
1050  FR_SBUFF_IN_BSTRCPY_BUFFER_RETURN(out, node->fmt);
1051  FR_SBUFF_IN_CHAR_RETURN(out, ':');
1052 
1053  if (node->child) {
1054  slen = xlat_print(out, node->child, &xlat_escape);
1055  if (slen < 0) goto error;
1056  }
1057  break;
1058 
1059  case XLAT_ALTERNATE:
1060  slen = xlat_print(out, node->child, &xlat_escape);
1061  if (slen < 0) goto error;
1062 
1063  FR_SBUFF_IN_STRCPY_LITERAL_RETURN(out, ":-");
1064  slen = xlat_print(out, node->alternate, &xlat_escape);
1065  if (slen < 0) goto error;
1066  break;
1067 
1068  fr_assert_fail(NULL);
1069  break;
1070 
1071  case XLAT_INVALID:
1072  case XLAT_LITERAL:
1073  case XLAT_ONE_LETTER:
1074  case XLAT_GROUP:
1075  fr_assert_fail(NULL);
1076  break;
1077  }
1078  FR_SBUFF_IN_CHAR_RETURN(out, '}');
1079  next:
1080  node = node->next;
1081  }
1082 
1083  return fr_sbuff_used_total(out) - at_in;
1084 }

1135 /** Tokenize an xlat expansion into a series of XLAT_TYPE_CHILD arguments
1136  *
1137  * @param[in] ctx to allocate nodes in. Note: All nodes will be
1138  * allocated in the same ctx. This is to allow
1139  * manipulation by xlat instantiation functions
1140  * later.
1141  * @param[out] head the head of the xlat list / tree structure.
1142  * @param[out] flags Populated with parameters that control xlat
1143  * evaluation and multi-pass parsing.
1144  * @param[in] in the format string to expand.
1145  * @param[in] p_rules controlling how to parse the string outside of
1146  * any expansions.
1147  * @param[in] t_rules controlling how attribute references are parsed.
1148  * @return
1149  * - <=0 on error.
1150  * - >0 on success which is the number of characters parsed.
1151  */
1152 ssize_t xlat_tokenize_argv(TALLOC_CTX *ctx, xlat_exp_t **head, xlat_flags_t *flags, fr_sbuff_t *in,
1153  fr_sbuff_parse_rules_t const *p_rules, tmpl_rules_t const *t_rules)
1154 {
1155  fr_sbuff_t our_in = FR_SBUFF_NO_ADVANCE(in);
1156  ssize_t slen;
1157  fr_cursor_t cursor;
1158  fr_sbuff_marker_t m;
1159  fr_sbuff_parse_rules_t const *our_p_rules; /* Bareword parse rules */
1160  fr_sbuff_parse_rules_t tmp_p_rules;
1161  xlat_flags_t tmp_flags = {};
1162 
1163  if (!flags) flags = &tmp_flags;
1164 
1165  *head = NULL;
1166 
1167  if (p_rules && p_rules->terminals) {
1168  tmp_p_rules = (fr_sbuff_parse_rules_t){ /* Stack allocated due to CL scope */
1169  .terminals = fr_sbuff_terminals_amerge(NULL, p_rules->terminals,
1170  tmpl_parse_rules_bareword_quoted.terminals),
1171  .escapes = tmpl_parse_rules_bareword_quoted.escapes
1172  };
1173  our_p_rules = &tmp_p_rules;
1174  } else {
1175  our_p_rules = &tmpl_parse_rules_bareword_quoted;
1176  }
1177 
1178  /*
1179  * skip spaces at the beginning as we
1180  * don't want them to become a whitespace
1181  * literal.
1182  */
1183  fr_sbuff_adv_past_whitespace(in, SIZE_MAX);
1184  fr_sbuff_marker(&m, &our_in);
1185 
1186  fr_cursor_init(&cursor, head);
1187  while (fr_sbuff_extend(&our_in)) {
1188  xlat_exp_t *node = NULL;
1189  fr_token_t quote;
1190  char *fmt;
1191  size_t len;
1192 
1193  fr_sbuff_set(&m, &our_in); /* Record start of argument */
1194 
1195  fr_sbuff_out_by_longest_prefix(&slen, &quote, xlat_quote_table, &our_in, T_BARE_WORD);
1196 
1197  /*
1198  * Alloc a new node to hold the child nodes
1199  * that make up the argument.
1200  */
1201  node = xlat_exp_alloc_null(ctx);
1202  xlat_exp_set_type(node, XLAT_GROUP);
1203  node->quote = quote;
1204 
1205  switch (quote) {
1206  /*
1207  * Barewords --may-contain=%{expansions}
1208  */
1209  case T_BARE_WORD:
1210  if (xlat_tokenize_literal(node, &node->child, &node->flags, &our_in,
1211  false, our_p_rules, t_rules) < 0) {
1212  error:
1213  if (our_p_rules != &tmpl_parse_rules_bareword_quoted) {
1214  talloc_const_free(our_p_rules->terminals);
1215  }
1216  talloc_free(node);
1217  fr_cursor_head(&cursor);
1218  fr_cursor_free_list(&cursor);
1219 
1220  return -fr_sbuff_used(&our_in); /* error */
1221  }
1222  xlat_flags_merge(flags, &node->flags);
1223  break;
1224 
1225  /*
1226  * "Double quoted strings may contain %{expansions}"
1227  */
1228  case T_DOUBLE_QUOTED_STRING:
1229  if (xlat_tokenize_literal(node, &node->child, &node->flags, &our_in,
1230  false, &tmpl_parse_rules_double_quoted, t_rules) < 0) goto error;
1231  xlat_flags_merge(flags, &node->flags);
1232  break;
1233 
1234  /*
1235  * 'Single quoted strings get parsed as literals'
1236  */
1237  case T_SINGLE_QUOTED_STRING:
1238  {
1239  char *str;
1240 
1241  node->child = xlat_exp_alloc_null(node);
1242  xlat_exp_set_type(node->child, XLAT_LITERAL);
1243  node->flags.needs_async = false; /* Literals are always needs_async */
1244 
1245  slen = fr_sbuff_out_aunescape_until(node->child, &str, &our_in, SIZE_MAX,
1246  tmpl_parse_rules_single_quoted.terminals,
1247  tmpl_parse_rules_single_quoted.escapes);
1248  if (slen < 0) goto error;
1249  xlat_exp_set_name_buffer_shallow(node->child, str);
1250  xlat_flags_merge(flags, &node->flags);
1251  }
1252  break;
1253 
1254  /*
1255  * `back quoted strings aren't supported`
1256  */
1257  case T_BACK_QUOTED_STRING:
1258  fr_strerror_printf("Unexpected `...` string");
1259  goto error;
1260 
1261  default:
1262  fr_assert(0);
1263  break;
1264  }
1265 
1266  if ((quote != T_BARE_WORD) && !fr_sbuff_next_if_char(&our_in, fr_token_quote[quote])) { /* Quoting */
1267  fr_strerror_printf("Unterminated string");
1268  fr_sbuff_set(&our_in, &m);
1269  goto error;
1270  }
1271 
1272  fmt = talloc_bstrndup(node, fr_sbuff_current(&m), fr_sbuff_behind(&m));
1273  xlat_exp_set_name_buffer_shallow(node, fmt);
1274 
1275  xlat_flags_merge(flags, &node->flags);
1276  fr_cursor_insert(&cursor, node);
1277  node = NULL;
1278 
1279  /*
1280  * If we're not and the end of the string
1281  * and there's no whitespace between tokens
