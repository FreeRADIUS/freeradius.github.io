45 /** "server local" time. This is the time in nanoseconds since the application started.
46  *
47  * This time is our *private* view of time. It should only be used
48  * for internal timers, events, etc. It can skew randomly as NTP
49  * plays with the local clock.
50  */
51 typedef int64_t fr_time_t;
52 
53 /** "Unix" time. This is the time in nanoseconds since midnight January 1, 1970
54  *
55  * Note that it is *unsigned*, as we don't use dates before 1970. Having it
56  * unsigned also allows the compiler to catch issues where people confuse the
57  * two types of time.
58  *
59  * The unix times are *public* times. i.e. times that we get from
60  * the network, or send to the network. We have no idea if the other
61  * parties idea of time is correct (or if ours is wrong), so we don't
62  * mangle unix time based on clock skew.
63  */
64 typedef uint64_t fr_unix_time_t;
65 
66 /** A time delta, a difference in time measured in nanoseconds.
67  *
68  * This is easier to distinguish where server epoch time is being
69  * used, and where relative time is being used.
70  */
71 typedef int64_t fr_time_delta_t;
