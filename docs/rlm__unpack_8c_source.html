52 /** Unpack data
53  *
54  * Example:
55 @verbatim
56 %{unpack:&Class 0 integer}
57 @endverbatim
58  * Expands Class, treating octet at offset 0 (bytes 0-3) as an "integer".
59  *
60  * @ingroup xlat_functions
61  */
62 static ssize_t unpack_xlat(UNUSED TALLOC_CTX *ctx, char **out, size_t outlen,
63  UNUSED void const *mod_inst, UNUSED void const *xlat_inst,
64  request_t *request, char const *fmt)
65 {
66  char *data_name, *data_size, *data_type;
67  char *p;
68  size_t len, input_len;
69  ssize_t slen;
70  int offset;
71  fr_type_t type;
72  fr_dict_attr_t const *da;
73  fr_pair_t *vp, *cast;
74  uint8_t const *input;
75  char buffer[256];
76  uint8_t blob[256];
77 
78  /*
79  * FIXME: copy only the fields here, as we parse them.
80  */
81  strlcpy(buffer, fmt, sizeof(buffer));
82 
83  p = buffer;
84  fr_skip_whitespace(p); /* skip leading spaces */
85 
86  data_name = p;
87 
88  fr_skip_not_whitespace(p);
89 
90  if (!*p) {
91  error:
92  REDEBUG("Format string should be '<data> <offset> <type>' e.g. '&Class 1 integer'");
93  nothing:
94  return -1;
95  }
96 
97  fr_zero_whitespace(p);
98  if (!*p) GOTO_ERROR;
99 
100  data_size = p;
101 
102  fr_skip_not_whitespace(p);
103  if (!*p) GOTO_ERROR;
104 
105  fr_zero_whitespace(p);
106  if (!*p) GOTO_ERROR;
107 
108  data_type = p;
109 
110  fr_skip_not_whitespace(p);
111  if (*p) GOTO_ERROR; /* anything after the type is an error */
112 
113  /*
114  * Attribute reference
115  */
116  if (*data_name == '&') {
117  if (xlat_fmt_get_vp(&vp, request, data_name) < 0) goto nothing;
118 
119  if ((vp->vp_type != FR_TYPE_OCTETS) &&
120  (vp->vp_type != FR_TYPE_STRING)) {
121  REDEBUG("unpack requires the input attribute to be 'string' or 'octets'");
122  goto nothing;
123  }
124  input = vp->vp_octets;
125  input_len = vp->vp_length;
126 
127  } else if ((data_name[0] == '0') && (data_name[1] == 'x')) {
128  /*
129  * Hex data.
130  */
131  len = strlen(data_name + 2);
132  if (len > 0) {
133  fr_sbuff_parse_error_t err;
134 
135  input = blob;
136  input_len = fr_hex2bin(&err, &FR_DBUFF_TMP(blob, sizeof(blob)),
137  &FR_SBUFF_IN(data_name + 2, len), true);
138  if (err) {
139  REDEBUG("Invalid hex string in '%s'", data_name);
140  goto nothing;
141  }
142  } else {
143  input = blob;
144  input_len = 0;
145  }
146  } else {
147  GOTO_ERROR;
148  }
149 
150  offset = (int) strtoul(data_size, &p, 10);
151  if (*p) {
152  REDEBUG("unpack requires a float64 number, not '%s'", data_size);
153  goto nothing;
154  }
155 
156  type = fr_table_value_by_str(fr_value_box_type_table, data_type, FR_TYPE_INVALID);
157  if (type == FR_TYPE_INVALID) {
158  REDEBUG("Invalid data type '%s'", data_type);
159  goto nothing;
160  }
161 
162  /*
163  * Output must be a non-zero limited size.
164  */
165  if ((dict_attr_sizes[type][0] == 0) ||
166  (dict_attr_sizes[type][0] != dict_attr_sizes[type][1])) {
167  REDEBUG("unpack requires fixed-size output type, not '%s'", data_type);
168  goto nothing;
169  }
170 
171  if (input_len < (offset + dict_attr_sizes[type][0])) {
172  REDEBUG("Insufficient data to unpack '%s' from '%s'", data_type, data_name);
173  goto nothing;
174  }
175 
176  da = fr_dict_attr_child_by_num(fr_dict_root(dict_freeradius), attr_cast_base->attr + type);
177  if (!da) {
178  REDEBUG("Cannot decode type '%s'", data_type);
179  goto nothing;
180  }
181 
182  MEM(cast = fr_pair_afrom_da(request, da));
183 
184  memcpy(&(cast->data), input + offset, dict_attr_sizes[type][0]);
185 
186  /*
187  * Hacks
188  */
189  switch (type) {
190  case FR_TYPE_INT32:
191  case FR_TYPE_UINT32:
192  cast->vp_uint32 = ntohl(cast->vp_uint32);
193  break;
194 
195  case FR_TYPE_UINT16:
196  cast->vp_uint16 = ((input[offset] << 8) | input[offset + 1]);
197  break;
198 
199  case FR_TYPE_UINT64:
200  cast->vp_uint64 = ntohll(cast->vp_uint64);
201  break;
202 
203  case FR_TYPE_DATE:
204  cast->vp_date = fr_time_from_timeval(&(struct timeval) {.tv_sec = ntohl(cast->vp_uint32)});
205  break;
206 
207  default:
208  break;
209  }
210 
211  slen = fr_pair_print_value_quoted(&FR_SBUFF_OUT(*out, outlen), cast, T_BARE_WORD);
212  talloc_free(cast);
213  if (slen < 0) {
214  REDEBUG("Insufficient buffer space to unpack data");
215  goto nothing;
216  }
217 
218  return slen;
219 }
