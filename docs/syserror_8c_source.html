165 /** Guaranteed to be thread-safe version of strerror
166  *
167  * @param num errno as returned by function or from global errno.
168  * @return local specific error string relating to errno.
169  *
170  * @hidecallergraph
171  */
172 char const *fr_syserror(int num)
173 {
174  char *buffer, *p, *end;
175 
176  buffer = fr_syserror_buffer;
177  if (!buffer) {
178  /*
179  * Try and produce something useful,
180  * even if the thread is exiting.
181  */
182  if (logging_stop) {
183  if (HAVE_DEFINITION(num)) return fr_syserror_macro_names[num];
184  return "";
185  }
186 
187  buffer = talloc_array(NULL, char, FR_SYSERROR_BUFSIZE);
188  if (!buffer) {
189  fr_perror("Failed allocating memory for system error buffer");
190  return NULL;
191  }
192  fr_thread_local_set_destructor(fr_syserror_buffer, _fr_logging_free, buffer);
193  }
194 
195  if (num == 0) return "No additional error information";
196 
197  p = buffer;
198  end = p + FR_SYSERROR_BUFSIZE;
199 
200  /*
201  * Prefix system errors with the macro name and number
202  * if we're debugging.
203  */
204  if (HAVE_DEFINITION(num)) {
205  p += snprintf(p, end - p, "%s: ", fr_syserror_macro_names[num]);
206  } else {
207  p += snprintf(p, end - p, "errno %i: ", num);
208  }
209  if (p >= end) return p;
210 
211  /*
212  * XSI-Compliant version
213  */
214 #if !defined(HAVE_FEATURES_H) || !defined(__GLIBC__) || ((_POSIX_C_SOURCE >= 200112L || _XOPEN_SOURCE >= 500) && ! _GNU_SOURCE)
215  {
216  int ret;
217 
218  ret = strerror_r(num, p, end - p);
219  if (ret != 0) {
220 # ifndef NDEBUG
221  fprintf(stderr, "strerror_r() failed to write error for errno %i to buffer %p (%zu bytes), "
222  "returned %i: %s\n", num, buffer, (size_t)FR_SYSERROR_BUFSIZE, ret, strerror(ret));
223 # endif
224  buffer[0] = '\0';
225  }
226  }
227  return buffer;
228  /*
229  * GNU Specific version
230  *
231  * The GNU Specific version returns a char pointer. That pointer may point
232  * the buffer you just passed in, or to an immutable static string.
233  */
234 #else
235  {
236  char *q;
237 
238  q = strerror_r(num, p, end - p);
239  if (!q) {
240 # ifndef NDEBUG
241  fprintf(stderr, "strerror_r() failed to write error for errno %i to buffer %p "
242  "(%zu bytes): %s\n", num, buffer, (size_t)FR_SYSERROR_BUFSIZE, strerror(errno));
243 # endif
244  buffer[0] = '\0';
245  return buffer;
246  }
247 
248  /*
249  * If strerror_r used a static string, copy it to the buffer
250  */
251  if (q != p) {
252  size_t len;
253 
254  len = strlen(q) + 1;
255  if (len >= (size_t)(end - p)) len = end - p;
256 
257  strlcpy(p, q, len);
258  }
259 
260  return buffer;
261  }
262 #endif
263 }
