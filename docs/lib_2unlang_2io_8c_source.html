30 /** Run the interpreter after creating a subrequest.
31  *
32  * Just run some "unlang", but don't do anything else.
33  *
34  * This is a shim function added to 'fake' requests by the subrequest and parallel keywords.
35  */
36 unlang_action_t unlang_io_process_interpret(rlm_rcode_t *p_result, UNUSED module_ctx_t const *mctx, request_t *request)
37 {
38  rlm_rcode_t rcode;
39 
40  REQUEST_VERIFY(request);
41 
42  rcode = unlang_interpret(request);
43 
44  /*
45  * We've yielded, and can keep running. Do so.
46  */
47  if ((rcode == RLM_MODULE_YIELD) &&
48  (request->master_state != REQUEST_STOP_PROCESSING)) {
49  *p_result = RLM_MODULE_YIELD;
50  return UNLANG_ACTION_YIELD;
51  }
52 
53  /*
54  * Either we're done naturally, or we're forcibly done. Stop.
55  *
56  * If we have a parent, then we're running synchronously
57  * with it. Allow the parent to resume.
58  */
59  if (request->parent) unlang_interpret_resumable(request->parent);
60 
61  /*
62  * Don't bother setting request->reply->code.
63  */
64  RETURN_MODULE_HANDLED;
65 }

67 /** Allocate a child request based on the parent.
68  *
69  * @param[in] parent spawning the child request.
70  * @param[in] namespace the child request operates in. If NULL the parent's namespace is used.
71  * @param[in] detachable Allow/disallow the child to be detached.
72  * @return
73  * - The new child request.
74  * - NULL on error.
75  */
76 request_t *unlang_io_subrequest_alloc(request_t *parent, fr_dict_t const *namespace, bool detachable)
77 {
78  request_t *child;
79 
80  if (!detachable) {
81  child = request_alloc_fake(parent, namespace);
82  } else {
83  child = request_alloc_detachable(parent, namespace);
84  }
85  if (!child) return NULL;
86 
87  /*
88  * Push the child, and set it's top frame to be true.
89  */
90 
91  child->log.unlang_indent = parent->log.unlang_indent;
92 
93  /*
94  * Initialize some basic information for the child.
95  *
96  * Note that we do NOT initialize child->backlog, as the
97  * child is never resumable... the parent is resumable.
98  */
99  child->number = parent->number;
100  child->el = parent->el;
101  child->server_cs = parent->server_cs;
102  child->backlog = parent->backlog;
103 
104  /*
105  * Initialize all of the async fields.
106  */
107  child->async = talloc_zero(child, fr_async_t);
108 
109 #define COPY_FIELD(_x) child->async->_x = parent->async->_x
110  COPY_FIELD(listen);
111  COPY_FIELD(recv_time);
112  child->async->fake = true;
113 
114  /*
115  * Always set the "process" function to the local
116  * bare-bones function which just runs on section of
117  * "unlang", and doesn't send replies or anything else.
118  */
119  child->async->process = unlang_io_process_interpret;
120 
121  return child;
122 }
