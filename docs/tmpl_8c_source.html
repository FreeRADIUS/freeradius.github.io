39 /*
40  * Clean up everything except the waitpid handler.
41  *
42  * If there is a waitpid handler, then this cleanup function MUST
43  * be called after setting the handler.
44  */
45 static void unlang_tmpl_exec_cleanup(request_t *request)
46 {
47  unlang_stack_t *stack = request->stack;
48  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
49  unlang_frame_state_tmpl_t *state = talloc_get_type_abort(frame->state,
50  unlang_frame_state_tmpl_t);
51 
52  if (state->fd >= 0) {
53  (void) fr_event_fd_delete(request->el, state->fd, FR_EVENT_FILTER_IO);
54  close(state->fd);
55  state->fd = -1;
56  }
57 
58  /*
59  * It still hasn't exited. Tell the event loop that we
60  * need to wait as long as necessary for the PID to exit,
61  * and that we don't care about the exit status.
62  */
63  if (state->pid) {
64  (void) fr_event_pid_wait(request->el, request->el, NULL, state->pid, NULL, NULL);
65  state->pid = 0;
66  }
67 
68  if (state->ev) fr_event_timer_delete(&state->ev);
69 }

71 /** Send a signal (usually stop) to a request
72  *
73  * This is typically called via an "async" action, i.e. an action
74  * outside of the normal processing of the request.
75  *
76  * If there is no #fr_unlang_tmpl_signal_t callback defined, the action is ignored.
77  *
78  * @param[in] request The current request.
79  * @param[in] action to signal.
80  */
81 static void unlang_tmpl_signal(request_t *request, fr_state_signal_t action)
82 {
83  unlang_stack_t *stack = request->stack;
84  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
85  unlang_frame_state_tmpl_t *state = talloc_get_type_abort(frame->state,
86  unlang_frame_state_tmpl_t);
87 
88  if (!state->signal) return;
89 
90  state->signal(request, state->rctx, action);
91 
92  /*
93  * If we're cancelled, then kill any child processes, and
94  * ignore future signals.
95  */
96  if (action == FR_SIGNAL_CANCEL) {
97  if (state->pid > 0) kill(state->pid, SIGKILL);
98  state->failed = true;
99 
100  unlang_tmpl_exec_cleanup(request);
101  state->signal = NULL;
102  }
103 }

105 /** Push a tmpl onto the stack for evaluation
106  *
107  * @param[in] ctx To allocate value boxes and values in.
108  * @param[out] out The value_box created from the tmpl. May be NULL,
109  * in which case the result is discarded.
110  * @param[in] request The current request.
111  * @param[in] tmpl the tmpl to expand
112  * @param[in] vps the input VPs. May be NULL. Used only for #TMPL_TYPE_EXEC
113  * @param[out] status where the status of exited programs will be stored.
114  */
115 int unlang_tmpl_push(TALLOC_CTX *ctx, fr_value_box_t **out, request_t *request, tmpl_t const *tmpl, fr_pair_t *vps, int *status)
116 {
117  unlang_stack_t *stack = request->stack;
118  unlang_stack_frame_t *frame;
119  unlang_frame_state_tmpl_t *state;
120 
121  unlang_tmpl_t *ut;
122 
123  static unlang_t tmpl_instruction = {
124  .type = UNLANG_TYPE_TMPL,
125  .name = "tmpl",
126  .debug_name = "tmpl",
127  .actions = {
128  [RLM_MODULE_REJECT] = 0,
129  [RLM_MODULE_FAIL] = 0,
130  [RLM_MODULE_OK] = 0,
131  [RLM_MODULE_HANDLED] = 0,
132  [RLM_MODULE_INVALID] = 0,
133  [RLM_MODULE_DISALLOW] = 0,
134  [RLM_MODULE_NOTFOUND] = 0,
135  [RLM_MODULE_NOOP] = 0,
136  [RLM_MODULE_UPDATED] = 0
137  },
138  };
139 
140  MEM(ut = talloc(stack, unlang_tmpl_t));
141  ut->self = tmpl_instruction;
142  ut->tmpl = tmpl;
143 
144  /*
145  * Push a new tmpl frame onto the stack
146  */
147  if (unlang_interpret_push(request, unlang_tmpl_to_generic(ut),
148  RLM_MODULE_UNKNOWN, UNLANG_NEXT_STOP, false) < 0) return -1;
149 
150  frame = &stack->frame[stack->depth];
151  state = talloc_get_type_abort(frame->state, unlang_frame_state_tmpl_t);
152 
153  *state = (unlang_frame_state_tmpl_t) {
154  .out = out,
155  .ctx = ctx,
156  .vps = vps,
157  .status_p = status
158  };
159 
160  return 0;
161 }

166 static void unlang_tmpl_exec_waitpid(UNUSED fr_event_list_t *el, UNUSED pid_t pid, int status, void *uctx)
167 {
168  request_t *request = uctx;
169  unlang_stack_t *stack = request->stack;
170  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
171  unlang_frame_state_tmpl_t *state = talloc_get_type_abort(frame->state,
172  unlang_frame_state_tmpl_t);
173 
174  state->status = status;
175  state->pid = 0;
176 
177  /*
178  * We may receive the "child exited" signal before the
179  * "pipe has been closed" signal.
180  */
181  if (state->fd >= 0) {
182  (void) fr_event_fd_delete(request->el, state->fd, FR_EVENT_FILTER_IO);
183  close(state->fd);
184  state->fd = -1;
185  }
186 
187  if (state->ev) fr_event_timer_delete(&state->ev);
188 
189  unlang_interpret_resumable(request);
190 }

192 static void unlang_tmpl_exec_read(UNUSED fr_event_list_t *el, UNUSED int fd, UNUSED int flags, void *uctx)
193 {
194  request_t *request = uctx;
195  unlang_stack_t *stack = request->stack;
196  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
197  unlang_frame_state_tmpl_t *state = talloc_get_type_abort(frame->state,
198  unlang_frame_state_tmpl_t);
199  ssize_t data_len;
200  char *p, *end;
201 
202  p = state->ptr;
203  end = state->buffer + talloc_array_length(state->buffer);
204 
205  data_len = read(state->fd, p, end - p);
206  if (data_len < 0) {
207  if (errno == EINTR) return;
208 
209  REDEBUG("Error reading from child program - %s", fr_syserror(errno));
210  unlang_tmpl_exec_cleanup(request);
211  unlang_interpret_resumable(request);
212  return;
213  }
214 
215  p += data_len;
216 
217  /*
218  * Done reading, close the pipe. We still wait for the
219  * caller to exit, so that we can get it's exit code.
220  */
221  if (data_len == 0) goto close_pipe;
222 
223  /*
224  * Ran out of buffer space. Kill the process and discard
225  * all output. We destroy the output so that we don't
226  * return truncated data to the caller. We also destroy
227  * the timer which waits for the exit code. We no longer
228  * care about it.
229  */
230  if (p >= end) {
231  fr_assert(state->ev != NULL);
232  fr_event_timer_delete(&state->ev);
233  state->ptr = state->buffer;
234 
235  REDEBUG("Too much output from program - killing it and failing the request");
236  kill(state->pid, SIGKILL);
237  state->failed = true;
238 
239  close_pipe:
240  (void) fr_event_fd_delete(request->el, state->fd, FR_EVENT_FILTER_IO);
241  close(state->fd);
242  state->fd = -1;
243  return;
244  }
245 
246  fr_assert(p < end);
247  state->ptr = p;
248 }

250 static void unlang_tmpl_exec_timeout(UNUSED fr_event_list_t *el, UNUSED fr_time_t now, void *uctx)
251 {
252  request_t *request = uctx;
253  unlang_stack_t *stack = request->stack;
254  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
255  unlang_frame_state_tmpl_t *state = talloc_get_type_abort(frame->state,
256  unlang_frame_state_tmpl_t);
257 
258  fr_assert(state->pid > 0);
259 
260 #ifdef __linux__
261  int status;
262 
263  /*
264  * libkqueue on Linux isn't quite there yet. Maybe the
265  * program has exited, and we haven't noticed. In which
266  * case, do a graceful cleanup.
267  */
268  if (waitpid(state->pid, &status, WNOHANG) == state->pid) {
269  unlang_tmpl_exec_waitpid(el, state->pid, status, request);
270  return;
271  }
272 #endif
273 
274  if (state->fd < 0) {
275  REDEBUG("Timeout waiting for program to exit - killing it and failing the request");
276  } else {
277  REDEBUG("Timeout running program - killing it and failing the request");
278  }
279  kill(state->pid, SIGKILL);
280  state->failed = true;
281 
282  unlang_tmpl_exec_cleanup(request);
283  unlang_interpret_resumable(request);
284 }

287 /** Wrapper to call a resumption function after a tmpl has been expanded
288  *
289  * If the resumption function returns YIELD, then this function is
290  * called repeatedly until the resumption function returns a final
291  * value.
292  */
293 static unlang_action_t unlang_tmpl_resume(rlm_rcode_t *p_result, request_t *request)
294 {
295  unlang_stack_t *stack = request->stack;
296  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
297  unlang_frame_state_tmpl_t *state = talloc_get_type_abort(frame->state,
298  unlang_frame_state_tmpl_t);
299 
300  if (state->out) *state->out = state->box;
301 
302  if (state->resume) {
303  rlm_rcode_t rcode;
304 
305  rcode = state->resume(request, state->rctx);
306  *p_result = rcode;
307  if (rcode == RLM_MODULE_YIELD) return UNLANG_ACTION_YIELD;
308  } else {
309  *p_result = RLM_MODULE_OK;
310  }
311 
312  return UNLANG_ACTION_CALCULATE_RESULT;
313 }

317 /** Wrapper to call exec after the program has finished executing
318  *
319  */
320 static unlang_action_t unlang_tmpl_exec_wait_final(rlm_rcode_t *p_result, request_t *request)
321 {
322  unlang_stack_t *stack = request->stack;
323  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
324  unlang_frame_state_tmpl_t *state = talloc_get_type_abort(frame->state,
325  unlang_frame_state_tmpl_t);
326 
327  unlang_tmpl_exec_cleanup(request);
328 
329  /*
330  * The exec failed for some internal reason. We don't
331  * care about output, and we don't care about the programs exit status.
332  */
333  if (state->failed) {
334  TALLOC_FREE(state->box);
335  goto resume;
336  }
337 
338  fr_assert(state->pid == 0);
339 
340  if (state->status != 0) {
341  if (WIFEXITED(state->status)) {
342  RDEBUG("Program failed with status code %d", WEXITSTATUS(state->status));
343  state->status = WEXITSTATUS(state->status);
344 
345  } else if (WIFSIGNALED(state->status)) {
346  RDEBUG("Program exited due to signal with status code %d", WTERMSIG(state->status));
347  state->status = -WTERMSIG(state->status);
348 
349  } else {
350  RDEBUG("Program exited due to unknown status %d", state->status);
351  state->status = -state->status;
352  }
353 
354  fr_assert(state->box == NULL);
355  goto resume;
356  }
357 
358  /*
359  * Save the *mangled* exit status, not the raw one.
360  */
361  if (state->status_p) *state->status_p = state->status;
362 
363  /*
364  * We might want to just get the status of the program,
365  * and not care about the output.
366  *
367  * If we do care about the output, it's unquoted, and tainted.
368  */
369  if (state->out) {
370  fr_type_t type = FR_TYPE_STRING;
371 
372  /*
373  * Remove any trailing LF / CR
374  */
375  if (state->ptr > state->buffer) {
376  char *p = state->ptr;
377 
378  while (p > state->buffer) {
379  if (p[-1] < ' ') {
380  p--;
381  continue;
382  }
383 
384  break;
385  }
386  state->ptr = p;
387  }
388 
389  MEM(state->box = fr_value_box_alloc(state->ctx, FR_TYPE_STRING, NULL, true));
390  if (fr_value_box_from_str(state->box, state->box, &type, NULL,
391  state->buffer, state->ptr - state->buffer, 0, true) < 0) {
392  talloc_free(state->box);
393  *p_result = RLM_MODULE_FAIL;
394  return UNLANG_ACTION_CALCULATE_RESULT;
395  }
396  }
397 
398  /*
399  * Ensure that the callers resume function is called.
400  */
401 resume:
402  frame->process = unlang_tmpl_resume;
403  return unlang_tmpl_resume(p_result, request);
404 }

407 /** Wrapper to call exec after a tmpl has been expanded
408  *
409  */
410 static unlang_action_t unlang_tmpl_exec_wait_resume(rlm_rcode_t *p_result, request_t *request)
411 {
412  unlang_stack_t *stack = request->stack;
413  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
414  unlang_frame_state_tmpl_t *state = talloc_get_type_abort(frame->state,
415  unlang_frame_state_tmpl_t);
416  pid_t pid;
417  int *fd_p = NULL;
418 
419  /*
420  * @todo - if there's no state->out, then we don't need
421  * to have an FD, and we don't need to have a buffer.
422  */
423  state->fd = -1;
424  if (state->out) fd_p = &state->fd;
425 
426  if (fr_exec_wait_start(request, state->box, state->vps, &pid, NULL, fd_p) < 0) {
427  RPEDEBUG("Failed executing program");
428  fail:
429  *p_result = RLM_MODULE_FAIL;
430  return UNLANG_ACTION_CALCULATE_RESULT;
431  }
432 
433  TALLOC_FREE(state->box); /* this is the xlat expansion, and not the output string we want */
434 
435  state->pid = pid;
436  state->status = -1; /* default to program didn't work */
437 
438  /*
439  * Tell the event loop that it needs to wait for this PID.
440  */
441  if (fr_event_pid_wait(state, request->el, &state->ev_pid, pid,
442  unlang_tmpl_exec_waitpid, request) < 0) {
443  state->pid = 0;
444  RPEDEBUG("Failed adding watcher for child process");
445  unlang_tmpl_exec_cleanup(request);
446  goto fail;
447  }
448 
449  /*
450  * Kill the child process after a period of time.
451  *
452  * @todo - make the timeout configurable
453  */
454  if (fr_event_timer_in(state->ctx, request->el, &state->ev, fr_time_delta_from_sec(10), unlang_tmpl_exec_timeout, request) < 0) {
455  unlang_tmpl_exec_cleanup(request);
456  goto fail;
457  }
458 
459  /*
460  * If the caller doesn't want the output box, we can just skip reading from the FD.
461  */
462  if (state->fd >= 0) {
463  if (fr_event_fd_insert(state->ctx, request->el, state->fd, unlang_tmpl_exec_read, NULL, NULL, request) < 0) {
464  RPEDEBUG("Failed adding event");
465  unlang_tmpl_exec_cleanup(request);
466  goto fail;
467  }
468 
469  MEM(state->buffer = talloc_array(state, char, 1024));
470  state->ptr = state->buffer;
471  }
472 
473  frame->process = unlang_tmpl_exec_wait_final;
474 
475  *p_result = RLM_MODULE_YIELD;
476  return UNLANG_ACTION_YIELD;
477 }

506 static unlang_action_t unlang_tmpl(rlm_rcode_t *p_result, request_t *request)
507 {
508  unlang_stack_t *stack = request->stack;
509  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
510  unlang_frame_state_tmpl_t *state = talloc_get_type_abort(frame->state,
511  unlang_frame_state_tmpl_t);
512  unlang_tmpl_t *ut = unlang_generic_to_tmpl(frame->instruction);
513  xlat_exp_t const *xlat;
514 
515  /*
516  * If we're not called from unlang_tmpl_push(), then
517  * ensure that we clean up the resulting value boxes.
518  */
519  if (!state->ctx) state->ctx = state;
520 
521  if (!tmpl_async_required(ut->tmpl)) {
522  if (!ut->inline_exec) {
523  if (tmpl_aexpand_type(state->ctx, &state->box, FR_TYPE_STRING, request, ut->tmpl, NULL, NULL) < 0) {
524  RPEDEBUG("Failed expanding %s", ut->tmpl->name);
525  *p_result = RLM_MODULE_FAIL;
526  }
527 
528  *p_result = RLM_MODULE_OK;
529  return UNLANG_ACTION_CALCULATE_RESULT;
530  }
531 
532  /*
533  * Inline exec's are only called from in-line
534  * text in the configuration files.
535  */
536  frame->process = unlang_tmpl_exec_nowait_resume;
537 
538  repeatable_set(frame);
539  if (unlang_xlat_push(state->ctx, &state->box, request, tmpl_xlat(ut->tmpl), false) < 0) {
540  *p_result = RLM_MODULE_FAIL;
541  return UNLANG_ACTION_STOP_PROCESSING;
542  }
543  return UNLANG_ACTION_PUSHED_CHILD;
544  }
545 
546  /*
547  * XLAT structs are allowed.
548  */
549  if (ut->tmpl->type == TMPL_TYPE_XLAT) {
550  frame->process = unlang_tmpl_resume;
551  repeatable_set(frame);
552  if (unlang_xlat_push(state->ctx, &state->box, request, tmpl_xlat(ut->tmpl), false) < 0) {
553  *p_result = RLM_MODULE_FAIL;
554  return UNLANG_ACTION_STOP_PROCESSING;
555  }
556  return UNLANG_ACTION_PUSHED_CHILD;
557  }
558 
559  if (ut->tmpl->type == TMPL_TYPE_XLAT_UNRESOLVED) {
560  REDEBUG("Xlat expansions MUST be fully resolved before being run asynchronously");
561  *p_result = RLM_MODULE_FAIL;
562  return UNLANG_ACTION_CALCULATE_RESULT;
563  }
564 
565  /*
566  * Attribute expansions, etc. don't require YIELD.
567  */
568  if (ut->tmpl->type != TMPL_TYPE_EXEC) {
569  REDEBUG("Internal error - template '%s' should not require async", ut->tmpl->name);
570  *p_result = RLM_MODULE_FAIL;
571  return UNLANG_ACTION_CALCULATE_RESULT;
572  }
573 
574  xlat = tmpl_xlat(ut->tmpl);
575  fr_assert(xlat);
576 
577  /*
578  * Expand the arguments to the program we're executing.
579  */
580  frame->process = unlang_tmpl_exec_wait_resume;
581  repeatable_set(frame);
582  if (unlang_xlat_push(state->ctx, &state->box, request, xlat, false) < 0) {
583  *p_result = RLM_MODULE_FAIL;
584  return UNLANG_ACTION_STOP_PROCESSING;
585  }
586 
587  return UNLANG_ACTION_PUSHED_CHILD;
588 }
