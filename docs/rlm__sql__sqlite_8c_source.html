69 /** Convert an sqlite status code to an sql_rcode_t
70  *
71  * @param status to convert.
72  * @return
73  * - RLM_SQL_OK - If no errors found.
74  * - RLM_SQL_ERROR - If a known, non-fatal, error occurred.
75  * - RLM_SQL_ALT_QUERY - If a constraints violation occurred.
76  * - RLM_SQL_RECONNECT - Anything else, we assume the connection can no longer be used.
77  */
78 static sql_rcode_t sql_error_to_rcode(int status)
79 {
80  /*
81  * Lowest byte is error category, other byte may contain
82  * the extended error, depending on version.
83  */
84  switch (status & 0xff) {
85  /*
86  * Not errors
87  */
88  case SQLITE_OK:
89  case SQLITE_DONE:
90  case SQLITE_ROW:
91  return RLM_SQL_OK;
92  /*
93  * User/transient errors
94  */
95  case SQLITE_ERROR: /* SQL error or missing database */
96  case SQLITE_FULL:
97  case SQLITE_MISMATCH:
98  case SQLITE_BUSY: /* Can be caused by database locking */
99  return RLM_SQL_ERROR;
100 
101  /*
102  * Constraints violations
103  */
104  case SQLITE_CONSTRAINT:
105  return RLM_SQL_ALT_QUERY;
106 
107  /*
108  * Errors with the handle, that probably require reinitialisation
109  */
110  default:
111  return RLM_SQL_RECONNECT;
112  }
113 }

115 /** Determine if an error occurred, and what type of error it was
116  *
117  * @param db handle to extract error from (may be NULL).
118  * @param status to check (if unused, set to SQLITE_OK).
119  * @return
120  * - RLM_SQL_OK - If no errors found.
121  * - RLM_SQL_ERROR - If a known, non-fatal, error occurred.
122  * - RLM_SQL_ALT_QUERY - If a constraints violation occurred.
123  * - RLM_SQL_RECONNECT - Anything else. We assume the connection can no longer be used.
124  */
125 static sql_rcode_t sql_check_error(sqlite3 *db, int status)
126 {
127  int hstatus = SQLITE_OK;
128 
129  if (db) {
130  hstatus = sqlite3_errcode(db);
131  switch (hstatus & 0xff) {
132  case SQLITE_OK:
133  case SQLITE_DONE:
134  case SQLITE_ROW:
135  hstatus = SQLITE_OK;
136  break;
137 
138  default:
139  break;
140  }
141  }
142 
143  switch (status & 0xff) {
144  case SQLITE_OK:
145  case SQLITE_DONE:
146  case SQLITE_ROW:
147  status = SQLITE_OK;
148  break;
149 
150  default:
151  break;
152  }
153 
154  if (status != SQLITE_OK) return sql_error_to_rcode(status);
155  if (hstatus != SQLITE_OK) return sql_error_to_rcode(status);
156 
157  return RLM_SQL_OK;
158 }

160 /** Print an error to the global debug log
161  *
162  * If status does not indicate success, write an error to the global error log.
163  *
164  * @note The error code will be appended to the fmt string in the format ": code 0x<hex> (<int>)[: <string>]".
165  *
166  * @param db handle to extract error from (may be NULL).
167  * @param status to check (if unused, set to SQLITE_OK).
168  * @param fmt to preprend.
169  * @param ... arguments to fmt.
170  */
171 static void sql_print_error(sqlite3 *db, int status, char const *fmt, ...)
172  CC_HINT(format (printf, 3, 4)) CC_HINT(nonnull (3));
173 static void sql_print_error(sqlite3 *db, int status, char const *fmt, ...)
174 {
175  va_list ap;
176  char *p;
177  int hstatus = SQLITE_OK;
178 
179  if (db) {
180  hstatus = sqlite3_errcode(db);
181  switch (hstatus & 0xff) {
182  case SQLITE_OK:
183  case SQLITE_DONE:
184  case SQLITE_ROW:
185  hstatus = SQLITE_OK;
186  break;
187 
188  default:
189  break;
190  }
191  }
192 
193  switch (status & 0xff) {
194  case SQLITE_OK:
195  case SQLITE_DONE:
196  case SQLITE_ROW:
197  status = SQLITE_OK;
198  break;
199 
200  default:
201  break;
202  }
203 
204  /*
205  * No errors!
206  */
207  if ((hstatus == SQLITE_OK) && (status == SQLITE_OK)) return;
208 
209  /*
210  * At least one error...
211  */
212  va_start(ap, fmt);
213  MEM(p = talloc_vasprintf(NULL, fmt, ap));
214  va_end(ap);
215 
216  /*
217  * Disagreement between handle, and function return code,
218  * print them both.
219  */
220  if ((status != SQLITE_OK) && (status != hstatus)) {
221 #ifdef HAVE_SQLITE3_ERRSTR
222  ERROR("%s: Code 0x%04x (%i): %s", p, status, status, sqlite3_errstr(status));
223 #else
224  ERROR("%s: Code 0x%04x (%i)", p, status, status);
225 #endif
226  }
227 
228  if (hstatus != SQLITE_OK) ERROR("%s: Code 0x%04x (%i): %s",
229  p, hstatus, hstatus, sqlite3_errmsg(db));
230 }

233 static int sql_loadfile(TALLOC_CTX *ctx, sqlite3 *db, char const *filename)
234 {
235  ssize_t len;
236  int statement_cnt = 0;
237  char *buffer;
238  char *p, *q;
239  int cl;
240  FILE *f;
241  struct stat finfo;
242 
243  int status;
244  sqlite3_stmt *statement;
245  char const *z_tail;
246 
247  INFO("Executing SQL statements from file \"%s\"", filename);
248 
249  f = fopen(filename, "r");
250  if (!f) {
251  ERROR("Failed opening SQL file \"%s\": %s", filename,
252  fr_syserror(errno));
253 
254  return -1;
255  }
256 
257  if (fstat(fileno(f), &finfo) < 0) {
258  ERROR("Failed stating SQL file \"%s\": %s", filename,
259  fr_syserror(errno));
260 
261  fclose(f);
262 
263  return -1;
264  }
265 
266  if (finfo.st_size > BOOTSTRAP_MAX) {
267  too_big:
268  ERROR("Size of SQL (%zu) file exceeds limit (%uk)",
269  (size_t) finfo.st_size / 1024, BOOTSTRAP_MAX / 1024);
270 
271  fclose(f);
272 
273  return -1;
274  }
275 
276  MEM(buffer = talloc_array(ctx, char, finfo.st_size + 1));
277  len = fread(buffer, sizeof(char), finfo.st_size, f);
278  if (len > finfo.st_size) {
279  talloc_free(buffer);
280  goto too_big;
281  }
282 
283  if (!len) {
284  if (ferror(f)) {
285  ERROR("Error reading SQL file: %s", fr_syserror(errno));
286 
287  fclose(f);
288  talloc_free(buffer);
289 
290  return -1;
291  }
292 
293  DEBUG("Ignoring empty SQL file");
294 
295  fclose(f);
296  talloc_free(buffer);
297 
298  return 0;
299  }
300 
301  buffer[len] = '\0';
302  fclose(f);
303 
304  /*
305  * Check if input data is UTF-8. Allow CR/LF \t, too.
306  */
307  for (p = buffer; p < (buffer + len); p += cl) {
308  if (*p < ' ') {
309  if ((*p != 0x0a) && (*p != 0x0d) && (*p != '\t')) break;
310  cl = 1;
311  } else {
312  cl = fr_utf8_char((uint8_t *) p, -1);
313  if (!cl) break;
314  }
315  }
316 
317  if ((p - buffer) != len) {
318  ERROR("Bootstrap file contains non-UTF8 char at offset %zu", p - buffer);
319  talloc_free(buffer);
320  return -1;
321  }
322 
323  /*
324  * Statement delimiter is ;\n
325  */
326  p = buffer;
327  while ((q = strchr(p, ';'))) {
328  uint32_t statement_len;
329 
330  if ((q[1] != '\n') && (q[1] != '\0')) {
331  p = q + 1;
332  statement_cnt++;
333  continue;
334  }
335 
336 #ifndef NDEBUG
337  if ((q - p) > (1 << 20)) {
338  ERROR("Failed preparing statement %i: too long!", statement_cnt);
339  talloc_free(buffer);
340  return -1;
341  }
342 #endif
343 
344  statement_len = q - p;
345 
346 #ifdef HAVE_SQLITE3_PREPARE_V2
347  status = sqlite3_prepare_v2(db, p, statement_len, &statement, &z_tail);
348 #else
349  status = sqlite3_prepare(db, p, statement_len, &statement, &z_tail);
350 #endif
351 
352  if (sql_check_error(db, status) != RLM_SQL_OK) {
353  sql_print_error(db, status, "Failed preparing statement %i", statement_cnt);
354  talloc_free(buffer);
355  return -1;
356  }
357 
358  status = sqlite3_step(statement);
359  if (sql_check_error(db, status) != RLM_SQL_OK) {
360  sql_print_error(db, status, "Failed executing statement %i", statement_cnt);
361  sqlite3_finalize(statement);
362  talloc_free(buffer);
363  return -1;
364  }
365 
366  status = sqlite3_finalize(statement);
367  if (sql_check_error(db, status) != RLM_SQL_OK) {
368  sql_print_error(db, status, "Failed finalizing statement %i", statement_cnt);
369  talloc_free(buffer);
370  return -1;
371  }
372 
373  statement_cnt++;
374  p = q + 1;
375  }
376 
377  talloc_free(buffer);
378  return 0;
379 }
380 #endif

397 static void _sql_greatest(sqlite3_context *ctx, int num_values, sqlite3_value **values)
398 {
399  int i;
400  sqlite3_int64 value, max = 0;
401 
402  for (i = 0; i < num_values; i++) {
403  value = sqlite3_value_int64(values[i]);
404  if (value > max) {
405  max = value;
406  }
407  }
408 
409  sqlite3_result_int64(ctx, max);

412 static int CC_HINT(nonnull) sql_socket_init(rlm_sql_handle_t *handle, rlm_sql_config_t *config,
413  UNUSED fr_time_delta_t timeout)
414 {
415  rlm_sql_sqlite_conn_t *conn;
416  rlm_sql_sqlite_t *inst = config->driver;
417 
418  int status;
419 
420  MEM(conn = handle->conn = talloc_zero(handle, rlm_sql_sqlite_conn_t));
421  talloc_set_destructor(conn, _sql_socket_destructor);
422 
423  INFO("Opening SQLite database \"%s\"", inst->filename);
424 #ifdef HAVE_SQLITE3_OPEN_V2
425  status = sqlite3_open_v2(inst->filename, &(conn->db), SQLITE_OPEN_READWRITE | SQLITE_OPEN_NOMUTEX, NULL);
426 #else
427  status = sqlite3_open(inst->filename, &(conn->db));
428 #endif
429 
430  if (!conn->db || (sql_check_error(conn->db, status) != RLM_SQL_OK)) {
431  sql_print_error(conn->db, status, "Error opening SQLite database \"%s\"", inst->filename);
432  return RLM_SQL_ERROR;
433  }
434  status = sqlite3_busy_timeout(conn->db, inst->busy_timeout);
435  if (sql_check_error(conn->db, status) != RLM_SQL_OK) {
436  sql_print_error(conn->db, status, "Error setting busy timeout");
437  return RLM_SQL_ERROR;
438  }
439 
440  /*
441  * Enable extended return codes for extra debugging info.
442  */
443 #ifdef HAVE_SQLITE3_EXTENDED_RESULT_CODES
444  status = sqlite3_extended_result_codes(conn->db, 1);
445  if (sql_check_error(conn->db, status) != RLM_SQL_OK) {
446  sql_print_error(conn->db, status, "Error enabling extended result codes");
447  return RLM_SQL_ERROR;
448  }
449 #endif
450 
451 #ifdef HAVE_SQLITE3_CREATE_FUNCTION_V2
452  status = sqlite3_create_function_v2(conn->db, "GREATEST", -1, SQLITE_ANY, NULL,
453  _sql_greatest, NULL, NULL, NULL);
454 #else
455  status = sqlite3_create_function(conn->db, "GREATEST", -1, SQLITE_ANY, NULL,
456  _sql_greatest, NULL, NULL);
457 #endif
458  if (sql_check_error(conn->db, status) != RLM_SQL_OK) {
459  sql_print_error(conn->db, status, "Failed registering 'GREATEST' sql function");
460  return RLM_SQL_ERROR;
461  }
462 
463  return RLM_SQL_OK;
464 }

531 static sql_rcode_t sql_fetch_row(rlm_sql_row_t *out, rlm_sql_handle_t *handle, rlm_sql_config_t *config)
532 {
533  int status;
534  rlm_sql_sqlite_conn_t *conn = handle->conn;
535 
536  int i = 0;
537 
538  char **row;
539 
540  *out = NULL;
541 
542  TALLOC_FREE(handle->row);
543 
544  /*
545  * Executes the SQLite query and interates over the results
546  */
547  status = sqlite3_step(conn->statement);
548 
549  /*
550  * Error getting next row
551  */
552  if (sql_check_error(conn->db, status) != RLM_SQL_OK) return RLM_SQL_ERROR;
553 
554  /*
555  * No more rows to process (we're done)
556  */
557  if (status == SQLITE_DONE) return RLM_SQL_NO_MORE_ROWS;
558 
559  /*
560  * We only need to do this once per result set, because
561  * the number of columns won't change.
562  */
563  if (conn->col_count == 0) {
564  conn->col_count = sql_num_fields(handle, config);
565  if (conn->col_count == 0) return RLM_SQL_ERROR;
566  }
567 
568  /*
569  * Free the previous result (also gets called on finish_query)
570  */
571  MEM(row = handle->row = talloc_zero_array(handle->conn, char *, conn->col_count + 1));
572 
573  for (i = 0; i < conn->col_count; i++) {
574  switch (sqlite3_column_type(conn->statement, i)) {
575  case SQLITE_INTEGER:
576  MEM(row[i] = talloc_typed_asprintf(row, "%d", sqlite3_column_int(conn->statement, i)));
577  break;
578 
579  case SQLITE_FLOAT:
580  MEM(row[i] = talloc_typed_asprintf(row, "%f", sqlite3_column_double(conn->statement, i)));
581  break;
582 
583  case SQLITE_TEXT:
584  {
585  char const *p;
586  p = (char const *) sqlite3_column_text(conn->statement, i);
587 
588  if (p) MEM(row[i] = talloc_typed_strdup(row, p));
589  }
590  break;
591 
592  case SQLITE_BLOB:
593  {
594  uint8_t const *p;
595  size_t len;
596 
597  p = sqlite3_column_blob(conn->statement, i);
598  if (p) {
599  len = sqlite3_column_bytes(conn->statement, i);
600 
601  MEM(row[i] = talloc_zero_array(row, char, len + 1));
602  memcpy(row[i], p, len);
603  }
604  }
605  break;
606 
607  default:
608  break;
609  }
610  }
611 
612  *out = row;
613 
614  return RLM_SQL_OK;
615 }

618 {

620

682 static int mod_instantiate(rlm_sql_config_t const *config, void *instance, CONF_SECTION *cs)
683 {
684  bool exists;
685  rlm_sql_sqlite_t *inst = instance;
686  struct stat buf;
687 
688  if (!inst->filename) {
689  MEM(inst->filename = talloc_typed_asprintf(inst, "%s/%s",
690  main_config->raddb_dir, config->sql_db));
691  }
692 
693  if (stat(inst->filename, &buf) == 0) {
694  exists = true;
695  } else if (errno == ENOENT) {
696  exists = false;
697  } else {
698  ERROR("Database exists, but couldn't be opened: %s", fr_syserror(errno));
699  return -1;
700  }
701 
702  if (cf_pair_find(cs, "bootstrap") && !exists) {
703 # ifdef HAVE_SQLITE3_OPEN_V2
704  int status;
705  int ret;
706  char const *p;
707  char *buff;
708  sqlite3 *db = NULL;

786  "Upgrade to SQLite >= 3.5.1 if you need this functionality");

788  }
789
