47 /** Send DHCP packet using a connectionless UDP socket
48  *
49  * @param packet to send
50  * @return
51  * - >= 0 if successful.
52  * - < 0 if failed.
53  */
54 int fr_dhcpv4_udp_packet_send(fr_radius_packet_t *packet)
55 {
56  int ret;
57  struct sockaddr_storage dst;
58  socklen_t sizeof_dst;
59  struct sockaddr_storage src;
60  socklen_t sizeof_src;
61 
62  fr_ipaddr_to_sockaddr(&src, &sizeof_src, &packet->socket.inet.src_ipaddr, packet->socket.inet.src_port);
63  fr_ipaddr_to_sockaddr(&dst, &sizeof_dst, &packet->socket.inet.dst_ipaddr, packet->socket.inet.dst_port);
64  if (packet->data_len == 0) {
65  fr_strerror_printf("No data to send");
66  return -1;
67  }
68 
69  errno = 0;
70 
71  ret = sendfromto(packet->socket.fd, packet->data, packet->data_len, 0,
72  packet->socket.inet.ifindex,
73  (struct sockaddr *)&src, sizeof_src,
74  (struct sockaddr *)&dst, sizeof_dst);
75  if ((ret < 0) && errno) fr_strerror_printf("dhcp_send_socket: %s", fr_syserror(errno));
76 
77  return ret;
78 }

80 /** Receive DHCP packet using a connectionless UDP socket
81  *
82  * @param sockfd handle.
83  * @return
84  * - pointer to fr_radius_packet_t if successful.
85  * - NULL if failed.
86  */
87 fr_radius_packet_t *fr_dhcpv4_udp_packet_recv(int sockfd)
88 {
89  struct sockaddr_storage src;
90  struct sockaddr_storage dst;
91  socklen_t sizeof_src;
92  socklen_t sizeof_dst;
93  fr_radius_packet_t *packet;
94  uint8_t *data;
95  ssize_t data_len;
96  fr_ipaddr_t src_ipaddr, dst_ipaddr;
97  uint16_t src_port, dst_port;
98  int ifindex = 0;
99  fr_time_t when;
100 
101  data = talloc_zero_array(NULL, uint8_t, MAX_PACKET_SIZE);
102  if (!data) {
103  fr_strerror_printf("Out of memory");
104  return NULL;
105  }
106 
107  sizeof_src = sizeof(src);
108  sizeof_dst = sizeof(dst);
109  data_len = recvfromto(sockfd, data, MAX_PACKET_SIZE, 0,
110  &ifindex,
111  (struct sockaddr *)&src, &sizeof_src,
112  (struct sockaddr *)&dst, &sizeof_dst, &when);
113 
114  if (data_len <= 0) {
115  fr_strerror_printf("Failed reading data from DHCP socket: %s", fr_syserror(errno));
116  talloc_free(data);
117  return NULL;
118  }
119 
120  if (!fr_cond_assert(data_len <= (ssize_t)talloc_array_length(data))) {
121  talloc_free(data); /* Bounds check for tainted scalar (Coverity) */
122  return NULL;
123  }
124  sizeof_dst = sizeof(dst);
125 
126  /*
127  * This should never fail...
128  */
129  if (getsockname(sockfd, (struct sockaddr *) &dst, &sizeof_dst) < 0) {
130  fr_strerror_printf("getsockname failed: %s", fr_syserror(errno));
131  talloc_free(data);
132  return NULL;
133  }
134 
135  fr_ipaddr_from_sockaddr(&dst_ipaddr, &dst_port, &dst, sizeof_dst);
136  fr_ipaddr_from_sockaddr(&src_ipaddr, &src_port, &src, sizeof_src);
137 
138  if (!fr_dhcpv4_ok(data, data_len, NULL, NULL)) return NULL;
139 
140  packet = fr_dhcpv4_packet_alloc(data, data_len);
141  if (!packet) return NULL;
142 
143  fr_socket_addr_init_inet(&packet->socket, IPPROTO_UDP, ifindex, &src_ipaddr, src_port, &dst_ipaddr, dst_port);
144 
145  talloc_steal(packet, data);
146  packet->data = data;
147  packet->socket.fd = sockfd;
148 
149  packet->timestamp = when;
150  return packet;
151 }
