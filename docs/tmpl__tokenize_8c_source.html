298 /** Resolve attribute name to a #pair_list_t value.
299  *
300  * Check the name string for #pair_list_t qualifiers and write a #pair_list_t value
301  * for that list to out. This value may be passed to #radius_list, along with the current
302  * #request_t, to get a pointer to the actual list in the #request_t.
303  *
304  * If we're sure we've definitely found a list qualifier token delimiter (``:``) but the
305  * string doesn't match a #radius_list qualifier, return 0 and write #PAIR_LIST_UNKNOWN
306  * to out.
307  *
308  * If we can't find a string that looks like a request qualifier, set out to def, and
309  * return 0.
310  *
311  * @note #radius_list_name should be called before passing a name string that may
312  * contain qualifiers to #fr_dict_attr_by_name.
313  *
314  * @param[out] out Where to write the list qualifier.
315  * @param[in] name String containing list qualifiers to parse.
316  * @param[in] def the list to return if no qualifiers were found.
317  * @return 0 if no valid list qualifier could be found, else the number of bytes consumed.
318  * The caller may then advanced the name pointer by the value returned, to get the
319  * start of the attribute name (if any).
320  *
321  * @see pair_list
322  * @see radius_list
323  */
324 size_t radius_list_name(pair_list_t *out, char const *name, pair_list_t def)
325 {
326  char const *p = name;
327  char const *q;
328 
329  /* This should never be a NULL pointer */
330  fr_assert(name);
331 
332  /*
333  * Try and determine the end of the token
334  */
335  for (q = p; fr_dict_attr_allowed_chars[(uint8_t) *q]; q++);
336 
337  switch (*q) {
338  /*
339  * It's a bareword made up entirely of dictionary chars
340  * check and see if it's a list qualifier, and if it's
341  * not, return the def and say we couldn't parse
342  * anything.
343  */
344  case '\0':
345  *out = fr_table_value_by_substr(pair_list_table, p, (q - p), PAIR_LIST_UNKNOWN);
346  if (*out != PAIR_LIST_UNKNOWN) return q - p;
347  *out = def;
348  return 0;
349 
350  /*
351  * It may be a list qualifier delimiter
352  */
353  case ':':
354  {
355  char const *d = q + 1;
356 
357  if (isdigit((int) *d)) {
358  while (isdigit((int) *d)) d++;
359 
360  if (!fr_dict_attr_allowed_chars[(uint8_t) *d]) {
361  *out = def;
362  return 0;
363  }
364  }
365 
366  *out = fr_table_value_by_substr(pair_list_table, p, (q - p), PAIR_LIST_UNKNOWN);
367  if (*out == PAIR_LIST_UNKNOWN) return 0;
368 
369  return (q + 1) - name; /* Consume the list and delimiter */
370  }
371 
372  default:
373  *out = def;
374  return 0;
375  }
376 }

378 /** Resolve attribute name to a #request_ref_t value.
379  *
380  * Check the name string for qualifiers that reference a parent #request_t.
381  *
382  * If we find a string that matches a #request_ref_t qualifier, return the number of chars
383  * we consumed.
384  *
385  * If we're sure we've definitely found a list qualifier token delimiter (``*``) but the
386  * qualifier doesn't match one of the #request_ref_t qualifiers, return 0 and set out to
387  * #REQUEST_UNKNOWN.
388  *
389  * If we can't find a string that looks like a request qualifier, set out to def, and
390  * return 0.
391  *
392  * @param[out] out The #request_ref_t value the name resolved to (or #REQUEST_UNKNOWN).
393  * @param[in] name of attribute.
394  * @param[in] def default request ref to return if no request qualifier is present.
395  * @return 0 if no valid request qualifier could be found, else the number of bytes consumed.
396  * The caller may then advanced the name pointer by the value returned, to get the
397  * start of the attribute list or attribute name(if any).
398  *
399  * @see radius_list_name
400  * @see request_ref_table
401  */
402 size_t radius_request_name(request_ref_t *out, char const *name, request_ref_t def)
403 {
404  char const *p, *q;
405 
406  p = name;
407  /*
408  * Try and determine the end of the token
409  */
410  for (q = p; fr_dict_attr_allowed_chars[(uint8_t) *q] && (*q != '.') && (*q != '-'); q++);
411 
412  /*
413  * First token delimiter wasn't a '.'
414  */
415  if (*q != '.') {
416  *out = def;
417  return 0;
418  }
419 
420  *out = fr_table_value_by_substr(request_ref_table, name, q - p, REQUEST_UNKNOWN);
421  if (*out == REQUEST_UNKNOWN) return 0;
422 
423  return (q + 1) - p;
424 }

727  *

730 {

736  */

738

742  switch (src_ar->type) {

1158 /** Parse array subscript and in future other filters
1159  *
1160  * @param[out] err Parse error code.
1161  * @param[in] ar to populate filter for.
1162  * @param[in] name containing more attribute ref data.
1163  * @return
1164  * - TMPL_ATTR_REF_HAS_FILTER on success with filter parsed.
1165  * - TMPL_ATTR_REF_NO_FILTER on success with no filter.
1166  * - TMPL_ATTR_REF_BAD_FILTER on failure.
1167  */
1168 static tmpl_attr_filter_t tmpl_attr_parse_filter(tmpl_attr_error_t *err, tmpl_attr_t *ar, fr_sbuff_t *name)
1169 {
1170  /*
1171  * Parse array subscript (and eventually complex filters)
1172  */
1173  if (!fr_sbuff_next_if_char(name, '[')) return TMPL_ATTR_REF_NO_FILTER;
1174 
1175  switch (*fr_sbuff_current(name)) {
1176  case '#':
1177  ar->num = NUM_COUNT;
1178  fr_sbuff_next(name);
1179  break;
1180 
1181  case '*':
1182  ar->num = NUM_ALL;
1183  fr_sbuff_next(name);
1184  break;
1185 
1186  case 'n':
1187  ar->num = NUM_LAST;
1188  fr_sbuff_next(name);
1189  break;
1190 
1191  default:
1192  {
1193  fr_sbuff_parse_error_t sberr = FR_SBUFF_PARSE_OK;
1194  fr_sbuff_t tmp = FR_SBUFF_NO_ADVANCE(name);
1195 
1196  if (fr_sbuff_out(&sberr, &ar->num, &tmp) == 0) {
1197  if (sberr == FR_SBUFF_PARSE_ERROR_NOT_FOUND) {
1198  fr_strerror_printf("Array index is not an integer");
1199  if (err) *err = TMPL_ATTR_ERROR_INVALID_ARRAY_INDEX;
1200  error:
1201  return TMPL_ATTR_REF_BAD_FILTER;
1202  }
1203 
1204  fr_strerror_printf("Invalid array index");
1205  if (err) *err = TMPL_ATTR_ERROR_INVALID_ARRAY_INDEX;
1206  goto error;
1207  }
1208 
1209  if ((ar->num > 1000) || (ar->num < 0)) {
1210  fr_strerror_printf("Invalid array index '%hi' (should be between 0-1000)", ar->num);
1211  ar->num = 0;
1212  if (err) *err = TMPL_ATTR_ERROR_INVALID_ARRAY_INDEX;
1213  goto error;
1214  }

1218  }

1221  * Always advance here, so the error
1222  * marker points to the bad char.

1233 /** Parse an unresolved attribute, i.e. one which can't be found in the current dictionary
1234  *
1235  * This function calls itself recursively to process additional OID
1236  * components once we've failed to resolve one component.
1237  *
1238  * @note Do not call directly.
1239  *
1240  * @param[in] ctx to allocate new attribute reference in.
1241  * @param[out] err Parse error.
1242  * @param[in,out] vpt to append this reference to.
1243  * @param[in] name to parse.
1244  * @param[in] rules see tmpl_attr_afrom_attr_substr.
1245  * @param[in] depth How deep we are. Used to check for maximum nesting level.
1246  * @return
1247  * - <0 on error.
1248  * - 0 on success.
1249  */
1250 static inline int tmpl_attr_afrom_attr_unresolved_substr(TALLOC_CTX *ctx, tmpl_attr_error_t *err,
1251  tmpl_t *vpt,
1252  fr_sbuff_t *name, tmpl_rules_t const *rules,
1253  unsigned int depth)

1256  int ret;
1257  char *unresolved;
1258  size_t len;
1259 
1260  if (depth > FR_DICT_MAX_TLV_STACK) {
1261  fr_strerror_printf("Attribute nesting too deep");
1262  if (err) *err = TMPL_ATTR_ERROR_NESTING_TOO_DEEP;
1263  return -1;
1264  }
1265 
1266  /*
1267  * Input too short
1268  */
1269  if (!fr_sbuff_extend(name)) {
1270  fr_strerror_printf("Missing attribute reference");
1271  if (err) *err = TMPL_ATTR_ERROR_INVALID_NAME;
1272  return -1;
1273  }
1274 
1275  /*
1276  * Mark the tmpl up as an unresolved attribute reference
1277  * the attribute reference will be resolved later.
1278  */
1279  vpt->type = TMPL_TYPE_ATTR_UNRESOLVED;
1280

1283  * Copy out a string of allowed dictionary chars to form
1284  * the unresolved attribute name.
1285  *
1286  * This will be resolved later (outside of this function).
1287  */
1288  len = fr_sbuff_out_abstrncpy_allowed(ar, &unresolved,
1289  name, FR_DICT_ATTR_MAX_NAME_LEN + 1,
1290  fr_dict_attr_allowed_chars);
1291  if (len == 0) {
1292  fr_strerror_printf("Invalid attribute name");
1293  if (err) *err = TMPL_ATTR_ERROR_INVALID_NAME;
1294  error:
1295  talloc_free(ar);
1296  return -1;
1297  }
1298  if (len > FR_DICT_ATTR_MAX_NAME_LEN) {
1299  fr_strerror_printf("Attribute name is too long");
1300  if (err) *err = TMPL_ATTR_ERROR_INVALID_NAME;
1301  goto error;
1302  }
1303 
1304  *ar = (tmpl_attr_t){
1305  .ar_num = NUM_ANY,
1306  .ar_type = TMPL_ATTR_TYPE_UNRESOLVED,
1307  .ar_unresolved = unresolved
1308  };
1309 
1310  if (tmpl_attr_parse_filter(err, ar, name) < 0) goto error;
1311 
1312  /*
1313  * Insert the ar into the list of attribute references
1314  */
1315  tmpl_attr_insert(vpt, ar);
1316 
1317  /*
1318  * Once one OID component is created as unresolved all
1319  * future OID components are also unresolved.
1320  */
1321  if (fr_sbuff_next_if_char(name, '.')) {
1322  ret = tmpl_attr_afrom_attr_unresolved_substr(ctx, err, vpt, name, rules, depth + 1);
1323  if (ret < 0) {
1324  fr_dlist_talloc_free_tail(&vpt->data.attribute.ar); /* Remove and free ar */
1325  return -1;
1326  }
1327  }
1328 
1329  return 0;
1330 }

1332 /** Parse an attribute reference, either an OID or attribute name
1333  *
1334  * @note Do not call directly.
1335  *
1336  * @param[in] ctx to allocate new attribute reference in.
1337  * @param[out] err Parse error.
1338  * @param[in,out] vpt to append this reference to.
1339  * @param[in] parent Result of parsing the previous attribute reference.
1340  * @param[in] name to parse.
1341  * @param[in] rules which places constraints on attribute reference parsing.
1342  * Rules interpreted by this function is:
1343  * - allow_unknown - If false unknown OID components
1344  * result in a parse error.
1345  * - allow_unresolved - If false unknown attribute names
1346  * result in a parse error.
1347  * - disallow_internal - If an attribute resolves in the
1348  * internal dictionary then that results in a parse
1349  * error.
1350  * - allow_foreign - If an attribute resolves in a dictionary
1351  * that does not match the parent
1352  * (exception being FR_TYPE_GROUP) then that results
1353  * in a parse error.
1354  * @param[in] depth How deep we are. Used to check for maximum nesting level.
1355  * @return
1356  * - <0 on error.
1357  * - 0 on success.
1358  */

1360  tmpl_t *vpt,
1361  fr_dict_attr_t const *parent,
1362  fr_sbuff_t *name, tmpl_rules_t const *rules, unsigned int depth)
1363 {

1365  ssize_t slen;
1366  tmpl_attr_t *ar = NULL;
1367  fr_dict_attr_t const *da;
1368  fr_sbuff_marker_t m_s;
1369  tmpl_attr_filter_t filter;
1370  fr_dict_attr_err_t dict_err;
1371

1373 
1374  if (depth > FR_DICT_MAX_TLV_STACK) {
1375  fr_strerror_printf("Attribute nesting too deep");
1376  if (err) *err = TMPL_ATTR_ERROR_NESTING_TOO_DEEP;
1377  error:
1378  fr_sbuff_marker_release(&m_s);
1379  return -1;
1380  }
1381 
1382  /*
1383  * Input too short
1384  */
1385  if (!fr_sbuff_extend(name)) {
1386  fr_strerror_printf("Missing attribute reference");
1387  if (err) *err = TMPL_ATTR_ERROR_INVALID_NAME;
1388  goto error;
1389  }
1390 
1391  /*
1392  * No parent means we need to go hunting through all the dictionaries
1393  */
1394  if (!parent) {
1395  slen = fr_dict_attr_by_qualified_name_substr(&dict_err, &da,
1396  rules->dict_def, name, !rules->disallow_internal);
1397  /*
1398  * Otherwise we're resolving in the context of the last component,
1399  * or its reference in the case of group attributes.
1400  */
1401  } else {
1402  slen = fr_dict_attr_child_by_name_substr(&dict_err, &da, parent, name, false);
1403  }
1404 
1405  /*
1406  * Fatal errors related to nesting...
1407  */
1408  switch (dict_err) {
1409  case FR_DICT_ATTR_NO_CHILDREN:
1410  if (parent && parent->flags.is_unknown) break;
1411  goto error;
1412 
1413  case FR_DICT_ATTR_NOT_DESCENDENT:
1414  goto error;
1415 
1416  default:
1417  break;
1418  }
1419 
1420  /*
1421  * The named component was a known attribute
1422  * so record it as a normal attribute
1423  * reference.
1424  */
1425  if (slen > 0) {
1426  MEM(ar = talloc(ctx, tmpl_attr_t));
1427  *ar = (tmpl_attr_t){
1428  .ar_num = NUM_ANY,
1429  .ar_type = TMPL_ATTR_TYPE_NORMAL,
1430  .ar_da = da
1431  };
1432 
1433  goto check_attr;
1434  }
1435 
1436  /*
1437  * See if the ref begins with an unsigned integer
1438  * if it does it's probably an OID component
1439  *
1440  * .<oid>
1441  */
1442  slen = fr_sbuff_out(NULL, &oid, name);
1443  if (slen > 0) {
1444  fr_dict_attr_t *da_unknown;
1445 
1446  fr_strerror(); /* Clear out any existing errors */
1447 
1448  /*
1449  * Locating OID attributes is different than
1450  * locating named attributes because we have
1451  * significantly more numberspace overlap
1452  * between the protocols.
1453  */
1454  if (!parent && rules->dict_def) parent = fr_dict_root(rules->dict_def);
1455  if (!parent && !rules->disallow_internal) parent = fr_dict_root(fr_dict_internal());
1456  if (!parent) {
1457  fr_strerror_printf("OID references must be qualified with a protocol when used here");
1458  if (err) *err = TMPL_ATTR_ERROR_UNQUALIFIED_NOT_ALLOWED;
1459  fr_sbuff_set(name, &m_s);
1460  goto error;
1461  }
1462  /*
1463  * If it's numeric and not a known attribute
1464  * then we create an unknown attribute with
1465  * the specified attribute number.
1466  */
1467  da = fr_dict_attr_child_by_num(parent, oid);
1468  if (da) {
1469  /*
1470  * The OID component was a known attribute
1471  * so record it as a normal attribute
1472  * reference.
1473  */
1474  MEM(ar = talloc(ctx, tmpl_attr_t));
1475  *ar = (tmpl_attr_t){
1476  .ar_num = NUM_ANY,
1477  .ar_type = TMPL_ATTR_TYPE_NORMAL,
1478  .ar_da = da,
1479  };
1480  vpt->data.attribute.was_oid = true;
1481 
1482  goto check_attr;
1483  }
1484 
1485  if (!rules->allow_unknown) {
1486  fr_strerror_printf("Unknown attributes not allowed here");
1487  if (err) *err = TMPL_ATTR_ERROR_UNKNOWN_NOT_ALLOWED;
1488  fr_sbuff_set(name, &m_s);
1489  goto error;
1490  }
1491 
1492  MEM(ar = talloc(ctx, tmpl_attr_t));
1493 
1494  switch (parent->type) {
1495  case FR_TYPE_VSA:
1496  if (fr_dict_unknown_vendor_afrom_num(ar, &da_unknown, parent, oid) < 0) {
1497  if (err) *err = TMPL_ATTR_ERROR_UNKNOWN_NOT_ALLOWED; /* strerror set by dict function */
1498  goto error;
1499  }
1500  break;
1501 
1502  default:
1503  if (fr_dict_unknown_attr_afrom_num(ar, &da_unknown, parent, oid) < 0) {
1504  if (err) *err = TMPL_ATTR_ERROR_UNKNOWN_NOT_ALLOWED; /* strerror set by dict function */
1505  goto error;
1506  }
1507  break;
1508  }
1509  da_unknown->flags.internal = 1;
1510 
1511  *ar = (tmpl_attr_t){
1512  .ar_num = NUM_ANY,
1513  .ar_type = TMPL_ATTR_TYPE_UNKNOWN,
1514  .ar_unknown = da_unknown,
1515  .ar_da = da_unknown,
1516  };
1517  da = da_unknown;
1518  vpt->data.attribute.was_oid = true;
1519  goto do_suffix;
1520  }
1521 
1522  /*
1523  * Can't parse it as an attribute, might be a literal string
1524  * let the caller decide.
1525  *
1526  * Don't alter the fr_strerror buffer, may contain useful
1527  * errors from the dictionary code.
1528  */
1529  if (!rules->allow_unresolved) {
1530  fr_strerror_printf_push("Unresolved attributes not allowed here");
1531  if (err) *err = TMPL_ATTR_ERROR_UNRESOLVED_NOT_ALLOWED;
1532  fr_sbuff_set(name, &m_s);
1533  goto error;
1534  }
1535 
1536  fr_sbuff_marker_release(&m_s);
1537 
1538  /*
1539  * Once we hit one unresolved attribute we have to treat
1540  * the rest of the components are unresolved as well.
1541  */
1542  return tmpl_attr_afrom_attr_unresolved_substr(ctx, err, vpt, name, rules, depth);
1543 
1544 check_attr:
1545  /*
1546  * Attribute location (dictionary) checks
1547  */
1548  if (!rules->allow_foreign || rules->disallow_internal) {
1549  fr_dict_t const *found_in = fr_dict_by_da(da);
1550 
1551  /*
1552  * Parent is the dict root if this is the first ref in the
1553  * chain.
1554  */
1555  if (!parent) parent = fr_dict_root(rules->dict_def);
1556 
1557  /*
1558  * Even if allow_foreign is false, if disallow_internal is not
1559  * true, we still allow the resolution.
1560  */
1561  if (rules->disallow_internal && (found_in == fr_dict_internal())) {
1562  fr_strerror_printf("Internal attributes not allowed here");
1563  if (err) *err = TMPL_ATTR_ERROR_INTERNAL_NOT_ALLOWED;
1564  fr_sbuff_set(name, &m_s);
1565  goto error;
1566  }
1567  /*
1568  * Check that the attribute we resolved was from an allowed
1569  * dictionary.
1570  *
1571  * We already checked if internal attributes were disallowed
1572  * above, so we skip this check if the attribute is internal.
1573  *
1574  * The reason this checks works with foreign attributes is
1575  * because when an attr ref resolves to a group parent is not
1576  * set to that attribute, but the foreign dictionary attribute
1577  * that it references.
1578  *
1579  * My-Dhcp-In-RADIUS-Attribute.My-DHCP-Attribute
1580  * | ||_ DHCP attribute
1581  * | |_ Lookup inside linking attribute triggers dictionary change
1582  * |_ RADIUS attribute
1583  */
1584  if (found_in != fr_dict_internal() &&
1585  !rules->allow_foreign && (found_in != fr_dict_by_da(parent))) {
1586  fr_strerror_printf("Foreign %s attribute found. Only %s attributes are allowed here",
1587  fr_dict_root(found_in)->name,
1588  fr_dict_root(rules->dict_def)->name);
1589  if (err) *err = TMPL_ATTR_ERROR_FOREIGN_NOT_ALLOWED;
1590  fr_sbuff_set(name, &m_s);
1591  goto error;
1592  }
1593  }
1594 
1595 do_suffix:
1596  /*
1597  * Parse the attribute reference filter
1598  *
1599  * Error out immediately if the filter is bad
1600  * otherwise determine whether to keep the
1601  * attribute reference or omit it based on:
1602  *
1603  * - Whether there was a filter present.
1604  * - The type of attribute.
1605  * - If this is the leaf attribute reference.
1606  */
1607  if ((filter = tmpl_attr_parse_filter(err, ar, name)) == TMPL_ATTR_REF_BAD_FILTER) goto error;
1608 
1609  /*
1610  * At the end of the attribute reference. If there's a
1611  * trailing '.' then there's another attribute reference
1612  * we need to parse, otherwise we're done.
1613  */
1614  if (fr_sbuff_next_if_char(name, '.')) {
1615  switch (da->type) {
1616  /*
1617  * If this is a group then the parent is the
1618  * group ref.
1619  */
1620  case FR_TYPE_GROUP:
1621  parent = fr_dict_attr_ref(da);
1622  break;
1623 
1624  case FR_TYPE_STRUCT:
1625  case FR_TYPE_TLV:
1626  case FR_TYPE_VENDOR:
1627  case FR_TYPE_VSA:
1628  /*
1629  * Omit nesting types where the relationship is already
1630  * described by the dictionaries and there's no filter.
1631  *
1632  * These attribute references would just use additional
1633  * memory for no real purpose.
1634  *
1635  * Because we pre-allocate an attribute reference in
1636  * each tmpl talloc pool, unless the attribute
1637  * reference list contains a group, there's no performance
1638  * penalty in repeatedly allocating and freeing this ar.
1639  */
1640  if ((filter == TMPL_ATTR_REF_NO_FILTER) && (ar->type == TMPL_ATTR_TYPE_NORMAL)) {
1641  TALLOC_FREE(ar);
1642  }
1643  parent = da;
1644  break;
1645 
1646  default:
1647  fr_strerror_printf("Parent type of nested attribute must be of "
1648  "\"struct\", \"tlv\", \"vendor\", \"vsa\" or \"group\", got \"%s\"",
1649  fr_table_str_by_value(fr_value_box_type_table,
1650  da->type, "<INVALID>"));
1651  fr_sbuff_set(name, &m_s);
1652  goto error;
1653  }
1654 
1655  if (ar) tmpl_attr_insert(vpt, ar);
1656  if (tmpl_attr_afrom_attr_substr(ctx, err, vpt, parent, name, rules, depth + 1) < 0) {
1657  if (ar) fr_dlist_talloc_free_tail(&vpt->data.attribute.ar); /* Remove and free ar */
1658  goto error;
1659  }
1660  /*
1661  * If it's a leaf we always insert the attribute
1662  * reference into the list, even if it's a
1663  * nesting attribute.
1664  *
1665  * This is useful for nested update sections
1666  * where the tmpl might be the name of a new
1667  * subsection.
1668  */
1669  } else {
1670  tmpl_attr_insert(vpt, ar);
1671  }
1672 
1673  fr_sbuff_marker_release(&m_s);
1674 
1675  return 0;
1676 }

1756  fr_dlist_talloc_free_tail(list); /* Remove and free rr */
1757  return -1;
1758  }
1759  }

1770  * @param[in] name of attribute including #request_ref_t and #pair_list_t qualifiers.
1771  * If only #request_ref_t #pair_list_t qualifiers are found,
1772  * a #TMPL_TYPE_LIST #tmpl_t will be produced.
1773  * @param[in] p_rules Formatting rules used to check for trailing garbage.
1774  * @param[in] t_rules Rules which control parsing:
1775  * - dict_def The default dictionary to use if attributes
1776  * are unqualified.
1777  * - request_def The default #request_t to set if no
1778  * #request_ref_t qualifiers are found in name.
1779  * - list_def The default list to set if no #pair_list_t
1780  * qualifiers are found in the name.
1781  * - allow_unknown If true attributes in the format accepted by
1782  * #fr_dict_unknown_afrom_oid_substr will be allowed,
1783  * even if they're not in the main dictionaries.
1784  * If an unknown attribute is found a #TMPL_TYPE_ATTR
1785  * #tmpl_t will be produced.
1786  * If #tmpl_afrom_attr_substr is being called on
1787  * startup, the #tmpl_t may be passed to
1788  * #tmpl_attr_unknown_add to
1789  * add the unknown attribute to the main dictionary.
1790  * If the unknown attribute is not added to
1791  * the main dictionary the #tmpl_t cannot be used
1792  * to search for a #fr_pair_t in a #request_t.
1793  * - allow_unresolved If true, we don't generate a parse error on
1794  * unknown attributes. If an unknown attribute is
1795  * found a #TMPL_TYPE_ATTR_UNRESOLVED
1796  * #tmpl_t will be produced.
1797  * - allow_foreign If true, allow attribute names to be qualified
1798  * with a protocol outside of the passed dict_def.
1799  * - disallow_internal If true, don't allow fallback to internal
1800  * attributes.
1801  *
1802  * @see REMARKER to produce pretty error markers from the return value.
1803  *
1804  * @return
1805  * - <= 0 on error (offset as negative integer)
1806  * - > 0 on success (number of bytes parsed).
1807  */
1808 ssize_t tmpl_afrom_attr_substr(TALLOC_CTX *ctx, tmpl_attr_error_t *err,
1809  tmpl_t **out, fr_sbuff_t *name,
1810  fr_sbuff_parse_rules_t const *p_rules,
1811  tmpl_rules_t const *t_rules)
1812 {
1813  int ret;
1814  size_t list_len;
1815  tmpl_t *vpt;
1816  fr_sbuff_t our_name = FR_SBUFF_NO_ADVANCE(name); /* Take a local copy in case we need to back track */
1817  bool ref_prefix = false;
1818  bool is_raw = false;
1819 
1820  if (!t_rules) t_rules = &default_attr_rules; /* Use the defaults */
1821 
1822  if (err) *err = TMPL_ATTR_ERROR_NONE;
1823 
1824  if (!fr_sbuff_extend(&our_name)) {
