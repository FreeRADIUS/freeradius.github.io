41 { \

68 /** Update all markers and pointers in the set of dbuffs to point to new_buff
69  *
70  * This function should be used if the underlying buffer is realloced.
71  *
72  * @param[in] dbuff to update.
73  * @param[in] new_buff to assign to to sbuff.
74  * @param[in] new_len Length of the new buffer.
75  */
76 void fr_dbuff_update(fr_dbuff_t *dbuff, uint8_t *new_buff, size_t new_len)
77 {
78  fr_dbuff_t *dbuff_i;
79  uint8_t *old_buff; /* Current buff */
80 
81  old_buff = dbuff->buff;
82 
83  /*
84  * Update pointers to point to positions
85  * in new buffer based on their relative
86  * offsets in the old buffer... but not
87  * past the end of the new buffer.
88  */
89  for (dbuff_i = dbuff; dbuff_i; dbuff_i = dbuff_i->parent) {
90  fr_dbuff_marker_t *m_i;
91 
92  dbuff_i->buff = new_buff;
93  dbuff_i->start = new_buff + min(new_len, dbuff_i->start - old_buff);
94  dbuff_i->end = dbuff_i->buff + new_len;
95  dbuff_i->p = new_buff + min(new_len, dbuff_i->p - old_buff);
96 
97  for (m_i = dbuff_i->m; m_i; m_i = m_i->next) m_i->p = new_buff + min(new_len, m_i->p - old_buff);
98  }
99 }

101 /** Reallocate the current buffer
102  *
103  * @param[in] dbuff to be extended.
104  * @param[in] extension How many additional bytes should be allocated
105  * in the buffer.
106  * @return
107  * - 0 the extension operation failed.
108  * - >0 the number of bytes the buffer was extended by.
109  */
110 size_t fr_dbuff_extend_talloc(fr_dbuff_t *dbuff, size_t extension)
111 {
112  fr_dbuff_uctx_talloc_t *tctx = dbuff->uctx;
113  size_t clen, nlen, elen = extension;
114  uint8_t *new_buff;
115 
116  CHECK_DBUFF_INIT(dbuff);
117 
118  clen = dbuff->buff ? talloc_array_length(dbuff->buff) : 0;
119  /*
120  * If the current buffer size + the extension
121  * is less than init, extend the buffer to init.
122  *
123  * This can happen if the buffer has been
124  * trimmed, and then additional data is added.
125  */
126  if ((clen + elen) < tctx->init) {
127  elen = tctx->init - clen;
128  /*
129  * Double the buffer size if it's more than the
130  * requested amount.
131  */
132  } else if (elen < clen){
133  elen = clen;
134  }
135 
136  /*
137  * Check we don't exceed the maximum buffer
138  * length.
139  */
140  if (tctx->max && ((clen + elen) > tctx->max)) {
141  elen = tctx->max - clen;
142  if (elen == 0) {
143  fr_strerror_printf("Failed extending buffer by %zu bytes to "
144  "%zu bytes, max is %zu bytes",
145  extension, clen + extension, tctx->max);
146  return 0;
147  }
148  }
149  nlen = clen + elen;
150 
151  new_buff = talloc_realloc(tctx->ctx, dbuff->buff, uint8_t, nlen);
152  if (unlikely(!new_buff)) {
153  fr_strerror_printf("Failed extending buffer by %zu bytes to %zu bytes", elen, nlen);
154  return 0;
155  }
156 
157  (void)fr_dbuff_update(dbuff, new_buff, nlen); /* Shouldn't fail as we're extending */
158 
159  return elen;
160 }
