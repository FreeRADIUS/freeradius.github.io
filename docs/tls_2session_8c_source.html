218 /** Copy data to the intermediate buffer, before we send it somewhere
219  *
220  * @param[in] record buffer to write to.
221  * @param[in] in data to write.
222  * @param[in] inlen Length of data to write.
223  * @return the amount of data written to the record buffer.
224  */
225 inline static unsigned int record_from_buff(fr_tls_record_t *record, void const *in, unsigned int inlen)
226 {
227  unsigned int added = FR_TLS_MAX_RECORD_SIZE - record->used;
228 
229  if (added > inlen) added = inlen;
230  if (added == 0) return 0;
231 
232  memcpy(record->data + record->used, in, added);
233  record->used += added;
234 
235  return added;
236 }
237 
238 /** Take data from the buffer, and give it to the caller
239  *
240  * @param[in] record buffer to read from.
241  * @param[out] out where to write data from record buffer.
242  * @param[in] outlen The length of the output buffer.
243  * @return the amount of data written to the output buffer.
244  */
245 inline static unsigned int record_to_buff(fr_tls_record_t *record, void *out, unsigned int outlen)
246 {
247  unsigned int taken = record->used;
248 
249  if (taken > outlen) taken = outlen;
250  if (taken == 0) return 0;
251  if (out) memcpy(out, record->data, taken);
252 
253  record->used -= taken;
254 
255  /*
256  * This is pretty bad...
257  */
258  if (record->used > 0) memmove(record->data, record->data + taken, record->used);
259 
260  return taken;
261 }

400 /** Determine the PSK to use for an incoming connection
401  *
402  * @param[in] ssl session.
403  * @param[in] identity The identity of the PSK to search for.
404  * @param[out] psk Where to write the PSK we found (if any).
405  * @param[in] max_psk_len The length of the buffer provided for PSK.
406  * @return
407  * - 0 if no PSK matching identity was found.
408  * - >0 if a PSK matching identity was found (the length of bytes written to psk).
409  */
410 unsigned int fr_tls_session_psk_server_cb(SSL *ssl, const char *identity,
411  unsigned char *psk, unsigned int max_psk_len)
412 {
413  size_t psk_len = 0;
414  fr_tls_conf_t *conf;
415  request_t *request;
416 
417  conf = (fr_tls_conf_t *)SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_CONF);
418  if (!conf) return 0;
419 
420  request = (request_t *)SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_REQUEST);
421  if (request && conf->psk_query) {
422  size_t hex_len;
423  fr_pair_t *vp;
424  char buffer[2 * PSK_MAX_PSK_LEN + 4]; /* allow for too-long keys */
425 
426  /*
427  * The passed identity is weird. Deny it.
428  */
429  if (!session_psk_identity_is_safe(identity)) {
430  RWDEBUG("Invalid characters in PSK identity %s", identity);
431  return 0;
432  }
433 
434  MEM(pair_update_request(&vp, attr_tls_psk_identity) >= 0);
435  if (fr_pair_value_from_str(vp, identity, -1, '\0', true) < 0) {
436  RPWDEBUG2("Failed parsing TLS PSK Identity");
437  talloc_free(vp);
438  return 0;
439  }
440 
441  hex_len = xlat_eval(buffer, sizeof(buffer), request, conf->psk_query, NULL, NULL);
442  if (!hex_len) {
443  RWDEBUG("PSK expansion returned an empty string.");
444  return 0;
445  }
446 
447  /*
448  * The returned key is truncated at MORE than
449  * OpenSSL can handle. That way we can detect
450  * the truncation, and complain about it.
451  */
452  if (hex_len > (2 * max_psk_len)) {
453  RWDEBUG("Returned PSK is too long (%u > %u)", (unsigned int) hex_len, 2 * max_psk_len);
454  return 0;
455  }
456 
457  /*
458  * Leave the TLS-PSK-Identity in the request, and
459  * convert the expansion from printable string
460  * back to hex.
461  */
462  return fr_hex2bin(NULL,
463  &FR_DBUFF_TMP((uint8_t *)psk, (size_t)max_psk_len),
464  &FR_SBUFF_IN(buffer, hex_len), false);
465  }
466 
467  if (!conf->psk_identity) {
468  DEBUG("No static PSK identity set. Rejecting the user");
469  return 0;
470  }
471 
472  /*
473  * No request_t, or no dynamic query. Just look for a
474  * static identity.
475  */
476  if (strcmp(identity, conf->psk_identity) != 0) {
477  ERROR("Supplied PSK identity %s does not match configuration. Rejecting.",
478  identity);
479  return 0;
480  }
481 
482  psk_len = strlen(conf->psk_password);
483  if (psk_len > (2 * max_psk_len)) return 0;
484 
485  return fr_hex2bin(NULL,
486  &FR_DBUFF_TMP((uint8_t *)psk, (size_t)max_psk_len),
487  &FR_SBUFF_IN(conf->psk_password, psk_len), false);
488 }

491 /** Record session state changes
492  *
493  * Called by OpenSSL whenever the session state changes, an alert is received or an error occurs.
494  *
495  * @param[in] ssl session.
496  * @param[in] where Which context the callback is being called in.
497  * See https://www.openssl.org/docs/manmaster/man3/SSL_CTX_set_info_callback.html
498  * for additional info.
499  * @param[in] ret 0 if an error occurred, or the alert type if an alert was received.
500  */
501 void fr_tls_session_info_cb(SSL const *ssl, int where, int ret)
502 {
503  char const *role, *state;
504  request_t *request = SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_REQUEST);
505 
506  if ((where & ~SSL_ST_MASK) & SSL_ST_CONNECT) {
507  role = "Client ";
508  } else if (((where & ~SSL_ST_MASK)) & SSL_ST_ACCEPT) {
509  role = "Server ";
510  } else {
511  role = "";
512  }
513 
514  state = SSL_state_string_long(ssl);
515  state = state ? state : "<INVALID>";
516 
517  if ((where & SSL_CB_LOOP) || (where & SSL_CB_HANDSHAKE_START) || (where & SSL_CB_HANDSHAKE_DONE)) {
518  if (RDEBUG_ENABLED3) {
519  char const *abbrv = SSL_state_string(ssl);
520  size_t len;
521 #if OPENSSL_VERSION_NUMBER >= 0x10100000L
522  STACK_OF(SSL_CIPHER) *client_ciphers;
523 #endif
524 
525  /*
526  * Trim crappy OpenSSL state strings...
527  */
528  len = strlen(abbrv);
529  if ((len > 1) && (abbrv[len - 1] == ' ')) len--;
530 
531  RDEBUG3("Handshake state [%.*s] - %s%s", (int)len, abbrv, role, state);
532 
533  /*
534  * After a ClientHello, list all the proposed ciphers from the client
535  */
536 #if OPENSSL_VERSION_NUMBER >= 0x10100000L
537  if (SSL_get_state(ssl) == TLS_ST_SR_CLNT_HELLO &&
538  (client_ciphers = SSL_get_client_ciphers(ssl))) {
539  int i;
540  int num_ciphers = sk_SSL_CIPHER_num(client_ciphers);
541  const SSL_CIPHER *this_cipher;
542 
543  RDEBUG3("Client's preferred ciphers (by priority)");
544  RINDENT();
545  for (i = 0; i < num_ciphers; i++) {
546  this_cipher = sk_SSL_CIPHER_value(client_ciphers, i);
547  RDEBUG3("[%i] %s", i, SSL_CIPHER_get_name(this_cipher));
548  }
549  REXDENT();
550  }
551 #endif
552  } else {
553  RDEBUG2("Handshake state - %s%s (%i)", role, state, SSL_get_state(ssl));
554  }
555  return;
556  }
557 
558  if (where & SSL_CB_ALERT) {
559  if ((ret & 0xff) == SSL_AD_CLOSE_NOTIFY) return;
560 
561  if (where & SSL_CB_READ) {
562  fr_pair_t *vp;
563 
564  REDEBUG("Client sent %s TLS alert: %s", SSL_alert_type_string_long(ret),
565  SSL_alert_desc_string_long(ret));
566 
567  /*
568  * Offer helpful advice... Should be expanded.
569  */
570  switch (ret & 0xff) {
571  case TLS1_AD_UNKNOWN_CA:
572  REDEBUG("Verify client has copy of CA certificate, and trusts CA");
573  break;
574 
575  default:
576  break;
577  }
578 
579  MEM(pair_update_request(&vp, attr_tls_client_error_code) >= 0);
580  vp->vp_uint8 = ret & 0xff;
581  RDEBUG2("&TLS-Client-Error-Code := %pV", &vp->data);
582  } else {
583  REDEBUG("Sending client %s TLS alert: %s %i", SSL_alert_type_string_long(ret),
584  SSL_alert_desc_string_long(ret), ret & 0xff);
585  }
586  return;
587  }
588 
589  if (where & SSL_CB_EXIT) {
590  if (ret == 0) {
591  REDEBUG("Handshake exit state %s%s", role, state);
592  return;
593  }
594

596  if (SSL_want_read(ssl)) {
597  RDEBUG2("Need more data from client"); /* State same as previous call, don't print */
598  return;
599  }
600  REDEBUG("Handshake exit state %s%s", role, state);
601  }
602  }

712 /** Record the progression of the TLS handshake
713  *
714  * This callback is called by OpenSSL whenever a protocol message relating to a handshake is sent
715  * or received.
716  *
717  * This function copies state information from the various arguments into the state->info
718  * structure of the #fr_tls_session_t, to allow us to track the progression of the handshake.
719  *
720  * @param[in] write_p
721  * - 0 when a message has been received.
722  * - 1 when a message has been sent.
723  *
724  * @param[in] msg_version The TLS version negotiated, should be one of:
725  * - TLS1_VERSION
726  * - TLS1_1_VERSION
727  * - TLS1_2_VERSION
728  * - TLS1_3_VERSION
729  *
730  * @param[in] content_type One of the contentType values defined for TLS:
731  * - SSL3_RT_CHANGE_CIPHER_SPEC (20)
732  * - SSL3_RT_ALERT (21)
733  * - SSL3_RT_HANDSHAKE (22)
734  * - TLS1_RT_HEARTBEAT (24)
735  *
736  * @param[in] inbuf The raw protocol message.
737  * @param[in] len Length of the raw protocol message.
738  * @param[in] ssl The SSL session.
739  * @param[in] arg The #fr_tls_session_t holding the SSL session.
740  */
741 void fr_tls_session_msg_cb(int write_p, int msg_version, int content_type,
742  void const *inbuf, size_t len,
743  SSL *ssl, void *arg)
744 {
745  uint8_t const *buf = inbuf;
746  fr_tls_session_t *session = talloc_get_type_abort(arg, fr_tls_session_t);
747  request_t *request = SSL_get_ex_data(session->ssl, FR_TLS_EX_INDEX_REQUEST);
748 
749  /*
750  * Mostly to check for memory corruption...
751  */
752  if (!fr_cond_assert(session->ssl = ssl)) {
753  ERROR("fr_tls_session_t and ssl arg do not match in fr_tls_session_msg_cb");
754  session->invalid = true;
755  return;
756  }
757 
758  /*
759  * As per https://tools.ietf.org/html/rfc7568
760  *
761  * We explicitly disable SSLv2/v3, hence the asserts.
762  */
763 #ifdef SSL2_VERSION
764  if (!fr_cond_assert(msg_version != SSL2_VERSION)) {
765  ROPTIONAL(REDEBUG, ERROR, "Invalid version (SSLv2) in handshake");
766  session->invalid = true;
767  return;
768  }
769 #endif
770 
771 #ifdef SSL3_VERSION
772  if (!fr_cond_assert(msg_version != SSL3_VERSION)) {
773  ROPTIONAL(REDEBUG, ERROR, "Invalid version (SSLv3) in handshake");
774  session->invalid = true;
775  return;
776  }
777 #endif
778 
779  /*
780  * OpenSSL >= 1.0.2 calls this function with 'pseudo'
781  * content types. Which breaks our tracking of
782  * the SSL Session state.
783  */
784  if ((msg_version == 0) && (content_type > UINT8_MAX)) {
785  DEBUG4("Ignoring fr_tls_session_msg_cb call with pseudo content type %i, version %i",
786  content_type, msg_version);
787  return;
788  }
789 
790  if ((write_p != 0) && (write_p != 1)) {
791  DEBUG4("Ignoring fr_tls_session_msg_cb call with invalid write_p %d", write_p);
792  return;
793  }
794 
795  /*
796  * 0 - received (from peer)
797  * 1 - sending (to peer)
798  */
799  session->info.origin = write_p;
800  session->info.content_type = content_type;
801  session->info.record_len = len;
802  session->info.version = msg_version;
803  session->info.initialized = true;
804 
805  switch (content_type) {
806  case SSL3_RT_ALERT:
807  session->info.alert_level = buf[0];
808  session->info.alert_description = buf[1];
809  session->info.handshake_type = 0x00;
810  break;
811 
812  case SSL3_RT_HANDSHAKE:
813  session->info.handshake_type = buf[0];
814  session->info.alert_level = 0x00;
815  session->info.alert_description = 0x00;
816  break;
817 
818 #ifdef SSL3_RT_HEARTBEAT
819  case TLS1_RT_HEARTBEAT:
820  uint8_t *p = buf;
821 
822  if ((len >= 3) && (p[0] == 1)) {
823  size_t payload_len;
824 
825  payload_len = (p[1] << 8) | p[2];
826  if ((payload_len + 3) > len) {
827  session->invalid = true;
828  ROPTIONAL(REDEBUG, ERROR, "OpenSSL Heartbeat attack detected. Closing connection");
829  return;
830  }
831  }
832  break;
833 #endif
834  default:
835  break;
836  }
837 
838  session_msg_log(request, session, (uint8_t const *)inbuf, len);
839 }

863 /** Extract attributes from an X509 certificate
864  *
865  * @param cursor to copy attributes to.
866  * @param ctx to allocate attributes in.
867  * @param session current TLS session.
868  * @param cert to validate.
869  * @param depth the certificate is in the certificate chain (0 == leaf).
870  * @return
871  * - 0 on success.
872  * - < 0 on failure.
873  */
874 int fr_tls_session_pairs_from_x509_cert(fr_cursor_t *cursor, TALLOC_CTX *ctx,
875  fr_tls_session_t *session, X509 *cert, int depth)
876 {
877  char buffer[1024];
878  char attribute[256];
879  char **identity;
880  int attr_index, loc;
881 
882 #if OPENSSL_VERSION_NUMBER >= 0x10100000L
883  STACK_OF(X509_EXTENSION) const *ext_list = NULL;
884 #else
885  STACK_OF(X509_EXTENSION) *ext_list = NULL;
886 #endif
887 
888  ASN1_INTEGER *sn = NULL;
889  ASN1_TIME *asn_time = NULL;
890 
891  fr_pair_t *vp = NULL;
892 
893  request_t *request;
894 
895 #define CERT_ATTR_ADD(_attr, _attr_index, _value) fr_tls_session_cert_attr_add(ctx, request, cursor, _attr, _attr_index, _value)
896 
897  fr_cursor_tail(cursor);
898 
899  attr_index = depth;
900  if (attr_index > 1) attr_index = 1;
901 
902  request = (request_t *)SSL_get_ex_data(session->ssl, FR_TLS_EX_INDEX_REQUEST);
903  fr_assert(request != NULL);
904 
905  identity = (char **)SSL_get_ex_data(session->ssl, FR_TLS_EX_INDEX_IDENTITY);
906 
907  if (RDEBUG_ENABLED3) {
908  buffer[0] = '\0';
909  X509_NAME_oneline(X509_get_subject_name(cert), buffer, sizeof(buffer));
910  buffer[sizeof(buffer) - 1] = '\0';
911  RDEBUG3("Creating attributes for \"%s\":", buffer[0] ? buffer : "Cert missing subject OID");
912  }
913 
914  /*
915  * Get the Serial Number
916  */
917  sn = X509_get_serialNumber(cert);
918  if (sn && ((size_t) sn->length < (sizeof(buffer) / 2))) {
919  char *p = buffer;
920  int i;
921 
922  for (i = 0; i < sn->length; i++) {
923  sprintf(p, "%02x", (unsigned int)sn->data[i]);
924  p += 2;
925  }
926 
927  CERT_ATTR_ADD(IDX_SERIAL, attr_index, buffer);
928  }
929 
930  /*
931  * Get the Expiration Date
932  */
933  buffer[0] = '\0';
934  asn_time = X509_get_notAfter(cert);
935  if (identity && asn_time && (asn_time->length < (int)sizeof(buffer))) {
936  time_t expires;
937 
938  /*
939  * Add expiration as a time since the epoch
940  */
941  if (fr_tls_utils_asn1time_to_epoch(&expires, asn_time) < 0) {
942  RPWDEBUG("Failed parsing certificate expiry time");
943  } else {
944  vp = CERT_ATTR_ADD(IDX_EXPIRATION, attr_index, NULL);
945  vp->vp_date = fr_time_from_timeval(&(struct timeval) {.tv_sec = expires});
946  }
947  }
948 
949  /*
950  * Get the Subject & Issuer
951  */
952  buffer[0] = '\0';
953  X509_NAME_oneline(X509_get_subject_name(cert), buffer, sizeof(buffer));
954  buffer[sizeof(buffer) - 1] = '\0';
955  if (identity && buffer[0]) {
956  CERT_ATTR_ADD(IDX_SUBJECT, attr_index, buffer);
957 
958  /*
959  * Get the Common Name, if there is a subject.
960  */
961  X509_NAME_get_text_by_NID(X509_get_subject_name(cert),
962  NID_commonName, buffer, sizeof(buffer));
963  buffer[sizeof(buffer) - 1] = '\0';
964 
965  if (buffer[0]) {
966  CERT_ATTR_ADD(IDX_COMMON_NAME, attr_index, buffer);
967  }
968  }
969 
970  X509_NAME_oneline(X509_get_issuer_name(cert), buffer, sizeof(buffer));
971  buffer[sizeof(buffer) - 1] = '\0';
972  if (identity && buffer[0]) {
973  CERT_ATTR_ADD(IDX_ISSUER, attr_index, buffer);
974  }
975 
976  /*
977  * Get the RFC822 Subject Alternative Name
978  */
979  loc = X509_get_ext_by_NID(cert, NID_subject_alt_name, 0);
980  if (loc >= 0) {
981  X509_EXTENSION *ext = NULL;
982  GENERAL_NAMES *names = NULL;
983  int i;
984 
985  ext = X509_get_ext(cert, loc);
986  if (ext && (names = X509V3_EXT_d2i(ext))) {
987  for (i = 0; i < sk_GENERAL_NAME_num(names); i++) {
988  GENERAL_NAME *name = sk_GENERAL_NAME_value(names, i);
989 
990  switch (name->type) {
991 #ifdef GEN_EMAIL
992  case GEN_EMAIL: {
993 #if OPENSSL_VERSION_NUMBER >= 0x10100000L
994  char const *rfc822Name = (char const *)ASN1_STRING_get0_data(name->d.rfc822Name);
995 #else
996  char *rfc822Name = (char *)ASN1_STRING_data(name->d.rfc822Name);
997 #endif
998 
999  CERT_ATTR_ADD(IDX_SUBJECT_ALT_NAME_EMAIL, attr_index, rfc822Name);
1000  break;
1001  }
1002 #endif /* GEN_EMAIL */
1003 #ifdef GEN_DNS
1004  case GEN_DNS: {
1005 #if OPENSSL_VERSION_NUMBER >= 0x10100000L
1006  char const *dNSName = (char const *)ASN1_STRING_get0_data(name->d.dNSName);
1007 #else
1008  char *dNSName = (char *)ASN1_STRING_data(name->d.dNSName);
1009 #endif
1010  CERT_ATTR_ADD(IDX_SUBJECT_ALT_NAME_DNS, attr_index, dNSName);
1011  break;
1012  }
1013 #endif /* GEN_DNS */
1014 #ifdef GEN_OTHERNAME
1015  case GEN_OTHERNAME:
1016  /* look for a MS UPN */
1017  if (NID_ms_upn != OBJ_obj2nid(name->d.otherName->type_id)) break;
1018 
1019  /* we've got a UPN - Must be ASN1-encoded UTF8 string */
1020  if (name->d.otherName->value->type == V_ASN1_UTF8STRING) {
1021  CERT_ATTR_ADD(IDX_SUBJECT_ALT_NAME_UPN, attr_index,
1022  (char *)name->d.otherName->value->value.utf8string);
1023  break;
1024  }
1025 
1026  RWARN("Invalid UPN in Subject Alt Name (should be UTF-8)");
1027  break;
1028 #endif /* GEN_OTHERNAME */
1029  default:
1030  /* XXX TODO handle other SAN types */
1031  break;
1032  }
1033  }
1034  }
1035  if (names != NULL) GENERAL_NAMES_free(names);
1036  }
1037 
1038  /*
1039  * Only add extensions for the actual client certificate
1040  */
1041  if (attr_index == 0) {
1042 #if OPENSSL_VERSION_NUMBER >= 0x10100000L
1043  ext_list = X509_get0_extensions(cert);
1044 #else
1045  ext_list = cert->cert_info->extensions;
1046 #endif
1047 
1048  /*
1049  * Grab the X509 extensions, and create attributes out of them.
1050  * For laziness, we re-use the OpenSSL names
1051  */
1052  if (sk_X509_EXTENSION_num(ext_list) > 0) {
1053  int i, len;
1054  char *p;
1055  BIO *out;
1056 
1057  MEM(out = BIO_new(BIO_s_mem()));
1058  strlcpy(attribute, "TLS-Client-Cert-", sizeof(attribute));
1059 
1060  for (i = 0; i < sk_X509_EXTENSION_num(ext_list); i++) {
1061  char value[1024];
1062  ASN1_OBJECT *obj;
1063  X509_EXTENSION *ext;
1064  fr_dict_attr_t const *da;
1065 
1066  ext = sk_X509_EXTENSION_value(ext_list, i);
1067 
1068  obj = X509_EXTENSION_get_object(ext);
1069  if (i2a_ASN1_OBJECT(out, obj) <= 0) {
1070  RPWDEBUG("Skipping X509 Extension (%i) conversion to attribute. "
1071  "Conversion from ASN1 failed...", i);
1072  continue;
1073  }
1074 
1075  len = BIO_read(out, attribute + 16 , sizeof(attribute) - 16 - 1);
1076  if (len <= 0) continue;
1077 
1078  attribute[16 + len] = '\0';
1079 
1080  for (p = attribute + 16; *p != '\0'; p++) if (*p == ' ') *p = '-';
1081 
1082  X509V3_EXT_print(out, ext, 0, 0);
1083  len = BIO_read(out, value , sizeof(value) - 1);
1084  if (len <= 0) continue;
1085 
1086  value[len] = '\0';
1087 
1088  da = fr_dict_attr_by_name(dict_freeradius, attribute);
1089  if (!da) {
1090  RWDEBUG3("Skipping attribute %s: "
1091  "Add dictionary definition if you want to access it", attribute);
1092  continue;
1093  }
1094 
1095  MEM(vp = fr_pair_afrom_da(request, da));
1096  if (fr_pair_value_from_str(vp, value, -1, '\0', true) < 0) {
1097  RPWDEBUG3("Skipping: %s += '%s'", attribute, value);
1098  talloc_free(vp);
1099  continue;
1100  }
1101 
1102  fr_cursor_append(cursor, vp);
1103  }
1104  BIO_free_all(out);
1105  }
1106  }
1107 
1108  return 0;
1109 }

1111 /** Decrypt application data
1112  *
1113  * @note Handshake must have completed before this function may be called.
1114  *
1115  * Feed data from dirty_in to OpenSSL, and read the clean data into clean_out.
1116  *
1117  * @param[in] request The current #request_t.
1118  * @param[in] session The current TLS session.
1119  * @return
1120  * - -1 on error.
1121  * - 1 if more fragments are required to fully reassemble the record for decryption.
1122  * - 0 if we decrypted a complete record.
1123  */
1124 int fr_tls_session_recv(request_t *request, fr_tls_session_t *session)
1125 {
1126  int ret;
1127 
1128  fr_tls_session_request_bind(request, session->ssl);
1129 
1130  if (!SSL_is_init_finished(session->ssl)) {
1131  REDEBUG("Attempted to read application data before handshake completed");
1132  error:
1133  ret = -1;
1134  goto finish;
1135  }
1136 
1137  /*
1138  * Decrypt the complete record.
1139  */
1140  if (session->dirty_in.used) {
1141  ret = BIO_write(session->into_ssl, session->dirty_in.data, session->dirty_in.used);
1142  if (ret != (int) session->dirty_in.used) {
1143  record_init(&session->dirty_in);
1144  REDEBUG("Failed writing %zd bytes to SSL BIO: %d", session->dirty_in.used, ret);
1145  goto error;
1146  }
1147 
1148  record_init(&session->dirty_in);
1149  }
1150 
1151  /*
1152  * Clear the dirty buffer now that we are done with it
1153  * and init the clean_out buffer to store decrypted data
1154  */
1155  record_init(&session->clean_out);
1156 
1157  /*
1158  * Read (and decrypt) the tunneled data from the
1159  * SSL session, and put it into the decrypted
1160  * data buffer.
1161  */
1162  ret = SSL_read(session->ssl, session->clean_out.data, sizeof(session->clean_out.data));
1163  if (ret < 0) {
1164  int code;
1165 
1166  code = SSL_get_error(session->ssl, ret);
1167  switch (code) {
1168  case SSL_ERROR_WANT_READ:
1169  RWDEBUG("Peer indicated record was complete, but OpenSSL returned SSL_WANT_READ. "
1170  "Attempting to continue");
1171  ret = 1;
1172  goto finish;
1173 
1174  case SSL_ERROR_WANT_WRITE:
1175  REDEBUG("Error in fragmentation logic: SSL_WANT_WRITE");
1176  goto error;
1177 
1178  case SSL_ERROR_NONE:
1179  RDEBUG2("No application data received. Assuming handshake is continuing...");
1180  ret = 0;
1181  break;
1182 
1183  default:
1184  REDEBUG("Error in fragmentation logic");
1185  fr_tls_log_io_error(request, session, ret, "Failed in SSL_read");
1186  goto error;
1187  }
1188 
1189  }
1190 
1191  /*
1192  * Passed all checks, successfully decrypted data
1193  */
1194  session->clean_out.used = ret;
1195  ret = 0;
1196 
1197  if (RDEBUG_ENABLED3) {
1198  RHEXDUMP3(session->clean_out.data, session->clean_out.used,
1199  "Decrypted TLS application data (%zu bytes)", session->clean_out.used);
1200  } else {
1201  RDEBUG2("Decrypted TLS application data (%zu bytes)", session->clean_out.used);
1202  }
1203 finish:
1204  fr_tls_session_request_unbind(session->ssl);
1205 
1206  return ret;
1207 }

1209 /** Encrypt application data
1210  *
1211  * @note Handshake must have completed before this function may be called.
1212  *
1213  * Take cleartext data from clean_in, and feed it to OpenSSL, reading
1214  * the encrypted data into dirty_out.
1215  *
1216  * @param request The current request.
1217  * @param session The current TLS session.
1218  * @return
1219  * - -1 on failure.
1220  * - 0 on success.
1221  */
1222 int fr_tls_session_send(request_t *request, fr_tls_session_t *session)
1223 {
1224  int ret = 0;
1225 
1226  fr_tls_session_request_bind(request, session->ssl);
1227 
1228  if (!SSL_is_init_finished(session->ssl)) {
1229  REDEBUG("Attempted to write application data before handshake completed");
1230  ret = -1;
1231  goto finish;
1232  }
1233 
1234  /*
1235  * If there's un-encrypted data in 'clean_in', then write
1236  * that data to the SSL session, and then call the BIO function
1237  * to get that encrypted data from the SSL session, into
1238  * a buffer which we can then package into an EAP packet.
1239  *
1240  * Based on Server's logic this clean_in is expected to
1241  * contain the data to send to the client.
1242  */
1243  if (session->clean_in.used > 0) {
1244  if (RDEBUG_ENABLED3) {
1245  RHEXDUMP3(session->clean_in.data, session->clean_in.used,
1246  "TLS application data to encrypt (%zu bytes)", session->clean_in.used);
1247  } else {
1248  RDEBUG2("TLS application data to encrypt (%zu bytes)", session->clean_in.used);
1249  }
1250

1252  record_to_buff(&session->clean_in, NULL, ret);
1253 
1254  /* Get the dirty data from Bio to send it */
1255  ret = BIO_read(session->from_ssl, session->dirty_out.data,
1256  sizeof(session->dirty_out.data));
1257  if (ret > 0) {
1258  session->dirty_out.used = ret;
1259  ret = 0;
1260  } else {
1261  if (fr_tls_log_io_error(request, session, ret, "Failed in SSL_write") < 0) ret = -1;
1262  }
1263  }
1264 
1265 finish:
1266  fr_tls_session_request_unbind(session->ssl);
1267 
1268  return ret;
1269 }

1318 /** Continue a TLS handshake
1319  *
1320  * Advance the TLS handshake by feeding OpenSSL data from dirty_in,
1321  * and reading data from OpenSSL into dirty_out.
1322  *
1323  * @param request The current request.
1324  * @param session The current TLS session.
1325  * @return
1326  * - -1 on error.
1327  * - 0 on success.
1328  */
1329 int fr_tls_session_handshake(request_t *request, fr_tls_session_t *session)
1330 {
1331  int ret;
1332 
1333  fr_tls_session_request_bind(request, session->ssl);
1334 
1335  /*
1336  * This is a logic error. fr_tls_session_handshake
1337  * must not be called if the handshake is
1338  * complete fr_tls_session_recv must be
1339  * called instead.
1340  */
1341  if (SSL_is_init_finished(session->ssl)) {
1342  REDEBUG("Attempted to continue TLS handshake, but handshake has completed");
1343  error:
1344  ret = -1;
1345  goto finish;
1346  }
1347 
1348  if (session->invalid) {
1349  REDEBUG("Preventing invalid session from continuing");
1350  goto error;
1351  }
1352 
1353  /*
1354  * Feed dirty data into OpenSSL, so that is can either
1355  * process it as Application data (decrypting it)
1356  * or continue the TLS handshake.
1357  */
1358  if (session->dirty_in.used) {
1359  ret = BIO_write(session->into_ssl, session->dirty_in.data, session->dirty_in.used);
1360  if (ret != (int)session->dirty_in.used) {
1361  REDEBUG("Failed writing %zd bytes to TLS BIO: %d", session->dirty_in.used, ret);
1362  record_init(&session->dirty_in);
1363  goto error;
1364  }
1365  record_init(&session->dirty_in);
1366  }
1367 
1368  /*
1369  * Magic/More magic? Although SSL_read is normally
1370  * used to read application data, it will also
1371  * continue the TLS handshake. Removing this call will
1372  * cause the handshake to fail.
1373  *
1374  * We don't ever expect to actually *receive* application
1375  * data here.
1376  *
1377  * The reason why we call SSL_read instead of SSL_accept,
1378  * or SSL_connect, as it allows this function
1379  * to be used, irrespective or whether we're acting
1380  * as a client or a server.
1381  *
1382  * If acting as a client SSL_set_connect_state must have
1383  * been called before this function.
1384  *
1385  * If acting as a server SSL_set_accept_state must have
1386  * been called before this function.
1387  */
1388  ret = SSL_read(session->ssl, session->clean_out.data + session->clean_out.used,
1389  sizeof(session->clean_out.data) - session->clean_out.used);
1390  if (ret > 0) {
1391  session->clean_out.used += ret;
1392  success:
1393  ret = 1;
1394  goto finish;
1395  }
1396 
1397  /*
1398  * Returns 0 if we can continue processing the handshake
1399  * Returns -1 if we encountered a fatal error.
1400  */
1401  if (fr_tls_log_io_error(request, session, ret, "Failed in SSL_read") < 0) goto error;
1402 
1403  /*
1404  * This only occurs once per session, where calling
1405  * SSL_read updates the state of the SSL session, setting
1406  * this flag to true.
1407  *
1408  * Callbacks provide enough info so we don't need to
1409  * print debug statements when the handshake is in other
1410  * states.
1411  */
1412  if (SSL_is_init_finished(session->ssl)) {
1413  SSL_CIPHER const *cipher;
1414  fr_pair_t *vp;
1415  char const *version;
1416 
1417  char cipher_desc[256], cipher_desc_clean[256];
1418  char *p = cipher_desc, *q = cipher_desc_clean;
1419 
1420  cipher = SSL_get_current_cipher(session->ssl);
1421  SSL_CIPHER_description(cipher, cipher_desc, sizeof(cipher_desc));
1422 
1423  /*
1424  * Cleanup the output from OpenSSL
1425  * Seems to print info in a tabular format.
1426  */
1427  while (*p != '\0') {
1428  if (isspace(*p)) {
1429  *q++ = *p;
1430  fr_skip_whitespace(p);
1431  continue;
1432  }
1433  *q++ = *p++;
1434  }
1435  *q = '\0';
1436 
1437  RDEBUG2("Cipher suite: %s", cipher_desc_clean);
1438 
1439  RDEBUG2("Adding TLS session information to request");
1440  vp = fr_pair_afrom_num(request->state_ctx, 0, FR_TLS_SESSION_CIPHER_SUITE);
1441  if (vp) {
1442  fr_pair_value_strdup(vp, SSL_CIPHER_get_name(cipher));
1443  fr_pair_add(&request->state, vp);
1444  RINDENT();
1445  RDEBUG2("&session-state.%pP", vp);
1446  REXDENT();
1447  }
1448 
1449  if (((size_t)session->info.version >= NUM_ELEMENTS(tls_version_str)) ||
1450  !tls_version_str[session->info.version]) {
1451  version = "UNKNOWN";
1452  } else {
1453  version = tls_version_str[session->info.version];
1454  }
1455 
1456  vp = fr_pair_afrom_num(request->state_ctx, 0, FR_TLS_SESSION_VERSION);
1457  if (vp) {
1458  fr_pair_value_strdup(vp, version);
1459  fr_pair_add(&request->state, vp);
1460  RINDENT();
1461  RDEBUG2("&session-state.TLS-Session-Version := \"%s\"", version);
1462  REXDENT();
1463  }
1464 
1465  /*
1466  * Cache the SSL_SESSION pointer.
1467  *
1468  * Which contains all the data we need for session resumption.
1469  */
1470  if (!session->session) {
1471  session->session = SSL_get_session(session->ssl);
1472  if (!session->session) {
1473  REDEBUG("Failed getting TLS session");
1474  goto error;
1475  }
1476  }
1477 
1478  if (RDEBUG_ENABLED3) {
1479  BIO *ssl_log;
1480 
1481  MEM(ssl_log = BIO_new(BIO_s_mem()));
1482 
1483  if (SSL_SESSION_print(ssl_log, session->session) == 1) {
1484  FR_OPENSSL_DRAIN_ERROR_QUEUE(RDEBUG3, "", ssl_log);
1485  } else {
1486  RDEBUG3("Failed retrieving session data");
1487  }
1488  BIO_free(ssl_log);
1489  }
1490 
1491  /*
1492  * Session was resumed, add attribute to mark it as such.
1493  */
1494  if (SSL_session_reused(session->ssl)) {
1495  /*
1496  * Mark the request as resumed.
1497  */
1498  MEM(pair_update_request(&vp, attr_eap_session_resumed) >= 0);
1499  vp->vp_bool = true;
1500  }
1501  }
1502 
1503  /*
1504  * Get data to pack and send back to the TLS peer.
1505  */
1506  ret = BIO_ctrl_pending(session->from_ssl);
1507  if (ret > 0) {
1508  ret = BIO_read(session->from_ssl, session->dirty_out.data,
1509  sizeof(session->dirty_out.data));
1510  if (ret > 0) {
1511  session->dirty_out.used = ret;
1512  } else if (BIO_should_retry(session->from_ssl)) {
1513  record_init(&session->dirty_in);
1514  RDEBUG2("Asking for more data in tunnel");
1515  goto success;
1516 
1517  } else {
1518  fr_tls_log_error(NULL, NULL);
1519  record_init(&session->dirty_in);
1520  goto error;
1521  }
1522  } else {
1523  /* Its clean application data, do whatever we want */
1524  record_init(&session->clean_out);
1525  }
1526 
1527  /*
1528  * Trash the current data in dirty_out, and synthesize
1529  * a TLS error record.
1530  *
1531  * OpensSL annoyingly provides no mechanism for us to
1532  * send alerts, and we need to send alerts as part of
1533  * RFC 5216, so this is our only option.
1534  */
1535  if (session->pending_alert) fr_tls_session_alert_send(request, session);
1536 
1537  /* We are done with dirty_in, reinitialize it */
1538  record_init(&session->dirty_in);
1539 
1540 finish:
1541  fr_tls_session_request_unbind(session->ssl);
1542 
1543  return ret;
1544 }

1671  fr_assert(conf->ctx_count > 0);
1672 
1673  RDEBUG2("Initiating new TLS session");
1674 
1675  ssl_ctx = conf->ctx[(conf->ctx_count == 1) ? 0 : conf->ctx_next++ % conf->ctx_count]; /* mutex not needed */
1676  fr_assert(ssl_ctx);
1677 
1678  new_tls = SSL_new(ssl_ctx);
1679  if (new_tls == NULL) {
1680  fr_tls_log_error(request, "Error creating new TLS session");
1681  return NULL;
1682  }
1683

1685  if (session == NULL) {

1689  }

1691  session->ctx = ssl_ctx;

1693  talloc_set_destructor(session, _fr_tls_session_free);

1699  */

1707  * SSL. This is *very important* as we want to handle
1708  * the transmission part. Now the only IO interface
1709  * that SSL is aware of, is our defined BIO buffers.
1710  *
1711  * This means that all SSL IO is done to/from memory,
1712  * and we can update those BIOs from the packets we've
1713  * received.
1714  */

1716  MEM(session->from_ssl = BIO_new(BIO_s_mem()));

1718

1720  * Add the message callback to identify what type of
1721  * message/handshake is passed
1722  */

1729  * This is to prevent sessions being created by one application
1730  * and used by another. In our case it prevents sessions being
1731  * reused between modules, or TLS server components such as
1732  * RADSEC.
1733  *
1734  * A context must always be set when doing session resumption
1735  * otherwise session resumption will fail.
1736  *
1737  * As the context ID must be <= 32, we digest the context
1738  * data with sha256.
1739  */

1744  uint8_t digest[SHA256_DIGEST_LENGTH];
1745

1760  EVP_DigestFinal_ex(md_ctx, digest, NULL);
1761  EVP_MD_CTX_destroy(md_ctx);

1782  fr_tls_log_error(request, "Failed loading TLS session certificate \"%s\"",

1785  }

1788  fr_tls_log_error(request, "Failed validating TLS session certificate \"%s\"",
1789  vp->vp_strvalue);

1791  }

1793  * Better to perform explicit checks, than rely
1794  * on OpenSSL's opaque error messages.
1795  */
1796  } else {
1797  if (!conf->chains || !conf->chains[0]->private_key_file) {
1798  ERROR("TLS Server requires a private key file");
1799  goto error;
1800  }
1801 
1802  if (!conf->chains || !conf->chains[0]->certificate_file) {
1803  ERROR("TLS Server requires a certificate file");
1804  goto error;
1805  }
1806  }

1809  * In Server mode we only accept.
1810  *
1811  * This sets up the SSL session to work correctly with
1812  * fr_tls_session_handhsake.
1813  */
1814  SSL_set_accept_state(session->ssl);
1815 
1816  /*
1817  * Verify the peer certificate, if asked.
1818  */
1819  if (client_cert) {
1820  RDEBUG2("Setting verify mode to require certificate from client");
1821  verify_mode = SSL_VERIFY_PEER;
1822  verify_mode |= SSL_VERIFY_FAIL_IF_NO_PEER_CERT;
1823  verify_mode |= SSL_VERIFY_CLIENT_ONCE;
1824  }
1825  SSL_set_verify(session->ssl, verify_mode, fr_tls_validate_cert_cb);
1826 
1827  SSL_set_ex_data(session->ssl, FR_TLS_EX_INDEX_CONF, (void *)conf);
1828  SSL_set_ex_data(session->ssl, FR_TLS_EX_INDEX_TLS_SESSION, (void *)session);
1829 
1830  /*
1831  * We use default fragment size, unless the Framed-MTU
1832  * tells us it's too big. Note that we do NOT account
1833  * for the EAP-TLS headers if conf->fragment_size is
1834  * large, because that config item looks to be confusing.
1835  *
1836  * i.e. it should REALLY be called MTU, and the code here
1837  * should figure out what that means for TLS fragment size.
