30 unlang_action_t unlang_group(rlm_rcode_t *p_result, request_t *request)
31 {
32  unlang_stack_t *stack = request->stack;
33  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
34  unlang_t *instruction = frame->instruction;
35  unlang_group_t *g;
36 
37  g = unlang_generic_to_group(instruction);
38 
39  /*
40  * The compiler catches most of these, EXCEPT for the
41  * top-level 'recv Access-Request' etc. Which can exist,
42  * and can be empty.
43  */
44  if (!g->children) {
45  RDEBUG2("} # %s ... <ignoring empty subsection>", instruction->debug_name);
46  return UNLANG_ACTION_EXECUTE_NEXT;
47  }
48 
49  if (unlang_interpret_push(request, g->children, frame->result, UNLANG_NEXT_SIBLING, UNLANG_SUB_FRAME) < 0) {
50  *p_result = RLM_MODULE_FAIL;
51  return UNLANG_ACTION_STOP_PROCESSING;
52  }
53 
54  return UNLANG_ACTION_PUSHED_CHILD;
55 }

57 static unlang_action_t unlang_policy(rlm_rcode_t *result, request_t *request)
58 {
59  unlang_stack_t *stack = request->stack;
60  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
61 
62  /*
63  * Ensure returns stop at the enclosing policy
64  */
65  return_point_set(frame);
66 
67  return unlang_group(result, request);
