68 /** Get a new our_realtime value
69  *
70  * Should be done regularly to adjust for changes in system time.
71  *
72  * @return
73  * - 0 on success.
74  * - -1 on failure.
75  */
76 int fr_time_sync(void)
77 {
78  struct tm tm;
79  time_t now;
80 
81  /*
82  * our_realtime represents system time
83  * at the start of our epoch.
84  *
85  * So to convert a realtime timeval
86  * to fr_time we just subtract
87  * our_realtime from the timeval,
88  * which leaves the number of nanoseconds
89  * elapsed since our epoch.
90  */
91 #ifdef HAVE_CLOCK_GETTIME
92  {
93  struct timespec ts_realtime, ts_monotime;
94 
95  /*
96  * Call these consecutively to minimise drift...
97  */
98  if (clock_gettime(CLOCK_REALTIME, &ts_realtime) < 0) return -1;
99  if (clock_gettime(CLOCK_MONOTONIC, &ts_monotime) < 0) return -1;
100 
101  atomic_store_explicit(&our_realtime,
102  fr_time_delta_from_timespec(&ts_realtime) -
103  (fr_time_delta_from_timespec(&ts_monotime) - our_epoch),
104  memory_order_release);
105 
106  now = ts_realtime.tv_sec;
107  }
108 #else
109  {
110  struct timeval tv_realtime;
111  uint64_t monotime;
112 
113  /*
114  * Call these consecutively to minimise drift...
115  */
116  (void) gettimeofday(&tv_realtime, NULL);
117  monotime = mach_absolute_time();
118 
119  atomic_store_explicit(&our_realtime,
120  fr_time_delta_from_timeval(&tv_realtime) -
121  (monotime - our_mach_epoch) * (timebase.numer / timebase.denom,
122  memory_order_release));
123 
124  now = tv_realtime.tv_sec;
125  }
126 #endif
127 
128  /*
129  * Get local time zone name, daylight savings, and GMT
130  * offsets.
131  */
132  (void) localtime_r(&now, &tm);
133 
134  isdst = (tm.tm_isdst != 0);
135  tz_names[isdst] = tm.tm_zone;
136  gmtoff[isdst] = tm.tm_gmtoff * NSEC; /* they store seconds, we store nanoseconds */
137 
138  return 0;
139 }

169 /** Return a relative time since the server our_epoch
170  *
171  * This time is useful for doing time comparisons, deltas, etc.
172  * Human (i.e. printable) time is something else.
173  *
174  * @returns fr_time_t time in nanoseconds since the server our_epoch.
175  *
176  * @hidecallergraph
177  */
178 fr_time_t fr_time(void)
179 {
180 #ifdef HAVE_CLOCK_GETTIME
181  struct timespec ts;
182  (void) clock_gettime(CLOCK_MONOTONIC, &ts);
183  return fr_time_delta_from_timespec(&ts) - our_epoch;
184 #else /* __MACH__ is defined */
185  uint64_t when;
186 
187  when = mach_absolute_time();
188  when -= our_mach_epoch;
189 
190  return when * (timebase.numer / timebase.denom);
191 #endif
192 }

312 /** Return time delta from the time zone.
313  *
314  * Returns the delta between UTC and the timezone specified by tz
315  *
316  * @param[in] tz time zone name
317  * @param[out] delta the time delta
318  * @return
319  * - 0 converted OK
320  * - <0 on error
321  *
322  * @note This function ONLY handles a limited number of time
323  * zones: local and gmt. It is impossible in general to parse
324  * arbitrary time zone strings, as there are duplicates.
325  */
326 int fr_time_delta_from_time_zone(char const *tz, fr_time_delta_t *delta)
327 {
328  *delta = 0;
329 
330  if ((strcmp(tz, "UTC") == 0) ||
331  (strcmp(tz, "GMT") == 0)) {
332  return 0;
333  }
334 
335  /*
336  * Our local time zone OR time zone with daylight savings.
337  */
338  if (tz_names[0] && (strcmp(tz, tz_names[0]) == 0)) {
339  *delta = gmtoff[0];
340  return 0;
341  }
342 
343  if (tz_names[1] && (strcmp(tz, tz_names[1]) == 0)) {
344  *delta = gmtoff[1];
345  return 0;
346  }
347 
348  return -1;
349 }

351 /** Create fr_time_delta_t from a string
352  *
353  * @param[out] out Where to write fr_time_delta_t
354  * @param[in] in String to parse.
355  * @param[in] hint scale for the parsing. Default is "seconds"
356  * @return
357  * - 0 on success.
358  * - -1 on failure.
359  */
360 int fr_time_delta_from_str(fr_time_delta_t *out, char const *in, fr_time_res_t hint)
361 {
362  int sec;
363  char *p, *end;
364  fr_time_delta_t delta;
365 
366  sec = strtoul(in, &end, 10);
367  if (in == end) {
368  failed:
369  fr_strerror_printf("Failed parsing \"%s\" as time_delta", in);
370  return -1;
371  }
372 
373  /*
374  * Cast before multiplication to avoid integer overflow
375  * in 'int' seconds.
376  */
377  delta = sec;
378  delta *= NSEC;
379 
380  /*
381  * Allow "1ns", etc.
382  */
383  if ((*end >= 'a') && (*end <= 'z')) {
384  p = end;
385  goto parse_precision;
386  }
387 
388  /*
389  * Decimal number
390  */
391  if (*end == '.') {
392  int len;
393 
394  len = sec = 0;
395 
396  end++;
397  p = end;
398 
399  /*
400  * Parse the decimal portion of a number like "0.1".
401  */
402  while ((*p >= '0') && (*p <= '9')) {
403  if (len > 9) {
404  fr_strerror_printf("Too much precision for time_delta");
405  }
406 
407  sec *= 10;
408  sec += *p - '0';
409  p++;
410  len++;
411  }
412 
413  /*
414  * We've just parsed "0.1" as "1". We need to
415  * shift it left to convert it to nanoseconds.
416  */
417  while (len < 9) {
418  sec *= 10;
419  len++;
420  }
421 
422  delta += sec;
423 
424  parse_precision:
425  /*
426  * Nothing else, it defaults to whatever scale the caller passed.
427  */
428  if (!*p) goto do_scale;
429 
430  if ((p[0] == 's') && !p[1]) goto done;
431 
432  /*
433  * Everything else has "ms" or "us" or "ns".
434  *
435  * "1.1ns" means "parse it as 1.1s, and then
436  * shift it right 9 orders of magnitude to
437  * convert it to nanoseconds.
438  */
439  if ((p[1] == 's') && (p[2] == '\0')) {
440  if (p[0] == 'm') {
441  delta /= 1000;
442  goto done;
443  }
444 
445  if (p[0] == 'u') {
446  delta /= 1000000;
447  goto done;
448  }
449 
450  if (p[0] == 'n') {
451  delta /= NSEC;
452  goto done;
453  }
454 
455  error:
456  fr_strerror_printf("Invalid time qualifier in \"%s\"", p);
457  return -1;
458  }
459 
460  if ((p[0] == 'm') && !p[1]) {
461  delta *= 60;
462  goto done;
463  }
464 
465  if ((p[0] == 'h') && !p[1]) {
466  delta *= 3600;
467  goto done;
468  }
469 
470  if ((p[0] == 'd') && !p[1]) {
471  delta *= 86400;
472  goto done;
473  }
474 
475  goto error;
476 
477  } else if (*end == ':') {
478  /*
479  * 00:01 is at least minutes, potentially hours
480  */
481  int minutes = sec;
482 
483  p = end + 1;
484  sec = strtoul(p, &end, 10);
485  if (p == end) goto failed;
486 
487  if (*end) goto failed;
488 
489  if (sec > 60) {
490  fr_strerror_printf("Too many seconds in \"%s\"", in);
491  return -1;
492  }
493 
494  if (minutes > 60) {
495  fr_strerror_printf("Too many minutes in \"%s\"", in);
496  return -1;
497  }
498 
499  /*
500  * @todo - do we want to allow decimals, as in
501  * "1:30.5"? Perhaps not for now.
502  *
503  * @todo - support hours, maybe. Even though
504  * pretty much nothing needs them right now.
505  */
506  if (*end) goto failed;
507 
508  delta = minutes * 60 + sec;
509  delta *= NSEC;
510 
511  } else if (!*end) {
512  do_scale:
513  switch (hint) {
514  case FR_TIME_RES_SEC:
515  break;
516 
517  case FR_TIME_RES_MSEC:
518  delta /= 1000;
519  break;
520 
521  case FR_TIME_RES_USEC:
522  delta /= 1000000;
523  break;
524 
525  case FR_TIME_RES_NSEC:
526  delta /= 1000000000;
527  break;
528 
529  default:
530  fr_strerror_printf("Invalid hint %d for time delta", hint);
531  return -1;
532  }
533 
534  } else {
535  goto failed;
536  }
537 
538 done:
539  *out = delta;
540 
541  return 0;
542 }

594 void fr_time_elapsed_update(fr_time_elapsed_t *elapsed, fr_time_t start, fr_time_t end)
595 {
596  fr_time_t delay;
597 
598  if (start >= end) {
599  delay = 0;
600  } else {
601  delay = end - start;
602  }
603 
604  if (delay < 1000) { /* microseconds */
605  elapsed->array[0]++;
606 
607  } else if (delay < 10000) {
608  elapsed->array[1]++;
609 
610  } else if (delay < 100000) {
611  elapsed->array[2]++;
612 
613  } else if (delay < 1000000) { /* milliseconds */
614  elapsed->array[3]++;
615 
616  } else if (delay < 10000000) {
617  elapsed->array[4]++;
618 
619  } else if (delay < (fr_time_t) 100000000) {
620  elapsed->array[5]++;
621 
622  } else if (delay < (fr_time_t) 1000000000) { /* seconds */
623  elapsed->array[6]++;
624 
625  } else { /* tens of seconds or more */
626  elapsed->array[7]++;
627 
628  }
629 }

672 /*
673  * Based on https://blog.reverberate.org/2020/05/12/optimizing-date-algorithms.html
674  */
675 time_t fr_time_from_utc(struct tm *tm)
676 {
677  static const uint16_t month_yday[12] = {0, 31, 59, 90, 120, 151,
678  181, 212, 243, 273, 304, 334};
679 
680  uint32_t year_adj = tm->tm_year + 4800 + 1900; /* Ensure positive year, multiple of 400. */
681  uint32_t febs = year_adj - (tm->tm_mon <= 2 ? 1 : 0); /* Februaries since base. */
682  uint32_t leap_days = 1 + (febs / 4) - (febs / 100) + (febs / 400);
683  uint32_t days = 365 * year_adj + leap_days + month_yday[tm->tm_mon] + tm->tm_mday - 1;
684 
685  /*
686  * 2472692 adjusts the days for Unix epoch. It is calculated as
687  * (365.2425 * (4800 + 1970))
688  */
689  return (days - 2472692) * 86400 + (tm->tm_hour * 3600) + (tm->tm_min * 60) + tm->tm_sec;
690 }
