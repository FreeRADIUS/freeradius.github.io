34 /*
35  * This file implements path-compressed, level-compressed
36  * patricia tries. The original research paper is:
37  *
38  * https://www.nada.kth.se/~snilsson/publications/Dynamic-trie-compression-implementation/
39  *
40  * The functionality has been extended to include intermediate
41  * nodes which consume 0 bits, but which hold user context data.
42  * These intermediate nodes allow for "longest prefix" matching.
43  * For example, in networking, you can have a routing table entry
44  * with 0/0 leading to one destination, and 10/8 leading to a
45  * different one. Looking up an address in the 10/8 network will
46  * return the 10/8 destination. Looking up any other address
47  * will return the default destination.
48  *
49  * In addition, we desire the ability to add and delete nodes
50  * dynamically. In the example given above, this means that
51  * after deleting 10/8, the trie should contain only the 0/0
52  * network and associated destination.
53  *
54  * As of yet, it does not do level compression. This can be
55  * added without (hopefully) too much work. That would require
56  * an additional step to "normalize" the trie.
57  *
58  * This code could be extended to do packet matching, through the
59  * inclusion of "don't care" paths. e.g. parsing an IP header,
60  * where the src/dst IP addresses are 32-bit "don't care" fields.
61  *
62  * It could also be extended via "count" paths, where the path
63  * holds a count that is used in another part of the trie. For
64  * example, in RADIUS. The attribute encoding is one byte
65  * attribute, one byte length, followed by "length - 2" bytes of
66  * data. At that point though, you might as well just use Ragel.
67  */

227 /** Get the longest prefix of the two keys.
228  *
229  */
230 static int fr_trie_key_lcp(uint8_t const *key1, int keylen1, uint8_t const *key2, int keylen2, int start_bit)
231 {
232  int lcp, end_bit;
233 
234  if (!keylen1 || !keylen2) return 0;
235  fr_cond_assert((start_bit & 0x07) == start_bit);
236 
237  end_bit = keylen1;
238  if (end_bit > keylen2) end_bit = keylen2;
239  end_bit += start_bit;
240 
241  MPRINT2("%.*sLCP %02x%02x %02x%02x start %d length %d, %d\n",
242  start_bit, spaces, key1[0], key1[1], key2[0], key2[1], start_bit, keylen1, keylen2);
243 
244  lcp = 0;
245 
246  while (end_bit > 0) {
247  int num_bits;
248  uint8_t cmp1, cmp2, xor;
249 
250  MPRINT2("END %d\n", end_bit);
251 
252  /*
253  * Default to grabbing the whole byte.
254  */
255  cmp1 = key1[0];
256  cmp2 = key2[0];
257  num_bits = 8;
258 
259  /*
260  * The LCP ends in this byte. Mask off the
261  * trailing bits so that they don't affect the
262  * result.
263  */
264  if (end_bit < 8) {
265  cmp1 &= used_bit_mask[end_bit - 1];
266  cmp2 &= used_bit_mask[end_bit - 1];
267  num_bits = end_bit;
268  }
269 
270  /*
271  * The key doesn't start on the leading bit.
272  * Shift the data left until it does start there.
273  */
274  if ((start_bit & 0x07) != 0) {
275  cmp1 <<= start_bit;
276  cmp2 <<= start_bit;
277  num_bits -= start_bit;
278  end_bit -= start_bit;
279 
280  /*
281  * For subsequent bytes we start on a
282  * byte boundary.
283  */
284  start_bit = 0;
285  }
286 
287  xor = cmp1 ^ cmp2;
288 
289  /*
290  * A table lookup is faster than looping through
291  * the bits. If the LCP is smaller than the
292  * number of bits we're looking up, we can stop.
293  *
294  * On the other hand, if it returns the same or
295  * too many bits, just do another round through
296  * the loop, so that we can update the pointers
297  * and check the exit conditions.
298  */
299  if (xor2lcp[xor] < num_bits) {
300  MPRINT2("RETURN %d + %d\n", lcp, xor2lcp[xor]);
301  return lcp + xor2lcp[xor];
302  }
303 
304  /*
305  * The LCP may be longer than num_bits if we're
306  * checking the first byte, which has only
307  * "start_bit" things we care about. Ignore that
308  * case, and just keep going.
309  */
310 
311  lcp += num_bits;
312  end_bit -= num_bits;
313  key1++;
314  key2++;
315  }
316 
317  return lcp;
318 }

339 /** Return a chunk of a key (in the low bits) for use in 2^N node de-indexing
340  *
341  */
342 static uint16_t get_chunk(uint8_t const *key, int start_bit, int num_bits)
343 {
344  uint16_t chunk;
345  int end_bit;
346 
347  fr_cond_assert(num_bits > 0);
348  fr_cond_assert(num_bits <= 16);
349 
350  /*
351  * Normalize it so that the caller doesn't have to.
352  */
353  if (start_bit > 7) {
354  key += BYTEOF(start_bit);
355  start_bit &= 0x07;
356  }
357 
358  /*
359  * Special-case 1-bit lookups.
360  */
361  if (num_bits == 1) {
362  chunk = key[0] >> (7 - start_bit);
363  chunk &= 0x01;
364  return chunk;
365  }
366 
367  /*
368  * Catch some simple use-cases.
369  */
370  if (start_bit == 0) {
371  if (num_bits < 7) return key[0] >> (8 - num_bits);
372  if (num_bits == 8) return key[0];
373  if (num_bits == 16) return (key[0] << 8) | key[1];
374  return ((key[0] << 8) | key[1]) >> (16 - num_bits);
375  }
376 
377  /*
378  * Load the first byte and mask off the bits we don't
379  * want.
380  */
381  chunk = key[0] & start_bit_mask[start_bit & 0x07];
382 
383  fr_cond_assert(BYTEOF(start_bit + num_bits - 1) <= 1);
384 
385  if (BYTEOF(start_bit + num_bits - 1) != 0) {
386  chunk <<= 8;
387  chunk |= key[1];
388  }
389 
390  /*
391  * The bits we want are now all in the higher bits
392  * of "chunk". But we only want some of them.
393  *
394  * Shift the chunk so that the bits we want are now in
395  * the low bits.
396  */
397  end_bit = (start_bit + num_bits) & 0x07;
398  if (end_bit != 0) chunk >>= 8 - end_bit;
399 
400  fr_cond_assert(chunk < (1 << num_bits));
401 
402  return chunk;
403 }

408 /** Write a chunk to an output buffer
409  *
410  */
411 static void write_chunk(uint8_t *out, int start_bit, int num_bits, uint16_t chunk)
412 {
413  fr_cond_assert(chunk < (1 << num_bits));
414 
415  /*
416  * Normalize it so that the caller doesn't have to.
417  */
418  if (start_bit > 7) {
419  out += BYTEOF(start_bit);
420  start_bit &= 0x07;
421  }
422 
423  /*
424  * Special-case 1-bit writes.
425  */
426  if (num_bits == 1) {
427  out[0] &= ~((1 << (7 - start_bit)) - 1);
428  out[0] |= chunk << (7 - start_bit);
429  return;
430  }
431 
432  /*
433  * Ensure that we don't write to more than 2 octets at
434  * the same time.
435  */
436  fr_cond_assert((start_bit + num_bits) <= 16);
437 
438  /*
439  * Shift the chunk to the high bits, but leave room for
440  * start_bit
441  */
442  if ((start_bit + num_bits) < 16) chunk <<= (16 - (start_bit + num_bits));
443 
444  /*
445  * Mask off the first bits that are already in the
446  * output. Then OR in the relevant bits of "chunk".
447  */
448  out[0] &= (used_bit_mask[start_bit] << 1);
449  out[0] |= chunk >> 8;
450 
451  if ((start_bit + num_bits) > 8) {
452  out[1] = chunk & 0xff;
453  }
454 }

763 static fr_trie_node_t *fr_trie_node_split(TALLOC_CTX *ctx, fr_trie_node_t *node, int bits)
764 {
765  fr_trie_node_t *split;
766  int i, remaining_bits;
767 
768  /*
769  * Can't split zero bits, more bits than the node has, or
770  * a node which has 1 bit.
771  */
772  if ((bits == 0) || (bits >= node->bits) || (node->bits == 1)) {
773  fr_strerror_printf("invalid value for node split (%d / %d)", bits, node->bits);
774  return NULL;
775  }
776 
777  split = fr_trie_node_alloc(ctx, bits);
778  if (!split) return NULL;
779 
780  remaining_bits = node->bits - bits;
781 
782  /*
783  * Allocate the children. For now, just brute-force all
784  * of the children. We take a later pass at optimizing this.
785  */
786  for (i = 0; i < (1 << bits); i++) {
787  int j;
788  fr_trie_node_t *child;
789 
790  child = fr_trie_node_alloc(ctx, remaining_bits);
791  if (!child) {
792  fr_trie_free((fr_trie_t *) split);
793  return NULL;
794  }
795 
796  for (j = 0; j < (1 << remaining_bits); j++) {
797  if (!node->trie[(i << remaining_bits) + j]) continue;
798 
799  child->trie[j] = node->trie[(i << remaining_bits) + j];
800  node->trie[(i << remaining_bits) + j] = NULL; /* so we don't free it when freeing 'node' */
801  child->used++;
802  }
803 
804  if (!child->used) {
805  talloc_free(child); /* no children, so no need to recurse */
806  continue;
807  }
808 
809  split->trie[i] = (fr_trie_t *) child;
810  split->used++;
811  }
812 
813  /*
814  * Note that we do NOT free "node". The caller still
815  * needs it for some activies.
816  */
817  return split;
818 }
