63 int isup_scan_for_reset(struct ss7_application *app, struct msgb *msg)
64 {
65  struct isup_msg_hdr *hdr;
66  int range;
67  uint16_t cic;
68 
69  /* too small for an isup message? */
70  if (msgb_l3len(msg) < sizeof(*hdr)) {
71  LOGP(DISUP, LOGL_ERROR, "Message too small for the header\n");
72  return -1;
73  }
74 
75  /* no trunk name, don't bother forwarding */
76  if (!app->trunk_name) {
77  LOGP(DISUP, LOGL_DEBUG,
78  "No trunk name defined for: %s\n", app->name);
79  return 0;
80  }
81 
82  hdr = (struct isup_msg_hdr *) msg->l3h;
83  cic = isup_cic_to_local(hdr);
84 
85  switch (hdr->msg_type) {
86  case ISUP_MSG_GRS:
87  range = isup_parse_status(&hdr->data[0],
88  msgb_l3len(msg) - sizeof(*hdr));
89  if (range <= 0) {
90  LOGP(DISUP, LOGL_ERROR,
91  "Failed to parse range on app %s\n", app->name);
92  return -1;
93  }
94 
95  LOGP(DISUP, LOGL_DEBUG,
96  "Going to reset ISUP for app %s, cic %d range %d\n",
97  app->name, cic, range);
98  reset_cics(app, cic, range);
99  break;
100  case ISUP_MSG_RSC:
101  LOGP(DISUP, LOGL_DEBUG,
102  "Going to reset single CIC %d on app %s\n",
103  cic, app->name);
104  reset_cic(app, cic);
105  break;
106  }
107 
108  return 0;
109 }
