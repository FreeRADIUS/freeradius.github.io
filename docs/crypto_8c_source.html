47 /** Generate keys according to RFC 2716 and add to the reply
48  *
49  */
50 int eap_crypto_mppe_keys(request_t *request, SSL *ssl, char const *prf_label, size_t prf_label_len)
51 {
52  uint8_t out[4 * EAP_TLS_MPPE_KEY_LEN];
53  uint8_t *p;
54 
55  if (SSL_export_keying_material(ssl, out, sizeof(out), prf_label, prf_label_len, NULL, 0, 0) != 1) {
56  fr_tls_log_error(request, "Failed generating MPPE keys");
57  return -1;
58  }
59 
60  if (RDEBUG_ENABLED3) {
61  uint8_t random[SSL3_RANDOM_SIZE];
62  size_t random_len;
63  uint8_t master_key[SSL_MAX_MASTER_KEY_LENGTH];
64  size_t master_key_len;
65 
66  RDEBUG3("Key Derivation Function input");
67  RINDENT();
68  RDEBUG3("prf label : %pV", fr_box_strvalue_len(prf_label, prf_label_len));
69  master_key_len = SSL_SESSION_get_master_key(SSL_get_session(ssl), master_key, sizeof(master_key));
70  RDEBUG3("master session key : %pH", fr_box_octets(master_key, master_key_len));
71  random_len = SSL_get_client_random(ssl, random, SSL3_RANDOM_SIZE);
72  RDEBUG3("client random : %pH", fr_box_octets(random, random_len));
73  random_len = SSL_get_server_random(ssl, random, SSL3_RANDOM_SIZE);
74  RDEBUG3("server random : %pH", fr_box_octets(random, random_len));
75  REXDENT();
76  }
77 
78  RDEBUG2("Adding session keys");
79  p = out;
80  eap_add_reply(request, attr_ms_mppe_recv_key, p, EAP_TLS_MPPE_KEY_LEN);
81  p += EAP_TLS_MPPE_KEY_LEN;
82  eap_add_reply(request, attr_ms_mppe_send_key, p, EAP_TLS_MPPE_KEY_LEN);
83 
84  eap_add_reply(request, attr_eap_msk, out, 64);
85  eap_add_reply(request, attr_eap_emsk, out + 64, 64);
86 
87  return 0;
88 }
