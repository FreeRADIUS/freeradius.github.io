31 /** Merge connection and call specific client and server controls
32  *
33  * LDAP_OPT_CLIENT_CONTROLS and LDAP_OPT_SERVER_CONTROLS are useless
34  * because they're overriden in their entirety if any call specific
35  * controls are specified.
36  *
37  * @param[out] serverctrls_out Where to write serverctrls.
38  * @param[out] clientctrls_out Where to write clientctrls.
39  * @param[in] serverctrls_len length of serverctrls array.
40  * @param[in] clientctrls_len length of clientctrls array.
41  * @param[in] conn to get controls from.
42  * @param[in] serverctrls_in from arguments.
43  * @param[in] clientctrls_in from_arguments.
44  */
45  void fr_ldap_control_merge(LDAPControl *serverctrls_out[],
46  LDAPControl *clientctrls_out[],
47  size_t serverctrls_len,
48  size_t clientctrls_len,
49  fr_ldap_connection_t *conn,
50  LDAPControl *serverctrls_in[],
51  LDAPControl *clientctrls_in[])
52 {
53  size_t i, num_serverctrls = 0, num_clientctrls = 0;
54 
55  if (serverctrls_in) {
56  for (i = 0; serverctrls_in[i] && (num_serverctrls < LDAP_MAX_CONTROLS); i++) {
57  serverctrls_out[num_serverctrls++] = serverctrls_in[i];
58  }
59  }
60 
61  if (clientctrls_in) {
62  for (i = 0; clientctrls_in[i] && (num_clientctrls < LDAP_MAX_CONTROLS); i++) {
63  clientctrls_out[num_clientctrls++] = clientctrls_in[i];
64  }
65  }
66 
67  for (i = 0; (i < (size_t)conn->serverctrls_cnt) && (num_serverctrls < serverctrls_len); i++) {
68  serverctrls_out[num_serverctrls++] = conn->serverctrls[i].control;
69  }
70 
71  for (i = 0; (i < (size_t)conn->clientctrls_cnt) && (num_clientctrls < clientctrls_len); i++) {
72  clientctrls_out[num_clientctrls++] = conn->clientctrls[i].control;
73  }
74 
75  serverctrls_out[num_serverctrls] = NULL;
76  clientctrls_out[num_clientctrls] = NULL;
77 }

79 /** Add a serverctrl to a connection handle
80  *
81  * All internal LDAP functions will pass this serverctrl to the server.
82  *
83  * @param conn to add control to.
84  * @param ctrl to add.
85  * @param freeit Whether the control should be freed when the handle is released or closed.
86  * @return
87  * - 0 on success.
88  * - -1 on failure (exceeded maximum controls).
89  */
90 int fr_ldap_control_add_server(fr_ldap_connection_t *conn, LDAPControl *ctrl, bool freeit)
91 {
92  if ((size_t)conn->serverctrls_cnt >= ((NUM_ELEMENTS(conn->serverctrls)) - 1)) {
93  return -1;
94  }
95 
96  conn->serverctrls[conn->serverctrls_cnt].control = ctrl;
97  conn->serverctrls[conn->serverctrls_cnt].freeit = freeit;
98  conn->serverctrls_cnt++;
99 
100  return 0;
101 }

127 /** Clear and free any controls associated with a connection
128  *
129  * @param conn to clear controls from.
130  */
131 void fr_ldap_control_clear(fr_ldap_connection_t *conn)
132 {
133  int i;
134 
135  for (i = 0; i < conn->serverctrls_cnt; i++) {
136  if (conn->serverctrls[i].freeit) ldap_control_free(conn->serverctrls[i].control);
137  conn->serverctrls[i].freeit = false;
138  conn->serverctrls[i].control = NULL;
139  }
140  conn->serverctrls_cnt = 0;
141 
142  for (i = 0; i < conn->clientctrls_cnt; i++) {
143  if (conn->clientctrls[i].freeit) ldap_control_free(conn->clientctrls[i].control);
144  conn->clientctrls[i].freeit = false;
145  conn->clientctrls[i].control = NULL;
146  }
147  conn->clientctrls_cnt = 0;
148 }

150 #ifdef LDAP_CONTROL_X_SESSION_TRACKING
151 /** Add session controls to a connection as per draft-wahl-ldap-session
152  *
153  * @note the RFC states that the username identifier, must be the authenticated
154  * user id, not the purported one. As order of operations is configurable,
155  * we're going to leave that up to the server admin to satisfy that
156  * requirement
157  *
158  * For once the RFC is pretty helpful about what should be inserted into the
159  * various values, and maps out RADIUS attributes to formatOIDs, so none of
160  * this is configurable.
161  *
162  * @param conn to add controls to.
163  * @param request to draw attributes from.
164  */
165 int fr_ldap_control_add_session_tracking(fr_ldap_connection_t *conn, request_t *request)
166 {
167  /*
168  * The OpenLDAP guys didn't declare the formatOID parameter to
169  * ldap_create_session_tracking_control as const *sigh*.
170  */
171  static char username_oid[] = LDAP_CONTROL_X_SESSION_TRACKING_USERNAME;
172  static char acctsessionid_oid[] = LDAP_CONTROL_X_SESSION_TRACKING_RADIUS_ACCT_SESSION_ID;
173  static char acctmultisessionid_oid[] = LDAP_CONTROL_X_SESSION_TRACKING_RADIUS_ACCT_MULTI_SESSION_ID;
174 
175  int ret;
176 
177  char ipaddress[INET6_ADDRSTRLEN];
178  char *username = NULL;
179  char *acctsessionid = NULL;
180  char *acctmultisessionid = NULL;
181  char *hostname;
182 
183  LDAPControl *username_control = NULL;
184  LDAPControl *acctsessionid_control = NULL;
185  LDAPControl *acctmultisessionid_control = NULL;
186  struct berval tracking_id;
187 
188  fr_cursor_t cursor;
189  fr_pair_t const *vp;
190 
191  memcpy(&hostname, main_config->name, sizeof(hostname)); /* const / non-const issues */
192 
193  for (vp = fr_cursor_init(&cursor, &request->request_pairs);
194  vp;
195  vp = fr_cursor_next(&cursor)) {
196  if (fr_dict_attr_is_top_level(vp->da)) switch (vp->da->attr) {
197  case FR_NAS_IP_ADDRESS:
198  case FR_NAS_IPV6_ADDRESS:
199  fr_pair_print_value_quoted(&FR_SBUFF_OUT(ipaddress, sizeof(ipaddress)), vp, T_BARE_WORD);
200  break;
201 
202  case FR_USER_NAME:
203  memcpy(&username, &vp->vp_strvalue, sizeof(username));
204  break;
205 
206  case FR_ACCT_SESSION_ID:
207  memcpy(&acctsessionid, &vp->vp_strvalue, sizeof(acctsessionid));
208  break;
209 
210  case FR_ACCT_MULTI_SESSION_ID:
211  memcpy(&acctmultisessionid, &vp->vp_strvalue, sizeof(acctmultisessionid));
212  break;
213  }
214  }
215 
216  if (username) {
217  tracking_id.bv_val = username;
218  tracking_id.bv_len = talloc_array_length(username) - 1;
219 
220  ret = ldap_create_session_tracking_control(conn->handle, ipaddress,
221  hostname,
222  username_oid,
223  &tracking_id,
224  &username_control);
225  if (ret != LDAP_SUCCESS) {
226  REDEBUG("Failed creating username session tracking control: %s", ldap_err2string(ret));
227  error:
228  if (username_control) ldap_control_free(username_control);
229  if (acctsessionid_control) ldap_control_free(acctsessionid_control);
230  if (acctmultisessionid_control) ldap_control_free(acctmultisessionid_control);
231  return -1;
232  }
233  }
234 
235  if (acctsessionid) {
236  tracking_id.bv_val = acctsessionid;
237  tracking_id.bv_len = talloc_array_length(acctsessionid) - 1;
238 
239  ret = ldap_create_session_tracking_control(conn->handle, ipaddress,
240  hostname,
241  acctsessionid_oid,
242  &tracking_id,
243  &acctsessionid_control);
244  if (ret != LDAP_SUCCESS) {
245  REDEBUG("Failed creating acctsessionid session tracking control: %s", ldap_err2string(ret));
246  goto error;
247  }
248  }
249 
250  if (acctmultisessionid) {
251  tracking_id.bv_val = acctmultisessionid;
252  tracking_id.bv_len = talloc_array_length(acctmultisessionid) - 1;
253 
254  ret = ldap_create_session_tracking_control(conn->handle, ipaddress,
255  hostname,
256  acctmultisessionid_oid,
257  &tracking_id,
258  &acctmultisessionid_control);
259  if (ret != LDAP_SUCCESS) {
260  REDEBUG("Failed creating acctmultisessionid session tracking control: %s",
261  ldap_err2string(ret));
262  goto error;
263  }
264  }
265 
266  if ((conn->serverctrls_cnt + 3) >= LDAP_MAX_CONTROLS) {
267  REDEBUG("Insufficient space to add session tracking controls");
268  goto error;
269  }
270 
271  if (username_control && (fr_ldap_control_add_server(conn, username_control, true) < 0)) goto error;
272 
273  if (acctsessionid_control && (fr_ldap_control_add_server(conn, acctsessionid_control, true) < 0)) {
274  conn->serverctrls_cnt--;
275  conn->serverctrls[conn->serverctrls_cnt].control = NULL;
276  goto error;
277  }
278 
279  if (acctmultisessionid_control && (fr_ldap_control_add_server(conn, acctmultisessionid_control, true) < 0)) {
280  conn->serverctrls_cnt--;
281  conn->serverctrls[conn->serverctrls_cnt].control = NULL;
282  conn->serverctrls_cnt--;
283  conn->serverctrls[conn->serverctrls_cnt].control = NULL;
284  goto error;
285  }
286 
287  return 0;
288 }
