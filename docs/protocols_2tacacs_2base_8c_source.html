132 int fr_tacacs_body_xor(fr_tacacs_packet_t const *pkt, uint8_t *body, size_t body_len, char const *secret, size_t secret_len)
133 {
134  uint8_t pad[MD5_DIGEST_LENGTH];
135  uint8_t *buf;
136  int pad_offset;
137 
138  if (!secret) {
139  if (pkt->hdr.flags & FR_TAC_PLUS_UNENCRYPTED_FLAG)
140  return 0;
141  else {
142  fr_strerror_printf("Packet is encrypted but no secret for the client is set");
143  return -1;
144  }
145  }
146 
147  if (pkt->hdr.flags & FR_TAC_PLUS_UNENCRYPTED_FLAG) {
148  fr_strerror_printf("Packet is unencrypted but a secret has been set for the client");
149  return -1;
150  }
151 
152  pad_offset = sizeof(pkt->hdr.session_id) + secret_len + sizeof(pkt->hdr.version) + sizeof(pkt->hdr.seq_no);
153 
154  /* MD5_1 = MD5{session_id, key, version, seq_no} */
155  /* MD5_n = MD5{session_id, key, version, seq_no, MD5_n-1} */
156  buf = talloc_array(NULL, uint8_t, pad_offset + MD5_DIGEST_LENGTH);
157 
158  memcpy(&buf[0], &pkt->hdr.session_id, sizeof(pkt->hdr.session_id));
159  memcpy(&buf[sizeof(pkt->hdr.session_id)], secret, secret_len);
160  memcpy(&buf[sizeof(pkt->hdr.session_id) + secret_len], &pkt->hdr.version, sizeof(pkt->hdr.version));
161  memcpy(&buf[sizeof(pkt->hdr.session_id) + secret_len + sizeof(pkt->hdr.version)], &pkt->hdr.seq_no, sizeof(pkt->hdr.seq_no));
162 
163  fr_md5_calc(pad, buf, pad_offset);
164 
165  size_t pos = 0;
166  do {
167  for (size_t i = 0; i < MD5_DIGEST_LENGTH && pos < body_len; i++, pos++)
168  body[pos] ^= pad[i];
169 
170  if (pos == body_len)
171  break;
172 
173  memcpy(&buf[pad_offset], pad, MD5_DIGEST_LENGTH);
174  fr_md5_calc(pad, buf, pad_offset + MD5_DIGEST_LENGTH);
175  } while (1);
176 
177  talloc_free(buf);
178 
179  return 0;
180 }

182 /**
183  * Return how long a TACACS+ packet is
184  *
185  * Note that we only look at the 12 byte packet header. We don't
186  * (yet) do validation on authentication / authorization /
187  * accounting headers. The packet may still be determined later
188  * to be invalid.
189  *
190  * @param buffer to check
191  * @param buffer_len length of the buffer
192  * @return
193  * >0 size of the TACACS+ packet. We want. MAY be larger than "buffer_len"
194  * <=0 error, packet should be discarded.
195  */
196 ssize_t fr_tacacs_length(uint8_t const *buffer, size_t buffer_len)
197 {
198  fr_tacacs_packet_t const *pkt = (fr_tacacs_packet_t const *) buffer;
199  size_t length, want;
200 
201  /*
202  * Check that we have a full TACACS+ header before
203  * decoding anything.
204  */
205  if (buffer_len < sizeof(pkt->hdr)) {
206  return sizeof(pkt->hdr);
207  }
208 
209  /*
210  * TACACS major / minor version MUST be 12.0 or 12.1
211  */
212  if (!((buffer[0] == 0xc0) || (buffer[0] == 0xc1))) {
213  fr_strerror_printf("Unsupported TACACS+ version %02x", buffer[0]);
214  return -1;
215  }
216 
217  /*
218  * There's no reason to accept 64K TACACS+ packets.
219  */
220  if ((buffer[8] != 0) || (buffer[9] != 0)) {
221  fr_strerror_printf("Packet is too large. Our limit is 64K");
222  return -1;
223  }
224 
225  /*
226  * There are only 3 types of packets which are supported.
227  */
228  if (!((pkt->hdr.type == FR_TAC_PLUS_AUTHEN) ||
229  (pkt->hdr.type == FR_TAC_PLUS_AUTHOR) ||
230  (pkt->hdr.type == FR_TAC_PLUS_ACCT))) {
231  fr_strerror_printf("Unknown packet type %u", pkt->hdr.type);
232  return -1;
233  }
234 
235  length = sizeof(pkt->hdr) + ntohl(pkt->hdr.length);
236 
237  if (buffer_len < length) return length;
238 
239  /*
240  * We want at least the headers for the various packet
241  * types. Note that we do NOT check the lengths in the
242  * headers against buffer / buffer_len. That process is
243  * complex and error-prone. It's best to leave it in one
244  * place: fr_tacacs_decode().
245  */
246  switch (pkt->hdr.type) {
247  default:
248  fr_assert(0); /* should have been caught above */
249  return -1;
250 
251  case FR_TAC_PLUS_AUTHEN:
252  if (packet_is_authen_start_request(pkt)) {
253  want = sizeof(pkt->hdr) + sizeof(pkt->authen.start);
254 
255  } else if (packet_is_authen_continue(pkt)) {
256  want = sizeof(pkt->hdr) + sizeof(pkt->authen.cont);
257 
258  } else {
259  fr_assert(packet_is_authen_reply(pkt));
260  want = sizeof(pkt->hdr) + sizeof(pkt->authen.reply);
261  }
262  break;
263 
264  case FR_TAC_PLUS_AUTHOR:
265  if (packet_is_author_request(pkt)) {
266  want = sizeof(pkt->hdr) + sizeof(pkt->author.req);
267  } else {
268  fr_assert(packet_is_author_response(pkt));
269  want = sizeof(pkt->hdr) + sizeof(pkt->author.res);
270  }
271  break;
272 
273  case FR_TAC_PLUS_ACCT:
274  if (packet_is_acct_request(pkt)) {
275  want = sizeof(pkt->hdr) + sizeof(pkt->acct.req);
276  } else {
277  fr_assert(packet_is_acct_reply(pkt));
278  want = sizeof(pkt->hdr) + sizeof(pkt->acct.reply);
279  }
280  break;
281  }
282 
283  if (want > length) {
284  fr_strerror_printf("Packet is too small. Want %zu, got %zu", want, length);
285  return -1;
286  }
287 
288  return length;
289 }
