309 /** Determines whether a list contains the number of boxes required
310  *
311  * @note More efficient than fr_value_box_list_len for argument validation as it
312  * doesn't walk the entire list.
313  *
314  * @param[in] head of the list of value boxes.
315  * @param[in] min The number of boxes required to return true.
316  * @return
317  * - true if the list has at least min boxes.
318  * - false if the list has fewer than min boxes.
319  */
320 static inline bool fr_value_box_list_len_min(fr_value_box_t const *head, size_t min)
321 {
322  size_t i;
323 
324  for (i = 0; head && i < min; head = head->next, i++);
325 
326  return (i == min);
327 }

496  *
497  * @param[in] dst Where to copy the ethernet address to.

503 static inline int fr_value_unbox_ethernet_addr(fr_ethernet_t *dst, fr_value_box_t *src)
504 {
505  if (unlikely(src->type != FR_TYPE_ETHERNET)) { \
