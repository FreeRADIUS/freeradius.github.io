50 /** De-queue curl requests and wake up the requests that initiated them
51  *
52  * @param[in] mhandle containing the event loop and request counter.
53  * @param[in] mandle to dequeue curl easy handles/responses from.
54  */
55 static inline void _fr_curl_io_demux(fr_curl_handle_t *mhandle, CURLM *mandle)
56 {
57  struct CURLMsg *m;
58  int msg_queued = 0;
59 
60  while ((m = curl_multi_info_read(mandle, &msg_queued))) {
61  switch (m->msg) {
62  case CURLMSG_DONE:
63  {
64  fr_curl_io_request_t *randle;
65  request_t *request = NULL;
66  CURL *candle = m->easy_handle;
67  CURLcode ret;
68 
69  fr_assert(candle);
70 
71  mhandle->transfers--;
72 
73  ret = curl_easy_getinfo(candle, CURLINFO_PRIVATE, &randle);
74  if (!fr_cond_assert_msg(ret == CURLE_OK,
75  "Failed retrieving request data from CURL easy handle (candle)")) {
76  curl_multi_remove_handle(mandle, candle);
77  return;
78  }
79  request = randle->request;
80 
81  REQUEST_VERIFY(request);
82 
83  /*
84  * If the request failed, say why...
85  */
86  if (m->data.result != CURLE_OK) {
87  REDEBUG("curl request failed: %s (%i)",
88  curl_easy_strerror(m->data.result), m->data.result);
89  }
90  randle->result = m->data.result;
91 
92  /*
93  * Looks like this needs to be done last,
94  * else m->data.result ends up being junk.
95  */
96  curl_multi_remove_handle(mandle, candle);
97 
98  unlang_interpret_resumable(request);
99  }
100  break;
101 
102  default:
103 #ifndef NDEBUG
104  DEBUG4("Got unknown msg (%i) when dequeueing curl responses", msg_queued);
105 #endif
106  break;
107  }
108  }
109 }

138 /** Service an IO event on a file descriptor
139  *
140  * @param[in] mhandle containing the event loop and request counter.
141  * @param[in] fd the IO event occurred for.
142  * @param[in] event type.
143  */
144 static inline void _fr_curl_io_service(fr_curl_handle_t *mhandle, int fd, int event)
145 {
146  CURLMcode ret;
147  CURLM *mandle = mhandle->mandle;
148  int running = 0;
149 
150  ret = curl_multi_socket_action(mandle, fd, event, &running);
151  if (ret != CURLM_OK) {
152  ERROR("multi-handle %p - Servicing failed - %s (%i)", mandle, curl_multi_strerror(ret), ret);
153  return;
154  }
155 
156  if (DEBUG_ENABLED3) {
157  char const *event_str;
158 
159  switch (event) {
160  case CURL_CSELECT_ERR:
161  event_str = "error";
162  break;
163 
164  case CURL_CSELECT_OUT:
165  event_str = "socket-writable";
166  break;
167 
168  case CURL_CSELECT_IN:
169  event_str = "socket-readable";
170  break;
171 
172  default:
173  event_str = "<INVALID>";
174  break;
175  }
176 
177  DEBUG3("multi-handle %p - Serviced on fd %i event (%s). "
178  "%i request(s) in progress, %" PRIu64 " requests(s) to dequeue",
179  mandle, fd, event_str, running, mhandle->transfers - (uint64_t)running);
180  }
181 
182 
183  _fr_curl_io_demux(mhandle, mandle);
184 }

235 /** Callback called by libcurl to set/unset timers
236  *
237  * Each rlm_fr_curl_thread_t has a timer event which is controller by libcurl.
238  * This allows libcurl to honour timeouts set on requests to remote hosts,
239  * and means we don't need to set timeouts for individual I/O events.
240  *
241  * @param[in] mandle handle requesting the timer be set/unset.
242  * @param[in] timeout_ms If > 0, how long to wait before calling curl_multi_socket_action.
243  * If == 0, we call curl_multi_socket_action as soon as possible.
244  * If < 0, we delete the timer.
245  * @param[in] ctx The rlm_fr_curl_thread_t specific to this thread.
246  * @return
247  * - 0 on success.
248  * - -1 on error.
249  */
250 static int _fr_curl_io_timer_modify(CURLM *mandle, long timeout_ms, void *ctx)
251 {
252  fr_curl_handle_t *mhandle = talloc_get_type_abort(ctx, fr_curl_handle_t);
253 
254  if (timeout_ms < 0) {
255  if (fr_event_timer_delete(&mhandle->ev) < 0) {
256  PERROR("Failed deleting multi-handle timer");
257  return -1;
258  }
259  DEBUG3("multi-handle %p - Timer removed", mandle);
260  return 0;
261  }
262 
263  DEBUG3("multi-handle %p will need servicing in %li ms", mandle, timeout_ms);
264 
265  /*
266  * https://curl.haxx.se/libcurl/c/CURLMOPT_TIMERFUNCTION.html
267  *
268  * WARNING: even if it feels tempting, avoid calling libcurl directly from within the
269  * callback itself when the timeout_ms value is zero, since it risks triggering an
270  * unpleasant recursive behavior that immediately calls another call to the callback
271  * with a zero timeout...
272  *
273  * Setting a timeout of zero when calling fr_event_timer_in should result in the event
274  * repeating at most twice during one iteration of the event loop.
275  *
276  * In a previous version of this code we called curl_multi_socket_action immediately
277  * if timeout_ms was 0. It was observed that this lead to this callback being called
278  * ~665 times per request which is why we no longer do that.
279  */
280  if (fr_event_timer_in(mhandle, mhandle->el, &mhandle->ev,
281  fr_time_delta_from_msec(timeout_ms), _fr_curl_io_timer_expired, mhandle) < 0) return -1;
282  return 0;
283 }

285 /** Called by libcurl to register a socket that it's intefr_curled in receiving IO events for
286  *
287  *
288  * @param[in] easy handle this fd relates to.
289  * @param[in] fd File descriptor curl wants to be notified about.
290  * @param[in] what Which events libcurl wants to be notified of, may be one of:
291  * - CURL_POLL_IN Wait for incoming data. For the socket
292  * to become readable.
293  * - CURL_POLL_OUT Wait for outgoing data. For the socket
294  * to become writable.
295  * - CURL_POLL_INOUT Wait for incoming and outgoing data.
296  * For the socket to become readable or writable.
297  * - CURL_POLL_REMOVE The specified socket/file descriptor is no
298  * longer used by libcurl.
299  * @param[in] ctx The fr_curl_handle_t specific to this thread.
300  * @param[in] fd_ctx Private data associated with the socket.
301  */
302 static int _fr_curl_io_event_modify(UNUSED CURL *easy, curl_socket_t fd, int what, void *ctx, UNUSED void *fd_ctx)
303 {
304  fr_curl_handle_t *mhandle = talloc_get_type_abort(ctx, fr_curl_handle_t);
305 
306  switch (what) {
307  case CURL_POLL_IN:
308  if (fr_event_fd_insert(mhandle, mhandle->el, fd,
309  _fr_curl_io_service_readable,
310  NULL,
311  _fr_curl_io_service_errored,
312  mhandle) < 0) {
313  PERROR("multi-handle %p registration failed for read+error events on FD %i",
314  mhandle->mandle, fd);
315  return -1;
316  }
317  DEBUG4("multi-handle %p registered for read+error events on FD %i", mhandle->mandle, fd);
318  break;
319 
320  case CURL_POLL_OUT:
321  if (fr_event_fd_insert(mhandle, mhandle->el, fd,
322  NULL,
323  _fr_curl_io_service_writable,
324  _fr_curl_io_service_errored,
325  mhandle) < 0) {
326  PERROR("multi-handle %p registration failed for write+error events on FD %i",
327  mhandle->mandle, fd);
328  return -1;
329  }
330  DEBUG4("multi-handle %p registered for write+error events on FD %i", mhandle->mandle, fd);
331  break;
332 
333  case CURL_POLL_INOUT:
334  if (fr_event_fd_insert(mhandle, mhandle->el, fd,
335  _fr_curl_io_service_readable,
336  _fr_curl_io_service_writable,
337  _fr_curl_io_service_errored,
338  mhandle) < 0) {
339  PERROR("multi-handle %p registration failed for read+write+error events on FD %i",
340  mhandle->mandle, fd);
341  return -1;
342  }
343  DEBUG4("multi-handle %p registered for read+write+error events on FD %i", mhandle->mandle, fd);
344  break;
345 
346  case CURL_POLL_REMOVE:
347  if (fr_event_fd_delete(mhandle->el, fd, FR_EVENT_FILTER_IO) < 0) {
348  PERROR("multi-handle %p de-registration failed for FD %i", mhandle->mandle, fd);
349  return -1;
350  }
351  DEBUG4("multi-handle %p unregistered events for FD %i", mhandle->mandle, fd);
352  break;
353 
354  default:
355  fr_assert(0);
356  return -1;
357  }
358 
359  return CURLM_OK;
360 }

362 /** Callback to receive debugging data from libcurl
363  *
364  * @note Should only be set on a handle if RDEBUG_ENABLED3 is true.
365  *
366  * @param[in] candle Curl handle the debugging data pertains to.
367  * @param[in] type The type of debugging data we received.
368  * @param[in] data Buffer containing debug data (not \0 terminated). Despite the
369  * type being char *, this can be binary data depending on the
370  * curl_infotype.
371  * @param[in] len The length of the data in the buffer.
372  * @param[in] uctx The current request.
373  * @return
374  * - 0 (we always indicate success)
375  */
376 static int curl_debug_log(UNUSED CURL *candle, curl_infotype type, char *data, size_t len, void *uctx)
377 {
378  request_t *request = talloc_get_type_abort(uctx, request_t);
379  char const *p = data, *q, *end = p + len;
380  char const *verb;
381 
382  switch (type) {
383  case CURLINFO_TEXT:
384  /*
385  * Curl debug output has trailing newlines, and could conceivably
386  * span multiple lines. Take care of both cases.
387  */
388  while (p < end) {
389  q = memchr(p, '\n', end - p);
390  if (!q) q = end;
391 
392  RDEBUG3("libcurl - %pV", fr_box_strvalue_len(p, q ? q - p : p - end));
393  p = q + 1;
394  }
395 
396  break;
397 
398  case CURLINFO_HEADER_IN:
399  verb = "received";
400  print_header:
401  while (p < end) {
402  q = memchr(p, '\n', end - p);
403  q = q ? q + 1 : end;
404 
405  if (RDEBUG_ENABLED4) {
406  RHEXDUMP4((uint8_t const *)p, q - p,
407  "%s header: %pV",
408  verb, fr_box_strvalue_len(p, q - p));
409  } else {
410  RDEBUG3("%s header: %pV",
411  verb, fr_box_strvalue_len(p, q - p));
412  }
413  p = q;
414  }
415  break;
416 
417  case CURLINFO_HEADER_OUT:
418  verb = "sending";
419  goto print_header;
420 
421  case CURLINFO_DATA_IN:
422  RHEXDUMP4((uint8_t const *)data, len, "received data[length %zu]", len);
423  break;
424 
425  case CURLINFO_DATA_OUT:
426  RHEXDUMP4((uint8_t const *)data, len, "sending data[length %zu]", len);
427  break;
428 
429  case CURLINFO_SSL_DATA_OUT:
430  RHEXDUMP4((uint8_t const *)data, len, "sending ssl-data[length %zu]", len);
431  break;
432 
433  case CURLINFO_SSL_DATA_IN:
434  RHEXDUMP4((uint8_t const *)data, len, "received ssl-data[length %zu]", len);
435  break;
436 
437  default:
438  RHEXDUMP3((uint8_t const *)data, len, "libcurl - debug data (unknown type %i)", (int)type);
439  break;
440  }
441 
442  return 0;
443 }

445 /** Sends a request using libcurl
446  *
447  * Send the actual curl request to the server. The response will be handled by
448  * the numerous callbacks configured for the easy handle.
449  *
450  * @param[in] mhandle Thread-specific mhandle wrapper.
451  * @param[in] request Current request.
452  * @param[in] randle representing the request.
453  * @return
454  * - 0 on success.
455  * - -1 on failure.
456  */
457 int fr_curl_io_request_enqueue(fr_curl_handle_t *mhandle, request_t *request, fr_curl_io_request_t *randle)
458 {
459  CURLcode ret;
460  CURLMcode mret;
461 
462  REQUEST_VERIFY(request);
463 
464  randle->request = request;
465 
466  /*
467  * Set debugging functions so we can track the
468  * IO request's progress.
469  */
470  if (RDEBUG_ENABLED3) {
471  FR_CURL_REQUEST_SET_OPTION(CURLOPT_DEBUGFUNCTION, curl_debug_log);
472  FR_CURL_REQUEST_SET_OPTION(CURLOPT_DEBUGDATA, request);
473  FR_CURL_REQUEST_SET_OPTION(CURLOPT_VERBOSE, 1L);
474  }
475 
476  /*
477  * Stick the current request in the curl handle's
478  * private data. This makes it simple to resume
479  * the request in the demux function later...
480  */
481  ret = curl_easy_setopt(randle->candle, CURLOPT_PRIVATE, randle);
482  if (ret != CURLE_OK) {
483  REDEBUG("Request failed: %i - %s", ret, curl_easy_strerror(ret));
484  return -1;
485  }
486 
487  /*
488  * Increment here, else the debug output looks
489  * messed up is curl_multi_add_handle triggers
490  * event loop modifications calls immediately.
491  */
492  mhandle->transfers++;
493  mret = curl_multi_add_handle(mhandle->mandle, randle->candle);
494  if (mret != CURLM_OK) {
495  mhandle->transfers--;
496  REDEBUG("Request failed: %i - %s", mret, curl_multi_strerror(mret));
497  return -1;
498  }
499 
500  return 0;
501 
502 error:
503  return -1;
504 }
