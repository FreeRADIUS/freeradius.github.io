28 /**
29  * $Id: b3c1211369802db06ea3ab76908cf0f90cb0bd1b $
30  * @file rlm_sigtran/event.c
31  * @brief separate eventing thread to run libosmocore's event loop
32  *
33  * libosmocore is not thread safe, none of the event loop access functions
34  * support synchronised access by multiple threads.
35  *
36  * Writing a shim layer and overloading the libosmo functions would work
37  * but is likely to be fragile.
38  *
39  * Instead, we run a special thread to handle all requests to SS7 entities.
40  *
41  * This thread runs the libosmocore event loop, and select()s over the SCTP
42  * FDs libosmo* creates, and a pipe we create (to allow communication
43  * with worker threads).
44  *
45  * You might except performance using this model to be terrible. But the
46  * fact that libosmo is entirely async, and there's no heavy crypto being
47  * performed, I suspect that this thread is unlikely to become a bottleneck.
48  *
49  * In future when we have our own async event loop, we can look at submitting
50  * patches to libosmocore to allow integration with external event loops.
51  *
52  * @note We rely on the fact that writes to a pipe of less than PIPE_BUF, are
53  * atomic and not interleaved.
54  *
55  * @author Arran Cudbard-Bell
56  *
57  * @copyright 2016 Network RADIUS SARL (license@networkradius.com)
58  */

275 /** Send response
276  *
277  * @note Works for both blocking and non-blocking sockets
278  *
279  * @param[in] ofd to write response notification to.
280  * @param[in] txn we're confirming.
281  */
282 int sigtran_event_submit(struct osmo_fd *ofd, sigtran_transaction_t *txn)
283 {
284  uint8_t buff[sizeof(void *)];
285  uint8_t *p = buff, *end = buff + sizeof(buff);
286 
287  memcpy(buff, &txn, sizeof(buff));
288 
289  for (p = buff; p < end; p++) {
290  ssize_t slen;
291 
292  slen = write(ofd->fd, p, end - p);
293  if (slen > 0) {
294  p += slen;
295  continue;
296  }
297 
298  if (errno == EAGAIN) {
299  int ret;
300  fd_set error_set;
301  fd_set write_set;
302 
303  DEBUG3("Server core - Got EAGAIN (no buffer space left), waiting for pipe to become writable");
304 
305  FD_ZERO(&error_set);
306  FD_ZERO(&write_set);
307 
308  FD_SET(ofd->fd, &error_set);
309  FD_SET(ofd->fd, &write_set);
310 
311  /*
312  * Don't let signals mess up the select
313  */
314  do {
315  ret = select(ofd->fd + 1, NULL, &write_set, &error_set, NULL);
316  } while ((ret == -1) && (errno == EINTR));
317 
318  /*
319  * If there wasn't an error try again...
320  */
321  if ((ret > 0) && !FD_ISSET(ofd->fd, &error_set)) continue;
322  }
323 
324  ERROR("Server core - Failed writing to pipe (%i): %s", ofd->fd, fr_syserror(errno));
325  return -1;
326  }
327 
328  return 0;
329 }

331 /** Processes a request for a new pipe from a worker thread
332  *
333  * @param ofd for the main ctrl_pipe.
334  * @param what happened.
335  * @return
336  * - 0 on success, with pointer written to registration pipe for new osmo_fd.
337  * - -1 on error, with NULL pointer written to registration pipe.
338  */
339 static int event_process_request(struct osmo_fd *ofd, unsigned int what)
340 {
341  sigtran_transaction_t *txn;
342 
343  void *ptr;
344  ssize_t len;
345 
346  if (what & BSC_FD_EXCEPT) {
347  ERROR("pipe (%i) closed by osmocom thread, event thread exiting", ofd->fd);
348  do_exit = true;
349  return -1;
350  }
351 
352  if (!(what & BSC_FD_READ)) return 0;
353 
354  len = read(ofd->fd, &ptr, sizeof(ptr));
355  if (len < 0) {
356  ERROR("osmocom thread - Failed reading from pipe (%i): %s", ofd->fd, fr_syserror(errno));
357  return -1;
358  }
359  if (len == 0) {
360  DEBUG4("Ignoring zero length read");
361  return 0;
362  }
363  if (len != sizeof(ptr)) {
364  ERROR("osmocom thread - Failed reading data from pipe (%i): Too short, "
365  "expected %zu bytes, got %zu bytes", ofd->fd, sizeof(ptr), len);
366  ptr = NULL;
367 
368  if (sigtran_event_submit(ofd, NULL) < 0) {
369  fatal_error:
370  DEBUG3("Event loop will exit");
371  do_exit = true;
372  return -1;
373  }
374 
375  return -1;
376  }
377 
378  DEBUG3("osmocom thread - Read %zu bytes from pipe %i (%p)", len, ofd->fd, ptr);
379 
380  txn = talloc_get_type_abort(ptr, sigtran_transaction_t);
381  txn->ctx.ofd = ofd;
382  switch (txn->request.type) {
383  case SIGTRAN_REQUEST_THREAD_REGISTER:
384  {
385  struct osmo_fd *req_ofd;
386  int fd;
387 
388  fd = *((int *)txn->request.data); /* Not talloced */
389 
390  DEBUG3("osmocom thread - Registering req_pipe (%i)", fd);
391 
392  req_ofd = ofd_create(ofd->data, fd, event_process_request, NULL);
393  if (!req_ofd) {
394  txn->response.type = SIGTRAN_RESPONSE_FAIL;
395  } else {
396  txn->response.type = SIGTRAN_RESPONSE_OK;
397  }
398  }
399  break;
400 
401  case SIGTRAN_REQUEST_THREAD_UNREGISTER:
402  DEBUG3("osmocom thread - Deregistering req_pipe (%i). Signalled by worker", ofd->fd);
403  txn->response.type = SIGTRAN_RESPONSE_OK;
404 
405  if (sigtran_event_submit(ofd, txn) < 0) goto fatal_error;
406  talloc_free(ofd); /* Ordering is important */
407  return 0;
408 
409  case SIGTRAN_REQUEST_LINK_UP:
410  DEBUG3("osmocom thread - Bringing link up");
411  if (event_link_up(ofd->data, (sigtran_conn_t **)&txn->response.data, txn->request.data) < 0) { /* Struct not talloced */
412  txn->response.type = SIGTRAN_RESPONSE_FAIL;
413  } else {
414  txn->response.type = SIGTRAN_RESPONSE_OK;
415  }
416  break;
417 
418  case SIGTRAN_REQUEST_LINK_DOWN:
419  DEBUG3("osmocom thread - Taking link down");
420  if (event_link_down(talloc_get_type_abort(txn->request.data, sigtran_conn_t)) < 0) {
421  txn->response.type = SIGTRAN_RESPONSE_FAIL;
422  } else {
423  txn->response.type = SIGTRAN_RESPONSE_OK;
424  }
425  break;
426 
427  case SIGTRAN_REQUEST_MAP_SEND_AUTH_INFO:
428  {
429  sigtran_map_send_auth_info_req_t *req = talloc_get_type_abort(txn->request.data,
430  sigtran_map_send_auth_info_req_t);
431  DEBUG3("osmocom thread - Processing map send auth info");
432  if (sigtran_tcap_outgoing(NULL, req->conn, txn, ofd) < 0) {
433  txn->response.type = SIGTRAN_RESPONSE_FAIL;
434  } else {
435  return 0; /* Keep caller blocked until we get a response */
436  }
437  }
438  break;
439 
440  case SIGTRAN_REQUEST_EXIT:
441  DEBUG3("osmocom thread - Event loop will exit");
442  do_exit = true;
443  txn->response.type = SIGTRAN_RESPONSE_OK;
444 
445  if (sigtran_event_submit(ofd, txn) < 0) goto fatal_error;
446  talloc_free(ofd); /* Ordering is important */
447  return 0;
448 
449 #ifndef NDEBUG
450  case SIGTRAN_REQUEST_TEST:
451  txn->response.type = SIGTRAN_RESPONSE_OK;
452  break;
453 #endif
454 
455  default:
456  fr_assert(0);
457  goto fatal_error;
458  }
459 
460  if (sigtran_event_submit(ofd, txn) < 0) goto fatal_error;
461 
462  return 0;
463 }

465 /** Enter the libosmo event loop
466  *
467  * Will run until the thread is killed, or signalled to exit on the ctrl_pipe.
468  */
469 static void *sigtran_event_loop(UNUSED void *instance)
470 {
471  TALLOC_CTX *ctx = talloc_init_const("sigtran_event_ctx");
472 
473  fr_assert((ctrl_pipe[0] < 0) && (ctrl_pipe[1] < 0)); /* Ensure only one instance exists */
474 
475  /*
476  * Patch in libosmo's logging system to ours
477  */
478  sigtran_log_init(ctx);
479 
480  if (socketpair(AF_UNIX, SOCK_STREAM, 0, ctrl_pipe) < 0) {
481  ERROR("osmocom thread - Failed creating ctrl_pipe: %s", fr_syserror(errno));
482  return NULL;
483  }
484  if (!ofd_create(ctx, ctrl_pipe[1], event_process_request, ctx)) return NULL;
485 
486  DEBUG2("osmocom thread - Entering event loop, listening on fd %i (client fd %i)", ctrl_pipe[1], ctrl_pipe[0]);
487 
488  sem_post(&event_thread_running); /* Up enough to be ok! */
489 
490  /*
491  * The main event loop.
492  */
493  while (true) {
494  osmo_select_main(0);
495  if (do_exit) {
496 #if 0
497  fd_set readset, writeset, exceptset;
498  int high_fd;
499 
500  FD_ZERO(&readset);
501  FD_ZERO(&writeset);
502  FD_ZERO(&exceptset);
503 
504  high_fd = osmo_fd_fill_fds(&readset, &writeset, &exceptset);
505  if (high_fd == 0) break;
506 
507  DEBUG3("osmocom thread - Deferring exit, waiting for fd %i", high_fd);
508 #else
509  break;
510 #endif
511  }
512  }
513 
514  talloc_free(ctx); /* Also frees ctrl pipe ofd (which closes ctrl_pipe[1]) */
515 
516  DEBUG2("osmocom thread - Event loop exiting");
517 
518  return NULL;
519 }
