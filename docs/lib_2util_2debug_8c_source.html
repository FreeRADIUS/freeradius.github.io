211 /** Determine if we're running under LSAN (Leak Sanitizer)
212  *
213  * @return
214  * - 0 if we're not.
215  * - 1 if we are.
216  * - -1 if we can't tell because of an error.
217  * - -2 if we can't tell because we were compiled with support for the LSAN interface.
218  */
219 int fr_get_lsan_state(void)
220 {
221  uint8_t ret = 0;
222 
223  if (lsan_state != INT_MAX) return lsan_state;/* Use cached result */
224 
225  if (pipe(lsan_test_pipe) < 0) {
226  fr_strerror_printf("Failed opening internal pipe: %s", fr_syserror(errno));
227  return -1;
228  }
229 
230  lsan_test_pid = fork();
231  if (lsan_test_pid == -1) {
232  fr_strerror_printf("Error forking: %s", fr_syserror(errno));
233  return -1;
234  }
235 
236  /* Child */
237  if (lsan_test_pid == 0) {
238  close(lsan_test_pipe[0]); /* Close parent's side */
239  exit(EXIT_SUCCESS); /* Results in LSAN calling __lsan_is_turned_off via onexit handler */
240  }
241 
242  /* Parent */
243  close(lsan_test_pipe[1]); /* Close child's side */
244 
245  while ((read(lsan_test_pipe[0], &ret, sizeof(ret)) < 0) && (errno == EINTR));
246 
247  close(lsan_test_pipe[0]); /* Close our side (so we don't leak FDs) */
248 
249  /* Collect child */
250  waitpid(lsan_test_pid, NULL, 0);
251 
252  lsan_state = ret; /* Cache test results */
253 
254  return ret;
255 }

264 /** Determine if we're running under a debugger by attempting to attach using pattach
265  *
266  * @return
267  * - 0 if we're not.
268  * - 1 if we are.
269  * - -1 if we can't tell because of an error.
270  * - -2 if we can't tell because we don't have the CAP_SYS_PTRACE capability.
271  */
272 int fr_get_debug_state(void)
273 {
274  int pid;
275 
276  int from_child[2] = {-1, -1};
277 
278 #ifdef HAVE_CAPABILITY_H
279  cap_flag_value_t state;
280  cap_t caps;
281 
282  /*
283  * If we're running under linux, we first need to check if we have
284  * permission to to ptrace. We do that using the capabilities
285  * functions.
286  */
287  caps = cap_get_proc();
288  if (!caps) {
289  fr_strerror_printf("Failed getting process capabilities: %s", fr_syserror(errno));
290  return DEBUGGER_STATE_UNKNOWN;
291  }
292 
293  if (cap_get_flag(caps, CAP_SYS_PTRACE, CAP_PERMITTED, &state) < 0) {
294  fr_strerror_printf("Failed getting CAP_SYS_PTRACE permitted state: %s",
295  fr_syserror(errno));
296  cap_free(caps);
297  return DEBUGGER_STATE_UNKNOWN;
298  }
299 
300  if ((state == CAP_SET) && (cap_get_flag(caps, CAP_SYS_PTRACE, CAP_EFFECTIVE, &state) < 0)) {
301  fr_strerror_printf("Failed getting CAP_SYS_PTRACE effective state: %s",
302  fr_syserror(errno));
303  cap_free(caps);
304  return DEBUGGER_STATE_UNKNOWN;
305  }
306 
307  /*
308  * We don't have permission to ptrace, so this test will always fail.
309  */
310  if (state == CAP_CLEAR) {
311  fr_strerror_printf("ptrace capability not set. If debugger detection is required run as root or: "
312  "setcap cap_sys_ptrace+ep <path_to_binary>");
313  cap_free(caps);
314  return DEBUGGER_STATE_UNKNOWN_NO_PTRACE_CAP;
315  }
316  cap_free(caps);
317 #endif
318 
319  if (pipe(from_child) < 0) {
320  fr_strerror_printf("Error opening internal pipe: %s", fr_syserror(errno));
321  return DEBUGGER_STATE_UNKNOWN;
322  }
323 
324  pid = fork();
325  if (pid == -1) {
326  fr_strerror_printf("Error forking: %s", fr_syserror(errno));
327  return DEBUGGER_STATE_UNKNOWN;
328  }
329 
330  /* Child */
331  if (pid == 0) {
332  int8_t ret = DEBUGGER_STATE_NOT_ATTACHED;
333  int ppid = getppid();
334  int flags;
335 
336  /*
337  * Disable the leak checker for this forked process
338  * so we don't get spurious leaks reported.
339  */
340 #ifdef HAVE_SANITIZER_LSAN_INTERFACE_H
341  lsan_disable = true;
342 #endif
343 
344 DIAG_OFF(deprecated-declarations)
345  flags = PT_ATTACH;
346 DIAG_ON(deprecated-declarations)
347 
348  /* Close parent's side */
349  close(from_child[0]);
350 
351  /*
352  * FreeBSD is extremely picky about the order of operations here
353  * we need to attach, wait *then* write whilst the parent is still
354  * suspended, then detach, continuing the process.
355  *
356  * If we don't do it in that order the read in the parent triggers
357  * a SIGKILL.
358  */
359 
360  if ((_PTRACE(flags, ppid) == 0)
361 #ifdef __APPLE__
362  || (errno == EPERM)
363 #endif
364  ) {
365  /* Wait for the parent to stop */
366  waitpid(ppid, NULL, 0);
367 
368  /* Tell the parent what happened */
369  if (write(from_child[1], &ret, sizeof(ret)) < 0) {
370  fprintf(stderr, "Writing ptrace status to parent failed: %s", fr_syserror(errno));
371  }
372 
373  /* Detach */
374  _PTRACE_DETACH(ppid);
375  exit(0);
376  }
377 
378  ret = DEBUGGER_STATE_ATTACHED;
379  /* Tell the parent what happened */
380  if (write(from_child[1], &ret, sizeof(ret)) < 0) {
381  fprintf(stderr, "Writing ptrace status to parent failed: %s", fr_syserror(errno));
382  }
383 
384  exit(0);
385  /* Parent */
386  } else {
387  int8_t ret = DEBUGGER_STATE_UNKNOWN;
388 
389  /*
390  * The child writes errno (reason) if pattach failed else 0.
391  *
392  * This read may be interrupted by pattach,
393  * which is why we need the loop.
394  */
395  while ((read(from_child[0], &ret, sizeof(ret)) < 0) && (errno == EINTR));
396 
397  /* Close the pipes here (if we did it above, it might race with pattach) */
398  close(from_child[1]);
399  close(from_child[0]);
400 
401  /* Collect the status of the child */
402  waitpid(pid, NULL, 0);
403 
404  return ret;
405  }
406 }

551 /** Inserts a backtrace marker into the provided context
552  *
553  * Allows for maximum laziness and will initialise a circular buffer if one has not already been created.
554  *
555  * Code augmentation should look something like:
556 @verbatim
557  // Create a static fringer pointer, the first call to backtrace_attach will initialise it
558  static fr_fring_t *my_obj_bt;
559 
560  my_obj_t *alloc_my_obj(TALLOC_CTX *ctx) {
561  my_obj_t *this;
562 
563  this = talloc(ctx, my_obj_t);
564 
565  // Attach backtrace marker to object
566  backtrace_attach(&my_obj_bt, this);
567 
568  return this;
569  }
570 @endverbatim
571  *
572  * Then, later when a double free occurs:
573 @verbatim
574  (gdb) call backtrace_print(&my_obj_bt, <pointer to double freed memory>)
575 @endverbatim
576  *
577  * which should print a limited backtrace to stderr. Note, this backtrace will not include any argument
578  * values, but should at least show the code path taken.
579  *
580  * @param fring this should be a pointer to a static *fr_fring_buffer.
581  * @param obj we want to generate a backtrace for.
582  */
583 fr_bt_marker_t *fr_backtrace_attach(fr_fring_t **fring, TALLOC_CTX *obj)
584 {
585  fr_bt_marker_t *marker;
586 
587  if (*fring == NULL) {
588  pthread_mutex_lock(&fr_debug_init);
589  /* Check again now we hold the mutex - eww*/
590  if (*fring == NULL) *fring = fr_fring_alloc(NULL, MAX_BT_CBUFF, true);
591  pthread_mutex_unlock(&fr_debug_init);
592  }
593 
594  marker = talloc(obj, fr_bt_marker_t);
595  if (!marker) {
596  return NULL;
597  }
598 
599  marker->obj = (void *) obj;
600  marker->fring = *fring;
601 
602  fprintf(stderr, "Backtrace attached to %s %p\n", talloc_get_name(obj), obj);
603  /*
604  * Generate the backtrace for memory allocation
605  */
606  fr_backtrace_do(marker);
607  talloc_set_destructor(marker, fr_backtrace_do);
608 
609  return marker;
610 }

814 /** Check to see if panic_action file is world writable
815  *
816  * @return
817  * - 0 if file is OK.
818  * - -1 if the file is world writable.
819  */
820 static int fr_fault_check_permissions(void)
821 {
822  char const *p, *q;
823  size_t len;
824  char filename[256];
825  struct stat statbuf;
826 
827  /*
828  * Try and guess which part of the command is the binary, and check to see if
829  * it's world writable, to try and save the admin from their own stupidity.
830  *
831  * @fixme we should do this properly and take into account single and double
832  * quotes.

837  * is a bad idea and can result in deadlock.

845  } else {
846  p = panic_action;
847  }

853  return -1;

861 /** Prints a simple backtrace (if execinfo is available) and calls panic_action if set.
862  *
863  * @param sig caught
864  */
865 NEVER_RETURNS void fr_fault(int sig)
866 {
867  char cmd[sizeof(panic_action) + 20];
868  char *out = cmd;
869  size_t left = sizeof(cmd), ret;
870 
871  char const *p = panic_action;
872  char const *q;
873 
874  int code;
875 
876  /*
877  * If a debugger is attached, we don't want to run the panic action,
878  * as it may interfere with the operation of the debugger.
879  * If something calls us directly we just raise the signal and let
880  * the debugger handle it how it wants.
881  */
882  if (fr_debug_state == DEBUGGER_STATE_ATTACHED) {
883  FR_FAULT_LOG("RAISING SIGNAL: %s", strsignal(sig));
884  raise(sig);
885  }
886 
887  /*
888  * Makes the backtraces slightly cleaner
889  */
890  memset(cmd, 0, sizeof(cmd));
891 
892  FR_FAULT_LOG("CAUGHT SIGNAL: %s", strsignal(sig));
893 
894  /*
895  * Run the callback if one was registered
896  */
897  if (panic_cb && (panic_cb(sig) < 0)) goto finish;
898 
899  /*
900  * Produce a simple backtrace - They're very basic but at least give us an
901  * idea of the area of the code we hit the issue in.
902  *
903  * See below in fr_fault_setup() and
904  * https://sourceware.org/bugzilla/show_bug.cgi?id=16159
905  * for why we only print backtraces in debug builds if we're using GLIBC.
906  */
907 #if defined(HAVE_EXECINFO) && (!defined(NDEBUG) || !defined(__GNUC__))
908  if (fr_fault_log_fd >= 0) {
909  size_t frame_count;
910  void *stack[MAX_BT_FRAMES];
911 
912  frame_count = backtrace(stack, MAX_BT_FRAMES);
913 
914  FR_FAULT_LOG("Backtrace of last %zu frames:", frame_count);
915 
916  backtrace_symbols_fd(stack, frame_count, fr_fault_log_fd);
917  }
918 #endif
919 
920  /* No panic action set... */
921  if (panic_action[0] == '\0') {
922  FR_FAULT_LOG("No panic action set");
923  goto finish;
924  }
925 
926  /*
927  * Check for administrator sanity.
928  */
929  if (fr_fault_check_permissions() < 0) {
930  FR_FAULT_LOG("Refusing to execute panic action: %s", fr_strerror());
931  goto finish;
932  }
933 
934  /* Substitute %p for the current PID (useful for attaching a debugger) */
935  while ((q = strstr(p, "%p"))) {
936  out += ret = snprintf(out, left, "%.*s%d", (int) (q - p), p, (int) getpid());
937  if (left <= ret) {
938  oob:
939  FR_FAULT_LOG("Panic action too long");
940  fr_exit_now(128 + sig);
941  }
942  left -= ret;
943  p = q + 2;
944  }
945  if (strlen(p) >= left) goto oob;
946  strlcpy(out, p, left);
947 
948  {
949  bool disable = false;
950 
951  FR_FAULT_LOG("Calling: %s", cmd);
952 
953  /*
954  * Here we temporarily enable the dumpable flag so if GBD or LLDB
955  * is called in the panic_action, they can pattach to the running
956  * process.
957  */
958  if (fr_get_pr_dumpable_flag() == 0) {
959  if ((fr_set_pr_dumpable_flag(true) < 0) || !fr_get_pr_dumpable_flag()) {
960  FR_FAULT_LOG("Failed setting dumpable flag, pattach may not work: %s", fr_strerror());
961  } else {
962  disable = true;
963  }
964  FR_FAULT_LOG("Temporarily setting PR_DUMPABLE to 1");
965  }
966 
967  code = system(cmd);
968 
969  /*
970  * We only want to error out here, if dumpable was originally disabled
971  * and we managed to change the value to enabled, but failed
972  * setting it back to disabled.
973  */
974  if (disable) {
975  FR_FAULT_LOG("Resetting PR_DUMPABLE to 0");
976  if (fr_set_pr_dumpable_flag(false) < 0) {
977  FR_FAULT_LOG("Failed resetting dumpable flag to off: %s", fr_strerror());
978  FR_FAULT_LOG("Exiting due to insecure process state");
979  fr_exit_now(EXIT_FAILURE);
980  }
981  }
982 
983  FR_FAULT_LOG("Panic action exited with %i", code);
984 
985  fr_exit_now(128 + sig);
986  }
987 
988 finish:
989  /*
990  * (Re-)Raise the signal, so that if we're running under
991  * a debugger.
992  *
993  * This allows debuggers to function normally and catch
994  * fatal signals.
995  */
996  fr_unset_signal(sig); /* Make sure we don't get into a loop */
997  raise(sig);
998  fr_exit_now(128 + sig); /* Function marked as noreturn */
999 }

1052 /** Generate a talloc memory report for a context and print to stderr/stdout
1053  *
1054  * @param ctx to generate a report for, may be NULL in which case the root context is used.
1055  */
1056 int fr_log_talloc_report(TALLOC_CTX const *ctx)
1057 {
1058 #define TALLOC_REPORT_MAX_DEPTH 20
1059 
1060  FILE *log;
1061  int fd;
1062 
1063  fd = dup(fr_fault_log_fd);
1064  if (fd < 0) {
1065  fr_strerror_printf("Couldn't write memory report, failed to dup log fd: %s", fr_syserror(errno));
1066  return -1;
1067  }
1068  log = fdopen(fd, "w");
1069  if (!log) {
1070  close(fd);
1071  fr_strerror_printf("Couldn't write memory report, fdopen failed: %s", fr_syserror(errno));
1072  return -1;
1073  }
1074 
1075  if (!ctx) {
1076  fprintf(log, "Current state of talloced memory:\n");
1077  talloc_report_full(talloc_null_ctx(), log);
1078  } else {
1079  int i;
1080 
1081  fprintf(log, "Talloc chunk lineage:\n");
1082  fprintf(log, "%p (%s)", ctx, talloc_get_name(ctx));
1083 
1084  i = 0;
1085  while ((i < TALLOC_REPORT_MAX_DEPTH) && (ctx = talloc_parent(ctx))) {
1086  fprintf(log, " < %p (%s)", ctx, talloc_get_name(ctx));
1087  i++;
1088  }
1089  fprintf(log, "\n");
1090 
1091  i = 0;
1092  do {
1093  fprintf(log, "Talloc context level %i:\n", i++);
1094  talloc_report_full(ctx, log);
1095  } while ((ctx = talloc_parent(ctx)) &&
1096  (i < TALLOC_REPORT_MAX_DEPTH) &&
1097  (talloc_parent(ctx) != talloc_autofree_ctx) && /* Stop before we hit the autofree ctx */
1098  (talloc_parent(ctx) != talloc_null_ctx())); /* Stop before we hit NULL ctx */
1099  }
1100 
1101  fclose(log);
1102 
1103  return 0;
1104 }

1138 /** Registers signal handlers to execute panic_action on fatal signal
1139  *
1140  * May be called multiple time to change the panic_action/program.
1141  *
1142  * @param[in] ctx to allocate autofreeable resources in.
1143  * @param[in] cmd to execute on fault. If present %p will be substituted
1144  * for the parent PID before the command is executed, and %e
1145  * will be substituted for the currently running program.
1146  * @param program Name of program currently executing (argv[0]).
1147  * @return
1148  * - 0 on success.
1149  * - -1 on failure.
1150  */
1151 int fr_fault_setup(TALLOC_CTX *ctx, char const *cmd, char const *program)
1152 {
1153  static bool setup = false;
1154 
1155  char *out = panic_action;
1156  size_t left = sizeof(panic_action);
1157 
1158  char const *p = cmd;
1159  char const *q;
1160 
1161  if (cmd) {
1162  size_t ret;
1163 
1164  /* Substitute %e for the current program */
1165  while ((q = strstr(p, "%e"))) {
1166  out += ret = snprintf(out, left, "%.*s%s", (int) (q - p), p, program ? program : "");
1167  if (left <= ret) {
1168  oob:
1169  fr_strerror_printf("Panic action too long");
1170  return -1;
1171  }
1172  left -= ret;
1173  p = q + 2;
1174  }
1175  if (strlen(p) >= left) goto oob;
1176  strlcpy(out, p, left);
1177  } else {
1178  *panic_action = '\0';
1179  }
1180 
1181  /*
1182  * Check for administrator sanity.
1183  */
1184  if (fr_fault_check_permissions() < 0) return -1;
1185 
1186  /* Unsure what the side effects of changing the signal handler mid execution might be */
1187  if (!setup) {
1188  char *env;
1189 
1190  /*
1191  * Installing signal handlers interferes with some debugging
1192  * operations. Give the developer control over whether the
1193  * signal handlers are installed or not.
1194  */
1195  env = getenv("DEBUGGER_ATTACHED");
1196  if (env && (strcmp(env, "yes") == 0)) {
1197  fr_debug_state = DEBUGGER_STATE_ATTACHED; /* i.e. disable signal handlers */
1198 
1199  } else if (env && (strcmp(env, "no") == 0)) {
1200  fr_debug_state = DEBUGGER_STATE_NOT_ATTACHED; /* i.e. enable signal handlers */
1201 
1202  /*
1203  * Figure out if we were started under a debugger
1204  */
1205  } else {
1206  if (fr_debug_state < 0) fr_debug_state = fr_get_debug_state();
1207  }
1208 
1209  talloc_set_log_fn(_fr_talloc_log);
1210 
1211  /*
1212  * These signals can't be properly dealt with in the debugger
1213  * if we set our own signal handlers.
1214  */
1215  switch (fr_debug_state) {
1216  default:
1217  case DEBUGGER_STATE_NOT_ATTACHED:
1218 #ifdef SIGABRT
1219  if (fr_set_signal(SIGABRT, fr_fault) < 0) return -1;
1220 
1221  /*
1222  * Use this instead of abort so we get a
1223  * full backtrace with broken versions of LLDB
1224  */
1225  talloc_set_abort_fn(_fr_talloc_fault);
1226 #endif
1227 #ifdef SIGILL
1228  if (fr_set_signal(SIGILL, fr_fault) < 0) return -1;
1229 #endif
1230 #ifdef SIGFPE
1231  if (fr_set_signal(SIGFPE, fr_fault) < 0) return -1;
1232 #endif
1233 #ifdef SIGSEGV
1234  if (fr_set_signal(SIGSEGV, fr_fault) < 0) return -1;
1235 #endif
1236  break;
1237 
1238  case DEBUGGER_STATE_ATTACHED:
1239  break;
1240  }
1241 
1242  /*
1243  * Needed for memory reports
1244  */
1245  fr_disable_null_tracking_on_free(ctx);
1246 
1247 #if defined(HAVE_MALLOPT) && !defined(NDEBUG)
1248  /*
1249  * If were using glibc malloc > 2.4 this scribbles over
1250  * uninitialised and freed memory, to make memory issues easier
1251  * to track down.
1252  */
1253  if (!getenv("TALLOC_FREE_FILL")) mallopt(M_PERTURB, 0x42);
1254  mallopt(M_CHECK_ACTION, 3);
1255 #endif
1256 
1257 #if defined(HAVE_EXECINFO) && defined(__GNUC__) && !defined(NDEBUG)
1258  /*
1259  * We need to pre-load lgcc_s, else we can get into a deadlock
1260  * in fr_fault, as backtrace() attempts to dlopen it.
1261  *
1262  * Apparently there's a performance impact of loading lgcc_s,
1263  * so only do it if this is a debug build.
1264  *
1265  * See: https://sourceware.org/bugzilla/show_bug.cgi?id=16159
1266  */
1267  {
1268  void *stack[10];
1269 
1270  backtrace(stack, 10);
1271  }
1272 #endif
1273  }
1274  setup = true;
1275 
1276  return 0;
1277 }
