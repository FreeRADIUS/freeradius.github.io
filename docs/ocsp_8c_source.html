60 /** Extract components of OCSP responser URL from a certificate
61  *
62  * @param[in] cert to extract URL from.
63  * @param[out] host_out Portion of the URL (must be freed with free()).
64  * @param[out] port_out Port portion of the URL (must be freed with free()).
65  * @param[out] path_out Path portion of the URL (must be freed with free()).
66  * @param[out] is_https Whether the responder should be contacted using https.
67  * @return
68  * - 0 if no valid URL is contained in the certificate.
69  * - 1 if a URL was found and parsed.
70  * - -1 if at least one URL was found, but none could be parsed.
71  */
72 static int ocsp_cert_url_parse(X509 *cert, char **host_out, char **port_out, char **path_out, int *is_https)
73 {
74  int i;
75  bool found_uri = false;
76 
77  AUTHORITY_INFO_ACCESS *aia;
78  ACCESS_DESCRIPTION *ad;
79 
80  aia = X509_get_ext_d2i(cert, NID_info_access, NULL, NULL);
81 
82  for (i = 0; i < sk_ACCESS_DESCRIPTION_num(aia); i++) {
83  ad = sk_ACCESS_DESCRIPTION_value(aia, i);
84  if (OBJ_obj2nid(ad->method) != NID_ad_OCSP) continue;
85  if (ad->location->type != GEN_URI) continue;
86  found_uri = true;
87 
88  if (OCSP_parse_url((char *) ad->location->d.ia5->data, host_out,
89  port_out, path_out, is_https)) return 1;
90  }
91  return found_uri ? -1 : 0;
92 }

94 /** Set the OCSP TLS stapling extension for a SSL session, from cached response data
95  *
96  * @param ssl The current SSL session.
97  * @param vp containing the response.
98  * @return
99  * - -1 on error.
100  * - 0 on success.
101  */
102 static int ocsp_staple_from_pair(request_t *request, SSL *ssl, fr_pair_t *vp)
103 {
104  uint8_t *p;
105 
106  /*
107  * OpenSSL should free the buffer itself.
108  */
109  p = OPENSSL_malloc(vp->vp_length);
110  if (!p) return -1;
111 
112  memcpy(p, vp->vp_octets, vp->vp_length);
113 
114  RDEBUG2("Adding OCSP stapling extension");
115  if (SSL_set_tlsext_status_ocsp_resp(ssl, p, vp->vp_length) == 0) {
116  OPENSSL_free(p);
117  return -1;
118  }
119 
120  return 0;
121 }

123 /** Store OCSP response as a TLS-OCSP-Response attribute
124  *
125  * @note Adds &request.TLS-OCSP-Response to the current request, and adds
126  * stapling data to the current SSL session.
127  *
128  * @param out Where to write the TLS-OCSP-Response attribute. Will also
129  * have been added to the request list of the current request.
130  * @param request The current request.
131  * @param resp to serialize.
132  * @return
133  * - -1 on error.
134  * - 0 on success.
135  */
136 static int ocsp_staple_to_pair(fr_pair_t **out, request_t *request, OCSP_RESPONSE *resp)
137 {
138  fr_pair_t *vp;
139  size_t len;
140  uint8_t *p;
141 
142  if (!resp) {
143  REDEBUG("No OCSP response available");
144  return -1;
145  }
146 
147  len = i2d_OCSP_RESPONSE(resp, NULL);
148  if (len <= 0) {
149  REDEBUG("Failed getting OCSP response length");
150  return -1;
151  }
152 
153  MEM(pair_update_request(&vp, attr_tls_ocsp_response) >= 0);
154  MEM(fr_pair_value_mem_alloc(vp, &p, len, true));
155  len = i2d_OCSP_RESPONSE(resp, &p);
156  if (len <= 0) {
157  REDEBUG("Failed serialising OCSP response");
158  pair_delete_request(vp);
159  return -1;
160  }
161 
162  RDEBUG2("Serializing OCSP response");
163  RINDENT();
164  RDEBUG2("&%pP", vp);
165  REXDENT();
166 
167  *out = vp;
168 
169  return 0;
170 }

172 /** Callback used to get stapling data for the current server cert
173  *
174  * @param ssl Current SSL session.
175  * @param data OCSP configuration.
176  */
177 int fr_tls_ocsp_staple_cb(SSL *ssl, void *data)
178 {
179  fr_tls_ocsp_conf_t *conf = data; /* Alloced as part of fr_tls_conf_t (not talloced) */
180  request_t *request = SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_REQUEST);
181 
182  X509 *cert;
183  X509 *issuer_cert;
184  X509_STORE *server_store = NULL;
185  X509_STORE_CTX *server_store_ctx = NULL;
186 
187  STACK_OF(X509) *our_chain = NULL;
188 
189  int ret;
190 
191  cert = SSL_get_certificate(ssl);
192  if (!cert) {
193  fr_tls_log_error(request, "No server certificate found in SSL session");
194  error:
195  X509_STORE_CTX_free(server_store_ctx);
196  X509_STORE_free(server_store);
197 
198  return conf->softfail ? SSL_TLSEXT_ERR_NOACK : SSL_TLSEXT_ERR_ALERT_FATAL;
199  }
200 
201  server_store = SSL_CTX_get_cert_store(SSL_get_SSL_CTX(ssl));
202  if (!server_store) {
203  fr_tls_log_error(request, "Failed retrieving SSL session cert store");
204  goto error;
205  }
206 
207 #if OPENSSL_VERSION_NUMBER >= 0x10102000L
208  if (SSL_get0_chain_certs(ssl, &our_chain) == 0) {
209 #else
210  /*
211  * Ignore the return code for older versions of
212  * OpenSSL.
213  *
214  * https://github.com/openssl/openssl/pull/9395
215  */
216  (void)SSL_get0_chain_certs(ssl, &our_chain);
217  if (!our_chain) {
218 #endif
219  fr_tls_log_error(request, "Failed retrieving chain certificates from current SSL session");
220  goto error;
221  }
222 
223  /*
224  * Print out the current chain in the certificate store
225  * to help with debugging issues where we can't find the
226  * server cert's issuer.
227  */
228  if (RDEBUG_ENABLED3) {
229  RDEBUG3("Current SSL session cert store contents");
230  RINDENT();
231  fr_tls_log_certificate_chain(request, our_chain, cert);
232  REXDENT();
233  }
234 
235  MEM(server_store = X509_STORE_new());
236  X509_STORE_set_trust(server_store, 1); /* All certs are trusted */
237 
238  /*
239  * Add the chain certificates from the current SSL*
240  * to the trusted store so that we can determine
241  * the issuer cert of the certificate we presented.
242  */
243  {
244  int num = sk_X509_num(our_chain);
245  int i;
246 
247  for (i = 0; i < num; i++) {
248  if (X509_STORE_add_cert(server_store, sk_X509_value(our_chain, i)) != 1) {
249  fr_tls_log_error(request, "Failed adding certificate to trusted store");
250  goto error;
251  }
252  }
253  }
254 
255  /*
256  * This is what OpenSSL uses to construct SSL chains
257  * for validation. We just need to use it to find
258  * who issued our server certificate.
259  */
260  MEM(server_store_ctx = X509_STORE_CTX_new());
261  if (X509_STORE_CTX_init(server_store_ctx, server_store, NULL, NULL) == 0) {
262  fr_tls_log_error(request, "Failed initialising SSL session cert store ctx");
263  goto error;
264  }
265 
266  ret = X509_STORE_CTX_get1_issuer(&issuer_cert, server_store_ctx, cert);
267  if (ret != 1) {
268  X509_NAME *subject;
269  X509_NAME *issuer;
270  char *subject_str;
271  char *issuer_str;
272 
273  subject = X509_get_subject_name(cert);
274  if (!subject) {
275  fr_tls_log_error(request, "Couldn't retrieve subject name of SSL session cert");
276  goto error;
277  }
278  MEM(subject_str = X509_NAME_oneline(subject, NULL, 0));
279 
280  issuer = X509_get_issuer_name(cert);
281  if (!issuer) {
282  fr_tls_log_error(request, "Couldn't retrieve issuer name of SSL session cert");
283  OPENSSL_free(subject_str);
284  goto error;
285  }
286  MEM(issuer_str = X509_NAME_oneline(issuer, NULL, 0));
287 
288  switch (ret) {
289  case 0:
290  fr_tls_log_error(request, "Issuer \"%s\" of \"%s\" not found in certificate store",
291  issuer_str, subject_str);
292  break;
293  default:
294  fr_tls_log_error(request, "Error retrieving issuer \"%s\" of \"%s\" from certificate store",
295  issuer_str, subject_str);
296  break;
297  }
298 
299  OPENSSL_free(subject_str);
300  OPENSSL_free(issuer_str);
301  goto error;
302  }
303 
304  fr_assert(issuer_cert);
305 
306  ret = fr_tls_ocsp_check(request, ssl, server_store, issuer_cert, cert, conf, true);
307  switch (ret) {
308  default:
309  case 0: /* server cert is invalid */
310  ret = SSL_TLSEXT_ERR_ALERT_FATAL;
311  break;
312 
313  case 1: /* yes */
314  ret = SSL_TLSEXT_ERR_OK;
315  break;
316 
317  case 2: /* skipped */
318  ret = SSL_TLSEXT_ERR_NOACK;
319  break;
320  }
321 
322  X509_free(issuer_cert); /* Decrement reference count on issuer cert */
323  X509_STORE_CTX_free(server_store_ctx);
324  X509_STORE_free(server_store);
325 
326  return ret;
327 }

329 /** Sends a OCSP request to a defined OCSP responder
330  *
331  */
332 int fr_tls_ocsp_check(request_t *request, SSL *ssl,
333  X509_STORE *store, X509 *issuer_cert, X509 *client_cert,
334  fr_tls_ocsp_conf_t *conf, bool staple_response)
335 {
336  OCSP_CERTID *certid;
337  OCSP_REQUEST *req = NULL;
338  OCSP_RESPONSE *resp = NULL;
339  OCSP_BASICRESP *bresp = NULL;
340  char *host = NULL;
341  char *port = NULL;
342  char *path = NULL;
343  char host_header[1024];
344  int use_ssl = -1;
345  long this_fudge = OCSP_MAX_VALIDITY_PERIOD, this_max_age = -1;
346  BIO *conn = NULL, *ssl_log = NULL;
347  ocsp_status_t ocsp_status = OCSP_STATUS_FAILED;
348  ocsp_status_t status;
349  ASN1_GENERALIZEDTIME *rev, *this_update, *next_update;
350  int reason;
351  OCSP_REQ_CTX *ctx;
352  int rc;
353 
354  fr_time_t start;
355  fr_pair_t *vp;
356 
357  if (conf->cache_server) {
358  rlm_rcode_t rcode;
359 
360  fr_tls_cache_process(&rcode, request, conf->cache.load);
361  switch (rcode) {
362  case RLM_MODULE_REJECT:
363  REDEBUG("Told to force OCSP validation failure from cached response");
364  return OCSP_STATUS_FAILED;
365 
366  case RLM_MODULE_OK:
367  case RLM_MODULE_UPDATED:
368  /*
369  * These are fine for OCSP too, we don't *expect* to always
370  * have a cached OCSP status.
371  */
372  case RLM_MODULE_NOTFOUND:
373  case RLM_MODULE_NOOP:
374  break;
375 
376  default:
377  RWDEBUG("Failed retrieving cached OCSP status");
378  break;
379  }
380  }
381 
382  /*
383  * Allow us to cache the OCSP verified state externally
384  */
385  vp = fr_pair_find_by_da(&request->control_pairs, attr_tls_ocsp_cert_valid);
386  if (vp) switch (vp->vp_uint32) {
387  case 0: /* no */
388  RDEBUG2("Found &control.TLS-OCSP-Cert-Valid = no, forcing OCSP failure");
389  return OCSP_STATUS_FAILED;
390 
391  case 1: /* yes */
392  RDEBUG2("Found &control.TLS-OCSP-Cert-Valid = yes, forcing OCSP success");
393 
394  /*
395  * If this fails, and an OCSP stapled response is required,
396  * we need to run the full OCSP check.
397  */
398  if (staple_response) {
399  vp = fr_pair_find_by_da(&request->control_pairs, attr_tls_ocsp_response);
400  if (!vp) {
401  RDEBUG2("No &control.TLS-OCSP-Response attribute found, performing full OCSP check");
402  break;
403  }
404  if (ocsp_staple_from_pair(request, ssl, vp) < 0) {
405  RWDEBUG("Failed setting OCSP staple response in SSL session");
406  return OCSP_STATUS_FAILED;
407  }
408  }
409 
410  return OCSP_STATUS_OK;
411 
412  case 2: /* skipped */
413  RDEBUG2("Found &control.TLS-OCSP-Cert-Valid = skipped, skipping OCSP check");
414  return conf->softfail ? OCSP_STATUS_OK : OCSP_STATUS_FAILED;
415 
416  case 3: /* unknown */
417  default:
418  break;
419  }
420 
421  if (issuer_cert == NULL) {
422  RWDEBUG("Could not get issuer certificate");
423  goto skipped;
424  }
425 
426  /*
427  * Setup logging for this OCSP operation
428  */
429  ssl_log = BIO_new(BIO_s_mem());
430  if (!ssl_log) {
431  REDEBUG("Failed creating log queue");
432  ocsp_status = OCSP_STATUS_SKIPPED;
433  goto finish;
434  }
435 
436  /*
437  * Create OCSP Request
438  */
439  certid = OCSP_cert_to_id(NULL, client_cert, issuer_cert);
440  req = OCSP_REQUEST_new();
441  OCSP_request_add0_id(req, certid);
442  if (conf->use_nonce) OCSP_request_add1_nonce(req, NULL, 8);
443 
444  /*
445  * Send OCSP Request and get OCSP Response
446  */
447 
448  /* Get OCSP responder URL */
449  if (conf->override_url) {
450  char *url;
451 
452  use_url:
453  memcpy(&url, &conf->url, sizeof(url));
454  /* Reading the libssl src, they do a strdup on the URL, so it could of been const *sigh* */
455  OCSP_parse_url(url, &host, &port, &path, &use_ssl);
456  if (!host || !port || !path) {
457  RWDEBUG("Host or port or path missing from configured URL \"%s\". Not doing OCSP", url);
458  goto skipped;
459  }
460  } else {
461  int ret;
462 
463  ret = ocsp_cert_url_parse(client_cert, &host, &port, &path, &use_ssl);
464  switch (ret) {
465  case -1:
466  RWDEBUG("Invalid URL in certificate. Not doing OCSP");
467  break;
468 
469  case 0:
470  if (conf->url) {
471  RWDEBUG("No OCSP URL in certificate, falling back to configured URL");
472  goto use_url;
473  }
474  RWDEBUG("No OCSP URL in certificate. Not doing OCSP");
475  goto skipped;
476 
477  case 1:
478  fr_assert(host && port && path);
479  break;
480  }
481  }
482 
483  RDEBUG2("Using responder URL \"http://%s:%s%s\"", host, port, path);
484 
485  /* Check host and port length are sane, then create Host: HTTP header */
486  if ((strlen(host) + strlen(port) + 2) > sizeof(host_header)) {
487  RWDEBUG("Host and port too long");
488  goto skipped;
489  }
490  snprintf(host_header, sizeof(host_header), "%s:%s", host, port);
491 
492  /* Setup BIO socket to OCSP responder */
493  conn = BIO_new_connect(host);
494  BIO_set_conn_port(conn, port);
495 
496  if (conf->timeout) BIO_set_nbio(conn, 1);
497 
498  rc = BIO_do_connect(conn);
499  if ((rc <= 0) && ((!conf->timeout) || !BIO_should_retry(conn))) {
500  REDEBUG("Couldn't connect to OCSP responder");
501  ocsp_status = OCSP_STATUS_SKIPPED;
502  goto finish;
503  }
504 
505  ctx = OCSP_sendreq_new(conn, path, NULL, -1);
506  if (!ctx) {
507  REDEBUG("Couldn't create OCSP request");
508  ocsp_status = OCSP_STATUS_SKIPPED;
509  goto finish;
510  }
511 
512  if (!OCSP_REQ_CTX_add1_header(ctx, "Host", host_header)) {
513  REDEBUG("Couldn't set Host header");
514  ocsp_status = OCSP_STATUS_SKIPPED;
515  goto finish;
516  }
517 
518  if (!OCSP_REQ_CTX_set1_req(ctx, req)) {
519  REDEBUG("Couldn't add data to OCSP request");
520  ocsp_status = OCSP_STATUS_SKIPPED;
521  goto finish;
522  }
523 
524  start = fr_time();
525  do {
526  rc = OCSP_sendreq_nbio(&resp, ctx);
527  if (conf->timeout) {
528  if (conf->timeout > (fr_time() - start)) break;
529  }
530  } while ((rc == -1) && BIO_should_retry(conn));
531 
532  if (conf->timeout && (rc == -1) && BIO_should_retry(conn)) {
533  REDEBUG("Response timed out");
534  ocsp_status = OCSP_STATUS_SKIPPED;
535  goto finish;
536  }
537 
538  OCSP_REQ_CTX_free(ctx);
539 
540  if (rc == 0) {
541  REDEBUG("Couldn't get OCSP response");
542  FR_OPENSSL_DRAIN_ERROR_QUEUE(REDEBUG, "", ssl_log);
543  ocsp_status = OCSP_STATUS_SKIPPED;
544  goto finish;
545  }
546 
547  /* Verify OCSP response status */
548  status = OCSP_response_status(resp);
549  if (status != OCSP_RESPONSE_STATUS_SUCCESSFUL) {
550  REDEBUG("Response status: %s", OCSP_response_status_str(status));
551  goto finish;
552  }
553  bresp = OCSP_response_get1_basic(resp);
554  if (conf->use_nonce && OCSP_check_nonce(req, bresp) != 1) {
555  REDEBUG("Response has wrong nonce value");
556  goto finish;
557  }
558  if (OCSP_basic_verify(bresp, NULL, store, 0) != 1){
559  REDEBUG("Couldn't verify OCSP basic response");
560  goto finish;
561  }
562 
563  /* Verify OCSP cert status */
564  if (!OCSP_resp_find_status(bresp, certid, (int *)&status, &reason, &rev, &this_update, &next_update)) {
565  REDEBUG("No Status found");
566  goto finish;
567  }
568 
569  /*
570  * Here we check the fields 'thisUpdate' and 'nextUpdate'
571  * from the OCSP response against the server's time.
572  *
573  * this_fudge is the number of seconds +- between the current
574  * time and this_update.
575  *
576  * The default for this_fudge is 300, defined by OCSP_MAX_VALIDITY_PERIOD.
577  */
578  if (!OCSP_check_validity(this_update, next_update, this_fudge, this_max_age)) {
579  /*
580  * We want this to show up in the global log
581  * so someone will fix it...
582  */
583  RATE_LIMIT_GLOBAL(RERROR, "Delta +/- between OCSP response time and our time is greater than %li "
584  "seconds. Check servers are synchronised to a common time source",
585  this_fudge);
586  FR_OPENSSL_DRAIN_ERROR_QUEUE(REDEBUG, "", ssl_log);
587  goto finish;
588  }
589 
590  /*
591  * Print any messages we may have accumulated
592  */
593  FR_OPENSSL_DRAIN_ERROR_QUEUE(RDEBUG2, "", ssl_log);
594  if (RDEBUG_ENABLED) {
595  RDEBUG2("OCSP response valid from:");
596  ASN1_GENERALIZEDTIME_print(ssl_log, this_update);
597  RINDENT();
598  FR_OPENSSL_DRAIN_LOG_QUEUE(RDEBUG2, "", ssl_log);
599  REXDENT();
600 
601  if (next_update) {
602  RDEBUG2("New information available at:");
603  ASN1_GENERALIZEDTIME_print(ssl_log, next_update);
604  RINDENT();
605  FR_OPENSSL_DRAIN_LOG_QUEUE(RDEBUG2, "", ssl_log);
606  REXDENT();
607  }
608  }
609 
610  /*
611  * When an OCSP validation command is used with OpenSSL
612  * next_update is NULL.
613  */
614  if (next_update) {
615  fr_time_t now;
616  time_t next;
617 
618  /*
619  * Sometimes we already know what 'now' is depending
620  * on the code path, other times we don't.
621  */
622  now = fr_time();
623 
624  if (fr_tls_utils_asn1time_to_epoch(&next, next_update) < 0) {
625  RPEDEBUG("Failed parsing next_update time");
626  ocsp_status = OCSP_STATUS_SKIPPED;
627  goto finish;
628  }
629  if (fr_time_to_sec(now) < next){
630  RDEBUG2("Adding OCSP TTL attribute");
631 
632  MEM(pair_update_request(&vp, attr_tls_ocsp_next_update) >= 0);
633  vp->vp_uint32 = next - fr_time_to_sec(now);
634  RINDENT();
635  RDEBUG2("&%pP", vp);
636  REXDENT();
637  } else {
638  RDEBUG2("Update time is in the past. Not adding &TLS-OCSP-Next-Update");
639  }
640  } else {
641  RDEBUG2("Update time not provided. Not adding &TLS-OCSP-Next-Update");
642  }
643 
644  switch (status) {
645  case V_OCSP_CERTSTATUS_GOOD:
646  RDEBUG2("Cert status: good");
647  ocsp_status = OCSP_STATUS_OK;
648  break;
649 
650  default:
651  /* REVOKED / UNKNOWN */
652  REDEBUG("Cert status: %s", OCSP_cert_status_str(status));
653  if (reason != -1) REDEBUG("Reason: %s", OCSP_crl_reason_str(reason));
654 
655  /*
656  * Print any messages we may have accumulated
657  */
658  FR_OPENSSL_DRAIN_LOG_QUEUE(RDEBUG, "", ssl_log);
659  if (RDEBUG_ENABLED2) {
660  RDEBUG2("Revocation time:");
661  ASN1_GENERALIZEDTIME_print(ssl_log, rev);
662  RINDENT();
663  FR_OPENSSL_DRAIN_LOG_QUEUE(RDEBUG2, "", ssl_log);
