156 /*
157  * The "detail.work" file doesn't exist. Let's see if we can rename one.
158  */
159 static int work_rename(proto_detail_file_thread_t *thread)
160 {
161  proto_detail_file_t const *inst = thread->inst;
162  unsigned int i;
163  int found;
164  time_t chtime;
165  char const *filename;
166  glob_t files;
167  struct stat st;
168 
169  DEBUG3("proto_detail (%s): polling for detail files in %s",
170  thread->name, inst->directory);
171 
172  memset(&files, 0, sizeof(files));
173  if (glob(inst->filename, 0, NULL, &files) != 0) {
174  noop:
175  DEBUG3("proto_detail (%s): no matching files for %s",
176  thread->name, inst->filename);
177  globfree(&files);
178  return -1;
179  }
180 
181  /*
182  * Loop over the glob'd files, looking for the
183  * oldest one.
184  */
185  chtime = 0;
186  found = -1;
187  for (i = 0; i < files.gl_pathc; i++) {
188  if (stat(files.gl_pathv[i], &st) < 0) continue;
189 
190  if ((i == 0) || (st.st_ctime < chtime)) {
191  chtime = st.st_ctime;
192  found = i;
193  }
194  }
195 
196  /*
197  * No matching files, reset the timer and continue.
198  */
199  if (found < 0) goto noop;
200 
201  /*
202  * Rename detail to detail.work
203  */
204  filename = files.gl_pathv[found];
205 
206  DEBUG("proto_detail (%s): Renaming %s -> %s", thread->name, filename, inst->filename_work);
207  if (rename(filename, inst->filename_work) < 0) {
208  ERROR("detail (%s): Failed renaming %s to %s: %s",
209  thread->name, filename, inst->filename_work, fr_syserror(errno));
210  goto noop;
211  }
212 
213  globfree(&files); /* Shouldn't be using anything in files now */
214 
215  /*
216  * The file should now exist, return the open'd FD.
217  */
218  return open(inst->filename_work, inst->mode);
219 }

231 /*
232  * The "detail.work" file exists, and is open in the 'fd'.
233  */
234 static int work_exists(proto_detail_file_thread_t *thread, int fd)
235 {
236  proto_detail_file_t const *inst = thread->inst;
237  bool opened = false;
238  proto_detail_work_thread_t *work;
239  fr_listen_t *li = NULL;
240  struct stat st;
241 
242  fr_event_vnode_func_t funcs = { .delete = mod_vnode_delete };
243 
244  DEBUG3("proto_detail (%s): Trying to lock %s", thread->name, inst->filename_work);
245 
246  /*
247  * "detail.work" exists, try to lock it.
248  */
249  if (rad_lockfd_nonblock(fd, 0) < 0) {
250  fr_time_t delay;
251 
252  DEBUG3("proto_detail (%s): Failed locking %s: %s",
253  thread->name, inst->filename_work, fr_syserror(errno));
254 
255  close(fd);
256 
257  delay = thread->lock_interval;
258 
259  /*
260  * Set the next interval, and ensure that we
261  * don't do massive busy-polling.
262  */
263  thread->lock_interval += thread->lock_interval / 2;
264  if (thread->lock_interval > ((fr_time_delta_t) 30) * NSEC) thread->lock_interval = ((fr_time_delta_t) 30) * NSEC;
265 
266  DEBUG3("proto_detail (%s): Waiting %d.%06ds for lock on file %s",
267  thread->name, (int) (delay / NSEC), (int) ((delay % NSEC) / 1000), inst->filename_work);
268 
269  if (fr_event_timer_in(thread, thread->el, &thread->ev,
270  delay, work_retry_timer, thread) < 0) {
271  ERROR("Failed inserting retry timer for %s", inst->filename_work);
272  }
273  return 0;
274  }
275 
276  DEBUG3("proto_detail (%s): Obtained lock and starting to process file %s",
277  thread->name, inst->filename_work);
278 
279  /*
280  * Ignore empty files.
281  */
282  if (fstat(fd, &st) < 0) {
283  ERROR("Failed opening %s: %s", inst->filename_work,
284  fr_syserror(errno));
285  unlink(inst->filename_work);
286  close(fd);
287  return 1;
288  }
289 
290  if (!st.st_size) {
291  DEBUG3("proto_detail (%s): %s file is empty, ignoring it.",
292  thread->name, inst->filename_work);
293  unlink(inst->filename_work);
294  close(fd);
295  return 1;
296  }
297 
298  /*
299  * This listener is allocated in a thread-specific
300  * context, so it doesn't need a destructor,
301  */
302  MEM(li = talloc_zero(NULL, fr_listen_t));
303 
304  /*
305  * Create a new listener, and insert it into the
306  * scheduler. Shamelessly copied from proto_detail.c
307  * mod_open(), with changes.
308  *
309  * This listener is parented from the worker. So that
310  * when the worker goes away, so does the listener.
311  */
312  li->app_io = inst->parent->work_io;
313 
314  li->app = inst->parent->self;
315  li->app_instance = inst->parent;
316  li->server_cs = inst->parent->server_cs;
317 
318  /*
319  * The worker may be in a different thread, so avoid
320  * talloc threading issues by using a NULL TALLOC_CTX.
321  */
322  MEM(li->thread_instance = work = talloc_zero(li, proto_detail_work_thread_t));
323 
324  li->app_io_instance = inst->parent->work_io_instance;
325  work->inst = li->app_io_instance;
326  work->file_parent = thread;
327  work->ev = NULL;
328 
329  li->fd = work->fd = dup(fd);
330  if (work->fd < 0) {
331  DEBUG("proto_detail (%s): Failed opening %s: %s",
332  thread->name, inst->filename_work, fr_syserror(errno));
333 
334  close(fd);
335  talloc_free(li);
336  return -1;
337  }
338 
339  /*
340  * Don't do anything until the file has been deleted.
341  *
342  * @todo - ensure that proto_detail_work is done the file...
343  * maybe by creating a new instance?
344  */
345  if (fr_event_filter_insert(thread, NULL, thread->el, fd, FR_EVENT_FILTER_VNODE,
346  &funcs, NULL, thread) < 0) {
347  PERROR("Failed adding work socket to event loop");
348  close(fd);
349  talloc_free(li);
350  return -1;
351  }
352 
353  /*
354  * Remember this for later.
355  */
356  thread->vnode_fd = fd;
357 
358  /*
359  * For us, this is the worker listener.
360  * For the worker, this is it's own parent
361  */
362  thread->listen = li;
363 
364  work->filename_work = talloc_strdup(work, inst->filename_work);
365 
366  /*
367  * Set configurable parameters for message ring buffer.
368  */
369  li->default_message_size = inst->parent->max_packet_size;
370  li->num_messages = inst->parent->num_messages;
371 
372  pthread_mutex_lock(&thread->worker_mutex);
373  thread->num_workers++;
374  pthread_mutex_unlock(&thread->worker_mutex);
375 
376  /*
377  * Open the detail.work file.
378  */
379  if (li->app_io->open(li) < 0) {
380  ERROR("Failed opening %s", li->app_io->name);
381  goto error;
382  }
383  opened = true;
384 
385  fr_assert(li->app_io->get_name);
386  li->name = li->app_io->get_name(li);
387 
388  if (!fr_schedule_listen_add(inst->parent->sc, li)) {
389  error:
390  if (fr_event_fd_delete(thread->el, thread->vnode_fd, FR_EVENT_FILTER_VNODE) < 0) {
391  PERROR("Failed removing DELETE callback when opening work file");
392  }
393  close(thread->vnode_fd);
394  thread->vnode_fd = -1;
395 
396  if (opened) {
397  (void) li->app_io->close(li);
398  thread->listen = NULL;
399  li = NULL;
400  }
401 
402  talloc_free(li);
403  return -1;
404  }
405 
406  /*
407  * Tell the worker to clean itself up.
408  */
409  work->listen = li;
410 
411  return 0;
412 }

415 static void mod_vnode_delete(fr_event_list_t *el, int fd, UNUSED int fflags, void *ctx)
416 {
417  proto_detail_file_thread_t *thread = talloc_get_type_abort(ctx, proto_detail_file_thread_t);
418  proto_detail_file_t const *inst = thread->inst;
419 
420  DEBUG("proto_detail (%s): Deleted %s", thread->name, inst->filename_work);
421 
422  /*
423  * Silently ignore notifications from the directory. We
424  * didn't ask for them, but libkqueue delivers them to
425  * us.
426  */
427  if (fd == thread->fd) return;
428 
429  if (fd != thread->vnode_fd) {
430  ERROR("Received DELETE for FD %d, when we were expecting one on FD %d - ignoring it",
431  fd, thread->vnode_fd);
432  return;
433  }
434 
435  if (fr_event_fd_delete(el, fd, FR_EVENT_FILTER_VNODE) < 0) {
436  PERROR("Failed removing DELETE callback after deletion");
437  }
438  close(fd);
439  thread->vnode_fd = -1;
440 
441  /*
442  * Re-initialize the state machine.
443  *
444  * Note that a "delete" may be the result of an atomic
445  * "move", which both deletes the old file, and creates
446  * the new one.
447  */
448  work_init(thread);
449 }

452 static void work_init(proto_detail_file_thread_t *thread)
453 {
454  proto_detail_file_t const *inst = thread->inst;
455  int fd, rcode;
456  bool has_worker;
457 
458  pthread_mutex_lock(&thread->worker_mutex);
459  has_worker = (thread->num_workers != 0);
460  pthread_mutex_unlock(&thread->worker_mutex);
461 
462  /*
463  * The worker is still processing the file, poll until
464  * it's done.
465  */
466  if (has_worker) {
467  DEBUG3("proto_detail (%s): worker %s is still alive, waiting for it to finish.",
468  thread->name, inst->filename_work);
469  goto delay;
470  }
471 
472  fr_assert(thread->vnode_fd < 0);
473 
474  /*
475  * See if there is a "detail.work" file. If not, try to
476  * rename an existing file to "detail.work".
477  */
478  DEBUG3("Trying to open %s", inst->filename_work);
479  fd = open(inst->filename_work, inst->mode);
480 
481  /*
482  * If the work file didn't exist, try to rename detail* ->
483  * detail.work, and return the newly opened file.
484  */
485  if (fd < 0) {
486  if (errno != ENOENT) {
487  DEBUG("proto_detail (%s): Failed opening %s: %s",
488  thread->name, inst->filename_work,
489  fr_syserror(errno));
490  goto delay;
491  }
492 
493 retry:
494  fd = work_rename(thread);
495  }
496 
497  /*
498  * The work file still doesn't exist. Go set up timers,
499  * or wait for an event which signals us that something
500  * in the directory changed.
501  */
502  if (fd < 0) {
503  /*
504  * Wait for the directory to change before
505  * looking for another "detail" file.
506  */
507  if (!inst->poll_interval) return;
508 
509 delay:
510  /*
511  * Check every N seconds.
512  */
513  DEBUG3("Waiting %d.000000s for new files in %s", inst->poll_interval, thread->name);
514 
515  if (fr_event_timer_in(thread, thread->el, &thread->ev,
516  fr_time_delta_from_sec(inst->poll_interval), work_retry_timer, thread) < 0) {
517  ERROR("Failed inserting poll timer for %s", inst->filename_work);
518  }
519  return;
520  }
521 
522  thread->lock_interval = NSEC / 10;
523 
524  /*
525  * It exists, go process it!
526  *
527  * We will get back to the main loop when the
528  * "detail.work" file is deleted.
529  */
530  rcode = work_exists(thread, fd);
531  if (rcode < 0) goto delay;
532 
533  /*
534  * The file was empty, so we try to get another one.
535  */
536  if (rcode == 1) goto retry;
537 
538  /*
539  * Otherwise the child is successfully processing the
540  * file.
541  */
542 }

585 static int mod_bootstrap(void *instance, CONF_SECTION *cs)
586 {
587  proto_detail_file_t *inst = talloc_get_type_abort(instance, proto_detail_file_t);
588  dl_module_inst_t const *dl_inst;
589  char *p;
590 
591 #ifdef __linux__
592  /*
593  * The kqueue API takes an FD, but inotify requires a filename.
594  * libkqueue uses /proc/PID/fd/# to look up the FD -> filename mapping.
595  *
596  * However, if you start the server as "root", and then swap to "radiusd",
597  * /proc/PID will be owned by "root" for security reasons. The only way
598  * to make /proc/PID owned by "radiusd" is to set the DUMPABLE flag.
599  *
600  * Instead of making the poor sysadmin figure this out,
601  * we check for this situation, and give them a
602  * descriptive message telling them what to do.
603  */
604  if (!main_config->allow_core_dumps &&
605  main_config->uid_is_set &&
606  main_config->server_uid != 0) {
607  cf_log_err(cs, "Cannot start detail file reader due to Linux limitations.");
608  cf_log_err(cs, "Please set 'allow_core_dumps = true' in the main configuration file.");
609  return -1;
610  }
611 #endif
612 
613  /*
614  * Find the dl_module_inst_t holding our instance data
615  * so we can find out what the parent of our instance
616  * was.
617  */
618  dl_inst = dl_module_instance_by_data(instance);
619  fr_assert(dl_inst);
620 
621 #ifndef __linux__
622  /*
623  * Linux inotify works. So we allow poll_interval==0
624  */
625  FR_INTEGER_BOUND_CHECK("poll_interval", inst->poll_interval, >=, 1);
626 #endif
627  FR_INTEGER_BOUND_CHECK("poll_interval", inst->poll_interval, <=, 3600);
628 
629  inst->parent = talloc_get_type_abort(dl_inst->parent->data, proto_detail_t);
630  inst->cs = cs;
631 
632  inst->directory = p = talloc_strdup(inst, inst->filename);
633 
634  p = strrchr(p, '/');
635  if (!p) {
636  cf_log_err(cs, "Filename must contain '/'");
637  return -1;
638  }
639 
640  *p = '\0';
641 
642  if (!inst->filename_work) {
643  inst->filename_work = talloc_typed_asprintf(inst, "%s/detail.work", inst->directory);
644  }
645 
646  /*
647  * We need this for the lock.
648  */
649  inst->mode = O_RDWR;
650 
651  return 0;
652 }

654 static int mod_close(fr_listen_t *li)
655 {
656  proto_detail_file_t const *inst = talloc_get_type_abort_const(li->app_io_instance, proto_detail_file_t);
657  proto_detail_file_thread_t *thread = talloc_get_type_abort(li->thread_instance, proto_detail_file_thread_t);
658 
659  if (thread->nr) (void) fr_network_socket_delete(thread->nr, inst->parent->listen);
660 
661  /*
662  * @todo - have our OWN event loop for timers, and a
663  * "copy timer from -> to, which means we only have to
664  * delete our child event loop from the parent on close.
665  */
666  close(thread->fd);
667 
668  if (thread->vnode_fd >= 0) {
669  if (thread->nr) {
670  (void) fr_network_socket_delete(thread->nr, inst->parent->listen);
671  } else {
672  if (fr_event_fd_delete(thread->el, thread->vnode_fd, FR_EVENT_FILTER_VNODE) < 0) {
673  PERROR("Failed removing DELETE callback on detach");
674  }
675  }
676  close(thread->vnode_fd);
677  thread->vnode_fd = -1;
678 
679  pthread_mutex_destroy(&thread->worker_mutex);
680  }
681 
682  return 0;
683 }
