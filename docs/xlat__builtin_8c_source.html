128 /** Convenience function to convert a string attribute reference to a cursor
129  *
130  * This is intended to be used by xlat functions which need to iterate over
131  * an attribute reference provided as a format string or as a boxed value.
132  *
133  * We can't add attribute reference support to the xlat parser
134  * as the inputs and outputs of xlat functions are all boxed values and
135  * couldn't represent a fr_pair_t.
136  *
137  * @param[in] ctx To allocate new cursor in.
138  * @param[out] out Where to write heap allocated cursor. Must be freed
139  * once it's done with. The heap based cursor is to
140  * simplify memory management, as all tmpls are heap
141  * allocated, and we need to bind the lifetime of the
142  * tmpl and tmpl cursor together.
143  * @param[in] tainted May be NULL. Set to true if one or more of the pairs
144  * in the cursor's scope have the tainted flag high.
145  * @param[in] request The current request.
146  * @param[in] fmt string. Leading whitespace will be ignored.
147  * @return
148  * - 0 on success.
149  * - -1 on failure.
150  */
151 int xlat_fmt_to_cursor(TALLOC_CTX *ctx, fr_cursor_t **out,
152  bool *tainted, request_t *request, char const *fmt)
153 {
154  tmpl_t *vpt;
155  fr_pair_t *vp;
156  fr_cursor_t *cursor;
157  tmpl_cursor_ctx_t cc;
158 
159  fr_skip_whitespace(fmt); /* Not binary safe, but attr refs should only contain printable chars */
160 
161  if (tmpl_afrom_attr_str(NULL, NULL, &vpt, fmt,
162  &(tmpl_rules_t){
163  .dict_def = request->dict,
164  .prefix = TMPL_ATTR_REF_PREFIX_AUTO
165  }) <= 0) {
166  RPEDEBUG("Failed parsing attribute reference");
167  return -1;
168  }
169 
170  MEM(cursor = talloc(ctx, fr_cursor_t));
171  talloc_steal(cursor, vpt);
172  vp = tmpl_cursor_init(NULL, NULL, &cc, cursor, request, vpt);
173  tmpl_cursor_clear(&cc);
174  *out = cursor;
175 
176  if (!tainted) return 0;
177 
178  *tainted = false; /* Needed for the rest of the code */
179 
180  if (!vp) return 0;
181 
182  do {
183  if (vp->vp_tainted) {
184  *tainted = true;
185  break;
186  }
187  } while ((vp = fr_cursor_next(cursor)));
188 
189  fr_cursor_head(cursor); /* Reset */
190 
191  return 0;
192 }

262  *

270  * @param[in] func xlat function to be called.
271  * @param[in] escape function to sanitize any sub expansions passed to the xlat function.
272  * @param[in] instantiate function to pre-parse any xlat specific data.
273  * @param[in] inst_size sizeof() this xlat's instance data.
274  * @param[in] buf_len Size of the output buffer to allocate when calling the function.
275  * May be 0 if the function allocates its own buffer.
276  * @return
277  * - 0 on success.
278  * - -1 on failure.
279  */
280 int xlat_register_legacy(void *mod_inst, char const *name,
281  xlat_func_legacy_t func, xlat_escape_legacy_t escape,
282  xlat_instantiate_t instantiate, size_t inst_size,
283  size_t buf_len)
284 {
285  xlat_t *c;
286  bool is_new = false;
287 
288  if (!xlat_root && (xlat_init() < 0)) return -1;;
289 
290  if (!name || !*name) {
291  ERROR("%s: Invalid xlat name", __FUNCTION__);
292  return -1;
293  }
294 
295  /*
296  * If it already exists, replace the instance.
297  */
298  c = rbtree_finddata(xlat_root, &(xlat_t){ .name = name });
299  if (c) {

301  ERROR("%s: Cannot re-define internal expansion %s", __FUNCTION__, name);

306  */
307  } else {
308  c = talloc_zero(xlat_root, xlat_t);
309  c->name = talloc_typed_strdup(c, name);
310  talloc_set_destructor(c, _xlat_func_talloc_free);
311  is_new = true;
312  }
313 
314  c->func.sync = func;
315  c->type = XLAT_FUNC_LEGACY;
316  c->buf_len = buf_len;
317  c->escape = escape;
318  c->mod_inst = mod_inst;
319  c->instantiate = instantiate;
320  c->inst_size = inst_size;
321  c->needs_async = false;
322 
323  DEBUG3("%s: %s", __FUNCTION__, c->name);
324 
325  if (is_new && !rbtree_insert(xlat_root, c)) {
326  ERROR("Failed inserting xlat registration for %s",
327  c->name);
328  talloc_free(c);
329  return -1;
330  }
331 
332  return 0;
333 }

337  *
338  * @param[in] ctx Used to automate deregistration of the xlat fnction.
339  * @param[in] name of the xlat.
340  * @param[in] func to register.
341  * @param[in] needs_async Requires asynchronous evaluation.
342  * @return
343  * - 0 on success.
344  * - -1 on failure.
345  */
346 xlat_t const *xlat_register(TALLOC_CTX *ctx, char const *name, xlat_func_t func, bool needs_async)
347 {
348  xlat_t *c;
349 
350  if (!xlat_root) xlat_init();
351 
352  if (!name || !*name) {
353  ERROR("%s: Invalid xlat name", __FUNCTION__);
354  return NULL;
355  }
356 
357  /*
358  * If it already exists, replace the instance.
359  */
360  c = rbtree_finddata(xlat_root, &(xlat_t){ .name = name });
361  if (c) {
362  if (c->internal) {
363  ERROR("%s: Cannot re-define internal expansion %s", __FUNCTION__, name);
364  return NULL;
365  }
366 
367  if ((c->type != XLAT_FUNC_NORMAL) || (c->needs_async != needs_async)) {
368  ERROR("%s: Cannot change async capability of %s", __FUNCTION__, name);
369  return NULL;
370  }
371 
372  if (c->func.async != func) {
373  ERROR("%s: Cannot change callback function for %s", __FUNCTION__, name);
374  return NULL;
375  }
376 
377  return c;
378  }
379 
380  /*
381  * Doesn't exist. Create it.
382  */
383  c = talloc_zero(ctx, xlat_t);
384  c->name = talloc_typed_strdup(c, name);
385  talloc_set_destructor(c, _xlat_func_talloc_free);
386 
387  c->func.async = func;
388  c->type = XLAT_FUNC_NORMAL;
389  c->needs_async = needs_async; /* this function may yield */
390 
391  DEBUG3("%s: %s", __FUNCTION__, c->name);
392 
393  if (!rbtree_insert(xlat_root, c)) {
394  ERROR("%s: Failed inserting xlat registration for %s", __FUNCTION__, c->name);

543 /** xlat "redundant" processing
544  *
545  * Processes xlat calls for modules defined in "redundant"
546  * sub-sections of the instantiate configuration.
547  *
548  * @ingroup xlat_functions
549  */
550 static ssize_t xlat_redundant(TALLOC_CTX *ctx, char **out, NDEBUG_UNUSED size_t outlen,
551  void const *mod_inst, UNUSED void const *xlat_inst,
552  request_t *request, char const *fmt)
553 {
554  xlat_redundant_t const *xr = mod_inst;
555  CONF_ITEM *ci;
556  char const *name;
557  xlat_t *xlat;
558 
559  fr_assert((*out == NULL) && (outlen == 0)); /* Caller must not have allocated buf */
560  fr_assert(xr->type == XLAT_REDUNDANT);
561 
562  /*
563  * Pick the first xlat which succeeds
564  */
565  for (ci = cf_item_next(xr->cs, NULL);
566  ci != NULL;
567  ci = cf_item_next(xr->cs, ci)) {
568  ssize_t ret;
569 
570  if (!cf_item_is_pair(ci)) continue;
571 
572  name = cf_pair_attr(cf_item_to_pair(ci));
573  fr_assert(name != NULL);
574 
575  xlat = xlat_func_find(name, -1);
576  if (!xlat) continue;
577 
578  if (xlat->buf_len > 0) {
579  *out = talloc_array(ctx, char, xlat->buf_len);
580  **out = '\0'; /* Be sure the string is \0 terminated */
581  } else {
582  *out = NULL;
583  }
584 
585  ret = xlat->func.sync(ctx, out, xlat->buf_len, xlat->mod_inst, NULL, request, fmt);
586  if (ret <= 0) {
587  TALLOC_FREE(*out);
588  continue;
589  }
590  return ret;
591  }
592 
593  /*
594  * Everything failed. Oh well.
595  */
596  *out = NULL;
597  return 0;
598 }

601 /** xlat "load-balance" processing
602  *
603  * Processes xlat calls for modules defined in "load-balance"
604  * sub-sections of the instantiate configuration.
605  *
606  * @ingroup xlat_functions
607  */
608 static ssize_t xlat_load_balance(TALLOC_CTX *ctx, char **out, NDEBUG_UNUSED size_t outlen,
609  void const *mod_inst, UNUSED void const *xlat_inst,
610  request_t *request, char const *fmt)
611 {
612  uint32_t count = 0;
613  xlat_redundant_t const *xr = mod_inst;
614  CONF_ITEM *ci;
615  CONF_ITEM *found = NULL;
616  char const *name;
617  xlat_t *xlat;
618 
619  fr_assert((*out == NULL) && (outlen == 0)); /* Caller must not have allocated buf */
620 
621  /*
622  * Choose a child at random.
623  */
624  for (ci = cf_item_next(xr->cs, NULL);
625  ci != NULL;
626  ci = cf_item_next(xr->cs, ci)) {
627  if (!cf_item_is_pair(ci)) continue;
628  count++;
629 
630  /*
631  * Replace the previously found one with a random
632  * new one.
633  */
634  if ((count * (fr_rand() & 0xffff)) < (uint32_t) 0x10000) {
635  found = ci;
636  }
637  }
638 
639  /*
640  * Plain load balancing: do one child, and only one child.
641  */
642  if (xr->type == XLAT_LOAD_BALANCE) {
643  ssize_t slen;
644  name = cf_pair_attr(cf_item_to_pair(found));
645  fr_assert(name != NULL);
646 
647  xlat = xlat_func_find(name, -1);
648  if (!xlat) return -1;
649 
650  if (xlat->buf_len > 0) {
651  *out = talloc_array(ctx, char, xlat->buf_len);
652  **out = '\0'; /* Be sure the string is \0 terminated */
653  } else {
654  *out = NULL;
655  }
656  slen = xlat->func.sync(ctx, out, xlat->buf_len, xlat->mod_inst, NULL, request, fmt);
657  if (slen <= 0) TALLOC_FREE(*out);
658 
659  return slen;
660  }
661 
662  fr_assert(xr->type == XLAT_REDUNDANT_LOAD_BALANCE);
663 
664  /*
665  * Try the random one we found. If it fails, keep going
666  * through the rest of the children.
667  */
668  ci = found;
669  do {
670  name = cf_pair_attr(cf_item_to_pair(ci));
671  fr_assert(name != NULL);
672 
673  xlat = xlat_func_find(name, -1);
674  if (xlat) {
675  ssize_t ret;
676 
677  if (xlat->buf_len > 0) {
678  *out = talloc_array(ctx, char, xlat->buf_len);
679  **out = '\0'; /* Be sure the string is \0 terminated */
680  } else {
681  *out = NULL;
682  }
683  ret = xlat->func.sync(ctx, out, xlat->buf_len, xlat->mod_inst, NULL, request, fmt);
684  if (ret > 0) return ret;
685  TALLOC_FREE(*out);
686  }
687 
688  /*
689  * Go to the next one, wrapping around at the end.
690  */
691  ci = cf_item_next(xr->cs, ci);
692  if (!ci) ci = cf_item_next(xr->cs, NULL);
693  } while (ci != found);
694 
695  return -1;
696 }

954  *

956  *

970  * Replaces Tmp-String-1 with three new attributes:
971 @verbatim
972 &Tmp-String-1 = "a"
973 &Tmp-String-1 = "b"
974 &Tmp-String-1 = "c"
975 @endverbatim
976  *
977  * @ingroup xlat_functions
978  */
979 static ssize_t xlat_func_explode(TALLOC_CTX *ctx, char **out, size_t outlen,
980  UNUSED void const *mod_inst, UNUSED void const *xlat_inst,
981  request_t *request, char const *fmt)
982 {
983  tmpl_t *vpt = NULL;
984  fr_pair_t *vp;
985  fr_cursor_t cursor, to_merge;
986  tmpl_cursor_ctx_t cc;
987  fr_pair_t *head = NULL;
988  ssize_t slen;
989  int count = 0;
990  char const *p = fmt;
991  char delim;
992 
993  /*
994  * Trim whitespace
995  */
996  fr_skip_whitespace(p);
997 
998  slen = tmpl_afrom_attr_substr(ctx, NULL, &vpt,
999  &FR_SBUFF_IN(p, strlen(p)),
1000  &xlat_arg_parse_rules,
1001  &(tmpl_rules_t){ .dict_def = request->dict });
1002  if (slen <= 0) {
1003  RPEDEBUG("Invalid input");
1004  return -1;
1005  }
1006 
1007  p += slen;
1008 
1009  if (*p++ != ' ') {
1010  arg_error:
1011  talloc_free(vpt);
1012  REDEBUG("explode needs exactly two arguments: &ref <delim>");
1013  return -1;
1014  }
1015 
1016  if (*p == '\0' || p[1]) goto arg_error;
1017 
1018  delim = *p;
1019 
1020  fr_cursor_init(&to_merge, &head);
1021 
1022  vp = tmpl_cursor_init(NULL, NULL, &cc, &cursor, request, vpt);
1023  while (vp) {
1024  fr_pair_t *nvp;
1025  char const *end;
1026  char const *q;
1027 
1028  /*
1029  * This can theoretically operate on lists too
1030  * so we need to check the type of each attribute.
1031  */
1032  switch (vp->vp_type) {
1033  case FR_TYPE_OCTETS:
1034  case FR_TYPE_STRING:
1035  break;
1036 
1037  default:
1038  goto next;
1039  }
1040 
1041  p = vp->vp_ptr;
1042  end = p + vp->vp_length;
1043  while (p < end) {
1044  q = memchr(p, delim, end - p);
1045  if (!q) {
1046  /* Delimiter not present in attribute */
1047  if (p == vp->vp_ptr) goto next;
1048  q = end;
1049  }
1050 
1051  /* Skip zero length */
1052  if (q == p) {
1053  p = q + 1;
1054  continue;
1055  }
1056 
1057  MEM(nvp = fr_pair_afrom_da(talloc_parent(vp), vp->da));
1058  switch (vp->vp_type) {
1059  case FR_TYPE_OCTETS:
1060  MEM(fr_pair_value_memdup(nvp, (uint8_t const *)p, q - p, vp->vp_tainted) == 0);
1061  break;
1062 
1063  case FR_TYPE_STRING:
1064  MEM(fr_pair_value_bstrndup(nvp, p, q - p, vp->vp_tainted) == 0);
1065  break;
1066 
1067  default:
1068  fr_assert(0);
1069  }
1070 
1071  fr_cursor_append(&to_merge, nvp);
1072 
1073  p = q + 1; /* next */
1074 
1075  count++;
1076  }
1077 
1078  /*
1079  * Remove the unexploded version
1080  */
1081  vp = fr_cursor_remove(&cursor);
1082  talloc_free(vp);
1083  /*
1084  * Remove sets cursor->current to
1085  * the next iter value.
1086  */
1087  vp = fr_cursor_current(&cursor);
1088  continue;
1089 
1090  next:
1091  vp = fr_cursor_next(&cursor);
1092  }
1093  tmpl_cursor_clear(&cc);
1094 
1095  fr_cursor_head(&to_merge);
1096  fr_cursor_merge(&cursor, &to_merge);
1097  talloc_free(vpt);
1098 
1099  return snprintf(*out, outlen, "%i", count);
1100 }

1209 /** Parse the 3 arguments to lpad / rpad.
1210  *
1211  * Parses a fmt string with the components @verbatim <tmpl> <pad_len> <pad_char>@endverbatim
1212  *
1213  * @param[out] vpt_p Template to retrieve value to pad.
1214  * @param[out] pad_len_p Length the string needs to be padded to.
1215  * @param[out] pad_char_p Char to use for padding.
1216  * @param[in] request The current request.
1217  * @param[in] fmt string to parse.
1218  *
1219  * @return
1220  * - <= 0 the negative offset the parse error ocurred at.
1221  * - >0 how many bytes of fmt were parsed.
1222  */
1223 static ssize_t parse_pad(tmpl_t **vpt_p, size_t *pad_len_p, char *pad_char_p, request_t *request, char const *fmt)
1224 {
1225  ssize_t slen;

1227  char const *p;
1228  char *end;
1229  tmpl_t *vpt;
1230 
1231 
1232  *pad_char_p = ' '; /* the default */
1233 
1234  *vpt_p = NULL;
1235 
1236  p = fmt;
1237  fr_skip_whitespace(p);
1238 
1239  if (*p != '&') {
1240  REDEBUG("First argument must be an attribute reference");
1241  return 0;
1242  }
1243 
1244  slen = tmpl_afrom_attr_substr(request, NULL, &vpt,
1245  &FR_SBUFF_IN(p, strlen(p)),
1246  &xlat_arg_parse_rules,
1247  &(tmpl_rules_t){ .dict_def = request->dict });
1248  if (slen <= 0) {
1249  RPEDEBUG("Failed parsing input string");
1250  return slen;
1251  }
1252 
1253  p = fmt + slen;
1254 
1255  fr_skip_whitespace(p);
1256 
1257  pad_len = strtoul(p, &end, 10);
1258  if ((pad_len == ULONG_MAX) || (pad_len > 8192)) {
1259  talloc_free(vpt);
1260  REDEBUG("Invalid pad_len found at: %s", p);
1261  return fmt - p;
1262  }
1263 
1264  p += (end - p);
1265 
1266  /*
1267  * The pad_char_p character is optional.
1268  *
1269  * But we must have a space after the previous number,
1270  * and we must have only ONE pad_char_p character.
1271  */
1272  if (*p) {
1273  if (!isspace(*p)) {
1274  talloc_free(vpt);
1275  REDEBUG("Invalid text found at: %s", p);
1276  return fmt - p;
1277  }
1278 
1279  fr_skip_whitespace(p);
1280 
1281  if (p[1] != '\0') {
1282  talloc_free(vpt);
1283  REDEBUG("Invalid text found at: %s", p);
1284  return fmt - p;
1285  }
1286 
1287  *pad_char_p = *p++;
1288  }
1289 
1290  *vpt_p = vpt;
1291  *pad_len_p = pad_len;
1292 
1293  return p - fmt;
1294 }

1301 @endverbatim
1302  *

1304 @verbatim
1305 "%{lpad:&User-Name 5 x}" == "xxfoo"
1306 @endverbatim
1307  *
1308  * @ingroup xlat_functions
1309  */

1341  */

1348  memmove(to_pad + (pad - len), to_pad, len + 1);
