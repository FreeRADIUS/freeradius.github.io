88 /** Convert yubikey modhex to normal hex
89  *
90  * The same buffer may be passed as modhex and hex to convert the modhex in place.
91  *
92  * Modhex and hex must be the same size.
93  *
94  * @param[in] modhex data.
95  * @param[in] len of input and output buffers.
96  * @param[out] hex where to write the standard hexits.
97  * @return
98  * - The number of bytes written to the output buffer.
99  * - -1 on failure.
100  */
101 static ssize_t modhex2hex(char const *modhex, uint8_t *hex, size_t len)
102 {
103  size_t i;
104  char *c1, *c2;
105 
106  for (i = 0; i < len; i++) {
107  if (modhex[i << 1] == '\0') {
108  break;
109  }
110 
111  /*
112  * We only deal with whole bytes
113  */
114  if (modhex[(i << 1) + 1] == '\0')
115  return -1;
116 
117  if (!(c1 = memchr(modhextab, tolower((int) modhex[i << 1]), 16)) ||
118  !(c2 = memchr(modhextab, tolower((int) modhex[(i << 1) + 1]), 16)))
119  return -1;
120 
121  hex[i] = hextab[c1 - modhextab];
122  hex[i + 1] = hextab[c2 - modhextab];
123  }
124 
125  return i;
126 }

128 /** Xlat to convert Yubikey modhex to standard hex
129  *
130  * Example:
131 @verbatim
132 "%{modhextohex:vvrbuctetdhc}" == "ffc1e0d3d260"
133 @endverbatim
134  *
135  * @ingroup xlat_functions
136  */
137 static ssize_t modhex_to_hex_xlat(UNUSED TALLOC_CTX *ctx, char **out, size_t outlen,
138  UNUSED void const *mod_inst, UNUSED void const *xlat_inst,
139  request_t *request, char const *fmt)
140 {
141  ssize_t len;
142 
143  if (outlen < strlen(fmt)) return 0;
144 
145  /*
146  * mod2hex allows conversions in place
147  */
148  len = modhex2hex(fmt, (uint8_t *) *out, strlen(fmt));
149  if (len <= 0) {
150  REDEBUG("Modhex string invalid");
151  return -1;
152  }
153 
154  return len;
155 }

241 /*
242  * Find the named user in this modules database. Create the set
243  * of attribute-value pairs to check and reply with for this user
244  * from the database. The authentication code only needs to check
245  * the password, the rest is done here.
246  */
247 static unlang_action_t CC_HINT(nonnull) mod_authorize(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
248 {
249  rlm_yubikey_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_yubikey_t);
250  char const *passcode;
251  size_t len;
252  fr_pair_t *vp, *password;
253  char const *otp;
254  size_t password_len;
255  int ret;
256 
257  /*
258  * Can't do yubikey auth if there's no password.
259  */
260  password = fr_pair_find_by_da(&request->request_pairs, attr_user_password);
261  if (!password) {
262  /*
263  * Don't print out debugging messages if we know
264  * they're useless.
265  */
266  if (request->packet->code != FR_CODE_ACCESS_CHALLENGE) {
267  RDEBUG2("No cleartext password in the request. Can't do Yubikey authentication");
268  }
269 
270  RETURN_MODULE_NOOP;
271  }
272 
273  passcode = password->vp_strvalue;
274  len = password->vp_length;
275 
276  /*
277  * Now see if the passcode is the correct length (in its raw
278  * modhex encoded form).
279  *
280  * <public_id (6-16 bytes)> + <aes-block (32 bytes)>
281  *
282  */
283  if (len < (inst->id_len + YUBIKEY_TOKEN_LEN)) {
284  RDEBUG2("User-Password value is not the correct length, expected at least %u bytes, got %zu bytes",
285  inst->id_len + YUBIKEY_TOKEN_LEN, len);
286  RETURN_MODULE_NOOP;
287  }
288 
289  password_len = (len - (inst->id_len + YUBIKEY_TOKEN_LEN));
290  otp = passcode + password_len;
291  ret = otp_string_valid(inst, otp, (inst->id_len + YUBIKEY_TOKEN_LEN));
292  if (ret <= 0) {
293  if (RDEBUG_ENABLED3) {
294  RDMARKER(otp, -(ret), "User-Password (aes-block) value contains non modhex chars");
295  } else {
296  RDEBUG2("User-Password (aes-block) value contains non modhex chars");
297  }
298  RETURN_MODULE_NOOP;
299  }
300 
301  /* May be a concatenation, check the last 32 bytes are modhex */
302  if (inst->split) {
303  /*
304  * Insert a new request attribute just containing the OTP
305  * portion.
306  */
307  MEM(pair_update_request(&vp, attr_yubikey_otp) >= 0);
308  fr_pair_value_strdup(vp, otp);
309 
310  /*
311  * Replace the existing string buffer for the password
312  * attribute with one just containing the password portion.
313  */
314  MEM(fr_pair_value_bstr_realloc(password, NULL, password_len) == 0);
315 
316  RINDENT();
317  if (RDEBUG_ENABLED3) {
318  RDEBUG3("&request.%pP", vp);
319  RDEBUG3("&request.%pP", password);
320  } else {
321  RDEBUG2("&request.%s := <<< secret >>>", vp->da->name);
322  RDEBUG2("&request.%s := <<< secret >>>", password->da->name);
323  }
324  REXDENT();
325 
326  /*
327  * So the ID split code works on the non password portion.
328  */
329  passcode = vp->vp_strvalue;
330  }
331 
332  /*
333  * Split out the Public ID in case another module in authorize
334  * needs to verify it's associated with the user.
335  *
336  * It's left up to the user if they want to decode it or not.
337  */
338  if (inst->id_len) {
339  MEM(pair_update_request(&vp, attr_yubikey_public_id) >= 0);
340  fr_pair_value_bstrndup(vp, passcode, inst->id_len, true);
341  }
342 
343  if (!inst->auth_type) {
344  WARN("No 'authenticate %s {...}' section or 'Auth-Type = %s' set. Cannot setup Yubikey authentication",
345  inst->name, inst->name);
346  RETURN_MODULE_NOOP;
347  }
348 
349  if (!module_section_type_set(request, attr_auth_type, inst->auth_type)) RETURN_MODULE_NOOP;
350 
351  RETURN_MODULE_OK;
352 }

356  * Authenticate the user with the given password.
357  */
358 static unlang_action_t CC_HINT(nonnull) mod_authenticate(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
359 {
360  rlm_yubikey_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_yubikey_t);
361  rlm_rcode_t rcode = RLM_MODULE_NOOP;
362  char const *passcode = NULL;
363  fr_pair_t const *vp;
364  size_t len;
365  int ret;
366 
367  vp = fr_pair_find_by_da(&request->request_pairs, attr_yubikey_otp);
368  if (!vp) {
369  RDEBUG2("No Yubikey-OTP attribute found, falling back to User-Password");
370  /*
371  * Can't do yubikey auth if there's no password.
372  */
373  vp = fr_pair_find_by_da(&request->request_pairs, attr_user_password);
374  if (!vp) {
375  REDEBUG("No User-Password in the request. Can't do Yubikey authentication");
376  RETURN_MODULE_INVALID;
377  }
378  }
379 
380  passcode = vp->vp_strvalue;
381  len = vp->vp_length;
382 
383  /*
384  * Verify the passcode is the correct length (in its raw
385  * modhex encoded form).
386  *
387  * <public_id (6-16 bytes)> + <aes-block (32 bytes)>
388  */
389  if (len != (inst->id_len + YUBIKEY_TOKEN_LEN)) {
390  REDEBUG("%s value is not the correct length, expected bytes %u, got bytes %zu",
391  vp->da->name, inst->id_len + YUBIKEY_TOKEN_LEN, len);
392  RETURN_MODULE_INVALID;
393  }
394 
395  ret = otp_string_valid(inst, passcode, (inst->id_len + YUBIKEY_TOKEN_LEN));
396  if (ret <= 0) {
397  if (RDEBUG_ENABLED3) {
398  REMARKER(passcode, -ret, "Passcode (aes-block) value contains non modhex chars");
399  } else {
400  RERROR("Passcode (aes-block) value contains non modhex chars");
401  }
402  RETURN_MODULE_INVALID;
403  }
404 
405 #ifdef HAVE_YUBIKEY
406  if (inst->decrypt) {
407 
408  rlm_yubikey_decrypt(&rcode, inst, request, passcode);
409  if (rcode != RLM_MODULE_OK) RETURN_MODULE_RCODE(rcode);
410  /* Fall-Through to doing ykclient auth in addition to local auth */
411  }
412 #endif
413 
414 #ifdef HAVE_YKCLIENT
415  if (inst->validate) return rlm_yubikey_validate(p_result, inst, request, passcode);
416 #endif
417  RETURN_MODULE_RCODE(rcode);
418 }
