121 static int submodule_parse(UNUSED TALLOC_CTX *ctx, void *out, UNUSED void *parent,
122  CONF_ITEM *ci, UNUSED CONF_PARSER const *rule)
123 { char const *name = cf_pair_value(cf_item_to_pair(ci));
124  char *our_name = NULL;
125  char *p;
126  CONF_SECTION *eap_cs = cf_item_to_section(cf_parent(ci));
127  CONF_SECTION *submodule_cs;
128  eap_type_t method;
129 
130  /*
131  * Search with underscores smashed to hyphens
132  * as that's what's used in the dictionary.
133  */
134  p = our_name = talloc_strdup(NULL, name);
135  while (*p) {
136  if (*p == '_') *p = '-';
137  p++;
138  }
139 
140  method = eap_name2type(our_name);
141  if (method == FR_EAP_METHOD_INVALID) {
142  talloc_free(our_name);
143  cf_log_err(ci, "Unknown EAP type %s", name);
144  return -1;
145  }
146 
147 #if !defined(HAVE_OPENSSL_SSL_H) || !defined(HAVE_LIBSSL)
148  /*
149  * This allows the default configuration to be
150  * shipped with EAP-TLS, etc. enabled. If the
151  * system doesn't have OpenSSL, they will be
152  * ignored.
153  *
154  * If the system does have OpenSSL, then this
155  * code will not be used. The administrator will
156  * then have to delete the tls,
157  * etc. configurations from eap.conf in order to
158  * have EAP without the TLS types.
159  */
160  switch (method) {
161  case FR_EAP_METHOD_TLS:
162  case FR_EAP_METHOD_TTLS:
163  case FR_EAP_METHOD_PEAP:
164  case FR_EAP_METHOD_PWD:
165  case FR_EAP_METHOD_AKA_PRIME:
166  case FR_EAP_METHOD_AKA:
167  case FR_EAP_METHOD_SIM:
168  {
169  rlm_eap_t *inst = ((dl_module_inst_t *)cf_data_value(cf_data_find(eap_cs,
170  dl_module_inst_t, "rlm_eap")))->data;
171 
172  WARN("Ignoring EAP method %s because we don't have OpenSSL support", name);
173 
174  talloc_free(our_name);
175  }
176  return 0;
177 
178  default:
179  break;
180  }
181 #endif
182 
183  /*
184  * A bit hacky, we should really figure out a better way
185  * of handling missing sections.
186  */
187  submodule_cs = cf_section_find(eap_cs, name, NULL);
188  if (!submodule_cs) {
189  submodule_cs = cf_section_alloc(eap_cs, eap_cs, name, NULL);
190  cf_filename_set(submodule_cs, cf_filename(ci));
191  cf_lineno_set(submodule_cs, cf_lineno(ci));
192  }
193 
194  *(void **)out = submodule_cs;
195 
196  talloc_free(our_name);
197 
198  return 0;
199 }

232 /** Process NAK data from EAP peer
233  *
234  */
235 static eap_type_t eap_process_nak(module_ctx_t const *mctx, request_t *request,
236  eap_type_t type,
237  eap_type_data_t *nak)
238 {
239  rlm_eap_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_eap_t);
240  unsigned int i;
241  fr_pair_t *vp;
242  eap_type_t method = FR_EAP_METHOD_INVALID;
243 
244  /*
245  * The NAK data is the preferred EAP type(s) of
246  * the client.
247  *
248  * RFC 3748 says to list one or more proposed
249  * alternative types, one per octet, or to use
250  * 0 for no alternative.
251  */
252  if (!nak->data) {
253  REDEBUG("Peer sent empty (invalid) NAK. Can't select method to continue with");
254 
255  return FR_EAP_METHOD_INVALID;
256  }
257 
258  /*
259  * Pick one type out of the one they asked for,
260  * as they may have asked for many.
261  */
262  vp = fr_pair_find_by_da(&request->control_pairs, attr_eap_type);
263  for (i = 0; i < nak->length; i++) {
264  /*
265  * Type 0 is valid, and means there are no
266  * common choices.
267  */
268  if (nak->data[i] == 0) {
269  REDEBUG("Peer NAK'd indicating it is not willing to continue");
270 
271  return FR_EAP_METHOD_INVALID;
272  }
273 
274  /*
275  * It is invalid to request identity,
276  * notification & nak in nak.
277  */
278  if (nak->data[i] < FR_EAP_METHOD_MD5) {
279  REDEBUG("Peer NAK'd asking for bad type %s (%d)", eap_type2name(nak->data[i]), nak->data[i]);
280 
281  return FR_EAP_METHOD_INVALID;
282  }
283 
284  if ((nak->data[i] >= FR_EAP_METHOD_MAX) ||
285  !inst->methods[nak->data[i]].submodule) {
286  RDEBUG2("Peer NAK'd asking for unsupported EAP type %s (%d), skipping...",
287  eap_type2name(nak->data[i]),
288  nak->data[i]);
289 
290  continue;
291  }
292 
293  /*
294  * Prevent a firestorm if the client is confused.
295  */
296  if (type == nak->data[i]) {
297  char const *type_str = eap_type2name(nak->data[i]);
298 
299  RDEBUG2("Peer NAK'd our request for %s (%d) with a request for %s (%d), skipping...",
300  type_str, nak->data[i], type_str, nak->data[i]);
301 
302  RWARN("!!! We requested to use EAP type %s (%i)", type_str, nak->data[i]);
303  RWARN("!!! The supplicant rejected that, and requested to use the same EAP type.");
304  RWARN("!!! i.e. the supplicant said 'I don't like %s, please use %s instead.",
305  type_str, type_str);
306  RWARN("!!! The supplicant software is broken and does not work properly.");
307  RWARN("!!! Please upgrade it to software that works.");
308 
309  continue;
310  }
311 
312  /*
313  * Enforce per-user configuration of EAP
314  * types.
315  */
316  if (vp && (vp->vp_uint32 != nak->data[i])) {
317  RDEBUG2("Peer wants %s (%d), while we require %s (%d), skipping",
318  eap_type2name(nak->data[i]), nak->data[i],
319  eap_type2name(vp->vp_uint32), vp->vp_uint32);
320 
321  continue;
322  }
323 
324  RDEBUG2("Found mutually acceptable type %s (%d)", eap_type2name(nak->data[i]), nak->data[i]);
325 
326  method = nak->data[i];
327 
328  break;
329  }
330 
331  if (method == FR_EAP_METHOD_INVALID) REDEBUG("No mutually acceptable types found");
332 
333  return method;
334 }

365 /** Process the result of calling a submodule
366  *
367  * @param[out] p_result Result of calling the module, one of:
368  * - RLM_MODULE_INVALID if the request or EAP session state is invalid.
369  * - RLM_MODULE_OK if this round succeeded.
370  * - RLM_MODULE_HANDLED if we're done with this round.
371  * - RLM_MODULE_REJECT if the user should be rejected.
372  * @param[in] request The current request.
373  * @param[in] mctx module calling ctx.
374  * @param[in] eap_session the EAP session
375  * @param[in] result the input result from the submodule
376  */
377 static unlang_action_t mod_authenticate_result(rlm_rcode_t *p_result, UNUSED module_ctx_t const *mctx,
378  request_t *request, eap_session_t *eap_session, rlm_rcode_t result)
379 {
380  rlm_rcode_t rcode;
381 
382  /*
383  * Cleanup the subrequest
384  */
385  (void)fr_cond_assert(request_detach(eap_session->subrequest, true) == 0);
386  TALLOC_FREE(eap_session->subrequest);
387 
388  /*
389  * The submodule failed. Die.
390  */
391  switch (result) {
392  case RLM_MODULE_FAIL:
393  case RLM_MODULE_INVALID:
394  eap_fail(eap_session);
395  eap_session_destroy(&eap_session);
396 
397  rcode = RLM_MODULE_INVALID;
398  goto finish;
399 
400  /*
401  * Inconsistent result from submodule...
402  */
403  case RLM_MODULE_REJECT:
404  case RLM_MODULE_DISALLOW:
405  eap_session->this_round->request->code = FR_EAP_CODE_FAILURE;
406  break;
407 
408  /*
409  * Definitely shouldn't get this.
410  */
411  case RLM_MODULE_YIELD:
412  fr_assert(0);
413  break;
414 
415  default:
416  break;
417  }
418 
419  /*
420  * We are done, wrap the EAP-request in RADIUS to send
421  * with all other required radius attributes
422  */
423  rcode = eap_compose(eap_session);
424 
425  /*
426  * Add to the list only if it is EAP-Request, OR if
427  * it's LEAP, and a response.
428  */
429  if (((eap_session->this_round->request->code == FR_EAP_CODE_REQUEST) &&
430  (eap_session->this_round->request->type.num >= FR_EAP_METHOD_MD5)) ||
431 
432  /*
433  * LEAP is a little different. At Stage 4,
434  * it sends an EAP-Success message, but we still
435  * need to keep the State attribute & session
436  * data structure around for the AP Challenge.
437  *
438  * At stage 6, LEAP sends an EAP-Response, which
439  * isn't put into the list.
440  */
441  ((eap_session->this_round->response->code == FR_EAP_CODE_RESPONSE) &&
442  (eap_session->this_round->response->type.num == FR_EAP_METHOD_LEAP) &&
443  (eap_session->this_round->request->code == FR_EAP_CODE_SUCCESS) &&
444  (eap_session->this_round->request->type.num == 0))) {
445  talloc_free(eap_session->prev_round);
446  eap_session->prev_round = eap_session->this_round;
447  eap_session->this_round = NULL;
448  } else {
449  RDEBUG2("Cleaning up EAP session");
450  eap_session_destroy(&eap_session);
451  }
452 
453  /*
454  * Freeze the eap_session so we can continue
455  * the authentication session later.
456  */
457  eap_session_freeze(&eap_session);
458 
459 finish:
460  RETURN_MODULE_RCODE(rcode);
461 }

479 /** Select the correct callback based on a response
480  *
481  * Based on the EAP response from the supplicant, and setup a call on the
482  * unlang stack to the appropriate submodule.
483  *
484  * Default to the configured EAP-Type for all Unsupported EAP-Types.
485  *
486  * @param[out] p_result the result of the operation.
487  * @param[in] mctx module calling ctx.
488  * @param[in] eap_session State data that persists over multiple rounds of EAP.
489  * @return
490  * - RLM_MODULE_INVALID destroy the EAP session as its invalid.
491  * - RLM_MODULE_YIELD Yield control back to the interpreter so it can
492  * call the submodule.
493  */
494 static unlang_action_t eap_method_select(rlm_rcode_t *p_result, module_ctx_t const *mctx, eap_session_t *eap_session)
495 {
496  rlm_eap_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_eap_t);
497  eap_type_data_t *type = &eap_session->this_round->response->type;
498  request_t *request = eap_session->request;
499 
500  rlm_eap_method_t const *method;
501 
502  eap_type_t next = inst->default_method;
503  fr_pair_t *vp;
504 
505  /*
506  * Session must have been thawed...
507  */
508  fr_assert(eap_session->request);
509 
510  /*
511  * Don't trust anyone.
512  */
513  if ((type->num == 0) || (type->num >= FR_EAP_METHOD_MAX)) {
514  REDEBUG("Peer sent EAP type number %d, which is outside known range", type->num);
515 
516  is_invalid:
517  RETURN_MODULE_INVALID;
518  }
519 
520  /*
521  * Multiple levels of TLS nesting are invalid. But if
522  * the parent has a home_server defined, then this
523  * request is being processed through a virtual
524  * server... so that's OK.
525  *
526  * i.e. we're inside an EAP tunnel, which means we have a
527  * parent. If the outer session exists, and doesn't have
528  * a home server, then it's multiple layers of tunneling.
529  */
530  if (eap_session->request->parent &&
531  eap_session->request->parent->parent) {
532  RERROR("Multiple levels of TLS nesting are invalid");
533  goto is_invalid;
534  }
535 
536  RDEBUG2("Peer sent packet with EAP method %s (%d)", eap_type2name(type->num), type->num);
537 
538  /*
539  * Figure out what to do.
540  */
541  switch (type->num) {
542  case FR_EAP_METHOD_IDENTITY:
543  /*
544  * Allow per-user configuration of EAP types.
545  */
546  vp = fr_pair_find_by_da(&eap_session->request->control_pairs, attr_eap_type);
547  if (vp) {
548  RDEBUG2("Using method from &control.EAP-Type");
549  next = vp->vp_uint32;
550  }
551 
552  /*
553  * Ensure it's valid.
554  */
555  if ((next < FR_EAP_METHOD_MD5) || (next >= FR_EAP_METHOD_MAX) || (!inst->methods[next].submodule)) {
556  REDEBUG2("Tried to start unsupported EAP type %s (%d)",
557  eap_type2name(next), next);
558  goto is_invalid;
559  }
560 
561  do_init:
562  /*
563  * If any of these fail, we messed badly somewhere
564  */
565  fr_assert(next >= FR_EAP_METHOD_MD5);
566  fr_assert(next < FR_EAP_METHOD_MAX);
567  fr_assert(inst->methods[next].submodule);
568 
569  eap_session->process = inst->methods[next].submodule->session_init;
570  eap_session->type = next;
571  break;
572 
573  case FR_EAP_METHOD_NAK:
574  /*
575  * Delete old data, if necessary. If we called a method
576  * before, and it initialized itself, we need to free
577  * the memory it alloced.
578  */
579  TALLOC_FREE(eap_session->opaque);
580  next = eap_process_nak(mctx, eap_session->request, eap_session->type, type);
581  if (!next) RETURN_MODULE_REJECT;
582 
583  /*
584  * Initialise the state machine for the next submodule
585  */
586  goto do_init;
587 
588  /*
589  * Only allow modules that are enabled to be called,
590  * treating any other requests as invalid.
591  *
592  * This may seem a bit harsh, but remember the server

676  eap_packet_raw_t *eap_packet;

679  if (!fr_pair_find_by_da(&request->request_pairs, attr_eap_message)) {

685  * Reconstruct the EAP packet from the EAP-Message
686  * attribute. The relevant decoder should have already
687  * concatenated the fragments into a single buffer.
688  */
689  eap_packet = eap_packet_from_vp(request, request->request_pairs);

693  }

697  * is part of an ongoing authentication session,
698  * retrieve the existing eap_session from the request
699  * data.
