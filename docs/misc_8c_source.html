310 /** Write out a vector to a file descriptor
311  *
312  * Wraps writev, calling it as necessary. If timeout is not NULL,
313  * timeout is applied to each call that returns EAGAIN or EWOULDBLOCK
314  *
315  * @note Should only be used on nonblocking file descriptors.
316  * @note Socket should likely be closed on timeout.
317  * @note iovec may be modified in such a way that it's not re-usable.
318  * @note Leaves errno set to the last error that occurred.
319  *
320  * @param fd to write to.
321  * @param vector to write.
322  * @param iovcnt number of elements in iovec.
323  * @param timeout how long to wait for fd to become writable before timing out.
324  * @return
325  * - Number of bytes written.
326  * - -1 on failure.
327  */
328 ssize_t fr_writev(int fd, struct iovec vector[], int iovcnt, fr_time_delta_t timeout)
329 {
330  struct iovec *vector_p = vector;
331  ssize_t total = 0;
332 
333  while (iovcnt > 0) {
334  ssize_t wrote;
335 
336  wrote = writev(fd, vector_p, iovcnt);
337  if (wrote > 0) {
338  total += wrote;
339  while (wrote > 0) {
340  /*
341  * An entire vector element was written
342  */
343  if (wrote >= (ssize_t)vector_p->iov_len) {
344  iovcnt--;
345  wrote -= vector_p->iov_len;
346  vector_p++;
347  continue;
348  }
349 
350  /*
351  * Partial vector element was written
352  */
353  vector_p->iov_len -= wrote;
354  vector_p->iov_base = ((char *)vector_p->iov_base) + wrote;
355  break;
356  }
357  continue;
358  } else if (wrote == 0) return total;
359 
360  switch (errno) {
361  /* Write operation would block, use select() to implement a timeout */
362 #if EWOULDBLOCK != EAGAIN
363  case EWOULDBLOCK:
364  case EAGAIN:
365 #else
366  case EAGAIN:
367 #endif
368  {
369  int ret;
370  fd_set write_set;
371 
372  FD_ZERO(&write_set);
373  FD_SET(fd, &write_set);
374 
375  /* Don't let signals mess up the select */
376  do {
377  ret = select(fd + 1, NULL, &write_set, NULL, &(fr_time_delta_to_timeval(timeout)));
378  } while ((ret == -1) && (errno == EINTR));
379 
380  /* Select returned 0 which means it reached the timeout */
381  if (ret == 0) {
382  fr_strerror_printf("Write timed out");
383  return -1;
384  }
385 
386  /* Other select error */
387  if (ret < 0) {
388  fr_strerror_printf("Failed waiting on socket: %s", fr_syserror(errno));
389  return -1;
390  }
391 
392  /* select said a file descriptor was ready for writing */
393  if (!fr_cond_assert(FD_ISSET(fd, &write_set))) return -1;
394 
395  break;
396  }
397 
398  default:
399  return -1;
400  }
401  }
402 
403  return total;
404 }

406 /** Convert UTF8 string to UCS2 encoding
407  *
408  * @note Borrowed from src/crypto/ms_funcs.c of wpa_supplicant project (http://hostap.epitest.fi/wpa_supplicant/)
409  *
410  * @param[out] out Where to write the ucs2 string.
411  * @param[in] outlen Size of output buffer.
412  * @param[in] in UTF8 string to convert.
413  * @param[in] inlen length of UTF8 string.
414  * @return the size of the UCS2 string written to the output buffer (in bytes).
415  */
416 ssize_t fr_utf8_to_ucs2(uint8_t *out, size_t outlen, char const *in, size_t inlen)
417 {
418  size_t i;
419  uint8_t *start = out;
420 
421  for (i = 0; i < inlen; i++) {
422  uint8_t c, c2, c3;
423 
424  c = in[i];
425  if ((size_t)(out - start) >= outlen) {
426  /* input too long */
427  return -1;
428  }
429 
430  /* One-byte encoding */
431  if (c <= 0x7f) {
432  out[0] = (uint8_t)c;
433  out[1] = 0;
434  out += 2;
435  continue;
436  } else if ((i == (inlen - 1)) || ((size_t)(out - start) >= (outlen - 1))) {
437  /* Incomplete surrogate */
438  return -1;
439  }
440 
441  c2 = in[++i];
442  /* Two-byte encoding */
443  if ((c & 0xe0) == 0xc0) {
444  FR_PUT_LE16(out, ((c & 0x1f) << 6) | (c2 & 0x3f));
445  out += 2;
446  continue;
447  }
448  if ((i == inlen) || ((size_t)(out - start) >= (outlen - 1))) {
449  /* Incomplete surrogate */
450  return -1;
451  }
452 
453  /* Three-byte encoding */
454  c3 = in[++i];
455  FR_PUT_LE16(out, ((c & 0xf) << 12) | ((c2 & 0x3f) << 6) | (c3 & 0x3f));
456  out += 2;
457  }
458 
459  return out - start;
460 }

462 /** Write 128bit unsigned integer to buffer
463  *
464  * @author Alexey Frunze
465  *
466  * @param out where to write result to.
467  * @param outlen size of out.
468  * @param num 128 bit integer.
469  */
470 size_t fr_snprint_uint128(char *out, size_t outlen, uint128_t const num)
471 {
472  char buff[] = "00000000000000000000000000000000000000000000";
473  uint64_t n[2];
474  char *p = buff;
475  int i;
476 #ifndef WORDS_BIGENDIAN
477  size_t const l = 0;
478  size_t const h = 1;
479 #else
480  size_t const l = 1;
481  size_t const h = 0;
482 #endif
483 
484  memcpy(n, &num, sizeof(n));
485 
486  for (i = 0; i < 128; i++) {
487  ssize_t j;
488  int carry;
489 
490  carry = (n[h] >= 0x8000000000000000);
491 
492  // Shift n[] left, doubling it
493  n[h] = ((n[h] << 1) & 0xffffffffffffffff) + (n[l] >= 0x8000000000000000);
494  n[l] = ((n[l] << 1) & 0xffffffffffffffff);
495 
496  // Add s[] to itself in float, doubling it
497  for (j = sizeof(buff) - 2; j >= 0; j--) {
498  buff[j] += buff[j] - '0' + carry;
499  carry = (buff[j] > '9');
500  if (carry) buff[j] -= 10;
501  }
502  }
503 
504  while ((*p == '0') && (p < &buff[sizeof(buff) - 2])) p++;
505 
506  return strlcpy(out, p, outlen);
507 }

568 /** Convert string in various formats to a fr_unix_time_t
569  *
570  * @param date_str input date string.
571  * @param date time_t to write result to.
572  * @return
573  * - 0 on success.
574  * - -1 on failure.
575  */
576 int fr_unix_time_from_str(fr_unix_time_t *date, char const *date_str)
577 {
578  int i;
579  time_t t;
580  struct tm *tm, s_tm;
581  char buf[64];
582  char *p;
583  char *f[4];
584  char *tail = NULL;
585  fr_time_delta_t gmtoff = 0;
586 
587  /*
588  * Test for unix timestamp, which is just a number and
589  * nothing else.
590  */
591  t = strtoul(date_str, &tail, 10);
592  if (*tail == '\0') {
593  *date = fr_unix_time_from_timeval(&(struct timeval) { .tv_sec = t });
594  return 0;
595  }
596 
597  tm = &s_tm;
598  memset(tm, 0, sizeof(*tm));
599  tm->tm_isdst = -1; /* don't know, and don't care about DST */
600 
601  /*
602  * Check for RFC 3339 dates. Note that we only support
603  * dates in a ~1000 year period. If the server is being
604  * used after 3000AD, someone can patch it then.
605  *
606  * %Y-%m-%dT%H:%M:%S
607  * [.%d] sub-seconds
608  * Z | (+/-)%H:%M time zone offset
609  *
610  */
611  if ((t > 1900) && (t < 3000) && *tail == '-') {
612  unsigned long subseconds;
613  int tz, tz_hour, tz_min;
614 
615  p = tail + 1;
616  s_tm.tm_year = t - 1900; /* 'struct tm' starts years in 1900 */
617 
618  if (get_part(&p, &s_tm.tm_mon, 1, 13, '-', "month") < 0) return -1;
619  s_tm.tm_mon--; /* ISO is 1..12, where 'struct tm' is 0..11 */
620 
621  if (get_part(&p, &s_tm.tm_mday, 1, 31, 'T', "day") < 0) return -1;
622  if (get_part(&p, &s_tm.tm_hour, 0, 23, ':', "hour") < 0) return -1;
623  if (get_part(&p, &s_tm.tm_min, 0, 59, ':', "minute") < 0) return -1;
624  if (get_part(&p, &s_tm.tm_sec, 0, 60, '\0', "seconds") < 0) return -1;
625 
626  if (*p == '.') {
627  p++;
628  subseconds = strtoul(p, &tail, 10);
629  if (subseconds > NSEC) {
630  fr_strerror_printf("Invalid nanosecond specifier");
631  return -1;
632  }
633 
634  /*
635  * Scale subseconds to nanoseconds by how
636  * many digits were parsed/
637  */
638  if ((tail - p) < 9) {
639  for (i = 0; i < 9 - (tail -p); i++) {
640  subseconds *= 10;
641  }
642  }
643 
644  p = tail;
645  } else {
646  subseconds = 0;
647  }
648 
649  /*
650  * Time zone is GMT. Leave well enough
651  * alone.
652  */
653  if (*p == 'Z') {
654  if (p[1] != '\0') {
655  fr_strerror_printf("Unexpected text '%c' after time zone", p[1]);
656  return -1;
657  }
658  tz = 0;
659  goto done;
660  }
661 
662  if ((*p != '+') && (*p != '-')) {
663  fr_strerror_printf("Invalid time zone specifier '%c'", *p);
664  return -1;
665  }
666  tail = p; /* remember sign for later */
667  p++;
668 
669  if (get_part(&p, &tz_hour, 0, 23, ':', "hour in time zone") < 0) return -1;
670  if (get_part(&p, &tz_min, 0, 59, '\0', "minute in time zone") < 0) return -1;
671 
672  if (*p != '\0') {
673  fr_strerror_printf("Unexpected text '%c' after time zone", *p);
674  return -1;
675  }
676 
677  /*
678  * We set this, but the timegm() function ignores
679  * it. Note also that mktime() ignores it too,
680  * and treats the time zone as local.
681  *
682  * We can't store this value in s_tm.gtmoff,
683  * because the timegm() function helpfully zeros
684  * it out.
685  */
686  tz = tz_hour * 3600 + tz_min;
687  if (*tail == '-') tz *= -1;
688 
689  done:
690  t = fr_time_from_utc(tm);
691  if (t == (time_t) -1) {
692  fr_strerror_printf("Failed calling system function to parse time - %s",
693  fr_syserror(errno));
694  return -1;
695  }
696 
697  /*
698  * Add in the time zone offset, which the posix
699  * functions are too stupid to do.
700  */
701  t += tz;
702 
703  *date = fr_unix_time_from_timeval(&(struct timeval) { .tv_sec = t });
704  *date += subseconds;
705  return 0;
706  }
707 
708  strlcpy(buf, date_str, sizeof(buf));
709 
710  p = buf;
711  f[0] = mystrtok(&p, " \t");
712  f[1] = mystrtok(&p, " \t");
713  f[2] = mystrtok(&p, " \t");
714  f[3] = mystrtok(&p, " \t"); /* may, or may not, be present */
715  if (!f[0] || !f[1] || !f[2]) {
716  fr_strerror_printf("Too few fields");
717  return -1;
718  }
719 
720  /*
721  * Try to parse the time zone. If it's GMT / UTC or a
722  * local time zone we're OK.
723  *
724  * Otherwise, ignore errors and assume GMT.
725  */
726  if (*p != '\0') {
727  fr_skip_whitespace(p);
728  (void) fr_time_delta_from_time_zone(p, &gmtoff);
729  }
730 
731  /*
732  * The time has a colon, where nothing else does.
733  * So if we find it, bubble it to the back of the list.
734  */
735  if (f[3]) {
736  for (i = 0; i < 3; i++) {
737  if (strchr(f[i], ':')) {
738  p = f[3];
739  f[3] = f[i];
740  f[i] = p;
741  break;
742  }
743  }
744  }
745 
746  /*
747  * The month is text, which allows us to find it easily.
748  */
749  tm->tm_mon = 12;
750  for (i = 0; i < 3; i++) {
751  if (isalpha((int) *f[i])) {
752  int j;
753 
754  /*
755  * Bubble the month to the front of the list
756  */
757  p = f[0];
758  f[0] = f[i];
759  f[i] = p;
760 
761  for (j = 0; j < 12; j++) {
762  if (strncasecmp(months[j], f[0], 3) == 0) {
763  tm->tm_mon = j;
764  break;
765  }
766  }
767  }
768  }
769 
770  /* month not found? */
771  if (tm->tm_mon == 12) {
772  fr_strerror_printf("No month found");
773  return -1;
774  }
775 
776  /*
777  * The year may be in f[1], or in f[2]
778  */
779  tm->tm_year = atoi(f[1]);
780  tm->tm_mday = atoi(f[2]);
781 
782  if (tm->tm_year >= 1900) {
783  tm->tm_year -= 1900;
784 
785  } else {
786  /*
787  * We can't use 2-digit years any more, they make it
788  * impossible to tell what's the day, and what's the year.
789  */
790  if (tm->tm_mday < 1900) {
791  fr_strerror_printf("Invalid year < 1900");
792  return -1;
793  }
794 
795  /*
796  * Swap the year and the day.
797  */
798  i = tm->tm_year;
799  tm->tm_year = tm->tm_mday - 1900;
800  tm->tm_mday = i;
801  }
802 
803  /*
804  * If the day is out of range, die.
805  */
806  if ((tm->tm_mday < 1) || (tm->tm_mday > 31)) {
807  fr_strerror_printf("Invalid day of month");
808  return -1;
809  }
810 
811  /*
812  * There may be %H:%M:%S. Parse it in a hacky way.
813  */
814  if (f[3]) {
815  f[0] = f[3]; /* HH */
816  f[1] = strchr(f[0], ':'); /* find : separator */
817  if (!f[1]) {
818  fr_strerror_printf("No ':' after hour");
819  return -1;
820  }
821 
822  *(f[1]++) = '\0'; /* nuke it, and point to MM:SS */
823 
824  f[2] = strchr(f[1], ':'); /* find : separator */
825  if (f[2]) {
826  *(f[2]++) = '\0'; /* nuke it, and point to SS */
827  tm->tm_sec = atoi(f[2]);
828  } /* else leave it as zero */
829 
830  tm->tm_hour = atoi(f[0]);
831  tm->tm_min = atoi(f[1]);
832  }
833 
834  /*
835  * Returns -1 on failure.
836  */
837  t = fr_time_from_utc(tm);
838  if (t == (time_t) -1) {
839  fr_strerror_printf("Failed calling system function to parse time - %s",
840  fr_syserror(errno));
841  return -1;
842  }
843 
844  /*
845  * Get the UTC time, and manually add in the offset from GMT.
846  */
847  *date = fr_unix_time_from_timeval(&(struct timeval) { .tv_sec = t });
848 
849  /*
850  * Add in the time zone offset, which the posix
851  * functions are too stupid to do.
852  */
853  *date += gmtoff;
854 
855  return 0;
856 }

943  *
944  * Avoids multiplication overflow.
945  *
946  * @param[in] lhs First operand.
