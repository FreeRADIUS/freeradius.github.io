36 /** Add a variable length extension to a talloc chunk
37  *
38  * This is used to build a structure from a primary struct type and one or more
39  * extension structures. The memory for the composed structure is contiguous which
40  * has performance benefits, and means we don't have the overhead of talloc headers
41  * for each of the extensions.
42  *
43  * @note When a new extension is allocated its memory will not be initialised.
44  *
45  * @note It is highly recommended to allocate composed structures within a talloc_pool
46  * to avoid the overhead of malloc+memcpy.
47  *
48  * @param[in] def Extension definitions.
49  * @param[in,out] chunk_p The chunk to add an extension for.
50  * Under certain circumstances the value of *chunk_p will
51  * be changed to point to a new memory block.
52  * All cached copies of the previous pointer should be
53  * updated.
54  * @param[in] ext to alloc.
55  * @param[in] ext_len The length of the extension.
56  * @return
57  * - NULL if we failed allocating an extension.
58  * - A pointer to the extension we allocated.
59  */
60 void *fr_ext_alloc_size(fr_ext_t const *def, void **chunk_p, int ext, size_t ext_len)
61 {
62  size_t aligned_len = ROUND_UP_POW2(ext_len, FR_EXT_ALIGNMENT);
63  size_t chunk_len;
64  size_t hdr_len = 0;
65 
66  size_t offset;
67 
68  fr_ext_info_t const *info;
69  void *n_chunk, *chunk = *chunk_p;
70  uint8_t *chunk_ext;
71  uint8_t *ext_ptr;
72  char const *type;
73 
74  chunk_ext = CHUNK_EXT(*chunk_p, def->offset_of_exts);
75  if (chunk_ext[ext]) return CHUNK_EXT_PTR(*chunk_p, chunk_ext, ext);
76 
77  info = &def->info[ext];
78  if (info->has_hdr) hdr_len = sizeof(dict_ext_hdr_t); /* Add space for a length prefix */
79 
80  /*
81  * Packing the offsets into a uint8_t array
82  * means the offset address of the final
83  * extension must be less than or equal to
84  * UINT8_MAX * FR_EXT_ALIGNMENT.
85  */
86  chunk_len = talloc_get_size(chunk);
87  offset = (chunk_len + hdr_len) / FR_EXT_ALIGNMENT;
88  if (unlikely(offset > UINT8_MAX)) {
89  fr_strerror_printf("Insufficient space remaining for extensions");
90  return NULL;
91  }
92 
93  /*
94  * talloc_realloc_size unhelpfully forgets
95  * the name of the chunk, so we need to
96  * record it and set it back again.
97  */
98  type = talloc_get_name(chunk);
99  n_chunk = talloc_realloc_size(NULL, chunk, chunk_len + hdr_len + aligned_len);
100  if (!n_chunk) {
101  fr_strerror_printf("Failed reallocing %s (%s). Tried to realloc %zu bytes -> %zu bytes",
102  type, fr_syserror(errno), chunk_len, chunk_len + aligned_len);
103  return NULL;
104  }
105  talloc_set_name_const(n_chunk, type);
106  *chunk_p = n_chunk;
107 
108  chunk_ext = CHUNK_EXT(*chunk_p, def->offset_of_exts);
109  chunk_ext[ext] = (uint8_t)offset;
110 
111  ext_ptr = ((uint8_t *)n_chunk) + chunk_len;
112 
113  if (info->has_hdr) {
114  dict_ext_hdr_t *ext_hdr = (dict_ext_hdr_t *)ext_ptr;
115 
116  ext_hdr->len = ext_len; /* Record the real size */
117  return &ext_hdr->data; /* Pointer to the data portion */
118  }
119 
120  return ext_ptr;
121 }

123 /** Return the length of an extension
124  *
125  * @param[in] def Extension definitions.
126  * @param[in] chunk to return extension length for.
127  * @param[in] ext to return length for.
128  * @return
129  * - 0 if no extension exists or is of zero length.
130  * - >0 the length of the extension.
131  */
132 size_t fr_ext_len(fr_ext_t const *def, void const *chunk, int ext)
133 {
134  uint8_t offset;
135  fr_ext_info_t const *info;
136  dict_ext_hdr_t *ext_hdr;
137  uint8_t *chunk_ext;
138 
139  chunk_ext = CHUNK_EXT(chunk, def->offset_of_exts);
140  offset = chunk_ext[ext];
141  if (!offset) return 0;
142 
143  info = &def->info[ext];
144  if (!info->has_hdr) return info->min; /* Fixed size */
145 
146  ext_hdr = (dict_ext_hdr_t *)((uintptr_t)chunk) + ((offset * FR_EXT_ALIGNMENT) - sizeof(dict_ext_hdr_t));
147  return ext_hdr->len;
148 }

150 /** Copy extension data from one attribute to another
151  *
152  * @param[in] def Extension definitions.
153  * @param[in,out] chunk_out to copy extension to.
154  * Under certain circumstances the value of *chunk_out will
155  * be changed to point to a new memory block.
156  * All cached copies of the previous pointer should be
157  * updated.
158  * @param[in] chunk_in to copy extension from.
159  * @param[in] ext to copy.
160  * @return
161  * - NULL if we failed to allocate an extension structure.
162  * - A pointer to the offset of the extension in da_out.
163  */
164 void *fr_ext_copy(fr_ext_t const *def, void **chunk_out, void const *chunk_in, int ext)
165 {
166  void *ext_ptr, *new_ext_ptr;
167  uint8_t *chunk_ext;
168  size_t ext_len;
169 
170  fr_ext_info_t const *info;
171 
172  info = &def->info[ext];
173  if (!info->can_copy) {
174  fr_strerror_printf("Extension cannot be copied");
175  return NULL;
176  }
177 
178  chunk_ext = CHUNK_EXT(chunk_in, def->offset_of_exts);
179  ext_ptr = CHUNK_EXT_PTR(chunk_in, chunk_ext, ext);
180  if (!ext_ptr) return NULL;
181 
182  ext_len = fr_ext_len(def, chunk_in, ext);
183 
184  /*
185  * Use the special copy function.
186  * Its responsible for allocating the extension in the
187  * destination attribute.
188  */
189  if (info->copy) return info->copy(chunk_out, ext, ext_ptr, ext_len);
190 
191  /*
192  * If there's no special function
193  * just memcpy the data over.
194  */
195  new_ext_ptr = fr_ext_alloc_size(def, chunk_out, ext, ext_len);
196  if (!new_ext_ptr) return NULL;
197  memcpy(new_ext_ptr, ext_ptr, ext_len);
198 
199  return new_ext_ptr;
200 }

202 /** Copy all the extensions from one attribute to another
203  *
204  * @param[in] def Extension definitions.
205  * @param[in,out] chunk_out to copy extensions to.
206  * Under certain circumstances the value of *chunk_out will
207  * be changed to point to a new memory block.
208  * All cached copies of the previous pointer should be
209  * updated.
210  * @param[in] chunk_in to copy extensions from.
211  * @return
212  * - 0 on success.
213  * - -1 if a copy operation failed.
214  */
215 int fr_ext_copy_all(fr_ext_t const *def, void **chunk_out, void const *chunk_in)
216 {
217  int i;
218  uint8_t *ext_in = CHUNK_EXT(chunk_in, def->offset_of_exts);
219 
220  for (i = 0; i < def->max; i++) {
221  if (!ext_in[i] || !def->info[i].can_copy) continue;
222  if (!fr_ext_copy(def, chunk_out, chunk_in, i)) return -1;
223  }
224 
225  return 0;
226 }

228 /** Print out all extensions and hexdump their contents
229  *
230  * This function is intended to be called from interactive debugging
231  * sessions only. It does not use the normal logging infrastructure.
232  *
233  * @param[in] def Extension definitions.
234  * @param[in] name the identifier of the structure
235  * being debugged i.e da->name.
236  * @param[in] chunk to debug.
237  */
238 void fr_ext_debug(fr_ext_t const *def, char const *name, void const *chunk)
239 {
240  int i;
241 
242  FR_FAULT_LOG("ext: %s - total len = %zu", name, talloc_get_size(chunk));
243  for (i = 0; i < (int)def->max; i++) {
244  uint8_t *chunk_ext = CHUNK_EXT(def->info, def->offset_of_exts);
245  if (chunk_ext[i]) {
246  void *ext = CHUNK_EXT_PTR(chunk, chunk_ext, i);
247  size_t ext_len = fr_ext_len(def, chunk, i);
248 
249  char const *ext_name = fr_table_ordered_str_by_num(def->name_table,
250  *def->name_table_len,
251  i, "<INVALID>");
252 
253  if (ext_len > 1024) {
254  FR_FAULT_LOG("ext: %s - possibly bad length %zu - limiting dump to 1024",
255  ext_name, ext_len);
256  ext_len = 1024;
257  }
258 
259  FR_FAULT_LOG("ext: %s - start=%p end=%p len=%zu",
260  ext_name, ext, ((uint8_t *)ext) + ext_len, ext_len);
261  FR_FAULT_LOG_HEX(ext, ext_len);
262  }
263  }
