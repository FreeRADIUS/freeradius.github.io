29 /*
30  * We use *inverse* numbers to avoid numerical calculation issues.
31  *
32  * i.e. The bad way is to take two small numbers divide them by
33  * alpha / beta and then add them. That process can drop the
34  * lower digits. Instead, we take two small numbers, add them,
35  * and then divide the result by alpha / beta.
36  */
37 #define IBETA (4)
38 #define IALPHA (8)
39 
40 #define DIFF(_rtt, _t) (_rtt < _t ? (_t - _rtt) : (_rtt - _t))
41 #define RTTVAR(_rtt, _rttvar, _t) ((((IBETA - 1) * _rttvar) + DIFF(_rtt, _t)) / IBETA)
42 #define RTT(_old, _new) ((_new + ((IALPHA - 1) * _old)) / IALPHA)

94  *
95  */
96 static void fr_load_generator_send(fr_load_t *l, fr_time_t now, int count)
97 {
98  int i;
99 
100  /*
101  * Send as many packets as necessary.
102  */
103  l->stats.sent += count;
104  l->stats.last_send = now;
105 
106  /*
107  * Run the callback AFTER we set the timer. Which makes
108  * it more likely that the next timer fires on time.
109  */
110  for (i = 0; i < count; i++) {
111  l->callback(now + i, l->uctx);
112  }
113 }

115 static void load_timer(fr_event_list_t *el, fr_time_t now, void *uctx)
116 {
117  fr_load_t *l = uctx;
118  fr_time_t delta;
119  int count;
120 
121  /*
122  * Keep track of the overall maximum backlog for the
123  * duration of the entire test run.
124  */
125  l->stats.backlog = l->stats.sent - l->stats.received;
126  if (l->stats.backlog > l->stats.max_backlog) l->stats.max_backlog = l->stats.backlog;
127 
128  /*
129  * If we're done this step, go to the next one.
130  */
131  if (l->next >= l->step_end) {
132  l->step_start = l->next;
133  l->step_end = l->next + ((uint64_t) l->config->duration) * NSEC;
134  l->step_received = l->stats.received;
135  l->pps += l->config->step;
136  l->stats.pps = l->pps;
137  l->stats.skipped = 0;
138  l->delta = (NSEC * ((uint64_t) l->config->parallel)) / l->pps;
139 
140  /*
141  * Stop at max PPS, if it's set. Otherwise
142  * continue without limit.
143  */
144  if (l->config->max_pps && (l->pps > l->config->max_pps)) {
145  l->state = FR_LOAD_STATE_DRAINING;
146  return;
147  }
148  }
149 
150  /*
151  * We don't have "pps" packets in the backlog, go send
152  * some more. We scale the backlog by 1000 milliseconds
153  * per second. Then, multiple the PPS by the number of
154  * milliseconds of backlog we want to keep.
155  *
156  * If the backlog is smaller than packets/s *
157  * milliseconds of backlog, then keep sending.
158  * Otherwise, switch to a gated mode where we only send
159  * new packets once a reply comes in.
160  */
161  if (((uint32_t) l->stats.backlog * 1000) < (l->pps * l->config->milliseconds)) {
162  l->state = FR_LOAD_STATE_SENDING;
163  l->stats.blocked = false;
164  count = l->config->parallel;
165  l->stats.skipped = 0;
166 
167  /*
168  * Limit "count" so that it doesn't over-run backlog.
169  */
170  if (((uint32_t) ((count + l->stats.backlog) * 1000)) > (l->pps * l->config->milliseconds)) {
171  count = (count + l->stats.backlog) - ((l->pps * l->config->milliseconds) / 1000);
172  }
173 
174  } else {
175 
176  /*
177  * We have too many packets in the backlog, we're
178  * gated. Don't send more packets until we have
179  * a reply.
180  *
181  * Note that we will send *these* packets.
182  */
183  l->state = FR_LOAD_STATE_GATED;
184  l->stats.blocked = true;
185  count = 0;
186  l->stats.skipped += l->count;
187  }
188 
189  /*
190  * Skip timers if we're too busy.
191  */
192  l->next += l->delta;
193  if (l->next < now) {
194  while ((l->next + l->delta) < now) {
195 // l->stats.skipped += l->count;
196  l->next += l->delta;
197  }
198  }
199  delta = l->next - now;
200 
201  /*
202  * Set the timer for the next packet.
203  */
204  if (fr_event_timer_in(l, el, &l->ev, delta, load_timer, l) < 0) {
205  l->state = FR_LOAD_STATE_DRAINING;
206  return;
207  }
208 
209  if (count) fr_load_generator_send(l, now, count);
210 }

213 /** Start the load generator.
214  *
215  */
216 int fr_load_generator_start(fr_load_t *l)
217 {
218  l->stats.start = fr_time();
219  l->step_start = l->stats.start;
220  l->step_end = l->step_start + ((uint64_t) l->config->duration) * NSEC;
221 
222  l->pps = l->config->start_pps;
223  l->stats.pps = l->pps;
224  l->count = l->config->parallel;
225 
226  l->delta = (NSEC * ((uint64_t) l->config->parallel)) / l->pps;
227  l->next = l->step_start + l->delta;
228 
229  load_timer(l->el, l->step_start, l);
230  return 0;
231 }

246 /** Tell the load generator that we have a reply to a packet we sent.
247  *
248  */
249 fr_load_reply_t fr_load_generator_have_reply(fr_load_t *l, fr_time_t request_time)
250 {
251  fr_time_t now;
252  fr_time_delta_t t;
253 
254  /*
255  * Note that the replies may come out of order with
256  * respect to the request. So we can't use this reply
257  * for any kind of timing.
258  */
259  now = fr_time();
260  t = now - request_time;
261 
262  l->stats.rttvar = RTTVAR(l->stats.rtt, l->stats.rttvar, t);
263  l->stats.rtt = RTT(l->stats.rtt, t);
264 
265  l->stats.received++;
266 
267  /*
268  * t is in nanoseconds.
269  */
270  if (t < 1000) {
271  l->stats.times[0]++; /* < microseconds */
272  } else if (t < 10000) {
273  l->stats.times[1]++; /* microseconds */
274  } else if (t < 100000) {
275  l->stats.times[2]++; /* 10s of microseconds */
276  } else if (t < 1000000) {
277  l->stats.times[3]++; /* 100s of microiseconds */
278  } else if (t < 10000000) {
279  l->stats.times[4]++; /* milliseconds */
280  } else if (t < 100000000) {
281  l->stats.times[5]++; /* 10s of milliseconds */
282  } else if (t < NSEC) {
283  l->stats.times[6]++; /* 100s of milliseconds */
284  } else {
285  l->stats.times[7]++; /* seconds */
286  }
287 
288  /*
289  * Still sending packets. Rely on the timer to send more
290  * packets.
291  */
292  if (l->state == FR_LOAD_STATE_SENDING) return FR_LOAD_CONTINUE;
293 
294  /*
295  * The send code has decided that the backlog is too
296  * high. New requests are blocked until replies come in.
297  * Since we have a reply, send another request.
298  */
299  if (l->state == FR_LOAD_STATE_GATED) {
300  if (l->stats.skipped > 0) {
301  l->stats.skipped--;
302  fr_load_generator_send(l, now, 1);
303  }
304  return FR_LOAD_CONTINUE;
305  }
306 
307  /*
308  * We're still sending or gated, tell the caller to
309  * continue.
310  */
311  if (l->state != FR_LOAD_STATE_DRAINING) {
312  return FR_LOAD_CONTINUE;
313  }
314  /*
315  * Not yet received all replies. Wait until we have all
316  * replies.
317  */
318  if (l->stats.received < l->stats.sent) return FR_LOAD_CONTINUE;
319 
320  l->stats.end = now;
321  return FR_LOAD_DONE;
322 }

324 /** Print load generator statistics in CVS format.
325  *
326  */
327 size_t fr_load_generator_stats_sprint(fr_load_t *l, fr_time_t now, char *buffer, size_t buflen)
328 {
329  double now_f, last_send_f;
330 
331  if (!l->header) {
332  l->header = true;
333  return snprintf(buffer, buflen, "\"time\",\"last_packet\",\"rtt\",\"rttvar\",\"pps\",\"pps_accepted\",\"sent\",\"received\",\"backlog\",\"max_backlog\",\"<usec\",\"us\",\"10us\",\"100us\",\"ms\",\"10ms\",\"100ms\",\"s\",\"blocked\"\n");
334  }
335 
336 
337  now_f = now - l->stats.start;
338  now_f /= NSEC;
339 
340  last_send_f = l->stats.last_send - l->stats.start;
341  last_send_f /= NSEC;
342 
343  /*
344  * Track packets/s. Since times are in nanoseconds, we
345  * have to scale the counters up by NSEC. And since NSEC
346  * is 1B, the calculations have to be done via 64-bit
347  * numbers, and then converted to a final 32-bit counter.
348  */
349  if (now > l->step_start) {
350  l->stats.pps_accepted = (((uint64_t) (l->stats.received - l->step_received)) * NSEC) / (now - l->step_start);
351  }
352 
353  return snprintf(buffer, buflen,
354  "%f,%f,"
355  "%" PRIu64 ",%" PRIu64 ","
356  "%d,%d,"
357  "%d,%d,"
358  "%d,%d,"
359  "%d,%d,%d,%d,%d,%d,%d,%d,"
360  "%d\n",
361  now_f, last_send_f,
362  l->stats.rtt, l->stats.rttvar,
363  l->stats.pps, l->stats.pps_accepted,
364  l->stats.sent, l->stats.received,
365  l->stats.backlog, l->stats.max_backlog,
366  l->stats.times[0], l->stats.times[1], l->stats.times[2], l->stats.times[3],
367  l->stats.times[4], l->stats.times[5], l->stats.times[6], l->stats.times[7],
368  l->stats.blocked);
369 }
