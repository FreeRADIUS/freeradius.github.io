29 /** Convert hex strings to binary data
30  *
31  * @param[out] err If non-null contains any parse errors.
32  * @param[out] out Output buffer to write to.
33  * @param[in] in Input string.
34  * @param[in] no_trailing Error and return 0 if we find an odd length hex
35  * string.
36  * @return
37  * - >=0 the number of bytes written to out.
38  * - 0 no input data, or parse error.
39  * - <0 number of bytes we would have needed to copy the next hexit.
40  */
41 ssize_t fr_hex2bin(fr_sbuff_parse_error_t *err, fr_dbuff_t *out, fr_sbuff_t *in, bool no_trailing)
42 {
43  size_t total = 0;
44  fr_sbuff_t our_in = FR_SBUFF_NO_ADVANCE(in);
45  fr_dbuff_t our_out = FR_DBUFF_NO_ADVANCE(out);
46 
47  while (fr_sbuff_extend_lowat(NULL, &our_in, 2) >= 2) {
48  char *c1, *c2 = NULL;
49 
50  if(!(c1 = memchr(hextab, tolower((int) *fr_sbuff_current(&our_in)), sizeof(hextab))) ||
51  !(c2 = memchr(hextab, tolower((int) *(fr_sbuff_current(&our_in) + 1)), sizeof(hextab)))) {
52  if (!c2 && no_trailing) {
53  got_trailing:
54  if (err) *err = FR_SBUFF_PARSE_ERROR_TRAILING;
55  return 0;
56  }
57  goto done;
58  }
59 
60  FR_DBUFF_IN_BYTES_RETURN(&our_out, ((c1 - hextab) << 4) + (c2 - hextab));
61 
62  fr_sbuff_advance(&our_in, 2);
63  total++;
64  };
65 
66  if (no_trailing && (fr_sbuff_remaining(&our_in) > 0) &&
67  memchr(hextab, tolower((int) *our_in.p), sizeof(hextab))) goto got_trailing;
68 
69 done:
70  fr_sbuff_set(in, &our_in);
71  fr_dbuff_set(out, &our_out);
72 
73  if (err) *err = FR_SBUFF_PARSE_OK;
74 
75  return total;
76 }

78 /** Convert binary data to a hex string
79  *
80  * Ascii encoded hex string will not be prefixed with '0x'
81  *
82  * @param[out] out Output buffer to write to.
83  * @param[in] in input.
84  * @param[in] len how many bytes convert to hex.
85  * Pass SIZE_MAX to copy all available data.
86  * @return
87  * - >=0 the number of bytes written to out.
88  * - <0 number of bytes we would have needed to print the next hexit.
89  */
90 ssize_t fr_bin2hex(fr_sbuff_t *out, fr_dbuff_t *in, size_t len)
91 {
92  size_t total = 0;
93 
94  while ((fr_dbuff_extend_lowat(NULL, in, 2) > 0) && (total < len)) {
95  FR_SBUFF_IN_CHAR_RETURN(out, hextab[((*in->p) >> 4) & 0x0f], hextab[*in->p & 0x0f]);
96 
97  fr_dbuff_advance(in, 1);
98  total++;
99  };
100  return total * 2;
101 }
