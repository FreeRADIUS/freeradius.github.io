56 /** Change the state of a connection to READONLY execute a command and switch to READWRITE
57  *
58  * @param[out] status_out Where to write the status from the command.
59  * @param[out] reply_out Where to write the reply associated with the highest priority status.
60  * @param[in] request The current request.
61  * @param[in] conn to issue commands with.
62  * @param[in] argc Redis command argument count.
63  * @param[in] argv Redis command arguments.
64  * @return
65  * - 0 success.
66  * - -1 normal failure.
67  * - -2 failure that may leave the connection in a READONLY state.
68  */
69 static int redis_command_read_only(fr_redis_rcode_t *status_out, redisReply **reply_out,
70  request_t *request, fr_redis_conn_t *conn, int argc, char const **argv)
71 {
72  bool maybe_more = false;
73  redisReply *reply;
74  fr_redis_rcode_t status;
75 
76  *reply_out = NULL;
77 
78  redisAppendCommand(conn->handle, "READONLY");
79  redisAppendCommandArgv(conn->handle, argc, argv, NULL);
80  redisAppendCommand(conn->handle, "READWRITE");
81 
82  /*
83  * Process the response for READONLY
84  */
85  reply = NULL; /* Doesn't set reply to NULL on error *sigh* */
86  if (redisGetReply(conn->handle, (void **)&reply) == REDIS_OK) maybe_more = true;
87  status = fr_redis_command_status(conn, reply);
88  if (status != REDIS_RCODE_SUCCESS) {
89  REDEBUG("Setting READONLY failed");
90 
91  *reply_out = reply;
92  *status_out = status;
93 
94  if (maybe_more) {
95  if (redisGetReply(conn->handle, (void **)&reply) != REDIS_OK) return -1;
96  fr_redis_reply_free(&reply);
97  if (redisGetReply(conn->handle, (void **)&reply) != REDIS_OK) return -1;
98  fr_redis_reply_free(&reply);
99  }
100  return -1;
101  }
102 
103  fr_redis_reply_free(&reply);
104 
105  /*
106  * Process the response for the command
107  */
108  if (redisGetReply(conn->handle, (void **)&reply) == REDIS_OK) maybe_more = true;
109  status = fr_redis_command_status(conn, reply);
110  if (status != REDIS_RCODE_SUCCESS) {
111  *reply_out = reply;
112  *status_out = status;
113 
114  if (maybe_more) {
115  if (redisGetReply(conn->handle, (void **)&reply) != REDIS_OK) return -1;
116  fr_redis_reply_free(&reply);
117  }
118  return -1;
119  }
120 
121  *reply_out = reply;
122  reply = NULL;
123  *status_out = status;
124 
125  /*
126  * Process the response for READWRITE
127  */
128  if ((redisGetReply(conn->handle, (void **)&reply) != REDIS_OK) ||
129  (fr_redis_command_status(conn, reply) != REDIS_RCODE_SUCCESS)) {
130  REDEBUG("Setting READWRITE failed");
131 
132  fr_redis_reply_free(&reply); /* There could be a response we need to free */
133  fr_redis_reply_free(reply_out);
134  *reply_out = reply;
135  *status_out = status;
136 
137  return -2;
138  }
139  fr_redis_reply_free(&reply); /* Free READWRITE response */
140 
141  return 0;
142 }

210 /** Return the node that is currently servicing a particular key
211  *
212  * @ingroup xlat_functions
213  */
214 static xlat_action_t redis_node_xlat(TALLOC_CTX *ctx, fr_cursor_t *out,
215  request_t *request, void const *xlat_inst,
216  UNUSED void *xlat_thread_inst,
217  fr_value_box_t **in)
218 {
219  rlm_redis_t const *inst = talloc_get_type_abort_const(*((void const * const *)xlat_inst),
220  rlm_redis_t);
221 
222  fr_redis_cluster_key_slot_t const *key_slot;
223  fr_redis_cluster_node_t const *node;
224  fr_ipaddr_t ipaddr;
225  uint16_t port;
226 
227  char const *p;
228  char *q;
229  char const *key;
230  size_t key_len;
231  unsigned long idx = 0;
232  fr_value_box_t *vb;
233 
234  if (!in) {
235  REDEBUG("Missing key");
236  return XLAT_ACTION_FAIL;
237  }
238 
239  if (fr_value_box_list_concat(ctx, *in, in, FR_TYPE_STRING, true) < 0) {
240  RPEDEBUG("Failed concatenating input");
241  return XLAT_ACTION_FAIL;
242  }
243 
244  key = p = (*in)->vb_strvalue;
245  p = strchr(p, ' '); /* Look for index */
246  if (p) {
247  key_len = p - key;
248 
249  idx = strtoul(p, &q, 10);
250  if (q == p) {
251  REDEBUG("Tailing garbage after node index");
252  return XLAT_ACTION_FAIL;
253  }
254  } else {
255  key_len = (*in)->vb_length;
256  }
257 
258  key_slot = fr_redis_cluster_slot_by_key(inst->cluster, request, (uint8_t const *)key, key_len);
259  if (idx == 0) {
260  node = fr_redis_cluster_master(inst->cluster, key_slot);
261  } else {
262  node = fr_redis_cluster_slave(inst->cluster, key_slot, idx - 1);
263  }
264 
265  if (!node) {
266  RDEBUG2("No node available for this key slot");
267  return XLAT_ACTION_DONE;
268  }
269 
270  if ((fr_redis_cluster_ipaddr(&ipaddr, node) < 0) || (fr_redis_cluster_port(&port, node) < 0)) {
271  REDEBUG("Failed retrieving node information");
272  return XLAT_ACTION_FAIL;
273  }
274 
275  MEM(vb = fr_value_box_alloc_null(ctx));
276  fr_value_box_asprintf(vb, vb, NULL, false, "%pV:%u", fr_box_ipaddr(ipaddr), port);
277  fr_cursor_append(out, vb);
278 
279  return XLAT_ACTION_DONE;
280 }

283 /** Xlat to make calls to redis
284  *
285 @verbatim
286 %{redis:<redis command>}
287 @endverbatim
288  *
289  * @ingroup xlat_functions
290  */
291 static ssize_t redis_xlat(UNUSED TALLOC_CTX *ctx, char **out, size_t outlen,
292  void const *mod_inst, UNUSED void const *xlat_inst,
293  request_t *request, char const *fmt)
294 {
295  rlm_redis_t const *inst = mod_inst;
296  fr_redis_conn_t *conn;
297 
298  bool read_only = false;
299  uint8_t const *key = NULL;
300  size_t key_len = 0;
301 
302  fr_redis_cluster_state_t state;
303  fr_redis_rcode_t status;
304  redisReply *reply = NULL;
305  int s_ret;
306 
307  size_t len;
308  int ret;
309 
310  char const *p = fmt, *q;
311 
312  int argc;
313  char const *argv[MAX_REDIS_ARGS];
314  char argv_buf[MAX_REDIS_COMMAND_LEN];
315 
316  if (p[0] == '-') {
317  p++;
318  read_only = true;
319  }
320 
321  /*
322  * Hack to allow querying against a specific node for testing
323  */
324  if (p[0] == '@') {
325  fr_socket_t node_addr;
326  fr_pool_t *pool;
327 
328  RDEBUG3("Overriding node selection");
329 
330  p++;
331  q = strchr(p, ' ');
332  if (!q) {
333  REDEBUG("Found node specifier but no command, format is [-][@<host>[:port]] <redis command>");
334  return -1;
335  }
336 
337  if (fr_inet_pton_port(&node_addr.inet.dst_ipaddr, &node_addr.inet.dst_port, p, q - p, AF_UNSPEC, true, true) < 0) {
338  RPEDEBUG("Failed parsing node address");
339  return -1;
340  }
341 
342  p = q + 1;
343 
344  if (fr_redis_cluster_pool_by_node_addr(&pool, inst->cluster, &node_addr, true) < 0) {
345  RPEDEBUG("Failed locating cluster node");
346  return -1;
347  }
348 
349  conn = fr_pool_connection_get(pool, request);
350  if (!conn) {
351  REDEBUG("No connections available for cluster node");
352  return -1;
353  }
354 
355  argc = rad_expand_xlat(request, p, MAX_REDIS_ARGS, argv, false, sizeof(argv_buf), argv_buf);
356  if (argc <= 0) {
357  RPEDEBUG("Invalid command: %s", p);
358  arg_error:
359  fr_pool_connection_release(pool, request, conn);
360  return -1;
361  }
362  if (argc >= (MAX_REDIS_ARGS - 1)) {
363  RPEDEBUG("Too many parameters; increase MAX_REDIS_ARGS and recompile: %s", p);
364  goto arg_error;
365  }
366 
367  RDEBUG2("Executing command: %s", argv[0]);
368  if (argc > 1) {
369  RDEBUG2("With argments");
370  RINDENT();
371  for (int i = 1; i < argc; i++) RDEBUG2("[%i] %s", i, argv[i]);
372  REXDENT();
373  }
374 
375  if (!read_only) {
376  reply = redisCommandArgv(conn->handle, argc, argv, NULL);
377  status = fr_redis_command_status(conn, reply);
378  } else if (redis_command_read_only(&status, &reply, request, conn, argc, argv) == -2) {
379  goto close_conn;
380  }
381 
382  if (!reply) goto fail;
383 
384  switch (status) {
385  case REDIS_RCODE_MOVE:
386  {
387  fr_value_box_t vb;
388 
389  if (fr_redis_reply_to_value_box(NULL, &vb, reply, FR_TYPE_STRING, NULL) == 0) {
390  REDEBUG("Key served by a different node: %pV", &vb);
391  fr_value_box_clear(&vb);
392  }
393  goto fail;
394  }
395 
396  case REDIS_RCODE_SUCCESS:
397  goto reply_parse;
398 
399  case REDIS_RCODE_RECONNECT:
400  close_conn:
401  fr_pool_connection_close(pool, request, conn);
402  ret = -1;
403  goto finish;
404 
405  default:
406  fail:
407  fr_pool_connection_release(pool, request, conn);
408  ret = -1;
409  goto finish;
410  }
411  }
412 
413  /*
414  * Normal node selection and execution based on key
415  */
416  argc = rad_expand_xlat(request, p, MAX_REDIS_ARGS, argv, false, sizeof(argv_buf), argv_buf);
417  if (argc <= 0) {
418  RPEDEBUG("Invalid command: %s", p);
419  ret = -1;
420  goto finish;
421  }
422 
423  if (argc >= (MAX_REDIS_ARGS - 1)) {
424  RPEDEBUG("Too many parameters; increase MAX_REDIS_ARGS and recompile: %s", p);
425  ret = -1;
426  goto finish;
427  }
428 
429  /*
430  * If we've got multiple arguments, the second one is usually the key.
431  * The Redis docs say commands should be analysed first to get key
432  * positions, but this involves sending them to the server, which is
433  * just as expensive as sending them to the wrong server and receiving
434  * a redirect.
435  */
436  if (argc > 1) {
437  key = (uint8_t const *)argv[1];
438  key_len = strlen((char const *)key);
439  }
440  for (s_ret = fr_redis_cluster_state_init(&state, &conn, inst->cluster, request, key, key_len, read_only);
441  s_ret == REDIS_RCODE_TRY_AGAIN; /* Continue */
442  s_ret = fr_redis_cluster_state_next(&state, &conn, inst->cluster, request, status, &reply)) {
443  RDEBUG2("Executing command: %s", argv[0]);
444  if (argc > 1) {
445  RDEBUG2("With arguments");
446  RINDENT();
447  for (int i = 1; i < argc; i++) RDEBUG2("[%i] %s", i, argv[i]);
448  REXDENT();
449  }
450  if (!read_only) {
451  reply = redisCommandArgv(conn->handle, argc, argv, NULL);
452  status = fr_redis_command_status(conn, reply);
453  } else if (redis_command_read_only(&status, &reply, request, conn, argc, argv) == -2) {
454  state.close_conn = true;
455  }
456  }
457  if (s_ret != REDIS_RCODE_SUCCESS) {
458  ret = -1;
459  goto finish;
460  }
461 
462  if (!fr_cond_assert(reply)) {
463  ret = -1;
464  goto finish;
465  }
466 
467 reply_parse:
468  switch (reply->type) {
469  case REDIS_REPLY_INTEGER:
470  ret = snprintf(*out, outlen, "%lld", reply->integer);
471  break;
472 
473  case REDIS_REPLY_STATUS:
474  case REDIS_REPLY_STRING:
475  len = (((size_t)reply->len) >= outlen) ? outlen - 1: (size_t) reply->len;
476  memcpy(*out, reply->str, len);
477  (*out)[len] = '\0';
478  ret = reply->len;
479  break;
480 
481  default:
482  REDEBUG("Server returned non-value type \"%s\"",
483  fr_table_str_by_value(redis_reply_types, reply->type, "<UNKNOWN>"));
484  ret = -1;
485  break;
486  }
487 
488 finish:
489  fr_redis_reply_free(&reply);
490  return ret;
491 }

493 static int mod_bootstrap(void *instance, CONF_SECTION *conf)
494 {
495  rlm_redis_t *inst = instance;
496  char *name;
497  xlat_t const *xlat;
498

506  */

516 
517  return 0;
518 }
