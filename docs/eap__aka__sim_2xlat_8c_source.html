32 /** Returns the SIM method EAP-SIM or EAP-AKA hinted at by the user identifier
33  *
34 @verbatim
35 %{aka_sim_id_method:&id_attr}
36 @endverbatim
37  *
38  * @ingroup xlat_functions
39  */
40 static ssize_t aka_sim_xlat_id_method_xlat(TALLOC_CTX *ctx, char **out, UNUSED size_t outlen,
41  UNUSED void const *mod_inst, UNUSED void const *xlat_inst,
42  request_t *request, char const *fmt)
43 {
44  tmpl_t *vpt;
45  TALLOC_CTX *our_ctx = talloc_init_const("aka_sim_xlat");
46  ssize_t slen, id_len;
47  char const *p = fmt, *id, *method;
48  fr_aka_sim_id_type_t type_hint;
49  fr_aka_sim_method_hint_t method_hint;
50 
51  /*
52  * Trim whitespace
53  */
54  fr_skip_whitespace(p);
55 
56  slen = tmpl_afrom_attr_substr(our_ctx, NULL, &vpt, &FR_SBUFF_IN(p, strlen(p)),
57  NULL,
58  &(tmpl_rules_t){
59  .dict_def = request->dict,
60  .prefix = TMPL_ATTR_REF_PREFIX_AUTO
61  });
62  if (slen <= 0) {
63  RPEDEBUG("Invalid attribute reference");
64  error:
65  talloc_free(our_ctx);
66  return -1;
67  }
68 
69  if (tmpl_aexpand(our_ctx, &id, request, vpt, NULL, NULL) < 0) {
70  RPEDEBUG2("Failing expanding ID attribute");
71  goto error;
72  }
73 
74  id_len = talloc_array_length(id) - 1;
75  if (fr_aka_sim_id_type(&type_hint, &method_hint, id, id_len) < 0) {
76  RPEDEBUG2("AKA/SIM Id \"%pV\" has unrecognised format", fr_box_strvalue_len(id, id_len));
77  goto error;
78  }
79 
80  switch (method_hint) {
81  default:
82  case AKA_SIM_METHOD_HINT_UNKNOWN:
83  *out = NULL;
84  return 0;
85 
86  case AKA_SIM_METHOD_HINT_SIM:
87  method = fr_dict_enum_name_by_value(attr_eap_aka_sim_method_hint,
88  fr_box_uint32(FR_METHOD_HINT_VALUE_SIM));
89  break;
90 
91  case AKA_SIM_METHOD_HINT_AKA:
92  method = fr_dict_enum_name_by_value(attr_eap_aka_sim_method_hint,
93  fr_box_uint32(FR_METHOD_HINT_VALUE_AKA));
94  break;
95 
96  case AKA_SIM_METHOD_HINT_AKA_PRIME:
97  method = fr_dict_enum_name_by_value(attr_eap_aka_sim_method_hint,
98  fr_box_uint32(FR_METHOD_HINT_VALUE_AKA_PRIME));
99  break;
100  }
101 
102  *out = talloc_typed_strdup(ctx, method);
103  talloc_free(our_ctx);
104 
105  return talloc_array_length(*out) - 1;
106 }

237 /** Decrypt a 3gpp pseudonym
238  *
239 @verbatim
240 %{3gpp_pseudonym_decrypt:&id_attr &key_attr}
241 @endverbatim
242  *
243  * @ingroup xlat_functions
244  */
245 static ssize_t aka_sim_3gpp_pseudonym_decrypt_xlat(TALLOC_CTX *ctx, char **out, UNUSED size_t outlen,
246  UNUSED void const *mod_inst, UNUSED void const *xlat_inst,
247  request_t *request, char const *fmt)
248 {
249  tmpl_t *id_vpt, *key_vpt;
250  TALLOC_CTX *our_ctx = talloc_init_const("aka_sim_xlat");
251  ssize_t slen, id_len, key_len;
252  uint8_t tag;
253  char out_tag;
254  uint8_t *key;
255  char decrypted[AKA_SIM_IMSI_MAX_LEN + 1];
256  char const *p = fmt, *id;
257 
258  /*
259  * Trim whitespace
260  */
261  fr_skip_whitespace(p);
262 
263  slen = tmpl_afrom_attr_substr(our_ctx, NULL, &id_vpt, &FR_SBUFF_IN(p, strlen(p)),
264  NULL,
265  &(tmpl_rules_t){
266  .dict_def = request->dict,
267  .prefix = TMPL_ATTR_REF_PREFIX_AUTO
268  });
269  if (slen <= 0) {
270  RPEDEBUG("Invalid ID attribute reference");
271  error:
272  talloc_free(our_ctx);
273  return -1;
274  }
275 
276  p += slen;
277  if (*p != ' ') {
278  REDEBUG2("Missing key argument");
279  goto error;
280  }
281  p++;
282 
283  slen = tmpl_afrom_attr_substr(our_ctx, NULL, &key_vpt, &FR_SBUFF_IN(p, strlen(p)),
284  NULL,
285  &(tmpl_rules_t){
286  .dict_def = request->dict,
287  .prefix = TMPL_ATTR_REF_PREFIX_AUTO
288  });
289  if (slen <= 0) {
290  RPEDEBUG("Invalid key attribute reference");
291  goto error;
292  }
293 
294  if (tmpl_aexpand(our_ctx, &id, request, id_vpt, NULL, NULL) < 0) {
295  RPEDEBUG2("Failing expanding ID attribute");
296  goto error;
297  }
298 
299 
300  if (tmpl_aexpand(our_ctx, &key, request, key_vpt, NULL, NULL) < 0) {
301  RPEDEBUG2("Failing expanding Key attribute");
302  goto error;
303  }
304 
305  id_len = talloc_array_length(id);
306  if (id_len != (AKA_SIM_3GPP_PSEUDONYM_LEN + 1)) {
307  REDEBUG2("3gpp pseudonym incorrect length, expected %i bytes, got %zu bytes",
308  AKA_SIM_3GPP_PSEUDONYM_LEN + 1, id_len);
309  goto error;
310  }
311 
312  key_len = talloc_array_length(key);
313  if (key_len != 16) {
314  REDEBUG2("Decryption key incorrect length, expected %i bytes, got %zu bytes", 16, key_len);
315  goto error;
316  }
317 
318  tag = fr_aka_sim_id_3gpp_pseudonym_tag(id);
319  switch (tag) {
320  case ID_TAG_SIM_PSEUDONYM_B64:
321  out_tag = ID_TAG_SIM_PERMANENT;
322  break;
323 
324  case ID_TAG_AKA_PSEUDONYM_B64:
325  out_tag = ID_TAG_AKA_PERMANENT;
326  break;
327 
328  case ID_TAG_AKA_PRIME_PSEUDONYM_B64:
329  out_tag = ID_TAG_AKA_PRIME_PERMANENT;
330  break;
331 
332  default:
333  REDEBUG2("Unexpected tag value (%u) in AKA/SIM Id \"%pV\"", tag, fr_box_strvalue_len(id, id_len));
334  goto error;
335  }
336 
337  RDEBUG2("Decrypting \"%pV\"", fr_box_strvalue_len(id, id_len));
338  if (fr_aka_sim_id_3gpp_pseudonym_decrypt(decrypted, id, key) < 0) {
339  RPEDEBUG2("Failed decrypting AKA/SIM Id");
340  goto error;
341  }
342 
343  /*
344  * Recombine unencrypted IMSI with tag
345  */
346  MEM(*out = talloc_typed_asprintf(ctx, "%c%s", out_tag, decrypted));
347  talloc_free(our_ctx);
348 
349  return talloc_array_length(*out) - 1;
350 }

352 /** Encrypts a 3gpp pseudonym
353  *
354 @verbatim
355 %{3gpp_pseudonym_encrypt:&id_attr &key_attr key_index}
356 @endverbatim
357  *
358  * @ingroup xlat_functions
359  */
360 static ssize_t aka_sim_3gpp_pseudonym_encrypt_xlat(TALLOC_CTX *ctx, char **out, UNUSED size_t outlen,
361  UNUSED void const *mod_inst, UNUSED void const *xlat_inst,
362  request_t *request, char const *fmt)
363 {
364  tmpl_t *id_vpt, *key_vpt;
365  TALLOC_CTX *our_ctx = talloc_init_const("aka_sim_xlat");
366  ssize_t slen, id_len, key_len;
367  uint8_t *key, tag = 0;
368  unsigned long key_index;
369  char encrypted[AKA_SIM_3GPP_PSEUDONYM_LEN + 1];
370  char const *p = fmt, *id;
371  char const *id_p, *id_end;
372  fr_aka_sim_id_type_t type_hint;
373  fr_aka_sim_method_hint_t method_hint;
374 
375  /*
376  * Trim whitespace
377  */
378  fr_skip_whitespace(p);
379 
380  slen = tmpl_afrom_attr_substr(our_ctx, NULL, &id_vpt, &FR_SBUFF_IN(p, strlen(p)),
381  NULL,
382  &(tmpl_rules_t){
383  .dict_def = request->dict,
384  .prefix = TMPL_ATTR_REF_PREFIX_AUTO
385  });
386  if (slen <= 0) {
387  RPEDEBUG("Invalid ID attribute reference");
388  error:
389  talloc_free(our_ctx);
390  return -1;
391  }
392 
393  p += slen;
394  if (*p != ' ') {
395  REDEBUG2("Missing key argument");
396  goto error;
397  }
398  p++;
399 
400  slen = tmpl_afrom_attr_substr(our_ctx, NULL, &key_vpt, &FR_SBUFF_IN(p, strlen(p)),
401  NULL,
402  &(tmpl_rules_t){
403  .dict_def = request->dict,
404  .prefix = TMPL_ATTR_REF_PREFIX_AUTO
405  });
406  if (slen <= 0) {
407  RPEDEBUG("Invalid key attribute reference");
408  goto error;
409  }
410  p += slen;
411 
412  if (*p != ' ') {
413  REDEBUG2("Missing key index");
414  goto error;
415  }
416  p++;
417 
418  /*
419  * Get the key index
420  */
421  key_index = strtoul(p, NULL, 10);
422  if (key_index > 15) {
423  REDEBUG2("Key index must be between 0-15");
424  goto error;
425  }
426 
427  /*
428  * Get the ID
429  */
430  if (tmpl_aexpand(our_ctx, &id, request, id_vpt, NULL, NULL) < 0) {
431  RPEDEBUG2("Failing expanding ID attribute");
432  goto error;
433  }
434 
435  /*
436  * Get the key
437  */
438  if (tmpl_aexpand(our_ctx, &key, request, key_vpt, NULL, NULL) < 0) {
439  RPEDEBUG2("Failing expanding Key attribute");
440  goto error;
441  }
442 
443  key_len = talloc_array_length(key);
444  if (key_len != 16) {
445  REDEBUG2("Encryption key incorrect length, expected %i bytes, got %zu bytes", 16, key_len);
446  goto error;
447  }
448 
449  /*
450  * Determine what type/method hints are in the current ID.
451  */
452  id_len = talloc_array_length(id) - 1;
453  if (id_len == (AKA_SIM_IMSI_MAX_LEN + 1)) { /* +1 for ID tag */
454  if (fr_aka_sim_id_type(&type_hint, &method_hint, id, id_len) < 0) {
455  RPEDEBUG2("SIM ID \"%pV\" has unrecognised format", fr_box_strvalue_len(id, id_len));
456  goto error;
457  }
458 
459  if (type_hint != AKA_SIM_ID_TYPE_PERMANENT) {
460  REDEBUG2("SIM ID \"%pV\" is not a permanent identity (IMSI)", fr_box_strvalue_len(id, id_len));
461  goto error;
462  }
463 
464  switch (method_hint) {
465  case AKA_SIM_METHOD_HINT_SIM:
466  tag = ID_TAG_SIM_PSEUDONYM_B64;
467  break;
468 
469  case AKA_SIM_METHOD_HINT_AKA:
470  tag = ID_TAG_AKA_PSEUDONYM_B64;
471  break;
472 
473  case AKA_SIM_METHOD_HINT_AKA_PRIME:
474  tag = ID_TAG_AKA_PRIME_PSEUDONYM_B64;
475  break;
476 
477  case AKA_SIM_METHOD_HINT_UNKNOWN:
478  case AKA_SIM_METHOD_HINT_MAX:
479  REDEBUG2("AKA/SIM ID \"%pV\" does not contain a method hint", fr_box_strvalue_len(id, id_len));
480  goto error;
481  }
482 
483  id_p = id + 1;
484  id_end = (id_p + id_len) - 1;
485  /*
486  * ID lacks a hint byte, figure it out from &control.EAP-Type
487  */
488  } else if ((id_len >= AKA_SIM_IMSI_MIN_LEN) && (id_len <= AKA_SIM_IMSI_MAX_LEN)) {
489  fr_pair_t *eap_type;
490 
491  eap_type = fr_pair_find_by_da(&request->request_pairs, attr_eap_type);
492  if (!eap_type) {
493  REDEBUG("SIM ID does not contain method hint, and no &control.EAP-Type found. "
494  "Don't know what tag to prepend to encrypted identity");
495  goto error;
496  }
497 
498  switch (eap_type->vp_uint32) {
499  case FR_EAP_TYPE_VALUE_SIM:
500  tag = ID_TAG_SIM_PSEUDONYM_B64;
501  break;
502 
503  case FR_EAP_TYPE_VALUE_AKA:
504  tag = ID_TAG_AKA_PSEUDONYM_B64;
505  break;
506 
507  case FR_EAP_TYPE_VALUE_AKA_PRIME:
508  tag = ID_TAG_AKA_PRIME_PSEUDONYM_B64;
509  break;
510 
511  default:
512  REDEBUG("&control.EAP-Type does not match a SIM based EAP-Type (SIM, AKA, AKA-Prime)");
513  break;
514  }
515 
516  id_p = id;
517  id_end = id_p + id_len;
518  } else {
519  REDEBUG2("IMSI incorrect length, expected %i bytes, got %zu bytes", AKA_SIM_IMSI_MAX_LEN + 1,
520  id_len);
521  goto error;
522 
523  }
524 
525  /*
526  * Encrypt the IMSI
527  *
528  * Strip existing tag from the permanent id
529  */
530  if (fr_aka_sim_id_3gpp_pseudonym_encrypt(encrypted, id_p, id_end - id_p, tag, (uint8_t)key_index, key) < 0) {
531  RPEDEBUG2("Failed encrypting SIM ID \"%pV\"", fr_box_strvalue_len(id, id_len));
532  return -1;
533  }
534 
535  MEM(*out = talloc_typed_asprintf(ctx, "%s", encrypted));
536  talloc_free(our_ctx);
537 
538  return talloc_array_length(*out) - 1;
539 }
