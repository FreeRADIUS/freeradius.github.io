73 ssize_t fr_struct_from_network(TALLOC_CTX *ctx, fr_cursor_t *cursor,
74  fr_dict_attr_t const *parent, uint8_t const *data, size_t data_len,
75  fr_dict_attr_t const **child_p,
76  fr_decode_value_t decode_value, void *decoder_ctx)
77 {
78  unsigned int child_num;
79  uint8_t const *p = data, *end = data + data_len;
80  fr_dict_attr_t const *child;
81  fr_pair_t *head = NULL;
82  fr_cursor_t child_cursor;
83  fr_pair_t *vp, *key_vp;
84  unsigned int offset = 0;
85 
86  if (data_len < 1) return -1; /* at least one byte of data */
87 
88  FR_PROTO_HEX_DUMP(data, data_len, "fr_struct_from_network");
89 
90  /*
91  * Record where we were in the list when this function was called
92  */
93  fr_cursor_init(&child_cursor, &head);
94  *child_p = NULL;
95  child_num = 1;
96  key_vp = NULL;
97 
98  /*
99  * Decode structs with length prefixes.
100  */
101  if (da_is_length_field(parent)) {
102  size_t struct_len;
103 
104  struct_len = (p[0] << 8) | p[1];
105  if ((struct_len + 2) > data_len) goto unknown;
106 
107  data_len = struct_len + 2;
108  end = data + data_len;
109  p += 2;
110  }
111 
112  while (p < end) {
113  size_t child_length;
114 
115  /*
116  * Go to the next child. If it doesn't exist, we're done.
117  */
118  child = fr_dict_attr_child_by_num(parent, child_num);
119  if (!child) break;
120 
121  FR_PROTO_HEX_DUMP(p, (end - p), "fr_struct_from_network - child %s (%d)", child->name, child->attr);
122 
123  /*
124  * Check for bit fields.
125  */
126  if (da_is_bit_field(child)) {
127  uint8_t array[8];
128  unsigned int num_bits;
129  uint64_t value;
130 
131  num_bits = offset + child->flags.length;
132  if ((end - p) < ((num_bits + 7) >> 3)) goto unknown;
133 
134  memset(array, 0, sizeof(array));
135  memcpy(&array[0], p, ((num_bits + 7) >> 3)); /* round up to nearest byte */
136 
137  if (offset > 0) array[0] &= (1 << (8 - offset)) - 1; /* mask off bits we don't care about */
138 
139  memcpy(&value, &array[0], sizeof(value));
140  value = htonll(value);
141  value >>= (8 - offset); /* move it to the lower bits */
142  value >>= (56 - child->flags.length);
143 
144  vp = fr_pair_afrom_da(ctx, child);
145  if (!vp) {
146  FR_PROTO_TRACE("fr_struct_from_network - failed allocating child VP");
147  goto unknown;
148  }
149 
150  switch (child->type) {
151  case FR_TYPE_BOOL:
152  vp->vp_bool = value;
153  break;
154 
155  case FR_TYPE_UINT8:
156  vp->vp_uint8 = value;
157  break;
158 
159  case FR_TYPE_UINT16:
160  vp->vp_uint16 = value;
161  break;
162 
163  case FR_TYPE_UINT32:
164  vp->vp_uint32 = value;
165  break;
166 
167  case FR_TYPE_UINT64:
168  vp->vp_uint64 = value;
169  break;
170 
171  default:
172  goto unknown;
173  }
174 
175  vp->type = VT_DATA;
176  vp->vp_tainted = true;
177  fr_cursor_append(&child_cursor, vp);
178  p += (num_bits >> 3);
179  offset = num_bits & 0x07;
180  child_num++;
181  continue;
182  }
183  offset = 0; /* reset for non-bit-field attributes */
184 
185  /*
186  * Decode child TLVs, according to the parent attribute.
187  *
188  * Return only PARTIALLY decoded data. Let the
189  * caller decode the rest.
190  *
191  * @todo - pass TLVs to the "decode_value"
192  * function, so it can decode them.
193  */
194  if (child->type == FR_TYPE_TLV) {
195  *child_p = child;
196 
197  fr_cursor_head(&child_cursor);
198  fr_cursor_tail(cursor);
199  fr_cursor_merge(cursor, &child_cursor); /* Wind to the end of the new pairs */
200  return (p - data);
201  }
202 
203  child_length = child->flags.length;
204 
205  /*
206  * If this field overflows the input, then *all*
207  * of the input is suspect.
208  */
209  if ((p + child_length) > end) {
210  FR_PROTO_TRACE("fr_struct_from_network - child length %zd overflows buffer", child_length);
211  goto unknown;
212  }
213 
214  if (!child_length) child_length = (end - p);
215 
216  /*
217  * Magic values get the callback called.
218  *
219  * Note that if this is an *array* of DNS labels,
220  * the callback should deal with this.
221  */
222  if (decode_value) {
223  ssize_t slen;
224 
225  slen = decode_value(ctx, &child_cursor, NULL, child, p, child_length, decoder_ctx);
226  if (slen < 0) return slen - (p - data);
227 
228  p += slen; /* not always the same as child->flags.length */
229  child_num++; /* go to the next child */
230  if (da_is_key_field(child)) key_vp = fr_cursor_tail(&child_cursor);
231  continue;
232  }
233 
234  /*
235  * We only allow a limited number of data types
236  * inside of a struct.
237  */
238  switch (child->type) {
239  default:
240  FR_PROTO_TRACE("fr_struct_from_network - unknown child type");
241  goto unknown;
242 
243  case FR_TYPE_VALUE:
244  break;
245  }
246 
247  vp = fr_pair_afrom_da(ctx, child);
248  if (!vp) {
249  FR_PROTO_TRACE("fr_struct_from_network - failed allocating child VP");
250  goto unknown;
251  }
252 
253  /*
254  * No protocol-specific data types here (yet).
255  *
256  * If we can't decode this field, then the entire
257  * structure is treated as a raw blob.
258  */
259  if (fr_value_box_from_network(vp, &vp->data, vp->da->type, vp->da, p, child_length, true) < 0) {
260  FR_PROTO_TRACE("fr_struct_from_network - failed decoding child VP");
261  fr_pair_list_free(&vp);
262  unknown:
263  fr_pair_list_free(&head);
264 
265  vp = fr_unknown_from_network(ctx, parent, data, data_len);
266  if (!vp) return -1;
267 
268  /*
269  * And append this one VP to the output cursor.
270  */
271  fr_cursor_append(cursor, vp);
272  return data_len;
273  }
274 
275  vp->type = VT_DATA;
276  vp->vp_tainted = true;
277  fr_cursor_append(&child_cursor, vp);
278 
279  if (da_is_key_field(vp->da)) key_vp = vp;
280 
281  /*
282  * Note that we're decoding fixed fields here.
283  * So we skip the input based on the *known*
284  * length, and not on the *decoded* length.
285  */
286  p += child_length;
287  child_num++; /* go to the next child */
288  }
289 
290  /*
291  * Is there a substructure after this one? If so, go
292  * decode it.
293  */
294  if (key_vp) {
295  ssize_t slen;
296  fr_dict_enum_t const *enumv;
297  child = NULL;
298 
299  FR_PROTO_HEX_DUMP(p, (end - p), "fr_struct_from_network - substruct");
300 
301  /*
302  * Nothing more to decode, don't decode it.
303  */
304  if (p >= end) goto done;
305 
306  enumv = fr_dict_enum_by_value(key_vp->da, &key_vp->data);
307  if (enumv) child = enumv->child_struct[0];
308 
309  if (!child) {
310  unknown_child:
311  /*
312  * Encode the unknown child as attribute
313  * number 0. This choice means we don't
314  * have to look up, or keep track of, the
315  * number of children of the key field.
316  */
317  child = fr_dict_unknown_afrom_fields(ctx, key_vp->da,
318  fr_dict_vendor_num_by_da(key_vp->da), 0);
319  if (!child) goto unknown;
320 
321  vp = fr_unknown_from_network(ctx, child, p, end - p);
322  if (!vp) {
323  fr_dict_unknown_free(&child);
324  return -(p - data);
325  }
326 
327  fr_cursor_append(&child_cursor, vp);
328  p = end;
329  } else {
330  fr_assert(child->type == FR_TYPE_STRUCT);
331 
332  slen = fr_struct_from_network(ctx, &child_cursor, child, p, end - p, child_p,
333  decode_value, decoder_ctx);
334  if (slen <= 0) goto unknown_child;
335  p += slen;
336  }
337 
338  fr_dict_unknown_free(&child);
339 
340  fr_cursor_head(&child_cursor);
341  fr_cursor_tail(cursor);
342  fr_cursor_merge(cursor, &child_cursor); /* Wind to the end of the new pairs */
343 
344  /*
345  * Else return whatever we decoded. Note that if
346  * the substruct ends in a TLV, we decode only
347  * the fixed-length portion of the structure.
348  */
349  return p - data;
350  }
351 
352 done:
353  fr_cursor_head(&child_cursor);
354  fr_cursor_tail(cursor);
355  fr_cursor_merge(cursor, &child_cursor); /* Wind to the end of the new pairs */
356 
357  return data_len;
358 }

384 /** Put bits into an output dbuff
385  *
386  * @param dbuff where the bytes go
387  * @param p where leftover bits go
388  * @param start_bit start bit in the dbuff where the data goes, 0..7
389  * @param num_bits number of bits to write to the output, 0..55
390  * @param data data to write, all in the lower "num_bits" of the uint64_t variable
391  * @return
392  * >= 0 the next value to pass in for start_bit
393  * < 0 no space or invalid start_bit or num_bits parameter
394  */
395 static int put_bits_dbuff(fr_dbuff_t *dbuff, uint8_t *p, int start_bit, uint8_t num_bits, uint64_t data)
396 {
397  uint64_t used_bits;
398 
399  if (start_bit < 0 || start_bit > 7) return -1;
400  if (num_bits < 1 || num_bits > 56) return -1;
401 
402  /* Get bits buffered in *p */
403  used_bits = *p & (-256 >> start_bit);
404 
405  /* Mask out all but the least significant num_bits bits of data */
406  data &= (((uint64_t) 1) << num_bits) - 1;
407 
408  /* Move it towards the most significant end and put used_bits at the top */
409  data <<= (64 - (start_bit + num_bits));
410  data |= used_bits << 56;
411 
412  data = htonll(data);
413 
414  start_bit += num_bits;
415  if (start_bit > 7) FR_DBUFF_IN_MEMCPY_RETURN(dbuff, (uint8_t const *) &data, (size_t)(start_bit / 8));
416 
417  *p = ((uint8_t *) &data)[start_bit / 8];
418  return start_bit % 8;
419 }

421 ssize_t fr_struct_to_network_dbuff(fr_dbuff_t *dbuff,
422  fr_da_stack_t *da_stack, unsigned int depth,
423  fr_cursor_t *cursor, void *encoder_ctx,
424  fr_encode_value_dbuff_t encode_value)
425 {
426  fr_dbuff_t work_dbuff = FR_DBUFF_NO_ADVANCE(dbuff);
427  fr_dbuff_t hdr_dbuff = FR_DBUFF_NO_ADVANCE(dbuff);
428  int offset = 0;
429  unsigned int child_num = 1;
430  bool do_length = false;
431  uint8_t bit_buffer = 0;
432  fr_pair_t const *vp = fr_cursor_current(cursor);
433  fr_dict_attr_t const *key_da, *parent;
434 
435  if (!vp) {
436  fr_strerror_printf("%s: Can't encode empty struct", __FUNCTION__);
437  return -1;
438  }
439 
440  VP_VERIFY(vp);
441  parent = da_stack->da[depth];
442 
443  if (parent->type != FR_TYPE_STRUCT) {
444  fr_strerror_printf("%s: Expected type \"struct\" got \"%s\"", __FUNCTION__,
445  fr_table_str_by_value(fr_value_box_type_table, parent->type, "?Unknown?"));
446  return -1;
447  }
448 
449  /*
450  * @todo - if we get a child which *eventually* has the
451  * given parent, then allow encoding of that struct, too.
452  * This allows us to encode structures automatically,
453  * even if key fields are omitted.
454  */
455  if (vp->da->parent != parent) {
456  fr_strerror_printf("%s: struct encoding is missing previous attributes", __FUNCTION__);
457  return -1;
458  }
459 
460  key_da = NULL;
461 
462  /*
463  * Some structs are prefixed by a 16-bit length.
464  */
465  if (da_is_length_field(parent)) {
466  FR_DBUFF_ADVANCE_RETURN(dbuff, 2);
467  do_length = true;
468  }
469 
470  for (;;) {
471  fr_dict_attr_t const *child;
472 
473  /*
474  * The child attributes should be in order. If
475  * they're not, we fill the struct with zeroes.
476  *
477  * The caller will encode TLVs.
478  */
479  child = fr_dict_attr_child_by_num(parent, child_num);
480  if (!child) break;
481 
482  /*
483  * Encode child TLVs at the end of a struct.
484  *
485  * @todo - just have a loop here looking for
486  * vp->da->parent == child, and encoding those.
487  */
488  if (child->type == FR_TYPE_TLV) {
489  break;
490  }
491 
492  /*
493  * Skipped a VP, or left one off at the end, fill the struct with zeros.
494  */
495  if (!vp || (vp->da != child)) {
496  /*
497  * Zero out the bit field.
498  */
499  if (da_is_bit_field(child)) {
500  offset = put_bits_dbuff(&work_dbuff, &bit_buffer, offset, child->flags.length, 0);
501  if (offset < 0) return offset;
502  child_num++;
503  continue;
504  }
505 
506  if (da_is_key_field(child)) {
507  key_da = child;
508  }
509 
510  /*
511  * Zero out the unused field.
512  */
513  FR_DBUFF_MEMSET_RETURN(&work_dbuff, 0, child->flags.length);
514  child_num++;
515  continue;
516  }
517 
518  /*
519  * The 'struct' encoder handles bit fields.
520  * They're just integers, so there's no need to
521  * call the protocol encoder.
522  *
523  * This limitation means that we can't have
524  * encrypted bit fields, but that's fine.
525  */
526  if (da_is_bit_field(child)) {
527  uint64_t value;
528 
529  switch (child->type) {
530  case FR_TYPE_BOOL:
531  value = vp->vp_bool;
532  break;
533 
534  case FR_TYPE_UINT8:
535  value = vp->vp_uint8;
536  break;
537 
538  case FR_TYPE_UINT16:
539  value = vp->vp_uint16;
540  break;
541 
542  case FR_TYPE_UINT32:
543  value = vp->vp_uint32;
544  break;
545 
546  case FR_TYPE_UINT64:
547  value = vp->vp_uint64;
548  break;
549 
550  default:
551  fr_strerror_printf("Invalid bit field");
552  return -1;
553  }
554 
555  offset = put_bits_dbuff(&work_dbuff, &bit_buffer, offset, child->flags.length, value);
556  if (offset < 0) return offset;
557 
558  do {
559  vp = fr_cursor_next(cursor);
560  if (!vp || !vp->da->flags.internal) break;
561  } while (vp != NULL);
562  goto next;
563 
564  }
565 
566  /* Not a bit field; insist that no buffered bits remain. */
567  if (offset != 0) {
568  leftover_bits:
569  fr_strerror_printf("leftover bits");
570  return -1;
571  }
572 
573  if (encode_value) {
574  ssize_t len;
575  /*
576  * Call the protocol encoder for non-bit fields.
577  */
578  fr_proto_da_stack_build(da_stack, child);
579  len = encode_value(&work_dbuff, da_stack, depth + 1, cursor, encoder_ctx);
580  if (len < 0) return len;
581  vp = fr_cursor_current(cursor);
582 
583  } else {
584  /*
585  * Encode fixed-size octets fields so that they
586  * are exactly the fixed size, UNLESS the entire
587  * output is truncated.
588  */
589  if ((vp->da->type == FR_TYPE_OCTETS) && vp->da->flags.length) {
590  size_t mylen = vp->da->flags.length;
591 
592  if (vp->vp_length < mylen) {
593  FR_DBUFF_IN_MEMCPY_RETURN(&work_dbuff, (uint8_t const *)(vp->vp_ptr),
594  vp->vp_length);
595  FR_DBUFF_MEMSET_RETURN(&work_dbuff, 0, mylen - vp->vp_length);
596  } else {
597  FR_DBUFF_IN_MEMCPY_RETURN(&work_dbuff, (uint8_t const *)(vp->vp_ptr), mylen);
598  }
599 
600  } else {
601  /*
602  * Determine the nested type and call the appropriate encoder
603  */
604  if (fr_value_box_to_network_dbuff(NULL, &work_dbuff, &vp->data) <= 0) return -1;
605  }
606 
607  do {
608  vp = fr_cursor_next(cursor);
609  if (!vp || !vp->da->flags.internal) break;
610  } while (vp != NULL);
611  }
612 
613  if (da_is_key_field(child)) {
614  key_da = child;
615  }
616 
617  next:
618  child_num++;
619  }
620 
621  /* Check for leftover bits */
622  if (offset != 0) goto leftover_bits;
623 
624  /*
625  * Check for keyed data to encode.
626  */
627  if (vp && key_da) {
628  /*
629  * If our parent is a struct, AND its parent is
630  * the key_da, then we have a keyed struct for
631  * the child. Go encode it.
632  */
633  if ((vp->da->parent->parent == key_da) &&
634  (vp->da->parent->type == FR_TYPE_STRUCT)) {
635  ssize_t len;
636  fr_proto_da_stack_build(da_stack, vp->da->parent);
637  len = fr_struct_to_network_dbuff(&work_dbuff, da_stack, depth + 2, /* note + 2 !!! */
638  cursor, encoder_ctx, encode_value);
639  if (len < 0) return len;
640  goto done;
641  }
642 
643  /*
644  * The next VP is likely octets and unknown.
645  */
646  if ((vp->da->parent == key_da) &&
647  (vp->da->type != FR_TYPE_TLV)) {
648  if (fr_value_box_to_network_dbuff(NULL, &work_dbuff, &vp->data) <= 0) return -1;
649  (void) fr_cursor_next(cursor);
650  goto done;
651  }
652 
653  /*
654  * We have no idea what to do. Ignore it.
655  */
656  }
657 
658 done:
659  if (do_length) {
660  uint32_t len = fr_dbuff_used(&work_dbuff) - 2;
661  if (len > 65535) return -1;
662  fr_dbuff_in(&hdr_dbuff, (uint16_t)len);
663  }
664 
665  return fr_dbuff_set(dbuff, &work_dbuff);
666 }
