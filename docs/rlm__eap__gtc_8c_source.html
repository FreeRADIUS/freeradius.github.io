73 /** Translate a string auth_type into an enumeration value
74  *
75  * @param[in] ctx to allocate data.
76  * @param[out] out Where to write the auth_type we created or resolved.
77  * @param[in] parent Base structure address.
78  * @param[in] ci #CONF_PAIR specifying the name of the auth_type.
79  * @param[in] rule unused.
80  * @return
81  * - 0 on success.
82  * - -1 on failure.
83  */
84 static int auth_type_parse(UNUSED TALLOC_CTX *ctx, void *out, UNUSED void *parent,
85  CONF_ITEM *ci, UNUSED CONF_PARSER const *rule)
86 {
87  char const *auth_type = cf_pair_value(cf_item_to_pair(ci));
88 
89  if (fr_dict_attr_enum_add_name_next(fr_dict_attr_unconst(attr_auth_type), auth_type) < 0) {
90  cf_log_err(ci, "Failed adding %s alias", attr_auth_type->name);
91  return -1;
92  }
93  *((fr_dict_enum_t **)out) = fr_dict_enum_by_name(attr_auth_type, auth_type, -1);
94 
95  return 0;
96 }

98 /*
99  * Keep processing the Auth-Type until it doesn't return YIELD.
100  */
101 static unlang_action_t mod_process_auth_type(rlm_rcode_t *p_result, UNUSED module_ctx_t const *mctx, request_t *request)
102 {
103  rlm_rcode_t rcode;
104 
105  eap_session_t *eap_session = eap_session_get(request->parent);
106  eap_round_t *eap_round = eap_session->this_round;
107 
108  rcode = unlang_interpret(request);
109 
110  if (request->master_state == REQUEST_STOP_PROCESSING) return UNLANG_ACTION_STOP_PROCESSING;
111 
112  if (rcode == RLM_MODULE_YIELD) return UNLANG_ACTION_YIELD;
113 
114  if (rcode != RLM_MODULE_OK) {
115  eap_round->request->code = FR_EAP_CODE_FAILURE;
116  RETURN_MODULE_RCODE(rcode);
117  }
118 
119  eap_round->request->code = FR_EAP_CODE_SUCCESS;
120  RETURN_MODULE_OK;
121 }

123 /*
124  * Authenticate a previously sent challenge.
125  */
126 static unlang_action_t mod_process(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
127 {
128  rlm_eap_gtc_t const *inst = talloc_get_type_abort(mctx->instance, rlm_eap_gtc_t);
129  rlm_rcode_t rcode;
130 
131  eap_session_t *eap_session = eap_session_get(request->parent);
132  eap_round_t *eap_round = eap_session->this_round;
133 
134  fr_pair_t *vp;
135  CONF_SECTION *unlang;
136 
137  /*
138  * Get the Cleartext-Password for this user.
139  */
140 
141  /*
142  * Sanity check the response. We need at least one byte
143  * of data.
144  */
145  if (eap_round->response->length <= 4) {
146  REDEBUG("Corrupted data");
147  eap_round->request->code = FR_EAP_CODE_FAILURE;
148  RETURN_MODULE_INVALID;
149  }
150 
151  /*
152  * EAP packets can be ~64k long maximum, and
153  * we don't like that.
154  */
155  if (eap_round->response->type.length > 128) {
156  REDEBUG("Response is too large to understand");
157  eap_round->request->code = FR_EAP_CODE_FAILURE;
158  RETURN_MODULE_INVALID;
159  }
160 
161  /*
162  * If there was a User-Password in the request,
163  * why the heck are they using EAP-GTC?
164  */
165  MEM(pair_update_request(&vp, attr_user_password) >= 0);
166  fr_pair_value_bstrndup(vp, (char const *)eap_round->response->type.data, eap_round->response->type.length, true);
167  vp->vp_tainted = true;
168 
169  unlang = cf_section_find(request->server_cs, "authenticate", inst->auth_type->name);
170  if (!unlang) {
171  /*
172  * Call the authenticate section of the *current* virtual server.
173  */
174  process_authenticate(&rcode, inst->auth_type->value->vb_uint32, request);
175  if (rcode != RLM_MODULE_OK) {
176  eap_round->request->code = FR_EAP_CODE_FAILURE;
177  RETURN_MODULE_RCODE(rcode);
178  }
179 
180  eap_round->request->code = FR_EAP_CODE_SUCCESS;
181  RETURN_MODULE_OK;
182  }
183 
184  if (unlang_interpret_push_section(request, unlang, RLM_MODULE_FAIL, UNLANG_TOP_FRAME) < 0) {
185  RETURN_MODULE_FAIL;
186  }
187 
188  eap_session->process = mod_process_auth_type;
189 
190  return eap_session->process(p_result, mctx, request);
191 }

194 /*
195  * Initiate the EAP-GTC session by sending a challenge to the peer.
196  */
197 static unlang_action_t mod_session_init(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
198 {
199  eap_session_t *eap_session = eap_session_get(request->parent);
200  char challenge_str[1024];
201  int length;
202  eap_round_t *eap_round = eap_session->this_round;
203  rlm_eap_gtc_t *inst = talloc_get_type_abort(mctx->instance, rlm_eap_gtc_t);
204 
205  if (xlat_eval(challenge_str, sizeof(challenge_str), request, inst->challenge, NULL, NULL) < 0) {
206  RETURN_MODULE_FAIL;
207  }
208 
209  length = strlen(challenge_str);
210 
211  /*
212  * We're sending a request...
213  */
214  eap_round->request->code = FR_EAP_CODE_REQUEST;
215 
216  eap_round->request->type.data = talloc_array(eap_round->request, uint8_t, length);
217  if (!eap_round->request->type.data) RETURN_MODULE_FAIL;
218 
219  memcpy(eap_round->request->type.data, challenge_str, length);
220  eap_round->request->type.length = length;
221 
222  /*
223  * We don't need to authorize the user at this point.
224  *
225  * We also don't need to keep the challenge, as it's
226  * stored in 'eap_session->this_round', which will be given back
227  * to us...
228  */
229  eap_session->process = mod_process;
230 
231  RETURN_MODULE_HANDLED;
232 }
