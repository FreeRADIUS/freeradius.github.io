46 int sigtran_client_do_transaction(int fd, sigtran_transaction_t *txn)
47 {
48  ssize_t len;
49  void *ptr;
50 
51  if (write(fd, &txn, sizeof(txn)) < 0) {
52  ERROR("worker - ctrl_pipe (%i) write failed: %s", fd, fr_syserror(errno));
53  return -1;
54  }
55 
56  /*
57  * Block until libosmo responds
58  */
59  len = read(fd, &ptr, sizeof(ptr));
60  if (len < 0) {
61  ERROR("worker - ctrl_pipe (%i) read failed : %s", fd, fr_syserror(errno));
62  return -1;
63  }
64 
65  if (len != sizeof(ptr)) {
66  ERROR("worker - ctrl_pipe (%i) data too short, expected %zu bytes, got %zi bytes",
67  fd, sizeof(ptr), len);
68  return -1;
69  }
70 
71  if (ptr != txn) {
72  ERROR("worker - ctrl_pipe (%i) response ptr (%p) does not match request (%p)", fd, ptr, txn);
73  return -1;
74  }
75 
76  /*
77  * Check talloc header is still OK
78  */
79  talloc_get_type_abort(ptr, sigtran_transaction_t);
80 
81  return 0;
82 }

106 /** Drain any data we received
107  *
108  * We don't care about this data, we just don't want the kernel to
109  * signal the other side that our read buffer's full.
110  */
111 static void _sigtran_pipe_read(UNUSED fr_event_list_t *el, int fd, UNUSED int flags, UNUSED void *uctx)
112 {
113  ssize_t len;
114  void *ptr;
115  sigtran_transaction_t *txn;
116 
117  len = read(fd, &ptr, sizeof(ptr));
118  if (len < 0) {
119  ERROR("worker - ctrl_pipe (%i) read failed : %s", fd, fr_syserror(errno));
120  return;
121  }
122 
123  if (len != sizeof(ptr)) {
124  ERROR("worker - ctrl_pipe (%i) data too short, expected %zu bytes, got %zi bytes",
125  fd, sizeof(ptr), len);
126  return;
127  }
128 
129  /*
130  * Check talloc header is still OK
131  */
132  txn = talloc_get_type_abort(ptr, sigtran_transaction_t);
133  if (txn->ctx.defunct) return; /* Request was stopped */
134 
135  fr_assert(txn->ctx.request);
136  unlang_interpret_resumable(txn->ctx.request); /* Continue processing */
137 }

139 /** Called by a new thread to register a new req_pipe
140  *
141  * @return
142  * - The client side of the req_pipe on success.
143  * - -1 on error.
144  */
145 int sigtran_client_thread_register(fr_event_list_t *el)
146 {
147  int req_pipe[2] = { -1, -1 };
148  sigtran_transaction_t *txn;
149 
150  /*
151  * Create the pipe on our side, and pass over
152  * the remote end to be registered.
153  */
154  if (socketpair(AF_UNIX, SOCK_STREAM, 0, req_pipe) < 0) {
155  ERROR("worker - Failed creating req_pipe: %s", fr_syserror(errno));
156  return -1;
157  }
158 
159  fr_assert((req_pipe[0] >= 0) && (req_pipe[1] >= 0));
160 
161  txn = talloc_zero(NULL, sigtran_transaction_t);
162  txn->request.type = SIGTRAN_REQUEST_THREAD_REGISTER;
163  txn->request.data = &req_pipe[1];
164 
165  if ((sigtran_client_do_ctrl_transaction(txn) < 0) || (txn->response.type != SIGTRAN_RESPONSE_OK)) {
166  ERROR("worker - Failed registering thread");
167  error:
168  close(req_pipe[0]);
169  close(req_pipe[1]);
170  talloc_free(txn);
171  return -1;
172  }
173  DEBUG3("worker - Thread register acked by osmocom thread");
174  talloc_free(txn);
175 
176  /*
177  * Read data coming back on the pipe,
178  * and resume requests which are
179  * waiting.
180  */
181  if (fr_event_fd_insert(NULL, el, req_pipe[0], _sigtran_pipe_read, NULL, _sigtran_pipe_error, NULL) < 0) {
182  ERROR("worker - Failed listening on osmocom pipe");
183  goto error;
184  }
185 
186  return req_pipe[0];
187 }

189 /** Signal that libosmo should unregister the other side of the pipe
190  *
191  * @param[in] el the request pipe was registered to.
192  * @param[in] req_pipe_fd The rlm_sigtran side of the req_pipe.
193  */
194 int sigtran_client_thread_unregister(fr_event_list_t *el, int req_pipe_fd)
195 {
196  sigtran_transaction_t *txn;
197 
198  txn = talloc_zero(NULL, sigtran_transaction_t);
199  txn->request.type = SIGTRAN_REQUEST_THREAD_UNREGISTER;
200 
201  /*
202  * The signal to unregister *MUST* be sent on the
203  * request pipe itself, so that the osmocom thread
204  * knows *WHICH* pipe to close on its side.
205  */
206  if ((sigtran_client_do_transaction(req_pipe_fd, txn) < 0) || (txn->response.type != SIGTRAN_RESPONSE_OK)) {
207  ERROR("worker - Failed unregistering thread");
208  talloc_free(txn);
209  return -1;
210  }
211  DEBUG3("worker - Thread unregister acked by osmocom thread");
212  talloc_free(txn);
213 
214  fr_event_fd_delete(el, req_pipe_fd, FR_EVENT_FILTER_IO);
215  close(req_pipe_fd);
216 
217  return 0;
218 }

258  txn->request.type = SIGTRAN_REQUEST_LINK_DOWN;
259  memcpy(&txn->request.data, conn, sizeof(txn->request.data));
260 
261  if ((sigtran_client_do_ctrl_transaction(txn) < 0) || (txn->response.type != SIGTRAN_RESPONSE_OK)) {
262  ERROR("worker - Failed taking down the link");

264  return -1;

267  talloc_free(txn);
268  *conn = NULL;

274  void *rctx, fr_state_signal_t action)
275 {
276  sigtran_transaction_t *txn = talloc_get_type_abort(rctx, sigtran_transaction_t);
277 
278  /*
279  * Ignore DUP signals, along with all others.
280  */

284  txn->ctx.request = NULL; /* remove the link to the (now dead) request */

290  rlm_rcode_t rcode;
291  fr_assert(request == txn->ctx.request);
292 
293  /*
294  * Process response
295  */

297  case SIGTRAN_RESPONSE_OK:
298  {
299  unsigned int i = 0;

301  fr_pair_t *vp;
302  sigtran_vector_t *vec;

306

378  i++;
379  break;
380  }
381  }

389

400  break;
401  }
402  talloc_free(txn);
403 
404  RETURN_MODULE_RCODE(rcode);
405 }
