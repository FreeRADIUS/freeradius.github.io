33 /** Compare two labels in a case-insensitive fashion.
34  *
35  * This function requires that the input is valid, i.e. all
36  * characters are within [-0-9A-Za-z]. If any other input is given,
37  * it will break.
38  */
39 static bool labelcmp(uint8_t const *a, uint8_t const *b, size_t len)
40 {
41  for (/* nothing */; len > 0; len--) {
42  if (*a == *b) {
43  a++;
44  b++;
45  continue;
46  }
47 
48  /*
49  * If one or the other isn't a letter, we can't
50  * do case insensitive comparisons of them, so
51  * they can't match.
52  */
53  if ((*a < 'A') || (*b < 'A')) {
54  return false;
55  }
56 
57  /*
58  * If they're equal but different case, then the
59  * only bit that's different is 0x20.
60  */
61  if (((*a)^(*b)) != 0x20) {
62  return false;
63  }
64 
65  a++;
66  b++;
67  }
68 
69  return true;
70 }

72 /** Compress "label" by looking at it recursively.
73  *
74  * For "ftp.example.com", it searches the input buffer for a matching
75  * "com". It only does string compares if it finds bytes "03 xx xx
76  * xx 00". This means that the scan is quick, because most bytes are
77  * skipped.
78  *
79  * If a matching string is found, the label is updated to replace
80  * "com" with a 2-byte pointer P1. The process then proceeds
81  * recursively, with "exampleP1".
82  *
83  * The input buffer is the scanned again for labels of matching
84  * length (7 here), AND which either end in the value of "P1", or end
85  * *at* P1. If we find a match, we replace "exampleP1" with "P2".
86  * The process then proceeds recursively with "ftpP2".
87  *
88  * Since the algorithm replaces known suffixes with pointers, we
89  * *never* have to compare full names. Instead, we only ever compare
90  * one label to one other label. And then only if the labels have
91  * the same lengths AND the same suffixes (00 byte, or a pointer P).
92  *
93  * As an extra optimization, we track the start of the label where we
94  * found the compressed pointer. e.g. "www.example.com" when
95  * compressing "com". We know that the "com" string CANNOT appear
96  * before this label. Because if it did, then the "www.example.com"
97  * name would have instead been compressed, as in "www.exampleP1".
98  *
99  * This optimization ensures that we scan as little of the buffer as
100  * possible, by moving the search start ahead in the buffer. This
101  * optimization also means that in many cases, the suffix we're
102  * looking for (e.g. "example.com") is in the first label we search.
103  * Which means that we end up ignoring most of the buffer.
104  *
105  * This algorithm is O(N * B), where N is the number of labels in a
106  * name (e.g. 3 for "ftp.example.com"), and "B" is the size of the
107  * buffer. It also does linear scans of the buffer, which are good
108  * for read-ahead. Each input label is compared to labels in the
109  * buffer only when very limited situations apply. And we never
110  * compare the full input name to full names in the buffer.
111  *
112  * In the case where we are adding many names from the same zone to
113  * the input buffer, the input buffer will start with the zone name.
114  * So any searches will match that. The only reason to continue
115  * scanning the buffer is to see if the name prefix already exists.
116  * If we assume that the records do not contain duplicates, then we
117  * can likely skip that scan, too.
118  *
119  * Adding that optimization, however, requires tracking the maximum
120  * size of a name across multiple invocations of the function. For
121  * example, if the maximum length name in the buffer is 3 labels, and
122  * we're adding a 3 label name, then we can stop scanning the buffer
123  * as soon as we compressed the 2 suffix labels. Since we are
124  * guaranteed that there are no duplicates, we are sure that there is
125  * no existing 3-label name which matches a 3-label name in the
126  * buffer.
127  *
128  *
129  * A different and more straightforward approach is to loop over all
130  * labels in the name from longest to shortest, and comparing them to
131  * each name in the buffer in turn. That algorithm ends up being
132  * O(L1 * T * L2), where L1 is the length of the input name, T is the
133  * total number of names in the buffer, and L2 is the average length
134  * of names in the buffer. This algorithm can result in the buffer
135  * being scanned many, many, times. The scan is also done forwards
136  * (due to comparing names one after the other), but also backwards
137  * (due to following pointers). Which makes for poor locality of
138  * reference.
139  *
140  * i.e. that approach *has* to scan the entire input buffer, because
141  * that's where all of the names are. Further, it has to scan it at
142  * least "N" times, because there are N labels in the input name. So
143  * O(N * B) is the *lower* bound for this algorithm.
144  *
145  * It gets worse when the straightforward algorithm does pointer
146  * following instead of pointer comparisons. It ends up scanning
147  * portions of the input buffer many, many, times. i.e. it can
148  * compare an input "com" name to "org" once for every "org" name in
149  * the input buffer. In contrast, because our algorithm does not do
150  * pointer following, it only compares "com" to "org" once.
151  *
152  * @param[in] start input buffer holding one or more labels
153  * @param[in] end end of the input buffer
154  * @param[out] new_search Where the parent call to dns_label_compress()
155  * should start searching from, instead of from "start".
156  * @param[in] label label to add to the buffer.
157  * @param[out] label_end updated end of the input label after compression.
158  * @return
159  * - false, we didn't compress the input
160  * - true, we did compress the input.
161  */
162 static bool dns_label_compress(uint8_t const *start, uint8_t const *end, uint8_t const **new_search,
163  uint8_t *label, uint8_t **label_end)
164 {
165  uint8_t *next;
166  uint8_t const *q, *ptr, *suffix, *search;
167  uint16_t offset;
168 
169  /*
170  * Don't compress "end of label" byte or pointers.
171  */
172  if (!*label || (*label > 63)) {
173  return false;
174  }
175 
176  /*
177  * Check the next label. Note that this is *after*
178  * "end". It also MUST be a valid, uncompressed label.
179  */
180  next = label + *label + 1;
181 
182  /*
183  * Note that by design, next > end. We don't care about
184  * the size of the buffer we put "label" into. We only
185  * care that all bytes of "label" are valid, and we don't
186  * access memroy after "label".
187  */
188 
189  /*
190  * On the first call, begin searching from the start of
191  * the buffer.
192  *
193  * For subsequent calls, begin from where we started
194  * searching before.
195  */
196  if (!new_search) {
197  search = start;
198  } else {
199  search = *new_search;
200  }
201 
202  /*
203  * We're at the last uncompressed label, scan the input
204  * buffer to see if there's a match.
205  *
206  * The scan skips ahead until it find both a label length
207  * that matches, AND "next" label which is 0x00. Only
208  * then does it do the string compare of the label
209  * values.
210  *
211  * We speed this up slightly by tracking the previous
212  * uncompressed pointer. If we do compress the current
213  * label, then we should also tell the caller where the
214  * previous uncompressed label started. That way the
215  * caller can start looking there for the next piece to
216  * compress. There's no need to search from the
217  * beginning of the input buffer, as we're sure that
218  * there is no earlier instance of the suffix we found.
219  *
220  * i.e. as we compress the current name, we start
221  * searching not at the beginning of the input buffer/
222  * Instead, we start searching at the name which contains
223  * the label that we just compressed. The previous
224  * sarching guarantees that no name *before* that one
225  * will match the suffix we're looking for. So we can
226  * skip all of the previous names in subsequent searches,
227  */
228  if (*next == 0x00) {
229  q = search;
230  while (q < end) {
231  if (*q == 0x00) {
232  q++;
233 
234  /*
235  * None of the previous names
236  * matched, so we tell the caller
237  * to start searching from the
238  * next name in the buffer.
239  */
240  search = q;
241  continue;
242  }
243 
244  /*
245  * Our label is a terminal one. Which
246  * can't point to a pointer.
247  */
248  if (*q > 63) {
249  q += 2;
250 
251  /*
252  * None of the previous
253  * uncompressed names matched,
254  * and this pointer refers to a
255  * compressed name. So it
256  * doesn't match, either.
257  */
258  search = q;
259  continue;
260  }
261 
262  /*
263  * We now have a label which MIGHT match.
264  * We have to walk down it until it does
265  * match. But we don't update "search"
266  * here, because there may be a suffix
267  * which matches.
268  */
269  ptr = q + *q + 1;
270  if (ptr > end) return false;
271 
272  /*
273  * Label lengths aren't the same, skip
274  * it.
275  */
276  if (*q != *label) {
277  q = ptr;
278  continue;
279  }
280 
281  /*
282  * Our input label ends with 0x00. If
283  * this label doesn't end with 0x00, skip
284  * it.
285  */
286  if (*ptr != 0x00) {
287  q = ptr;
288  continue;
289  }
290 
291  /*
292  * The pointer is too far away. Don't
293  * point to it. This check is mainly for
294  * static analyzers.
295  */
296  if ((q - start) > (1 << 14)) return false;
297 
298  /*
299  * Only now do case-insensitive
300  * comparisons.
301  */
302  if (!labelcmp(q + 1, label + 1, *label)) {
303  q = ptr;
304  continue;
305  }
306 
307  /*
308  * We have a match. Replace the input
309  * label with a compressed pointer. Tell
310  * the caller the start of the found
311  * name, so subsequent searches can start
312  * from there. Then return to the caller
313  * that we managed to compress this
314  * label.
315  */
316  offset = (q - start);
317  label[0] = (offset >> 8) | 0xc0;
318  label[1] = offset & 0xff;
319  *label_end = label + 2;
320  if (new_search) *new_search = search;
321  return true;
322  }
323 
324  return false;
325  }
326 
327  /*
328  * The next label is still uncompressed, so we call
329  * ourselves recursively in order to compress it.
330  */
331  if (*next < 63) {
332  if (!dns_label_compress(start, end, &search, next, label_end)) return false;
333 
334  /*
335  * Else it WAS compressed.
336  */
337  }
338 
339  /*
340  * The next label wasn't compressed, OR it is invalid,
341  * skip it. This check is here only to shut up the
342  * static analysis tools.
343  */
344  if (*next < 0xc0) {
345  return false;
346  }
347 
348  /*
349  * Remember where our suffix points to.
350  */
351  suffix = start + ((next[0] & ~0xc0) << 8) + next[1];
352 
353  /*
354  * Our label now ends with a compressed pointer. Scan
355  * the input until we find either an uncompressed label
356  * which ends with the same compressed pointer, OR we
357  * find an uncompressed label which ends AT our
358  * compressed pointer.
359  *
360  * Note that we start searching from the beginning of the
361  * label which resulted in us finding the compressed
362  * pointer!
363  *
364  * We're guaranteed that any label BEFORE that one
365  * doesn't end with a matching compressed pointer.
366  */
367  q = search;
368  while (q < end) {
369  if (*q == 0x00) {
370  q++;
371 
372  /*
373  * None of the previous stuff matched, so
374  * we tell the caller to start searching
375  * from the next name.
376  */
377  search = q;
378  continue;
379  }
380 
381  /*
382  * Skip compressed pointers. We can't point to
383  * compressed pointers.
384  */
385  if (*q > 63) {
386  q += 2;
387 
388  /*
389  * None of the previous uncompressed
390  * names matched, and this pointer refers
391  * to a compressed name. So it doesn't
392  * match, either.
393  */
394  search = q;
395  continue;
396  }
397 
398  /*
399  * We now have an uncompressed label in the input
400  * buffer. Check for a match.
401  */
402  ptr = q + *q + 1;
403  if (ptr > end) return false;
404 
405  /*
406  * Label lengths aren't the same, skip it.
407  */
408  if (*q != *label) {
409  q = ptr;
410  continue;
411  }
412 
413  /*
414  * If the NEXT label is uncompressed, then skip
415  * it unless it's the suffix we're pointing to.
416  */
417  if (*ptr < 63) {
418  if (ptr != suffix) {
419  q = ptr;
420  continue;
421  }
422 
423  goto check_label;
424  }
425 
426  /*
427  * The next label is a compressed pointer. If
428  * the compressed pointers are different, then
429  * skip both this label and the compressed
430  * pointer after it.
431  */
432  if ((ptr[0] != next[0]) ||
433  (ptr[1] != next[1])) {
434  q = ptr + 2;
435 
436  /*
437  * None of the previous uncompressed
438  * names matched, and this pointer refers
439  * to a compressed name. So it doesn't
440  * match, either.
441  */
442  search = q;
443  continue;
444  }
445 
446  check_label:
447  /*
448  * Pointer is too far away. Don't point
449  * to it.
450  */
451  if ((q - start) > (1 << 14)) return false;
452 
453  /*
454  * Only now do case-insensitive
455  * comparisons.
456  */
457  if (!labelcmp(q + 1, label + 1, *label)) {
458  q = ptr;
459  continue;
460  }
461 
462  /*
463  * We have a match. Replace the input
464  * label with a compressed pointer. Tell
465  * the caller the start of the found
466  * name, so subsequent searches can start
467  * from there. Then return to the caller
468  * that we managed to compress this
469  * label.
470  */
471  offset = (q - start);
472  label[0] = (offset >> 8) | 0xc0;
473  label[1] = offset & 0xff;
474  *label_end = label + 2;
475  if (new_search) *new_search = search;
476  return true;
477  }
478 
479  /*
480  * Who knows what it is, we couldn't compress it.
481  */
482  return false;
483 }

486 /** Encode a single value box of type string, serializing its contents to a dns label
487  * in a dbuff
488  *
489  * @param[in] dbuff Buffer where labels are written
490  * @param[in] compression Whether or not to do DNS label compression.
491  * @param[in] value to encode.
492  * @return
493  * - >0 the number of bytes written to the dbuff
494  * - 0 could not encode anything, an error has occurred.
495  * - <0 the number of bytes the dbuff should have had, instead of "remaining".
496  */
497 ssize_t fr_dns_label_from_value_box_dbuff(fr_dbuff_t *dbuff, bool compression, fr_value_box_t const *value)
498 {
499  ssize_t slen;
500  size_t need = 0;
501 
502  slen = fr_dns_label_from_value_box(&need, dbuff->p, fr_dbuff_remaining(dbuff), dbuff->p, compression, value);
503  if (slen < 0) return 0;
504 
505  if (slen == 0) return -need;
506 
507  fr_dbuff_advance(dbuff, slen);
508  return slen;
509 }

511 /** Encode a single value box of type string, serializing its contents to a dns label
512  *
513  * This functions takes a large buffer and encodes the label in part
514  * of the buffer. This API is necessary in order to allow DNS label
515  * compression.
516  *
517  * @param[out] need if not NULL, how long "buf_len" should be to
518  * serialize the rest of the data.
519  * Note: Only variable length types will be partially
520  * encoded. Fixed length types will not be partially encoded.
521  * @param[out] buf Buffer where labels are stored
522  * @param[in] buf_len The length of the output buffer
523  * @param[out] where Where to write this label
524  * @param[in] compression Whether or not to do DNS label compression.
525  * @param[in] value to encode.
526  * @return
527  * - 0 no bytes were written, see need value to determine
528  * - >0 the number of bytes written to "where", NOT "buf + where + outlen"
529  * - <0 on error.
530  */
531 ssize_t fr_dns_label_from_value_box(size_t *need, uint8_t *buf, size_t buf_len, uint8_t *where, bool compression,
532  fr_value_box_t const *value)
533 {
534  uint8_t *label;
535  uint8_t const *end = buf + buf_len;
536  uint8_t const *q, *strend, *last;
537  uint8_t *data;
538 
539  if (!buf || !buf_len || !where || !value) {
540  fr_strerror_printf("Invalid input");
541  return -1;
542  }
543 
544  /*
545  * Don't allow stupidities
546  */
547  if (!((where >= buf) && (where < (buf + buf_len)))) {
548  fr_strerror_printf("Label to write is outside of buffer");
549  return -1;
550  }
551 
552  /*
553  * We can only encode strings.
554  */
555  if (value->type != FR_TYPE_STRING) {
556  fr_strerror_printf("Asked to encode non-string type");
557  return -1;
558  }
559 
560  /*
561  * '.' or empty string is special, and is encoded as a
562  * plain zero byte.
563  *
564  * Since "where < end", we can always write 1 byte to it.
565  */
566  if ((value->vb_length == 0) ||
567  ((value->vb_length == 1) && (value->vb_strvalue[0] == '.'))) {
568  *where = 0x00;
569  return 1;
570  }
571 
572  /*
573  * Sanity check the name before writing anything to the
574  * buffer.
575  *
576  * Only encode [-0-9a-zA-Z]. Anything else is forbidden.
577  * Dots at the start are forbidden. Double dots are
578  * forbidden.
579  */
580  q = (uint8_t const *) value->vb_strvalue;
581  strend = q + value->vb_length;
582  last = q;
583 
584  if (*q == '.') {
585  fr_strerror_printf("Empty labels are invalid");
586  return -1;
587  }
588 
589  while (q < strend) {
590  if (*q == '.') {
591  /*
592  * Don't count final dot as an
593  * intermediate dot, and don't bother
594  * encoding it.
595  */
596  if ((q + 1) == strend) {
597  strend--;
598  break;
599  }
600 
601  if (q[1] == '.') {
602  fr_strerror_printf("Double dots '..' are forbidden");
603  return -1;
604  }
605  last = q;
606 
607  } else if (!((*q == '-') || ((*q >= '0') && (*q <= '9')) ||
608  ((*q >= 'A') && (*q <= 'Z')) || ((*q >= 'a') && (*q <= 'z')))) {
609  fr_strerror_printf("Invalid character %02x in label", *q);
610  return -1;
611  }
612 
613  q++;
614 
615  if ((q - last) > 63) {
616  fr_strerror_printf("Label is larger than 63 characters");
617  return -1;
618  }
619  }
620 
621  q = (uint8_t const *) value->vb_strvalue;
622 
623  /*
624  * For now, just encode the value as-is. We do
625  * compression as a second step.
626  *
627  * We need a minimum length of string + beginning length
628  * + trailing zero. Intermediate '.' are converted to
629  * length bytes.
630  */
631  if ((where + (strend - q) + 2) > end) {
632  if (need) *need = (where + (strend - q) + 2) - buf;
633  return 0;
634  }
635 
636  label = where;
637  *label = 0;
638  data = label + 1;
639 
640  /*
641  * @todo - encode into a local buffer, and then try to
642  * compress that into the output buffer. This means that
643  * the output buffer can be a little bit smaller.
644  */
645  while (q < strend) {
646  fr_assert(data < end);
647  fr_assert((data - where) < 255);
648 
649  /*
650  * '.' is a label delimiter.
651  *
652  * We've already checked above for '.' at the
653  * start, for double dots, and have already
654  * suppressed '.' at the end of the string.
655  *
656  * Start a new label.
657  */
658  if (*q == '.') {
659  label = data;
660  *label = 0;
661  data = label + 1;
662 
663  q++;
664  continue;
665  }
666 
667  *(data++) = *(q++);
668  (*label)++;
669  fr_assert(*label <= 63);
670  }
671 
672  *(data++) = 0; /* end of label */
673 
674  /*
675  * Only one label, don't compress it. Or, the label is
676  * already compressed.
677  */
678  if (!compression || (buf == where) || ((data - where) <= 2)) goto done;
679 
680  /*
681  * Compress it, AND tell us where the new end buffer is located.
682  */
683  (void) dns_label_compress(buf, where, NULL, where, &data);
684 
685 done:
686  fr_assert(data > where);
687  return data - where;
688 }

690 /** Get the *uncompressed* length of a DNS label in a network buffer.
691  *
692  * i.e. how bytes are required to store the uncompressed version of
693  * the label.
694  *
695  * Note that a bare 0x00 byte has length 1, to account for '.'
696  *
697  * @param[in] buf buffer holding one or more DNS labels
698  * @param[in] buf_len total length of the buffer
699  * @param[in,out] next the DNS label to check, updated to point to the next label
700  * @return
701  * - <=0 on error, offset from buf where the invalid label is located.
702  * - > 0 decoded size of this particular DNS label
703  */
704 ssize_t fr_dns_label_uncompressed_length(uint8_t const *buf, size_t buf_len, uint8_t const **next)
705 {
706  uint8_t const *p, *q, *end;
707  uint8_t const *current, *start;
708  size_t length;
709  bool at_first_label, set_next;
710 
711  if (!buf || (buf_len == 0) || !next) return 0;
712 
713  start = *next;
714 
715  /*
716  * Don't allow stupidities
717  */
718  if (!((start >= buf) && (start < (buf + buf_len)))) return 0;
719 
720  end = buf + buf_len;
721  p = current = start;
722  length = 0;
723  at_first_label = true;
724  set_next = false;
725 
726  /*
727  * We silently accept labels *without* a trailing 0x00,
728  * so long as they end at the end of the input buffer.
729  */
730  while (p < end) {
731  /*
732  * End of label byte. Skip it.
733  *
734  * Empty labels are length 1, to account for the
735  * '.'. The caller has to take care of this
736  * manually.
737  */
738  if (*p == 0x00) {
739  p++;
740  if (at_first_label) length++;
741 
742  /*
743  * We're still processing the first
744  * label, tell the caller where the next
745  * one is located.
746  */
747  if (current == start) {
748  *next = p;
749  set_next = true;
750  }
751 
752  break;
753  }
754 
755  /*
756  * 0b10 and 0b10 are forbidden
757  */
758  if ((*p > 63) && (*p < 0xc0)) {
759  fr_strerror_printf("Data with invalid high bits");
760  return -(p - buf);
761  }
762 
763  /*
764  * Maybe it's a compressed pointer.
765  */
766  if (*p > 63) {
767  uint16_t offset;
768 
769  if ((p + 2) > end) {
770  overflow:
771  fr_strerror_printf("Label overflows buffer");
772  return -(p - buf);
773  }
774 
775  offset = p[1];
776  offset += ((*p & ~0xc0) << 8);
777 
778  /*
779  * Forward references are forbidden,
780  * including self-references.
781  */
782  if (offset >= (p - buf)) {
783  fr_strerror_printf("Pointer %04x is an invalid forward reference", offset);
784  return -(p - buf);
785  }
786 
787  q = buf + offset;
788 
789  /*
790  * As an additional sanity check, the
791  * pointer MUST NOT point to something
792  * within the label we're parsing. If
793  * that happens, we have a loop.
794  *
795  * i.e. the pointer must be backwards to
796  * *before* our current label. When that
797  * limitation is enforced, pointer loops
798  * are impossible.
799  */
800  if (q >= current) {
801  fr_strerror_printf("Pointer %04x creates a loop within a label", offset);
802  return -(p - buf);
803  }
804 
805 
806  /*
807  * The pointer MUST point to a valid
808  * label length, and not to another
809  * pointer.
810  */
811  if (*q > 63) {
812  fr_strerror_printf("Pointer %04x does not point to the start of a label", offset);
813  return -(p - buf);
814  }
815 
816  /*
817  * If we're jumping away from the label
818  * we started with, tell the caller where
819  * the next label is in the network
820  * buffer.
821  */
822  if (current == start) {
823  *next = p + 2;
824  set_next = true;
825  }
826 
827  p = current = q;
828  continue;
829  }
830 
831  /*
832  * Else it's an uncompressed label
833  */
834  if ((p + *p + 1) > end) goto overflow;
835 
836  /*
837  * Account for the '.' on every label after the
838  * first one.
839  */
840  if (!at_first_label) length++;
841  at_first_label = false;
842  length += *p;
843 
844  /*
845  * DNS names can be no more than 255 octets.
846  */
847  if (length > 255) {
848  fr_strerror_printf("Total length of labels is > 255");
849  return -(p - buf);
850  }
851 
852  /*
853  * Verify that the contents of the label are OK.
854  */
855  for (q = p + 1; q < p + *p + 1; q++) {
856  if (!((*q == '-') || ((*q >= '0') && (*q <= '9')) ||
857  ((*q >= 'A') && (*q <= 'Z')) || ((*q >= 'a') && (*q <= 'z')))) {
858  fr_strerror_printf("Invalid character %02x in label", *q);
859  return -(q - buf);
860  }
861  }
862 
863  p += *p + 1;
864  }
865 
866  /*
867  * Return the length of this label.
868  */
869  if (!set_next) *next = p; /* should be 'end'< */
870 
871  return length;
872 }

899 
900  p = *start;
901 
902  if (*p == 0x00) {
903  *next = p + 1;

929 /** Decode a #fr_value_box_t from one DNS label
930  *
931  * The output type is always FR_TYPE_STRING
932  *
933  * Note that the caller MUST call fr_dns_labels_network_verify(src, len)
934  * before calling this function. Otherwise bad things will happen.
935  *
936  * @param[in] ctx Where to allocate any talloc buffers required.
937  * @param[out] dst value_box to write the result to.
938  * @param[in] src Start of the buffer containing DNS labels
939  * @param[in] len Length of the buffer to decode
940  * @param[in] label This particular label
941  * @param[in] tainted Whether the value came from a trusted source.
942  * @return
943  * - >= 0 The number of network bytes consumed.
944  * - <0 on error.
945  */
946 ssize_t fr_dns_label_to_value_box(TALLOC_CTX *ctx, fr_value_box_t *dst,
947  uint8_t const *src, size_t len, uint8_t const *label,
948  bool tainted)
949 {
950  ssize_t slen;
951  uint8_t const *after = label;
952  uint8_t const *current, *next;
953  uint8_t *p;
954  char *q;
955 
956  /*
957  * Get the uncompressed length of the label, and the
958  * label after this one.
959  */
960  slen = fr_dns_label_uncompressed_length(src, len, &after);
961  if (slen <= 0) return slen;
962 
963  fr_value_box_init_null(dst);
964 
965  /*
966  * An empty label is a 0x00 byte. Just create an empty
967  * string.
968  */
969  if (slen == 1) {
970  if (fr_value_box_bstr_alloc(ctx, &q, dst, NULL, 1, tainted) < 0) return -1;
971  q[0] = '.';
972  return after - label;
973  }
974 
975  /*
976  * Allocate the string and set up the value_box
977  */
978  if (fr_value_box_bstr_alloc(ctx, &q, dst, NULL, slen, tainted) < 0) return -1;
979 
980  current = label;
981  p = (uint8_t *) q;
982  q += slen;
983 
984  while ((current < after) && (*current != 0x00)) {
985  /*
986  * Get how many bytes this label has, and where
987  * we will go to obtain the next label.
988  *
989  * Note that slen > 0 here, as dns_label_decode()
990  * only returns 0 when the current byte is 0x00,
991  * which it can't be.
992  */
993  slen = dns_label_decode(src, &current, &next);
994 
995  /*
996  * As a sanity check, ensure we don't have a
997  * buffer overflow.
998  */
999  if ((p + slen) > (uint8_t *) q) {
1000  fail:
1001  fr_value_box_clear(dst);
1002  return -1;
1003  }
1004 
1005  /*
1006  * Add '.' before the label, but only for the
1007  * second and subsequent labels.
1008  */
1009  if (p != (uint8_t const *) dst->vb_strvalue) {
1010  *(p++) = '.';
1011  }
1012 
1013  /*
1014  * Copy the raw bytes from the network.
1015  */
1016  memcpy(p, current + 1, slen);
1017 
1018  /*
1019  * Go ahead in the output string, and go to the
1020  * next label for decoding.
1021  */
1022  p += slen;
1023  current = next;
1024  }
1025 
1026  /*
1027  * As a last sanity check, ensure that we've filled the
1028  * buffer exactly.
1029  */
1030  if (p != (uint8_t *) q) {
1031  goto fail;
1032  }
1033 
1034  *p = '\0';
1035 
1036  /*
1037  * Return the number of network bytes used to parse this
1038  * part of the label.
1039  */
1040  return after - label;
1041 }

1043 /** Get the *network* length of a DNS label in a buffer
1044  * i.e. the number of bytes in the encoded representation of the dns label.
1045  *
1046  * @param[in] buf buffer holding one or more DNS labels
1047  * @param[in] buf_len total length of the buffer
1048  * @return
1049  * - <=0 on error, offset from buf where the invalid label is located.
1050  * - > 0 network length of this particular DNS label
1051  */
1052 ssize_t fr_dns_label_network_length(uint8_t const *buf, size_t buf_len)
1053 {

1055  uint8_t const *end = buf + buf_len;
1056

1058  if (*p == 0x00) {
1059  p++;
1060  break;
1061  }
1062 
1063  if (*p < 64) {
1064  if ((p + *p + 1) > end) {
1065  return -(p - buf);
1066  }
1067 
1068  p += *p + 1;
1069  continue;

1071

1073  return -(p - buf);

1075

1077  return -(p - buf);
1078  }
1079 
1080  p += 2;
1081  break;
1082  }
1083 
1084  return p - buf;
