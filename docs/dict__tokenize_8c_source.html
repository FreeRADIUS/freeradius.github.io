103 int fr_dict_str_to_argv(char *str, char **argv, int max_argc)
104 {
105  int argc = 0;
106 
107  while (*str) {
108  if (argc >= max_argc) break;
109 
110  /*
111  * Chop out comments early.
112  */
113  if (*str == '#') {
114  *str = '\0';
115  break;
116  }
117 
118  while ((*str == ' ') ||
119  (*str == '\t') ||
120  (*str == '\r') ||
121  (*str == '\n'))
122  *(str++) = '\0';
123 
124  if (!*str) break;
125 
126  argv[argc] = str;
127  argc++;
128 
129  while (*str &&
130  (*str != ' ') &&
131  (*str != '\t') &&
132  (*str != '\r') &&
133  (*str != '\n'))
134  str++;
135  }
136 
137  return argc;
138 }

140 static int dict_read_sscanf_i(unsigned int *pvalue, char const *str)
141 {
142  int ret = 0;
143  int base = 10;
144  static char const *tab = "0123456789";
145 
146  if ((str[0] == '0') &&
147  ((str[1] == 'x') || (str[1] == 'X'))) {
148  tab = "0123456789abcdef";
149  base = 16;
150 
151  str += 2;
152  }
153 
154  while (*str) {
155  char const *c;
156 
157  if (*str == '.') break;
158 
159  c = memchr(tab, tolower((int)*str), base);
160  if (!c) return 0;
161 
162  ret *= base;
163  ret += (c - tab);
164  str++;
165  }
166 
167  *pvalue = ret;
168  return 1;
169 }

219 static int dict_process_type_field(dict_tokenize_ctx_t *ctx, char const *name, fr_type_t *type_p,
220  fr_dict_attr_flags_t *flags)
221 {
222  char *p;
223  fr_type_t type;
224 
225  /*
226  * Some types can have fixed length
227  */
228  p = strchr(name, '[');
229  if (p) {
230  char *q;
231  unsigned int length;
232 
233  *p = '\0';
234  q = strchr(p + 1, ']');
235  if (!q) {
236  fr_strerror_printf("Invalid format for '%s[...]'", name);
237  return -1;
238  }
239 
240  *q = '\0';
241 
242  if (!dict_read_sscanf_i(&length, p + 1)) {
243  fr_strerror_printf("Invalid length for '%s[...]'", name);
244  return -1;
245  }
246 
247  if ((length == 0) || (length > 253)) {
248  fr_strerror_printf("Invalid length for '%s[...]'", name);
249  return -1;
250  }
251 
252  /*
253  * Now that we have a length, check the data type.
254  */
255  if (strcmp(name, "octets") == 0) {
256  type = FR_TYPE_OCTETS;
257 
258  } else if (strcmp(name, "struct") == 0) {
259  type = FR_TYPE_STRUCT;
260 
261  } else if (strcmp(name, "bit") == 0) {
262  if (ctx->stack[ctx->stack_depth].da->type != FR_TYPE_STRUCT) {
263  fr_strerror_printf("Bit fields can only be used inside of a STRUCT");
264  return -1;
265  }
266 
267  flags->extra = 1;
268  flags->subtype = FLAG_BIT_FIELD;
269 
270  if (length == 1) {
271  type = FR_TYPE_BOOL;
272  } else if (length < 8) {
273  type = FR_TYPE_UINT8;
274  } else if (length < 16) {
275  type = FR_TYPE_UINT16;
276  } else if (length < 32) {
277  type = FR_TYPE_UINT32;
278  } else if (length < 56) { /* for laziness in encode / decode */
279  type = FR_TYPE_UINT64;
280  } else {
281  fr_strerror_printf("Invalid length for bit field");
282  return -1;
283  }
284 
285  /*
286  * We track where on a byte boundary this bit field ends.
287  */
288  flags->type_size = length;
289 
290  } else {
291  fr_strerror_printf("Only 'octets' types can have a 'length' parameter");
292  return -1;
293  }
294 
295  flags->length = length;
296  *type_p = type;
297  return 0;
298  }
299 
300  /*
301  * find the type of the attribute.
302  */
303  type = fr_table_value_by_str(fr_value_box_type_table, name, FR_TYPE_INVALID);
304  if (type == FR_TYPE_INVALID) {
305  fr_strerror_printf("Unknown data type '%s'", name);
306  return -1;
307  }
308 
309  *type_p = type;
310  return 0;
311 }

314 static int dict_process_flag_field(dict_tokenize_ctx_t *ctx, char *name, fr_type_t type, fr_dict_attr_flags_t *flags,
315  char **ref)
316 {
317  char *p, *next = NULL;
318 
319  if (ref) *ref = NULL;
320 
321  for (p = name; p && *p != '\0' ; p = next) {
322  char *key, *value;
323 
324  key = p;
325 
326  /*
327  * Allow for "key1,key2". But is has to be the
328  * last string in the flags field.
329  */
330  if (ctx->dict->subtype_table) {
331  int subtype;
332 
333  subtype = fr_table_value_by_str(ctx->dict->subtype_table, key, -1);
334  if (subtype >= 0) {
335  flags->subtype = subtype;
336  break;
337  }
338  }
339 
340 
341  /*
342  * Search for the first '=' or ','
343  */
344  for (next = p + 1; *next && (*next != '=') && (*next != ','); next++) {
345  /* do nothing */
346  }
347 
348  /*
349  * We have a value, zero out the '=' and point to the value.
350  */
351  if (*next == '=') {
352  *(next++) = '\0';
353  value = next;
354 
355  if (!*value || (*value == ',')) {
356  fr_strerror_printf("Missing value after '%s='", key);
357  return -1;
358  }
359  } else {
360  value = NULL;
361  }
362 
363  /*
364  * Skip any trailing text in the value.
365  */
366  for (/* nothing */; *next; next++) {
367  if (*next == ',') {
368  *(next++) = '\0';
369  break;
370  }
371  }
372 
373  /*
374  * Marks the attribute up as internal.
375  * This means it can use numbers outside of the allowed
376  * protocol range, and also means it will not be included
377  * in replies or proxy requests.
378  */
379  if (strcmp(key, "internal") == 0) {
380  flags->internal = 1;
381 
382  } else if (strcmp(key, "array") == 0) {
383  flags->array = 1;
384 
385  } else if (strcmp(key, "virtual") == 0) {
386  flags->virtual = 1;
387 
388  } else if (strcmp(key, "key") == 0) {
389  if ((type != FR_TYPE_UINT8) && (type != FR_TYPE_UINT16) && (type != FR_TYPE_UINT32)) {
390  fr_strerror_printf("The 'key' flag can only be used for attributes of type 'uint8', 'uint16', or 'uint32'");
391  return -1;
392  }
393 
394  if (flags->extra) {
395  fr_strerror_printf("Bit fields cannot be key fields");
396  return -1;
397  }
398 
399  flags->extra = 1;
400  flags->subtype = FLAG_KEY_FIELD;
401 
402  } else if (strcmp(key, "length") == 0) {
403  if (!value || (strcmp(value, "uint16") != 0)) {
404  fr_strerror_printf("The 'length' flag can only be used with value 'uint16'");
405  }
406 
407  flags->extra = 1;
408  flags->subtype = FLAG_LENGTH_UINT16;
409 
410  } else if ((type == FR_TYPE_DATE) || (type == FR_TYPE_TIME_DELTA)) {
411  flags->length = 4;
412  flags->type_size = FR_TIME_RES_SEC;
413 
414  if (strncmp(key, "uint", 4) == 0) {
415  fr_type_t subtype;
416 
417  subtype = fr_table_value_by_str(fr_value_box_type_table, name, FR_TYPE_INVALID);
418  if (subtype == FR_TYPE_INVALID) {
419  unknown_type:
420  fr_strerror_printf("Unknown or unsupported %s type '%s'",
421  fr_table_str_by_value(fr_value_box_type_table, type, "<UNKNOWN>"),
422  key);
423  return -1;
424  }
425 
426  switch (subtype) {
427  default:
428  goto unknown_type;
429 
430  case FR_TYPE_INT16:
431  if (type == FR_TYPE_DATE) goto unknown_type;
432  FALL_THROUGH;
433 
434  case FR_TYPE_UINT16:
435  flags->length = 2;
436  break;
437 
438  case FR_TYPE_INT32:
439  if (type == FR_TYPE_DATE) goto unknown_type;
440  FALL_THROUGH;
441 
442  case FR_TYPE_UINT32:
443  flags->length = 4;
444  break;
445 
446  case FR_TYPE_INT64:
447  if (type == FR_TYPE_DATE) goto unknown_type;
448  FALL_THROUGH;
449 
450  case FR_TYPE_UINT64:
451  flags->length = 8;
452  break;
453  }
454  } else {
455  int precision;
456 
457  precision = fr_table_value_by_str(date_precision_table, key, -1);
458  if (precision < 0) {
459  fr_strerror_printf("Unknown %s precision '%s'",
460  fr_table_str_by_value(fr_value_box_type_table, type, "<UNKNOWN>"),
461  key);
462  return -1;
463  }
464  flags->type_size = precision;
465  }
466 
467  } else if (strcmp(key, "ref") == 0) {
468  if (!value) {
469  fr_strerror_printf("Missing attribute name for 'ref=...'");
470  return -1;
471  }
472 
473  if ((type != FR_TYPE_GROUP) && (type != FR_TYPE_TLV)) {
474  fr_strerror_printf("The 'ref' flag can only be used for attributes of type 'group'");
475  return -1;
476  }
477 
478  *ref = talloc_strdup(ctx->dict->pool, value);
479  flags->extra = true;
480  flags->subtype = FLAG_HAS_REF;
481 
482  } else if (ctx->dict->subtype_table) {
483  int subtype;
484 
485  if (value) value[-1] = '='; /* hackity hack */
486 
487  /*
488  * Protocol should use strings
489  * "key1,key2" to allow for multiple
490  * flags.
491  */
492  if (flags->extra || flags->subtype) {
493  fr_strerror_printf("Cannot add flag '%s' - another flag is already set",
494  key);
495  return -1;
496  }
497 
498  subtype = fr_table_value_by_str(ctx->dict->subtype_table, key, -1);
499  if (subtype < 0) goto unknown_option;
500 
501  flags->subtype = subtype;
502 
503  } else {
504  unknown_option:
505  fr_strerror_printf("Unknown option '%s'", key);
506  return -1;
507  }
508  }
509 
510  /*
511  * Check that the flags are valid.
512  */
513  if (!dict_attr_flags_valid(ctx->dict, ctx->stack[ctx->stack_depth].da, name, NULL, type, flags)) return -1;
514 
515  return 0;
516 }

576  */

581

583

602 /*
603  * Process the ATTRIBUTE command
604  */
605 static int dict_read_process_attribute(dict_tokenize_ctx_t *ctx, char **argv, int argc,
606  fr_dict_attr_flags_t *base_flags)
607 {
608  bool set_relative_attr = true;
609 
610  ssize_t slen;
611  unsigned int attr;
612 
613  fr_type_t type;
614  fr_dict_attr_flags_t flags;
615  fr_dict_attr_t const *parent, *da;
616  char *ref = NULL;
617 
618  if ((argc < 3) || (argc > 4)) {
619  fr_strerror_printf("Invalid ATTRIBUTE syntax");
620  return -1;
621  }
622 
623  /*
624  * Dictionaries need to have real names, not shitty ones.
625  */
626  if (strncmp(argv[0], "Attr-", 5) == 0) {
627  fr_strerror_printf("Invalid ATTRIBUTE name");
628  return -1;
629  }
630 
631  memcpy(&flags, base_flags, sizeof(flags));
632 
633  if (dict_process_type_field(ctx, argv[2], &type, &flags) < 0) return -1;
634 
635  if (flags.extra && (flags.subtype == FLAG_BIT_FIELD)) {
636  fr_strerror_printf("Bit fields can only be defined as a MEMBER of a STRUCT");
637  return -1;
638  }
639 
640  /*
641  * Relative OIDs apply ONLY to attributes of type 'tlv'.
642  */
643  if (type != FR_TYPE_TLV) set_relative_attr = false;
644 
645  /*
646  * A non-relative ATTRIBUTE definition means that it is
647  * in the context of the previous BEGIN-FOO. So we
648  * unwind the stack to match.
649  */
650  if (argv[1][0] != '.') {
651  parent = dict_gctx_unwind(ctx);
652 
653  /*
654  * Allow '0xff00' as attribute numbers, but only
655  * if there is no OID component.
656  */
657  if (strchr(argv[1], '.') == 0) {
658  if (!dict_read_sscanf_i(&attr, argv[1])) {
659  fr_strerror_printf("Invalid ATTRIBUTE number");
660  return -1;
661  }
662 
663  } else {
664  slen = fr_dict_attr_by_oid(ctx->dict, &parent, &attr, argv[1]);
665  if (slen <= 0) return -1;
666  }
667 
668  } else {
669  if (!ctx->relative_attr) {
670  fr_strerror_printf("Unknown parent for partial OID");
671  return -1;
672  }
673 
674  parent = ctx->relative_attr;
675  set_relative_attr = false;
676 
677  slen = fr_dict_attr_by_oid(ctx->dict, &parent, &attr, argv[1]);
678  if (slen <= 0) return -1;
679  }
680 
681  if (!fr_cond_assert(parent)) return -1; /* Should have provided us with a parent */
682 
683  /*
684  * Members of a 'struct' MUST use MEMBER, not ATTRIBUTE.
685  */
686  if (parent->type == FR_TYPE_STRUCT) {
687  fr_strerror_printf("Member %s of ATTRIBUTE %s type 'struct' MUST use the \"MEMBER\" keyword",
688  argv[0], parent->name);
689  return -1;
690  }
691 
692  /*
693  * Parse options.
694  */
695  if (argc >= 4) {
696  if (dict_process_flag_field(ctx, argv[3], type, &flags, &ref) < 0) return -1;
697  } else {
698  if (!dict_attr_flags_valid(ctx->dict, parent, argv[3], NULL, type, &flags)) return -1;
699  }
700 
701 #ifdef WITH_DICTIONARY_WARNINGS
702  /*
703  * Hack to help us discover which vendors have illegal
704  * attributes.
705  */
706  if (!vendor && (attr < 256) &&
707  !strstr(fn, "rfc") && !strstr(fn, "illegal")) {
708  fprintf(stderr, "WARNING: Illegal Attribute %s in %s\n",
709  argv[0], fn);
710  }
711 #endif
712 
713 #ifdef __clang_analyzer__
714  if (!ctx->dict) return -1;
715 #endif
716 
717  /*
718  * Dynamically define where VSAs go. Note that we CANNOT
719  * define VSAs until we define an attribute of type VSA!
720  */
721  if ((type == FR_TYPE_VSA) && (parent->flags.is_root)) {
722  ctx->dict->vsa_parent = attr;
723  }
724 
725  /*
726  * TLVs can only refer to attributes in the same dictionary.
727  */

743  */
744  da = dict_attr_child_by_num(parent, attr);

764  */
765  if (!ref) {

770  }

821  */
822  fixup->next = ctx->group_fixup;
823  ctx->group_fixup = fixup;
824

832  */

854  self->dict = dict;

862  * BEGIN-STRUCT.
863  */
864  if (type == FR_TYPE_STRUCT) {
865  if (dict_gctx_push(ctx, da) < 0) return -1;
866  }
867

873  * Process the MEMBER command
874  */
875 static int dict_read_process_member(dict_tokenize_ctx_t *ctx, char **argv, int argc,
876  fr_dict_attr_flags_t *base_flags)
877 {
878  fr_type_t type;
879  fr_dict_attr_flags_t flags;
880 
881  if ((argc < 2) || (argc > 3)) {
882  fr_strerror_printf("Invalid MEMBER syntax");
883  return -1;
884  }
885 
886  if (ctx->stack[ctx->stack_depth].da->type != FR_TYPE_STRUCT) {
887  fr_strerror_printf("MEMBER can only be used for ATTRIBUTEs of type 'struct'");
888  return -1;
889  }
890 
891  /*
892  * Dictionaries need to have real names, not shitty ones.
893  */
894  if (strncmp(argv[0], "Attr-", 5) == 0) {
895  fr_strerror_printf("Invalid MEMBER name");
896  return -1;
897  }
898 
899  memcpy(&flags, base_flags, sizeof(flags));
900 
901  if (dict_process_type_field(ctx, argv[1], &type, &flags) < 0) return -1;
902 
903  /*
904  * Parse options.
905  */
906  if (argc >= 3) {
907  if (dict_process_flag_field(ctx, argv[2], type, &flags, NULL) < 0) return -1;
908  } else {
909  if (!dict_attr_flags_valid(ctx->dict, ctx->stack[ctx->stack_depth].da, argv[2], NULL, type, &flags)) return -1;
910  }
911 
912 #ifdef __clang_analyzer__
913  if (!ctx->dict) return -1;
914 #endif
915 
916  /*
917  * Double check bit field magic
918  */
919  if (ctx->stack[ctx->stack_depth].member_num > 0) {
920  fr_dict_attr_t const *previous;
921 
922  previous = dict_attr_child_by_num(ctx->stack[ctx->stack_depth].da,
923  ctx->stack[ctx->stack_depth].member_num);
924  if (!previous) {
925  fr_strerror_printf("Failed to find previous MEMBER");
926  return -1;
927  }
928 
929  /*
930  * Check that the previous bit field ended on a
931  * byte boundary.
932  */
933  if (previous->flags.extra && (previous->flags.subtype == FLAG_BIT_FIELD)) {
934  /*
935  * This attribute is a bit field. Keep
936  * track of where in the byte we are
937  * located.
938  */
939  if (flags.extra && (flags.subtype == FLAG_BIT_FIELD)) {
940  flags.type_size += previous->flags.type_size;
941  flags.type_size &= 0x07;
942 
943  } else {
944  if (previous->flags.type_size != 0) {
945  fr_strerror_printf("Previous bitfield %s did not end on a byte boundary",
946  previous->name);
947  return -1;
948  }
949  }
950  }
951  }
952 
953  /*
954  * Check if the parent 'struct' is fixed size. And if
955  * so, complain if we're adding a variable sized member.
956  */
957  if (ctx->stack[ctx->stack_depth].da->flags.length &&
958  ((type == FR_TYPE_STRING) || (type == FR_TYPE_TLV) ||
959  ((type == FR_TYPE_OCTETS) && !flags.length))) {
960  fr_strerror_printf("'struct' %s has fixed size %u, we cannot add a variable-sized member.",
961  ctx->stack[ctx->stack_depth].da->name, ctx->stack[ctx->stack_depth].da->flags.length);
962  return -1;
963  }
964 
965  /*
966  * Add the MEMBER to the parent.
967  */
968  if (fr_dict_attr_add(ctx->dict,
969  ctx->stack[ctx->stack_depth].da,
970  argv[0],
971  ++ctx->stack[ctx->stack_depth].member_num,
972  type, &flags) < 0) return -1;
973 
974  /*
975  * A 'struct' can have a MEMBER of type 'tlv', but ONLY
976  * as the last entry in the 'struct'. If we see that,
977  * set the previous attribute to the TLV we just added.
978  * This allows the children of the TLV to be parsed as
979  * partial OIDs, so we don't need to know the full path
980  * to them.
981  */
982  if (type == FR_TYPE_TLV) {
983  ctx->relative_attr = dict_attr_child_by_num(ctx->stack[ctx->stack_depth].da,
984  ctx->stack[ctx->stack_depth].member_num);
985  if (dict_gctx_push(ctx, ctx->relative_attr) < 0) return -1;
986 
987  } else {
988 
989  /*
990  * Add the size of this member to the parent struct.
991  */
992  ctx->stack[ctx->stack_depth].struct_size += flags.length;
993 
994  /*
995  * Check for overflow.
996  */
997  if (ctx->stack[ctx->stack_depth].da->flags.length &&
998  (ctx->stack[ctx->stack_depth].struct_size > ctx->stack[ctx->stack_depth].da->flags.length)) {
999  fr_strerror_printf("'struct' %s has fixed size %u, but member %s overflows that length",
1000  ctx->stack[ctx->stack_depth].da->name, ctx->stack[ctx->stack_depth].da->flags.length,
1001  argv[0]);
1002  return -1;
1003  }
1004  }
1005 
1006  return 0;
1007 }

1139  */

1141 {

1143  fr_dict_attr_t *parent;

1146  unsigned int attr;
1147  fr_dict_attr_flags_t flags;

1150

1152  fr_strerror_printf("Invalid STRUCT syntax");
1153  return -1;
1154  }
1155 
1156  /*
1157  * This SHOULD be the "key" field.
1158  */
1159  parent = dict_attr_by_name(ctx->dict, key_attr);

1161  fr_strerror_printf("Unknown attribute '%s'", key_attr);
1162  return -1;

1207  */
1208  if (!ctx->stack[ctx->stack_depth].da->flags.is_root &&

1218  */

1410  if (!dict->root) return -1;
1411 #endif
1412

1414  fr_strerror_printf("Conflicting numbers %u vs %u for PROTOCOL \"%s\"",
1415  dict->root->attr, value, dict->root->name);
1416  return -1;
1417  }
1418 
1419  } else if ((dict = dict_by_protocol_num(value)) != NULL) {
1420 #ifdef __clang_analyzer__
1421  if (!dict->root || !dict->root->name || !argv[0]) return -1;
1422 #endif
1423 
1424  if (strcasecmp(dict->root->name, argv[0]) != 0) {
1425  fr_strerror_printf("Conflicting names current \"%s\" vs new \"%s\" for PROTOCOL %u",
1426  dict->root->name, argv[0], dict->root->attr);
1427  return -1;
1428  }
1429  }
1430 
1431  /*
1432  * And check types no matter what.
1433  */
1434  if (dict) {
1435  if (type_size && (dict->root->flags.type_size != type_size)) {
1436  fr_strerror_printf("Conflicting flags for PROTOCOL \"%s\" (current %d versus new %d)",
1437  dict->root->name, dict->root->flags.type_size, type_size);
1438  return -1;
1439  }
1440  return 0;
1441  }
1442

1444 
1445  /*
1446  * Try to load protocol-specific validation routines.
1447  * Some protocols don't need them, so it's OK if the
1448  * validation routines don't exist.
1449  */
1450  if ((dict_dlopen(dict, argv[0]) < 0) && require_dl) {
1451  error:
1452  talloc_free(dict);
1453  return -1;
1454  }
1455 
1456  /*
1457  * Set the root attribute with the protocol name
1458  */
1459  if (dict_root_set(dict, argv[0], value) < 0) goto error;
1460 
1461  if (dict_protocol_add(dict) < 0) goto error;
1462 
1463  memcpy(&mutable, &dict->root, sizeof(mutable));
1464 
1465  if (!type_size) {
1466  mutable->flags.type_size = dict->default_type_size;
1467  mutable->flags.length = dict->default_type_length;
1468  } else {
1469  mutable->flags.type_size = type_size;
1470  mutable->flags.length = 1; /* who knows... */
1471  }
1472 
1473  return 0;
1474 }

1528 static int fr_dict_finalise(dict_tokenize_ctx_t *ctx)
1529 {
1530  /*
1531  * Resolve any VALUE aliases (enums) that were defined
1532  * before the attributes they reference.
1533  */
1534  if (ctx->enum_fixup) {
1535  fr_dict_attr_t *da;
1536  dict_enum_fixup_t *this, *next;
1537 
1538  for (this = ctx->enum_fixup; this != NULL; this = next) {
1539  fr_value_box_t value;
1540  fr_type_t type;
1541  int ret;
1542 
1543  next = this->next;
1544  da = dict_attr_by_name(ctx->dict, this->attribute);
1545  if (!da) {
1546  fr_strerror_printf("No ATTRIBUTE '%s' defined for VALUE '%s' at %s[%d]",
1547  this->attribute, this->name, this->filename, this->line);
1548  return -1;
1549  }
1550  type = da->type;
1551 
1552  if (fr_value_box_from_str(this, &value, &type, NULL,
1553  this->value, talloc_array_length(this->value) - 1, '\0', false) < 0) {
1554  fr_strerror_printf_push("Invalid VALUE for ATTRIBUTE \"%s\" at %s[%d]",
1555  da->name, this->filename, this->line);
1556  return -1;
1557  }
1558 
1559  ret = fr_dict_attr_enum_add_name(da, this->name, &value, false, false);
1560  fr_value_box_clear(&value);
1561 
1562  if (ret < 0) return -1;
1563 
1564  dict_attr_fixup_mark(ctx, da);
1565 
1566  /*
1567  * Just so we don't lose track of things.
1568  */
1569  ctx->enum_fixup = next;
1570  }
1571  }
1572 
1573  if (ctx->group_fixup) {
1574  dict_group_fixup_t *mine, *this, *next;
1575 
1576  mine = ctx->group_fixup;
1577  ctx->group_fixup = NULL;
1578 
1579  /*
1580  * Loop over references, adding the dictionaries
1581  * and attributes to the da.
1582  *
1583  * We avoid refcount loops by using the "autoref"
1584  * table. If a "group" attribute refers to a
1585  * dictionary which does not exist, we load it,
1586  * increment its reference count, and add it to
1587  * the autoref table.
1588  *
1589  * If a group attribute refers to a dictionary
1590  * which does exist, we check that dictionaries
1591  * "autoref" table. If OUR dictionary is there,
1592  * then we do nothing else. That dictionary
1593  * points to us via refcounts, so we can safely
1594  * point to it. The refcounts ensure that we
1595  * won't be free'd before the other one is
1596  * free'd.
1597  *
1598  * If our dictionary is NOT in the other
1599  * dictionaries autoref table, then it was loaded
1600  * via some other method. We increment its
1601  * refcount, and add it to our autoref table.
1602  *
1603  * Then when this dictionary is being free'd, we
1604  * also free the dictionaries in our autoref
1605  * table.
1606  */
1607  for (this = mine; this != NULL; this = next) {
1608  fr_dict_t *dict;
1609  fr_dict_attr_t const *da;
1610  char *p;
1611  ssize_t slen;
1612 
1613  da = dict_attr_by_name(ctx->dict, this->ref);
1614  if (da) {
1615  dict = ctx->dict;
1616  goto check;
1617  }
1618 
1619  /*
1620  * The attribute doesn't exist, and the reference
1621  * isn't in a "PROTO.ATTR" format, die.
1622  */
1623  p = strchr(this->ref, '.');
1624 
1625  /*
1626  * Get / skip protocol name.
1627  */
1628  slen = dict_by_protocol_substr(NULL,
1629  &dict, &FR_SBUFF_IN(this->ref, strlen(this->ref)),
1630  ctx->dict);
1631  if (slen <= 0) {
1632  fr_dict_t *other;
1633 
1634  if (p) *p = '\0';
1635 
1636  if (fr_dict_protocol_afrom_file(&other, this->ref, NULL) < 0) {
1637  return -1;
1638  }
1639 
1640  if (p) *p = '.';
1641 
1642  /*
1643  * Grab the protocol name again
1644  */
1645  dict = other;
1646  if (!p) {
1647  dict = other;
1648  da = other->root;
1649  goto check;
1650  }
1651 
1652  slen = p - this->ref;
1653  }
1654 
1655  if (slen < 0) {
1656  invalid_reference:
1657  fr_strerror_printf("Invalid reference '%s' at %s[%d]",
1658  this->ref, this->filename, this->line);
1659  group_error:
1660  /*
1661  * Just so we don't lose track of things.
1662  */
1663  // @todo - don't leak group_fixup stuff? things?
1664  return -1;
1665  }
1666 
1667  /*
1668  * No known dictionary, so we're asked to just
1669  * use the whole string. Which we did above. So
1670  * either it's a bad ref, OR it's a ref to a
1671  * dictionary which doesn't exist.
1672  */
1673  if (slen == 0) goto invalid_reference;
1674 
1675  /*
1676  * Look up the attribute.
1677  */
1678  da = dict_attr_by_name(dict, this->ref + slen + 1);
1679  if (!da) {
1680  fr_strerror_printf("No such attribute '%s' in reference at %s[%d]",
1681  this->ref + slen + 1, this->filename, this->line);
1682  goto group_error;
1683  }
1684 
1685  check:
1686  if (da->type != FR_TYPE_TLV) {
1687  fr_strerror_printf("References MUST be to attributes of type 'tlv' at %s[%d]",
1688  this->filename, this->line);
1689  goto group_error;
1690  }
1691 
1692  if (da_has_ref(da)) {
1693  fr_strerror_printf("References MUST NOT refer to an ATTRIBUTE which also has 'ref=...' at %s[%d]",
1694  this->filename, this->line);
1695  goto group_error;
1696  }
1697 
1698  talloc_free(this->ref);
1699  dict_attr_ref_set(this->da, da);
1700 
1701  next = this->next;
1702  }
1703  }
1704 
1705  if (ctx->ext_fixup) {
1706  fr_dict_attr_t *this;
1707 
1708  for (this = ctx->ext_fixup; this; this = this->fixup) {
1709  fr_dict_attr_ext_enumv_t *ext;
1710 
1711  ext = fr_dict_attr_ext(this, FR_DICT_ATTR_EXT_ENUMV);
1712  if (!ext) continue;
1713 
1714  fr_hash_table_fill(ext->value_by_name);
1715  fr_hash_table_fill(ext->name_by_value);
1716 
1717  ctx->ext_fixup = this->fixup;
1718  this->fixup = NULL;
1719  }
1720  }
1721 
1722  TALLOC_FREE(ctx->fixup_pool);
1723 
1724  /*
1725  * Walk over all of the hash tables to ensure they're
1726  * initialized. We do this because the threads may perform
1727  * lookups, and we don't want multi-threaded re-ordering
1728  * of the table entries. That would be bad.
1729  */
1730  fr_hash_table_fill(ctx->dict->vendors_by_name);
1731  fr_hash_table_fill(ctx->dict->vendors_by_num);
1732 
1733  ctx->value_attr = NULL;
1734  ctx->relative_attr = NULL;
1735 
1736  return 0;
1737 }

1739 /** Parse a dictionary file
1740  *
1741  * @param[in] ctx Contains the current state of the dictionary parser.
1742  * Used to track what PROTOCOL, VENDOR or TLV block
1743  * we're in. Block context changes in $INCLUDEs should
1744  * not affect the context of the including file.
1745  * @param[in] dir_name Directory containing the dictionary we're loading.
1746  * @param[in] filename we're parsing.
1747  * @param[in] src_file The including file.
1748  * @param[in] src_line Line on which the $INCLUDE or $INCLUDE- statement was found.
1749  * @return
1750  * - 0 on success.
1751  * - -1 on failure.
1752  */
1753 static int _dict_from_file(dict_tokenize_ctx_t *ctx,
1754  char const *dir_name, char const *filename,
1755  char const *src_file, int src_line)
1756 {
1757  FILE *fp;
1758  char dir[256], fn[256];
1759  char buf[256];
1760  char *p;
1761  int line = 0;
1762  bool was_member = false;
1763 
1764  struct stat statbuf;
1765  char *argv[MAX_ARGV];
1766  int argc;
1767  fr_dict_attr_t const *da;
1768 
1769  /*
1770  * Base flags are only set for the current file
1771  */
1772  fr_dict_attr_flags_t base_flags;
1773 
1774  if (!fr_cond_assert(!ctx->dict->root || ctx->stack[ctx->stack_depth].da)) return -1;
1775 
1776  if ((strlen(dir_name) + 3 + strlen(filename)) > sizeof(dir)) {
1777  fr_strerror_printf_push("%s: Filename name too long", "Error reading dictionary");
1778  return -1;
1779  }
1780 
1781  /*
1782  * If it's an absolute dir, forget the parent dir,
1783  * and remember the new one.
1784  *
1785  * If it's a relative dir, tack on the current filename
1786  * to the parent dir. And use that.
1787  */
1788  if (!FR_DIR_IS_RELATIVE(filename)) {
1789  strlcpy(dir, filename, sizeof(dir));
1790  p = strrchr(dir, FR_DIR_SEP);
1791  if (p) {
1792  p[1] = '\0';
1793  } else {
1794  strlcat(dir, "/", sizeof(dir));
1795  }
1796 
1797  strlcpy(fn, filename, sizeof(fn));
1798  } else {
1799  strlcpy(dir, dir_name, sizeof(dir));
1800  p = strrchr(dir, FR_DIR_SEP);
1801  if (p) {
1802  if (p[1]) strlcat(dir, "/", sizeof(dir));
1803  } else {
1804  strlcat(dir, "/", sizeof(dir));
1805  }
1806  strlcat(dir, filename, sizeof(dir));
1807  p = strrchr(dir, FR_DIR_SEP);
1808  if (p) {
1809  p[1] = '\0';
1810  } else {
1811  strlcat(dir, "/", sizeof(dir));
1812  }
1813 
1814  p = strrchr(filename, FR_DIR_SEP);
1815  if (p) {
1816  snprintf(fn, sizeof(fn), "%s%s", dir, p);
1817  } else {
1818  snprintf(fn, sizeof(fn), "%s%s", dir, filename);
1819  }
1820  }
1821 
1822  ctx->stack[ctx->stack_depth].filename = fn;
1823 
1824  if ((fp = fopen(fn, "r")) == NULL) {
1825  if (!src_file) {
1826  fr_strerror_printf_push("Couldn't open dictionary %s: %s", fr_syserror(errno), fn);
1827  } else {
1828  fr_strerror_printf_push("Error reading dictionary: %s[%d]: Couldn't open dictionary '%s': %s",
1829  src_file, src_line, fn,
1830  fr_syserror(errno));
1831  }
1832  return -2;
1833  }
1834 
1835  /*
1836  * If fopen works, this works.
1837  */
1838  if (stat(fn, &statbuf) < 0) {
1839  fclose(fp);
1840  return -1;
1841  }
1842 
1843  if (!S_ISREG(statbuf.st_mode)) {
1844  fclose(fp);
1845  fr_strerror_printf_push("Dictionary is not a regular file: %s", fn);
1846  return -1;
1847  }
1848 
1849  /*
1850  * Globally writable dictionaries means that users can control
1851  * the server configuration with little difficulty.
1852  */
1853 #ifdef S_IWOTH
1854  if ((statbuf.st_mode & S_IWOTH) != 0) {
1855  fclose(fp);
1856  fr_strerror_printf_push("Dictionary is globally writable: %s. "
1857  "Refusing to start due to insecure configuration", fn);
1858  return -1;
1859  }
1860 #endif
1861 
1862  /*
1863  * Seed the random pool with data.
1864  */
1865  fr_rand_seed(&statbuf, sizeof(statbuf));
1866 
1867  memset(&base_flags, 0, sizeof(base_flags));
1868 
1869  while (fgets(buf, sizeof(buf), fp) != NULL) {
1870  ctx->stack[ctx->stack_depth].line = line++;
1871 
1872  switch (buf[0]) {
1873  case '#':
1874  case '\0':
1875  case '\n':
1876  case '\r':
1877  continue;
1878  }
1879 
1880  /*
1881  * Comment characters should NOT be appearing anywhere but
1882  * as start of a comment;
1883  */
1884  p = strchr(buf, '#');
1885  if (p) *p = '\0';
1886 
1887  argc = fr_dict_str_to_argv(buf, argv, MAX_ARGV);
1888  if (argc == 0) continue;
1889 
1890  if (argc == 1) {
1891  fr_strerror_printf("Invalid entry");
1892 
1893  error:
1894  fr_strerror_printf_push("Error reading %s[%d]", fn, line);
1895  fclose(fp);
1896  return -1;
1897  }
1898 
1899  /*
1900  * Process VALUE lines.
1901  */
1902  if (strcasecmp(argv[0], "VALUE") == 0) {
1903  if (dict_read_process_value(ctx, argv + 1, argc - 1) == -1) goto error;
1904  continue;
1905  }
1906 
1907  /*
1908  * Perhaps this is a MEMBER of a struct
1909  *
1910  * @todo - create child ctx, so that we can have
1911  * nested structs.
1912  */
1913  if (strcasecmp(argv[0], "MEMBER") == 0) {
1914  if (dict_read_process_member(ctx,
1915  argv + 1, argc - 1,
1916  &base_flags) == -1) goto error;
1917  was_member = true;
1918  continue;
1919  }
1920 
1921  /*
1922  * Finalise a STRUCT.
1923  */
1924  if (was_member) {
1925  da = ctx->stack[ctx->stack_depth].da;
1926 
1927  if (da->type == FR_TYPE_STRUCT) {
1928 
1929  /*
1930  * The structure was fixed-size,
1931  * but the fields don't fill it.
1932  * That's an error.
1933  *
1934  * Since process_member() checks
1935  * for overflow, the check here
1936  * is really only for underflow.
1937  */
1938  if (da->flags.length &&
1939  (ctx->stack[ctx->stack_depth].struct_size != da->flags.length)) {
1940  fr_strerror_printf("MEMBERs of 'struct' %s do not exactly fill the fixed-size structure",
1941  da->name);
1942  goto error;
1943  }
1944 
1945  /*
1946  * If the structure is fixed
1947  * size, AND small enough to fit
1948  * into an 8-bit length field,
1949  * then update the length field
1950  * with the structure size/
1951  */
1952  if (ctx->stack[ctx->stack_depth].struct_size <= 255) {
1953  fr_dict_attr_t *mutable;
1954 
1955  memcpy(&mutable, &da, sizeof(mutable));
1956  mutable->flags.length = ctx->stack[ctx->stack_depth].struct_size;
1957  } /* else length 0 means "unknown / variable size / too large */
1958  } else {
1959  fr_assert(da->type == FR_TYPE_TLV);
1960  }
1961 
1962  was_member = false;
1963  }
1964 
1965  /*
1966  * Perhaps this is an attribute.
1967  */
1968  if (strcasecmp(argv[0], "ALIAS") == 0) {
1969  if (dict_read_process_alias(ctx,
1970  argv + 1, argc - 1) == -1) goto error;
1971  continue;
1972  }
1973 
1974  /*
1975  * Perhaps this is an attribute.
1976  */
1977  if (strcasecmp(argv[0], "ATTRIBUTE") == 0) {
1978  if (dict_read_process_attribute(ctx,
1979  argv + 1, argc - 1,
1980  &base_flags) == -1) goto error;
1981  continue;
1982  }
1983 
1984  /*
1985  * Process FLAGS lines.
1986  */
1987  if (strcasecmp(argv[0], "FLAGS") == 0) {
1988  if (dict_read_process_flags(ctx->dict, argv + 1, argc - 1, &base_flags) == -1) goto error;
1989  continue;
1990  }
1991 
1992  /*
1993  * Process STRUCT lines.
1994  */
1995  if (strcasecmp(argv[0], "STRUCT") == 0) {
1996  if (dict_read_process_struct(ctx, argv + 1, argc - 1) == -1) goto error;
1997  continue;
1998  }
1999 
2000  /*
2001  * See if we need to import another dictionary.
2002  */
2003  if (strncasecmp(argv[0], "$INCLUDE", 8) == 0) {
2004  int ret;
2005  int stack_depth = ctx->stack_depth;
2006 
2007  /*
2008  * Allow "$INCLUDE" or "$INCLUDE-", but
2009  * not anything else.
2010  */
2011  if ((argv[0][8] != '\0') && ((argv[0][8] != '-') || (argv[0][9] != '\0'))) goto invalid_keyword;
2012 
2013  /*
2014  * Included files operate on a copy of the context.
2015  *
2016  * This copy means that they inherit the
2017  * current context, including parents,
2018  * TLVs, etc. But if the included file
2019  * leaves a "dangling" TLV or "last
2020  * attribute", then it won't affect the
2021  * parent.
2022  */
2023 
2024  ret = _dict_from_file(ctx, dir, argv[1], fn, line);
2025  if ((ret == -2) && (argv[0][8] == '-')) {
2026  fr_strerror_printf(NULL); /* delete all errors */
2027  ret = 0;
2028  }
2029 
2030  if (ret < 0) {
2031  fr_strerror_printf_push("from $INCLUDE at %s[%d]", fn, line);
2032  fclose(fp);
2033  return -1;
2034  }
2035 
2036  if (ctx->stack_depth < stack_depth) {
2037  fr_strerror_printf_push("unexpected END-??? in $INCLUDE at %s[%d]", fn, line);
2038  fclose(fp);
2039  return -1;
2040  }
2041 
2042  while (ctx->stack_depth > stack_depth) {
2043  if (ctx->stack[ctx->stack_depth].nest == FR_TYPE_INVALID) {
2044  ctx->stack_depth--;
2045  continue;
2046  }
2047 
2048  fr_strerror_printf_push("BEGIN-??? without END-... in file $INCLUDEd from %s[%d]", fn, line);
2049  fclose(fp);
2050  return -1;
2051  }
2052 
2053  /*
2054  * Reset the filename.
2055  */
2056  ctx->stack[ctx->stack_depth].filename = fn;
2057  continue;
2058  } /* $INCLUDE */
2059 
2060  /*
2061  * Reset the previous attribute when we see
2062  * VENDOR or PROTOCOL or BEGIN/END-VENDOR, etc.
2063  */
2064  ctx->value_attr = NULL;
2065  ctx->relative_attr = NULL;
2066 
2067  /*
2068  * Process VENDOR lines.
2069  */
2070  if (strcasecmp(argv[0], "VENDOR") == 0) {
2071  if (dict_read_process_vendor(ctx->dict, argv + 1, argc - 1) == -1) goto error;
2072  continue;
2073  }
2074 
2075  /*
2076  * Process PROTOCOL line. Defines a new protocol.
2077  */
2078  if (strcasecmp(argv[0], "PROTOCOL") == 0) {
2079  if (argc < 2) {
2080  fr_strerror_printf_push("Invalid PROTOCOL entry");
2081  goto error;
2082  }
2083  if (dict_read_process_protocol(argv + 1, argc - 1) == -1) goto error;
2084  continue;
2085  }
2086 
2087  /*
2088  * Switches the current protocol context
2089  */
2090  if (strcasecmp(argv[0], "BEGIN-PROTOCOL") == 0) {
2091  fr_dict_t *found;
2092 
2093  if (argc != 2) {
