27 /** Return an EAP-Type for a particular name
28  *
29  * Converts a name into an IANA EAP type.
30  *
31  * @param name to convert.
32  * @return
33  * - IANA EAP type.
34  * - #FR_EAP_METHOD_INVALID if the name doesn't match any known types.
35  */
36 eap_type_t eap_name2type(char const *name)
37 {
38  fr_dict_enum_t *dv;
39 
40  dv = fr_dict_enum_by_name(attr_eap_type, name, -1);
41  if (!dv) return FR_EAP_METHOD_INVALID;
42 
43  if (dv->value->vb_uint32 >= FR_EAP_METHOD_MAX) return FR_EAP_METHOD_INVALID;
44

46 }
