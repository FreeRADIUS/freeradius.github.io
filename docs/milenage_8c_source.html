32 static inline int aes_128_encrypt_block(EVP_CIPHER_CTX *evp_ctx,
33  uint8_t const key[16], uint8_t const in[16], uint8_t out[16])
34 {
35  size_t len;
36 
37  if (unlikely(EVP_EncryptInit_ex(evp_ctx, EVP_aes_128_ecb(), NULL, key, NULL) != 1)) {
38  tls_strerror_printf("Failed initialising AES-128-ECB context");
39  return -1;
40  }
41 
42  /*
43  * By default OpenSSL will try and pad out a 16 byte
44  * plaintext to 32 bytes so that it's detectable that
45  * there was padding.
46  *
47  * In this case we know the length of the plaintext
48  * we're trying to recover, so we explicitly tell
49  * OpenSSL not to pad here, and not to expected padding
50  * when decrypting.
51  */
52  EVP_CIPHER_CTX_set_padding(evp_ctx, 0);
53  if (unlikely(EVP_EncryptUpdate(evp_ctx, out, (int *)&len, in, 16) != 1) ||
54  unlikely(EVP_EncryptFinal_ex(evp_ctx, out + len, (int *)&len) != 1)) {
55  tls_strerror_printf("Failed encrypting data");
56  return -1;
57  }
58 
59  return 0;
60 }

62 /** milenage_f1 - Milenage f1 and f1* algorithms
63  *
64  * @param[in] opc 128-bit value derived from OP and K.
65  * @param[in] k 128-bit subscriber key.
66  * @param[in] rand 128-bit random challenge.
67  * @param[in] sqn 48-bit sequence number.
68  * @param[in] amf 16-bit authentication management field.
69  * @param[out] mac_a Buffer for MAC-A = 64-bit network authentication code, or NULL
70  * @param[out] mac_s Buffer for MAC-S = 64-bit resync authentication code, or NULL
71  * @return
72  * - 0 on success.
73  * - -1 on failure.
74  */
75 static int milenage_f1(uint8_t mac_a[MILENAGE_MAC_A_SIZE],
76  uint8_t mac_s[MILENAGE_MAC_S_SIZE],
77  uint8_t const opc[MILENAGE_OPC_SIZE],
78  uint8_t const k[MILENAGE_KI_SIZE],
79  uint8_t const rand[MILENAGE_RAND_SIZE],
80  uint8_t const sqn[MILENAGE_SQN_SIZE],
81  uint8_t const amf[MILENAGE_AMF_SIZE])
82 {
83  uint8_t tmp1[16], tmp2[16], tmp3[16];
84  int i;
85  EVP_CIPHER_CTX *evp_ctx;
86 
87  /* tmp1 = TEMP = E_K(RAND XOR OP_C) */
88  for (i = 0; i < 16; i++) tmp1[i] = rand[i] ^ opc[i];
89 
90  evp_ctx = EVP_CIPHER_CTX_new();
91  if (!evp_ctx) {
92  tls_strerror_printf("Failed allocating EVP context");
93  return -1;
94  }
95 
96  if (aes_128_encrypt_block(evp_ctx, k, tmp1, tmp1) < 0) {
97  error:
98  EVP_CIPHER_CTX_free(evp_ctx);
99  return -1;
100  }
101 
102  /* tmp2 = IN1 = SQN || AMF || SQN || AMF */
103  memcpy(tmp2, sqn, 6);
104  memcpy(tmp2 + 6, amf, 2);
105  memcpy(tmp2 + 8, tmp2, 8);
106 
107  /* OUT1 = E_K(TEMP XOR rot(IN1 XOR OP_C, r1) XOR c1) XOR OP_C */
108 
109  /*
110  * rotate (tmp2 XOR OP_C) by r1 (= 0x40 = 8 bytes)
111  */
112  for (i = 0; i < 16; i++) tmp3[(i + 8) % 16] = tmp2[i] ^ opc[i];
113 
114  /*
115  * XOR with TEMP = E_K(RAND XOR OP_C)
116  */
117  for (i = 0; i < 16; i++) tmp3[i] ^= tmp1[i];
118  /* XOR with c1 (= ..00, i.e., NOP) */
119 
120  /*
121  * f1 || f1* = E_K(tmp3) XOR OP_c
122  */
123  if (aes_128_encrypt_block(evp_ctx, k, tmp3, tmp1) < 0) goto error; /* Reuses existing key */
124 
125  for (i = 0; i < 16; i++) tmp1[i] ^= opc[i];
126 
127  if (mac_a) memcpy(mac_a, tmp1, 8); /* f1 */
128  if (mac_s) memcpy(mac_s, tmp1 + 8, 8); /* f1* */
129 
130  EVP_CIPHER_CTX_free(evp_ctx);
131 
132  return 0;
133 }

135 /** milenage_f2345 - Milenage f2, f3, f4, f5, f5* algorithms
136  *
137  * @param[out] res Buffer for RES = 64-bit signed response (f2), or NULL
138  * @param[out] ck Buffer for CK = 128-bit confidentiality key (f3), or NULL
139  * @param[out] ik Buffer for IK = 128-bit integrity key (f4), or NULL
140  * @param[out] ak Buffer for AK = 48-bit anonymity key (f5), or NULL
141  * @param[out] ak_resync Buffer for AK = 48-bit anonymity key (f5*), or NULL
142  * @param[in] opc 128-bit value derived from OP and K.
143  * @param[in] k 128-bit subscriber key
144  * @param[in] rand 128-bit random challenge
145  * @return
146  * - 0 on success.
147  * - -1 on failure.
148  */
149 static int milenage_f2345(uint8_t res[MILENAGE_RES_SIZE],
150  uint8_t ik[MILENAGE_IK_SIZE],
151  uint8_t ck[MILENAGE_CK_SIZE],
152  uint8_t ak[MILENAGE_AK_SIZE],
153  uint8_t ak_resync[MILENAGE_AK_SIZE],
154  uint8_t const opc[MILENAGE_OPC_SIZE],
155  uint8_t const k[MILENAGE_KI_SIZE],
156  uint8_t const rand[MILENAGE_RAND_SIZE])
157 {
158  uint8_t tmp1[16], tmp2[16], tmp3[16];
159  int i;
160  EVP_CIPHER_CTX *evp_ctx;
161 
162  /* tmp2 = TEMP = E_K(RAND XOR OP_C) */
163  for (i = 0; i < 16; i++) tmp1[i] = rand[i] ^ opc[i];
164 
165  evp_ctx = EVP_CIPHER_CTX_new();
166  if (!evp_ctx) {
167  tls_strerror_printf("Failed allocating EVP context");
168  return -1;
169  }
170 
171  if (aes_128_encrypt_block(evp_ctx, k, tmp1, tmp2) < 0) {
172  error:
173  EVP_CIPHER_CTX_free(evp_ctx);
174  return -1;
175  }
176 
177  /* OUT2 = E_K(rot(TEMP XOR OP_C, r2) XOR c2) XOR OP_C */
178  /* OUT3 = E_K(rot(TEMP XOR OP_C, r3) XOR c3) XOR OP_C */
179  /* OUT4 = E_K(rot(TEMP XOR OP_C, r4) XOR c4) XOR OP_C */
180  /* OUT5 = E_K(rot(TEMP XOR OP_C, r5) XOR c5) XOR OP_C */
181 
182  /* f2 and f5 */
183  /* rotate by r2 (= 0, i.e., NOP) */
184  for (i = 0; i < 16; i++) tmp1[i] = tmp2[i] ^ opc[i];
185  tmp1[15] ^= 1; /* XOR c2 (= ..01) */
186  /* f5 || f2 = E_K(tmp1) XOR OP_c */
187 
188  if (aes_128_encrypt_block(evp_ctx, k, tmp1, tmp3) < 0) goto error;
189 
190  for (i = 0; i < 16; i++) tmp3[i] ^= opc[i];
191  if (res) memcpy(res, tmp3 + 8, 8); /* f2 */
192  if (ak) memcpy(ak, tmp3, 6); /* f5 */
193 
194  /* f3 */
195  if (ck) {
196  /* rotate by r3 = 0x20 = 4 bytes */
197  for (i = 0; i < 16; i++) tmp1[(i + 12) % 16] = tmp2[i] ^ opc[i];
198  tmp1[15] ^= 2; /* XOR c3 (= ..02) */
199 
200  if (aes_128_encrypt_block(evp_ctx, k, tmp1, ck) < 0) goto error;
201 
202  for (i = 0; i < 16; i++) ck[i] ^= opc[i];
203  }
204 
205  /* f4 */
206  if (ik) {
207  /* rotate by r4 = 0x40 = 8 bytes */
208  for (i = 0; i < 16; i++) tmp1[(i + 8) % 16] = tmp2[i] ^ opc[i];
209  tmp1[15] ^= 4; /* XOR c4 (= ..04) */
210 
211  if (aes_128_encrypt_block(evp_ctx, k, tmp1, ik) < 0) goto error;
212 
213  for (i = 0; i < 16; i++) ik[i] ^= opc[i];
214  }
215 
216  /* f5* */
217  if (ak_resync) {
218  /* rotate by r5 = 0x60 = 12 bytes */
219  for (i = 0; i < 16; i++) tmp1[(i + 4) % 16] = tmp2[i] ^ opc[i];
220  tmp1[15] ^= 8; /* XOR c5 (= ..08) */
221 
222  if (aes_128_encrypt_block(evp_ctx, k, tmp1, tmp1) < 0) goto error;
223 
224  for (i = 0; i < 6; i++) ak_resync[i] = tmp1[i] ^ opc[i];
225  }
226  EVP_CIPHER_CTX_free(evp_ctx);
227 
228  return 0;
229 }

264 /** Generate AKA AUTN, IK, CK, RES
265  *
266  * @param[out] autn Buffer for AUTN = 128-bit authentication token.
267  * @param[out] ik Buffer for IK = 128-bit integrity key (f4), or NULL.
268  * @param[out] ck Buffer for CK = 128-bit confidentiality key (f3), or NULL.
269  * @param[out] ak Buffer for AK = 48-bit anonymity key (f5), or NULL
270  * @param[out] res Buffer for RES = 64-bit signed response (f2), or NULL.
271  * @param[in] opc 128-bit operator variant algorithm configuration field (encr.).
272  * @param[in] amf 16-bit authentication management field.
273  * @param[in] ki 128-bit subscriber key.
274  * @param[in] sqn 48-bit sequence number (host byte order).
275  * @param[in] rand 128-bit random challenge.
276  * @return
277  * - 0 on success.
278  * - -1 on failure.
279  */
280 int milenage_umts_generate(uint8_t autn[MILENAGE_AUTN_SIZE],
281  uint8_t ik[MILENAGE_IK_SIZE],
282  uint8_t ck[MILENAGE_CK_SIZE],
283  uint8_t ak[MILENAGE_AK_SIZE],
284  uint8_t res[MILENAGE_RES_SIZE],
285  uint8_t const opc[MILENAGE_OPC_SIZE],
286  uint8_t const amf[MILENAGE_AMF_SIZE],
287  uint8_t const ki[MILENAGE_KI_SIZE],
288  uint64_t sqn,
289  uint8_t const rand[MILENAGE_RAND_SIZE])
290 {
291  uint8_t mac_a[8], ak_buff[MILENAGE_AK_SIZE];
292  uint8_t sqn_buff[MILENAGE_SQN_SIZE];
293  uint8_t *p = autn;
294  size_t i;
295 
296  if ((milenage_f1(mac_a, NULL, opc, ki, rand,
297  uint48_to_buff(sqn_buff, sqn), amf) < 0) ||
298  (milenage_f2345(res, ik, ck, ak_buff, NULL, opc, ki, rand) < 0)) return -1;
299 
300  /*
301  * AUTN = (SQN ^ AK) || AMF || MAC_A
302  */
303  for (i = 0; i < sizeof(sqn_buff); i++) *p++ = sqn_buff[i] ^ ak_buff[i];
304  memcpy(p, amf, MILENAGE_AMF_SIZE);
305  p += MILENAGE_AMF_SIZE;
306  memcpy(p, mac_a, sizeof(mac_a));
307 
308  /*
309  * Output the anonymity key if required
310  */
311  if (ak) memcpy(ak, ak_buff, sizeof(ak_buff));
312 
313  return 0;
314 }

400  *
401  * @param[out] ik Buffer for IK = 128-bit integrity key (f4), or NULL.
402  * @param[out] ck Buffer for CK = 128-bit confidentiality key (f3), or NULL.
403  * @param[out] res Buffer for RES = 64-bit signed response (f2), or NULL.
404  * @param[in] auts 112-bit buffer for AUTS.
405  * @param[in] opc 128-bit operator variant algorithm configuration field (encr.).
406  * @param[in] ki 128-bit subscriber key.
407  * @param[in] sqn 48-bit sequence number.
408  * @param[in] rand 128-bit random challenge.
409  * @param[in] autn 128-bit authentication token.
410  * @return
411  * - 0 on success.
412  * - -1 on failure.
413  * - -2 on synchronization failure
414  */
415 int milenage_check(uint8_t ik[MILENAGE_IK_SIZE],
416  uint8_t ck[MILENAGE_CK_SIZE],
417  uint8_t res[MILENAGE_RES_SIZE],
418  uint8_t auts[MILENAGE_AUTS_SIZE],
419  uint8_t const opc[MILENAGE_OPC_SIZE],
420  uint8_t const ki[MILENAGE_KI_SIZE],
421  uint64_t sqn,
422  uint8_t const rand[MILENAGE_RAND_SIZE],
423  uint8_t const autn[MILENAGE_AUTN_SIZE])
424 {
425 
426  uint8_t mac_a[MILENAGE_MAC_A_SIZE], ak[MILENAGE_AK_SIZE], rx_sqn[MILENAGE_SQN_SIZE];
427  uint8_t sqn_buff[MILENAGE_SQN_SIZE];
428  const uint8_t *amf;
429  size_t i;
430 
431  uint48_to_buff(sqn_buff, sqn);
432 
433  FR_PROTO_HEX_DUMP(autn, MILENAGE_AUTN_SIZE, "AUTN");
434  FR_PROTO_HEX_DUMP(rand, MILENAGE_RAND_SIZE, "RAND");
435

437 
438  FR_PROTO_HEX_DUMP(res, MILENAGE_RES_SIZE, "RES");

441  FR_PROTO_HEX_DUMP(ak, MILENAGE_AK_SIZE, "AK");
442 
443  /* AUTN = (SQN ^ AK) || AMF || MAC */
444  for (i = 0; i < 6; i++) rx_sqn[i] = autn[i] ^ ak[i];
445  FR_PROTO_HEX_DUMP(rx_sqn, MILENAGE_SQN_SIZE, "SQN");
446 
447  if (CRYPTO_memcmp(rx_sqn, sqn_buff, sizeof(rx_sqn)) <= 0) {
448  uint8_t auts_amf[MILENAGE_AMF_SIZE] = { 0x00, 0x00 }; /* TS 33.102 v7.0.0, 6.3.3 */
449 
450  if (milenage_f2345(NULL, NULL, NULL, NULL, ak, opc, ki, rand)) return -1;
451 
452  FR_PROTO_HEX_DUMP(ak, sizeof(ak), "AK*");
453  for (i = 0; i < 6; i++) auts[i] = sqn_buff[i] ^ ak[i];
454 
455  if (milenage_f1(NULL, auts + 6, opc, ki, rand, sqn_buff, auts_amf) < 0) return -1;
456  FR_PROTO_HEX_DUMP(auts, 14, "AUTS");
457  return -2;
458  }
459 
460  amf = autn + 6;
461  FR_PROTO_HEX_DUMP(amf, MILENAGE_AMF_SIZE, "AMF");
462  if (milenage_f1(mac_a, NULL, opc, ki, rand, rx_sqn, amf) < 0) return -1;
463 
464  FR_PROTO_HEX_DUMP(mac_a, MILENAGE_MAC_A_SIZE, "MAC_A");
465 
466  if (CRYPTO_memcmp(mac_a, autn + 8, 8) != 0) {
467  FR_PROTO_HEX_DUMP(autn + 8, 8, "Received MAC_A");
468  fr_strerror_printf("MAC mismatch");
469  return -1;
470  }
471 
472  return 0;
473 }
