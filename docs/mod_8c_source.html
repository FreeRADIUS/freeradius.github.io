83 /** Build a couchbase_opts_t structure from the configuration "couchbase_api" list
84  *
85  * Parse the "couchbase_api" list from the module configuration file and store this
86  * as a couchbase_opts_t object (key/value list).
87  *
88  * @param conf Configuration list.
89  * @param instance The module instance.
90  * @return
91  * - 0 on success.
92  * - -1 on failure.
93  */
94 int mod_build_api_opts(CONF_SECTION *conf, void *instance)
95 {
96  rlm_couchbase_t *inst = instance; /* our module instance */
97  CONF_SECTION *cs; /* module config list */
98  CONF_ITEM *ci; /* config item */
99  CONF_PAIR *cp; /* config pair */
100  couchbase_opts_t *entry = NULL; /* couchbase api options */
101 
102  /* find opts list */
103  cs = cf_section_find(conf, "opts", NULL);
104 
105  /* check list */
106  if (!cs) return 0;
107 
108  /* parse libcouchbase_opts list */
109  cf_log_debug(cs, "opts {");
110 
111  for (ci = cf_item_next(cs, NULL); ci != NULL; ci = cf_item_next(cs, ci)) {
112  /*
113  * Ignore things we don't care about.
114  */
115  if (!cf_item_is_pair(ci)) {
116  continue;
117  }
118 
119  /* get value pair from item */
120  cp = cf_item_to_pair(ci);
121 
122  /* create opts object */
123  if (!entry) {
124  entry = talloc_zero(inst, couchbase_opts_t);
125  inst->api_opts = entry;
126  } else {
127  entry->next = talloc_zero(inst->api_opts, couchbase_opts_t);
128  entry = entry->next;
129  }
130  entry->next = NULL;
131  entry->key = talloc_typed_strdup(entry, cf_pair_attr(cp));
132  entry->val = talloc_typed_strdup(entry, cf_pair_value(cp));
133 
134  /* debugging */
135  cf_log_debug(cs, "\t%s = \"%s\"", entry->key, entry->val);
136  }
137 
138  cf_log_debug(cs, "}");
139 
140  /* return */
141  return 0;
142 }

144 /** Create a new connection pool handle
145  *
146  * Create a new connection to Couchbase within the pool and initialize
147  * information associated with the connection instance.
148  *
149  * @param ctx The connection parent context.
150  * @param instance The module instance.
151  * @param timeout Maximum time to establish the connection.
152  * @return
153  * - New connection handle.
154  * - NULL on error.
155  */
156 void *mod_conn_create(TALLOC_CTX *ctx, void *instance, fr_time_delta_t timeout)
157 {
158  rlm_couchbase_t const *inst = talloc_get_type_abort_const(instance, rlm_couchbase_t); /* module instance pointer */
159  rlm_couchbase_handle_t *chandle = NULL; /* connection handle pointer */
160  cookie_t *cookie = NULL; /* couchbase cookie */
161  lcb_t cb_inst; /* couchbase connection instance */
162  lcb_error_t cb_error; /* couchbase error status */
163  couchbase_opts_t const *opts = inst->api_opts; /* couchbase extra API settings */
164 
165  /* create instance */
166  cb_error = couchbase_init_connection(&cb_inst, inst->server, inst->bucket, inst->username,
167  inst->password, fr_time_delta_to_sec(timeout), opts);
168 
169  /* check couchbase instance */
170  if (cb_error != LCB_SUCCESS) {
171  ERROR("failed to initiate couchbase connection: %s (0x%x)",
172  lcb_strerror(NULL, cb_error), cb_error);
173  /* destroy/free couchbase instance */
174  lcb_destroy(cb_inst);
175  /* fail */
176  return NULL;
177  }
178 
179  /* allocate memory for couchbase connection instance abstraction */
180  chandle = talloc_zero(ctx, rlm_couchbase_handle_t);
181  talloc_set_destructor(chandle, _mod_conn_free);
182 
183  /* allocate cookie off handle */
184  cookie = talloc_zero(chandle, cookie_t);
185 
186  /* init tokener error and json object */
187  cookie->jerr = json_tokener_success;
188  cookie->jobj = NULL;
189 
190  /* populate handle */
191  chandle->cookie = cookie;
192  chandle->handle = cb_inst;
193 
194  /* return handle struct */
195  return chandle;
196 }

227 /** Build a JSON object map from the configuration "map" list
228  *
229  * Parse the "map" list from the module configuration file and store this
230  * as a JSON object (key/value list) in the module instance. This map will be
231  * used to lookup and map attributes for all incoming accounting requests.
232  *
233  * @param conf Configuration list.
234  * @param instance The module instance.
235  * @return
236  * - 0 on success.
237  * - -1 on failure.
238  */
239 int mod_build_attribute_element_map(CONF_SECTION *conf, void *instance)
240 {
241  rlm_couchbase_t *inst = instance; /* our module instance */
242  CONF_SECTION *cs; /* module config list */
243  CONF_ITEM *ci; /* config item */
244  CONF_PAIR *cp; /* conig pair */
245  const char *attribute, *element; /* attribute and element names */
246 
247  /* find update list */
248  cs = cf_section_find(conf, "update", NULL);
249 
250  /* backwards compatibility */
251  if (!cs) {
252  cs = cf_section_find(conf, "map", NULL);
253  WARN("found deprecated 'map' list - please change to 'update'");
254  }
255 
256  /* check list */
257  if (!cs) {
258  ERROR("failed to find 'update' list in config");
259  /* fail */
260  return -1;
261  }
262 
263  /* create attribute map object */
264  inst->map = json_object_new_object();
265 
266  /* parse update list */
267  for (ci = cf_item_next(cs, NULL); ci != NULL; ci = cf_item_next(cs, ci)) {
268  /* validate item */
269  if (!cf_item_is_pair(ci)) {
270  ERROR("failed to parse invalid item in 'update' list");
271  /* free map */
272  if (inst->map) {
273  json_object_put(inst->map);
274  }
275  /* fail */
276  return -1;
277  }
278 
279  /* get value pair from item */
280  cp = cf_item_to_pair(ci);
281 
282  /* get pair name (attribute name) */
283  attribute = cf_pair_attr(cp);
284 
285  /* get pair value (element name) */
286  element = cf_pair_value(cp);
287 
288  /* add pair name and value */
289  json_object_object_add(inst->map, attribute, json_object_new_string(element));
290 
291  /* debugging */
292  DEBUG3("added attribute '%s' to element '%s' mapping", attribute, element);
293  }
294 
295  /* debugging */
296  DEBUG3("built attribute to element mapping %s", json_object_to_json_string(inst->map));
297 
298  /* return */
299  return 0;
300 }

302 /** Map attributes to JSON element names
303  *
304  * Attempt to map the passed attribute name to the configured JSON element
305  * name using the JSON object map mod_build_attribute_element_map().
306  *
307  * @param name The character name of the requested attribute.
308  * @param map The JSON object map to use for the lookup.
309  * @param buf The buffer where the given element will be stored if found.
310  * @return
311  * - 0 on success.
312  * - -1 on failure.
313  */
314 int mod_attribute_to_element(const char *name, json_object *map, void *buf)
315 {
316  json_object *j_value; /* json object values */
317 
318  /* clear buffer */
319  memset((char *) buf, 0, MAX_KEY_SIZE);
320 
321  /* attempt to map attribute */
322  if (json_object_object_get_ex(map, name, &j_value)) {
323  /* copy and check size */
324  if (strlcpy(buf, json_object_get_string(j_value), MAX_KEY_SIZE) >= MAX_KEY_SIZE) {
325  /* oops ... this value is bigger than our buffer ... error out */
326  ERROR("json map value larger than MAX_KEY_SIZE - %d", MAX_KEY_SIZE);
327  /* return fail */
328  return -1;
329  }
330  /* looks good */
331  return 0;
332  }
333 
334  /* debugging */
335  DEBUG("skipping attribute with no map entry - %s", name);
336 
337  /* default return */
338  return -1;

341 /** Build value pairs from the passed JSON object and add to the request
342  *
343  * Parse the passed JSON object and create value pairs that will be injected into
344  * the given request for authorization.
345  *
346  * Example JSON document structure:
347  * @code{.json}
348  * {
349  * "docType": "raduser",
350  * "userName": "test",
351  * "control": {
352  * "SHA-Password": {
353  * "value": "a94a8fe5ccb19ba61c4c0873d391e987982fbbd3",
354  * "op": ":="
355  * }
356  * },
357  * "reply": {
358  * "Reply-Message": {
359  * "value": "Hidey Ho!",
360  * "op": "="
361  * }
362  * }
363  * }
364  * @endcode
365  *
366  * @param[in] ctx to allocate maps in.
367  * @param[in] out Cursor to append maps to.
368  * @param[in] request The request to which the generated pairs should be added.
369  * @param[in] json The JSON object representation of the user document.
370  * @param[in] list The pair list PAIR_LIST_CONTROL or PAIR_LIST_REPLY.
371  * @return
372  * - 1 if no section found.
373  * - 0 on success.
374  * - <0 on error.
375  */
376 int mod_json_object_to_map(TALLOC_CTX *ctx, fr_cursor_t *out, request_t *request, json_object *json, pair_list_t list)
377 {
378  json_object *list_obj;
379  char const *list_name = fr_table_str_by_value(pair_list_table, list, "<INVALID>");
380 
381  /*
382  * Check for a section matching the specified list
383  */
384  if (!json_object_object_get_ex(json, list_name, &list_obj)) {
385  RDEBUG2("Couldn't find \"%s\" key in json object - Not adding value pairs for this attribute list",
386  list_name);
387 
388  return 1;
389  }
390 
391  /*
392  * Check the key representing the list is a JSON object
393  */
394  if (!fr_json_object_is_type(list_obj, json_type_object)) {
395  RERROR("Invalid json type for \"%s\" key - Attribute lists must be json objects", list_name);
396 
397  return -1;
398  }
399 
400  fr_cursor_tail(out); /* Wind to the end */
401 
402  /*
403  * Loop through the keys in this object.
404  *
405  * Where attr_name is the key, and attr_value_obj is
406  * the object containing the attributes value and
407  * operator.
408  */
409  json_object_object_foreach(list_obj, attr_name, attr_value_obj) {
410  json_object *value_obj, *op_obj;
411  fr_dict_attr_t const *da;
412  fr_token_t op;
413 
414  if (!fr_json_object_is_type(attr_value_obj, json_type_object)) {
415  REDEBUG("Invalid json type for \"%s\" key - Attributes must be json objects", attr_name);
416 
417  error:
418  fr_cursor_free_list(out); /* Free any maps we added */
419  return -1;
420  }
421 
422  RDEBUG3("Parsing %s - \"%s\" : { %s }", list_name,
423  attr_name, json_object_to_json_string(attr_value_obj));
424 
425  /*
426  * Check we have a value key
427  */
428  if (!json_object_object_get_ex(attr_value_obj, "value", &value_obj)) {
429  REDEBUG("Missing \"value\" key in: %s - \"%s\" : { %s }", list_name,
430  attr_name, json_object_to_json_string(attr_value_obj));
431 
432  goto error;
433  }
434 
435  /*
436  * Parse the operator and check its valid
437  */
438  if (json_object_object_get_ex(attr_value_obj, "op", &op_obj)) {
439  char const *op_str;
440 
441  op_str = json_object_get_string(op_obj);
442  if (!op_str) {
443  bad_op:
444  REDEBUG("Invalid \"op\" key in: %s - \"%s\" : { %s }", list_name,
445  attr_name, json_object_to_json_string(attr_value_obj));
446 
447  goto error;
448  }
449 
450  op = fr_table_value_by_str(fr_tokens_table, op_str, T_INVALID);
451  if (!fr_assignment_op[op] && !fr_equality_op[op]) goto bad_op;
452  } else {
453  op = T_OP_SET; /* The default */
454  }
455 
456  /*
457  * Lookup the string attr_name in the
458  * request dictionary.
459  */
460  da = fr_dict_attr_by_name(request->dict, attr_name);
461  if (!da) {
462  RPERROR("Invalid attribute \"%s\"", attr_name);
463  goto error;
464  }
465 
466  /*
467  * Create a map representing the operation
468  */
469  {
470  fr_value_box_t tmp = { .type = FR_TYPE_INVALID };
471  map_t *map;
472 
473  if (fr_json_object_to_value_box(ctx, &tmp, value_obj, da, true) < 0) {
474  bad_value:
475  RPERROR("Failed parsing value for \"%s\"", attr_name);
476  goto error;
477  }
478 
479  if (fr_value_box_cast_in_place(ctx, &tmp, da->type, da) < 0) {
480  fr_value_box_clear(&tmp);
481  goto bad_value;
482  }
483 
484  if (map_afrom_value_box(ctx, &map,
485  attr_name, T_BARE_WORD,
486  &(tmpl_rules_t){
487  .dict_def = request->dict,
488  .list_def = list,
489  },
490  op,
491  &tmp, true) < 0) {
492  fr_value_box_clear(&tmp);
493  goto bad_value;
494  }
495 
496  fr_cursor_insert(out, map);
497  }
498  }
499 
500  return 0;
501 }

503 /** Convert value pairs to json objects
504  *
505  * Take the passed value pair and convert it to a json-c JSON object.
506  * This code is heavily based on the fr_json_from_pair() function
507  * from src/lib/print.c.
508  *
509  * @param request The request object.
510  * @param vp The value pair to convert.
511  * @return A JSON object.
512  */
513 json_object *mod_value_pair_to_json_object(request_t *request, fr_pair_t *vp)
514 {
515  char value[255]; /* radius attribute value */
516 
517  /* add this attribute/value pair to our json output */
518  {
519  unsigned int i;
520 
521  switch (vp->vp_type) {
522  case FR_TYPE_UINT32:
523  i = vp->vp_uint32;
524  goto print_int;
525 
526  case FR_TYPE_UINT16:
527  i = vp->vp_uint16;
528  goto print_int;
529 
530  case FR_TYPE_UINT8:
531  i = vp->vp_uint8;
532 
533  print_int:
534  /* skip if we have flags */
535  if (vp->da->flags.has_value) break;
536 #ifdef HAVE_JSON_OBJECT_NEW_INT64
537  /* debug */
538  RDEBUG3("creating new int64 for unsigned 32 bit int/byte/short '%s'", vp->da->name);
539  /* return as 64 bit int - JSON spec does not support unsigned ints */
540  return json_object_new_int64(i);
541 #else
542  /* debug */
543  RDEBUG3("creating new int for unsigned 32 bit int/byte/short '%s'", vp->da->name);
544  /* return as 64 bit int - JSON spec does not support unsigned ints */
545  return json_object_new_int(i);
546 #endif
547 
548  case FR_TYPE_INT32:
549 #ifdef HAVE_JSON_OBJECT_NEW_INT64
550  /* debug */
551  RDEBUG3("creating new int64 for signed 32 bit integer '%s'", vp->da->name);
552  /* return as 64 bit int - json-c represents all ints as 64 bits internally */
553  return json_object_new_int64(vp->vp_int32);
554 #else
555  RDEBUG3("creating new int for signed 32 bit integer '%s'", vp->da->name);
556  /* return as signed int */
557  return json_object_new_int(vp->vp_int32);
558 #endif
559 
560  case FR_TYPE_UINT64:
561 #ifdef HAVE_JSON_OBJECT_NEW_INT64
562  /* debug */
563  RDEBUG3("creating new int64 for 64 bit integer '%s'", vp->da->name);
564  /* return as 64 bit int - because it is a 64 bit int */
565  return json_object_new_int64(vp->vp_uint64);
566 #else
567  /* warning */
568  RWARN("skipping 64 bit integer attribute '%s' - please upgrade json-c to 0.10+", vp->da->name);
569  break;
570 #endif
571 
572  default:
573  /* silence warnings - do nothing */
574  break;
575  }
576  }
577 
578  /* keep going if not set above */
579  switch (vp->vp_type) {
580  case FR_TYPE_STRING:
581  /* debug */
582  RDEBUG3("assigning string '%s' as string", vp->da->name);
583  /* return string value */
584  return json_object_new_string(vp->vp_strvalue);
585 
586  default:
587  /* debug */
588  RDEBUG3("assigning unhandled '%s' as string", vp->da->name);
589  /* get standard value */
590  fr_pair_print_value_quoted(&FR_SBUFF_OUT(value, sizeof(value)), vp, T_BARE_WORD);
591  /* return string value from above */
592  return json_object_new_string(value);
593  }
594 }

596 /** Ensure accounting documents always contain a valid timestamp
597  *
598  * Inspect the given JSON object representation of an accounting document
599  * fetched from Couchbase and ensuse it contains a valid (non NULL) timestamp value.
600  *
601  * @param json JSON object representation of an accounting document.
602  * @param vps The value pairs associated with the current accounting request.
603  * @return
604  * - 0 on success.
605  * - -1 on failure.
606  */
607 int mod_ensure_start_timestamp(json_object *json, fr_pair_t *vps)
608 {
609  json_object *j_value; /* json object value */
610  struct tm tm; /* struct to hold event time */
611  time_t ts = 0; /* values to hold time in seconds */
612  fr_pair_t *vp; /* values to hold value pairs */
613  char value[255]; /* store radius attribute values and our timestamp */
614 
615  /* get our current start timestamp from our json body */
616  if (json_object_object_get_ex(json, "startTimestamp", &j_value) == 0) {
617  /* debugging ... this shouldn't ever happen */
618  DEBUG("failed to find 'startTimestamp' in current json body");
619  /* return */
620  return -1;
621  }
622 
623  /* check for null value */
624  if (json_object_get_string(j_value) != NULL) {
625  /* already set - nothing left to do */
626  return 0;
627  }
628 
629  /* get current event timestamp */
630  if ((vp = fr_pair_find_by_da(&vps, attr_event_timestamp)) != NULL) {
631  /* get seconds value from attribute */
632  ts = fr_time_to_sec(vp->vp_date);
633  } else {
634  /* debugging */
635  DEBUG("failed to find event timestamp in current request");
636  /* return */
637  return -1;
638  }
639 
640  /* clear value */
641  memset(value, 0, sizeof(value));
642 
643  /* get elapsed session time */
644  if ((vp = fr_pair_find_by_da(&vps, attr_acct_session_time)) != NULL) {
645  /* calculate diff */
646  ts = (ts - vp->vp_uint32);
647  /* calculate start time */
648  size_t length = strftime(value, sizeof(value), "%b %e %Y %H:%M:%S %Z", localtime_r(&ts, &tm));
649  /* check length */
650  if (length > 0) {
651  /* debugging */
652  DEBUG("calculated start timestamp: %s", value);
653  /* store new value in json body */
654  json_object_object_add(json, "startTimestamp", json_object_new_string(value));
655  } else {
656  /* debugging */
657  DEBUG("failed to format calculated timestamp");
658  /* return */
659  return -1;
660  }
661  }
662 
663  /* default return */
664  return 0;
665 }

667 /** Handle client value processing for client_map_section()
668  *
669  * @param out Character output
670  * @param cp Configuration pair
671  * @param data The client data
672  * @return
673  * - 0 on success.
674  * - -1 on failure.
675  */
676 static int _get_client_value(char **out, CONF_PAIR const *cp, void *data)
677 {
678  json_object *j_value;
679 
680  if (!json_object_object_get_ex((json_object *)data, cf_pair_value(cp), &j_value)) {
681  *out = NULL;
682  return 0;
683  }
684

693 /** Load client entries from Couchbase client documents on startup
694  *
695  * This function executes the view defined in the module configuration and loops
696  * through all returned rows. The view is called with "stale=false" to ensure the
697  * most accurate data available when the view is called. This will force an index
698  * rebuild on this design document in Couchbase. However, since this function is only
699  * run once at server startup this should not be a concern.
700  *
701  * @param inst The module instance.
702  * @param tmpl Default values for new clients.
703  * @param map The client attribute configuration list.
704  * @return
705  * - 0 on success.
706  * - -1 on failure.
707  */
708 int mod_load_client_documents(rlm_couchbase_t *inst, CONF_SECTION *tmpl, CONF_SECTION *map)
709 {
710  rlm_couchbase_handle_t *handle = NULL; /* connection pool handle */
711  char vpath[256], vid[MAX_KEY_SIZE], vkey[MAX_KEY_SIZE]; /* view path and fields */
712  char error[512]; /* view error return */
713  int idx = 0; /* row array index counter */
714  int retval = 0; /* return value */
715  lcb_error_t cb_error = LCB_SUCCESS; /* couchbase error holder */
716  json_object *json, *j_value; /* json object holders */
717  json_object *jrows = NULL; /* json object to hold view rows */
718  CONF_SECTION *client; /* freeradius config list */
719  RADCLIENT *c; /* freeradius client */
720 
721  /* get handle */
722  handle = fr_pool_connection_get(inst->pool, NULL);
723 
724  /* check handle */
725  if (!handle) return -1;
726 
727  /* set couchbase instance */
728  lcb_t cb_inst = handle->handle;
729 
730  /* set cookie */
731  cookie_t *cookie = handle->cookie;
732 
733  /* build view path */
734  snprintf(vpath, sizeof(vpath), "%s?stale=false", inst->client_view);
735 
736  /* query view for document */
737  cb_error = couchbase_query_view(cb_inst, cookie, vpath, NULL);
738 
739  /* check error and object */
740  if (cb_error != LCB_SUCCESS || cookie->jerr != json_tokener_success || !cookie->jobj) {
741  /* log error */
742  ERROR("failed to execute view request or parse return");
743  /* set return */
744  retval = -1;
745  /* return */
746  goto free_and_return;
747  }
748 
749  /* debugging */
750  DEBUG3("cookie->jobj == %s", json_object_to_json_string(cookie->jobj));
751 
752  /* check for error in json object */
753  if (json_object_object_get_ex(cookie->jobj, "error", &json)) {
754  /* build initial error buffer */
755  strlcpy(error, json_object_get_string(json), sizeof(error));
756  /* get error reason */
757  if (json_object_object_get_ex(cookie->jobj, "reason", &json)) {
758  /* append divider */
759  strlcat(error, " - ", sizeof(error));
760  /* append reason */
761  strlcat(error, json_object_get_string(json), sizeof(error));
762  }
763  /* log error */
764  ERROR("view request failed with error: %s", error);
765  /* set return */
766  retval = -1;
767  /* return */
768  goto free_and_return;
769  }
770 
771  /* check for document id in return */
772  if (!json_object_object_get_ex(cookie->jobj, "rows", &json)) {
773  /* log error */
774  ERROR("failed to fetch rows from view payload");
775  /* set return */
776  retval = -1;
777  /* return */
778  goto free_and_return;
779  }
780 
781  /* get and hold rows */
782  jrows = json_object_get(json);
783 
784  /* free cookie object */
785  if (cookie->jobj) {
786  json_object_put(cookie->jobj);
787  cookie->jobj = NULL;
788  }
789 
790  /* debugging */
791  DEBUG3("jrows == %s", json_object_to_json_string(jrows));
792 
793  /* check for valid row value */
794  if (!fr_json_object_is_type(jrows, json_type_array) || json_object_array_length(jrows) < 1) {
795  /* log error */
796  ERROR("no valid rows returned from view: %s", vpath);
797  /* set return */
798  retval = -1;
799  /* return */
800  goto free_and_return;
801  }
802 
803  /* loop across all row elements */
804  for (idx = 0; (size_t)idx < (size_t)json_object_array_length(jrows); idx++) {
805  /* fetch current index */
806  json = json_object_array_get_idx(jrows, idx);
807 
808  /* get view id */
809  if (json_object_object_get_ex(json, "id", &j_value)) {
810  /* clear view id */
811  memset(vid, 0, sizeof(vid));
812  /* copy and check length */
813  if (strlcpy(vid, json_object_get_string(j_value), sizeof(vid)) >= sizeof(vid)) {
814  ERROR("id from row longer than MAX_KEY_SIZE (%d)",
815  MAX_KEY_SIZE);
816  continue;
817  }
818  } else {
819  WARN("failed to fetch id from row - skipping");
820  continue;
821  }
822 
823  /* get view key */
824  if (json_object_object_get_ex(json, "key", &j_value)) {
825  /* clear view key */
826  memset(vkey, 0, sizeof(vkey));
827  /* copy and check length */
828  if (strlcpy(vkey, json_object_get_string(j_value), sizeof(vkey)) >= sizeof(vkey)) {
829  ERROR("key from row longer than MAX_KEY_SIZE (%d)",
830  MAX_KEY_SIZE);
831  continue;
832  }
833  } else {
834  WARN("failed to fetch key from row - skipping");
835  continue;
836  }
837 
838  /* fetch document */
839  cb_error = couchbase_get_key(cb_inst, cookie, vid);
840 
841  /* check error and object */
842  if (cb_error != LCB_SUCCESS || cookie->jerr != json_tokener_success || !cookie->jobj) {
843  /* log error */
844  ERROR("failed to execute get request or parse return");
845  /* set return */
846  retval = -1;
847  /* return */
848  goto free_and_return;
849  }
850 
851  /* debugging */
852  DEBUG3("cookie->jobj == %s", json_object_to_json_string(cookie->jobj));
853 
854  /* allocate conf list */
855  client = tmpl ? cf_section_dup(NULL, NULL, tmpl, "client", vkey, true) :
856  cf_section_alloc(NULL, NULL, "client", vkey);
857 
858  if (client_map_section(client, map, _get_client_value, cookie->jobj) < 0) {
859  /* free config setion */
860  talloc_free(client);
861  /* set return */
862  retval = -1;
863  /* return */
864  goto free_and_return;
865  }
866 
867  /*
868  * @todo These should be parented from something.
869  */
870  c = client_afrom_cs(NULL, client, false);
871  if (!c) {
872  ERROR("failed to allocate client");
873  /* free config setion */
874  talloc_free(client);
875  /* set return */
876  retval = -1;
877  /* return */
878  goto free_and_return;
879  }
880 
881  /*
882  * Client parents the CONF_SECTION which defined it.
883  */
884  talloc_steal(c, client);
885 
886  /* attempt to add client */
887  if (!client_add(NULL, c)) {
888  ERROR("failed to add client '%s' from '%s', possible duplicate?", vkey, vid);
889  /* free client */
890  client_free(c);
891  /* set return */
892  retval = -1;
893  /* return */
894  goto free_and_return;
895  }
896 
897  /* debugging */
898  DEBUG("client '%s' added", c->longname);
899 
900  /* free json object */
901  if (cookie->jobj) {
902  json_object_put(cookie->jobj);
903  cookie->jobj = NULL;
904  }
905  }
906 
907  free_and_return:
908 
909  /* free rows */
910  if (jrows) {
911  json_object_put(jrows);
912  }
913 
914  /* free json object */
915  if (cookie->jobj) {
916  json_object_put(cookie->jobj);
917  cookie->jobj = NULL;
918  }
919 
920  /* release handle */
921  if (handle) fr_pool_connection_release(inst->pool, NULL, handle);
922 
923  /* return */
924  return retval;
925 }
