81 static int timecmp(UNUSED void *instance, request_t *req, UNUSED fr_pair_t *request, fr_pair_t *check,
82  UNUSED fr_pair_t *check_list)
83 {
84  /*
85  * If there's a request, use that timestamp.
86  */
87  if (timestr_match(check->vp_strvalue, req ? fr_time_to_sec(req->packet->timestamp) : time(NULL)) >= 0) return 0;
88 
89  return -1;
90 }

93 /*
94  * Time-Of-Day support
95  */
96 static int time_of_day(UNUSED void *instance, request_t *request,
97  UNUSED fr_pair_t *request_list, fr_pair_t *check,
98  UNUSED fr_pair_t *check_list)
99 {
100  int scan;
101  int hhmmss, when;
102  char const *p;
103  struct tm *tm, s_tm;
104  time_t now;
105 
106  if (strspn(check->vp_strvalue, "0123456789: ") != strlen(check->vp_strvalue)) {
107  RDEBUG2("Bad Time-Of-Day value \"%pV\"", &check->data);
108  return -1;
109  }
110 
111  now = fr_time_to_sec(request->packet->timestamp);
112  tm = localtime_r(&now, &s_tm);
113  hhmmss = (tm->tm_hour * 3600) + (tm->tm_min * 60) + tm->tm_sec;
114 
115  /*
116  * Time of day is a 24-hour clock
117  */
118  p = check->vp_strvalue;
119  scan = atoi(p);
120  p = strchr(p, ':');
121  if ((scan > 23) || !p) {
122  RDEBUG2("Bad Time-Of-Day value \"%pV\"", &check->data);
123  return -1;
124  }
125  when = scan * 3600;
126  p++;
127 
128  scan = atoi(p);
129  if (scan > 59) {
130  RDEBUG2("Bad Time-Of-Day value \"%pV\"", &check->data);
131  return -1;
132  }
133  when += scan * 60;
134 
135  p = strchr(p, ':');
136  if (p) {
137  scan = atoi(p + 1);
138  if (scan > 59) {
139  RDEBUG2("Bad Time-Of-Day value \"%pV\"", &check->data);
140  return -1;
141  }
142  when += scan;
143  }
144 
145  fprintf(stderr, "returning %d - %d\n",
146  hhmmss, when);
147 
148  return hhmmss - when;
149 }

151 /*
152  * Check if account has expired, and if user may login now.
153  */
154 static unlang_action_t CC_HINT(nonnull) mod_authorize(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
155 {
156  rlm_logintime_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_logintime_t);
157  fr_pair_t *ends, *vp;
158  int32_t left;
159 
160  ends = fr_pair_find_by_da(&request->control_pairs, attr_login_time);
161  if (!ends) RETURN_MODULE_NOOP;
162 
163  /*
164  * Authentication is OK. Now see if this user may login at this time of the day.
165  */
166  RDEBUG2("Checking Login-Time");
167 
168  /*
169  * Compare the time the request was received with the current Login-Time value
170  */
171  left = timestr_match(ends->vp_strvalue, fr_time_to_sec(request->packet->timestamp));
172  if (left < 0) RETURN_MODULE_DISALLOW; /* outside of the allowed time */
173 
174  /*
175  * Do nothing, login time is not controlled (unendsed).
176  */
177  if (left == 0) RETURN_MODULE_OK;
178 
179  /*
180  * The min_time setting is to deal with NAS that won't allow Session-vp values below a certain value
181  * For example some Alcatel Lucent products won't allow a Session-vp < 300 (5 minutes).
182  *
183  * We don't know were going to get another chance to lock out the user, so we need to do it now.
184  */
185  if ((uint32_t)left < inst->min_time) {
186  REDEBUG("Login outside of allowed time-slot (session end %s, with lockout %i seconds before)",
187  ends->vp_strvalue, inst->min_time);
188 
189  RETURN_MODULE_DISALLOW;
190  }
191 
192  /* else left > inst->min_time */
193 
194  /*
195  * There's time left in the users session, inform the NAS by including a Session-vp
196  * attribute in the reply, or modifying the existing one.
197  */
198  RDEBUG2("Login within allowed time-slot, %d seconds left in this session", left);
199 
200  switch (pair_update_reply(&vp, attr_session_timeout)) {
201  case 1:
202  /* just update... */
203  if (vp->vp_uint32 > (uint32_t)left) {
204  vp->vp_uint32 = (uint32_t)left;
205  RDEBUG2("&reply.Session-Timeout := %pV", &vp->data);
206  }
207  break;
208 
209  case 0: /* no pre-existing */
210  vp->vp_uint32 = (uint32_t)left;
211  RDEBUG2("&reply.Session-Timeout := %pV", &vp->data);
212  break;
213 
214  case -1: /* malloc failure */
215  MEM(NULL);
216  }
217 
218  RETURN_MODULE_OK;
219 }
