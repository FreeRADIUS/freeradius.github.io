61 {\
62  rlm_lua_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_lua_t);\

109 /** Close the global interpreter
110  *
111  */
112 static int mod_detach(void *instance)
113 {
114  rlm_lua_t *inst = instance;
115  rlm_rcode_t ret = 0;
116 
117  /*
118  * May be NULL if fr_lua_init failed
119  */
120  if (inst->interpreter) {
121  if (inst->func_detach) {
122  fr_lua_run(&ret, &(module_ctx_t){
123  .instance = inst,
124  .thread = &(rlm_lua_thread_t){
125  .interpreter = inst->interpreter
126  }
127  },
128  NULL, inst->func_detach);
129  }
130  lua_close(inst->interpreter);
131  }
132 
133  return ret;
134 }

136 static int mod_instantiate(void *instance, CONF_SECTION *conf)
137 {
138  rlm_lua_t *inst = instance;
139  rlm_rcode_t rcode;
140 
141  inst->xlat_name = cf_section_name2(conf);
142  if (!inst->xlat_name) inst->xlat_name = cf_section_name1(conf);
143 
144  /*
145  * Get an instance global interpreter to use with various things...
146  */
147  if (fr_lua_init(&inst->interpreter, inst) < 0) return -1;
148  inst->jit = fr_lua_isjit(inst->interpreter);
149  if (!inst->jit) WARN("Using standard Lua interpreter, performance will be suboptimal");
150 
151  DEBUG("Using %s interpreter", fr_lua_version(inst->interpreter));
152 
153  if (inst->func_instantiate) {
154  fr_lua_run(&rcode, &(module_ctx_t){
155  .instance = inst,
156  .thread = &(rlm_lua_thread_t){
157  .interpreter = inst->interpreter
158  }
159  },
160  NULL, inst->func_instantiate);
161  }
162 
163  return 0;
164 }
