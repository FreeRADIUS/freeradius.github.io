197 /** Add a deferred signal to the signal list
198  *
199  * Processing signals whilst in handlers usually leads to weird
200  * inconsistent states within the connection.
201  *
202  * If a public signal function is called, and detects its being called
203  * from within the handler, it instead adds a deferred signal entry
204  * and immediately returns.
205  *
206  * Once the handler is complete, and all pending C stack state changes
207  * are complete, the deferred signals are drained and processed.
208  */
209 static inline void connection_deferred_signal_add(fr_connection_t *conn, connection_dsignal_t signal)
210 {
211  connection_dsignal_entry_t *dsignal, *prev;
212 
213  prev = fr_dlist_tail(&conn->deferred_signals);
214  if (prev && (prev->signal == signal)) return; /* Don't insert duplicates */
215 
216  dsignal = talloc_zero(conn, connection_dsignal_entry_t);
217  dsignal->signal = signal;
218  fr_dlist_insert_tail(&conn->deferred_signals, dsignal);
219 
220 // DEBUG4("Adding deferred signal - %s", fr_table_str_by_value(connection_dsignals, signal, "<INVALID>"));
221 }

234 /** Process any deferred signals
235  *
236  */
237 static void connection_deferred_signal_process(fr_connection_t *conn)
238 {
239  connection_dsignal_entry_t *dsignal;
240  bool freed = false;
241 
242  /*
243  * We're inside and an instance of this function
244  * higher in the call stack. Don't do anything.
245  */
246  if (conn->processing_signals) return;
247 
248  /*
249  * Get notified if the connection gets freed
250  * out from under us...
251  */
252  fr_connection_watch_enable_set_uctx(conn->on_halted, &freed);
253  conn->processing_signals = true;
254 
255  while ((dsignal = fr_dlist_head(&conn->deferred_signals))) {
256  connection_dsignal_t signal;
257  fr_dlist_remove(&conn->deferred_signals, dsignal);
258  signal = dsignal->signal;
259  talloc_free(dsignal);
260 
261  DEBUG4("Processing deferred signal - %s",
262  fr_table_str_by_value(connection_dsignals, signal, "<INVALID>"));
263 
264  switch (signal) {
265  case CONNECTION_DSIGNAL_INIT:
266  fr_connection_signal_init(conn);
267  break;
268 
269  case CONNECTION_DSIGNAL_CONNECTED:
270  fr_connection_signal_connected(conn);
271  break;
272 
273  case CONNECTION_DSIGNAL_RECONNECT_FAILED: /* Reconnect - Failed */
274  fr_connection_signal_reconnect(conn, FR_CONNECTION_FAILED);
275  break;
276 
277  case CONNECTION_DSIGNAL_RECONNECT_EXPIRED: /* Reconnect - Expired */
278  fr_connection_signal_reconnect(conn, FR_CONNECTION_EXPIRED);
279  break;
280 
281  case CONNECTION_DSIGNAL_SHUTDOWN:
282  fr_connection_signal_shutdown(conn);
283  break;
284 
285  case CONNECTION_DSIGNAL_HALT:
286  fr_connection_signal_halt(conn);
287  break;
288 
289  case CONNECTION_DSIGNAL_FREE: /* Freed */
290  talloc_free(conn);
291  return;
292  }
293 
294  /*
295  * One of the signal handlers freed the connection
296  * return immediately.
297  */
298  if (freed) return;
299  }
300 
301  conn->processing_signals = false;
302  fr_connection_watch_disable(conn->on_halted);
303 }

352 /** Call a list of watch functions associated with a state
353  *
354  */
355 static inline void connection_watch_call(fr_connection_t *conn, fr_dlist_head_t *list)
356 {
357  /*
358  * Nested watcher calls are not allowed
359  * and shouldn't be possible because of
360  * deferred signal processing.
361  */
362  fr_assert(conn->next_watcher == NULL);
363 
364  while ((conn->next_watcher = fr_dlist_next(list, conn->next_watcher))) {
365  fr_connection_watch_entry_t *entry = conn->next_watcher;
366  bool oneshot = entry->oneshot; /* Watcher could be freed, so store now */
367 
368  if (!entry->enabled) continue;
369  if (oneshot) conn->next_watcher = fr_dlist_remove(list, entry);
370 
371 /*
372  DEBUG4("Notifying %swatcher - (%p)(conn=%p, prev=%s, state=%s, uctx=%p)",
373  entry->oneshot ? "oneshot " : "",
374  entry->func,
375  conn,
376  fr_table_str_by_value(fr_connection_states, conn->pub.prev, "<INVALID>"),
377  fr_table_str_by_value(fr_connection_states, conn->pub.state, "<INVALID>"),
378  entry->uctx);
379 */
380 
381  entry->func(conn, conn->pub.prev, conn->pub.state, entry->uctx);
382 
383  if (oneshot) talloc_free(entry);
384  }
385  conn->next_watcher = NULL;
386 }

410 /** Remove a watch function from a pre/post[state] list
411  *
412  */
413 static int connection_del_watch(fr_connection_t *conn, fr_dlist_head_t *state_lists,
414  fr_connection_state_t state, fr_connection_watch_t watch)
415 {
416  fr_connection_watch_entry_t *entry = NULL;
417  fr_dlist_head_t *list = &state_lists[state];
418 
419  while ((entry = fr_dlist_next(list, entry))) {
420  if (entry->func == watch) {
421 /*
422  DEBUG4("Removing %s watcher %p",
423  fr_table_str_by_value(fr_connection_states, state, "<INVALID>"),
424  watch);
425 */
426  if (conn->next_watcher == entry) {
427  conn->next_watcher = fr_dlist_remove(list, entry);
428  } else {
429  fr_dlist_remove(list, entry);
430  }
431  talloc_free(entry);
432  return 0;
433  }
434  }
435 
436  return -1;
437 }

613  *

615  * @param[in] now The current time.
616  * @param[in] uctx The #fr_connection_t the fd is associated with.
617  */

621 
622  switch (conn->pub.state) {
623  case FR_CONNECTION_STATE_FAILED:
624  case FR_CONNECTION_STATE_CLOSED:

634 /** Close the connection, then wait for another state change
635  *
636  */
637 static void connection_state_enter_closed(fr_connection_t *conn)
638 {
639  switch (conn->pub.state) {
640  case FR_CONNECTION_STATE_CONNECTING:
641  case FR_CONNECTION_STATE_CONNECTED:
642  case FR_CONNECTION_STATE_FAILED:
643  break;
644 
645  default:
646  BAD_STATE_TRANSITION(FR_CONNECTION_STATE_CLOSED);
647  return;
648  }
649 
650  STATE_TRANSITION(FR_CONNECTION_STATE_CLOSED);
651 
652  fr_event_timer_delete(&conn->ev);
653 
654  /*
655  * If there's a close callback, call it, so that the
656  * API client can free any resources associated
657  * with the connection handle.
658  */
659  WATCH_PRE(conn);
660 
661  /*
662  * is_closed is for pure paranoia. If everything
663  * is working correctly this state should never
664  * be entered if the connection is closed.
665  */
666  fr_assert(!conn->is_closed);
667  if (conn->close && !conn->is_closed) {
668  HANDLER_BEGIN(conn, conn->close);
669  DEBUG4("Calling close(el=%p, h=%p, uctx=%p)", conn->pub.el, conn->pub.h, conn->uctx);
670  conn->close(conn->pub.el, conn->pub.h, conn->uctx);
671  conn->is_closed = true; /* Ensure close doesn't get called twice if the connection is freed */
672  HANDLER_END(conn);
673  } else {
674  conn->is_closed = true;
675  }
676  WATCH_POST(conn);
677 }

694 /** Gracefully shutdown the handle
695  *
696  */
697 static void connection_state_enter_shutdown(fr_connection_t *conn)
698 {
699  fr_connection_state_t ret;
700 
701  switch (conn->pub.state) {
702  case FR_CONNECTION_STATE_CONNECTED:
703  break;
704 
705  default:
706  BAD_STATE_TRANSITION(FR_CONNECTION_STATE_SHUTDOWN);
707  return;
708  }
709 
710  STATE_TRANSITION(FR_CONNECTION_STATE_SHUTDOWN);
711 
712  WATCH_PRE(conn);
713  {
714  HANDLER_BEGIN(conn, conn->shutdown);
715  DEBUG4("Calling shutdown(el=%p, h=%p, uctx=%p)", conn->pub.el, conn->pub.h, conn->uctx);
716  ret = conn->shutdown(conn->pub.el, conn->pub.h, conn->uctx);
717  HANDLER_END(conn);
718  }
719  switch (ret) {
720  case FR_CONNECTION_STATE_SHUTDOWN:
721  break;
722 
723  default:
724  connection_state_enter_failed(conn);
725  return;
726  }
727  WATCH_POST(conn);
728 
729  /*
730  * If there's a connection timeout,
731  * set, then add the timer.
732  *
733  * The connection may be bad, in which
734  * case we want to automatically fail
735  * if it doesn't shutdown within the
736  * timeout period.
737  */
738  if (conn->connection_timeout) {
739  if (fr_event_timer_in(conn, conn->pub.el, &conn->ev,
740  conn->connection_timeout, _connection_timeout, conn) < 0) {
741  /*
742  * Can happen when the event loop is exiting
743  */
744  PERROR("Failed setting connection_timeout timer, closing connection");
745  connection_state_enter_closed(conn);
746  }
747  }
748 }

750 /** Connection failed
751  *
752  * Transition to the FR_CONNECTION_STATE_FAILED state.
753  *
754  * If the connection was open, or couldn't be opened wait for reconnection_delay before transitioning
755  * back to init.
756  *
757  * If no reconnection_delay was set, transition to halted.
758  *
759  * @param[in] conn that failed.
760  */
761 static void connection_state_enter_failed(fr_connection_t *conn)
762 {
763  fr_connection_state_t prev;
764  fr_connection_state_t ret = FR_CONNECTION_STATE_INIT;
765 
766  fr_assert(conn->pub.state != FR_CONNECTION_STATE_FAILED);
767 
768  /*
769  * Explicit error occurred, delete the connection timer
770  */
771  fr_event_timer_delete(&conn->ev);
772 
773  /*
774  * Record what state the connection is currently in
775  * so we can figure out what to do next.
776  */
777  prev = conn->pub.state;
778 
779  /*
780  * Now transition to failed
781  */
782  STATE_TRANSITION(FR_CONNECTION_STATE_FAILED);
783 
784  /*
785  * If there's a failed callback, give it the
786  * opportunity to suspend/destroy the
787  * connection.
788  */
789  WATCH_PRE(conn);
790  if (conn->failed) {
791  HANDLER_BEGIN(conn, conn->failed);
792  DEBUG4("Calling failed(h=%p, state=%s, uctx=%p)", conn->pub.h,
793  fr_table_str_by_value(fr_connection_states, prev, "<INVALID>"), conn->uctx);
794  ret = conn->failed(conn->pub.h, prev, conn->uctx);
795  HANDLER_END(conn);
796  }
797  WATCH_POST(conn);
798 
799  /*
800  * Enter the closed state if we failed during
801  * connecting, or when we were connected.
802  */
803  switch (prev) {
804  case FR_CONNECTION_STATE_CONNECTED:
805  case FR_CONNECTION_STATE_CONNECTING:
806  case FR_CONNECTION_STATE_TIMEOUT: /* Timeout means the connection progress past init */
807  case FR_CONNECTION_STATE_SHUTDOWN: /* Shutdown means the connection failed whilst shutting down */
808  connection_state_enter_closed(conn);
809  break;
810 
811  default:
812  break;
813  }
814 
815  if (conn->failed) {
816  switch (ret) {
817  /*
818  * The callback signalled it wants the
819  * connection to be reinitialised
820  * after reconnection_delay, or
821  * immediately if the failure was due
822  * to a connection timeout.
823  */
824  case FR_CONNECTION_STATE_INIT:
825  break;
826 
827  /*
828  * The callback signalled it wants the
829  * connection to stop.
830  */
831  case FR_CONNECTION_STATE_HALTED:
832  default:
833  connection_state_enter_halted(conn);
834  return;
835  }
836  }
837 
838  /*
839  * What previous state we were in
840  * determines if we need to apply the
841  * reconnect timeout.
842  */
843  switch (prev) {
844  case FR_CONNECTION_STATE_INIT: /* Failed during initialisation */
845  case FR_CONNECTION_STATE_CONNECTED: /* Failed after connecting */
846  case FR_CONNECTION_STATE_CONNECTING: /* Failed during connecting */
847  case FR_CONNECTION_STATE_SHUTDOWN: /* Failed during shutdown */
848  if (conn->reconnection_delay) {
849  DEBUG2("Delaying reconnection by %pVs", fr_box_time_delta(conn->reconnection_delay));
850  if (fr_event_timer_in(conn, conn->pub.el, &conn->ev,
851  conn->reconnection_delay, _reconnect_delay_done, conn) < 0) {
852  /*
853  * Can happen when the event loop is exiting
854  */
855  PERROR("Failed inserting reconnection_delay timer event, halting connection");
856  connection_state_enter_halted(conn);
857  }
858  return;
859  }
860 
861  /*
862  * If there's no reconnection
863  * delay, then don't automatically
864  * reconnect, and wait to be
865  * signalled.
866  */
867  connection_state_enter_halted(conn);
868  break;
869 
870  case FR_CONNECTION_STATE_TIMEOUT: /* Failed during connecting due to timeout */
871  connection_state_enter_init(conn);
872  break;
873 
874  default:
875  fr_assert(0);
876  }
877 }

880  *
881  * The connection took took long to open. Timeout the attempt and transition
882  * to the failed state.
883  */
884 static void connection_state_enter_timeout(fr_connection_t *conn)

886  switch (conn->pub.state) {
887  case FR_CONNECTION_STATE_CONNECTING:
888  case FR_CONNECTION_STATE_SHUTDOWN:
889  break;
890 
891  default:
892  BAD_STATE_TRANSITION(FR_CONNECTION_STATE_TIMEOUT);

896

904 /** Enter the halted state
905  *
906  * Here we wait, until signalled by fr_connection_signal_reconnect.
907  */
908 static void connection_state_enter_halted(fr_connection_t *conn)
909 {
910  fr_assert(conn->is_closed);
911 
912  switch (conn->pub.state) {
913  case FR_CONNECTION_STATE_FAILED: /* Init failure */
914  case FR_CONNECTION_STATE_CLOSED:
915  break;
916 
917  default:
918  BAD_STATE_TRANSITION(FR_CONNECTION_STATE_HALTED);
919  }
920 
921  fr_event_timer_delete(&conn->ev);
922

924  WATCH_PRE(conn);

928 /** Enter the connected state
929  *
930  * The connection is now fully connected. At this point we call the open callback
931  * so that the API client can install its normal set of I/O callbacks to deal with
932  * sending/receiving actual data.
933  *
934  * After this, the connection will only transition states if an API client
935  * explicitly calls fr_connection_signal_reconnect.
936  *
937  * The connection API cannot monitor the connection for failure conditions.
938  *
939  * @param[in] conn Entering the connecting state.
940  */
941 static void connection_state_enter_connected(fr_connection_t *conn)
942 {
943  int ret;
944 
945  fr_assert(conn->pub.state == FR_CONNECTION_STATE_CONNECTING);
946 
947  STATE_TRANSITION(FR_CONNECTION_STATE_CONNECTED);
948 
949  fr_event_timer_delete(&conn->ev);
950  WATCH_PRE(conn);
951  if (conn->open) {
952  HANDLER_BEGIN(conn, conn->open);
953  DEBUG4("Calling open(el=%p, h=%p, uctx=%p)", conn->pub.el, conn->pub.h, conn->uctx);
954  ret = conn->open(conn->pub.el, conn->pub.h, conn->uctx);
955  HANDLER_END(conn);
956  } else {
957  ret = FR_CONNECTION_STATE_CONNECTED;
958  }
959 
960  switch (ret) {
961  /*
962  * Callback agrees everything is connected
963  */
964  case FR_CONNECTION_STATE_CONNECTED:
965  DEBUG2("Connection established");
966  WATCH_POST(conn); /* Only call if we successfully connected */
967  return;
968 
969  /*
970  * Open callback failed
971  */
972  case FR_CONNECTION_STATE_FAILED:
973  default:
974  PERROR("Connection failed");
975  connection_state_enter_failed(conn);
976  return;
977  }
978 }

980 /** Enter the connecting state
981  *
982  * After this function returns we wait to be signalled with fr_connection_singal_connected
983  * or for the connection timer to expire.
984  *
985  * @param[in] conn Entering the connecting state.
986  */
987 static void connection_state_enter_connecting(fr_connection_t *conn)
988 {
989  switch (conn->pub.state) {
990  case FR_CONNECTION_STATE_INIT:
991  break;
992 
993  default:
994  BAD_STATE_TRANSITION(FR_CONNECTION_STATE_CONNECTING);
995  return;
996  }
997 
998  STATE_TRANSITION(FR_CONNECTION_STATE_CONNECTING);
999 
1000  WATCH_PRE(conn);
1001  WATCH_POST(conn);
1002 
1003  /*
1004  * If there's a connection timeout,
1005  * set, then add the timer.
1006  */
1007  if (conn->connection_timeout) {
1008  if (fr_event_timer_in(conn, conn->pub.el, &conn->ev,
1009  conn->connection_timeout, _connection_timeout, conn) < 0) {
1010  PERROR("Failed setting connection_timeout event, failing connection");
1011 
1012  /*
1013  * This can happen when the event loop
1014  * is exiting.
1015  *
1016  * Entering fail will close partially
1017  * open connection and then, if we still
1018  * can't insert a timer, then the connection
1019  * will be halted and sit idle until its
1020  * freed.
1021  */
1022  connection_state_enter_failed(conn);
1023  }
1024  }
1025 }

1027 /** Initial state of the connection
1028  *
1029  * Calls the init function we were passed to allocate a library specific handle or
1030  * file descriptor.
1031  *
1032  * @param[in] conn To initialise.
1033  */
1034 static void connection_state_enter_init(fr_connection_t *conn)
1035 {
1036  fr_connection_state_t ret;
1037 
1038  switch (conn->pub.state) {
1039  case FR_CONNECTION_STATE_HALTED:
1040  case FR_CONNECTION_STATE_CLOSED:
1041  case FR_CONNECTION_STATE_FAILED:
1042  break;
1043 
1044  default:
1045  BAD_STATE_TRANSITION(FR_CONNECTION_STATE_INIT);
1046  return;
1047  }
1048 
1049  /*
1050  * Increment every time we enter
1051  * We have to do this, as we don't know
1052  * whether the connection was halted by
1053  * the failed callback, and is now being
1054  * reconnected, or was automatically
1055  * reconnected.
1056  */
1057  conn->pub.reconnected++;
1058 
1059  STATE_TRANSITION(FR_CONNECTION_STATE_INIT);
1060 
1061  /*
1062  * If we have an init callback, call it.
1063  */
1064  WATCH_PRE(conn);
1065  if (conn->init) {
1066  HANDLER_BEGIN(conn, conn->init);
1067  DEBUG4("Calling init(h_out=%p, conn=%p, uctx=%p)", &conn->pub.h, conn, conn->uctx);
1068  ret = conn->init(&conn->pub.h, conn, conn->uctx);
1069  HANDLER_END(conn);
1070  } else {
1071  ret = FR_CONNECTION_STATE_CONNECTING;
1072  }
1073 
1074  switch (ret) {
1075  case FR_CONNECTION_STATE_CONNECTING:
1076  conn->is_closed = false; /* We now have a handle */
1077  WATCH_POST(conn); /* Only call if we successfully initialised the handle */
1078  connection_state_enter_connecting(conn);
1079  return;
1080 
1081  /*
1082  * Initialisation callback failed
1083  */
1084  case FR_CONNECTION_STATE_FAILED:
1085  default:
1086  PERROR("Connection initialisation failed");
1087  connection_state_enter_failed(conn);
1088  break;
1089  }
1090 }

1147 /** Asynchronously signal the connection should be reconnected
1148  *
1149  * Should be called if the caller has knowledge that the connection is bad
1150  * and should be reconnected.
1151  *
1152  * @param[in] conn to reconnect.
1153  * @param[in] reason Why the connection was signalled to reconnect.
1154  */
1155 void fr_connection_signal_reconnect(fr_connection_t *conn, fr_connection_reason_t reason)
1156 {
1157  DEBUG2("Signalled to reconnect from %s state",
1158  fr_table_str_by_value(fr_connection_states, conn->pub.state, "<INVALID>"));
1159 
1160  if (DEFER_SIGNALS(conn)) {
1161  if ((reason == FR_CONNECTION_EXPIRED) && conn->shutdown) {
1162  connection_deferred_signal_add(conn, CONNECTION_DSIGNAL_RECONNECT_EXPIRED);
1163  return;
1164  }
1165 
1166  connection_deferred_signal_add(conn, CONNECTION_DSIGNAL_RECONNECT_FAILED);
1167  return;
1168  }
1169 
1170  switch (conn->pub.state) {
1171  case FR_CONNECTION_STATE_CLOSED: /* Don't circumvent reconnection_delay */
1172  case FR_CONNECTION_STATE_INIT: /* Already initialising */
1173  break;
1174 
1175  case FR_CONNECTION_STATE_HALTED:
1176  fr_connection_signal_init(conn);
1177  break;
1178 
1179  case FR_CONNECTION_STATE_SHUTDOWN:
1180  if (reason == FR_CONNECTION_EXPIRED) break; /* Already shutting down */
1181  connection_state_enter_failed(conn);
1182  break;
1183 
1184  case FR_CONNECTION_STATE_CONNECTED:
1185  if (reason == FR_CONNECTION_EXPIRED) {
1186  if (conn->shutdown) {
1187  connection_state_enter_shutdown(conn);
1188  break;
1189  }
1190  connection_state_enter_closed(conn);
1191  break;
1192  }
1193  FALL_THROUGH;
1194 
1195  case FR_CONNECTION_STATE_CONNECTING:
1196  case FR_CONNECTION_STATE_TIMEOUT:
1197  case FR_CONNECTION_STATE_FAILED:
1198  connection_state_enter_failed(conn);
1199  break;
1200 
1201  case FR_CONNECTION_STATE_MAX:
1202  fr_assert(0);
1203  return;
1204  }
1205 }

1207 /** Shuts down a connection gracefully
1208  *
1209  * If a shutdown function has been provided, it is called.
1210  * It's then up to the shutdown function to install I/O handlers to signal
1211  * when the connection has finished shutting down and should be closed
1212  * via #fr_connection_signal_halt.
1213  *
1214  * @param[in] conn to shutdown.
1215  */
1216 void fr_connection_signal_shutdown(fr_connection_t *conn)
1217 {
1218  DEBUG2("Signalled to shutdown from %s state",
1219  fr_table_str_by_value(fr_connection_states, conn->pub.state, "<INVALID>"));
1220 
1221  if (DEFER_SIGNALS(conn)) {
1222  connection_deferred_signal_add(conn, CONNECTION_DSIGNAL_SHUTDOWN);
1223  return;
1224  }
1225 
1226  switch (conn->pub.state) {
1227  case FR_CONNECTION_STATE_HALTED:
1228  case FR_CONNECTION_STATE_SHUTDOWN:
1229  break;
1230 
1231  case FR_CONNECTION_STATE_INIT:
1232  connection_state_enter_halted(conn);
1233  break;
1234 
1235  /*
1236  * If the connection is connected it needs to be
1237  * shutdown first.
1238  *
1239  * The shutdown callback or an FD event it inserts then
1240  * to signal that the connection should be closed.
1241  */
1242  case FR_CONNECTION_STATE_CONNECTED:
1243  if (conn->shutdown) {
1244  connection_state_enter_shutdown(conn);
1245  break;
1246  }
1247  FALL_THROUGH;
1248 
1249  /*
1250  * If the connection is any of these states it
1251  * must have completed INIT which means it has
1252  * an active handle which needs to be closed before
1253  * the connection is halted.
1254  */
1255  case FR_CONNECTION_STATE_CONNECTING:
1256  case FR_CONNECTION_STATE_TIMEOUT:
1257  case FR_CONNECTION_STATE_FAILED:
1258  connection_state_enter_closed(conn);
1259  fr_assert(conn->is_closed);
1260 
1261  FALL_THROUGH;
1262  case FR_CONNECTION_STATE_CLOSED:
1263  connection_state_enter_halted(conn);
1264  break;
1265 
1266  case FR_CONNECTION_STATE_MAX:
1267  fr_assert(0);
1268  return;
1269  }
1270 }

1272 /** Shuts down a connection ungracefully
1273  *
1274  * If a connection is in an open or connection state it will be closed immediately.
1275  * Otherwise the connection will transition directly to the halted state.
1276  *
1277  * @param[in] conn to halt.
1278  */
1279 void fr_connection_signal_halt(fr_connection_t *conn)
1280 {
1281  DEBUG2("Signalled to halt from %s state",
1282  fr_table_str_by_value(fr_connection_states, conn->pub.state, "<INVALID>"));
1283 
1284  if (DEFER_SIGNALS(conn)) {
1285  connection_deferred_signal_add(conn, CONNECTION_DSIGNAL_HALT);
1286  return;
1287  }
1288 
1289  switch (conn->pub.state) {
1290  case FR_CONNECTION_STATE_HALTED:
1291  break;
1292 
1293  case FR_CONNECTION_STATE_INIT:
1294  case FR_CONNECTION_STATE_CLOSED:
1295  connection_state_enter_halted(conn);
1296  break;
1297 
1298  /*
1299  * If the connection is any of these states it
1300  * must have completed INIT which means it has
1301  * an active handle which needs to be closed before
1302  * the connection is halted.
1303  */
1304  case FR_CONNECTION_STATE_CONNECTED:
1305  case FR_CONNECTION_STATE_CONNECTING:
1306  case FR_CONNECTION_STATE_SHUTDOWN:
1307  case FR_CONNECTION_STATE_TIMEOUT:
1308  case FR_CONNECTION_STATE_FAILED:
1309  connection_state_enter_closed(conn);
1310  fr_assert(conn->is_closed);
1311  connection_state_enter_halted(conn);
1312  break;
1313 
1314  case FR_CONNECTION_STATE_MAX:
