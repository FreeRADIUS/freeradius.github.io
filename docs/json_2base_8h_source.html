89 /** Value formatting options for fr_json_afrom_pair_list()
90  *
91  * Controls how values are formatted in JSON documents
92  * produced from fr_json_afrom_pair_list().
93  *
94  * Not all these options are valid for all output modes.
95  * @see fr_json_format_verify(), fr_json_format_s
96  *
97  *
98  * If an attribute appears only once then the value will normally
99  * be written as an object. When an attribute appears more than
100  * once then the values will be added as an array instead. Setting
101  * **value_as_array** will ensure that values are always written as
102  * an array, even if containing only a single entry.
103  *
104  * Example with output_mode `JSON_MODE_OBJECT_SIMPLE` and `value_as_array` is false:
105 @verbatim
106 {"User-Name":"john","Filter-Id":["f1","f2"]}
107 @endverbatim
108  *
109  * Example with output_mode `JSON_MODE_OBJECT_SIMPLE` and `value_as_array` is true:
110 @verbatim
111 {"User-Name":["john"],"Filter-Id":["f1","f2"]}
112 @endverbatim
113  *
114  *
115  * Set **enum_as_int** to write enumerated values in their integer form.
116  *
117  * When false, the string form is output:
118 @verbatim
119 {"Service-Type":{"type":"uint32","value":"Login-User"}}
120 @endverbatim
121  *
122  * When true, the integer is output:
123 @verbatim
124 {"Service-Type":{"type":"uint32","value":1}}
125 @endverbatim
126  *
127  *
128  * Numeric data types will usually be written to the JSON document
129  * as numbers. **always_string** ensures that all values are written as
130  * strings:
131  *
132  * Example when `always_string` is false:
133 @verbatim
134 {"NAS-Port":{"type":"uint32","value":999}}
135 @endverbatim
136  *
137  * Example when `always_string` is true:
138 @verbatim
139 {"NAS-Port":{"type":"uint32","value":"999"}}
140 @endverbatim
141  *
142  */
143 typedef struct {
144  bool value_as_array; //!< Use JSON array for multiple attribute values.
145  bool enum_as_int; //!< Output enums as value, not their string representation.
146  bool always_string; //!< Output all data types as strings.
147 } fr_json_format_value_t;

152  * These options control the format of JSON document which is

154  *
155  * The **output_mode** determines the format of JSON that is created:
156  *
157  * When JSON_MODE_OBJECT:
158 @verbatim
159 {
160  "<attribute0>": {
161  "type":"<str-type0>",
162  "value":["value0"]
163  },
164  "<attribute1>": {
165  "type":"<str-type1>",
166  "value":["value1.0", "value1.1"]
167  },
168  "<attribute2>": {
169  "type":"<int-type2>",
170  "value":[2]
171  },
172 }
173 @endverbatim
174  *
175  * When JSON_MODE_OBJECT_SIMPLE:
176 @verbatim
177 {
178  "<attribute0>":"<value0>",
179  "<attribute1>":["<value1.0>","<value1.1>"],
180  "<attribute2>":2
181 }
182 @endverbatim
183  *
184  * When JSON_MODE_ARRAY:
185 @verbatim
186 [
187  {"name":"<attribute0>","type":"<str-type0>","value":"<value0>"},
188  {"name":"<attribute1>","type":"<str-type1>","value":"<value1.0>"},
189  {"name":"<attribute1>","type":"<str-type1>","value":"<value1.1>"},
190  {"name":"<attribute2>","type":"<int-type2>","value":2}
191 ]
192 @endverbatim
193  *
194  * When JSON_MODE_ARRAY_OF_VALUES:
195 @verbatim
196 [
197  <value0>,
198  <value1.0>,
199  <value1.1>,
200  <value2>
201 ]
202 @endverbatim
203  *
204  * When JSON_MODE_ARRAY_OF_NAMES:
205 @verbatim
206 [
207  <attribute0>,
208  <attribute1>,

211 ]

213  *
214  */
