34 static unlang_action_t unlang_load_balance_next(rlm_rcode_t *p_result, request_t *request)
35 {
36  unlang_stack_t *stack = request->stack;
37  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
38  unlang_t *instruction = frame->instruction;
39  unlang_frame_state_redundant_t *redundant;
40 
41  unlang_group_t *g;
42 
43  g = unlang_generic_to_group(instruction);
44 
45  redundant = talloc_get_type_abort(frame->state, unlang_frame_state_redundant_t);
46 
47 #ifdef __clang_analyzer__
48  if (!redundant->found) {
49  *p_result = RLM_MODULE_FAIL;
50  return UNLANG_ACTION_CALCULATE_RESULT;
51  }
52 #endif
53  /*
54  * Set up the first round versus subsequent ones.
55  */
56  if (!redundant->child) {
57  redundant->child = redundant->found;
58 
59  } else {
60  /*
61  * child is NULL on the first pass. But if it's
62  * back to the found one, then we're done.
63  */
64  if (redundant->child == redundant->found) {
65  /* DON'T change p_result, as it is taken from the child */
66  return UNLANG_ACTION_CALCULATE_RESULT;
67  }
68 
69  RDEBUG4("%s resuming", frame->instruction->debug_name);
70 
71  /*
72  * We are in a resumed frame. The module we
73  * chose failed, so we have to go through the
74  * process again.
75  */
76 
77  fr_assert(instruction->type != UNLANG_TYPE_LOAD_BALANCE); /* this is never called again */
78 
79  /*
80  * If the current child says "return", then do
81  * so.
82  */
83  if (redundant->child->actions[*p_result] == MOD_ACTION_RETURN) {
84  /* DON'T change p_result, as it is taken from the child */
85  return UNLANG_ACTION_CALCULATE_RESULT;
86  }
87  }
88 
89  /*
90  * Push the child, and yield for a later return.
91  */
92  if (unlang_interpret_push(request, redundant->child, frame->result, UNLANG_NEXT_STOP, UNLANG_SUB_FRAME) < 0) {
93  *p_result = RLM_MODULE_FAIL;
94  return UNLANG_ACTION_STOP_PROCESSING;
95  }
96 
97  /*
98  * Now that we've pushed this child, make the next call
99  * use the next child, wrapping around to the beginning.
100  *
101  * @todo - track the one we chose, and if it fails, do
102  * the load-balancing again, except this time skipping
103  * the failed module. AND, keep track of multiple failed
104  * modules in the unlang_frame_state_redundant_t
105  * structure.
106  */
107  redundant->child = redundant->child->next;
108  if (!redundant->child) redundant->child = g->children;
109 
110  repeatable_set(frame);
111 
112  return UNLANG_ACTION_PUSHED_CHILD;
113 }

115 static unlang_action_t unlang_load_balance(rlm_rcode_t *p_result, request_t *request)
116 {
117  unlang_stack_t *stack = request->stack;
118  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
119  unlang_t *instruction = frame->instruction;
120  unlang_frame_state_redundant_t *redundant;
121 
122  unlang_group_t *g;
123  unlang_load_balance_t *gext = NULL;
124 
125  uint32_t count = 0;
126 
127  g = unlang_generic_to_group(instruction);
128  if (!g->num_children) {
129  *p_result = RLM_MODULE_NOOP;
130  return UNLANG_ACTION_CALCULATE_RESULT;
131  }
132 
133  gext = unlang_group_to_load_balance(g);
134 
135  RDEBUG4("%s setting up", frame->instruction->debug_name);
136 
137  redundant = talloc_get_type_abort(frame->state, unlang_frame_state_redundant_t);
138 
139  if (gext && gext->vpt) {
140  uint32_t hash, start;
141  ssize_t slen;
142  char const *p = NULL;
143  char buffer[1024];
144 
145  /*
146  * Integer data types let the admin
147  * select which frame is being used.
148  */
149  if (tmpl_is_attr(gext->vpt) &&
150  ((tmpl_da(gext->vpt)->type == FR_TYPE_UINT8) ||
151  (tmpl_da(gext->vpt)->type == FR_TYPE_UINT16) ||
152  (tmpl_da(gext->vpt)->type == FR_TYPE_UINT32) ||
153  (tmpl_da(gext->vpt)->type == FR_TYPE_UINT64))) {
154  fr_pair_t *vp;
155 
156  slen = tmpl_find_vp(&vp, request, gext->vpt);
157  if (slen < 0) {
158  REDEBUG("Failed finding attribute %s", gext->vpt->name);
159  goto randomly_choose;
160  }
161 
162  switch (tmpl_da(gext->vpt)->type) {
163  case FR_TYPE_UINT8:
164  start = ((uint32_t) vp->vp_uint8) % g->num_children;
165  break;
166 
167  case FR_TYPE_UINT16:
168  start = ((uint32_t) vp->vp_uint16) % g->num_children;
169  break;
170 
171  case FR_TYPE_UINT32:
172  start = vp->vp_uint32 % g->num_children;
173  break;
174 
175  case FR_TYPE_UINT64:
176  start = (uint32_t) (vp->vp_uint64 % ((uint64_t) g->num_children));
177  break;
178 
179  default:
180  goto randomly_choose;
181  }
182 
183  } else {
184  slen = tmpl_expand(&p, buffer, sizeof(buffer), request, gext->vpt, NULL, NULL);
185  if (slen < 0) {
186  REDEBUG("Failed expanding template");
187  goto randomly_choose;
188  }
189 
190  hash = fr_hash(p, slen);
191 
192  start = hash % g->num_children;;
193  }
194 
195  RDEBUG3("load-balance starting at child %d", (int) start);
196 
197  count = 0;
198  for (redundant->child = redundant->found = g->children;
199  redundant->child != NULL;
200  redundant->child = redundant->child->next) {
201  count++;
202  if (count == start) {
203  redundant->found = redundant->child;
204  break;
205  }
206  }
207 
208  } else {
209  randomly_choose:
210  count = 0;
211 
212  /*
213  * Choose a child at random.
214  *
215  * @todo - leverage the "power of 2", as per
216  * lib/io/network.c. This is good enough for
217  * most purposes. However, in order to do this,
218  * we need to track active callers across
219  * *either* multiple modules in one thread, *or*
220  * across multiple threads.
221  *
222  * We don't have thread-specific instance data
223  * for this load-balance section. So for now,
224  * just pick a random child.
225  */
226  for (redundant->child = redundant->found = g->children;
227  redundant->child != NULL;
228  redundant->child = redundant->child->next) {
229  count++;
230 
231  if ((count * (fr_rand() & 0xffffff)) < (uint32_t) 0x1000000) {
232  redundant->found = redundant->child;
233  }
234  }
235  }
236 
237  /*
238  * Plain "load-balance". Just do one child.
239  */
240  if (instruction->type == UNLANG_TYPE_LOAD_BALANCE) {
241  if (unlang_interpret_push(request, redundant->found,
242  frame->result, UNLANG_NEXT_STOP, UNLANG_SUB_FRAME) < 0) {
243  *p_result = RLM_MODULE_FAIL;
244  return UNLANG_ACTION_STOP_PROCESSING;
245  }
246  return UNLANG_ACTION_PUSHED_CHILD;
247  }
248 
249  /*
250  * "redundant" and "redundant-load-balance" starts at
251  * "found", but we need to indicate that we're at the
252  * first child.
253  */
254  redundant->child = NULL;
255 
256  frame->process = unlang_load_balance_next;
257  return unlang_load_balance_next(p_result, request);
258 }
