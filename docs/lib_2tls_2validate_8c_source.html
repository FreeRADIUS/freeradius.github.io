42 /** Validates a certificate using custom logic
43  *
44  * Before trusting a certificate, we make sure that the certificate is
45  * 'valid'. There are several checks we perform to verify its validity.
46  *
47  * 1. Verify the certificate's signature, and verifying that the certificate has
48  * been issued by a trusted Certificate Authority (this is done for us by OpenSSL).
49  *
50  * 2. Verify that the certificate is valid for the present date (i.e. it is being
51  * presented within its validity dates).
52  *
53  * 3. Verify that the certificate has not been revoked by its issuing Certificate
54  * Authority, by checking with respect to a Certificate Revocation List (CRL).
55  *
56  * 4. Verify that the credentials presented by the certificate fulfill additional
57  * requirements specific to the application, such as with respect to access control
58  * lists or with respect to OCSP (Online Certificate Status Processing).
59  *
60  * @note This callback will be called multiple times based on the depth of the root
61  * certificate chain.
62  *
63  * @note As a byproduct of validation, various OIDs will be extracted from the
64  * certificates, and inserted into the session-state: list as fr_pair_t.
65  *
66  * @param ok preverify ok. 1 if true, 0 if false.
67  * @param x509_ctx containing certs to verify.
68  * @return
69  * - 0 if not valid.
70  * - 1 if valid.
71  */
72 int fr_tls_validate_cert_cb(int ok, X509_STORE_CTX *x509_ctx)
73 {
74  X509 *cert;
75  SSL *ssl;
76  fr_tls_session_t *tls_session;
77  int err, depth;
78  fr_tls_conf_t *conf;
79  int my_ok = ok;
80 
81  fr_pair_t *cert_vps = NULL;
82  fr_cursor_t cursor;
83 
84  char const **identity_p;
85  char const *identity = NULL;
86 
87  char subject[1024];
88  char common_name[1024];
89  char issuer[1024];
90 
91  request_t *request;
92 
93  cert = X509_STORE_CTX_get_current_cert(x509_ctx);
94  err = X509_STORE_CTX_get_error(x509_ctx);
95  depth = X509_STORE_CTX_get_error_depth(x509_ctx);
96 
97  /*
98  * Retrieve the pointer to the SSL of the connection currently treated
99  * and the application specific data stored into the SSL object.
100  */
101  ssl = X509_STORE_CTX_get_ex_data(x509_ctx, SSL_get_ex_data_X509_STORE_CTX_idx());
102  conf = talloc_get_type_abort(SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_CONF), fr_tls_conf_t);
103  tls_session = talloc_get_type_abort(SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_TLS_SESSION), fr_tls_session_t);
104  request = talloc_get_type_abort(SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_REQUEST), request_t);
105 
106  identity_p = SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_IDENTITY);
107  if (identity_p && *identity_p) identity = talloc_get_type_abort_const(*identity_p, char);
108 
109  if (RDEBUG_ENABLED3) {
110  STACK_OF(X509) *our_chain = X509_STORE_CTX_get_chain(x509_ctx);
111  int i;
112 
113 #if OPENSSL_VERSION_NUMBER >= 0x10100000L
114  RDEBUG3("Certificate chain - %i cert(s) untrusted", X509_STORE_CTX_get_num_untrusted(x509_ctx));
115 #else
116  RDEBUG3("Certificate chain");
117 #endif
118 
119  for (i = sk_X509_num(our_chain); i > 0 ; i--) {
120  X509 *this_cert = sk_X509_value(our_chain, i - 1);
121 
122  X509_NAME_oneline(X509_get_subject_name(this_cert), subject, sizeof(subject));
123  subject[sizeof(subject) - 1] = '\0';
124 
125  RDEBUG3("%s [%i] %s", this_cert == cert ? ">" : " ", i - 1, subject);
126  }
127  }
128 
129  /*
130  * For this next bit, we create the attributes *only* if
131  * we're at the client or issuing certificate, AND we
132  * have a user identity. i.e. we don't create the
133  * attributes for RadSec connections.
134  *
135  * We do not repopulate the attribute list with cert
136  * attributes as these should have been cached previously.
137  *
138  * If we do not have a copy of the cert that issued the
139  * client's cert in SSL_CTX's X509_STORE.
140  */
141  if (identity && (depth <= 1) && !SSL_session_reused(ssl)) {
142  fr_cursor_init(&cursor, &cert_vps);
143  fr_tls_session_pairs_from_x509_cert(&cursor, request, tls_session, cert, depth);
144 
145  /*
146  * Add a copy of the cert_vps to session state.
147  *
148  * Both PVS studio and Coverity detect the condition
149  * below as logically dead code unless we explicitly
150  * set cert_vps. This is because they're too dumb
151  * to realise that the cursor argument passed to
152  * fr_tls_session_pairs_from_x509_cert contains a
153  * reference to cert_vps.
154  */
155  cert_vps = fr_cursor_head(&cursor);
156  if (cert_vps) {
157  RDEBUG2("Adding certificate attributes to session-state");
158 
159  /*
160  * Print out all the pairs we have so far
161  */
162  log_request_pair_list(L_DBG_LVL_2, request, cert_vps, "&session-state.");
163 
164  /*
165  * cert_vps have a different talloc parent, so we
166  * can't just reference them.
167  */
168  MEM(fr_pair_list_copy(request->state_ctx, &request->state, &cert_vps) >= 0);
169  fr_pair_list_free(&cert_vps);
170  }
171  }
172 
173  /*
174  * Get the Subject & Issuer
175  */
176  subject[0] = issuer[0] = '\0';
177  X509_NAME_oneline(X509_get_subject_name(cert), subject, sizeof(subject));
178  subject[sizeof(subject) - 1] = '\0';
179 
180  X509_NAME_oneline(X509_get_issuer_name(cert), issuer, sizeof(issuer));
181  issuer[sizeof(issuer) - 1] = '\0';
182 
183  /*
184  * Get the Common Name, if there is a subject.
185  */
186  X509_NAME_get_text_by_NID(X509_get_subject_name(cert),
187  NID_commonName, common_name, sizeof(common_name));
188  common_name[sizeof(common_name) - 1] = '\0';
189 
190  /*
191  * If the CRL has expired, that might still be OK.
192  */
193  if (!my_ok && (conf->allow_expired_crl) && (err == X509_V_ERR_CRL_HAS_EXPIRED)) {
194  my_ok = 1;
195  X509_STORE_CTX_set_error(x509_ctx, 0);
196  }
197 
198  if (!my_ok) {
199  char const *p = X509_verify_cert_error_string(err);
200  RERROR("TLS error: %s (%i)", p, err);
201  return my_ok;
202  }
203 
204  switch (X509_STORE_CTX_get_error(x509_ctx)) {
205  case X509_V_ERR_UNABLE_TO_GET_ISSUER_CERT:
206  RERROR("issuer=%s", issuer);
207  break;
208 
209  case X509_V_ERR_CERT_NOT_YET_VALID:
210  case X509_V_ERR_ERROR_IN_CERT_NOT_BEFORE_FIELD:
211  RERROR("notBefore=");
212 #if 0
213  ASN1_TIME_print(bio_err, X509_get_notBefore(x509_ctx->current_cert));
214 #endif
215  break;
216 
217  case X509_V_ERR_CERT_HAS_EXPIRED:
218  case X509_V_ERR_ERROR_IN_CERT_NOT_AFTER_FIELD:
219  RERROR("notAfter=");
220 #if 0
221  ASN1_TIME_print(bio_err, X509_get_notAfter(x509_ctx->current_cert));
222 #endif
223  break;
224  }
225 
226  /*
227  * Stop checking if this is an intermediary.
228  *
229  * Client certificates get better OCSP checks.
230  */
231  if (depth > 0) {
232  RDEBUG2("[verify chain] = %s", my_ok ? "ok" : "invalid");
233  return my_ok;
234  }
235 
236  /*
237  * If the conf tells us to, check cert issuer
238  * against the specified value and fail
239  * verification if they don't match.
240  */
241  if (conf->check_cert_issuer && (strcmp(issuer, conf->check_cert_issuer) != 0)) {
242  REDEBUG("Certificate issuer (%s) does not match specified value (%s)!",
243  issuer, conf->check_cert_issuer);
244  my_ok = 0;
245  }
246 
247  /*
248  * If the conf tells us to, check the CN in the
249  * cert against xlat'ed value, but only if the
250  * previous checks passed.
251  */
252  if (my_ok && conf->check_cert_cn) {
253  char cn_str[1024];
254 
255  if (xlat_eval(cn_str, sizeof(cn_str), request, conf->check_cert_cn, NULL, NULL) < 0) {
256  /* if this fails, fail the verification */
257  my_ok = 0;
258  } else {
259  RDEBUG2("checking certificate CN (%s) with xlat'ed value (%s)", common_name, cn_str);
260  if (strcmp(cn_str, common_name) != 0) {
261  REDEBUG("Certificate CN (%s) does not match specified value (%s)!",
262  common_name, cn_str);
263  my_ok = 0;
264  }
265  }
266  } /* check_cert_cn */
267 
268  while (conf->verify_client_cert_cmd) {
269  char filename[256];
270  int fd;
271  FILE *fp;
272  fr_pair_t *vp;
273 
274  snprintf(filename, sizeof(filename), "%s/client.XXXXXXXX", conf->verify_tmp_dir);
275 
276 #ifdef __COVERITY__
277  /*
278  * POSIX-2008 requires that mkstemp creates the file
279  * with 0600 permissions. So setting umask is pointless
280  * and although it won't cause crashes, will cause
281  * race conditions in threaded environments.
282  */
283  umask(0600);
284 #endif
285  fd = mkstemp(filename);
286  if (fd < 0) {
287  RDEBUG2("Failed creating file in %s: %s",
288  conf->verify_tmp_dir, fr_syserror(errno));
289  break;
290  }
291 
292  fp = fdopen(fd, "w");
293  if (!fp) {
294  close(fd);
295  REDEBUG("Failed opening file \"%s\": %s", filename, fr_syserror(errno));
296  break;
297  }
298 
299  if (!PEM_write_X509(fp, cert)) {
300  fclose(fp);
301  REDEBUG("Failed writing certificate to file");
302  goto do_unlink;
303  }
304  fclose(fp);
305 
306  MEM(pair_update_request(&vp, attr_tls_client_cert_filename) >= 0);
307  fr_pair_value_strdup(vp, filename);
308 
309  RDEBUG2("Verifying client certificate with cmd");
310  if (radius_exec_program(request, NULL, 0, NULL, request, conf->verify_client_cert_cmd,
311  request->request_pairs, true, true, fr_time_delta_from_sec(EXEC_TIMEOUT)) != 0) {
312  REDEBUG("Client certificate CN \"%s\" failed external verification", common_name);
313  my_ok = 0;
314  } else {
315  RDEBUG2("Client certificate CN \"%s\" passed external validation", common_name);
316  }
317 
318  do_unlink:
319  unlink(filename);
320  break;
321  }
322 
323 #ifdef HAVE_OPENSSL_OCSP_H
324  /*
325  * Do OCSP last, so we have the complete set of attributes
326  * available for the virtual server.
327  *
328  * Fixme: Do we want to store the matching TLS-Client-cert-Filename?
329  */
330  if (my_ok && conf->ocsp.enable){
331  X509 *issuer_cert;
332 
333  RDEBUG2("Starting OCSP Request");
334 
335  /*
336  * If we don't have an issuer, then we can't send
337  * and OCSP request, but pass the NULL issuer in
338  * so fr_tls_ocsp_check can decide on the correct
339  * return code.
340  */
341  issuer_cert = X509_STORE_CTX_get0_current_issuer(x509_ctx);
342  my_ok = fr_tls_ocsp_check(request, ssl, conf->ocsp.store, issuer_cert, cert, &(conf->ocsp), false);
343  }
344 #endif
345 
346  RDEBUG2("[verify client] = %s", my_ok ? "ok" : "invalid");
347  return my_ok;
348 }

350 /** Revalidates the client's certificate chain
351  *
352  * Wraps the fr_tls_validate_cert_cb callback, allowing us to use the same
353  * validation logic whenever we need to.
354  *
355  * @note Only use so far is forcing the chain to be re-validated on session
356  * resumption.
357  *
358  * @return
359  * - 1 if the chain could be validated.
360  * - 0 if the chain failed validation.
361  */
362 int fr_tls_validate_client_cert_chain(SSL *ssl)
363 {
364  int err;
365  int verify;
366  int ret = 1;
367 
368  STACK_OF(X509) *chain;
369  X509 *cert;
370  X509_STORE *store;
371  X509_STORE_CTX *store_ctx;
372 
373  request_t *request;
374 
375  request = talloc_get_type_abort(SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_REQUEST), request_t);
376 
377  /*
378  * If there's no client certificate, we just return OK.
379  */
380  cert = SSL_get_peer_certificate(ssl); /* Increases ref count */
381  if (!cert) return 1;
382 
383  store_ctx = X509_STORE_CTX_new();
384  chain = SSL_get_peer_cert_chain(ssl); /* Does not increase ref count */
385  store = SSL_CTX_get_cert_store(SSL_get_SSL_CTX(ssl)); /* Does not increase ref count */
386 
387  X509_STORE_CTX_init(store_ctx, store, cert, chain);
388  X509_STORE_CTX_set_ex_data(store_ctx, SSL_get_ex_data_X509_STORE_CTX_idx(), ssl);
389  X509_STORE_CTX_set_verify_cb(store_ctx, fr_tls_validate_cert_cb);
390 
391  verify = X509_verify_cert(store_ctx);
392  if (verify != 1) {
393  err = X509_STORE_CTX_get_error(store_ctx);
394 
395  if (err != X509_V_OK) {
396  REDEBUG("Failed re-validating resumed session: %s", X509_verify_cert_error_string(err));
397  ret = 0;
398  }
399  }
400 
401  X509_free(cert);
402  X509_STORE_CTX_free(store_ctx);
403 
404  return ret;
405 }
