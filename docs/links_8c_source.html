63 void mtp_link_up(struct mtp_link *link)
64 {
65  int one_up;
66 
67  if (link->blocked) {
68  LOGP(DINP, LOGL_ERROR,
69  "Ignoring link up on blocked link %d/%s of linkset %d/%s.\n",
70  link->nr, link->name, link->set->nr, link->set->name);
71  return;
72  }
73 
74  one_up = is_one_up(link->set);
75  link->available = 1;
76  link->was_up = 0;
77 
78  mtp_link_set_init_slc(link->set);
79  if (!one_up)
80  mtp_linkset_up(link->set);
81  else
82  mtp_link_start_link_test(link);
83 }

110 int link_init(struct bsc_data *bsc, struct mtp_link_set *set)
111 {
112  int i;
113  struct mtp_udp_link *lnk;
114  struct mtp_link *blnk;
115 
116 
117  if (!bsc->udp_src_port) {
118  LOGP(DINP, LOGL_ERROR, "You need to set a UDP address.\n");
119  return -1;
120  }
121 
122  LOGP(DINP, LOGL_NOTICE, "Using UDP MTP mode.\n");
123 
124  if (link_global_init(&bsc->udp_data) != 0)
125  return -1;
126 
127  if (link_global_bind(&bsc->udp_data, bsc->udp_src_port) != 0)
128  return -1;
129 
130  for (i = 1; i <= bsc->udp_nr_links; ++i) {
131  blnk = mtp_link_alloc(set);
132  lnk = mtp_udp_link_init(blnk);
133 
134  lnk->link_index = i;
135 
136  /* now connect to the transport */
137  if (snmp_mtp_peer_name(lnk->session, bsc->udp_ip) != 0)
138  return -1;
139 
140  if (link_udp_init(lnk, bsc->udp_ip, bsc->udp_port) != 0)
141  return -1;
142  }
143 
144  return 0;
145 }
