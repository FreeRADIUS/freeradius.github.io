156 void handle_options(int argc, char **argv)
157 {
158  while (1) {
159  int option_index = 0, c;
160  static struct option long_options[] = {
161  {"help", 0, 0, 'h'},
162  {"config", 1, 0, 'c'},
163  {"pcap", 1, 0, 'p'},
164  {"version", 0, 0, 0},
165  {0, 0, 0, 0},
166  };
167 
168  c = getopt_long(argc, argv, "hc:p:v",
169  long_options, &option_index);
170  if (c == -1)
171  break;
172 
173  switch (c) {
174  case 'h':
175  print_usage(argv[0]);
176  print_help();
177  fr_exit_now(0);
178  case 'p':
179  if (bsc->pcap_fd >= 0)
180  close(bsc->pcap_fd);
181  bsc->pcap_fd = open(optarg, O_WRONLY | O_TRUNC | O_CREAT, S_IRUSR | S_IWUSR | S_IRGRP| S_IROTH);
182  if (bsc->pcap_fd < 0) {
183  fprintf(stderr, "Failed to open PCAP file.\n");
184  fr_exit_now(0);
185  }
186  mtp_pcap_write_header(bsc->pcap_fd);
187  break;
188  case 'c':
189  config = optarg;
190  break;
191  case 'v':
192  printf("This is %s version %s.\n", PACKAGE, VERSION);
193  fr_exit_now(0);
194  break;
195  default:
196  fprintf(stderr, "Unknown option.\n");
197  break;
198  }
199  }
200 
201  signal(SIGPIPE, SIG_IGN);
202  signal(SIGINT, sigint);
203  signal(SIGUSR1, sigusr1);
204  signal(SIGUSR2, sigusr2);
205 }
