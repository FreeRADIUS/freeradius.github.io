83 /** Check the proto value is sane/supported
84  *
85  * @param[in] proto to check
86  * @return
87  * - true if it is.
88  * - false if it's not.
89  */
90 static inline bool fr_socket_is_inet(int proto)
91 {
92  /*
93  * Check the protocol is sane
94  */
95  switch (proto) {
96  case IPPROTO_UDP:
97  case IPPROTO_TCP:
98 #ifdef IPPROTO_SCTP
99  case IPPROTO_SCTP:
100 #endif
101  return true;
102 
103  default:
104  fr_strerror_printf("Unknown IP protocol %d", proto);
105  return false;
106  }
107 }

115 /** Swap src/dst information of a fr_socket_t
116  *
117  * @param[out] dst Where to write the swapped addresses. May be the same as src.
118  * @param[in] src Socket address to swap.
119  */
120 static inline void fr_socket_addr_swap(fr_socket_t *dst, fr_socket_t const *src)
121 {
122  fr_socket_t tmp = *src;
123 
124  if (dst != src) *dst = tmp; /* copy non-address fields over */
125 
126  switch (src->proto) {
127  case IPPROTO_UDP:
128  case IPPROTO_TCP:
129 #ifdef IPPROTO_SCTP
130  case IPPROTO_SCTP:
131 #endif
132  dst->inet.dst_ipaddr = tmp.inet.src_ipaddr;
133  dst->inet.dst_port = tmp.inet.src_port;
134  dst->inet.src_ipaddr = tmp.inet.dst_ipaddr;
135  dst->inet.src_port = tmp.inet.dst_port;
136  break;
137 
138  default:
139  return;
140  }
141 }

143 /** Initialise a fr_socket_t for connecting to a remote host using a specific src interface, address and port
144  *
145  * Can also be used to record information from an incoming packet so that we can
146  * identify the correct return path later.
147  *
148  * @param[out] addr to initialise.
149  * @param[in] proto one of the IPPROTO_* macros, i.e. IPPROTO_TCP, IPPROTO_UDP
150  * @param[in] ifindex The interface to originate the packet from Pass <= 0 to
151  * indicate an unknown or unspecified interface.
152  * @param[in] src_ipaddr The source IP address of the packet, or source interface for
153  * packets to egress out of.
154  * @param[in] src_port The source port of the packet or the source
155  * @param[in] dst_ipaddr The destination IP address of the packet.
156  * @param[in] dst_port The destination port of the packet.
157  * @return
158  * - NULL if invalid parameters are provided.
159  * - An initialised fr_socket_t struct.
160  */
161 static inline fr_socket_t *fr_socket_addr_init_inet(fr_socket_t *addr,
162  int proto,
163  int ifindex, fr_ipaddr_t const *src_ipaddr, int src_port,
164  fr_ipaddr_t const *dst_ipaddr, int dst_port)
165 {
166  if (!fr_socket_is_inet(proto)) return NULL;
167 
168  *addr = (fr_socket_t){
169  .proto = proto,
170  .inet = {
171  .ifindex = ifindex,
172  .src_ipaddr = *src_ipaddr,
173  .src_port = src_port,
174  .dst_ipaddr = *dst_ipaddr,
175  .dst_port = dst_port
176  }
177  };
178 
179  return addr;
180 }
