76 /** Temporarily remove the filter for a func from kevent
77  *
78  * Use to populate elements in an array of #fr_event_update_t.
79  *
80  @code {.c}
81  static fr_event_update_t pause_read[] = {
82  FR_EVENT_SUSPEND(fr_event_io_func_t, read),
83  { 0 }
84  }
85  @endcode
86  *
87  * @param[in] _s the structure containing the func to suspend.
88  * @param[in] _f the func to suspend.
89  */
90 #define FR_EVENT_SUSPEND(_s, _f) { .offset = offsetof(_s, _f), .op = FR_EVENT_OP_SUSPEND }
91 
92 /** Re-add the filter for a func from kevent
93  *
94  * Use to populate elements in an array of #fr_event_update_t.
95  *
96  @code {.c}
97  static fr_event_update_t resume_read[] = {
98  FR_EVENT_RESUME(fr_event_io_func_t, read),

100  }
101  @endcode
102  *
103  * @param[in] _s the structure containing the func to suspend.
104  * @param[in] _f the func to resume.

106 #define FR_EVENT_RESUME(_s, _f) { .offset = offsetof(_s, _f), .op = FR_EVENT_OP_RESUME }
