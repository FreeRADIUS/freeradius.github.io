123 /** Calls panic_action ifndef NDEBUG, else logs error and evaluates to value of _x
124  *
125  * Should be wrapped in a condition, and if false, should cause function to return
126  * an error code. This allows control to return to the caller if a precondition is
127  * not satisfied and we're not debugging.
128  *
129  * Example:
130  @verbatim
131  if (!fr_cond_assert(request)) return -1
132  @endverbatim
133  *
134  * @param[in] _x expression to test (should evaluate to true)

238 /** Manual validation of structures.
239  *
240  * typedef struct {
241  * char *a;
242  * int b;
243  * FR_SIGNATURE // no semicolon!
244  * } foo_t;
245  *
246  * and then once the structure is initialized (and will never be changed)
247  *
248  * foo_t *ptr;
249  * FR_STRUCT_SIGN(ptr);
250  *
251  * and some time later...
252  *
253  * foo_t *ptr;
254  * FR_STRUCT_VERIFY(ptr);
255  *
256  * Note that the structure can't contain variable elements such as fr_dlist_t.
257  * And that we're not verifying the contents of the members which are pointers.
258  */
259 #ifndef NDEBUG
260 #define FR_STRUCT_SIGN(_ptr) fr_sign_struct(_ptr, sizeof(__typeof__(*_ptr)), offsetof(__typeof__(*_ptr), _signature));
261 #define FR_STRUCT_VERIFY(_ptr) fr_verify_struct(_ptr, sizeof(__typeof__(*_ptr)), offsetof(__typeof__(*_ptr), _signature))
262 #define FR_STRUCT_SIGNATURE uint32_t _signature;
263 
264 #define FR_STRUCT_MEMBER_SIGN(_ptr, _member, _len) _ptr->_signature_##_member = fr_hash(_ptr->_member, _len)
265 #define FR_STRUCT_MEMBER_VERIFY(_ptr, _member, _len) fr_verify_struct_member(_ptr->_member, _len, &(_ptr->_signature_##_member))
266 #define FR_STRUCT_MEMBER_SIGNATURE(_member) uint32_t _signature_##_member;
267 #else
268 #define FR_STRUCT_SIGN(_ptr)
269 #define FR_STRUCT_VERIFY(_ptr)
270 #define FR_STRUCT_SIGNATURE
271 
272 #define FR_STRUCT_MEMBER_SIGN(_ptr, _member, _len)
273 #define FR_STRUCT_MEMBER_VERIFY(_ptr, _member, _len)
274 #define FR_STRUCT_MEMBER_SIGNATURE(_member)
275 #endif
