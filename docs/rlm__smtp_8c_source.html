105 static int cf_table_parse_tmpl(UNUSED TALLOC_CTX *ctx, void *out, UNUSED void *parent,
106  CONF_ITEM *ci, CONF_PARSER const *rule)
107 {
108  int ret = 0;
109  ssize_t slen;
110  int type = rule->type;
111  bool tmpl = (type & FR_TYPE_TMPL);
112  CONF_PAIR *cp = cf_item_to_pair(ci);
113  tmpl_t *vpt;
114 
115  static tmpl_rules_t rules = {
116  .allow_unknown = true,
117  .allow_unresolved = true,
118  .allow_foreign = true
119  };
120 
121  if (!tmpl) {
122  cf_log_err(cp, "Failed parsing attribute reference");
123  ret = -1;
124  goto finish;
125  }
126 
127  slen = tmpl_afrom_substr(cp, &vpt, &FR_SBUFF_IN(cf_pair_value(cp), strlen(cf_pair_value(cp))),
128  cf_pair_value_quote(cp),
129  tmpl_parse_rules_unquoted[cf_pair_value_quote(cp)],
130  &rules);
131 
132  /* There was an error */
133  if (slen < 0) {
134  char *spaces, *text;
135 
136  fr_canonicalize_error(ctx, &spaces, &text, slen, cp->value);
137 
138  cf_log_err(cp, "Failed parsing attribute reference:");
139  cf_log_err(cp, "%s", text);
140  cf_log_perr(cp, "%s^", spaces);
141 
142  talloc_free(spaces);
143  talloc_free(text);
144  /* Return error */
145  ret = -1;
146  goto finish;
147  }
148 
149  if(tmpl_is_list(vpt)) {
150  ret = -1;
151  goto finish;
152  }
153 
154  /* Only string values should be used for SMTP components */
155  if(tmpl_expanded_type(vpt) != FR_TYPE_STRING) {
156  cf_log_err(cp, "Attribute reference must be a string");
157  ret = -1;
158  goto finish;
159  }
160 
161  *(tmpl_t **)out = vpt;
162 
163  cp->parsed = true;
164 
165 finish:
166  return ret;
167 }

240  * Parse through an array of tmpl * elements and add them to an slist
241  */
242 static int tmpl_arr_to_slist (rlm_smtp_thread_t *t, fr_mail_ctx *uctx, struct curl_slist **out, tmpl_t ** const tmpl)
243 {
244  request_t *request = uctx->request;
245  int count = 0;
246  char *expanded_str;
247 
248  talloc_foreach(tmpl, current) {
249  if (current->type == TMPL_TYPE_ATTR) {
250  /* If the element contains a reference to an attribute, parse every value it references */
251  count += tmpl_attr_to_slist(uctx, out, current);
252  } else {
253  /* If the element is just a normal string, add it's name to the slist*/
254  if( tmpl_aexpand(t, &expanded_str, request, current, NULL, NULL) < 0) {
255  RDEBUG2("Could not expand the element %s", current->name);
256  break;
257  }
258  *out = curl_slist_append(*out, expanded_str);
259  count += 1;
260  }
261  }
262 
263  return count;
264 }

290 /*
291  * Adds every value in a dict_attr to a curl_slist as a comma separated list with a preposition
292  */
293 static int tmpl_arr_to_header (rlm_smtp_thread_t *t, fr_mail_ctx *uctx, struct curl_slist **out, tmpl_t ** const tmpl,
294  const char *preposition)
295 {
296  request_t *request = uctx->request;
297  fr_sbuff_t sbuff;
298  fr_sbuff_uctx_talloc_t sbuff_ctx;
299  ssize_t out_len = 0;
300  char *expanded_str;
301 
302  /* Initialize the buffer for the recipients. Used for TO */
303  fr_sbuff_init_talloc(uctx, &sbuff, &sbuff_ctx, 256, SIZE_MAX);
304  /* Add the preposition for the header element */
305  fr_sbuff_in_strcpy(&sbuff, preposition);
306 
307  talloc_foreach(tmpl, vpt) {
308  /* If there have already been elements added, keep them comma separated */
309  if (out_len > 0) {
310  out_len += fr_sbuff_in_strcpy(&sbuff, ", ");
311  }
312  /* Add the tmpl to the header sbuff */
313  if (vpt->type == TMPL_TYPE_ATTR) {
314  /* If the element contains a reference to an attribute, parse every value it references */
315  out_len += tmpl_attr_to_sbuff(uctx, &sbuff, vpt, ", ");
316  } else {
317  /* If the element is just a normal string, add it's name to the slist*/
318  if( tmpl_aexpand(t, &expanded_str, request, vpt, NULL, NULL) < 0) {
319  RDEBUG2("Could not expand the element %s", vpt->name);
320  break;
321  }
322  out_len += fr_sbuff_in_bstrncpy(&sbuff, expanded_str, vpt->len);
323  }
324  }
325 
326  /* Add the generated buffer the the curl_slist if elements were added */
327  if (out_len > 0) {
328  *out = curl_slist_append(*out, sbuff.buff);
329  talloc_free(sbuff.buff);
330  /* one element was successfully added */
331  return 1;
332  }
333  talloc_free(sbuff.buff);
334  /* The element failed to be added */
335  RDEBUG2("Failed to add the element to the header");
336  return 0;
337 }

342 static int str_to_attachments (fr_mail_ctx *uctx, curl_mime *mime, char const * str, size_t len,
343  fr_sbuff_t *path_buffer, fr_sbuff_marker_t *m)
344 {
345  int attachments_set = 0;
346  request_t *request = uctx->request;
347  curl_mimepart *part;
348 
349  /* Move to the end of the template directory filepath */
350  fr_sbuff_set(path_buffer, m);
351 
352  /* Check to see if the file attachment is valid, skip it if not */
353  RDEBUG2("Trying to set attachment: %s", str);
354 
355  if(strncmp(str, "/", 1) == 0) {
356  RDEBUG2("File attachments cannot be an absolute path");
357  return 0;
358  }
359  if(strncmp(str, "..", 2) == 0) {
360  RDEBUG2("Cannot access values outside of template_directory");
361  return 0;
362  }
363 
364  /* Copy the filename into the buffer */
365  fr_sbuff_in_bstrncpy(path_buffer, str, len);
366  /* Add the file attachment as a mime encoded part */
367  attachments_set++;
368  part = curl_mime_addpart(mime);
369  curl_mime_encoder(part, "base64");
370  curl_mime_filedata(part, path_buffer->buff);
371  return 1;
372 }

501 /*
502  * Generates a curl_slist of header elements header elements
503  */
504 static int header_source(rlm_smtp_thread_t *t, fr_mail_ctx *uctx, UNUSED rlm_smtp_t const *inst)
505 {
506  fr_sbuff_t time_out;
507  char const *to = "TO: ";
508  char const *cc = "CC: ";
509  request_t *request = uctx->request;
510  fr_sbuff_t conf_buffer;
511  fr_sbuff_uctx_talloc_t conf_ctx;
512  map_t *conf_map;
513 
514  char *expanded_rhs;
515 
516  /* Initialize the sbuff for writing the config elements as header attributes */
517  fr_sbuff_init_talloc(uctx, &conf_buffer, &conf_ctx, 256, SIZE_MAX);
518  conf_map = inst->header_maps;
519  /* Load in all of the header elements supplies in the config */
520  while (conf_map->rhs && conf_map->lhs) {
521  /* Do any string expansion required in the rhs */
522  if( tmpl_aexpand(t, &expanded_rhs, request, conf_map->rhs, NULL, NULL) < 0) {
523  RDEBUG2("Skipping: %s's could not parse: %s", conf_map->lhs->name, conf_map->rhs->name);
524  goto next;
525  }
526  /* Format the conf item to be a valid SMTP header */
527  fr_sbuff_in_bstrncpy(&conf_buffer, conf_map->lhs->name, conf_map->lhs->len);
528  fr_sbuff_in_strcpy(&conf_buffer, ": ");
529  fr_sbuff_in_bstrncpy(&conf_buffer, expanded_rhs, strlen(expanded_rhs));
530  /* Add the header to the curl slist */
531  uctx->header = curl_slist_append(uctx->header, conf_buffer.buff);
532  talloc_free(conf_buffer.buff);
533  /* Check if there are more values to parse */
534  next:
535  if (!conf_map->next) break;
536  /* reinitialize the buffer and move to the next value */
537  fr_sbuff_init_talloc(uctx, &conf_buffer, &conf_ctx, 256, SIZE_MAX);
538  conf_map = conf_map->next;
