65 /** Call the exit handler
66  *
67  */
68 static int _exit_handler_entry_free(fr_exit_handler_entry_t *e)
69 {
70  if (fr_dlist_entry_in_list(&e->entry)) fr_dlist_entry_unlink(&e->entry);
71 
72  /*
73  * If the exit handler wasn't disarmed, call it...
74  */
75  if (e->func) e->func(e->uctx);
76 
77  return 0;
78 }

80 /** Allocate a new exit handler entry
81  *
82  */
83 static fr_exit_handler_entry_t *exit_handler_entry_alloc(fr_exit_handler_list_t *list,
84  fr_thread_local_atexit_t func, void const *uctx)
85 {
86  fr_exit_handler_entry_t *e;
87 
88  e = talloc_zero(list, fr_exit_handler_entry_t);
89  if (unlikely(!e)) return NULL;
90 
91  e->func = func;
92  memcpy(&e->uctx, &uctx, sizeof(e->uctx));
93  fr_dlist_insert_head(&list->head, e);
94  talloc_set_destructor(e, _exit_handler_entry_free);
95 
96  return e;
97 }

127 /** Free any thread-local exit handler lists that pthread_key failed to fre

140 /** Setup the atexit handler, should be called at the start of a program's execution
141  *
142  */
143 int fr_thread_local_atexit_setup(void)
144 {
145  if (global_atexit) return 0;
146 
147  global_atexit = talloc_zero(NULL, fr_exit_handler_list_t);
148  if (unlikely(!global_atexit)) return -1;
149 
150  fr_dlist_talloc_init(&global_atexit->head, fr_exit_handler_entry_t, entry);
151  talloc_set_destructor(global_atexit, _global_list_free);
152  atexit(_global_free); /* Call all remaining destructors at process exit */
153 
154  return 0;
155 }

157 /** Add a new destructor
158  *
159  * @return
160  * - 0 on success.
161  * - -1 on memory allocation failure;
162  */
163 int fr_thread_local_atexit(fr_thread_local_atexit_t func, void const *uctx)
164 {
165  int ret = 0;
166 
167  /*
168  * Initialise the global list containing all the thread-local
169  * dlist destructors.
170  */
171  pthread_mutex_lock(&global_atexit_mutex);
172  fr_cond_assert_msg(!is_exiting, "New atexit handlers should not be allocated whilst exiting");
173  if (!global_atexit) ret = fr_thread_local_atexit_setup();
174  pthread_mutex_unlock(&global_atexit_mutex);
175  if (ret < 0) return ret;
176 
177  /*
178  * Initialise the thread local list, just for pthread_exit().
179  */
180  if (!thread_local_atexit) {
181  /*
182  * Must be heap allocated, because thread local
183  * structures can be freed before the key
184  * destructor is run (depending on platform).
185  */
186  thread_local_atexit = talloc_zero(NULL, fr_exit_handler_list_t);
187  if (unlikely(!thread_local_atexit)) return -1;
188 
189  fr_dlist_talloc_init(&thread_local_atexit->head, fr_exit_handler_entry_t, entry);
190  (void) pthread_key_create(&thread_local_atexit->key, _thread_local_free);
191 
192  /*
193  * We need to pass in a pointer to the heap
194  * memory because, again, the thread local
195  * indirection table may have disappeared
196  * by the time the thread destructor is
197  * called.
198  */
199  (void) pthread_setspecific(thread_local_atexit->key, thread_local_atexit);
200  talloc_set_destructor(thread_local_atexit, _thread_local_list_free);
201 
202  /*
203  * Add a destructor for the thread-local list
204  * The pthread based destructor will disarm
205  * this if it fires, but leave it enabled if
206  * it doesn't, thus ensuring the memory is
207  * *always* freed one way or another.
208  */
209  pthread_mutex_lock(&global_atexit_mutex);
210  thread_local_atexit->e = exit_handler_entry_alloc(global_atexit,
211  _thread_local_free,
212  thread_local_atexit);
213  pthread_mutex_unlock(&global_atexit_mutex);
214  }
215 
216  /*
217  * Now allocate the actual atexit handler entry
218  */
219  if (exit_handler_entry_alloc(thread_local_atexit, func, uctx) == NULL) return -1;
220 
221  return 0;
222 }
