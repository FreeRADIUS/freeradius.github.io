95  MEM(ctx = talloc_init_const("control_master"));

97  MPRINT1("Master started.\n");
98 
99  /*
100  * Busy loop. We're stupid.
101  */
102  while (true) {
103  int num_events;
104  ssize_t data_size;
105  my_message_t m;
106  struct kevent kev;
107 
108  wait_for_events:
109  MPRINT1("Master waiting for events.\n");
110 
111  num_events = kevent(kq, NULL, 0, &kev, 1, NULL);
112  if (num_events < 0) {
113  fprintf(stderr, "Failed reading kevent: %s\n", fr_syserror(errno));
114  fr_exit_now(EXIT_FAILURE);
115  }
116 
117  MPRINT1("Master draining the control plane.\n");
118 
119  while (true) {
120  uint32_t id;
121 
122  data_size = fr_control_message_pop(aq, &id, &m, sizeof(m));
123  if (data_size == 0) goto wait_for_events;
124 
125  if (data_size < 0) {
126  fprintf(stderr, "Failed reading control message\n");
127  fr_exit_now(EXIT_FAILURE);
128  }
129 
130  fr_assert(data_size == sizeof(m));
131  fr_assert(id == FR_CONTROL_ID_CHANNEL);
132 
133  MPRINT1("Master got message %zu.\n", m.counter);
134 
135  fr_assert(m.header == CONTROL_MAGIC);
136 
137  if (m.counter == (max_messages - 1)) goto do_exit;
138  }
139  }
140 
141 do_exit:
142  MPRINT1("Master exiting.\n");
143 
144  talloc_free(ctx);
145 
146  return NULL;
147 }
