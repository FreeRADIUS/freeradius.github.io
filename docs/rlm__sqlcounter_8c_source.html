37 /*
38  * Note: When your counter spans more than 1 period (ie 3 months
39  * or 2 weeks), this module probably does NOT do what you want! It
40  * calculates the range of dates to count across by first calculating
41  * the End of the Current period and then subtracting the number of
42  * periods you specify from that to determine the beginning of the
43  * range.
44  *
45  * For example, if you specify a 3 month counter and today is June 15th,
46  * the end of the current period is June 30. Subtracting 3 months from
47  * that gives April 1st. So, the counter will sum radacct entries from
48  * April 1st to June 30. Then, next month, it will sum entries from
49  * May 1st to July 31st.
50  *
51  * To fix this behavior, we need to add some way of storing the Next
52  * Reset Time.
53  */

114 static int find_next_reset(rlm_sqlcounter_t *inst, time_t timeval)
115 {
116  int ret = 0;
117  size_t len;
118  unsigned int num = 1;
119  char last = '\0';
120  struct tm *tm, s_tm;
121  char sCurrentTime[40], sNextTime[40];
122  time_t date;
123 
124  tm = localtime_r(&timeval, &s_tm);
125  len = strftime(sCurrentTime, sizeof(sCurrentTime), "%Y-%m-%d %H:%M:%S", tm);
126  if (len == 0) *sCurrentTime = '\0';
127  tm->tm_sec = tm->tm_min = 0;
128 
129  fr_assert(inst->reset != NULL);
130 
131  if (isdigit((int) inst->reset[0])){
132  len = strlen(inst->reset);
133  if (len == 0)
134  return -1;
135  last = inst->reset[len - 1];
136  if (!isalpha((int) last))
137  last = 'd';
138  num = atoi(inst->reset);
139  DEBUG("num=%d, last=%c",num,last);
140  }
141  if (strcmp(inst->reset, "hourly") == 0 || last == 'h') {
142  /*
143  * Round up to the next nearest hour.
144  */
145  tm->tm_hour += num;
146  date = mktime(tm);
147  } else if (strcmp(inst->reset, "daily") == 0 || last == 'd') {
148  /*
149  * Round up to the next nearest day.
150  */
151  tm->tm_hour = 0;
152  tm->tm_mday += num;
153  date = mktime(tm);
154  } else if (strcmp(inst->reset, "weekly") == 0 || last == 'w') {
155  /*
156  * Round up to the next nearest week.
157  */
158  tm->tm_hour = 0;
159  tm->tm_mday += (7 - tm->tm_wday) +(7*(num-1));
160  date = mktime(tm);
161  } else if (strcmp(inst->reset, "monthly") == 0 || last == 'm') {
162  tm->tm_hour = 0;
163  tm->tm_mday = 1;
164  tm->tm_mon += num;
165  date = mktime(tm);
166  } else if (strcmp(inst->reset, "never") == 0) {
167  date = 0;
168  } else {
169  return -1;
170  }
171 
172  inst->reset_time = fr_time_from_sec(date);
173 
174  len = strftime(sNextTime, sizeof(sNextTime),"%Y-%m-%d %H:%M:%S",tm);
175  if (len == 0) *sNextTime = '\0';
176  DEBUG2("Current Time: %" PRId64 " [%s], Next reset %" PRId64 " [%s]",
177  (int64_t) timeval, sCurrentTime, (int64_t) date, sNextTime);
178 
179  return ret;
180 }

251 /*
252  * Replace %<whatever> in a string.
253  *
254  * %b last_reset
255  * %e reset_time
256  * %S sqlmod_inst
257  *
258  */
259 static ssize_t sqlcounter_expand(char *out, int outlen, rlm_sqlcounter_t const *inst, UNUSED request_t *request, char const *fmt)
260 {
261  int freespace;
262  char const *p;
263  char *q;
264  char tmpdt[40]; /* For temporary storing of dates */
265 
266  q = out;
267  p = fmt;
268  while (*p) {
269  /* Calculate freespace in output */
270  freespace = outlen - (q - out);
271  if (freespace <= 1) {
272  return -1;
273  }
274 
275  /*
276  * Non-% get copied as-is.
277  */
278  if (*p != '%') {
279  *q++ = *p++;
280  continue;
281  }
282  p++;
283  if (!*p) { /* % and then EOS --> % */
284  *q++ = '%';
285  break;
286  }
287 
288  if (freespace <= 2) return -1;
289 
290  /*
291  * We need TWO %% in a row before we do our expansions.
292  * If we only get one, just copy the %s as-is.
293  */
294  if (*p != '%') {
295  *q++ = '%';
296  *q++ = *p++;
297  continue;
298  }
299  p++;
300  if (!*p) {
301  *q++ = '%';
302  *q++ = '%';
303  break;
304  }
305 
306  if (freespace <= 3) return -1;
307 
308  switch (*p) {
309  case 'b': /* last_reset */
310  snprintf(tmpdt, sizeof(tmpdt), "%" PRId64, (int64_t) inst->last_reset);
311  strlcpy(q, tmpdt, freespace);
312  q += strlen(q);
313  p++;
314  break;
315  case 'e': /* reset_time */
316  snprintf(tmpdt, sizeof(tmpdt), "%" PRId64, (int64_t) inst->reset_time);
317  strlcpy(q, tmpdt, freespace);
318  q += strlen(q);
319  p++;
320  break;
321 
322  /*
323  * %%s gets copied over as-is.
324  */
325  default:
326  *q++ = '%';
327  *q++ = '%';
328  *q++ = *p++;
329  break;
330  }
331  }
332  *q = '\0';
333 
334  DEBUG2("sqlcounter_expand: '%s'", out);
335 
336  return strlen(out);
337 }

343 static int counter_cmp(void *instance, request_t *request, UNUSED fr_pair_t *req , fr_pair_t *check,
344  UNUSED fr_pair_t *check_list)
345 {
346  rlm_sqlcounter_t const *inst = talloc_get_type_abort_const(instance, rlm_sqlcounter_t);
347  uint64_t counter;
348 
349  char query[MAX_QUERY_LEN], subst[MAX_QUERY_LEN];
350  char *expanded = NULL;
351  size_t len;
352 
353  /* First, expand %k, %b and %e in query */
354  if (sqlcounter_expand(subst, sizeof(subst), inst, request, inst->query) <= 0) {
355  REDEBUG("Insufficient query buffer space");
356 
357  return -1;
358  }
359 
360  /* Then combine that with the name of the module were using to do the query */
361  len = snprintf(query, sizeof(query), "%%{%s:%s}", inst->sqlmod_inst, subst);
362  if (len >= sizeof(query) - 1) {
363  REDEBUG("Insufficient query buffer space");
364 
365  return -1;
366  }
367 
368  /* Finally, xlat resulting SQL query */
369  if (xlat_aeval(request, &expanded, request, query, NULL, NULL) < 0) {
370  return -1;
371  }
372 
373  if (sscanf(expanded, "%" PRIu64, &counter) != 1) {
374  RDEBUG2("No integer found in string \"%s\"", expanded);
375  }
376  talloc_free(expanded);
377 
378  if (counter < check->vp_uint64) return -1;
379  if (counter > check->vp_uint64) return 1;
380  return 0;
381 }

383 /*
384  * Find the named user in this modules database. Create the set
385  * of attribute-value pairs to check and reply with for this user
386  * from the database. The authentication code only needs to check
387  * the password, the rest is done here.
388  */
389 static unlang_action_t CC_HINT(nonnull) mod_authorize(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
390 {
391  rlm_sqlcounter_t *inst = talloc_get_type_abort(mctx->instance, rlm_sqlcounter_t);
392  uint64_t counter, res;
393  fr_pair_t *limit;
394  fr_pair_t *reply_item;
395  char msg[128];
396  int ret;
397 
398  char query[MAX_QUERY_LEN], subst[MAX_QUERY_LEN];
399  char *expanded = NULL;
400 
401  size_t len;
402 
403  /*
404  * Before doing anything else, see if we have to reset
405  * the counters.
406  */
407  if (inst->reset_time && (inst->reset_time <= fr_time_to_sec(request->packet->timestamp))) {
408  /*
409  * Re-set the next time and prev_time for this counters range
410  */
411  inst->last_reset = inst->reset_time;
412  find_next_reset(inst, fr_time_to_sec(request->packet->timestamp));
413  }
414 
415  if (tmpl_find_vp(&limit, request, inst->limit_attr) < 0) {
416  RWDEBUG2("Couldn't find limit attribute, %s, doing nothing...", inst->limit_attr->name);
417  RETURN_MODULE_NOOP;
418  }
419 
420  /* First, expand %k, %b and %e in query */
421  if (sqlcounter_expand(subst, sizeof(subst), inst, request, inst->query) <= 0) {
422  REDEBUG("Insufficient query buffer space");
423 
424  RETURN_MODULE_FAIL;
425  }
426 
427  /* Then combine that with the name of the module were using to do the query */
428  len = snprintf(query, sizeof(query), "%%{%s:%s}", inst->sqlmod_inst, subst);
429  if (len >= (sizeof(query) - 1)) {
430  REDEBUG("Insufficient query buffer space");
431 
432  RETURN_MODULE_FAIL;
433  }
434 
435  /* Finally, xlat resulting SQL query */
436  if (xlat_aeval(request, &expanded, request, query, NULL, NULL) < 0) {
437  RETURN_MODULE_FAIL;
438  }
439  talloc_free(expanded);
440 
441  if (sscanf(expanded, "%" PRIu64, &counter) != 1) {
442  RDEBUG2("No integer found in result string \"%s\". May be first session, setting counter to 0",
443  expanded);
444  counter = 0;
445  }
446 
447  /*
448  * Check if check item > counter
449  */
450  if (limit->vp_uint64 <= counter) {
451  fr_pair_t *vp;
452 
453  /* User is denied access, send back a reply message */
454  snprintf(msg, sizeof(msg), "Your maximum %s usage time has been reached", inst->reset);
455 
456  MEM(pair_update_reply(&vp, attr_reply_message) >= 0);
457  fr_pair_value_strdup(vp, msg);
458 
459  REDEBUG2("Maximum %s usage time reached", inst->reset);
460  REDEBUG2("Rejecting user, %s value (%" PRIu64 ") is less than counter value (%" PRIu64 ")",
461  inst->limit_attr->name, limit->vp_uint64, counter);
462 
463  RETURN_MODULE_REJECT;
464  }
465 
466  res = limit->vp_uint64 - counter;
467  RDEBUG2("Allowing user, %s value (%" PRIu64 ") is greater than counter value (%" PRIu64 ")",
468  inst->limit_attr->name, limit->vp_uint64, counter);
469 
470  /*
471  * We are assuming that simultaneous-use=1. But
472  * even if that does not happen then our user
473  * could login at max for 2*max-usage-time Is
474  * that acceptable?
475  */
476  if (inst->reply_attr) {
477  /*
478  * If we are near a reset then add the next
479  * limit, so that the user will not need to login
480  * again. Do this only for Session-Timeout.
481  */
482  if ((tmpl_da(inst->reply_attr) == attr_session_timeout) &&
483  inst->reset_time &&
484  (res >= (uint64_t)(inst->reset_time - fr_time_to_sec(request->packet->timestamp)))) {
485  uint64_t to_reset = inst->reset_time - fr_time_to_sec(request->packet->timestamp);
486 
487  RDEBUG2("Time remaining (%" PRIu64 "s) is greater than time to reset (%" PRIu64 "s). "
488  "Adding %" PRIu64 "s to reply value", to_reset, res, to_reset);
489  res = to_reset + limit->vp_uint32;
490  }
491 
492  /*
493  * Limit the reply attribute to the minimum of the existing value, or this new one.
494  */
495  ret = tmpl_find_or_add_vp(&reply_item, request, inst->reply_attr);
496  switch (ret) {
497  case 1: /* new */
498  break;
499 
500  case 0: /* found */
501  if (reply_item->vp_uint64 <= res) {
502  RDEBUG2("Leaving existing %s value of %" PRIu64, inst->reply_attr->name,
503  reply_item->vp_uint64);
504  RETURN_MODULE_OK;
505  }
506  break;
507 
508  case -1: /* alloc failed */
509  REDEBUG("Error allocating attribute %s", inst->reply_attr->name);
510  RETURN_MODULE_FAIL;
511 
512  default: /* request or list unavailable */
513  RDEBUG2("List or request context not available for %s, skipping...", inst->reply_attr->name);
514  RETURN_MODULE_OK;
515  }
516  reply_item->vp_uint64 = res;
517  RDEBUG2("&%pP", reply_item);
518 
519  RETURN_MODULE_UPDATED;
520  }
521 
522  RETURN_MODULE_OK;
523 }

535 static int mod_instantiate(void *instance, CONF_SECTION *conf)
536 {
537  rlm_sqlcounter_t *inst = instance;
538  time_t now;
539 
540  fr_assert(inst->query && *inst->query);
541 
542  now = time(NULL);
543  inst->reset_time = 0;
544 
545  if (find_next_reset(inst, now) == -1) {
546  cf_log_err(conf, "Invalid reset '%s'", inst->reset);
547  return -1;
548  }
549 
550  /*
551  * Discover the beginning of the current time period.
552  */
553  inst->last_reset = 0;
554 
555  if (find_prev_reset(inst, now) < 0) {
556  cf_log_err(conf, "Invalid reset '%s'", inst->reset);
557  return -1;
558  }
559 
560  return 0;
561 }

563 static int mod_bootstrap(void *instance, CONF_SECTION *conf)
564 {
565  rlm_sqlcounter_t *inst = instance;
566  fr_dict_attr_flags_t flags;
567 
568  /*
569  * Create a new attribute for the counter.
570  */

572  fr_assert(inst->limit_attr);
573 
574  memset(&flags, 0, sizeof(flags));
575  if (tmpl_attr_unresolved_add(fr_dict_unconst(dict_freeradius), inst->paircmp_attr, FR_TYPE_UINT64, &flags) < 0) {
576  cf_log_perr(conf, "Failed defining counter attribute");
577  return -1;
578  }
579 
580  if (tmpl_attr_unresolved_add(fr_dict_unconst(dict_freeradius), inst->limit_attr, FR_TYPE_UINT64, &flags) < 0) {
581  cf_log_perr(conf, "Failed defining check attribute");
582  return -1;
583  }
584 
585  if (tmpl_da(inst->paircmp_attr)->type != FR_TYPE_UINT64) {
586  cf_log_err(conf, "Counter attribute %s MUST be uint64", tmpl_da(inst->paircmp_attr)->name);
587  return -1;
588  }
589  if (paircmp_register_by_name(tmpl_da(inst->paircmp_attr)->name, NULL, true,
590  counter_cmp, inst) < 0) {
591  cf_log_perr(conf, "Failed registering comparison function for counter attribute %s",
592  tmpl_da(inst->paircmp_attr)->name);
593  return -1;
594  }
595 
596  if (tmpl_da(inst->limit_attr)->type != FR_TYPE_UINT64) {
597  cf_log_err(conf, "Check attribute %s MUST be uint64", tmpl_da(inst->limit_attr)->name);
598  return -1;
599  }
600 
601  return 0;
602 }
