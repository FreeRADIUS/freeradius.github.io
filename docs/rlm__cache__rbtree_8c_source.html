41 /** Compare two entries by key
42  *
43  * There may only be one entry with the same key.
44  */
45 static int cache_entry_cmp(void const *one, void const *two)
46 {
47  rlm_cache_entry_t const *a = one, *b = two;
48  int ret;
49 
50  ret = (a->key_len > b->key_len) - (a->key_len < b->key_len);
51  if (ret != 0) return ret;
52 
53  return memcmp(a->key, b->key, a->key_len);
54 }

56 /** Compare two entries by expiry time
57  *
58  * There may be multiple entries with the same expiry time.
59  */
60 static int8_t cache_heap_cmp(void const *one, void const *two)
61 {
62  rlm_cache_entry_t const *a = one, *b = two;
63 
64  return (a->expires > b->expires) - (a->expires < b->expires);
65 }

99 /** Create a new cache_rbtree instance
100  *
101  * @param instance A uint8_t array of inst_size if inst_size > 0, else NULL,
102  * this should contain the result of parsing the driver's
103  * CONF_PARSER array that it specified in the interface struct.
104  * @param conf section holding driver specific #CONF_PAIR (s).
105  * @return
106  * - 0 on success.
107  * - -1 on failure.
108  */
109 static int mod_instantiate(void *instance, UNUSED CONF_SECTION *conf)
110 {
111  rlm_cache_rbtree_t *driver = talloc_get_type_abort(instance, rlm_cache_rbtree_t);
112 
113  /*
114  * The cache.
115  */
116  driver->cache = rbtree_talloc_alloc(NULL, cache_entry_cmp, rlm_cache_rbtree_entry_t, NULL, 0);
117  if (!driver->cache) {
118  ERROR("Failed to create cache");
119  return -1;
120  }
121  talloc_link_ctx(driver, driver->cache);
122 
123  /*
124  * The heap of entries to expire.
125  */
126  driver->heap = fr_heap_talloc_alloc(driver, cache_heap_cmp, rlm_cache_rbtree_entry_t, heap_id);
127  if (!driver->heap) {
128  ERROR("Failed to create heap for the cache");
129  return -1;
130  }
131 
132  if (pthread_mutex_init(&driver->mutex, NULL) < 0) {
133  ERROR("Failed initializing mutex: %s", fr_syserror(errno));
134  return -1;
135  }
136 
137  return 0;
138 }

161 /** Locate a cache entry
162  *
163  * @note handle not used except for sanity checks.
164  *
165  * @copydetails cache_entry_find_t
166  */
167 static cache_status_t cache_entry_find(rlm_cache_entry_t **out,
168  UNUSED rlm_cache_config_t const *config, void *instance,
169  request_t *request, UNUSED void *handle, uint8_t const *key, size_t key_len)
170 {
171  rlm_cache_rbtree_t *driver = talloc_get_type_abort(instance, rlm_cache_rbtree_t);
172 
173  rlm_cache_entry_t *c;
174 
175  fr_assert(driver->cache);
176 
177  /*
178  * Clear out old entries
179  */
180  c = fr_heap_peek(driver->heap);
181  if (c && (c->expires < fr_time_to_unix_time(request->packet->timestamp))) {
182  fr_heap_extract(driver->heap, c);
183  rbtree_deletebydata(driver->cache, c);
184  talloc_free(c);
185  }
186 
187  /*
188  * Is there an entry for this key?
189  */
190  c = rbtree_finddata(driver->cache, &(rlm_cache_entry_t){ .key = key, .key_len = key_len });
191  if (!c) {
192  *out = NULL;
193  return CACHE_MISS;
194  }
195  *out = c;
196 
197  return CACHE_OK;
198 }

206 static cache_status_t cache_entry_expire(UNUSED rlm_cache_config_t const *config, void *instance,
207  request_t *request, UNUSED void *handle,
208  uint8_t const *key, size_t key_len)
209 {
210  rlm_cache_rbtree_t *driver = talloc_get_type_abort(instance, rlm_cache_rbtree_t);
211  rlm_cache_entry_t *c;
212 
213  if (!request) return CACHE_ERROR;
214 
215  c = rbtree_finddata(driver->cache, &(rlm_cache_entry_t){ .key = key, .key_len = key_len });
216  if (!c) return CACHE_MISS;
217 
218  fr_heap_extract(driver->heap, c);
219  rbtree_deletebydata(driver->cache, c);
220  talloc_free(c);
221 
222  return CACHE_OK;
223 }

225 /** Insert a new entry into the data store
226  *
227  * @note handle not used except for sanity checks.
228  *
229  * @copydetails cache_entry_insert_t
230  */
231 static cache_status_t cache_entry_insert(rlm_cache_config_t const *config, void *instance,
232  request_t *request, void *handle,
233  rlm_cache_entry_t const *c)
234 {
235  cache_status_t status;
236 
237  rlm_cache_rbtree_t *driver = talloc_get_type_abort(instance, rlm_cache_rbtree_t);
238  rlm_cache_entry_t *my_c;
239 
240  fr_assert(handle == request);
241 
242  if (!request) return CACHE_ERROR;
243 
244  memcpy(&my_c, &c, sizeof(my_c));
245 
246  /*
247  * Allow overwriting
248  */
249  if (!rbtree_insert(driver->cache, my_c)) {
250  status = cache_entry_expire(config, instance, request, handle, c->key, c->key_len);
251  if ((status != CACHE_OK) && !fr_cond_assert(0)) return CACHE_ERROR;
252 
253  if (!rbtree_insert(driver->cache, my_c)) {
254  RERROR("Failed adding entry");
255 
256  return CACHE_ERROR;
257  }
258  }
259 
260  if (fr_heap_insert(driver->heap, my_c) < 0) {
261  rbtree_deletebydata(driver->cache, my_c);
262  RERROR("Failed adding entry to expiry heap");
263 
264  return CACHE_ERROR;
265  }
266 
267  return CACHE_OK;
268 }

272  * @note handle not used except for sanity checks.

274  * @copydetails cache_entry_set_ttl_t

276 static cache_status_t cache_entry_set_ttl(UNUSED rlm_cache_config_t const *config, void *instance,
277  request_t *request, UNUSED void *handle,
278  rlm_cache_entry_t *c)
279 {
280  rlm_cache_rbtree_t *driver = talloc_get_type_abort(instance, rlm_cache_rbtree_t);
281 
282 #ifdef NDEBUG
283  if (!request) return CACHE_ERROR;
284 #endif
285 
286  if (!fr_cond_assert(fr_heap_extract(driver->heap, c) == 0)) {
287  RERROR("Entry not in heap");
288  return CACHE_ERROR;
289  }
290 
291  if (fr_heap_insert(driver->heap, c) < 0) {
292  rbtree_deletebydata(driver->cache, c); /* make sure we don't leak entries... */
293  RERROR("Failed updating entry TTL. Entry was forcefully expired");
294  return CACHE_ERROR;
295  }
296  return CACHE_OK;
297 }
