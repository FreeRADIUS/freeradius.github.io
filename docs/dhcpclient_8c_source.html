144 static fr_radius_packet_t *request_init(char const *filename)
145 {
146  FILE *fp;
147  fr_cursor_t cursor;
148  fr_pair_t *vp;
149  bool filedone = false;
150  fr_radius_packet_t *packet;
151 
152  /*
153  * Determine where to read the VP's from.
154  */
155  if (filename) {
156  fp = fopen(filename, "r");
157  if (!fp) {
158  ERROR("Error opening %s: %s", filename, fr_syserror(errno));
159  return NULL;
160  }
161  } else {
162  fp = stdin;
163  }
164 
165  packet = fr_radius_alloc(NULL, false);
166 
167  /*
168  * Read the VP's.
169  */
170  if (fr_pair_list_afrom_file(packet, dict_dhcpv4, &packet->vps, fp, &filedone) < 0) {
171  fr_perror("dhcpclient");
172  fr_radius_packet_free(&packet);
173  if (fp && (fp != stdin)) fclose(fp);
174  return NULL;
175  }
176 
177  /*
178  * Fix / set various options
179  */
180  for (vp = fr_cursor_init(&cursor, &packet->vps);
181  vp;
182  vp = fr_cursor_next(&cursor)) {
183 
184  /*
185  * Xlat expansions are not supported. Convert xlat to value box (if possible).
186  */
187  if (vp->type == VT_XLAT) {
188  fr_type_t type = vp->da->type;
189  if (fr_value_box_from_str(vp, &vp->data, &type, NULL, vp->xlat, -1, '\0', false) < 0) {
190  fr_perror("dhcpclient");
191  fr_radius_packet_free(&packet);
192  if (fp && (fp != stdin)) fclose(fp);
193  return NULL;
194  }
195  vp->type = VT_DATA;
196  }
197 
198  /*
199  * Allow to set packet type using DHCP-Message-Type
200  */
201  if (vp->da == attr_dhcp_message_type) {
202  packet->code = vp->vp_uint8;
203 
204  /*
205  * Allow it to set the packet type in
206  * the attributes read from the file.
207  * (this takes precedence over the command argument.)
208  */
209  } else if (vp->da == attr_packet_type) {
210  packet->code = vp->vp_uint32;
211 
212  } else if (vp->da == attr_packet_dst_port) {
213  packet->socket.inet.dst_port = vp->vp_uint16;
214 
215  } else if ((vp->da == attr_packet_dst_ip_address) ||
216  (vp->da == attr_packet_dst_ipv6_address)) {
217  memcpy(&packet->socket.inet.dst_ipaddr, &vp->vp_ip, sizeof(packet->socket.inet.src_ipaddr));
218 
219  } else if (vp->da == attr_packet_src_port) {
220  packet->socket.inet.src_port = vp->vp_uint16;
221 
222  } else if ((vp->da == attr_packet_src_ip_address) ||
223  (vp->da == attr_packet_src_ipv6_address)) {
224  memcpy(&packet->socket.inet.src_ipaddr, &vp->vp_ip, sizeof(packet->socket.inet.src_ipaddr));
225  } /* switch over the attribute */
226 
227  } /* loop over the VP's we read in */
228 
229  if (fp && (fp != stdin)) fclose(fp);
230 
231  /*
232  * And we're done.
233  */
234  return packet;
235 }

238 /*
239  * Loop waiting for DHCP replies until timer expires.
240  * Note that there may be more than one reply: multiple DHCP servers can respond to a broadcast discover.
241  * A real client would pick one of the proposed replies.
242  * We'll just return the first eligible reply, and display the others.
243  */
244 #if defined(HAVE_LINUX_IF_PACKET_H) || defined (HAVE_LIBPCAP)
245 static fr_radius_packet_t *fr_dhcpv4_recv_raw_loop(int lsockfd,
246 #ifdef HAVE_LINUX_IF_PACKET_H
247  struct sockaddr_ll *p_ll,
248 #endif
249  fr_radius_packet_t *packet_p)
250 {
251  fr_time_delta_t our_timeout;
252  fr_radius_packet_t *found = NULL;
253  fr_radius_packet_t *reply = NULL;
254  int nb_reply = 0;
255  int nb_offer = 0;
256  dc_offer_t *offer_list = NULL;
257  fd_set read_fd;
258  int retval;
259  fr_cursor_t cursor;
260 
261  our_timeout = timeout;
262 
263  /* Loop waiting for DHCP replies until timer expires */
264  while (our_timeout) {
265  if ((!found) || (reply)) { // only debug at start and each time we get a valid DHCP reply on raw socket
266  DEBUG("Waiting for %s DHCP replies for: %d.%06d",
267  (nb_reply > 0) ? " additional ":" ",
268  (int)(our_timeout / NSEC),
269  (int)(our_timeout % NSEC));
270  }
271 
272  reply = NULL;
273  FD_ZERO(&read_fd);
274  FD_SET(lsockfd, &read_fd);
275  retval = select(lsockfd + 1, &read_fd, NULL, NULL, &fr_time_delta_to_timeval(our_timeout));
276  if (retval < 0) {
277  fr_strerror_printf("Select on DHCP socket failed: %s", fr_syserror(errno));
278  return NULL;
279  }
280 
281  if (retval > 0 && FD_ISSET(lsockfd, &read_fd)) {
282  /* There is something to read on our socket */
283 
284 #ifdef HAVE_LINUX_IF_PACKET_H
285  reply = fr_dhcv4_raw_packet_recv(lsockfd, p_ll, packet_p);
286 #else
287 # ifdef HAVE_LIBPCAP
288  reply = fr_dhcpv4_pcap_recv(pcap);
289 # else
290 # error Need <if/packet.h> or <pcap.h>
291 # endif
292 #endif
293  } else {
294  our_timeout = 0;
295  }
296 
297  if (reply) {
298  nb_reply ++;
299 
300  if (fr_debug_lvl > 1) fr_dhcpv4_print_hex(stdout, reply->data, reply->data_len);
301 
302  fr_cursor_init(&cursor, &reply->vps);
303  if (fr_dhcpv4_decode(reply, reply->data, reply->data_len, &cursor, &reply->code) < 0) {
304  ERROR("Failed decoding reply");
305  return NULL;
306  }
307 
308  if (!found) found = reply;
309 
310  if (reply->code == FR_DHCP_OFFER) {
311  fr_pair_t *vp1 = fr_pair_find_by_da(&reply->vps,
312  attr_dhcp_dhcp_server_identifier);
313  fr_pair_t *vp2 = fr_pair_find_by_da(&reply->vps,
314  attr_dhcp_your_ip_address);
315 
316  if (vp1 && vp2) {
317  nb_offer++;
318  offer_list = talloc_realloc(packet_p, offer_list, dc_offer_t, nb_offer);
319  offer_list[nb_offer - 1].server_addr = vp1->vp_ipv4addr;
320  offer_list[nb_offer - 1].offered_addr = vp2->vp_ipv4addr;
321  }
322  }
323  }
324  }
325 
326  if (0 == nb_reply) {
327  DEBUG("No valid DHCP reply received");
328  return NULL;
329  }
330 
331  /* display offer(s) received */
332  if (nb_offer > 0 ) {
333  DEBUG("Received %d DHCP Offer(s):", nb_offer);
334  int i;
335  for (i = 0; i < nb_reply; i++) {
336  char server_addr_buf[INET6_ADDRSTRLEN];
337  char offered_addr_buf[INET6_ADDRSTRLEN];
338 
339  DEBUG("IP address: %s offered by DHCP server: %s",
340  inet_ntop(AF_INET, &offer_list[i].offered_addr,
341  offered_addr_buf, sizeof(offered_addr_buf)),
342  inet_ntop(AF_INET, &offer_list[i].server_addr,
343  server_addr_buf, sizeof(server_addr_buf))
344  );
345  }
346  }
347 
348  return found;
349 }
350 #endif /* <if/packet.h> or <pcap.h> */

352 static int send_with_socket(fr_radius_packet_t **reply, fr_radius_packet_t *packet)
353 {
354  int on = 1;
355 
356 #ifdef HAVE_LINUX_IF_PACKET_H
357  if (raw_mode) {
358  sockfd = fr_dhcpv4_raw_socket_open(&ll, iface_ind);
359  if (sockfd < 0) {
360  ERROR("Error opening socket");
361  return -1;
362  }
363  } else
364 #endif
365  {
366  sockfd = fr_socket_server_udp(&packet->socket.inet.src_ipaddr, &packet->socket.inet.src_port, NULL, false);
367  if (sockfd < 0) {
368  ERROR("Error opening socket - %s", fr_strerror());
369  return -1;
370  }
371 
372  if (fr_socket_bind(sockfd, &packet->socket.inet.src_ipaddr, &packet->socket.inet.src_port, NULL) < 0) {
373  ERROR("Error binding socket - %s", fr_strerror());
374  return -1;
375  }
376  }
377 
378 
379  /*
380  * Set option 'receive timeout' on socket.
381  * Note: in case of a timeout, the error will be "Resource temporarily unavailable".
382  */
383  if (setsockopt(sockfd, SOL_SOCKET, SO_RCVTIMEO,
384  &fr_time_delta_to_timeval(timeout), sizeof(struct timeval)) == -1) {
385  ERROR("Failed setting socket timeout: %s", fr_syserror(errno));
386  return -1;
387  }
388 
389  if (setsockopt(sockfd, SOL_SOCKET, SO_BROADCAST, &on, sizeof(on)) < 0) {
390  ERROR("Can't set broadcast option: %s", fr_syserror(errno));
391  return -1;
392  }
393  packet->socket.fd = sockfd;
394 
395 #ifdef HAVE_LINUX_IF_PACKET_H
396  if (raw_mode) {
397  if (fr_dhcpv4_raw_packet_send(sockfd, &ll, packet) < 0) {
398  ERROR("Failed sending (fr_dhcpv4_raw_packet_send): %s", fr_syserror(errno));
399  return -1;
400  }
401  if (!reply_expected) return 0;
402 
403  *reply = fr_dhcpv4_recv_raw_loop(sockfd, &ll, packet);
404  if (!*reply) {
405  ERROR("Error receiving reply (fr_dhcpv4_recv_raw_loop)");
406  return -1;
407  }
408  } else
409 #endif
410  {
411  if (fr_dhcpv4_udp_packet_send(packet) < 0) {
412  ERROR("Failed sending: %s", fr_syserror(errno));
413  return -1;
414  }
415  if (!reply_expected) return 0;
416 
417  *reply = fr_dhcpv4_udp_packet_recv(sockfd);
418  if (!*reply) {
419  if (errno == EAGAIN) {
420  fr_strerror(); /* clear error */
421  ERROR("Timed out waiting for reply");
422  } else {
423  ERROR("Error receiving reply");
424  }
425  return -1;
426  }
427  }
428 
429  return 0;
430 }

434 {
435  char ip[16];
436  char pcap_filter[255];
437 
438  pcap = fr_pcap_init(NULL, iface, PCAP_INTERFACE_IN_OUT);
439  if (!pcap) {
440  ERROR("Failed creating pcap");

442  }
443 
444  if (fr_pcap_open(pcap) < 0) {
445  ERROR("Failed opening interface");
446  talloc_free(pcap);
447  return -1;
448  }
449

451  sprintf(pcap_filter, "udp and dst port %d", packet->socket.inet.src_port);
452

454  ERROR("Failing setting filter");
455  talloc_free(pcap);

457  }

462  return -1;

464 
465  if (!reply_expected) return 0;
466 
467  *reply = fr_dhcpv4_recv_raw_loop(pcap->fd,
468 #ifdef HAVE_LINUX_IF_PACKET_H

471  packet);
472 
473  if (!*reply) {
474  ERROR("Error receiving reply");

476  return -1;
477  }
478

701  if (packet->socket.inet.dst_ipaddr.af == AF_UNSPEC) packet->socket.inet.dst_ipaddr = server_ipaddr;

709  (argc >= 3) ? "'auto'" : "unspecified");
710  fr_exit(EXIT_FAILURE);
711  }

715  */
716  if ((packet->code == FR_DHCP_RELEASE) || (packet->code == FR_DHCP_DECLINE)) {
717  reply_expected = false;
718  }

726  }
727

730  */

733  fr_cursor_init(&cursor, &packet->vps);
734

736 
737  fr_dhcpv4_decode(packet, packet->data, packet->data_len, &cursor, &packet->code);
738  dhcp_packet_debug(packet, false);
739  }
740 
741 #ifdef HAVE_LIBPCAP
742  if (raw_mode) {
743  ret = send_with_pcap(&reply, packet);
744  } else
745 #endif
746  {
747  ret = send_with_socket(&reply, packet);
748  }
749

752  fr_cursor_init(&cursor, &reply->vps);
753  if (fr_dhcpv4_decode(reply, reply->data, reply->data_len, &cursor, &reply->code) < 0) {
754  ERROR("Failed decoding packet");
755  ret = -1;
756  }
757  dhcp_packet_debug(reply, true);
758  }
759 
760  fr_dhcpv4_global_free();
761  fr_dict_autofree(dhcpclient_dict);
762 
763  return ret < 0 ? 1 : 0;
764 }
