32 static bool chbind_build_response(request_t *request, CHBIND_REQ *chbind)
33 {
34  ssize_t slen;
35  size_t total;
36  uint8_t *ptr, *end;
37  fr_pair_t const *vp;
38  fr_cursor_t cursor;
39 
40  total = 0;
41  for (vp = fr_cursor_init(&cursor, &request->reply_pairs);
42  vp != NULL;
43  vp = fr_cursor_next(&cursor)) {
44  /*
45  * Skip things which shouldn't be in channel bindings.
46  */
47  if (vp->da->flags.internal || (!vp->da->flags.extra && vp->da->flags.subtype)) continue;
48  if (vp->da == attr_message_authenticator) continue;
49 
50  total += 2 + vp->vp_length;
51  }
52 
53  /*
54  * No attributes: just send a 1-byte response code.
55  */
56  if (!total) {
57  ptr = talloc_zero_array(chbind, uint8_t, 1);
58  } else {
59  ptr = talloc_zero_array(chbind, uint8_t, total + 4);
60  }
61  if (!ptr) return false;
62  chbind->response = (chbind_packet_t *) ptr;
63 
64  /*
65  * Set the response code. Default to "fail" if none was
66  * specified.
67  */
68  vp = fr_pair_find_by_da(&request->control_pairs, attr_chbind_response_code);
69  if (vp) {
70  ptr[0] = vp->vp_uint32;
71  } else {
72  ptr[0] = CHBIND_CODE_FAILURE;
73  }
74 
75  if (!total) return true; /* nothing to encode */
76 
77  /* Write the length field into the header */
78  ptr[1] = (total >> 8) & 0xff;
79  ptr[2] = total & 0xff;
80  ptr[3] = CHBIND_NSID_RADIUS;
81 
82  RDEBUG2("Sending chbind response: code %i", (int )(ptr[0]));
83  log_request_pair_list(L_DBG_LVL_1, request, request->reply_pairs, NULL);
84 
85  /* Encode the chbind attributes into the response */
86  ptr += 4;
87  end = ptr + total;
88 
89  fr_cursor_init(&cursor, &request->reply_pairs);
90  while ((vp = fr_cursor_current(&cursor)) && (ptr < end)) {
91  /*
92  * Skip things which shouldn't be in channel bindings.
93  * i.e. tagged, encrypted, or extended attributes
94  */
95  if (vp->da->flags.subtype) {
96  next:
97  fr_cursor_next(&cursor);
98  continue;
99  }
100  if (vp->da == attr_message_authenticator) goto next;
101 
102  slen = fr_radius_encode_pair(&FR_DBUFF_TMP(ptr, end), &cursor, NULL);
103  if (slen < 0) {
104  if (slen == PAIR_ENCODE_SKIPPED) goto next;
105 
106  RPERROR("Failed encoding chbind response");
107 
108  talloc_free(ptr);
109  return false;
110  }
111  ptr += slen;
112  }
113 
114  return true;
115 }

118 /*
119  * Parse channel binding packet to obtain data for a specific
120  * NSID.
121  *
122  * See:
123  * http://tools.ietf.org/html/draft-ietf-emu-chbind-13#section-5.3.2
124  */
125 static size_t chbind_get_data(chbind_packet_t const *packet,
126  int desired_nsid,
127  uint8_t const **data)
128 {
129  uint8_t const *ptr;
130  uint8_t const *end;
131 
132  if (packet->code != CHBIND_CODE_REQUEST) {
133  return 0;
134  }
135 
136  ptr = (uint8_t const *) packet;
137  end = ptr + talloc_array_length((uint8_t const *) packet);
138 
139  ptr++; /* skip the code at the start of the packet */
140  while (ptr < end) {
141  uint8_t nsid;
142  size_t length;
143 
144  /*
145  * Need room for length(2) + NSID + data.
146  */
147  if ((end - ptr) < 4) return 0;
148 
149  length = (ptr[0] << 8) | ptr[1];
150  if (length == 0) return 0;
151 
152  if ((ptr + length + 3) > end) return 0;
153 
154  nsid = ptr[2];
155  if (nsid == desired_nsid) {
156  ptr += 3;
157  *data = ptr;
158  return length;
159  }
160 
161  ptr += 3 + length;
162  }
163 
164  return 0;
165 }

168 FR_CODE chbind_process(request_t *request, CHBIND_REQ *chbind)
169 {
170  FR_CODE code;
171  rlm_rcode_t rcode;
172  request_t *fake = NULL;
173  uint8_t const *attr_data;
174  size_t data_len = 0;
175  fr_pair_t *vp;
176 
177  /* check input parameters */
178  fr_assert((request != NULL) &&
179  (chbind != NULL) &&
180  (chbind->request != NULL) &&
181  (chbind->response == NULL));
182 
183  /* Set-up the fake request */
184  fake = request_alloc_fake(request, NULL);
185  MEM(fr_pair_add_by_da(fake->packet, &vp, &fake->request_pairs, attr_freeradius_proxied_to) >= 0);
186  fr_pair_value_from_str(vp, "127.0.0.1", sizeof("127.0.0.1"), '\0', false);
187 
188  /* Add the username to the fake request */
189  if (chbind->username) {
190  vp = fr_pair_copy(fake->packet, chbind->username);
191  fr_pair_add(&fake->request_pairs, vp);
192  }
193 
194  /*
195  * Maybe copy the State over, too?
196  */
197 
198  /* Add the channel binding attributes to the fake packet */
199  data_len = chbind_get_data(chbind->request, CHBIND_NSID_RADIUS, &attr_data);
200  if (data_len) {
201  fr_cursor_t cursor;
202 
203  fr_assert(data_len <= talloc_array_length((uint8_t const *) chbind->request));
204 
205  fr_cursor_init(&cursor, &fake->request_pairs);
206  while (data_len > 0) {
207  ssize_t attr_len;
208 
209  attr_len = fr_radius_decode_pair(fake->packet, &cursor, dict_radius,
210  attr_data, data_len, NULL);
211  if (attr_len <= 0) {
212  /*
213  * If fr_radius_decode_pair fails, return NULL string for
214  * channel binding response.
215  */
216  talloc_free(fake);
217 
218  return FR_CODE_ACCESS_ACCEPT;
219  }
220  attr_data += attr_len;
221  data_len -= attr_len;
222  }
223  }
224 
225  /*
226  * Set virtual server based on configuration for channel
227  * bindings, this is hard-coded for now.
228  */
229  fake->server_cs = virtual_server_find("channel_bindings");
230  fake->packet->code = FR_CODE_ACCESS_REQUEST;
231 
232  rad_virtual_server(&rcode, fake);
233  switch (rcode) {
234  /* If the virtual server succeeded, build a reply */
235  case RLM_MODULE_OK:
236  case RLM_MODULE_HANDLED:
237  if (chbind_build_response(fake, chbind)) {
238  code = FR_CODE_ACCESS_ACCEPT;
239  break;
240  }
241  FALL_THROUGH;
242 
243  /* If we got any other response from the virtual server, it maps to a reject */
244  default:
245  code = FR_CODE_ACCESS_REJECT;
246  break;
247  }
248 
249  talloc_free(fake);
250 
251  return code;
252 }

258 chbind_packet_t *eap_chbind_vp2packet(TALLOC_CTX *ctx, fr_pair_t *vps)
259 {
260  size_t length;
261  uint8_t *ptr;
262  fr_pair_t *vp;
263  chbind_packet_t *packet;
264  fr_cursor_t cursor;
265 
266  if (!fr_cursor_iter_by_da_init(&cursor, &vps, attr_eap_channel_binding_message)) return NULL;
267 
268  /*
269  * Compute the total length of the channel binding data.
270  */
271  length = 0;
272  for (vp = fr_cursor_current(&cursor);
273  vp;
274  vp = fr_cursor_next(&cursor)) {
275  length += vp->vp_length;
276  }
277 
278  if (length < 4) {
279  DEBUG("Invalid length %u for channel binding data", (unsigned int) length);
280  return NULL;
281  }
282 
283  /*
284  * Now that we know the length, allocate memory for the packet.
285  */
286  ptr = talloc_zero_array(ctx, uint8_t, length);
287  if (!ptr) return NULL;
288 
289  /*
290  * Copy the data over to our packet.
291  */
292  packet = (chbind_packet_t *) ptr;
293  for (vp = fr_cursor_head(&cursor);
294  vp != NULL;
295  vp = fr_cursor_next(&cursor)) {
296  memcpy(ptr, vp->vp_octets, vp->vp_length);
297  ptr += vp->vp_length;
298  }
299 
300  return packet;
301 }
