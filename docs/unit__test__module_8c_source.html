142 static request_t *request_from_file(TALLOC_CTX *ctx, FILE *fp, fr_event_list_t *el, RADCLIENT *client)
143 {
144  fr_pair_t *vp;
145  request_t *request;
146  fr_cursor_t cursor;
147 
148  static int number = 0;
149 
150  /*
151  * Create and initialize the new request.
152  */
153  request = request_local_alloc(ctx);
154 
155  /*
156  * FIXME - Should be less RADIUS centric, but everything
157  * else assumes RADIUS at the moment so we can fix this later.
158  */
159  request->dict = fr_dict_by_protocol_name(PROTOCOL_NAME);
160  if (!request->dict) {
161  ERROR("%s dictionary failed to load", PROTOCOL_NAME);
162  talloc_free(request);
163  return NULL;
164  }
165 
166  request->packet = fr_radius_alloc(request, false);
167  if (!request->packet) {
168  ERROR("No memory");
169  talloc_free(request);
170  return NULL;
171  }
172  request->packet->timestamp = fr_time();
173 
174  request->reply = fr_radius_alloc(request, false);
175  if (!request->reply) {
176  ERROR("No memory");
177  talloc_free(request);
178  return NULL;
179  }
180 
181  request->client = client;
182 
183  request->number = number++;
184  request->name = talloc_typed_asprintf(request, "%" PRIu64, request->number);
185 
186  request->master_state = REQUEST_ACTIVE;
187  request->server_cs = virtual_server_find("default");
188  fr_assert(request->server_cs != NULL);
189 
190  request->config = main_config;
191 
192  /*
193  * Read packet from fp
194  */
195  if (fr_pair_list_afrom_file(request->packet, dict_protocol, &request->request_pairs, fp, &filedone) < 0) {
196  fr_perror("%s", main_config->name);
197  talloc_free(request);
198  return NULL;
199  }
200 
201  /*
202  * Set the defaults for IPs, etc.
203  */
204  request->packet->code = access_request;
205 
206  request->packet->socket.proto = IPPROTO_UDP;
207  request->packet->socket.inet.src_ipaddr.af = AF_INET;
208  request->packet->socket.inet.src_ipaddr.prefix = 32;
209  request->packet->socket.inet.src_ipaddr.addr.v4.s_addr = htonl(INADDR_LOOPBACK);
210  request->packet->socket.inet.src_port = 18120;
211 
212  request->packet->socket.inet.dst_ipaddr.af = AF_INET;
213  request->packet->socket.inet.dst_ipaddr.prefix = 32;
214  request->packet->socket.inet.dst_ipaddr.addr.v4.s_addr = htonl(INADDR_LOOPBACK);
215  request->packet->socket.inet.dst_port = 1812;
216 
217  for (vp = fr_cursor_init(&cursor, &request->request_pairs);
218  vp;
219  vp = fr_cursor_next(&cursor)) {
220  /*
221  * Double quoted strings get marked up as xlat expansions,
222  * but we don't support that here.
223  */
224  if (vp->type == VT_XLAT) {
225  vp->vp_strvalue = vp->xlat;
226  vp->xlat = NULL;
227  vp->type = VT_DATA;
228  }
229 
230  if (vp->da == attr_packet_type) {
231  request->packet->code = vp->vp_uint32;
232  } else if (vp->da == attr_packet_dst_port) {
233  request->packet->socket.inet.dst_port = vp->vp_uint16;
234  } else if ((vp->da == attr_packet_dst_ip_address) ||
235  (vp->da == attr_packet_dst_ipv6_address)) {
236  memcpy(&request->packet->socket.inet.dst_ipaddr, &vp->vp_ip, sizeof(request->packet->socket.inet.dst_ipaddr));
237  } else if (vp->da == attr_packet_src_port) {
238  request->packet->socket.inet.src_port = vp->vp_uint16;
239  } else if ((vp->da == attr_packet_src_ip_address) ||
240  (vp->da == attr_packet_src_ipv6_address)) {
241  memcpy(&request->packet->socket.inet.src_ipaddr, &vp->vp_ip, sizeof(request->packet->socket.inet.src_ipaddr));
242  }
243  } /* loop over the VP's we read in */
244 
245  /*
246  * "0" is uniformly the "bad packet" type.
247  */
248  if (!request->packet->code) {
249  fr_strerror_printf("No 'Packet-Type' was found in the request list. Cannot send unknown packet");
250  return NULL;
251  }
252 
253  if (fr_debug_lvl) {
254  for (vp = fr_cursor_init(&cursor, &request->request_pairs);
255  vp;
256  vp = fr_cursor_next(&cursor)) {
257  /*
258  * Take this opportunity to verify all the fr_pair_ts are still valid.
259  */
260  if (!talloc_get_type(vp, fr_pair_t)) {
261  ERROR("Expected fr_pair_t pointer got \"%s\"", talloc_get_name(vp));
262 
263  fr_log_talloc_report(vp);
264  fr_assert(0);
265  }
266 
267  fr_log(&default_log, L_DBG, __FILE__, __LINE__, "%pP", vp);
268  }
269  }
270 
271  /*
272  * Build the reply template from the request.
273  */
274  fr_socket_addr_swap(&request->reply->socket, &request->packet->socket);
275 
276  request->reply->id = request->packet->id;
277  request->reply->code = 0; /* UNKNOWN code */
278  memcpy(request->reply->vector, request->packet->vector, sizeof(request->reply->vector));
279  request->reply_pairs = NULL;
280  request->reply->data = NULL;
281  request->reply->data_len = 0;
282 
283  /*
284  * Debugging
285  */
286  request->log.dst = talloc_zero(request, log_dst_t);
287  request->log.dst->func = vlog_request;
288  request->log.dst->uctx = &default_log;
289  request->log.lvl = fr_debug_lvl;
290 
291  fr_request_async_bootstrap(request, el);
292 
293  return request;
294 }

330 /*
331  * Read a file compose of xlat's and expected results
332  */
333 static bool do_xlats(char const *filename, FILE *fp)
334 {
335  int lineno = 0;
336  ssize_t len;
337  char *p;
338  char input[8192];
339  char output[8192];
340  request_t *request;
341 
342  /*
343  * Create and initialize the new request.
344  */
345  request = request_alloc(NULL);
346 
347  request->log.dst = talloc_zero(request, log_dst_t);
348  request->log.dst->func = vlog_request;
349  request->log.dst->uctx = &default_log;
350 
351  request->master_state = REQUEST_ACTIVE;
352  request->server_cs = virtual_server_find("default");
353  fr_assert(request->server_cs != NULL);
354 
355  request->config = main_config;
356 
357  request->log.lvl = fr_debug_lvl;
358  output[0] = '\0';
359 
360  while (fgets(input, sizeof(input), fp) != NULL) {
361  lineno++;
362 
363  /*
364  * Ignore blank lines and comments
365  */
366  p = input;
367  fr_skip_whitespace(p);
368 
369  if (*p < ' ') continue;
370  if (*p == '#') continue;
371 
372  p = strchr(p, '\n');
373  if (!p) {
374  if (!feof(fp)) {
375  fprintf(stderr, "Line %d too long in %s\n",
376  lineno, filename);
377  TALLOC_FREE(request);
378  return false;
379  }
380  } else {
381  *p = '\0';
382  }
383 
384  /*
385  * Look for "xlat"
386  */
387  if (strncmp(input, "xlat ", 5) == 0) {
388  ssize_t slen;
389  TALLOC_CTX *xlat_ctx = talloc_init_const("xlat");
390  char *fmt = talloc_typed_strdup(xlat_ctx, input + 5);
391  xlat_exp_t *head = NULL;
392  fr_sbuff_parse_rules_t p_rules = { .escapes = &fr_value_unescape_double };
393 
394  slen = xlat_tokenize_ephemeral(xlat_ctx, &head, NULL,
395  &FR_SBUFF_IN(fmt, talloc_array_length(fmt) - 1), &p_rules, NULL);
396  if (slen <= 0) {
397  talloc_free(xlat_ctx);
398  snprintf(output, sizeof(output), "ERROR offset %d '%s'", (int) -slen,
399  fr_strerror());
400  continue;
401  }
402 
403  if (input[slen + 5] != '\0') {
404  talloc_free(xlat_ctx);
405  snprintf(output, sizeof(output), "ERROR offset %d 'Too much text' ::%s::",
406  (int) slen, input + slen + 5);
407  continue;
408  }
409 
410  len = xlat_eval_compiled(output, sizeof(output), request, head, NULL, NULL);
411  if (len < 0) {
412  talloc_free(xlat_ctx);
413  snprintf(output, sizeof(output), "ERROR expanding xlat: %s", fr_strerror());
414  continue;
415  }
416 
417  TALLOC_FREE(xlat_ctx); /* also frees 'head' */
418  continue;
419  }
420 
421  /*
422  * Look for "data".
423  */
424  if (strncmp(input, "data ", 5) == 0) {
425  if (strcmp(input + 5, output) != 0) {
426  fprintf(stderr, "Mismatch at line %d of %s\n\tgot : %s\n\texpected : %s\n",
427  lineno, filename, output, input + 5);
428  TALLOC_FREE(request);
429  return false;
430  }
431  continue;
432  }
433 
434  fprintf(stderr, "Unknown keyword in %s[%d]\n", filename, lineno);
435  TALLOC_FREE(request);
436  return false;
437  }
438 
439  TALLOC_FREE(request);
440  return true;
441 }

464 static void request_run(fr_event_list_t *el, request_t *request)
465 {
466  rlm_rcode_t rcode;
467  module_method_t process;
468  void *inst;
469  fr_dict_enum_t *dv;
470  fr_heap_t *backlog;
471  request_t *child;
472 
473  dv = fr_dict_enum_by_value(attr_packet_type, fr_box_uint32(request->packet->code));
474  if (!dv) return;
475 
476  if (virtual_server_get_process_by_name(request->server_cs, dv->name, &process, &inst) < 0) {
477  REDEBUG("Cannot run virtual server '%s' - %s", cf_section_name2(request->server_cs), fr_strerror());
478  return;
479  }
480 
481  MEM(backlog = fr_heap_talloc_alloc(request, fr_pointer_cmp, request_t, runnable_id));
482  request->backlog = backlog;
483  request->el = el;
484 
485  if (process(&rcode, &(module_ctx_t){ .instance = inst }, request) != UNLANG_ACTION_YIELD) goto done;
486 
487  while (true) {
488  bool wait_for_event;
489  int num_events;
490 
491  wait_for_event = (fr_heap_num_elements(backlog) == 0);
492 
493  corral:
494  num_events = fr_event_corral(el, fr_time(), wait_for_event);
495  if (num_events < 0) {
496  PERROR("Failed retrieving events");
497  break;
498  }
499 
500  /*
501  * Service outstanding events.
502  */
503  if (num_events > 0) fr_event_service(el);
504 
505  /*
506  * The request is not runnable. Wait for events.
507  *
508  * Note that we do NOT run any child requests.
509  * There may be child requests in the backlog,
510  * but we just ignore them.
511  */
512  if (request->runnable_id < 0) {
513  wait_for_event = true;
514  goto corral;
515  }
516 
517  /*
518  * Run the parent request in preference to any
519  * child requests.
520  */
521  (void) fr_heap_extract(backlog, request);
522 
523  if (process(&rcode, &(module_ctx_t){ .instance = inst }, request) != UNLANG_ACTION_YIELD) break;
524  }
525 
526 done:
527  /*
528  * Parallel-detach creates detached, but runnable
529  * children. We don't want to run them, so we just clean
530  * them up here.
531  */
532  while ((child = fr_heap_pop(backlog)) != NULL) talloc_free(child);
533 
534  /*
535  * We do NOT run detached child requests. We just ignore
536  * them.
537  */
538  talloc_free(backlog);
539 }

541 static request_t *request_clone(request_t *old)
542 {
543  request_t *request;
544 
545  request = request_alloc(NULL);
546  if (!request) return NULL;
547 
548  if (!request->packet) request->packet = fr_radius_alloc(request, false);
549  if (!request->reply) request->reply = fr_radius_alloc(request, false);
550 
551  memcpy(request->packet, old->packet, sizeof(*request->packet));
552  (void) fr_pair_list_copy(request->packet, &request->request_pairs, &old->packet->vps);
553  request->packet->timestamp = fr_time();
554  request->number = old->number++;
555 
556  return request;
557 }
