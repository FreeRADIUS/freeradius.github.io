37 static ssize_t _fr_mkdir(int *fd_out, char const *path, mode_t mode, fr_mkdir_func_t func, void *uctx)
38 {
39  int ret, fd;
40  char *p;
41 
42  /*
43  * Try to make the path. If it exists, chmod it.
44  * If a path doesn't exist, that's OK. Otherwise
45  * return with an error.
46  *
47  * Directories permissions are initially set so
48  * that only we should have access. This prevents
49  * an attacker removing them and swapping them
50  * out for a link to somewhere else.
51  * We change them to the correct permissions later.
52  */
53  ret = mkdir(path, 0700);
54  if (ret >= 0) {
55  fd = open(path, O_DIRECTORY);
56  if (fd < 0) {
57  fr_strerror_printf("Failed opening directory we created: %s",
58  fr_syserror(errno));
59  mkdir_error:
60  p = strrchr(path, FR_DIR_SEP);
61  if (!p) return 0;
62 
63  return -(p - path);
64  }
65 
66  if (fchmod(fd, mode) < 0) {
67  fr_strerror_printf("Failed setting permissions on "
68  "directory we created: %s", fr_syserror(errno));
69  close(fd);
70  goto mkdir_error;
71  }
72  *fd_out = fd;
73  return PATH_MAX;
74  }
75 
76  /*
77  * EEXIST is only OK when we're calling
78  * mkdir on the whole path, and it exists
79  * which should have been caught by
80  * fr_mkdir before calling this function.
81  */
82  if (errno != ENOENT) {
83  fr_strerror_printf("Unexpected error creating "
84  "directory: %s", fr_syserror(errno));
85  goto mkdir_error;
86  }
87 
88  /*
89  * A component in the path path doesn't
90  * exist. Look for the LAST path name. Try
91  * to create that. If there's an error, we leave
92  * the path path as the one at which the
93  * error occured.
94  */
95  p = strrchr(path, FR_DIR_SEP);
96  if (!p || (p == path)) return -(p - path);
97 
98  *p = '\0';
99  if (_fr_mkdir(fd_out, path, mode, func, uctx) < 0) return -(p - path);
100 
101  /*
102  * At this point *fd_out, should be an FD
103  * for the containing directory.
104  */
105  if (mkdirat(*fd_out, p + 1, 0700) < 0) {
106  fr_strerror_printf("Failed creating directory: %s", fr_syserror(errno));
107  mkdirat_error:
108  close(*fd_out);
109  *fd_out = -1;
110  return -(p - path);
111  }
112 
113  fd = openat(*fd_out, p + 1, O_DIRECTORY);
114  if (fd < 0) {
115  fr_strerror_printf("Failed opening directory we "
116  "created: %s", fr_syserror(errno));
117  goto mkdirat_error;
118  }
119 
120  if (fchmod(fd, mode) < 0) {
121  fr_strerror_printf("Failed setting permissions on "
122  "directory we created: %s", fr_syserror(errno));
123  goto mkdirat_error;
124  }
125  *p = FR_DIR_SEP;
126 
127  /*
128  * Call the user function
129  */
130  if (func && (func(fd, path, uctx) < 0)) {
131  fr_strerror_printf_push("Callback failed processing directory \"%s\"", path);
132  goto mkdirat_error;
133  }
134 
135  /*
136  * Swap active *fd_out to point to the dir
137  * we just created.
138  */
139  close(*fd_out);
140  *fd_out = fd;
141 
142  return PATH_MAX;
143 }

145 /** Create directories that are missing in the specified path
146  *
147  * @param[out] fd_out If not NULL, will contain a file descriptor
148  * for the deepest path component created.
149  * @param[in] path to populate with directories.
150  * @param[in] len Length of the path string.
151  * @param[in] mode for new directories.
152  * @param[in] func to call each time a new directory is created.
153  * @param[in] uctx to pass to func.
154  * @return
155  * - >0 on success.
156  * - <= 0 on failure. Negative offset pointing to the
157  * path separator of the path component that caused the error.
158  */
159 ssize_t fr_mkdir(int *fd_out, char const *path, ssize_t len, mode_t mode, fr_mkdir_func_t func, void *uctx)
160 {
161  char *our_path;
162  int fd = -1;
163  ssize_t slen;
164 
165  if (len < 0) len = strlen(path);
166  if (len == 0) return 0;
167 
168  /*
169  * Fast path (har har)
170  *
171  * Avoids duping the input for the
172  * common case.
173  */
174  fd = open(path, O_DIRECTORY);
175  if (fd >= 0) goto done;
176 
177  /*
178  * Dup the amount of input path
179  * we need.
180  */
181  our_path = talloc_bstrndup(NULL, path, (size_t)len);
182  if (!our_path) {
183  fr_strerror_printf("Out of memory");
184  return -1;
185  }
186 
187  /*
188  * Call the recursive function to
189  * create any missing dirs in the
190  * specified path.
191  */
192  slen = _fr_mkdir(&fd, our_path, mode, func, uctx);
193  talloc_free(our_path);
194  if (slen <= 0) return slen;
195 
196 done:
197  if (fd_out) {
198  *fd_out = fd;
199  } else {
200  close(fd);
201  }
202 
203  return len;
204 }

206 /** Convenience wrapper around realpath
207  *
208  * Wraps realpath, but takes a path with an explicit length, and returns
209  * the result in a talloced buffer.
210  *
211  * On error, errno is set, and the string version of the error is
212  * available with fr_strerror().
213  *
214  * @param[in] ctx in which to allocate the result.
215  * @param[in] path To convert to an absolute path.
216  * @param[in] len How much of 'path' to read. If < 0, then
217  * the entire path will be used.
218  * @return
219  * - NULL on error.
220  * - The absolute version of the input path on success.
221  */
222 char *fr_realpath(TALLOC_CTX *ctx, char const *path, ssize_t len)
223 {
224  char *tmp_path = NULL, *abs_path, *talloc_abs_path;
225 
226  if (len > 0) path = tmp_path = talloc_bstrndup(NULL, path, (size_t)len);
227 
228  abs_path = realpath(path, NULL);
229  if (!abs_path) {
230  fr_strerror_printf("Failed resolving path \"%pV\": %s",
231  fr_box_strvalue_buffer(path), fr_syserror(errno));
232  talloc_free(tmp_path);
233  return NULL;
234  }
235 
236  talloc_abs_path = talloc_strdup(ctx, abs_path);
237  free(abs_path);
238  if (!talloc_abs_path) {
239  fr_strerror_printf("Out of Memory");
240  return NULL;
241  }
242 
243  return talloc_abs_path;
244 }

246 /** Create an empty file
247  *
248  * @param[out] fd_out If not NULL, will contain a file descriptor
249  * for the file we just opened.
250  * @param[in] filename path to file.
251  * @param[in] mode Specifies the file mode bits be applied.
252  * @param[in] mkdir Whether we should create directories
253  * for any missing path components.
254  * @param[in] dir_mode Mode of any directories created.
255  * @return
256  * - >0 on success.
257  * - <= 0 on failure. Error available in error stack (use fr_strerror())
258  */
259 ssize_t fr_touch(int *fd_out, char const *filename, mode_t mode, bool mkdir, mode_t dir_mode) {
260  int fd;
261 
262  fd = open(filename, O_WRONLY | O_CREAT, mode);
263  if (fd < 0) {
264  ssize_t slen = 0;
265  char *q;
266 
267  if (mkdir && (errno == ENOENT) && (q = strrchr(filename, FR_DIR_SEP))) {
268  int dir_fd;
269 
270  slen = fr_mkdir(&dir_fd, filename, q - filename, dir_mode, NULL, NULL);
271  if (slen <= 0) return slen;
272 
273  fd = openat(dir_fd, q + 1, O_WRONLY | O_CREAT, mode);
274  if (fd >= 0) {
275  close(dir_fd);
276  close(fd);
277  return strlen(filename);
278  }
279  close(dir_fd);
280  slen = -(q - filename);
281  }
282  fr_strerror_printf("Failed creating file: %s", fr_syserror(errno));
283  return slen;
284  }
285 
286  if (fd_out) {
287  *fd_out = fd;
288  } else {
289  close(fd);
290  }
291 
292  return strlen(filename);
293 }
