242 /** Iterate over range of IP addresses
243  *
244  * Mutates the ipaddr passed in, adding one to the prefix bits on each call.
245  *
246  * @param[in,out] ipaddr to increment.
247  * @param[in] end ipaddr to stop at.
248  * @param[in] prefix Length of the prefix.
249  * @return
250  * - true if the prefix bits are not high (continue).
251  * - false if the prefix bits are high (stop).
252  */
253 static bool ipaddr_next(fr_ipaddr_t *ipaddr, fr_ipaddr_t const *end, uint8_t prefix)
254 {
255  switch (ipaddr->af) {
256  default:
257  case AF_UNSPEC:
258  fr_assert(0);
259  return false;
260 
261  case AF_INET6:
262  {
263  uint128_t ip_curr, ip_end;
264 
265  if (!fr_cond_assert((prefix > 0) && (prefix <= 128))) return false;
266 
267  /* Don't be tempted to cast */
268  memcpy(&ip_curr, ipaddr->addr.v6.s6_addr, sizeof(ip_curr));
269  memcpy(&ip_end, end->addr.v6.s6_addr, sizeof(ip_curr));
270 
271  ip_curr = ntohlll(ip_curr);
272  ip_end = ntohlll(ip_end);
273 
274  /* We're done */
275  if (uint128_eq(ip_curr, ip_end)) return false;
276 
277  /* Increment the prefix */
278  ip_curr = uint128_add(ip_curr, uint128_lshift(uint128_new(0, 1), (128 - prefix)));
279  ip_curr = htonlll(ip_curr);
280  memcpy(&ipaddr->addr.v6.s6_addr, &ip_curr, sizeof(ipaddr->addr.v6.s6_addr));
281  return true;
282  }
283 
284  case AF_INET:
285  {
286  uint32_t ip_curr, ip_end;
287 
288  if (!fr_cond_assert((prefix > 0) && (prefix <= 32))) return false;
289 
290  ip_curr = ntohl(ipaddr->addr.v4.s_addr);
291  ip_end = ntohl(end->addr.v4.s_addr);
292 
293  /* We're done */
294  if (ip_curr == ip_end) return false;
295 
296  /* Increment the prefix */
297  ip_curr += 1 << (32 - prefix);
298  ipaddr->addr.v4.s_addr = htonl(ip_curr);
299  return true;
300  }
301  }
302 }

308 static int driver_do_lease(void *out, void *instance, ippool_tool_operation_t const *op,
309  redis_ippool_queue_t enqueue, redis_ippool_process_t process)
310 {
311  redis_driver_conf_t *inst = talloc_get_type_abort(instance, redis_driver_conf_t);
312 
313  int i;
314  bool more = true;
315  fr_redis_conn_t *conn;
316 
317  fr_redis_cluster_state_t state;
318  fr_redis_rcode_t status;
319 
320  fr_ipaddr_t ipaddr = op->start;
321  fr_redis_rcode_t s_ret = REDIS_RCODE_SUCCESS;
322  request_t *request;
323  redisReply **replies = NULL;
324 
325  unsigned int pipelined = 0;
326 
327  request = request_alloc(inst);
328  while (more) {
329  fr_ipaddr_t acked = ipaddr; /* Record our progress */
330  size_t reply_cnt = 0;
331 
332  for (s_ret = fr_redis_cluster_state_init(&state, &conn, inst->cluster, request,
333  op->pool, op->pool_len, false);
334  s_ret == REDIS_RCODE_TRY_AGAIN;
335  s_ret = fr_redis_cluster_state_next(&state, &conn, inst->cluster, request, status, &replies[0])) {
336  more = true; /* Reset to true, may have errored last loop */
337  status = REDIS_RCODE_SUCCESS;
338 
339  /*
340  * If we got a redirect, start back at the beginning of the block.
341  */
342  ipaddr = acked;
343 
344  for (i = 0; (i < MAX_PIPELINED) && more; i++, more = ipaddr_next(&ipaddr, &op->end,
345  op->prefix)) {
346  int enqueued;
347 
348  enqueued = enqueue(inst, conn, op->pool, op->pool_len,
349  op->range, op->range_len, &ipaddr, op->prefix);
350  if (enqueued < 0) break;
351  pipelined += enqueued;
352  }
353 
354  if (!replies) replies = talloc_zero_array(inst, redisReply *, pipelined);
355  if (!replies) return -1;
356 
357  reply_cnt = fr_redis_pipeline_result(&pipelined, &status, replies,
358  talloc_array_length(replies), conn);
359  for (i = 0; (size_t)i < reply_cnt; i++) fr_redis_reply_print(L_DBG_LVL_3,
360  replies[i], request, i);
361  }
362  if (s_ret != REDIS_RCODE_SUCCESS) {
363  fr_redis_pipeline_free(replies, reply_cnt);
364  talloc_free(replies);
365  talloc_free(request);
366  return -1;
367  }
368 
369  if (process) {
370  fr_ipaddr_t to_process = acked;
371 
372  for (i = 0; (size_t)i < reply_cnt; i++) {
373  int ret;
374 
375  ret = process(out, &to_process, replies[i]);
376  if (ret < 0) continue;
377  ipaddr_next(&to_process, &op->end, op->prefix);
378  }
379  }
380  fr_redis_pipeline_free(replies, reply_cnt);
381  TALLOC_FREE(replies);
382  }
383  talloc_free(request);
384 
385  return 0;
386 }

391 static int _driver_show_lease_process(void *out, fr_ipaddr_t const *ipaddr, redisReply const *reply)
392 {
393  size_t existing;
394  ippool_tool_lease_t ***modified = out;
395  ippool_tool_lease_t *lease;
396 
397  if (!*modified) *modified = talloc_array(NULL, ippool_tool_lease_t *, 1);
398 
399  /*
400  * The exec command is the only one that produces an array.
401  */
402  if (reply->type != REDIS_REPLY_ARRAY) return -1;
403  if (reply->elements < 4) return -1;
404 
405  if (reply->element[0]->type == REDIS_REPLY_NIL) return 0; /* A nil result (IP didn't exist) */
406  if (reply->element[0]->type != REDIS_REPLY_STRING) return -1; /* Something bad */
407  lease = talloc_zero(*modified, ippool_tool_lease_t);
408  lease->ipaddr = *ipaddr;
409  lease->next_event = (time_t)strtoull(reply->element[0]->str, NULL, 10);
410 
411  if (reply->element[1]->type == REDIS_REPLY_STRING) {
412  lease->device = talloc_memdup(lease, reply->element[1]->str, reply->element[1]->len);
413  lease->device_len = reply->element[1]->len;
414  }
415  if (reply->element[2]->type == REDIS_REPLY_STRING) {
416  lease->gateway = talloc_memdup(lease, reply->element[2]->str, reply->element[2]->len);
417  lease->gateway_len = reply->element[2]->len;
418  }
419  if (reply->element[3]->type == REDIS_REPLY_STRING) {
420  lease->range = talloc_memdup(lease, reply->element[3]->str, reply->element[3]->len);
421  lease->range_len = reply->element[3]->len;
422  }
423 
424  /*
425  * Grow the result array...
426  */
427  existing = talloc_array_length(*modified);
428  MEM(*modified = talloc_realloc(NULL, *modified, ippool_tool_lease_t *, existing + 1));
429  (*modified)[existing - 1] = lease;
430 
431  return 0;
432 }

710 /** Return the pools available across the cluster
711  *
712  * @param[in] ctx to allocate range names in.
713  * @param[out] out Array of pool names.
714  * @param[in] instance Driver specific instance data.
715  * @return
716  * - < 0 on failure.
717  * - >= 0 the number of ranges in the array we allocated.
718  */
719 static ssize_t driver_get_pools(TALLOC_CTX *ctx, uint8_t **out[], void *instance)
720 {
721  fr_socket_t *master;
722  size_t k;
723  ssize_t ret, i, used = 0;
724  fr_redis_conn_t *conn = NULL;
725  redis_driver_conf_t *inst = talloc_get_type_abort(instance, redis_driver_conf_t);
726  uint8_t key[IPPOOL_MAX_POOL_KEY_SIZE];
727  uint8_t *key_p = key;
728  request_t *request;
729  uint8_t **result;
730 
731  request = request_alloc(inst);
732 
733  IPPOOL_BUILD_KEY(key, key_p, "*}:pool", 1);
734 
735  *out = NULL; /* Initialise output pointer */
736 
737  /*
738  * Get the addresses of all masters in the pool
739  */
740  ret = fr_redis_cluster_node_addr_by_role(ctx, &master, inst->cluster, true, false);
741  if (ret <= 0) {
742  result = NULL;
743  return ret;
744  }
745 
746  result = talloc_zero_array(ctx, uint8_t *, 1);
747  if (!result) {
748  ERROR("Failed allocating array of pool names");
749  talloc_free(master);
750  return -1;
751  }
752 
753  /*
754  * Iterate over the masters, getting the pools on each
755  */
756  for (i = 0; i < ret; i++) {
757  fr_pool_t *pool;
758  redisReply *reply;
759  char const *p;
760  size_t len;
761  char cursor[19] = "0";
762 
763  if (fr_redis_cluster_pool_by_node_addr(&pool, inst->cluster, &master[i], false) < 0) {
764  ERROR("Failed retrieving pool for node");
765  error:
766  TALLOC_FREE(result);
767  talloc_free(master);
768  talloc_free(request);
769  return -1;
770  }
771 
772  conn = fr_pool_connection_get(pool, request);
773  if (!conn) goto error;
774  do {
775  /*
776  * Break up the scan so we don't block any single
777  * Redis node too long.
778  */
779  reply = redisCommand(conn->handle, "SCAN %s MATCH %b COUNT 20", cursor, key, key_p - key);
780  if (!reply) {
781  ERROR("Failed reading reply");
782  fr_pool_connection_release(pool, request, conn);
783  goto error;
784  }
785  fr_redis_reply_print(L_DBG_LVL_3, reply, request, 0);
786  if (fr_redis_command_status(conn, reply) != REDIS_RCODE_SUCCESS) {
787  PERROR("Error retrieving keys %s", cursor);
788 
789  reply_error:
790  fr_pool_connection_release(pool, request, conn);
791  fr_redis_reply_free(&reply);
792  goto error;
793  }
794 
795  if (reply->type != REDIS_REPLY_ARRAY) {
796  ERROR("Failed retrieving result, expected array got %s",
797  fr_table_str_by_value(redis_reply_types, reply->type, "<UNKNOWN>"));
798 
799  goto reply_error;
800  }
801 
802  if (reply->elements != 2) {
803  ERROR("Failed retrieving result, expected array with two elements, got %zu elements",
804  reply->elements);
805  fr_redis_reply_free(&reply);
806  goto reply_error;
807  }
808 
809  if (reply->element[0]->type != REDIS_REPLY_STRING) {
810  ERROR("Failed retrieving result, expected string got %s",
811  fr_table_str_by_value(redis_reply_types, reply->element[0]->type, "<UNKNOWN>"));
812  goto reply_error;
813  }
814 
815  if (reply->element[1]->type != REDIS_REPLY_ARRAY) {
816  ERROR("Failed retrieving result, expected array got %s",
817  fr_table_str_by_value(redis_reply_types, reply->element[1]->type, "<UNKNOWN>"));
818  goto reply_error;
819  }
820 
821  if ((talloc_array_length(result) - used) < reply->element[1]->elements) {
822  MEM(result = talloc_realloc(ctx, result, uint8_t *,
823  used + reply->element[1]->elements));
824  if (!result) {
825  ERROR("Failed expanding array of pool names");
826  goto reply_error;
827  }
828  }
829  strlcpy(cursor, reply->element[0]->str, sizeof(cursor));
830 
831  for (k = 0; k < reply->element[1]->elements; k++) {
832  redisReply *pool_key = reply->element[1]->element[k];
833 
834  /*
835  * Skip over things which are not pool names
836  */
837  if (pool_key->len < 7) continue; /* { + [<name>] + }:pool */
838 
839  if ((pool_key->str[0]) != '{') continue;
840  p = memchr(pool_key->str + 1, '}', pool_key->len - 1);
841  if (!p) continue;
842 
843  len = (pool_key->len - ((p + 1) - pool_key->str));
844  if (len != (sizeof(IPPOOL_POOL_KEY) - 1) + 1) continue;
845  if (memcmp(p + 1, ":" IPPOOL_POOL_KEY, (sizeof(IPPOOL_POOL_KEY) - 1) + 1) != 0) {
846  continue;
847  }
848 
849  /*
850  * String between the curly braces is the pool name
851  */
852  result[used++] = talloc_memdup(result, pool_key->str + 1, (p - pool_key->str) - 1);
853  }
854 
855  fr_redis_reply_free(&reply);
856  } while (!((cursor[0] == '0') && (cursor[1] == '\0'))); /* Cursor value of 0 means no more results */
857 
858  fr_pool_connection_release(pool, request, conn);
859  }
860 
861  if (used == 0) {
862  *out = NULL;
863  talloc_free(result);
864  return 0;
865  }
866 
867  /*
868  * Sort the results
869  */
870  {
871  uint8_t const **to_sort;
872 
873  memcpy(&to_sort, &result, sizeof(to_sort));
874 
875  fr_quick_sort((void const **)to_sort, 0, used, pool_cmp);
876  }
877 
878  *out = talloc_array(ctx, uint8_t *, used);
879  if (!*out) {
880  ERROR("Failed allocating file pool name array");
881  talloc_free(result);
882  return -1;
883  }
884 
885  /*
886  * SCAN can produce duplicates, remove them here
887  */
888  i = 0;
889  k = 0;
890  do { /* stop before last entry */
891  (*out)[k++] = talloc_steal(*out, result[i++]);
892  while ((i < used) && (pool_cmp(result[i - 1], result[i]) == 0)) i++;
893  } while (i < used);
894 
895  talloc_free(request);
896  talloc_free(result);
897 
898  return used;
899 }

1033 /** Convert an IP range or CIDR mask to a start and stop address
1034  *
1035  * @param[out] start_out Where to write the start address.
1036  * @param[out] end_out Where to write the end address.
1037  * @param[in] ip_str Unparsed IP string.
1038  * @param[in] prefix length of prefixes we'll be allocating.
1039  * @return
1040  * - 0 on success.
1041  * - -1 on failure.
1042  */
1043 static int parse_ip_range(fr_ipaddr_t *start_out, fr_ipaddr_t *end_out, char const *ip_str, uint8_t prefix)
1044 {
1045  fr_ipaddr_t start, end;
1046  bool ex_broadcast;
1047  char const *p;
1048 
1049  p = strchr(ip_str, '-');
1050  if (p) {
1051  char start_buff[INET6_ADDRSTRLEN + 4];
1052  char end_buff[INET6_ADDRSTRLEN + 4];
1053  size_t len;
1054 
1055  if ((size_t)(p - ip_str) >= sizeof(start_buff)) {
1056  ERROR("Start address too long");
1057  return -1;
1058  }
1059 
1060  len = strlcpy(start_buff, ip_str, (p - ip_str) + 1);
1061  if (is_truncated(len, sizeof(start_buff))) {
1062  ERROR("Start address too long");
1063  return -1;
1064  }
1065 
1066  len = strlcpy(end_buff, p + 1, sizeof(end_buff));
1067  if (is_truncated(len, sizeof(end_buff))) {
1068  ERROR("End address too long");
1069  return -1;
1070  }
1071 
1072  if (fr_inet_pton(&start, start_buff, -1, AF_UNSPEC, false, true) < 0) {
1073  PERROR("Failed parsing \"%s\" as start address", start_buff);
1074  return -1;
1075  }
1076 
1077  if (fr_inet_pton(&end, end_buff, -1, AF_UNSPEC, false, true) < 0) {
1078  PERROR("Failed parsing \"%s\" end address", end_buff);
1079  return -1;
1080  }
1081 
1082  if (start.af != end.af) {
1083  ERROR("Start and end address must be of the same address family");
1084  return -1;
1085  }
1086 
1087  if (!prefix) prefix = IPADDR_LEN(start.af);
1088 
1089  /*
1090  * IPv6 addresses
1091  */
1092  if (start.af == AF_INET6) {
1093  uint128_t start_int, end_int;
1094 
1095  memcpy(&start_int, start.addr.v6.s6_addr, sizeof(start_int));
1096  memcpy(&end_int, end.addr.v6.s6_addr, sizeof(end_int));
1097  if (uint128_gt(ntohlll(start_int), ntohlll(end_int))) {
1098  ERROR("End address must be greater than or equal to start address");
1099  return -1;
1100  }
1101  /*
1102  * IPv4 addresses
1103  */
1104  } else {
1105  if (ntohl((uint32_t)(start.addr.v4.s_addr)) >
1106  ntohl((uint32_t)(end.addr.v4.s_addr))) {
1107  ERROR("End address must be greater than or equal to start address");
1108  return -1;
1109  }
1110  }
1111 
1112  /*
1113  * Mask start and end so we can do prefix ranges too
1114  */
1115  fr_ipaddr_mask(&start, prefix);
1116  fr_ipaddr_mask(&end, prefix);
1117  start.prefix = prefix;
1118  end.prefix = prefix;
1119 
1120  *start_out = start;
1121  *end_out = end;
1122 
1123  return 0;
1124  }
1125 
1126  if (fr_inet_pton(&start, ip_str, -1, AF_UNSPEC, false, false) < 0) {
1127  ERROR("Failed parsing \"%s\" as IPv4/v6 subnet", ip_str);
1128  return -1;
1129  }
1130 
1131  if (!prefix) prefix = IPADDR_LEN(start.af);
1132 
1133  if (prefix < start.prefix) {
1134  ERROR("-p must be greater than or equal to /<mask> (%u)", start.prefix);
1135  return -1;
1136  }
1137  if (prefix > IPADDR_LEN(start.af)) {
1138  ERROR("-p must be less than or equal to address length (%u)", IPADDR_LEN(start.af));
1139  return -1;
1140  }
1141 
1142  if ((prefix - start.prefix) > 64) {
1143  ERROR("-p must be less than or equal to %u", start.prefix + 64);
1144  return -1;
1145  }
1146 
1147  /*
1148  * Exclude the broadcast address only if we're dealing with IPv4 addresses
1149  * if we're allocating IPv6 addresses or prefixes we don't need to.
1150  */
1151  ex_broadcast = (start.af == AF_INET) && (IPADDR_LEN(start.af) == prefix);
1152 
1153  /*
1154  * Excluding broadcast, 31/32 or 127/128 start/end are the same
1155  */
1156  if (ex_broadcast && (start.prefix >= (IPADDR_LEN(start.af) - 1))) {
1157  *start_out = start;
1158  *end_out = start;
1159  return 0;
1160  }
1161 
1162  /*
1163  * Set various fields (we only overwrite the IP later)
1164  */
1165  end = start;
1166 
1167  if (start.af == AF_INET6) {
1168  uint128_t ip, p_mask;
1169 
1170  /* cond assert to satisfy clang scan */
1171  if (!fr_cond_assert((prefix > 0) && (prefix <= 128))) return -1;
1172 
1173  /* Don't be tempted to cast */
1174  memcpy(&ip, start.addr.v6.s6_addr, sizeof(ip));
1175  ip = ntohlll(ip);
1176 
1177  /* Generate a mask that covers the prefix bits, and sets them high */
1178  p_mask = uint128_lshift(uint128_gen_mask(prefix - start.prefix), (128 - prefix));
1179  ip = htonlll(uint128_bor(p_mask, ip));
1180 
1181  /* Decrement by one */
1182  if (ex_broadcast) ip = uint128_sub(ip, uint128_new(0, 1));
1183  memcpy(&end.addr.v6.s6_addr, &ip, sizeof(end.addr.v6.s6_addr));
1184  } else {
1185  uint32_t ip;
