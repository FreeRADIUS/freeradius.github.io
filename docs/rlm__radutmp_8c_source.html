104 /*
105  * Zap all users on a NAS from the radutmp file.
106  */
107 static unlang_action_t radutmp_zap(rlm_rcode_t *p_result, request_t *request, char const *filename, uint32_t nasaddr, time_t t)
108 {
109  struct radutmp u;
110  int fd;
111 
112  if (t == 0) time(&t);
113 
114  fd = open(filename, O_RDWR);
115  if (fd < 0) {
116  REDEBUG("Error accessing file %s: %s", filename, fr_syserror(errno));
117  RETURN_MODULE_FAIL;
118  }
119 
120  /*
121  * Lock the utmp file, prefer lockf() over flock().
122  */
123  if (rad_lockfd(fd, LOCK_LEN) < 0) {
124  REDEBUG("Failed to acquire lock on file %s: %s", filename, fr_syserror(errno));
125  close(fd);
126  RETURN_MODULE_FAIL;
127  }
128 
129  /*
130  * Find the entry for this NAS / portno combination.
131  */
132  while (read(fd, &u, sizeof(u)) == sizeof(u)) {
133  if ((nasaddr != 0 && nasaddr != u.nas_address) || u.type != P_LOGIN) {
134  continue;
135  }
136  /*
137  * Match. Zap it.
138  */
139  if (lseek(fd, -(off_t)sizeof(u), SEEK_CUR) < 0) {
140  REDEBUG("radutmp_zap: negative lseek!");
141  lseek(fd, (off_t)0, SEEK_SET);
142  }
143  u.type = P_IDLE;
144  u.time = t;
145 
146  if (write(fd, &u, sizeof(u)) < 0) {
147  REDEBUG("Failed writing: %s", fr_syserror(errno));
148 
149  close(fd);
150  RETURN_MODULE_FAIL;
151  }
152  }
153  close(fd); /* and implicitely release the locks */
154 
155  RETURN_MODULE_OK;
156 }

161 static NAS_PORT *nas_port_find(NAS_PORT *nas_port_list, uint32_t nasaddr, uint16_t port)
162 {
163  NAS_PORT *cl;
164 
165  for(cl = nas_port_list; cl; cl = cl->next) {
166  if (nasaddr == cl->nasaddr &&
167  port == cl->port)
168  break;
169  }
170 
171  return cl;
172 }

175 /*
176  * Store logins in the RADIUS utmp file.
177  */
178 static unlang_action_t CC_HINT(nonnull) mod_accounting(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
179 {
180  rlm_radutmp_t *inst = talloc_get_type_abort(mctx->instance, rlm_radutmp_t);
181  rlm_rcode_t rcode = RLM_MODULE_OK;
182  struct radutmp ut, u;
183  fr_cursor_t cursor;
184  fr_pair_t *vp;
185  int status = -1;
186  int protocol = -1;
187  time_t t;
188  int fd = -1;
189  bool port_seen = false;
190  int off;
191  char ip_name[INET_ADDRSTRLEN]; /* 255.255.255.255 */
192  char const *nas;
193  NAS_PORT *cache;
194  int r;
195 
196  char *filename = NULL;
197  char *expanded = NULL;
198 
199  if (request->packet->socket.inet.src_ipaddr.af != AF_INET) {
200  RDEBUG2("IPv6 not supported!");
201  RETURN_MODULE_NOOP;
202  }
203 
204  /*
205  * Which type is this.
206  */
207  if ((vp = fr_pair_find_by_da(&request->request_pairs, attr_acct_status_type)) == NULL) {
208  RDEBUG2("No Accounting-Status-Type record");
209  RETURN_MODULE_NOOP;
210  }
211  status = vp->vp_uint32;
212 
213  /*
214  * Look for weird reboot packets.
215  *
216  * ComOS (up to and including 3.5.1b20) does not send
217  * standard FR_STATUS_ACCOUNTING_XXX messages.
218  *
219  * Check for: o no Acct-Session-Time, or time of 0
220  * o Acct-Session-Id of "00000000".
221  *
222  * We could also check for NAS-Port, that attribute
223  * should NOT be present (but we don't right now).
224  */
225  if ((status != FR_STATUS_ACCOUNTING_ON) &&
226  (status != FR_STATUS_ACCOUNTING_OFF)) do {
227  int check1 = 0;
228  int check2 = 0;
229 
230  if ((vp = fr_pair_find_by_da(&request->request_pairs, attr_acct_session_time))
231  == NULL || vp->vp_uint32 == 0)
232  check1 = 1;
233  if ((vp = fr_pair_find_by_da(&request->request_pairs, attr_acct_session_id))
234  != NULL && vp->vp_length == 8 &&
235  memcmp(vp->vp_strvalue, "00000000", 8) == 0)
236  check2 = 1;
237  if (check1 == 0 || check2 == 0) {
238  break;
239  }
240  RIDEBUG("Converting reboot records");
241  if (status == FR_STATUS_STOP) status = FR_STATUS_ACCOUNTING_OFF;
242  else if (status == FR_STATUS_START) status = FR_STATUS_ACCOUNTING_ON;
243  } while(0);
244 
245  time(&t);
246  memset(&ut, 0, sizeof(ut));
247  ut.porttype = 'A';
248  ut.nas_address = htonl(INADDR_NONE);
249 
250  /*
251  * First, find the interesting attributes.
252  */
253  for (vp = fr_cursor_init(&cursor, &request->request_pairs);
254  vp;
255  vp = fr_cursor_next(&cursor)) {
256  if ((vp->da == attr_login_ip_host) ||
257  (vp->da == attr_framed_ip_address)) {
258  ut.framed_address = vp->vp_ipv4addr;
259  } else if (vp->da == attr_framed_protocol) {
260  protocol = vp->vp_uint32;
261  } else if (vp->da == attr_nas_ip_address) {
262  ut.nas_address = vp->vp_ipv4addr;
263  } else if (vp->da == attr_nas_port) {
264  ut.nas_port = vp->vp_uint32;
265  port_seen = true;
266  } else if (vp->da == attr_acct_delay_time) {
267  ut.delay = vp->vp_uint32;
268  } else if (vp->da == attr_acct_session_id) {
269  /*
270  * If length > 8, only store the
271  * last 8 bytes.
272  */
273  off = vp->vp_length - sizeof(ut.session_id);
274  /*
275  * Ascend is br0ken - it adds a \0
276  * to the end of any string.
277  * Compensate.
278  */
279  if ((vp->vp_length > 0) && (vp->vp_strvalue[vp->vp_length - 1] == 0)) off--;
280  if (off < 0) off = 0;
281  memcpy(ut.session_id, vp->vp_strvalue + off, sizeof(ut.session_id));
282  break;
283  } else if (vp->da == attr_nas_port_type) {
284  if (vp->vp_uint32 <= 4) ut.porttype = porttypes[vp->vp_uint32];
285  } else if (vp->da == attr_calling_station_id) {
286  if (inst->caller_id_ok) strlcpy(ut.caller_id, vp->vp_strvalue, sizeof(ut.caller_id));
287  }
288  }
289 
290  /*
291  * If we didn't find out the NAS address, use the
292  * originator's IP address.
293  */
294  if (ut.nas_address == htonl(INADDR_NONE)) {
295  ut.nas_address = request->packet->socket.inet.src_ipaddr.addr.v4.s_addr;
296  nas = request->client->shortname;
297 
298  } else if (request->packet->socket.inet.src_ipaddr.addr.v4.s_addr == ut.nas_address) { /* might be a client, might not be. */
299  nas = request->client->shortname;
300 
301  /*
302  * The NAS isn't a client, it's behind
303  * a proxy server. In that case, just
304  * get the IP address.
305  */
306  } else {
307  nas = inet_ntop(AF_INET, &ut.nas_address, ip_name, sizeof(ip_name));
308  }
309 
310  /*
311  * Set the protocol field.
312  */
313  if (protocol == FR_PPP) {
314  ut.proto = 'P';
315  } else if (protocol == FR_SLIP) {
316  ut.proto = 'S';
317  } else {
318  ut.proto = 'T';
319  }
320 
321  ut.time = t - ut.delay;
322 
323  /*
324  * Get the utmp filename, via xlat.
325  */
326  filename = NULL;
327  if (xlat_aeval(request, &filename, request, inst->filename, NULL, NULL) < 0) {
328  RETURN_MODULE_FAIL;
329  }
330 
331  /*
332  * See if this was a reboot.
333  *
334  * Hmm... we may not want to zap all of the users when the NAS comes up, because of issues with receiving
335  * UDP packets out of order.
336  */
337  if (status == FR_STATUS_ACCOUNTING_ON && (ut.nas_address != htonl(INADDR_NONE))) {
338  RIDEBUG("NAS %s restarted (Accounting-On packet seen)", nas);
339  radutmp_zap(&rcode, request, filename, ut.nas_address, ut.time);
340 
341  goto finish;
342  }
343 
344  if (status == FR_STATUS_ACCOUNTING_OFF && (ut.nas_address != htonl(INADDR_NONE))) {
345  RIDEBUG("NAS %s rebooted (Accounting-Off packet seen)", nas);
346  radutmp_zap(&rcode, request, filename, ut.nas_address, ut.time);
347 
348  goto finish;
349  }
350 
351  /*
352  * If we don't know this type of entry pretend we succeeded.
353  */
354  if (status != FR_STATUS_START && status != FR_STATUS_STOP && status != FR_STATUS_ALIVE) {
355  REDEBUG("NAS %s port %u unknown packet type %d)", nas, ut.nas_port, status);
356  rcode = RLM_MODULE_NOOP;
357 
358  goto finish;
359  }
360 
361  /*
362  * Translate the User-Name attribute, or whatever else they told us to use.
363  */
364  if (xlat_aeval(request, &expanded, request, inst->username, NULL, NULL) < 0) {
365  rcode = RLM_MODULE_FAIL;
366 
367  goto finish;
368  }
369  strlcpy(ut.login, expanded, RUT_NAMESIZE);
370  TALLOC_FREE(expanded);
371 
372  /*
373  * Perhaps we don't want to store this record into
374  * radutmp. We skip records:
375  *
376  * - without a NAS-Port (telnet / tcp access)
377  * - with the username "!root" (console admin login)
378  */
379  if (!port_seen) {
380  RWDEBUG2("No NAS-Port seen. Cannot do anything. Checkrad will probably not work!");
381  rcode = RLM_MODULE_NOOP;
382 
383  goto finish;
384  }
385 
386  if (strncmp(ut.login, "!root", RUT_NAMESIZE) == 0) {
387  RDEBUG2("Not recording administrative user");
388  rcode = RLM_MODULE_NOOP;
389 
390  goto finish;
391  }
392 
393  /*
394  * Enter into the radutmp file.
395  */
396  fd = open(filename, O_RDWR|O_CREAT, inst->permission);
397  if (fd < 0) {
398  REDEBUG("Error accessing file %s: %s", filename, fr_syserror(errno));
399  rcode = RLM_MODULE_FAIL;
400 
401  goto finish;
402  }
403 
404  /*
405  * Lock the utmp file, prefer lockf() over flock().
406  */
407  if (rad_lockfd(fd, LOCK_LEN) < 0) {
408  REDEBUG("Error acquiring lock on %s: %s", filename, fr_syserror(errno));
409  rcode = RLM_MODULE_FAIL;
410 
411  goto finish;
412  }
413 
414  /*
415  * Find the entry for this NAS / portno combination.
416  */
417  if ((cache = nas_port_find(inst->nas_port_list, ut.nas_address, ut.nas_port)) != NULL) {
418  if (lseek(fd, (off_t)cache->offset, SEEK_SET) < 0) {
419  rcode = RLM_MODULE_FAIL;
420  goto finish;
421  }
422  }
423 
424  r = 0;
425  off = 0;
426  while (read(fd, &u, sizeof(u)) == sizeof(u)) {
427  off += sizeof(u);
428  if ((u.nas_address != ut.nas_address) || (u.nas_port != ut.nas_port)) {
429  continue;
430  }
431 
432  /*
433  * Don't compare stop records to unused entries.
434  */
435  if (status == FR_STATUS_STOP && u.type == P_IDLE) {
436  continue;
437  }
438 
439  if ((status == FR_STATUS_STOP) && strncmp(ut.session_id, u.session_id, sizeof(u.session_id)) != 0) {
440  /*
441  * Don't complain if this is not a
442  * login record (some clients can
443  * send _only_ logout records).
444  */
445  if (u.type == P_LOGIN) {
446  RWDEBUG("Logout entry for NAS %s port %u has wrong ID", nas, u.nas_port);
447  }
448 
449  r = -1;
450  break;
451  }
452 
453  if ((status == FR_STATUS_START) && strncmp(ut.session_id, u.session_id, sizeof(u.session_id)) == 0 &&
454  u.time >= ut.time) {
455  if (u.type == P_LOGIN) {
456  RIDEBUG("Login entry for NAS %s port %u duplicate", nas, u.nas_port);
457  r = -1;
458  break;
459  }
460 
461  RWDEBUG("Login entry for NAS %s port %u wrong order", nas, u.nas_port);
462  r = -1;
463  break;
464  }
465 
466  /*
467  * FIXME: the ALIVE record could need some more checking, but anyway I'd
468  * rather rewrite this mess -- miquels.
469  */
470  if ((status == FR_STATUS_ALIVE) && strncmp(ut.session_id, u.session_id, sizeof(u.session_id)) == 0 &&
471  u.type == P_LOGIN) {
472  /*
473  * Keep the original login time.
474  */
475  ut.time = u.time;
476  }
477 
478  if (lseek(fd, -(off_t)sizeof(u), SEEK_CUR) < 0) {
479  RWDEBUG("negative lseek!");
480  lseek(fd, (off_t)0, SEEK_SET);
481  off = 0;
482  } else {
483  off -= sizeof(u);
484  }
485 
486  r = 1;
487  break;
488  } /* read the file until we find a match */
489 
490  /*
491  * Found the entry, do start/update it with
492  * the information from the packet.
493  */
494  if ((r >= 0) && (status == FR_STATUS_START || status == FR_STATUS_ALIVE)) {
495  /*
496  * Remember where the entry was, because it's
497  * easier than searching through the entire file.
498  */
499  if (!cache) {
500  cache = talloc_zero(NULL, NAS_PORT);
501  if (cache) {
502  cache->nasaddr = ut.nas_address;
503  cache->port = ut.nas_port;
504  cache->offset = off;
505  cache->next = inst->nas_port_list;
506  inst->nas_port_list = cache;
507  }
508  }
509 
510  ut.type = P_LOGIN;
511  if (write(fd, &ut, sizeof(u)) < 0) {
512  REDEBUG("Failed writing: %s", fr_syserror(errno));
513 
514  rcode = RLM_MODULE_FAIL;
515  goto finish;
516  }
517  }
518 
519  /*
520  * The user has logged off, delete the entry by
521  * re-writing it in place.
522  */
523  if (status == FR_STATUS_STOP) {
524  if (r > 0) {
525  u.type = P_IDLE;
526  u.time = ut.time;
527  u.delay = ut.delay;
528  if (write(fd, &u, sizeof(u)) < 0) {
529  REDEBUG("Failed writing: %s", fr_syserror(errno));
530 
531  rcode = RLM_MODULE_FAIL;
532  goto finish;
533  }
534  } else if (r == 0) {
535  RWDEBUG("Logout for NAS %s port %u, but no Login record", nas, ut.nas_port);
536  }
537  }
538 
539  finish:
540 
541  talloc_free(filename);
542 
543  if (fd > -1) {
544  close(fd); /* and implicitely release the locks */
545  }
546 
547  RETURN_MODULE_RCODE(rcode);
548 }
