52 /** Extract useful information from the rootDSE of the LDAP server
53  *
54  * @param[in] ctx to allocate fr_ldap_directory_t in.
55  * @param[out] out where to write pointer to new fr_ldap_directory_t struct.
56  * @param[in,out] pconn connection for querying the directory.
57  * @return
58  * - 0 on success.
59  * - 1 if we failed identifying the directory server.
60  * - -1 on error.
61  */
62 int fr_ldap_directory_alloc(TALLOC_CTX *ctx, fr_ldap_directory_t **out, fr_ldap_connection_t **pconn)
63 {
64  static char const *attrs[] = { "vendorname",
65  "vendorversion",
66  "isGlobalCatalogReady",
67  "objectClass",
68  "orcldirectoryversion",
69  NULL };
70  fr_ldap_rcode_t status;
71  int entry_cnt;
72  int ldap_errno;
73  int i, num;
74  int rcode = 0;
75  struct berval **values = NULL;
76  fr_ldap_directory_t *directory;
77 
78  LDAPMessage *result = NULL, *entry;
79 
80  *out = NULL;
81 
82  directory = talloc_zero(ctx, fr_ldap_directory_t);
83  if (!directory) return -2;
84  *out = directory;
85 
86  directory->type = FR_LDAP_DIRECTORY_UNKNOWN;
87 
88  status = fr_ldap_search(&result, NULL, pconn, "", LDAP_SCOPE_BASE, "(objectclass=*)",
89  attrs, NULL, NULL);
90  switch (status) {
91  case LDAP_PROC_SUCCESS:
92  break;
93 
94  case LDAP_PROC_NO_RESULT:
95  WARN("Capability check failed: Can't access rootDSE");
96  rcode = 1;
97  goto finish;
98 
99  default:
100  rcode = 1;
101  goto finish;
102  }
103 
104  entry_cnt = ldap_count_entries((*pconn)->handle, result);
105  if (entry_cnt != 1) {
106  WARN("Capability check failed: Ambiguous result for rootDSE, expected 1 entry, got %i", entry_cnt);
107  rcode = 1;
108  goto finish;
109  }
110 
111  entry = ldap_first_entry((*pconn)->handle, result);
112  if (!entry) {
113  ldap_get_option((*pconn)->handle, LDAP_OPT_RESULT_CODE, &ldap_errno);
114 
115  WARN("Capability check failed: Failed retrieving entry: %s", ldap_err2string(ldap_errno));
116  rcode = 1;
117  goto finish;
118  }
119 
120  values = ldap_get_values_len((*pconn)->handle, entry, "vendorname");
121  if (values) {
122  directory->vendor_str = fr_ldap_berval_to_string(directory, values[0]);
123  INFO("Directory vendor: %s", directory->vendor_str);
124 
125  ldap_value_free_len(values);
126  }
127 
128  values = ldap_get_values_len((*pconn)->handle, entry, "vendorversion");
129  if (values) {
130  directory->version_str = fr_ldap_berval_to_string(directory, values[0]);
131  INFO("Directory version: %s", directory->version_str);
132 
133  ldap_value_free_len(values);
134  }
135 
136  if (directory->vendor_str) {
137  if (strcasestr(directory->vendor_str, "International Business Machines")) {
138  directory->type = FR_LDAP_DIRECTORY_EDIRECTORY;
139  }
140 
141  goto found;
142  }
143 
144  if (directory->version_str) {
145  /*
146  * Novell eDirectory vendorversion contains eDirectory
147  */
148  if (strcasestr(directory->version_str, "eDirectory")) {
149  directory->type = FR_LDAP_DIRECTORY_EDIRECTORY;
150  /*
151  * Oracle unified directory vendorversion contains Oracle Unified Directory
152  */
153  } else if (strcasestr(directory->version_str, "Oracle Unified Directory")) {
154  directory->type = FR_LDAP_DIRECTORY_ORACLE_UNIFIED_DIRECTORY;
155  /*
156  * Unbound directory vendorversion contains UnboundID
157  */
158  } else if (strcasestr(directory->version_str, "UnboundID")) {
159  directory->type = FR_LDAP_DIRECTORY_UNBOUND_ID;
160  /*
161  * NetScape directory venderversion contains Netscape-Directory
162  */
163  } else if (strcasestr(directory->version_str, "Netscape-Directory")) {
164  directory->type = FR_LDAP_DIRECTORY_NETSCAPE;
165  /*
166  * Siemens AG directory vendorversion contains DirX Directory
167  */
168  } else if (strcasestr(directory->version_str, "DirX Directory")) {
169  directory->type = FR_LDAP_DIRECTORY_SIEMENS_AG;
170  /*
171  * Sun One Directory vendorversion contains Sun Java
172  */
173  } else if (strcasestr(directory->version_str, "Sun Java")) {
174  directory->type = FR_LDAP_DIRECTORY_SUN_ONE_DIRECTORY;
175  }
176  goto found;
177  }
178 
179  /*
180  * isGlobalCatalogReady is only present on ActiveDirectory
181  * instances. AD doesn't provide vendorname or vendorversion
182  */
183  values = ldap_get_values_len((*pconn)->handle, entry, "isGlobalCatalogReady");
184  if (values) {
185  directory->type = FR_LDAP_DIRECTORY_ACTIVE_DIRECTORY;
186  ldap_value_free_len(values);
187  goto found;
188  }
189 
190  /*
191  * OpenLDAP has a special objectClass for its RootDSE
192  */
193  values = ldap_get_values_len((*pconn)->handle, entry, "objectClass");
194  if (values) {
195  num = ldap_count_values_len(values);
196  for (i = 0; i < num; i++) {
197  if (strncmp("OpenLDAProotDSE", values[i]->bv_val, values[i]->bv_len) == 0) {
198  directory->type = FR_LDAP_DIRECTORY_OPENLDAP;
199  }
200  }
201  ldap_value_free_len(values);
202  goto found;
203  }
204 
205  /*
206  * Oracle Virtual Directory and Oracle Internet Directory
207  */
208  values = ldap_get_values_len((*pconn)->handle, entry, "orcldirectoryversion");
209  if (values) {
210  if (memmem(values[0]->bv_val, values[0]->bv_len, "OID", 3)) {
211  directory->type = FR_LDAP_DIRECTORY_ORACLE_INTERNET_DIRECTORY;
212  } else if (memmem(values[0]->bv_val, values[0]->bv_len, "OVD", 3)) {
213  directory->type = FR_LDAP_DIRECTORY_ORACLE_VIRTUAL_DIRECTORY;
214  }
215  ldap_value_free_len(values);
216  }
217 
218 found:
219  INFO("Directory type: %s", fr_table_str_by_value(fr_ldap_directory_type_table, directory->type, "<INVALID>"));
220 
221  switch (directory->type) {
222  case FR_LDAP_DIRECTORY_ACTIVE_DIRECTORY:
223  case FR_LDAP_DIRECTORY_EDIRECTORY:
224  directory->cleartext_password = false;
225  break;
226 
227  default:
228  directory->cleartext_password = true;
229  break;
230  }
231 
232 finish:
233  if (result) ldap_msgfree(result);
234 
235  return rcode;
236 }
