69 /** Convert json object to fr_value_box_t
70  *
71  * @param[in] ctx to allocate any value buffers in (should usually be the same as out).
72  * @param[in] out Where to write value. Must be initialised.
73  * @param[in] object to convert.
74  * @param[in] enumv Any string values are assumed to be in PRESENTATION format, meaning
75  * that if an enumv is specified, they'll be checked against the list
76  * of aliases for that enumeration, and possibly converted into one of
77  * the enumeration values (which may not be a string).
78  * @param[in] tainted Whether the data source is untrusted.
79  * @return
80  * - 0 on success.
81  * - -1 on failure.
82  */
83 int fr_json_object_to_value_box(TALLOC_CTX *ctx, fr_value_box_t *out, json_object *object,
84  fr_dict_attr_t const *enumv, bool tainted)
85 {
86  switch (fr_json_object_get_type(object)) {
87  case json_type_string:
88  {
89  char const *value;
90  size_t len;
91  fr_dict_enum_t *found;
92 
93  value = json_object_get_string(object);
94  len = json_object_get_string_len(object);
95 
96  if (!enumv) goto no_enumv;
97 
98  if (fr_dict_valid_name(value, len) < 0) goto no_enumv;
99 
100  /*
101  * If an alias exists, use that value instead
102  */
103  found = fr_dict_enum_by_name(enumv, value, len);
104  if (found) {
105  if (fr_value_box_copy(ctx, out, found->value) < 0) return -1;
106  return 0;
107  }
108 
109  no_enumv:
110  /*
111  * Just copy the string to the box.
112  */
113  fr_value_box_bstrndup(ctx, out, NULL, value, len, tainted);
114  }
115  break;
116 
117  case json_type_double:
118  fr_value_box_shallow(out, json_object_get_double(object), tainted);
119  break;
120 
121  case json_type_int:
122  {
123 #ifdef HAVE_JSON_OBJECT_GET_INT64
124  int64_t num;
125 #else
126  int32_t num;
127 #endif
128 
129 #ifndef HAVE_JSON_OBJECT_GET_INT64
130  num = json_object_get_int(object);
131 #else
132  num = json_object_get_int64(object);
133  if (num < INT32_MIN) { /* 64bit signed*/
134  fr_value_box_shallow(out, (int64_t)num, tainted);
135  } else if (num > UINT32_MAX) { /* 64bit unsigned */
136  fr_value_box_shallow(out, (uint64_t)num, tainted);
137  } else
138 #endif
139  if (num < INT16_MIN) { /* 32bit signed */
140  fr_value_box_shallow(out, (int32_t)num, tainted);
141  } else if (num < INT8_MIN) { /* 16bit signed */
142  fr_value_box_shallow(out, (int16_t)num, tainted);
143  } else if (num < 0) { /* 8bit signed */
144  fr_value_box_shallow(out, (int8_t)num, tainted);
145  } else if (num > UINT16_MAX) { /* 32bit unsigned */
146  fr_value_box_shallow(out, (uint32_t)num, tainted);
147  } else if (num > UINT8_MAX) { /* 16bit unsigned */
148  fr_value_box_shallow(out, (int16_t)num, tainted);
149  } else { /* 8bit unsigned */
150  fr_value_box_shallow(out, (int8_t)num, tainted);
151  }
152  }
153  break;
154 
155  case json_type_boolean:
156  fr_value_box_shallow(out, (bool)json_object_get_boolean(object), tainted);
157  break;
158 
159  case json_type_null:
160  case json_type_array:
161  case json_type_object:
162  {
163  char const *value = json_object_to_json_string(object);
164 
165  fr_value_box_bstrndup(ctx, out, NULL, value, strlen(value), tainted);
166  }
167  break;
168  }
169 
170  out->tainted = tainted;
171 
172  return 0;
173 }

175 /** Convert boxed value_box to a JSON object
176  *
177  * @param[in] ctx to allocate temporary buffers in
178  * @param[in] data to convert.
179  */
180 json_object *json_object_from_value_box(TALLOC_CTX *ctx, fr_value_box_t const *data)
181 {
182  /*
183  * We're converting to PRESENTATION format
184  * so any attributes with enumeration values
185  * should be converted to string types.
186  */
187  if (data->enumv) {
188  fr_dict_enum_t *enumv;
189 
190  enumv = fr_dict_enum_by_value(data->enumv, data);
191  if (enumv) return json_object_new_string(enumv->name);
192  }
193 
194  switch (data->type) {
195  default:
196  do_string:
197  {
198  char *p;
199  json_object *obj;
200 
201  fr_value_box_aprint(ctx, &p, data, NULL);
202  if (!p) return NULL;
203 
204  obj = json_object_new_string(p);
205  talloc_free(p);
206 
207  return obj;
208  }
209 
210  case FR_TYPE_BOOL:
211  return json_object_new_boolean(data->vb_uint8);
212 
213  case FR_TYPE_UINT8:
214  return json_object_new_int(data->vb_uint8);
215 
216  case FR_TYPE_UINT16:
217  return json_object_new_int(data->vb_uint16);
218 
219 #ifdef HAVE_JSON_OBJECT_GET_INT64
220  case FR_TYPE_UINT32:
221  return json_object_new_int64((int64_t)data->vb_uint64); /* uint32_t (max) > int32_t (max) */
222 
223  case FR_TYPE_UINT64:
224  if (data->vb_uint64 > INT64_MAX) goto do_string;
225  return json_object_new_int64(data->vb_uint64);
226 #else
227  case FR_TYPE_UINT32:
228  if (data->vb_uint32 > INT32_MAX) goto do_string;
229  return json_object_new_in(data->vb_uint32);
230 #endif
231 
232  case FR_TYPE_INT8:
233  return json_object_new_int(data->vb_int8);
234 
235  case FR_TYPE_INT16:
236  return json_object_new_int(data->vb_int16);
237 
238  case FR_TYPE_INT32:
239  return json_object_new_int(data->vb_int32);
240 
241 #ifdef HAVE_JSON_OBJECT_GET_INT64
242  case FR_TYPE_INT64:
243  return json_object_new_int64(data->vb_int16);
244 #endif
245  }
246 }

248 /** Escapes string for use as a JSON string
249  *
250  * @param ctx Talloc context to allocate this string
251  * @param s Input string
252  * @param include_quotes Include the surrounding quotes of JSON strings
253  * @return New allocated character string, or NULL if something failed.
254  */
255 char *fr_json_from_string(TALLOC_CTX *ctx, char const *s, bool include_quotes)
256 {
257  char const *p;
258  char *out = NULL;
259  struct json_object *json;
260  int len;
261 
262  json = json_object_new_string(s);
263  if (!json) return NULL;
264 
265  if ((p = json_object_to_json_string(json))) {
266  if (include_quotes) {
267  out = talloc_typed_strdup(ctx, p);
268  } else {
269  len = strlen(p);
270  out = talloc_bstrndup(ctx, p + 1, len - 2); /* to_json_string adds quotes (") */
271  }
272  }
273 
274  json_object_put(json);
275  return out;
276 }

278 /** Prints attribute as string, escaped suitably for use as JSON string
279  *
280  * Returns < 0 if the buffer may be (or have been) too small to write the encoded
281  * JSON value to.
282  *
283  * @param out Where to write the string.
284  * @param outlen Length of output buffer.
285  * @param vp to print.
286  * @return
287  * - Length of data written to out.
288  * - value >= outlen on truncation.
289  */
290 size_t fr_json_from_pair(char *out, size_t outlen, fr_pair_t const *vp)
291 {
292  ssize_t slen;
293  size_t freespace = outlen;
294 
295  switch (vp->vp_type) {
296  case FR_TYPE_UINT32:
297  if (vp->da->flags.has_value) break;
298 
299  return snprintf(out, freespace, "%u", vp->vp_uint32);
300 
301  case FR_TYPE_UINT16:
302  if (vp->da->flags.has_value) break;
303 
304  return snprintf(out, freespace, "%u", (unsigned int) vp->vp_uint16);
305 
306  case FR_TYPE_UINT8:
307  if (vp->da->flags.has_value) break;
308 
309  return snprintf(out, freespace, "%u", (unsigned int) vp->vp_uint8);
310 
311  case FR_TYPE_INT32:
312  return snprintf(out, freespace, "%d", vp->vp_int32);
313 
314  default:
315  break;
316  }
317 
318  if (vp->vp_type == FR_TYPE_STRING) {
319  char *tmp = fr_json_from_string(NULL, vp->vp_strvalue, true);
320 
321  /* Indicate truncation */
322  if (!tmp) return outlen + 1;
323  slen = strlen(tmp);
324  if (freespace <= (size_t)slen) return outlen + 1;
325 
326  strcpy(out, tmp);
327  talloc_free(tmp);
328 
329  return slen;
330  }
331 
332  /* Indicate truncation */
333  if (freespace < 2) return outlen + 1;
334  *out++ = '"';
335  freespace--;
336 
337  slen = fr_pair_print_value_quoted(&FR_SBUFF_OUT(out, freespace), vp, T_BARE_WORD);
338  if (slen < 0) return slen;
339 
340  out += (size_t)slen;
341  freespace -= (size_t)slen;
342 
343  /* Indicate truncation */
344  if (freespace < 2) return outlen + 1;
345  *out++ = '"';
346  freespace--;
347  *out = '\0'; // We don't increment out, because the nul byte should not be included in the length
348 
349  return outlen - freespace;
350 }

365 /** Convert fr_pair_t into a JSON object
366  *
367  * If format.value.enum_as_int is set, and the given VP is an enum
368  * value, the integer value is returned as a json_object rather
369  * than the text representation.
370  *
371  * If format.value.always_string is set then a numeric value pair
372  * will be returned as a JSON string object.
373  *
374  * @param[in] ctx Talloc context.
375  * @param[out] out returned json object.
376  * @param[in] vp to get the value of.
377  * @param[in] format format definition, or NULL.
378  * @return
379  * - 1 if 'out' is the integer enum value, 0 otherwise
380  * - -1 on error.
381  */
382 static int json_afrom_value_box(TALLOC_CTX *ctx, json_object **out,
383  fr_pair_t *vp, fr_json_format_t const *format)
384 {
385  struct json_object *obj;
386  fr_value_box_t const *vb;
387  fr_value_box_t vb_str;
388  int is_enum = 0;
389 
390  fr_assert(vp);
391 
392  vb = &vp->data;
393 
394  if (format && format->value.enum_as_int) {
395  is_enum = fr_pair_value_enum_box(&vb, vp);
396  fr_assert(is_enum >= 0);
397  }
398 
399  if (format && format->value.always_string) {
400  if (fr_value_box_cast(ctx, &vb_str, FR_TYPE_STRING, NULL, vb) == 0) {
401  vb = &vb_str;
402  } else {
403  return -1;
404  }
405  }
406 
407  MEM(obj = json_object_from_value_box(ctx, vb));
408 
409  if (format && format->value.always_string) {
410  fr_value_box_clear(&vb_str);
411  }
412 
413  *out = obj;
414  return is_enum;
415 }

418 /** Add prefix to attribute name
419  *
420  * If the format "attr.prefix" string is set then prepend this
421  * to the given attribute name, otherwise return name unchanged.
422  *
423  * @param[out] buf where to write the new name, if set
424  * @param[in] buf_len length of buf
425  * @param[in] name original attribute name
426  * @param[in] format json format structure
427  * @return pointer to name, or buf if the prefix was added
428  */
429 static inline char const *attr_name_with_prefix(char *buf, size_t buf_len, const char *name, fr_json_format_t const *format)
430 {
431  int len;
432 
433  if (!format->attr.prefix) return name;
434 
435  len = snprintf(buf, buf_len, "%s:%s", format->attr.prefix, name);
436 
437  if (len == (int)strlen(buf)) {
438  return buf;
439  }
440 
441  return name;
442 }

445 /** Verify that the options in fr_json_format_t are valid
446  *
447  * Warnings are optional, will fatal error if the format is corrupt.
448  *
449  * @param[in] format the format structure to check
450  * @param[in] verbose print out warnings if set
451  * @return true if format is good, otherwise false
452  */
453 bool fr_json_format_verify(fr_json_format_t const *format, bool verbose)
454 {
455  bool ret = true;
456 
457  fr_assert(format);
458 
459  switch (format->output_mode) {
460  case JSON_MODE_OBJECT:
461  case JSON_MODE_OBJECT_SIMPLE:
462  case JSON_MODE_ARRAY:
463  /* all options are valid */
464  return true;
465  case JSON_MODE_ARRAY_OF_VALUES:
466  if (format->attr.prefix) {
467  if (verbose) WARN("attribute name prefix not valid in output_mode 'array_of_values' and will be ignored");
468  ret = false;
469  }
470  if (format->value.value_as_array) {
471  if (verbose) WARN("'value_as_array' not valid in output_mode 'array_of_values' and will be ignored");
472  ret = false;
473  }
474  return ret;
475  case JSON_MODE_ARRAY_OF_NAMES:
476  if (format->value.value_as_array) {
477  if (verbose) WARN("'value_as_array' not valid in output_mode 'array_of_names' and will be ignored");
478  ret = false;
479  }
480  if (format->value.enum_as_int) {
481  if (verbose) WARN("'enum_as_int' not valid in output_mode 'array_of_names' and will be ignored");
482  ret = false;
483  }
484  if (format->value.always_string) {
485  if (verbose) WARN("'always_string' not valid in output_mode 'array_of_names' and will be ignored");
486  ret = false;
487  }
488  return ret;
489  default:
490  ERROR("JSON format output mode is invalid");
491  }
492 
493  /* If we get here, something has gone wrong */
494  fr_assert(0);
495 
496  return false;
497 }

500 /** Returns a JSON object representation of a list of value pairs
501  *
502  * The result is a struct json_object, which should be free'd with
503  * json_object_put() by the caller. Intended to only be called by
504  * fr_json_afrom_pair_list().
505  *
506  * This function generates the "object" format, JSON_MODE_OBJECT.
507  * @see fr_json_format_s
508  *
509  * @param[in] ctx Talloc context.
510  * @param[in] vps a list of value pairs.
511  * @param[in] format Formatting control, must be set.
512  * @return JSON object with the generated representation.
513  */
514 static json_object *json_object_afrom_pair_list(TALLOC_CTX *ctx, fr_pair_t *vps,
515  fr_json_format_t const *format)
516 {
517  fr_cursor_t cursor;
518  fr_pair_t *vp;
519  struct json_object *obj;
520  char buf[FR_DICT_ATTR_MAX_NAME_LEN + 32];
521 
522  /* Check format and type */
523  fr_assert(format);
524  fr_assert(format->output_mode == JSON_MODE_OBJECT);
525 
526  MEM(obj = json_object_new_object());
527 
528  for (vp = fr_cursor_init(&cursor, &vps);
529  vp;
530  vp = fr_cursor_next(&cursor)) {
531  char const *attr_name;
532  struct json_object *vp_object, *values, *value, *type_name;
533 
534  /*
535  * Get attribute name and value.
536  */
537  attr_name = attr_name_with_prefix(buf, sizeof(buf), vp->da->name, format);
538 
539  if (json_afrom_value_box(ctx, &value, vp, format) < 0) {
540  fr_strerror_printf("Failed to convert attribute value to JSON object");
541  error:
542  json_object_put(obj);
543  return NULL;
544  }
545 
546  /*
547  * Look in the table to see if we already have
548  * a key for the attribute we're working on.
549  */
550  if (!json_object_object_get_ex(obj, attr_name, &vp_object)) {
551  /*
552  * Wasn't there, so create a new object for this attribute.
553  */
554  MEM(vp_object = json_object_new_object());
555  json_object_object_add(obj, attr_name, vp_object);
556 
557  /*
558  * Add "type" to newly created keys.
559  */
560  MEM(type_name = json_object_new_string(fr_table_str_by_value(fr_value_box_type_table,
561  vp->vp_type, "<INVALID>")));
562  json_object_object_add(vp_object, "type", type_name);
563 
564  /*
565  * Create a "value" array to hold any attribute values for this attribute...
566  */
567  if (format->value.value_as_array) {
568  MEM(values = json_object_new_array());
569  json_object_object_add(vp_object, "value", values);
570  } else {
571  /*
572  * ...unless this is the first time we've seen the attribute and
573  * value_as_array is false, in which case just add the value directly
574  * and move on to the next attribute.
575  */
576  json_object_object_add(vp_object, "value", value);
577  continue;
578  }
579  } else {
580  /*
581  * Find the 'values' array to add the current value to.
582  */
583  if (!fr_cond_assert(json_object_object_get_ex(vp_object, "value", &values))) {
584  fr_strerror_printf("Inconsistent JSON tree");
585  goto error;
586  }
587 
588  /*
589  * If value_as_array is no set then "values" may not be an array, so it will
590  * need converting to an array to add this extra attribute.
591  */
592  if (!format->value.value_as_array) {
593  json_type type;
594  struct json_object *convert_value = values;
595 
596  /* Check "values" type */
597  type = json_object_get_type(values);
598 
599  /* It wasn't an array, so turn it into one with the old value as the first entry */
600  if (type != json_type_array) {
601  MEM(values = json_object_new_array());
602  json_object_array_add(values, json_object_get(convert_value));
603  json_object_object_del(vp_object, "value");
604  json_object_object_add(vp_object, "value", values);
605  }
606  }
607  }
608 
609  /*
610  * Append to the JSON array.
611  */
612  json_object_array_add(values, value);
613  }
614 
615  return obj;
616 }

645 
646  MEM(obj = json_object_new_object());

723 /** Returns a JSON array representation of a list of value pairs
724  *
725  * The result is a struct json_object, which should be free'd with
726  * json_object_put() by the caller. Intended to only be called by
727  * fr_json_afrom_pair_list().
728  *
729  * This function generates the "array" format, JSON_MODE_ARRAY.
730  * @see fr_json_format_s
731  *
732  * @param[in] ctx Talloc context.
733  * @param[in] vps a list of value pairs.
734  * @param[in] format Formatting control, must be set.
735  * @return JSON object with the generated representation.
736  */
737 static struct json_object *json_array_afrom_pair_list(TALLOC_CTX *ctx, fr_pair_t *vps,
738  fr_json_format_t const *format)
739 {
740  fr_cursor_t cursor;
741  fr_pair_t *vp;
742  struct json_object *obj;
743  struct json_object *seen_attributes = NULL;
744  char buf[FR_DICT_ATTR_MAX_NAME_LEN + 32];
745 
746  /* Check format and type */
747  fr_assert(format);
748  fr_assert(format->output_mode == JSON_MODE_ARRAY);
749 
750  MEM(obj = json_object_new_array());
751 
752  /*
753  * If attribute values should be in a list format, then keep track
754  * of the attributes we've previously seen in a JSON object.
755  */
756  if (format->value.value_as_array) {
757  seen_attributes = json_object_new_object();
758  }
759 
760  for (vp = fr_cursor_init(&cursor, &vps);
761  vp;
762  vp = fr_cursor_next(&cursor)) {
763  char const *attr_name;
764  struct json_object *name, *value, *type_name;
765  struct json_object *values = NULL;
766  struct json_object *attrobj = NULL;
767  bool already_seen = false;
768 
769  /*
770  * Get attribute name and value.
771  */
772  attr_name = attr_name_with_prefix(buf, sizeof(buf), vp->da->name, format);
773 
774  if (json_afrom_value_box(ctx, &value, vp, format) < 0) {
775  fr_strerror_printf("Failed to convert attribute value to JSON object");
776  json_object_put(obj);
777  return NULL;
778  }
779 
780  if (format->value.value_as_array) {
781  /*
782  * Try and find this attribute in the "seen_attributes" object. If it is
783  * there then get the "values" array to add this attribute value to.
784  */
785  already_seen = json_object_object_get_ex(seen_attributes, attr_name, &values);
786  }
787 
788  /*
789  * If we're adding all attributes to the toplevel array, or we're adding values
790  * to an array of an existing attribute but haven't seen it before, then we need
791  * to create a new JSON object for this attribute.
792  */
793  if (!format->value.value_as_array || !already_seen) {
794  /*
795  * Create object and add it to top-level array
796  */
797  MEM(attrobj = json_object_new_object());
798  json_object_array_add(obj, attrobj);
799 
800  /*
801  * Add the attribute name in the "name" key and the type in the "type" key
802  */
803  MEM(name = json_object_new_string(attr_name));
804  json_object_object_add(attrobj, "name", name);
805 
806  MEM(type_name = json_object_new_string(fr_table_str_by_value(fr_value_box_type_table,
807  vp->vp_type, "<INVALID>")));
808  json_object_object_add(attrobj, "type", type_name);
809  }
810 
811  if (format->value.value_as_array) {
812  /*
813  * We're adding values to an array for the first copy of this attribute
814  * that we saw. First time around we need to create an array.
815  */
816  if (!already_seen) {
817  MEM(values = json_object_new_array());
818  /*
819  * Add "value":[] key to the attribute object
820  */
821  json_object_object_add(attrobj, "value", values);
822 
823  /*
824  * Also add to "seen_attributes" to check later
825  */
826  json_object_object_add(seen_attributes, attr_name, json_object_get(values));
827  }
828 
829  /*
830  * Always add the value to the respective "values" array.
831  */
832  json_object_array_add(values, value);
833  } else {
834  /*
835  * This is simpler; just add a "value": key to the attribute object.
836  */
837  json_object_object_add(attrobj, "value", value);
838  }
839 
840  }
841 
842  /*
843  * No longer need the "seen_attributes" object, it was just used for tracking.
844  */
845  if (format->value.value_as_array) {
846  json_object_put(seen_attributes);
847  }
848 
849  return obj;
850 }

961 @verbatim
962 {

989

991 
992  switch (format->output_mode) {

994  MEM(obj = json_object_afrom_pair_list(ctx, vps, format));
995  break;
996  case JSON_MODE_OBJECT_SIMPLE:

999  case JSON_MODE_ARRAY:
1000  MEM(obj = json_array_afrom_pair_list(ctx, vps, format));
1001  break;

1014  MEM(out = talloc_typed_strdup(ctx, p));
1015 
1016  /*
1017  * Free the JSON structure, it's not needed any more
1018  */
1019  json_object_put(obj);
1020 
1021  return out;
1022 }
