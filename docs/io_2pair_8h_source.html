54 /** Return the correct adjusted slen for errors
55  *
56  * @param[in] slen returned from the function we called.
57  * @param[in] start of the buffer.
58  * @param[in] p offset passed to function which returned the slen.
59  */
60 static inline ssize_t fr_pair_decode_slen(ssize_t slen, uint8_t const *start, uint8_t const *p)
61 {
62  if (slen > 0) return slen;
63 
64  switch (slen) {
65  case PAIR_DECODE_OOM:
66  case PAIR_DECODE_FATAL_ERROR:
67  return slen;
68 
69  default:
70  return slen - (p - start);
71  }
72 }

90 /** Generic interface for encoding one or more fr_pair_ts
91  *
92  * An encoding function should consume at most, one top level fr_pair_t and encode
93  * it in the appropriate wire format for the protocol, writing the encoded data to
94  * out, and returning the encoded length.
95  *
96  * The exception to processing one fr_pair_t is if multiple fr_pair_ts can be aggregated
97  * into a single TLV, in which case the encoder may consume as many fr_pair_ts as will
98  * fit into that TLV.
99  *
100  * Outlen provides the length of the buffer to write the encoded data to. The return
101  * value must not be greater than outlen.
102  *
103  * The cursor is used to track how many pairs there are remaining.
104  *
105  * @param[out] out Where to write the encoded data.
106  * @param[in] cursor Cursor containing the list of attributes to process.
107  * @param[in] encoder_ctx Any encoder specific data such as secrets or configurables.
108  * @return
109  * - PAIR_ENCODE_SKIPPED - The current pair is not valid for encoding and should be skipped.
110  * - PAIR_ENCODE_FATAL_ERROR - Encoding failed in a fatal way. Encoding the packet should be
111  * aborted in its entirety.
112  * - <0 - The encoder ran out of space and returned the number of bytes as a negative
113  * integer that would be required to encode the attribute.
114  * - >0 - The number of bytes written to out.
115  */
116 typedef ssize_t (*fr_pair_encode_t)(fr_dbuff_t *out, fr_cursor_t *cursor, void *encoder_ctx);
