90 static void alloc_blocks(fr_message_set_t *ms, uint32_t *seed, UNUSED int *start, int *end)
91 {
92  int i;
93  uint32_t hash;
94 
95  /*
96  * We can't allocated the entire array, and we can't
97  * over-fill the array.
98  */
99  fr_assert((size_t) my_alloc_size < MY_ARRAY_SIZE);
100 
101  MPRINT2("BLOCK ALLOC %d\n", my_alloc_size);
102 
103  for (i = 0; i < my_alloc_size; i++) {
104  int index;
105  fr_message_t *m;
106 
107  index = (*end + i) & (MY_ARRAY_SIZE - 1);
108 
109  hash = fr_hash_update(seed_string, seed_string_len, *seed);
110  *seed = hash;
111 
112  hash &= allocation_mask;
113  hash++; /* can't have it zero... */
114 
115  array[index] = hash;
116 
117  m = fr_message_reserve(ms, reserve_size);
118  fr_assert(m != NULL);
119 
120  messages[index] = (fr_test_t *) fr_message_alloc(ms, m, hash);
121  fr_assert(messages[index] == (void *) m);
122 
123  if (touch_memory) {
124  size_t j;
125  size_t k = 0;
126 
127  for (j = 0; j < m->data_size; j++) {
128  k += m->data[j];
129  }
130 
131  m->data[0] = k;
132  }
133 
134  if (debug_lvl > 1) printf("%08x\t", hash);
135 
136  fr_assert(m->status == FR_MESSAGE_USED);
137 
138  used += hash;
139 // fr_assert(fr_ring_buffer_used(rb) == used);
140  }
141 
142  *end += my_alloc_size;
143 }

145 static void free_blocks(UNUSED fr_message_set_t *ms, UNUSED uint32_t *seed, int *start, int *end)
146 {
147  int i;
148 
149  fr_assert(my_alloc_size < MY_ARRAY_SIZE);
150 
151  MPRINT2("BLOCK FREE %d\n", my_alloc_size);
152 
153  for (i = 0; i < my_alloc_size; i++) {
154  int index;
155  int ret;
156  fr_message_t *m;
157 
158  index = (*start + i) & (MY_ARRAY_SIZE - 1);
159 
160  m = &messages[index]->m;
161 
162  fr_assert(m->status == FR_MESSAGE_USED);
163 
164  ret = fr_message_done(m);
165 #ifndef NDEBUG
166  fr_assert(ret == 0);
167 #else
168  if (ret != 0) fr_exit_now(EXIT_FAILURE);
169 #endif
170 
171  used -= array[index];
172 
173  array[index] = 0;
174  messages[index] = NULL;
175  }
176 
177  *start += my_alloc_size;
178  if (*start > MY_ARRAY_SIZE) {
179  *start -= MY_ARRAY_SIZE;
180  *end -= MY_ARRAY_SIZE;
181  fr_assert(*start <= *end);
182  fr_assert(*start < MY_ARRAY_SIZE);
183  }
184 }
