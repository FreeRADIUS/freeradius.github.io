334 /** @name Attribute comparisons
335  *
336  * @{
337  */
338 static inline CC_HINT(nonnull) int8_t fr_dict_attr_cmp(fr_dict_attr_t const *a, fr_dict_attr_t const *b)
339 {
340  int8_t ret;
341 
342  /*
343  * Comparing unknowns or raws is expensive
344  * because we need to check the lineage.
345  */
346  if (a->flags.is_unknown | a->flags.is_raw | b->flags.is_unknown | b->flags.is_raw) {
347  ret = STABLE_COMPARE(a->depth, b->depth);
348  if (ret != 0) return ret;
349 
350  ret = STABLE_COMPARE(a->attr, b->attr);
351  if (ret != 0) return ret;
352 
353  ret = (a->parent == NULL) - (b->parent == NULL);
354  if ((ret != 0) || !a->parent) return ret;
355 
356  return fr_dict_attr_cmp(a->parent, b->parent);
357  }
358 
359  /*
360  * Comparing knowns is cheap because the
361  * DAs are unique.
362  */
363  return STABLE_COMPARE(a, b);
364 }
