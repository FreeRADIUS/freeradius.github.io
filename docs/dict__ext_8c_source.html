42 /** Copy all enumeration values from one attribute to another
43  *
44  */
45 static void *fr_dict_attr_ext_enumv_copy(void **chunk_p, int ext, void *ext_ptr, UNUSED size_t ext_len)
46 {
47  fr_dict_attr_t **da_p = (fr_dict_attr_t **)chunk_p;
48  fr_dict_attr_ext_enumv_t *new_ext, *old_ext;
49  fr_hash_iter_t iter;
50  fr_dict_enum_t *enumv;
51  bool has_child = da_is_key_field(*da_p);
52 
53  new_ext = dict_attr_ext_alloc(da_p, ext);
54  if (!new_ext) return NULL;
55 
56  old_ext = ext_ptr;
57 
58  if (!old_ext->value_by_name && !old_ext->name_by_value) {
59  memset(new_ext, 0, sizeof(*new_ext));
60  return new_ext;
61  }
62 
63  /*
64  * Add all the enumeration values from
65  * the old attribute to the new attribute.
66  */
67  for (enumv = fr_hash_table_iter_init(old_ext->value_by_name, &iter);
68  enumv;
69  enumv = fr_hash_table_iter_next(old_ext->value_by_name, &iter)) {
70  fr_dict_attr_t const *child_struct;
71 
72  if (!has_child) {
73  child_struct = NULL;
74  } else {
75  child_struct = enumv->child_struct[0];
76  }
77 
78  /*
79  * Fixme - Child struct copying is probably wrong
80  */
81  if (dict_attr_enum_add_name(*da_p, enumv->name, enumv->value, true, true, child_struct) < 0) {
82  return NULL;
83  }
84  }
85 
86  return new_ext;
87 }

89 /** Rediscover the parent of this attribute, and cache it
90  *
91  */
92 static void *fr_dict_attr_ext_vendor_copy(void **chunk_p, int ext, void *ext_ptr, UNUSED size_t ext_len)
93 {
94  fr_dict_attr_t **da_p = (fr_dict_attr_t **)chunk_p;
95  fr_dict_attr_ext_vendor_t *new_ext, *old_ext = ext_ptr;
96  fr_dict_attr_t const **da_stack;
97  fr_dict_attr_t const *old_vendor = old_ext->vendor;
98  fr_dict_attr_t const *new_vendor, *da;
99 
100  /*
101  * If we have a da stack, see if we can
102  * find a vendor at the same depth as
103  * the old depth.
104  */
105  da_stack = fr_dict_attr_da_stack(*da_p);
106  if (da_stack) {
107  new_vendor = da_stack[old_vendor->depth];
108  if ((new_vendor->type == old_vendor->type) && (new_vendor->attr == old_vendor->attr)) {
109  new_ext = dict_attr_ext_alloc(da_p, ext);
110  if (!new_ext) return NULL;
111 
112  new_ext->vendor = new_vendor;
113  return new_ext;
114  }
115  }
116 
117  /*
118  * Otherwise traverse the parent list
119  * looking for the vendor.
120  *
121  * Theoretically the attribute could
122  * have been moved to a different depth.
123  */
124  for (da = (*da_p)->parent; da; da = da->parent) {
125  if ((da->type == old_vendor->type) && (da->attr == old_vendor->attr)) {
126  new_ext = dict_attr_ext_alloc(da_p, ext);
127  if (!new_ext) return NULL;
128 
129  new_ext->vendor = da;
130  return new_ext;
131  }
132  }
133 
134  return NULL;
135 }
