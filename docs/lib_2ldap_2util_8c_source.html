39 /** Converts "bad" strings into ones which are safe for LDAP
40  *
41  * @note RFC 4515 says filter strings can only use the @verbatim <hex><hex> @endverbatim
42  * format, whereas RFC 4514 indicates that some chars in DNs, may be escaped simply
43  * with a backslash. For simplicity, we always use the hex escape sequences.
44  * In other areas where we're doing DN comparison, the DNs need to be normalised first
45  * so that they both use only hex escape sequences.
46  *
47  * @note This is a callback for xlat operations.
48  *
49  * Will escape any characters in input strings that would cause the string to be interpreted
50  * as part of a DN and or filter. Escape sequence is @verbatim <hex><hex> @endverbatim.
51  *
52  * @param request The current request.
53  * @param out Pointer to output buffer.
54  * @param outlen Size of the output buffer.
55  * @param in Raw unescaped string.
56  * @param arg Any additional arguments (unused).
57  */
58 size_t fr_ldap_escape_func(UNUSED request_t *request, char *out, size_t outlen, char const *in, UNUSED void *arg)
59 {
60 
61  size_t left = outlen;
62 
63  if (*in && ((*in == ' ') || (*in == '#'))) goto encode;
64 
65  while (*in) {
66  /*
67  * Encode unsafe characters.
68  */
69  if (memchr(specials, *in, sizeof(specials) - 1)) {
70  encode:
71  /*
72  * Only 3 or less bytes available.
73  */
74  if (left <= 3) break;
75 
76  *out++ = '\\';
77  *out++ = hextab[(*in >> 4) & 0x0f];
78  *out++ = hextab[*in & 0x0f];
79  in++;
80  left -= 3;
81 
82  continue;
83  }
84 
85  if (left <= 1) break;
86 
87  /*
88  * Doesn't need encoding
89  */
90  *out++ = *in++;
91  left--;
92  }
93 
94  *out = '\0';
95 
96  return outlen - left;
97 }

99 /** Converts escaped DNs and filter strings into normal
100  *
101  * @note RFC 4515 says filter strings can only use the @verbatim <hex><hex> @endverbatim
102  * format, whereas RFC 4514 indicates that some chars in DNs, may be escaped simply
103  * with a backslash..
104  *
105  * Will unescape any special characters in strings, or @verbatim <hex><hex> @endverbatim
106  * sequences.
107  *
108  * @param request The current request.
109  * @param out Pointer to output buffer.
110  * @param outlen Size of the output buffer.
111  * @param in Escaped string string.
112  * @param arg Any additional arguments (unused).
113  */
114 size_t fr_ldap_unescape_func(UNUSED request_t *request, char *out, size_t outlen, char const *in, UNUSED void *arg)
115 {
116  char const *p;
117  char *c1, *c2, c3;
118  size_t freespace = outlen;
119 
120  if (outlen <= 1) return 0;
121 
122  p = in;
123  while (*p && (--freespace > 0)) {
124  if (*p != '\\') {
125  next:
126  *out++ = *p++;
127  continue;
128  }
129 
130  p++;
131 
132  /* It's an escaped special, just remove the slash */
133  if (memchr(specials, *in, sizeof(specials) - 1)) {
134  *out++ = *p++;
135  continue;
136  }
137 
138  /* Is a hex sequence */
139  if (!(c1 = memchr(hextab, tolower(p[0]), 16)) ||
140  !(c2 = memchr(hextab, tolower(p[1]), 16))) goto next;
141  c3 = ((c1 - hextab) << 4) + (c2 - hextab);
142 
143  *out++ = c3;
144  p += 2;
145  }
146 
147  *out = '\0';
148 
149  return outlen - freespace;
150 }

153 /** Check whether a string looks like a DN
154  *
155  * @param[in] in Str to check.
156  * @param[in] inlen Length of string to check.
157  * @return
158  * - true if string looks like a DN.
159  * - false if string does not look like DN.
160  */
161 bool fr_ldap_util_is_dn(char const *in, size_t inlen)
162 {
163  char const *p;
164 
165  char want = '=';
166  bool too_soon = true;
167  int comp = 1;
168 
169  for (p = in; inlen > 0; p++, inlen--) {
170  if (p[0] == '\\') {
171  char c;
172 
173  too_soon = false;
174 
175  /*
176  * Invalid escape sequence, not a DN
177  */
178  if (inlen < 2) return false;
179 
180  /*
181  * Double backslash, consume two chars
182  */
183  if (p[1] == '\\') {
184  inlen--;
185  p++;
186  continue;
187  }
188 
189  /*
190  * Special, consume two chars
191  */
192  switch (p[1]) {
193  case ' ':
194  case '#':
195  case '=':
196  case '"':
197  case '+':
198  case ',':
199  case ';':
200  case '<':
201  case '>':
202  case '\'':
203  inlen -= 1;
204  p += 1;
205  continue;
206 
207  default:
208  break;
209  }
210 
211  /*
212  * Invalid escape sequence, not a DN
213  */
214  if (inlen < 3) return false;
215 
216  /*
217  * Hex encoding, consume three chars
218  */
219  if (fr_hex2bin(NULL, &FR_DBUFF_TMP((uint8_t *) &c, 1), &FR_SBUFF_IN(p + 1, 2), false) == 1) {
220  inlen -= 2;
221  p += 2;
222  continue;
223  }
224 
225  /*
226  * Invalid escape sequence, not a DN
227  */
228  return false;
229  }
230 
231  switch (*p) {
232  case '=':
233  if (too_soon || (*p != want)) return false; /* Too soon after last , or = */
234  want = ',';
235  too_soon = true;
236  break;
237 
238  case ',':
239  if (too_soon || (*p != want)) return false; /* Too soon after last , or = */
240  want = '=';
241  too_soon = true;
242  comp++;
243  break;
244 
245  default:
246  too_soon = false;
247  break;
248  }
249  }
250 
251  /*
252  * If the string ended with , or =, or the number
253  * of components was less than 2
254  *
255  * i.e. we don't have <attr>=<val>,<attr>=<val>
256  */
257  if (too_soon || (comp < 2)) return false;
258 
259  return true;
260 }

262 /** Parse a subset (just server side sort for now) of LDAP URL extensions
263  *
264  * @param[out] sss Where to write a pointer to the server side sort control
265  * we created.
266  * @param[in] request The current request.
267  * @param[in] conn Handle to allocate controls under.

269  * @return

271  * - -1 on failure.
272  */
273 int fr_ldap_parse_url_extensions(LDAPControl **sss, request_t *request, fr_ldap_connection_t *conn, char **extensions)
274 {
275  int i;
276 
277  *sss = NULL;
278 
279  if (!extensions) return 0;
280 
281  /*
282  * Parse extensions in the LDAP URL
283  */
284  for (i = 0; extensions[i]; i++) {
285  char *p;
286  bool is_critical = false;
287 
288  p = extensions[i];
289  if (*p == '!') {
290  is_critical = true;
291  p++;
292  }
293 
294 #ifdef HAVE_LDAP_CREATE_SORT_CONTROL
295  /*
296  * Server side sort control
297  */
298  if (strncmp(p, "sss", 3) == 0) {
299  LDAPSortKey **keys;
300  int ret;
301 
302  p += 3;
303  p = strchr(p, '=');
304  if (!p) {
305  REDEBUG("Server side sort extension must be in the format \"[!]sss=<key>[,key]\"");
306  return -1;
307  }
308  p++;
309 
310  ret = ldap_create_sort_keylist(&keys, p);
311  if (ret != LDAP_SUCCESS) {
312  REDEBUG("Invalid server side sort value \"%s\": %s", p, ldap_err2string(ret));
313  return -1;
314  }
315 
316  if (*sss) ldap_control_free(*sss);
317 
318  ret = ldap_create_sort_control(conn->handle, keys, is_critical ? 1 : 0, sss);
319  ldap_free_sort_keylist(keys);
320  if (ret != LDAP_SUCCESS) {
321  ERROR("Failed creating server sort control: %s", ldap_err2string(ret));
322  return -1;
323  }
324 
325  continue;
326  }
327 #endif
328 
329  RWDEBUG("URL extension \"%s\" ignored", p);
330  }
331 
332  return 0;
333 }

378 /** Normalise escape sequences in a DN
379  *
380  * Characters in a DN can either be escaped as
381  * @verbatim <hex><hex> @endverbatim or @verbatim <special> @endverbatim
382  *
383  * The LDAP directory chooses how characters are escaped, which can make
384  * local comparisons of DNs difficult.
385  *
386  * Here we search for hex sequences that match special chars, and convert
387  * them to the @verbatim <special> @endverbatim form.
388  *
389  * @note the resulting output string will only ever be shorter than the
390  * input, so it's fine to use the same buffer for both out and in.
391  *
392  * @param out Where to write the normalised DN.
393  * @param in The input DN.
394  * @return The number of bytes written to out.
395  */
396 size_t fr_ldap_util_normalise_dn(char *out, char const *in)
397 {
398  char const *p;
399  char *o = out;
400 
401  for (p = in; *p != '\0'; p++) {
402  if (p[0] == '\\') {
403  char c;
404 
405  /*
406  * Double backslashes get processed specially
407  */
408  if (p[1] == '\\') {
409  p += 1;
410  *o++ = p[0];
411  *o++ = p[1];
412  continue;
413  }
414 
415  /*
416  * Hex encodings that have an alternative
417  * special encoding, get rewritten to the
418  * special encoding.
419  */
420  if (fr_hex2bin(NULL, &FR_DBUFF_TMP((uint8_t *) &c, 1), &FR_SBUFF_IN(p + 1, 2), false) == 1) {
421  switch (c) {
422  case ' ':
423  case '#':
424  case '=':
425  case '"':
426  case '+':
427  case ',':
428  case ';':
429  case '<':
430  case '>':
431  case '\'':
432  *o++ = '\\';
433  *o++ = c;
434  p += 2;
435  continue;
436 
437  default:
438  break;
439  }
440  }
441  }
442  *o++ = *p;
443  }
444  *o = '\0';
445 
446  return o - out;
447 }

450  *
451  * Returns the length of the non matching string in full.
452  *
453  * @param full DN.
454  * @param part Partial DN as returned by ldap_parse_result.

459 size_t fr_ldap_common_dn(char const *full, char const *part)
460 {
461  size_t f_len, p_len, i;
462 
463  if (!full) return -1;
464 
465  f_len = strlen(full);
466 
467  if (!part) return -1;
468 
469  p_len = strlen(part);
470  if (!p_len) return f_len;
471 
472  if ((f_len < p_len) || !f_len) return -1;
473 
474  for (i = 0; i < p_len; i++) if (part[p_len - i] != full[f_len - i]) return -1;
475
