144 /*
145  * (Re-)read radiusd.conf into memory.
146  */
147 static int mod_instantiate(void *instance, CONF_SECTION *conf)
148 {
149  rlm_detail_t *inst = instance;
150  CONF_SECTION *cs;
151 
152  inst->name = cf_section_name2(conf);
153  if (!inst->name) inst->name = cf_section_name1(conf);
154 
155  /*
156  * Escape filenames only if asked.
157  */
158  if (inst->escape) {
159  inst->escape_func = rad_filename_escape;
160  } else {
161  inst->escape_func = rad_filename_make_safe;
162  }
163 
164  inst->ef = module_exfile_init(inst, conf, 256, 30, inst->locking, NULL, NULL);
165  if (!inst->ef) {
166  cf_log_err(conf, "Failed creating log file context");
167  return -1;
168  }
169 
170  /*
171  * Suppress certain attributes.
172  */
173  cs = cf_section_find(conf, "suppress", NULL);
174  if (cs) {
175  CONF_ITEM *ci;
176 
177  inst->ht = fr_hash_table_create(NULL, detail_hash, detail_cmp, NULL);
178 
179  for (ci = cf_item_next(cs, NULL);
180  ci != NULL;
181  ci = cf_item_next(cs, ci)) {
182  char const *attr;
183  fr_dict_attr_t const *da;
184 
185  if (!cf_item_is_pair(ci)) continue;
186 
187  attr = cf_pair_attr(cf_item_to_pair(ci));
188  if (!attr) continue; /* pair-anoia */
189 
190  if (fr_dict_attr_by_qualified_name(&da, dict_radius, attr, false) != FR_DICT_ATTR_OK) {
191  cf_log_perr(conf, "Failed resolving attribute");
192  return -1;
193  }
194 
195  /*
196  * Be kind to minor mistakes.
197  */
198  if (fr_hash_table_find_by_data(inst->ht, da)) {
199  WARN("Ignoring duplicate entry '%s'", attr);
200  continue;
201  }
202 
203 
204  if (!fr_hash_table_insert(inst->ht, da)) {
205  ERROR("Failed inserting '%s' into suppression table", attr);
206  return -1;
207  }
208 
209  DEBUG("'%s' suppressed, will not appear in detail output", attr);
210  }
211 
212  /*
213  * If we didn't suppress anything, delete the hash table.
214  */
215  if (fr_hash_table_num_elements(inst->ht) == 0) {
216  fr_hash_table_free(inst->ht);
217  inst->ht = NULL;
218  }
219  }
220 
221  return 0;
222 }

241 /** Write a single detail entry to file pointer
242  *
243  * @param[in] out Where to write entry.
244  * @param[in] inst Instance of rlm_detail.
245  * @param[in] request The current request.
246  * @param[in] packet associated with the request (request, reply...).
247  * @param[in] compat Write out entry in compatibility mode.
248  */
249 static int detail_write(FILE *out, rlm_detail_t const *inst, request_t *request, fr_radius_packet_t *packet, bool compat)
250 {
251  fr_pair_t *vp;
252  char timestamp[256];
253  char *header;
254 
255  if (tmpl_expand(&header, timestamp, sizeof(timestamp), request, inst->header, NULL, NULL) < 0) {
256  return -1;
257  }
258 
259  if (!packet->vps) {
260  RWDEBUG("Skipping empty packet");
261  return 0;
262  }
263 
264 #define WRITE(fmt, ...) do {\
265  if (fprintf(out, fmt, ## __VA_ARGS__) < 0) {\
266  RERROR("Failed writing to detail file: %s", fr_syserror(errno));\
267  return -1;\
268  }\
269 } while(0)
270 
271  WRITE("%s\n", timestamp);
272 
273  /*
274  * Write the information to the file.
275  */
276  if (!compat) {
277  /*
278  * Print out names, if they're OK.
279  * Numbers, if not.
280  */
281  if (is_radius_code(packet->code)) {
282  WRITE("\tPacket-Type = %s\n",
283  fr_dict_enum_name_by_value(attr_packet_type, fr_box_uint32(packet->code)));
284  } else {
285  WRITE("\tPacket-Type = %u\n", packet->code);
286  }
287  }
288 
289  if (inst->log_srcdst) {
290  fr_pair_t src_vp, dst_vp;
291 
292  memset(&src_vp, 0, sizeof(src_vp));
293  memset(&dst_vp, 0, sizeof(dst_vp));
294 
295  switch (packet->socket.inet.src_ipaddr.af) {
296  case AF_INET:
297  src_vp.da = attr_packet_src_ipv4_address;
298  fr_value_box_shallow(&src_vp.data, &packet->socket.inet.src_ipaddr, true);
299 
300  dst_vp.da = attr_packet_dst_ipv4_address;
301  fr_value_box_shallow(&dst_vp.data, &packet->socket.inet.dst_ipaddr, true);
302  break;
303 
304  case AF_INET6:
305  src_vp.da = attr_packet_src_ipv6_address;
306  fr_value_box_shallow(&src_vp.data, &packet->socket.inet.src_ipaddr, true);
307 
308  dst_vp.da = attr_packet_dst_ipv6_address;
309  fr_value_box_shallow(&dst_vp.data, &packet->socket.inet.dst_ipaddr, true);
310  break;
311 
312  default:
313  break;
314  }
315 
316  detail_fr_pair_fprint(request, out, &src_vp);
317  detail_fr_pair_fprint(request, out, &dst_vp);
318 
319  src_vp.da = attr_packet_src_port;
320  fr_value_box_shallow(&src_vp.data, packet->socket.inet.src_port, true);
321 
322  dst_vp.da = attr_packet_dst_port;
323  fr_value_box_shallow(&dst_vp.data, packet->socket.inet.dst_port, true);
324 
325  detail_fr_pair_fprint(request, out, &src_vp);
326  detail_fr_pair_fprint(request, out, &dst_vp);
327  }
328 
329  {
330  fr_cursor_t cursor;
331  /* Write each attribute/value to the log file */
332  for (vp = fr_cursor_init(&cursor, &packet->vps);
333  vp;
334  vp = fr_cursor_next(&cursor)) {
335  fr_token_t op;
336 
337  if (inst->ht && fr_hash_table_find_by_data(inst->ht, vp->da)) continue;
338 
339  /*
340  * Don't print passwords in old format...
341  */
342  if (compat && (vp->da == attr_user_password)) continue;
343 
344  /*
345  * Print all of the attributes, operator should always be '='.
346  */
347  op = vp->op;
348  vp->op = T_OP_EQ;
349  fr_pair_fprint(out, vp);
350  vp->op = op;
351  }
352  }
353 
354  /*
355  * Add the original protocol of the request, this should
356  * be used by the detail reader to set the default
357  * dictionary used for decoding.
358  */
359 // WRITE("\t%s = %s", attr_protocol->name, fr_dict_root(request->dict)->name);
360  WRITE("\tTimestamp = %lu\n", (unsigned long) fr_time_to_sec(request->packet->timestamp));
361 
362  WRITE("\n");
363 
364  return 0;
365 }

367 /*
368  * Do detail, compatible with old accounting
369  */
370 static unlang_action_t CC_HINT(nonnull) detail_do(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request,
371  fr_radius_packet_t *packet, bool compat)
372 {
373  int outfd, dupfd;
374  char buffer[DIRLEN];
375 
376  FILE *outfp;
377 
378 #ifdef HAVE_GRP_H
379  gid_t gid;
380  char *endptr;
381 #endif
382 
383  rlm_detail_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_detail_t);
384 
385  /*
386  * Generate the path for the detail file. Use the same
387  * format, but truncate at the last /. Then feed it
388  * through xlat_eval() to expand the variables.
389  */
390  if (xlat_eval(buffer, sizeof(buffer), request, inst->filename, inst->escape_func, NULL) < 0) {
391  RETURN_MODULE_FAIL;
392  }
393 
394  RDEBUG2("%s expands to %s", inst->filename, buffer);
395 
396  outfd = exfile_open(inst->ef, request, buffer, inst->perm);
397  if (outfd < 0) {
398  RPERROR("Couldn't open file %s", buffer);
399  /* coverity[missing_unlock] */
400  RETURN_MODULE_FAIL;
401  }
402 
403  if (inst->group != NULL) {
404  gid = strtol(inst->group, &endptr, 10);
405  if (*endptr != '\0') {
406  if (rad_getgid(request, &gid, inst->group) < 0) {
407  RDEBUG2("Unable to find system group '%s'", inst->group);
408  goto skip_group;
409  }
410  }
411 
412  if (chown(buffer, -1, gid) == -1) {
413  RDEBUG2("Unable to change system group of '%s'", buffer);
414  }
415  }
416 
417 skip_group:
418  outfp = NULL;
419  dupfd = dup(outfd);
420  if (dupfd < 0) {
421  RERROR("Failed to dup() file descriptor for detail file");
422  goto fail;
423  }
424 
425  /*
426  * Open the output fp for buffering.
427  */
428  if ((outfp = fdopen(dupfd, "a")) == NULL) {
429  RERROR("Couldn't open file %s: %s", buffer, fr_syserror(errno));
430  fail:
431  if (outfp) fclose(outfp);
432  exfile_close(inst->ef, request, outfd);
433  RETURN_MODULE_FAIL;
434  }
435 
436  if (detail_write(outfp, inst, request, packet, compat) < 0) goto fail;
437 
438  /*
439  * Flush everything
440  */
441  fclose(outfp);
442  exfile_close(inst->ef, request, outfd);
443 
444  /*
445  * And everything is fine.
446  */
447  RETURN_MODULE_OK;
448 }
