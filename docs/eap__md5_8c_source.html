48 /*
49  * We expect only RESPONSE for which SUCCESS or FAILURE is sent back
50  */
51 MD5_PACKET *eap_md5_extract(eap_round_t *eap_round)
52 {
53  md5_packet_t *data;
54  MD5_PACKET *packet;
55  unsigned short name_len;
56 
57  /*
58  * We need a response, of type EAP-MD5, with at least
59  * one byte of type data (EAP-MD5) following the 4-byte
60  * EAP-Packet header.
61  */
62  if (!eap_round ||
63  !eap_round->response ||
64  (eap_round->response->code != FR_MD5_RESPONSE) ||
65  eap_round->response->type.num != FR_EAP_METHOD_MD5 ||
66  !eap_round->response->type.data ||
67  (eap_round->response->length <= MD5_HEADER_LEN) ||
68  (eap_round->response->type.data[0] == 0)) {
69  ERROR("corrupted data");
70  return NULL;
71  }
72 
73  packet = talloc_zero(eap_round, MD5_PACKET);
74  if (!packet) return NULL;
75 
76  /*
77  * Code & id for MD5 & EAP are same
78  *
79  * but md5_length = length of the EAP-MD5 data, which
80  * doesn't include the EAP header, or the octet saying
81  * EAP-MD5.
82  */
83  packet->code = eap_round->response->code;
84  packet->id = eap_round->response->id;
85  packet->length = eap_round->response->length - (MD5_HEADER_LEN + 1);
86 
87  /*
88  * Sanity check the EAP-MD5 packet sent to us
89  * by the client.
90  */
91  data = (md5_packet_t *)eap_round->response->type.data;
92 
93  /*
94  * Already checked the size above.
95  */
96  packet->value_size = data->value_size;
97 
98  /*
99  * Allocate room for the data, and copy over the data.
100  */
101  packet->value = talloc_array(packet, uint8_t, packet->value_size);
102  if (!packet->value) {
103  talloc_free(packet);
104  return NULL;
105  }
106  memcpy(packet->value, data->value_name, packet->value_size);
107 
108  /*
109  * Name is optional and is present after Value, but we
110  * need to check for it, as eap_md5_compose()
111  */
112  name_len = packet->length - (packet->value_size + 1);
113  if (name_len) {
114  packet->name = talloc_array(packet, char, name_len + 1);
115  if (!packet->name) {
116  talloc_free(packet);
117  return NULL;
118  }
119  memcpy(packet->name, data->value_name + packet->value_size,
120  name_len);
121  packet->name[name_len] = 0;
122  }
123 
124  return packet;
125 }

128 /*
129  * verify = MD5(id+password+challenge_sent)
130  */
131 int eap_md5_verify(MD5_PACKET *packet, fr_pair_t* password,
132  uint8_t *challenge)
133 {
134  char *ptr;
135  char string[1 + FR_MAX_STRING_LEN*2];
136  uint8_t digest[16];
137  unsigned short len;
138 
139  /*
140  * Sanity check it.
141  */
142  if (packet->value_size != 16) {
143  ERROR("Expected 16 bytes of response to challenge, got %d", packet->value_size);
144  return 0;
145  }
146 
147  len = 0;
148  ptr = string;
149 
150  /*
151  * This is really rad_chap_pwencode()...
152  */
153  *ptr++ = packet->id;
154  len++;
155  memcpy(ptr, password->vp_strvalue, password->vp_length);
156  ptr += password->vp_length;
157  len += password->vp_length;
158 
159  /*
160  * The challenge size is hard-coded.
161  */
162  memcpy(ptr, challenge, MD5_CHALLENGE_LEN);
163  len += MD5_CHALLENGE_LEN;
164 
165  fr_md5_calc(digest, (uint8_t *)string, len);
166 
167  /*
168  * The length of the response is always 16 for MD5.
169  */
170  if (fr_digest_cmp(digest, packet->value, 16) != 0) {
171  return 0;
172  }
173 
174  return 1;
175 }

177 /*
178  * Compose the portions of the reply packet specific to the
179  * EAP-MD5 protocol, in the EAP reply typedata
180  */
181 int eap_md5_compose(eap_round_t *eap_round, MD5_PACKET *reply)
182 {
183  uint8_t *ptr;
184  unsigned short name_len;
185 
186  /*
187  * We really only send Challenge (EAP-Identity),
188  * and EAP-Success, and EAP-Failure.
189  */
190  if (reply->code < 3) {
191  eap_round->request->type.num = FR_EAP_METHOD_MD5;
192 
193  fr_assert(reply->length > 0);
194 
195  eap_round->request->type.data = talloc_array(eap_round->request,
196  uint8_t,
197  reply->length);
198  if (!eap_round->request->type.data) {
199  talloc_free(reply);
200  return 0;
201  }
202  ptr = eap_round->request->type.data;
203  *ptr++ = (uint8_t)(reply->value_size & 0xFF);
204  memcpy(ptr, reply->value, reply->value_size);
205 
206  /* Just the Challenge length */
207  eap_round->request->type.length = reply->value_size + 1;
208 
209  /*
210  * Return the name, if necessary.
211  *
212  * Don't see why this is *ever* necessary...
213  */
214  name_len = reply->length - (reply->value_size + 1);
215  if (name_len && reply->name) {
216  ptr += reply->value_size;
217  memcpy(ptr, reply->name, name_len);
218  /* Challenge length + Name length */
219  eap_round->request->type.length += name_len;
220  }
221  } else {
222  eap_round->request->type.length = 0;
223  /* TODO: In future we might add message here wrt rfc1994 */
224  }
225  eap_round->request->code = reply->code;
226  talloc_free(reply);
227 
228  return 1;
229 }
