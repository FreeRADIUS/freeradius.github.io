44 /** Calculate powers
45  *
46  * @author Orson Peters
47  * @note Borrowed from the gist here: https://gist.github.com/nightcracker/3551590.
48  *
49  * @param base a 32bit signed integer.
50  * @param exp amount to raise base by.
51  * @return base ^ pow, or 0 on underflow/overflow.
52  */
53 static int64_t fr_pow(int64_t base, int64_t exp)
54 {
55  static const uint8_t highest_bit_set[] = {
56  0, 1, 2, 2, 3, 3, 3, 3,
57  4, 4, 4, 4, 4, 4, 4, 4,
58  5, 5, 5, 5, 5, 5, 5, 5,
59  5, 5, 5, 5, 5, 5, 5, 5,
60  6, 6, 6, 6, 6, 6, 6, 6,
61  6, 6, 6, 6, 6, 6, 6, 6,
62  6, 6, 6, 6, 6, 6, 6, 6,
63  6, 6, 6, 6, 6, 6, 6, 6 // anything past 63 is a guaranteed overflow with base > 1
64  };
65 
66  int64_t result = 1;
67 
68  if (exp > 63) {
69  if (base == 1) return 1;
70  if (base == -1) return 1 - 2 * (exp & 1);
71  return 0; /* overflow */
72  }
73 
74  switch (highest_bit_set[exp]) {
75  case 6:
76  if (exp & 1) result *= base;
77  exp >>= 1;
78  base *= base;
79  FALL_THROUGH;
80  case 5:
81  if (exp & 1) result *= base;
82  exp >>= 1;
83  base *= base;
84  FALL_THROUGH;
85  case 4:
86  if (exp & 1) result *= base;
87  exp >>= 1;
88  base *= base;
89  FALL_THROUGH;
90  case 3:
91  if (exp & 1) result *= base;
92  exp >>= 1;
93  base *= base;
94  FALL_THROUGH;
95  case 2:
96  if (exp & 1) result *= base;
97  exp >>= 1;
98  base *= base;
99  FALL_THROUGH;
100  case 1:
101  if (exp & 1) result *= base;
102  FALL_THROUGH;
103  default:
104  return result;
105  }
106 }

159 static bool get_number(request_t *request, char const **string, int64_t *answer)
160 {
161  fr_sbuff_term_t const bareword_terminals =
162  FR_SBUFF_TERMS(
163  L("\t"),
164  L("\n"),
165  L(" "),
166  L("%"),
167  L("&"),
168  L(")"),
169  L("+"),
170  L("-"),
171  L("/"),
172  L("^"),
173  L("|")
174  );
175  fr_sbuff_parse_rules_t const p_rules = { .terminals = &bareword_terminals };
176  int64_t x;
177  bool invert = false;
178  bool negative = false;
179  char const *p = *string;
180  tmpl_t *vpt = NULL;
181 
182  /*
183  * Look for a number.
184  */
185  fr_skip_whitespace(p);
186 
187  /*
188  * ~1 == 0xff...ffe
189  */
190  if (*p == '~') {
191  invert = true;
192  p++;
193  }
194 
195  /*
196  * No algrebraic operator found, the next thing
197  * MUST be a number.
198  *
199  * If it isn't, then we die.
200  */
201  if ((*p == '0') && (p[1] == 'x')) {
202  char *end;
203 
204  x = strtoul(p, &end, 16);
205  p = end;
206  goto done;
207  }
208 
209  if (*p == '-') {
210  negative = true;
211  p++;
212  }
213 
214  /*
215  * Look for an attribute.
216  */
217  if (*p == '&') {
218  int i, max, err;
219  ssize_t slen;
220  fr_pair_t *vp;
221  fr_cursor_t cursor;
222  tmpl_cursor_ctx_t cc;
223 
224  slen = tmpl_afrom_attr_substr(request, NULL, &vpt,
225  &FR_SBUFF_IN(p, strlen(p)),
226  &p_rules,
227  &(tmpl_rules_t){ .dict_def = request->dict });
228  if (slen <= 0) {
229  RPEDEBUG("Failed parsing attribute name '%s'", p);
230  return false;
231  }
232 
233  p += slen;
234 
235  if (tmpl_num(vpt) == NUM_COUNT) {
236  REDEBUG("Attribute count is not supported");
237  return false;
238  }
239 
240  if (tmpl_num(vpt) == NUM_ALL) {
241  max = 65535;
242  } else {
243  max = 1;
244  }
245 
246  x = 0;
247  for (i = 0, vp = tmpl_cursor_init(&err, NULL, &cc, &cursor, request, vpt);
248  (i < max) && (vp != NULL);
249  i++, vp = fr_cursor_next(&cursor)) {
250  int64_t y;
251  fr_value_box_t value;
252 
253  if (vp->vp_type != FR_TYPE_UINT64) {
254  if (fr_value_box_cast(vp, &value, FR_TYPE_UINT64, NULL, &vp->data) < 0) {
255  RPEDEBUG("Failed converting &%.*s to an integer value", (int) vpt->len,
256  vpt->name);
257  error:
258  tmpl_cursor_clear(&cc);
259  return false;
260  }
261  if (value.vb_uint64 > INT64_MAX) {
262  overflow:
263  talloc_free(vpt);
264  REDEBUG("Value of &%.*s (%pV) would overflow a signed 64bit integer "
265  "(our internal arithmetic type)", (int)vpt->len, vpt->name, &value);
266  goto error;
267  }
268  y = (int64_t)value.vb_uint64;
269 
270  RINDENT();
271  RDEBUG3("&%.*s --> %" PRIu64, (int)vpt->len, vpt->name, y);
272  REXDENT();
273  } else {
274  if (vp->vp_uint64 > INT64_MAX) {
275  /*
276  * So we can print out the correct value
277  * in the overflow error message.
278  */
279  fr_value_box_copy(NULL, &value, &vp->data);
280  goto overflow;
281  }
282  y = (int64_t)vp->vp_uint64;
283  }
284 
285  /*
286  * Check for overflow without actually overflowing.
287  */
288  if ((y > 0) && (x > (int64_t) INT64_MAX - y)) goto overflow;
289 
290  if ((y < 0) && (x < (int64_t) INT64_MIN - y)) goto overflow;
291 
292  x += y;
293  } /* loop over all found VPs */
294  tmpl_cursor_clear(&cc);
295 
296  if (err != 0) {
297  RWDEBUG("Can't find &%.*s. Using 0 as operand value", (int)vpt->len, vpt->name);
298  goto done;
299  }
300 
301  goto done;
302  }
303 
304  /*
305  * Do brackets recursively
306  */
307  if (*p == '(') {
308  p++;
309  if (!get_expression(request, &p, &x, TOKEN_NONE)) return false;
310 
311  if (*p != ')') {
312  REDEBUG("No trailing ')'");
313  return false;
314  }
315  p++;
316  goto done;
317  }
318 
319  if ((*p < '0') || (*p > '9')) {
320  REDEBUG("Not a number at \"%s\"", p);
321  return false;
322  }
323 
324  /*
325  * This is doing it the hard way, but it also allows
326  * us to increment 'p'.
327  */
328  x = 0;
329  while ((*p >= '0') && (*p <= '9')) {
330  x *= 10;
331  x += (*p - '0');
332  p++;
333  }
334 
335 done:
336  if (vpt) talloc_free(vpt);
337 
338  if (invert) x = ~x;
339 
340  if (negative) x = -x;
341 
342  *string = p;
343  *answer = x;
344  return true;
345 }

347 static bool calc_result(request_t *request, int64_t lhs, expr_token_t op, int64_t rhs, int64_t *answer)
348 {
349  switch (op) {
350  default:
351  case TOKEN_SUBTRACT:
352  rhs = -rhs;
353  FALL_THROUGH;
354 
355  case TOKEN_ADD:
356  if ((rhs > 0) && (lhs > (int64_t) INT64_MAX - rhs)) {
357  overflow:
358  REDEBUG("Numerical overflow in expression!");
359  return false;
360  }
361 
362  if ((rhs < 0) && (lhs < (int64_t) INT64_MIN - rhs)) goto overflow;
363 
364  *answer = lhs + rhs;
365  break;
366 
367  case TOKEN_DIVIDE:
368  if (rhs == 0) {
369  REDEBUG("Division by zero in expression!");
370  return false;
371  }
372 
373  *answer = lhs / rhs;
374  break;
375 
376  case TOKEN_REMAINDER:
377  if (rhs == 0) {
378  REDEBUG("Division by zero!");
379  return false;
380  }
381 
382  *answer = lhs % rhs;
383  break;
384 
385  case TOKEN_MULTIPLY:
386  *answer = lhs * rhs;
387  break;
388 
389  case TOKEN_LSHIFT:
390  if (rhs > 62) {
391  REDEBUG("Shift must be less than 62 (was %lld)", (long long int) rhs);
392  return false;
393  }
394 
395  *answer = lhs << rhs;
396  break;
397 
398  case TOKEN_RSHIFT:
399  if (rhs > 62) {
400  REDEBUG("Shift must be less than 62 (was %lld)", (long long int) rhs);
401  return false;
402  }
403 
404  *answer = lhs >> rhs;
405  break;
406 
407  case TOKEN_AND:
408  *answer = lhs & rhs;
409  break;
410 
411  case TOKEN_OR:
412  *answer = lhs | rhs;
413  break;
414 
415  case TOKEN_POWER:
416  if (rhs > 63) {
417  REDEBUG("Exponent must be between 0-63 (was %lld)", (long long int) rhs);
418  return false;
419  }
420 
421  if (lhs > 65535) {
422  REDEBUG("Base must be between 0-65535 (was %lld)", (long long int) lhs);
423  return false;
424  }
425 
426  *answer = fr_pow(lhs, rhs);
427  break;
428  }
429 
430  return true;
431 }

433 static bool get_operator(request_t *request, char const **string, expr_token_t *op)
434 {
435  int i;
436  char const *p = *string;
437 
438  /*
439  * All tokens are one character.
440  */
441  for (i = 0; map[i].token != TOKEN_LAST; i++) {
442  if (*p == map[i].op) {
443  *op = map[i].token;
444  *string = p + 1;
445  return true;
446  }
447  }
448 
449  if ((p[0] == '<') && (p[1] == '<')) {
450  *op = TOKEN_LSHIFT;
451  *string = p + 2;
452  return true;
453  }
454 
455  if ((p[0] == '>') && (p[1] == '>')) {
456  *op = TOKEN_RSHIFT;
457  *string = p + 2;
458  return true;
459  }
460 
461  REDEBUG("Expected operator at \"%s\"", p);
462 
463  return false;
464 }

467 static bool get_expression(request_t *request, char const **string, int64_t *answer, expr_token_t prev)
468 {
469  int64_t lhs, rhs;
470  char const *p, *op_p;
471  expr_token_t this;
472 
473  p = *string;
474 
475  if (!get_number(request, &p, &lhs)) return false;
476 
477 redo:
478  fr_skip_whitespace(p);
479 
480  /*
481  * A number by itself is OK.
482  */
483  if (!*p || (*p == ')')) {
484  *answer = lhs;
485  *string = p;
486  return true;
487  }
488 
489  /*
490  * Peek at the operator.
491  */
492  op_p = p;
493  if (!get_operator(request, &p, &this)) return false;
494 
495  /*
496  * a + b + c ... = (a + b) + c ...
497  * a * b + c ... = (a * b) + c ...
498  *
499  * Feed the current number to the caller, who will take
500  * care of continuing.
501  */
502  if (precedence[this] <= precedence[prev]) {
503  *answer = lhs;
504  *string = op_p;
505  return true;
506  }
507 
508  /*
509  * a + b * c ... = a + (b * c) ...
510  */
511  if (!get_expression(request, &p, &rhs, this)) return false;
512 
513  if (!calc_result(request, lhs, this, rhs, answer)) return false;
514 
515  /*
516  * There may be more to calculate. The answer we
517  * calculated here is now the LHS of the lower priority
518  * operation which follows the current expression. e.g.
519  *
520  * a * b + c ... = (a * b) + c ...
521  * = d + c ...
522  */
523  lhs = *answer;
524  goto redo;
525 }
