30 /** Initialize a retransmission counter
31  *
32  * @param[in,out] r the retransmission structure
33  * @param now when the retransmission starts
34  * @param config the counters to track. They shouldn't change while the retransmission is happening
35  */
36 int fr_retry_init(fr_retry_t *r, fr_time_t now, fr_retry_config_t const *config)
37 {
38  fr_time_delta_t scale, rt;
39  uint128_t delay;
40 
41  memset(r, 0, sizeof(*r));
42 
43  r->config = config;
44  r->count = 1;
45  r->start = now;
46  r->updated = now;
47 
48  /*
49  * Initial:
50  *
51  * RT = IRT + RAND * IRT
52  * = IRT * (1 + RAND)
53  */
54  scale = fr_rand();
55  scale += ((fr_time_delta_t) 1) << 32; /* multiple it by 1 * 2^32 */
56  scale -= ((fr_time_delta_t) 1) << 31; /* scale it -2^31..+2^31 */
57 
58  delay = uint128_mul64(scale, r->config->irt);
59  rt = uint128_to_64(uint128_rshift(delay, 32));
60 
61  r->rt = rt;
62  r->next = now + rt;
63 
64  return 0;
65 }

67 /** Initialize a retransmission counter
68  *
69  * @param[in,out] r the retransmission structure
70  * @param now the current time
71  * @return
72  * - FR_RETRTY_CONTINUE - continue retransmitting
73  * - FR_RETRY_MRC - stop, maximum retransmission count has been reached
74  * - FR_RETRY_MDR - stop, maximum retransmission duration has been reached.
75  */
76 fr_retry_state_t fr_retry_next(fr_retry_t *r, fr_time_t now)
77 {
78  fr_time_delta_t scale, rt;
79  uint128_t delay;
80 
81  /*
82  * Increment retransmission counter
83  */
84  r->count++;
85  r->updated = now;
86 
87  /*
88  * We retried too many times. Fail.
89  */
90  if (r->config->mrc && (r->count > r->config->mrc)) {
91  return FR_RETRY_MRC;
92  }
93 
94 redo:
95  /*
96  * Cap delay at MRD
97  */
98  if (r->config->mrd) {
99  fr_time_t end;
100 
101  end = r->start;
102  end += r->config->mrd;
103 
104  if (now > end) {
105  return FR_RETRY_MRD;
106  }
107  }
108 
109  /*
110  * RFC 5080 Section 2.2.1
111  *
112  * RAND gives a random number between -0.1 and +0.1
113  *
114  * Our random number generator returns 0..2^32, so we
115  * have to scale everything relative to that.
116  *
117  * RT = 2*RTprev + RAND*RTprev
118  * = RTprev * (2 + RAND)
119  */
120  scale = fr_rand();
121  scale -= ((fr_time_delta_t) 1) << 31; /* scale it -2^31..+2^31 */
122  scale += ((fr_time_delta_t) 1) << 33; /* multiple it by 2 * 2^32 */
123 
124  delay = uint128_mul64(scale, r->rt);
125  rt = uint128_to_64(uint128_rshift(delay, 32));
126 
127  /*
128  * Cap delay at MRT.
129  *
130  * RT = MRT + RAND * MRT
131  * = MRT * (1 + RAND)
132  */
133  if (r->config->mrt && (rt > r->config->mrt)) {
134  scale = fr_rand();
135  scale -= ((fr_time_delta_t) 1) << 31; /* scale it -2^31..+2^31 */
136  scale += ((fr_time_delta_t) 1) << 32; /* multiple it by 1 * 2^32 */
137 
138  delay = uint128_mul64(scale, r->config->mrt);
139  rt = uint128_to_64(uint128_rshift(delay, 32));
140  }
141 
142  /*
143  * And finally set the retransmission timer.
144  */
145  r->rt = rt;
146 
147  /*
148  * Add in the retransmission delay. Note that we send
149  * the packet at "next + rt", and not "now + rt". That
150  * way the timer won't drift.
151  */
152  r->next += rt;
153 
154  /*
155  * The "next" retransmission time is in the past, AND
156  * we're already halfway through the time after that.
157  * Skip this retransmission, and set the time for the
158  * next one.
159  *
160  * i.e. if we weren't serviced for one event, just skip
161  * it, and go to the next one.
162  */
163  if ((r->next + (rt / 2)) < now) goto redo;
164 
165  return FR_RETRY_CONTINUE;
166 }
