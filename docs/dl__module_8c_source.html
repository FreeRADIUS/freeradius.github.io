141 /** Check if the magic number in the module matches the one in the library
142  *
143  * This is used to detect potential ABI issues caused by running with modules which
144  * were built for a different version of the server.
145  *
146  * @param[in] cs being parsed.
147  * @param[in] module Common fields from module's exported interface struct.
148  * @returns
149  * - 0 on success.
150  * - -1 if prefix mismatch.
151  * - -2 if version mismatch.
152  * - -3 if commit mismatch.
153  */
154 static int dl_module_magic_verify(CONF_SECTION const *cs, dl_module_common_t const *module)
155 {
156 #ifdef HAVE_DLADDR
157  Dl_info dl_info;
158  dladdr(module, &dl_info);
159 #endif
160 
161  if (MAGIC_PREFIX(module->magic) != MAGIC_PREFIX(RADIUSD_MAGIC_NUMBER)) {
162 #ifdef HAVE_DLADDR
163  cf_log_err(cs, "Failed loading module rlm_%s from file %s", module->name, dl_info.dli_fname);
164 #endif
165  cf_log_err(cs, "Application and rlm_%s magic number (prefix) mismatch."
166  " application: %x module: %x", module->name,
167  MAGIC_PREFIX(RADIUSD_MAGIC_NUMBER),
168  MAGIC_PREFIX(module->magic));
169  return -1;
170  }
171 
172  if (MAGIC_VERSION(module->magic) != MAGIC_VERSION(RADIUSD_MAGIC_NUMBER)) {
173 #ifdef HAVE_DLADDR
174  cf_log_err(cs, "Failed loading module rlm_%s from file %s", module->name, dl_info.dli_fname);
175 #endif
176  cf_log_err(cs, "Application and rlm_%s magic number (version) mismatch."
177  " application: %lx module: %lx", module->name,
178  (unsigned long) MAGIC_VERSION(RADIUSD_MAGIC_NUMBER),
179  (unsigned long) MAGIC_VERSION(module->magic));
180  return -2;
181  }
182 
183  if (MAGIC_COMMIT(module->magic) != MAGIC_COMMIT(RADIUSD_MAGIC_NUMBER)) {
184 #ifdef HAVE_DLADDR
185  cf_log_err(cs, "Failed loading module rlm_%s from file %s", module->name, dl_info.dli_fname);
186 #endif
187  cf_log_err(cs, "Application and rlm_%s magic number (commit) mismatch."
188  " application: %lx module: %lx", module->name,
189  (unsigned long) MAGIC_COMMIT(RADIUSD_MAGIC_NUMBER),
190  (unsigned long) MAGIC_COMMIT(module->magic));
191  return -3;
192  }
193 
194  return 0;
195 }

261 /** Allocate module instance data, and parse the module's configuration
262  *
263  * @param[in] dl_inst to allocate this instance data in.
264  * @param[in] module to alloc instance data for.
265  */
266 static void dl_module_instance_data_alloc(dl_module_inst_t *dl_inst, dl_module_t const *module)
267 {
268  void *data;
269 
270  /*
271  * If there is supposed to be instance data, allocate it now.
272  *
273  * If the structure is zero length then allocation will still
274  * succeed, and will create a talloc chunk header.
275  *
276  * This is needed so we can resolve instance data back to
277  * dl_module_instance_t/dl_module_t/dl_t.
278  */
279  MEM(data = talloc_zero_array(dl_inst, uint8_t, module->common->inst_size));
280 
281  if (!module->common->inst_type) {
282  talloc_set_name(data, "%s_t", module->dl->name ? module->dl->name : "config");
283  } else {
284  talloc_set_name(data, "%s", module->common->inst_type);
285  }
286  dl_inst->data = data;
287 
288  /*
289  * Must be done before setting the destructor to ensure the
290  * destructor can find the dl_module_inst_t associated
291  * with the data.
292  */
293  fr_assert(dl_module_loader != NULL);
294  rbtree_insert(dl_module_loader->inst_data_tree, dl_inst); /* Duplicates not possible */
295 
296  talloc_set_destructor(data, _dl_module_instance_data_free);
297 }

299 /** Decrement the reference count of the dl, eventually freeing it
300  *
301  */
302 static int _dl_module_free(dl_module_t *dl_module)
303 {
304  /*
305  * dl is empty if we tried to load it and failed.
306  */
307  if (dl_module->dl) {
308  if (DEBUG_ENABLED4) {
309  DEBUG4("%s unloaded. Handle address %p, symbol address %p", dl_module->dl->name,
310  dl_module->dl->handle, dl_module->common);
311  } else {
312  DEBUG3("%s unloaded", dl_module->dl->name);
313  }
314  }
315 
316  if (dl_module->in_tree) {
317  rbtree_deletebydata(dl_module_loader->module_tree, dl_module);
318  dl_module->in_tree = false;
319  }
320 
321  dl_free(dl_module->dl);
322 
323  return 0;
324 }

326 /** Load a module library using dlopen() or return a previously loaded module from the cache
327  *
328  * When the dl_module_t is no longer used, talloc_free() may be used to free it.
329  *
330  * When all references to the original dlhandle are freed, dlclose() will be called on the
331  * dlhandle to unload the module.
332  *
333  * @param[in] conf section describing the module's configuration. This is only used
334  * to give error messages context, and for initialization.
335  * @param[in] parent The dl_module_t of the parent module, e.g. rlm_sql for rlm_sql_postgresql.
336  * @param[in] name of the module e.g. sql for rlm_sql.
337  * @param[in] type Used to determine module name prefixes. Must be one of:
338  * - DL_MODULE_TYPE_MODULE
339  * - DL_MODULE_TYPE_PROTO
340  * - DL_MODULE_TYPE_SUBMODULE
341  * @return
342  * - Module handle holding dlhandle, and module's public interface structure.
343  * - NULL if module couldn't be loaded, or some other error occurred.
344  */
345 dl_module_t const *dl_module(CONF_SECTION *conf, dl_module_t const *parent, char const *name, dl_module_type_t type)
346 {
347  dl_module_t *dl_module = NULL;
348  dl_t *dl = NULL;
349  char *module_name = NULL;
350  char *p, *q;
351  dl_module_common_t const *common;
352 
353  DL_INIT_CHECK;
354 
355  if (parent) {
356  module_name = talloc_typed_asprintf(NULL, "%s_%s_%s",
357  fr_table_str_by_value(dl_module_type_prefix,
358  parent->type, "<INVALID>"),
359  parent->common->name, name);
360  } else {
361  module_name = talloc_typed_asprintf(NULL, "%s_%s",
362  fr_table_str_by_value(dl_module_type_prefix, type, "<INVALID>"),
363  name);
364  }
365 
366  if (!module_name) return NULL;
367 
368  for (p = module_name, q = p + talloc_array_length(p) - 1; p < q; p++) *p = tolower(*p);
369 
370  /*
371  * If the module's already been loaded, increment the reference count.
372  */
373  dl_module = rbtree_finddata(dl_module_loader->module_tree,
374  &(dl_module_t){ .dl = &(dl_t){ .name = module_name }});
375  if (dl_module) {
376  talloc_free(module_name);
377  talloc_increase_ref_count(dl_module);
378  return dl_module;
379  }
380 
381  dl_module = talloc_zero(dl_module_loader, dl_module_t);
382  dl_module->parent = parent;
383  dl_module->type = type;
384  talloc_set_destructor(dl_module, _dl_module_free); /* Do this late */
385 
386  /*
387  * Pass in dl_module as the uctx so that
388  * we can get at it in any callbacks.
389  */
390  dl = dl_by_name(dl_module_loader->dl_loader, module_name, dl_module, false);
391  if (!dl) {
392  cf_log_perr(conf, "Failed to link to module \"%s\"", module_name);
393  cf_log_err(conf, "Make sure it (and all its dependent libraries!) are in the search path"
394  " of your system's ld");
395  error:
396  talloc_free(module_name);
397  talloc_free(dl_module); /* Do not free dl explicitly, it's handled by the destructor */
398  return NULL;
399  }
400  dl_module->dl = dl;
401 
402  DEBUG3("%s loaded, checking if it's valid", module_name);
403 
404  common = dlsym(dl->handle, module_name);
405  if (!common) {
406  cf_log_err(conf, "Could not find \"%s\" symbol in module: %s", module_name, dlerror());
407  goto error;
408  }
409  dl_module->common = common;
410 
411  /*
412  * Before doing anything else, check if it's sane.
413  */
414  if (dl_module_magic_verify(conf, common) < 0) goto error;
415 
416  DEBUG3("%s validated. Handle address %p, symbol address %p", module_name, dl, common);
417 
418  if (dl_symbol_init(dl_module_loader->dl_loader, dl) < 0) {
419  cf_log_perr(conf, "Failed calling initializers for module \"%s\"", module_name);
420  goto error;
421  }
422 
423  cf_log_info(conf, "Loaded module \"%s\"", module_name);
424 
425  /*
426  * Add the module to the dl cache
427  */
428  dl_module->in_tree = rbtree_insert(dl_module_loader->module_tree, dl_module);
429  if (!dl_module->in_tree) {
430  cf_log_err(conf, "Failed caching module \"%s\"", module_name);
431  goto error;
432  }
433 
434  talloc_free(module_name);
435 
436  return dl_module;
437 }

439 /** Free a module instance, removing it from the instance tree
440  *
441  * Also decrements the reference count of the module potentially unloading it.
442  *
443  * @param[in] dl_inst to free.
444  * @return 0.
445  */
446 static int _dl_module_instance_free(dl_module_inst_t *dl_inst)
447 {
448  /*
449  * Ensure sane free order, and that all destructors
450  * run before the .so/.dylib is unloaded.
451  *
452  * This *MUST* be done *BEFORE* decrementing the
453  * reference count on the module.
454  *
455  * It also *MUST* be done before removing this struct
456  * from the inst_data_tree, so the detach destructor
457  * can find the dl_module_inst_t associated with
458  * the opaque data.
459  */
460  talloc_free_children(dl_inst);
461 
462  /*
463  * Remove this instance from the tracking tree.
464  */
465  fr_assert(dl_module_loader != NULL);
466  rbtree_deletebydata(dl_module_loader->inst_data_tree, dl_inst);
467 
468  /*
469  * Decrements the reference count. The module object
470  * won't be unloaded until all instances of that module
471  * have been destroyed.
472  */
473  talloc_decrease_ref_count(dl_inst->module);
474 
475  return 0;
476 }

496 /** Load a module and parse its #CONF_SECTION in one operation
497  *
498  * When this instance is no longer needed, it should be freed with talloc_free().
499  * When all instances of a particular module are unloaded, the dl handle will be closed,
500  * unloading the module.
501  *
502  * @param[in] ctx to allocate structures in.
503  * @param[out] out where to write our #dl_module_inst_t containing the module
504  * handle and instance.
505  * @param[in] conf section to parse.
506  * @param[in] parent of module instance.
507  * @param[in] name of the module to load .e.g. 'udp' for 'proto_radius_udp'
508  * if the parent were 'proto_radius'.
509  * @param[in] type of module to load.
510  *
511  * @return
512  * - 0 on success.
513  * - -1 on failure.
514  */
515 int dl_module_instance(TALLOC_CTX *ctx, dl_module_inst_t **out,
516  CONF_SECTION *conf, dl_module_inst_t const *parent,
517  char const *name, dl_module_type_t type)
518 {
519  dl_module_inst_t *dl_inst;
520  char const *name2;
521 
522  DL_INIT_CHECK;
523 
524  MEM(dl_inst = talloc_zero(ctx, dl_module_inst_t));
525 
526  /*
527  * Find a section with the same name as the module
528  */
529  dl_inst->module = dl_module(conf, parent ? parent->module : NULL, name, type);
530  if (!dl_inst->module) {
531  talloc_free(dl_inst);
532  return -1;
533  }
534 
535  /*
536  * ctx here is the main module's instance data
537  */
538  dl_module_instance_data_alloc(dl_inst, dl_inst->module);
539 
540  talloc_set_destructor(dl_inst, _dl_module_instance_free);
541 
542  /*
543  * Associate the module instance with the conf section
544  * *before* executing any parse rules that might need it.
545  */
546  cf_data_add(conf, dl_inst, dl_inst->module->dl->name, false);
547 
548  if (dl_inst->module->common->config && conf) {
549  if ((cf_section_rules_push(conf, dl_inst->module->common->config)) < 0 ||
550  (cf_section_parse(dl_inst->data, dl_inst->data, conf) < 0)) {
551  cf_log_err(conf, "Failed evaluating configuration for module \"%s\"",
552  dl_inst->module->dl->name);
553  talloc_free(dl_inst);
554  return -1;
555  }
556  }
557 
558  name2 = cf_section_name2(conf);
559  if (name2) {
560  dl_inst->name = talloc_typed_strdup(dl_inst, name2);
561  } else {
562  dl_inst->name = talloc_typed_strdup(dl_inst, cf_section_name1(conf));
563  }
564 
565  dl_inst->conf = conf;
566  dl_inst->parent = parent;
567 
568  *out = dl_inst;
569 
570  return 0;
571 }

584 static int _dl_module_loader_free(dl_module_loader_t *dl_module_l)
585 {
586  int ret = 0;
587 
588  if (rbtree_num_elements(dl_module_l->inst_data_tree) > 0) {
589  ret = -1;
590 #ifndef NDEBUG
591  WARN("Refusing to cleanup dl loader, the following module instances are still in use:");
592  rbtree_walk(dl_module_l->inst_data_tree, RBTREE_IN_ORDER, _dl_inst_walk_print, NULL);
593 #endif
594  goto finish;
595  }
596 
597  /*
598  * Do this as an explicit step, as this free can fail
599  */
600  ret = talloc_free(dl_module_l->dl_loader);
601  if (ret != 0) {
602  PWARN("dl loader not freed");
603  }
604 
605 finish:
606  if (ret != 0) {
607 #ifndef NDEBUG
608  WARN("This may appear as a leak in talloc memory reports");
609 #endif
610  } else {
611  dl_module_loader = NULL;
612  }
613 
614  return ret;
615 }

628  *
629  */
630 dl_module_loader_t *dl_module_loader_init(char const *lib_dir)
631 {
632  if (dl_module_loader) {
633  /*
634  * Allow it to update the search path.
635  */
636  if (dl_search_path_set(dl_module_loader->dl_loader, lib_dir) < 0) {

638  }
639 
640  return dl_module_loader;
641  }
642 
643  dl_module_loader = talloc_zero(NULL, dl_module_loader_t);
644  if (!dl_module_loader) {
645  ERROR("Failed initialising uctx for dl_loader");
646  return NULL;
647  }
648 
649  dl_module_loader->dl_loader = dl_loader_init(NULL, dl_module_loader, false, true);
650  if (!dl_module_loader) {
651  PERROR("Failed initialising dl_loader");
652  error:
653  TALLOC_FREE(dl_module_loader);
654  return NULL;
655  }
656  dl_search_path_prepend(dl_module_loader->dl_loader, lib_dir);
