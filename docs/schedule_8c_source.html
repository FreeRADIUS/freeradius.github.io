162 /** Entry point for worker threads
163  *
164  * @param[in] arg the fr_schedule_worker_t
165  * @return NULL
166  */
167 static void *fr_schedule_worker_thread(void *arg)
168 {
169  TALLOC_CTX *ctx;
170  fr_schedule_worker_t *sw = talloc_get_type_abort(arg, fr_schedule_worker_t);
171  fr_schedule_t *sc = sw->sc;
172  fr_schedule_child_status_t status = FR_CHILD_FAIL;
173  fr_schedule_network_t *sn;
174  char worker_name[32];
175 
176  worker_id = sw->id; /* Store the current worker ID */
177 
178  snprintf(worker_name, sizeof(worker_name), "Worker %d", sw->id);
179 
180  sw->ctx = ctx = talloc_init("%s", worker_name);
181  if (!ctx) {
182  ERROR("%s - Failed allocating memory", worker_name);
183  goto fail;
184  }
185 
186  INFO("%s - Starting", worker_name);
187 
188  sw->el = fr_event_list_alloc(ctx, NULL, NULL);
189  if (!sw->el) {
190  PERROR("%s - Failed creating event list", worker_name);
191  goto fail;
192  }
193 
194 
195  sw->worker = fr_worker_create(ctx, sw->el, worker_name, sc->log, sc->lvl, &sc->config->worker);
196  if (!sw->worker) {
197  PERROR("%s - Failed creating worker", worker_name);
198  goto fail;
199  }
200 
201  /*
202  * @todo make this a registry
203  */
204  if (sc->worker_thread_instantiate) {
205  CONF_SECTION *cs;
206  char section_name[32];
207 
208  snprintf(section_name, sizeof(section_name), "%u", sw->id);
209 
210  cs = cf_section_find(sc->cs, "worker", section_name);
211  if (!cs) cs = cf_section_find(sc->cs, "worker", NULL);
212 
213  if (sc->worker_thread_instantiate(sw->ctx, sw->el, cs) < 0) {
214  PERROR("%s - Failed calling thread instantiate", worker_name);
215  goto fail;
216  }
217  }
218 
219  sw->status = FR_CHILD_RUNNING;
220 
221  /*
222  * Add this worker to all network threads.
223  */
224  for (sn = fr_dlist_head(&sc->networks);
225  sn != NULL;
226  sn = fr_dlist_next(&sc->networks, sn)) {
227  (void) fr_network_worker_add(sn->nr, sw->worker);
228  }
229 
230  DEBUG3("%s - Started", worker_name);
231 
232  /*
233  * Tell the originator that the thread has started.
234  */
235  sem_post(&sc->worker_sem);
236 
237  /*
238  * Do all of the work.
239  *
240  * @todo check for child processes.
241  */
242  fr_worker(sw->worker);
243 
244  status = FR_CHILD_EXITED;
245 
246 fail:
247  sw->status = status;
248 
249  if (sw->worker) {
250  fr_worker_destroy(sw->worker);
251  sw->worker = NULL;
252  }
253 
254  INFO("%s - Exiting", worker_name);
255 
256  if (sc->worker_thread_detach) sc->worker_thread_detach(NULL); /* Fixme once we figure out what uctx should be */
257 
258  /*
259  * Not looping at this point, but may catch timer/fd
260  * insertions being done after the thread should have
261  * exited.
262  */
263  if (sw->el) fr_event_loop_exit(sw->el, 1);
264 
265  /*
266  * Tell the scheduler we're done.
267  */
268  sem_post(&sc->worker_sem);
269 
270  return NULL;
271 }

283 /** Initialize and run the network thread.
284  *
285  * @param[in] arg the fr_schedule_network_t
286  * @return NULL
287  */
288 static void *fr_schedule_network_thread(void *arg)
289 {
290  TALLOC_CTX *ctx;
291  fr_schedule_network_t *sn = talloc_get_type_abort(arg, fr_schedule_network_t);
292  fr_schedule_t *sc = sn->sc;
293  fr_schedule_child_status_t status = FR_CHILD_FAIL;
294  fr_event_list_t *el;
295  char network_name[32];
296 
297  snprintf(network_name, sizeof(network_name), "Network %d", sn->id);
298 
299  INFO("%s - Starting", network_name);
300 
301  sn->ctx = ctx = talloc_init("%s", network_name);
302  if (!ctx) {
303  ERROR("%s - Failed allocating memory", network_name);
304  goto fail;
305  }
306 
307  el = fr_event_list_alloc(ctx, NULL, NULL);
308  if (!el) {
309  PERROR("%s - Failed creating event list", network_name);
310  goto fail;
311  }
312 
313  sn->nr = fr_network_create(ctx, el, network_name, sc->log, sc->lvl, &sc->config->network);
314  if (!sn->nr) {
315  PERROR("%s - Failed creating network", network_name);
316  goto fail;
317  }
318 
319  sn->status = FR_CHILD_RUNNING;
320 
321  /*
322  * Tell the originator that the thread has started.
323  */
324  sem_post(&sc->network_sem);
325 
326  DEBUG3("%s - Started", network_name);
327 
328  /*
329  * Print out statistics for this network IO handler.
330  */
331  if (sc->config->stats_interval) (void) fr_event_timer_in(sn, el, &sn->ev, sn->sc->config->stats_interval, stats_timer, sn);
332 
333  /*
334  * Call the main event processing loop of the network
335  * thread Will not return until the worker is about
336  * to exit.
337  */
338  fr_network(sn->nr);
339 
340  status = FR_CHILD_EXITED;
341 
342 fail:
343  sn->status = status;
344 
345  INFO("%s - Exiting", network_name);
346 
347  /*
348  * Tell the scheduler we're done.
349  */
350  sem_post(&sc->network_sem);
351 
352  return NULL;
353 }

355 /** Creates a new thread using our standard set of options
356  *
357  * New threads are:
358  * - Joinable, i.e. you can call pthread_join on them to confirm they've exited
359  * - Immune to catchable signals.
360  *
361  * @param[out] thread handled that was created by pthread_create.
362  * @param[in] func entry point for the thread.
363  * @param[in] arg Argument to pass to func.
364  * @return
365  * - 0 on success.
366  * - -1 on failure.
367  */
368 int fr_schedule_pthread_create(pthread_t *thread, void *(*func)(void *), void *arg)
369 {
370  pthread_attr_t attr;
371  int ret;
372 
373  /*
374  * Set the thread to wait around after it's exited
375  * so it can be joined. This is more of a useful
376  * mechanism for the parent to determine if all
377  * the threads have exited so it can continue with
378  * a graceful shutdown.
379  */
380  pthread_attr_init(&attr);
381  pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_JOINABLE);
382 
383  ret = pthread_create(thread, &attr, func, arg);
384  if (ret != 0) {
385  fr_strerror_printf("Failed creating thread: %s", fr_syserror(ret));
386  return -1;
387  }
388 
389  return 0;
390 }

392 /** Create a scheduler and spawn the child threads.
393  *
394  * @param[in] ctx talloc context.
395  * @param[in] el event list, only for single-threaded mode.
396  * @param[in] logger destination for all logging messages.
397  * @param[in] lvl log level.
398  * @param[in] worker_thread_instantiate callback for new worker threads.
399  * @param[in] worker_thread_detach callback to destroy resources
400  * allocated by worker_thread_instantiate.
401  * @param[in] config configuration for the scheduler
402  * @return
403  * - NULL on error
404  * - fr_schedule_t new scheduler
405  */
406 fr_schedule_t *fr_schedule_create(TALLOC_CTX *ctx, fr_event_list_t *el,
407  fr_log_t *logger, fr_log_lvl_t lvl,
408  fr_schedule_thread_instantiate_t worker_thread_instantiate,
409  fr_schedule_thread_detach_t worker_thread_detach,
410  fr_schedule_config_t *config)
411 {
412  unsigned int i;
413  fr_schedule_worker_t *sw, *next_sw;
414  fr_schedule_network_t *sn, *next_sn;
415  fr_schedule_t *sc;
416 
417  sc = talloc_zero(ctx, fr_schedule_t);
418  if (!sc) {
419  fr_strerror_printf("Failed allocating memory");
420  return NULL;
421  }
422 
423  /*
424  * Glue workers into the trigger code.
425  */
426  trigger_worker_request_add = fr_worker_request_add;
427 
428  sc->config = config;
429  sc->el = el;
430  sc->log = logger;
431  sc->lvl = lvl;
432 
433  sc->worker_thread_instantiate = worker_thread_instantiate;
434  sc->worker_thread_detach = worker_thread_detach;
435  sc->running = true;
436 
437  /*
438  * If we're single-threaded, create network / worker, and insert them into the event loop.
439  */
440  if (el) {
441  sc->single_network = fr_network_create(sc, el, "Network", sc->log, sc->lvl, &sc->config->network);
442  if (!sc->single_network) {
443  PERROR("Failed creating network");
444  pre_instantiate_st_fail:
445  talloc_free(sc);
446  return NULL;
447  }
448 
449  sc->single_worker = fr_worker_create(sc, el, "Worker", sc->log, sc->lvl, &sc->config->worker);
450  if (!sc->single_worker) {
451  PERROR("Failed creating worker");
452  fr_network_destroy(sc->single_network);
453  goto pre_instantiate_st_fail;
454  }
455 
456  /*
457  * Parent thread-specific data from the single_worker
458  */
459  if (sc->worker_thread_instantiate) {
460  CONF_SECTION *subcs;
461 
462  subcs = cf_section_find(sc->cs, "worker", "0");
463  if (!subcs) subcs = cf_section_find(sc->cs, "worker", NULL);
464 
465  if (sc->worker_thread_instantiate(sc->single_worker, el, subcs) < 0) {
466  PERROR("Failed calling thread instantiate");
467  destroy_both:
468  fr_network_destroy(sc->single_network);
469  fr_worker_destroy(sc->single_worker);
470  goto pre_instantiate_st_fail;
471  }
472  }
473 
474  if (fr_command_register_hook(NULL, "0", sc->single_worker, cmd_worker_table) < 0) {
475  PERROR("Failed adding worker commands");
476  st_fail:
477  if (sc->worker_thread_detach) sc->worker_thread_detach(NULL);
478  goto destroy_both;
479  }
480 
481  if (fr_command_register_hook(NULL, "0", sc->single_network, cmd_network_table) < 0) {
482  PERROR("Failed adding network commands");
483  goto st_fail;
484  }
485 
486  (void) fr_network_worker_add(sc->single_network, sc->single_worker);
487  DEBUG("Scheduler created in single-threaded mode");
488 
489  if (fr_event_pre_insert(el, fr_worker_pre_event, sc->single_worker) < 0) {
490  fr_strerror_printf("Failed adding pre-check to event list");
491  goto st_fail;
492  }
493 
494  /*
495  * Add the event which processes request_t packets.
496  */
497  if (fr_event_post_insert(el, fr_worker_post_event, sc->single_worker) < 0) {
498  fr_strerror_printf("Failed inserting post-processing event");
499  goto st_fail;
500  }
501 
502  return sc;
503  }
504 
505  /*
506  * Parse any scheduler-specific configuration.
507  */
508  if (!config) {
509  MEM(sc->config = talloc_zero(sc, fr_schedule_config_t));
510  sc->config->max_networks = 1;
511  sc->config->max_workers = 4;
512  } else {
513  sc->config = config;
514 
515  if (sc->config->max_networks < 1) sc->config->max_networks = 1;
516  if (sc->config->max_networks > 64) sc->config->max_networks = 64;
517  if (sc->config->max_workers < 1) sc->config->max_workers = 1;
518  if (sc->config->max_workers > 64) sc->config->max_workers = 64;
519  }
520 
521  /*
522  * Create the lists which hold the workers and networks.
523  */
524  fr_dlist_init(&sc->workers, fr_schedule_worker_t, entry);
525  fr_dlist_init(&sc->networks, fr_schedule_network_t, entry);
526 
527  memset(&sc->network_sem, 0, sizeof(sc->network_sem));
528  if (sem_init(&sc->network_sem, 0, SEMAPHORE_LOCKED) != 0) {
529  ERROR("Failed creating semaphore: %s", fr_syserror(errno));
530  talloc_free(sc);
531  return NULL;
532  }
533 
534  memset(&sc->worker_sem, 0, sizeof(sc->worker_sem));
535  if (sem_init(&sc->worker_sem, 0, SEMAPHORE_LOCKED) != 0) {
536  ERROR("Failed creating semaphore: %s", fr_syserror(errno));
537  talloc_free(sc);
538  return NULL;
539  }
540 
541  /*
542  * Create the network threads first.
543  */
544  for (i = 0; i < sc->config->max_networks; i++) {
545  DEBUG3("Creating %u/%u networks", i, sc->config->max_networks);
546 
547  /*
548  * Create a worker "glue" structure
549  */
550  sn = talloc_zero(sc, fr_schedule_network_t);
551  if (!sn) {
552  ERROR("Network %u - Failed allocating memory", i);
553  break;
554  }
555 
556  sn->id = i;
557  sn->sc = sc;
558  sn->status = FR_CHILD_INITIALIZING;
559  fr_dlist_insert_head(&sc->networks, sn);
560 
561  if (fr_schedule_pthread_create(&sn->pthread_id, fr_schedule_network_thread, sn) < 0) {
562  PERROR("Failed creating network %u", i);
563  break;
564  }
565  }
566 
567  /*
568  * Wait for all of the networks to signal us that either
569  * they've started, OR there's been a problem and they
570  * can't start.
571  */
572  for (i = 0; i < (unsigned int)fr_dlist_num_elements(&sc->networks); i++) {
573  DEBUG3("Waiting for semaphore from network %u/%u",
574  i, (unsigned int)fr_dlist_num_elements(&sc->networks));
575  SEM_WAIT_INTR(&sc->network_sem);
576  }
577 
578  /*
579  * See if all of the networks have started.
580  */
581  for (sn = fr_dlist_head(&sc->networks);
582  sn != NULL;
583  sn = next_sn) {
584  next_sn = fr_dlist_next(&sc->networks, sn);
585 
586  if (sn->status != FR_CHILD_RUNNING) {
587  fr_dlist_remove(&sc->networks, sn);
588  continue;
589  }
590  }
591 
592  /*
593  * Failed to start some workers, refuse to do anything!
594  */
595  if ((unsigned int)fr_dlist_num_elements(&sc->networks) < sc->config->max_networks) {
596  fr_schedule_destroy(&sc);
597  return NULL;
598  }
599 
600  /*
601  * Create all of the workers.
602  */
603  for (i = 0; i < sc->config->max_workers; i++) {
604  DEBUG3("Creating %u/%u workers", i, sc->config->max_workers);
605 
606  /*
607  * Create a worker "glue" structure
608  */
609  sw = talloc_zero(sc, fr_schedule_worker_t);
610  if (!sw) {
611  ERROR("Worker %u - Failed allocating memory", i);
612  break;
613  }
614 
615  sw->id = i;
616  sw->sc = sc;
617  sw->status = FR_CHILD_INITIALIZING;
618  fr_dlist_insert_head(&sc->workers, sw);
619 
620  if (fr_schedule_pthread_create(&sw->pthread_id, fr_schedule_worker_thread, sw) < 0) {
621  PERROR("Failed creating worker %u", i);
622  break;
623  }
624  }
625 
626  /*
627  * Wait for all of the workers to signal us that either
628  * they've started, OR there's been a problem and they
629  * can't start.
630  */
631  for (i = 0; i < (unsigned int)fr_dlist_num_elements(&sc->workers); i++) {
632  DEBUG3("Waiting for semaphore from worker %u/%u",
633  i, (unsigned int)fr_dlist_num_elements(&sc->workers));
634  SEM_WAIT_INTR(&sc->worker_sem);
635  }
636 
637  /*
638  * See if all of the workers have started.
639  */
640  for (sw = fr_dlist_head(&sc->workers);
641  sw != NULL;
642  sw = next_sw) {
643 
644  next_sw = fr_dlist_next(&sc->workers, sw);
645 
646  if (sw->status != FR_CHILD_RUNNING) {
647  fr_dlist_remove(&sc->workers, sw);
648  continue;
649  }
650  }
651 
652  /*
653  * Failed to start some workers, refuse to do anything!
654  */
655  if ((unsigned int)fr_dlist_num_elements(&sc->workers) < sc->config->max_workers) {
656  fr_schedule_destroy(&sc);
657  return NULL;
658  }
659 
660  for (sw = fr_dlist_head(&sc->workers), i = 0;
661  sw != NULL;
662  sw = next_sw, i++) {
663  char buffer[32];
664 
665  next_sw = fr_dlist_next(&sc->workers, sw);
666 
667  snprintf(buffer, sizeof(buffer), "%d", i);
668  if (fr_command_register_hook(NULL, buffer, sw->worker, cmd_worker_table) < 0) {
669  PERROR("Failed adding worker commands");
670  goto st_fail;
671  }
672  }
673 
674  for (sn = fr_dlist_head(&sc->networks), i = 0;
675  sn != NULL;
676  sn = next_sn, i++) {
677  char buffer[32];
678 
679  next_sn = fr_dlist_next(&sc->networks, sn);
680 
681  snprintf(buffer, sizeof(buffer), "%d", i);
682  if (fr_command_register_hook(NULL, buffer, sn->nr, cmd_network_table) < 0) {
683  PERROR("Failed adding network commands");
684  goto st_fail;
685  }
686  }

689  sc->config->max_networks, (unsigned int)fr_dlist_num_elements(&sc->workers));

694 /** Destroy a scheduler, and tell its child threads to exit.
695  *
696  * @param[in] sc_to_free the scheduler
697  * @return
698  * - <0 on error
699  * - 0 on success
700  */
701 int fr_schedule_destroy(fr_schedule_t **sc_to_free)
702 {
703  fr_schedule_t *sc = *sc_to_free;
704  unsigned int i;
705  fr_schedule_worker_t *sw;
706  fr_schedule_network_t *sn;
707 
708  if (!sc) return 0;
709 
710  sc->running = false;
711 
712  /*
713  * Single threaded mode: kill the only network / worker we have.
714  */
715  if (sc->el) {
716  /*
717  * Destroy the network side first. It tells the
718  * workers to close.
719  */
720  fr_network_destroy(sc->single_network);
721  fr_worker_destroy(sc->single_worker);
722  goto done;
723  }
724 
725  if (!fr_cond_assert(fr_dlist_num_elements(&sc->networks) > 0)) return -1;
726  if (!fr_cond_assert(fr_dlist_num_elements(&sc->workers) > 0)) return -1;
727 
728  /*
729  * Signal each network thread to exit.
730  */
731  for (sn = fr_dlist_head(&sc->networks);
732  sn != NULL;
733  sn = fr_dlist_next(&sc->networks, sn)) {
734  fr_network_exit(sn->nr);
735  }
736 
737  /*
738  * If the network threads are running, tell them to exit,
739  * and wait for them to do so. Each network thread tells
740  * all of its worker threads that it's exiting. It then
741  * closes the channels. When the workers see that there
742  * are no input channels, they exit, too.
743  */
744  for (i = 0; i < (unsigned int)fr_dlist_num_elements(&sc->networks); i++) {
745  DEBUG2("Scheduler - Waiting for semaphore indicating network exit %u/%u", i,
746  (unsigned int)fr_dlist_num_elements(&sc->networks));
747  SEM_WAIT_INTR(&sc->network_sem);
748  }
749  DEBUG2("Scheduler - All networks indicated exit complete");
750 
751  while ((sn = fr_dlist_head(&sc->networks)) != NULL) {
752  fr_dlist_remove(&sc->networks, sn);
753 
754  /*
755  * Ensure that the thread has exited before
756  * cleaning up the context.
757  *
758  * This also ensures that the child threads have
759  * exited before the main thread cleans up the
760  * module instances.
761  */
762  if (pthread_join(sn->pthread_id, NULL) != 0) {
763  ERROR("Failed joining network %i: %s", sn->id, fr_syserror(errno));
764  } else {
765  DEBUG2("Network %i joined (cleaned up)", sn->id);
766  }
767  talloc_free(sn->ctx);
768  }
769 
770  /*
771  * Wait for all worker threads to finish. THEN clean up
772  * modules. Otherwise, the modules will be removed from
773  * underneath the workers!
774  */
775  for (i = 0; i < (unsigned int)fr_dlist_num_elements(&sc->workers); i++) {
776  DEBUG2("Scheduler - Waiting for semaphore indicating worker exit %u/%u", i + 1,
777  (unsigned int)fr_dlist_num_elements(&sc->workers));
778  SEM_WAIT_INTR(&sc->worker_sem);
779  }
780  DEBUG2("Scheduler - All workers indicated exit complete");
781 
782  /*
783  * Clean up the exited workers.
784  */
785  while ((sw = fr_dlist_head(&sc->workers)) != NULL) {
786  fr_dlist_remove(&sc->workers, sw);
787 
788  /*
789  * Ensure that the thread has exited before
790  * cleaning up the context.
791  *
792  * This also ensures that the child threads have
793  * exited before the main thread cleans up the
794  * module instances.
795  */
796  if (pthread_join(sw->pthread_id, NULL) != 0) {
797  ERROR("Failed joining worker %i: %s", sw->id, fr_syserror(errno));
798  } else {
799  DEBUG2("Worker %i joined (cleaned up)", sw->id);
800  }
801  talloc_free(sw->ctx);
802  }
803 
804  sem_destroy(&sc->network_sem);
805  sem_destroy(&sc->worker_sem);
806 done:
807  /*
808  * Now that all of the workers are done, we can return to
809  * the caller, and have it dlclose() the modules.
810  */
811  talloc_free(sc);
812  *sc_to_free = NULL;
813 
814  return 0;
815 }
