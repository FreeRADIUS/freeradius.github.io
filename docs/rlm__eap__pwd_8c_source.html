87 static int send_pwd_request(request_t *request, pwd_session_t *session, eap_round_t *eap_round)
88 {
89  size_t len;
90  uint16_t totlen;
91  pwd_hdr *hdr;
92 
93  len = (session->out_len - session->out_pos) + sizeof(pwd_hdr);
94  fr_assert(len > 0);
95  eap_round->request->code = FR_EAP_CODE_REQUEST;
96  eap_round->request->type.num = FR_EAP_METHOD_PWD;
97  eap_round->request->type.length = (len > session->mtu) ? session->mtu : len;
98  eap_round->request->type.data = talloc_zero_array(eap_round->request, uint8_t, eap_round->request->type.length);
99  hdr = (pwd_hdr *)eap_round->request->type.data;
100 
101  switch (session->state) {
102  case PWD_STATE_ID_REQ:
103  EAP_PWD_SET_EXCHANGE(hdr, EAP_PWD_EXCH_ID);
104  break;
105 
106  case PWD_STATE_COMMIT:
107  EAP_PWD_SET_EXCHANGE(hdr, EAP_PWD_EXCH_COMMIT);
108  break;
109 
110  case PWD_STATE_CONFIRM:
111  EAP_PWD_SET_EXCHANGE(hdr, EAP_PWD_EXCH_CONFIRM);
112  break;
113 
114  default:
115  REDEBUG("PWD state is invalid. Can't send request");
116  return -1;
117  }
118 
119  /*
120  * are we fragmenting?
121  */
122  if (((session->out_len - session->out_pos) + sizeof(pwd_hdr)) > session->mtu) {
123  EAP_PWD_SET_MORE_BIT(hdr);
124  if (session->out_pos == 0) {
125 
126  /*
127  * the first fragment, add the total length
128  */
129  EAP_PWD_SET_LENGTH_BIT(hdr);
130  totlen = ntohs(session->out_len);
131  memcpy(hdr->data, (char *)&totlen, sizeof(totlen));
132  memcpy(hdr->data + sizeof(uint16_t),
133  session->out,
134  session->mtu - sizeof(pwd_hdr) - sizeof(uint16_t));
135  session->out_pos += (session->mtu - sizeof(pwd_hdr) - sizeof(uint16_t));
136  } else {
137  /*
138  * an intermediate fragment
139  */
140  memcpy(hdr->data, session->out + session->out_pos, (session->mtu - sizeof(pwd_hdr)));
141  session->out_pos += (session->mtu - sizeof(pwd_hdr));
142  }
143  } else {
144  /*
145  * either it's not a fragment or it's the last fragment.
146  * The out buffer isn't needed anymore though so get rid of it.
147  */
148  memcpy(hdr->data, session->out + session->out_pos,
149  (session->out_len - session->out_pos));
150  talloc_free(session->out);
151  session->out = NULL;
152  session->out_pos = session->out_len = 0;
153  }
154  return 0;
155 }

157 static unlang_action_t mod_process(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
158 {
159  rlm_eap_pwd_t *inst = talloc_get_type_abort(mctx->instance, rlm_eap_pwd_t);
160  eap_session_t *eap_session = eap_session_get(request->parent);
161 
162  pwd_session_t *session;
163 
164  pwd_hdr *hdr;
165  pwd_id_packet_t *packet;
166  eap_packet_t *response;
167 
168  eap_round_t *eap_round;
169  size_t in_len;
170  rlm_rcode_t rcode = RLM_MODULE_OK;
171  uint16_t offset;
172  uint8_t exch, *in, *ptr, msk[MSK_EMSK_LEN], emsk[MSK_EMSK_LEN];
173  uint8_t peer_confirm[SHA256_DIGEST_LENGTH];
174 
175  if (((eap_round = eap_session->this_round) == NULL) || !inst) return 0;
176 
177  session = talloc_get_type_abort(eap_session->opaque, pwd_session_t);
178  response = eap_session->this_round->response;
179  hdr = (pwd_hdr *)response->type.data;
180 
181  /*
182  * The header must be at least one byte.
183  */
184  if (!hdr || (response->type.length < sizeof(pwd_hdr))) {
185  REDEBUG("Packet with insufficient data");
186  RETURN_MODULE_INVALID;
187  }
188 
189  in = hdr->data;
190  in_len = response->type.length - sizeof(pwd_hdr);
191 
192  /*
193  * See if we're fragmenting, if so continue until we're done
194  */
195  if (session->out_pos) {
196  if (in_len) REDEBUG("PWD got something more than an ACK for a fragment");
197  if (send_pwd_request(request, session, eap_round) < 0) RETURN_MODULE_FAIL;
198 
199  RETURN_MODULE_OK;
200  }
201 
202  /*
203  * The first fragment will have a total length, make a
204  * buffer to hold all the fragments
205  */
206  if (EAP_PWD_GET_LENGTH_BIT(hdr)) {
207  if (session->in) {
208  REDEBUG("PWD already alloced buffer for fragments");
209  RETURN_MODULE_FAIL;
210  }
211 
212  if (in_len < 2) {
213  REDEBUG("Invalid packet: length bit set, but no length field");
214  RETURN_MODULE_INVALID;
215  }
216 
217  session->in_len = ntohs(in[0] * 256 | in[1]);
218  if (!session->in_len) {
219  DEBUG("EAP-PWD malformed packet (input length)");
220  RETURN_MODULE_FAIL;
221  }
222 
223  MEM(session->in = talloc_zero_array(session, uint8_t, session->in_len));
224 
225  session->in_pos = 0;
226  in += sizeof(uint16_t);
227  in_len -= sizeof(uint16_t);
228  }
229 
230  /*
231  * All fragments, including the 1st will have the M(ore) bit set,
232  * buffer those fragments!
233  */
234  if (EAP_PWD_GET_MORE_BIT(hdr)) {
235  if (!session->in) {
236  RDEBUG2("Unexpected fragment.");
237  return 0;
238  }
239 
240  if ((session->in_pos + in_len) > session->in_len) {
241  REDEBUG("Fragment overflows packet");
242  RETURN_MODULE_INVALID;
243  }
244 
245  memcpy(session->in + session->in_pos, in, in_len);
246  session->in_pos += in_len;
247 
248  /*
249  * send back an ACK for this fragment
250  */
251  exch = EAP_PWD_GET_EXCHANGE(hdr);
252  eap_round->request->code = FR_EAP_CODE_REQUEST;
253  eap_round->request->type.num = FR_EAP_METHOD_PWD;
254  eap_round->request->type.length = sizeof(pwd_hdr);
255 
256  MEM(eap_round->request->type.data = talloc_array(eap_round->request, uint8_t, sizeof(pwd_hdr)));
257 
258  hdr = (pwd_hdr *)eap_round->request->type.data;
259  EAP_PWD_SET_EXCHANGE(hdr, exch);
260  RETURN_MODULE_OK;
261  }
262 
263 
264  if (session->in) {
265  /*
266  * The last fragment...
267  */
268  if ((session->in_pos + in_len) > session->in_len) {
269  REDEBUG("PWD will overflow a fragment buffer");
270  RETURN_MODULE_INVALID;
271  }
272  memcpy(session->in + session->in_pos, in, in_len);
273  in = session->in;
274  in_len = session->in_len;
275  }
276 
277  switch (session->state) {
278  case PWD_STATE_ID_REQ:
279  {
280  fr_pair_t *known_good;
281  fr_dict_attr_t const *allowed_passwords[] = { attr_cleartext_password };
282  int ret;
283  bool ephemeral;
284  BIGNUM *x = NULL, *y = NULL;
285 
286  if (EAP_PWD_GET_EXCHANGE(hdr) != EAP_PWD_EXCH_ID) {
287  REDEBUG("PWD exchange is incorrect, Not ID");
288  RETURN_MODULE_INVALID;
289  }
290 
291  packet = (pwd_id_packet_t *) in;
292  if (in_len < sizeof(*packet)) {
293  REDEBUG("Packet is too small (%zd < %zd).", in_len, sizeof(*packet));
294  RETURN_MODULE_INVALID;
295  }
296 
297  if ((packet->prf != EAP_PWD_DEF_PRF) ||
298  (packet->random_function != EAP_PWD_DEF_RAND_FUN) ||
299  (packet->prep != EAP_PWD_PREP_NONE) ||
300  (CRYPTO_memcmp(packet->token, &session->token, 4)) ||
301  (packet->group_num != ntohs(session->group_num))) {
302  REDEBUG("PWD ID response is malformed");
303  RETURN_MODULE_INVALID;
304  }
305 
306  /*
307  * We've agreed on the ciphersuite, record it...
308  */
309  ptr = (uint8_t *)&session->ciphersuite;
310  memcpy(ptr, (char *)&packet->group_num, sizeof(uint16_t));
311  ptr += sizeof(uint16_t);
312  *ptr = EAP_PWD_DEF_RAND_FUN;
313  ptr += sizeof(uint8_t);
314  *ptr = EAP_PWD_DEF_PRF;
315 
316  session->peer_id_len = in_len - sizeof(pwd_id_packet_t);
317  if (session->peer_id_len >= sizeof(session->peer_id)) {
318  REDEBUG("PWD ID response is malformed");
319  RETURN_MODULE_INVALID;
320  }
321 
322  memcpy(session->peer_id, packet->identity, session->peer_id_len);
323  session->peer_id[session->peer_id_len] = '\0';
324 
325  known_good = password_find(&ephemeral, request, request->parent,
326  allowed_passwords, NUM_ELEMENTS(allowed_passwords), false);
327  if (!known_good) {
328  REDEBUG("No \"known good\" password found for user");
329  RETURN_MODULE_FAIL;
330  }
331 
332  ret = compute_password_element(request, session, session->group_num,
333  known_good->vp_strvalue, known_good->vp_length,
334  inst->server_id, strlen(inst->server_id),
335  session->peer_id, strlen(session->peer_id),
336  &session->token, inst->bnctx);
337  if (ephemeral) talloc_list_free(&known_good);
338  if (ret < 0) {
339  REDEBUG("Failed to obtain password element");
340  RETURN_MODULE_FAIL;
341  }
342 
343  /*
344  * Compute our scalar and element
345  */
346  if (compute_scalar_element(request, session, inst->bnctx)) {
347  REDEBUG("Failed to compute server's scalar and element");
348  RETURN_MODULE_FAIL;
349  }
350 
351  MEM(x = BN_new());
352  MEM(y = BN_new());
353 
354  /*
355  * Element is a point, get both coordinates: x and y
356  */
357  if (!EC_POINT_get_affine_coordinates_GFp(session->group, session->my_element, x, y, inst->bnctx)) {
358  REDEBUG("Server point assignment failed");
359  BN_clear_free(x);
360  BN_clear_free(y);
361  RETURN_MODULE_FAIL;
362  }
363 
364  /*
365  * Construct request
366  */
367  session->out_len = BN_num_bytes(session->order) + (2 * BN_num_bytes(session->prime));
368  MEM(session->out = talloc_zero_array(session, uint8_t, session->out_len));
369 
370  ptr = session->out;
371  offset = BN_num_bytes(session->prime) - BN_num_bytes(x);
372  BN_bn2bin(x, ptr + offset);
373  BN_clear_free(x);
374 
375  ptr += BN_num_bytes(session->prime);
376  offset = BN_num_bytes(session->prime) - BN_num_bytes(y);
377  BN_bn2bin(y, ptr + offset);
378  BN_clear_free(y);
379 
380  ptr += BN_num_bytes(session->prime);
381  offset = BN_num_bytes(session->order) - BN_num_bytes(session->my_scalar);
382  BN_bn2bin(session->my_scalar, ptr + offset);
383 
384  session->state = PWD_STATE_COMMIT;
385  rcode = send_pwd_request(request, session, eap_round) < 0 ? RLM_MODULE_FAIL : RLM_MODULE_OK;
386  }
387  break;
388 
389  case PWD_STATE_COMMIT:
390  if (EAP_PWD_GET_EXCHANGE(hdr) != EAP_PWD_EXCH_COMMIT) {
391  REDEBUG("PWD exchange is incorrect, not commit!");
392  RETURN_MODULE_INVALID;
393  }
394 
395  /*
396  * Process the peer's commit and generate the shared key, k
397  */
398  if (process_peer_commit(request, session, in, in_len, inst->bnctx)) {
399  REDEBUG("Failed processing peer's commit");
400  RETURN_MODULE_FAIL;
401  }
402 
403  /*
404  * Compute our confirm blob
405  */
406  if (compute_server_confirm(request, session, session->my_confirm, inst->bnctx)) {
407  REDEBUG("Failed computing confirm");
408  RETURN_MODULE_FAIL;
409  }
410 
411  /*
412  * Construct a response...which is just our confirm blob
413  */
414  session->out_len = SHA256_DIGEST_LENGTH;
415  MEM(session->out = talloc_array(session, uint8_t, session->out_len));
416 
417  memset(session->out, 0, session->out_len);
418  memcpy(session->out, session->my_confirm, SHA256_DIGEST_LENGTH);
419 
420  session->state = PWD_STATE_CONFIRM;
421  rcode = send_pwd_request(request, session, eap_round) < 0 ? RLM_MODULE_FAIL : RLM_MODULE_OK;
422  break;
423 
424  case PWD_STATE_CONFIRM:
425  if (in_len < SHA256_DIGEST_LENGTH) {
426  REDEBUG("Peer confirm is too short (%zd < %d)", in_len, SHA256_DIGEST_LENGTH);
427  RETURN_MODULE_INVALID;
428  }
429 
430  if (EAP_PWD_GET_EXCHANGE(hdr) != EAP_PWD_EXCH_CONFIRM) {
431  REDEBUG("PWD exchange is incorrect, not commit");
432  RETURN_MODULE_INVALID;
433  }
434  if (compute_peer_confirm(request, session, peer_confirm, inst->bnctx)) {
435  REDEBUG("Cannot compute peer's confirm");
436  RETURN_MODULE_FAIL;
437  }
438  if (CRYPTO_memcmp(peer_confirm, in, SHA256_DIGEST_LENGTH)) {
439  REDEBUG("PWD exchange failed, peer confirm is incorrect");
440  RETURN_MODULE_FAIL;
441  }
442  if (compute_keys(request, session, peer_confirm, msk, emsk)) {
443  REDEBUG("Failed generating (E)MSK");
444  RETURN_MODULE_FAIL;
445  }
446  eap_round->request->code = FR_EAP_CODE_SUCCESS;
447 
448  /*
449  * Return the MSK (in halves).
450  */
451  eap_add_reply(request->parent, attr_ms_mppe_recv_key, msk, MPPE_KEY_LEN);
452  eap_add_reply(request->parent, attr_ms_mppe_send_key, msk + MPPE_KEY_LEN, MPPE_KEY_LEN);
453 
454  rcode = RLM_MODULE_OK;
455  break;
456 
457  default:
458  REDEBUG("Unknown PWD state");
459  RETURN_MODULE_FAIL;
460  }
461 
462  /*
463  * We processed the buffered fragments, get rid of them.
464  */
465  if (session->in) {
466  talloc_free(session->in);
467  session->in = NULL;
468  }
469 
470  RETURN_MODULE_RCODE(rcode);
471 }

489 static unlang_action_t mod_session_init(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
490 {
491  rlm_eap_pwd_t *inst = talloc_get_type_abort(mctx->instance, rlm_eap_pwd_t);
492  eap_session_t *eap_session = eap_session_get(request->parent);
493  pwd_session_t *session;
494  fr_pair_t *vp;
495  pwd_id_packet_t *packet;
496 
497  MEM(session = talloc_zero(eap_session, pwd_session_t));
498  talloc_set_destructor(session, _free_pwd_session);
499  /*
500  * set things up so they can be free'd reliably
501  */
502  session->group_num = inst->group;
503 
504  /*
505  * The admin can dynamically change the MTU.
506  */
507  session->mtu = inst->fragment_size;
508  vp = fr_pair_find_by_da(&request->request_pairs, attr_framed_mtu);
509 
510  /*
511  * session->mtu is *our* MTU. We need to subtract off the EAP
512  * overhead.
513  *
514  * 9 = 4 (EAPOL header) + 4 (EAP header) + 1 (EAP type)
515  *
516  * The fragmentation code deals with the included length
517  * so we don't need to subtract that here.
518  */
519  if (vp && (vp->vp_uint32 > 100) && (vp->vp_uint32 < session->mtu)) session->mtu = vp->vp_uint32 - 9;
520 
521  session->state = PWD_STATE_ID_REQ;
522  session->out_pos = 0;
523  eap_session->opaque = session;
524 
525  /*
526  * construct an EAP-pwd-ID/Request
527  */
528  session->out_len = sizeof(pwd_id_packet_t) + strlen(inst->server_id);
529  MEM(session->out = talloc_zero_array(session, uint8_t, session->out_len));
530 
531  packet = (pwd_id_packet_t *)session->out;
532  packet->group_num = htons(session->group_num);
533  packet->random_function = EAP_PWD_DEF_RAND_FUN;
534  packet->prf = EAP_PWD_DEF_PRF;
535  session->token = fr_rand();
536  memcpy(packet->token, (char *)&session->token, 4);
537  packet->prep = EAP_PWD_PREP_NONE;
538  memcpy(packet->identity, inst->server_id, session->out_len - sizeof(pwd_id_packet_t) );
539 
540  if (send_pwd_request(request, session, eap_session->this_round) < 0) RETURN_MODULE_FAIL;
541 
542  eap_session->process = mod_process;
543 
544  RETURN_MODULE_HANDLED;
545 }
