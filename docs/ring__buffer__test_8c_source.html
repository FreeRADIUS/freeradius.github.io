72 static void alloc_blocks(fr_ring_buffer_t *rb, uint32_t *seed, UNUSED int *start, int *end)
73 {
74  int i;
75  uint32_t hash;
76 
77  for (i = 0; i < ALLOC_SIZE; i++) {
78  int index;
79  uint8_t *p;
80 
81  index = (*end + i) & (ARRAY_SIZE - 1);
82 
83  hash = fr_hash_update(seed_string, seed_string_len, *seed);
84  *seed = hash;
85 
86  hash &= 0x3ff;
87  hash += 16; /* can't have it zero... */
88 
89  array[index] = hash;
90  p = fr_ring_buffer_reserve(rb, 2048);
91 
92  if (!fr_cond_assert(p != NULL)) fr_exit_now(EXIT_FAILURE);
93 
94  data[index] = fr_ring_buffer_alloc(rb, hash);
95  if (!fr_cond_assert(data[index] == p)) fr_exit_now(EXIT_FAILURE);
96 
97  if (debug_lvl > 1) printf("%08x\t", hash);
98 
99  used += hash;
100  fr_assert(fr_ring_buffer_used(rb) == used);
101  }
102 
103  *end += ALLOC_SIZE;
104 }

106 static void free_blocks(fr_ring_buffer_t *rb, UNUSED uint32_t *seed, int *start, int *end)
107 {
108  int i;
109 
110  for (i = 0; i < ALLOC_SIZE; i++) {
111  int index;
112  int rcode;
113 
114  index = (*start + i) & (ARRAY_SIZE - 1);
115 
116  rcode = fr_ring_buffer_free(rb, array[index]);
117  if (!fr_cond_assert(rcode == 0)) fr_exit_now(EXIT_FAILURE);
118 
119  used -= array[index];
120  fr_assert(fr_ring_buffer_used(rb) == used);
121 
122  array[index] = 0;
123  data[index] = NULL;
124  }
125 
126  *start += ALLOC_SIZE;
127  if (*start > ARRAY_SIZE) {
128  *start -= ARRAY_SIZE;
129  *end -= ARRAY_SIZE;
130  }
131 }
