56 /** Auto cast from the input type to CONF_ITEM (which is the base type)

58  * Automatically casts:

61  * - #CONF_DATA

68  CONF_SECTION const * : cf_section_to_item((CONF_SECTION const *)_cf), \

70  CONF_PAIR * : cf_pair_to_item((CONF_PAIR const *)_cf), \

72  CONF_PAIR const * const : cf_pair_to_item((CONF_PAIR const * const)_cf), \
73  CONF_DATA * : cf_data_to_item((CONF_DATA const *)_cf), \
74  CONF_DATA const * : cf_data_to_item((CONF_DATA const *)_cf), \
75  CONF_DATA const * const : cf_data_to_item((CONF_DATA const * const)_cf), \
