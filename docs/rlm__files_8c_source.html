105 static int getusersfile(TALLOC_CTX *ctx, char const *filename, rbtree_t **ptree)
106 {
107  int rcode;
108  fr_pair_t *vp;
109  PAIR_LIST *users = NULL;
110  PAIR_LIST *entry, *next;
111  PAIR_LIST *user_list, *default_list, **default_tail;
112  rbtree_t *tree;
113 
114  if (!filename) {
115  *ptree = NULL;
116  return 0;
117  }
118 
119  rcode = pairlist_read(ctx, dict_radius, filename, &users, 1);
120  if (rcode < 0) {
121  return -1;
122  }
123 
124  /*
125  * Walk through the 'users' file list
126  */
127  entry = users;
128  while (entry) {
129  fr_cursor_t cursor;
130 
131  /*
132  * Look for improper use of '=' in the
133  * check items. They should be using
134  * '==' for on-the-wire RADIUS attributes,
135  * and probably ':=' for server
136  * configuration items.
137  */
138  for (vp = fr_cursor_init(&cursor, &entry->check);
139  vp;
140  vp = fr_cursor_next(&cursor)) {
141  /*
142  * Ignore attributes which are set
143  * properly.
144  */
145  if (vp->op != T_OP_EQ) {
146  continue;
147  }
148 
149  /*
150  * If it's a vendor attribute,
151  * or it's a wire protocol,
152  * ensure it has '=='.
153  */
154  if ((fr_dict_vendor_num_by_da(vp->da) != 0) ||
155  (vp->da->attr < 0x100)) {
156  WARN("[%s]:%d Changing '%s =' to '%s =='\n\tfor comparing RADIUS attribute in check item list for user %s",
157  filename, entry->lineno,
158  vp->da->name, vp->da->name,
159  entry->name);
160  vp->op = T_OP_CMP_EQ;
161  continue;
162  }
163  } /* end of loop over check items */
164 
165  /*
166  * Look for server configuration items
167  * in the reply list.
168  *
169  * It's a common enough mistake, that it's
170  * worth doing.
171  */
172  for (vp = fr_cursor_init(&cursor, &entry->reply);
173  vp;
174  vp = fr_cursor_next(&cursor)) {
175  /*
176  * If it's NOT a vendor attribute,
177  * and it's NOT a wire protocol
178  * and we ignore Fall-Through,
179  * then bitch about it, giving a
180  * good warning message.
181  */
182  if (fr_dict_attr_is_top_level(vp->da) && (vp->da->attr > 1000)) {
183  WARN("[%s]:%d Check item \"%s\"\n"
184  "\tfound in reply item list for user \"%s\".\n"
185  "\tThis attribute MUST go on the first line"
186  " with the other check items", filename, entry->lineno, vp->da->name,
187  entry->name);
188  }
189  }
190 
191  entry = entry->next;
192  }
193 
194  tree = rbtree_alloc(ctx, pairlist_cmp, NULL, RBTREE_FLAG_NONE);
195  if (!tree) {
196  pairlist_free(&users);
197  return -1;
198  }
199 
200  default_list = NULL;
201  default_tail = &default_list;
202 
203  /*
204  * We've read the entries in linearly, but putting them
205  * into an indexed data structure would be much faster.
206  * Let's go fix that now.
207  */
208  for (entry = users; entry != NULL; entry = next) {
209  /*
210  * Remove this entry from the input list.
211  */
212  next = entry->next;
213  entry->next = NULL;
214 
215  /*
216  * DEFAULT entries get their own list.
217  */
218  if (strcmp(entry->name, "DEFAULT") == 0) {
219  if (!default_list) {
220  default_list = entry;
221 
222  /*
223  * Insert the first DEFAULT into the tree.
224  */
225  if (!rbtree_insert(tree, entry)) {
226  error:
227  pairlist_free(&entry);
228  pairlist_free(&next);
229  talloc_free(tree);
230  return -1;
231  }
232 
233  } else {
234  /*
235  * Tack this entry onto the tail
236  * of the DEFAULT list.
237  */
238  *default_tail = entry;
239  }
240 
241  default_tail = &entry->next;
242  continue;
243  }
244 
245  /*
246  * Not DEFAULT, must be a normal user.
247  */
248  user_list = rbtree_finddata(tree, entry);
249  if (!user_list) {
250  /*
251  * Insert the first one.
252  */
253  if (!rbtree_insert(tree, entry)) goto error;
254  } else {
255  /*
256  * Find the tail of this list, and add it
257  * there.
258  */
259  while (user_list->next) user_list = user_list->next;
260 
261  user_list->next = entry;
262  }
263  }
264 
265  *ptree = tree;
266 
267  return 0;
268 }

292  * Common code called by everything below.
293  */
294 static unlang_action_t file_common(rlm_rcode_t *p_result, rlm_files_t const *inst,
295  request_t *request, char const *filename, rbtree_t *tree,
296  fr_radius_packet_t *packet, fr_radius_packet_t *reply)
297 {
298  char const *name;
299  fr_pair_t *check_tmp = NULL;
300  fr_pair_t *reply_tmp = NULL;
301  PAIR_LIST const *user_pl, *default_pl;
302  bool found = false;
303  PAIR_LIST my_pl;
304  char buffer[256];
305 
306 
307  if (tmpl_expand(&name, buffer, sizeof(buffer), request, inst->key, NULL, NULL) < 0) {
308  REDEBUG("Failed expanding key %s", inst->key->name);
309  RETURN_MODULE_FAIL;
310  }
311 
312  if (!tree) RETURN_MODULE_NOOP;
313 
314  my_pl.name = name;
315  user_pl = rbtree_finddata(tree, &my_pl);
316  my_pl.name = "DEFAULT";
317  default_pl = rbtree_finddata(tree, &my_pl);
318 
319  /*
320  * Find the entry for the user.
321  */
322  while (user_pl || default_pl) {
323  fr_cursor_t cursor;
324  fr_pair_t *vp;
325  PAIR_LIST const *pl;
326 
327  /*
328  * Figure out which entry to match on.
329  */
330 
331  if (!default_pl && user_pl) {
332  pl = user_pl;
333  user_pl = user_pl->next;
334 
335  } else if (!user_pl && default_pl) {
336  pl = default_pl;
337  default_pl = default_pl->next;
338 
339  } else if (user_pl->order < default_pl->order) {
340  pl = user_pl;
341  user_pl = user_pl->next;
342 
343  } else {
344  pl = default_pl;
345  default_pl = default_pl->next;
346  }
347 
348  MEM(fr_pair_list_copy(request, &check_tmp, &pl->check) >= 0);
349  for (vp = fr_cursor_init(&cursor, &check_tmp);
350  vp;
351  vp = fr_cursor_next(&cursor)) {
352  if (xlat_eval_pair(request, vp) < 0) {
353  RPWARN("Failed parsing expanded value for check item, skipping entry");
354  fr_pair_list_free(&check_tmp);
355  continue;
356  }
357  }
358 
359  if (paircmp(request, packet->vps, check_tmp) == 0) {
360  RDEBUG2("Found match \"%s\" one line %d of %s", pl->name, pl->lineno, filename);
361  found = true;
362 
363  /* ctx may be reply */
364  if (pl->reply) {
365  MEM(fr_pair_list_copy(reply, &reply_tmp, &pl->reply) >= 0);
366  radius_pairmove(request, &reply->vps, reply_tmp, true);
367  }
368  fr_pair_list_move(&request->control_pairs, &check_tmp);
369 
370  reply_tmp = NULL; /* radius_pairmove() frees input attributes */
371  fr_pair_list_free(&check_tmp);
372 
373  /*
374  * Fallthrough?
375  */
376  if (!fall_through(pl->reply)) break;
377  }
378 
379  /* Ensure temporary check list is clear before next match */
380  fr_pair_list_free(&check_tmp);
381  }
382 
383  /*
384  * Remove server internal parameters.
385  */
386  fr_pair_delete_by_da(&reply->vps, attr_fall_through);
387 
388  /*
389  * See if we succeeded.
390  */
391  if (!found)
392  RETURN_MODULE_NOOP; /* on to the next module */
393 
394  RETURN_MODULE_OK;
395 
396 }
