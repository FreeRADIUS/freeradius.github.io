51 /** Encode a packet
52  *
53  */
54 ssize_t fr_radius_packet_encode(fr_radius_packet_t *packet, fr_radius_packet_t const *original, char const *secret)
55 {
56  uint8_t const *original_data;
57  ssize_t slen;
58 
59  /*
60  * A 4K packet, aligned on 64-bits.
61  */
62  uint8_t data[MAX_PACKET_LEN];
63 
64 #ifndef NDEBUG
65  if (fr_debug_lvl >= L_DBG_LVL_4) fr_radius_packet_log_hex(&default_log, packet);
66 #endif
67 
68  if (original) {
69  original_data = original->data;
70  } else {
71  original_data = NULL;
72  }
73 
74  /*
75  * This has to be initialized for Access-Request packets
76  */
77  memcpy(data + 4, packet->vector, sizeof(packet->vector));
78 
79  slen = fr_radius_encode(data, sizeof(data), original_data, secret, talloc_array_length(secret) - 1,
80  packet->code, packet->id, packet->vps);
81  if (slen < 0) return slen;
82 
83  /*
84  * Fill in the rest of the fields, and copy the data over
85  * from the local stack to the newly allocated memory.
86  *
87  * Yes, all this 'memcpy' is slow, but it means
88  * that we only allocate the minimum amount of
89  * memory for a request.
90  */
91  packet->data_len = (size_t) slen;
92  packet->data = talloc_array(packet, uint8_t, packet->data_len);
93  if (!packet->data) {
94  fr_strerror_printf("Out of memory");
95  return -1;
96  }
97 
98  memcpy(packet->data, data, packet->data_len);
99 
100  return 0;
101 }

104 /** Calculate/check digest, and decode radius attributes
105  *
106  * @return
107  * - 0 on success
108  * - -1 on decoding error.
109  */
110 int fr_radius_packet_decode(fr_radius_packet_t *packet, fr_radius_packet_t *original,
111  uint32_t max_attributes, bool tunnel_password_zeros, char const *secret)
112 {
113  int packet_length;
114  uint32_t num_attributes;
115  uint8_t *ptr;
116  radius_packet_t *hdr;
117  fr_pair_t *head = NULL;
118  fr_cursor_t cursor, out;
119  fr_radius_ctx_t packet_ctx = {
120  .secret = secret,
121  .tunnel_password_zeros = tunnel_password_zeros
122  };
123 
124 #ifndef NDEBUG
125  if (fr_debug_lvl >= L_DBG_LVL_4) fr_radius_packet_log_hex(&default_log, packet);
126 #endif
127 
128  switch (packet->code) {
129  case FR_CODE_ACCESS_REQUEST:
130  case FR_CODE_STATUS_SERVER:
131  memcpy(packet_ctx.vector, packet->vector, sizeof(packet_ctx.vector));
132  break;
133 
134  case FR_CODE_ACCESS_ACCEPT:
135  case FR_CODE_ACCESS_REJECT:
136  case FR_CODE_ACCESS_CHALLENGE:
137  case FR_CODE_ACCOUNTING_RESPONSE:
138  case FR_CODE_COA_ACK:
139  case FR_CODE_COA_NAK:
140  case FR_CODE_DISCONNECT_ACK:
141  case FR_CODE_DISCONNECT_NAK:
142  /*
143  * radsniff doesn't always have a response
144  */
145  if (original) {
146  memcpy(packet_ctx.vector, original->vector, sizeof(packet_ctx.vector));
147  } else {
148  memset(packet->vector, 0, sizeof(packet->vector));
149  memset(packet_ctx.vector, 0, sizeof(packet_ctx.vector));
150  }
151  break;
152 
153  case FR_CODE_ACCOUNTING_REQUEST:
154  case FR_CODE_COA_REQUEST:
155  case FR_CODE_DISCONNECT_REQUEST:
156  memset(packet->vector, 0, sizeof(packet->vector));
157  memset(packet_ctx.vector, 0, sizeof(packet_ctx.vector));
158  break;
159 
160  default:
161  fr_strerror_printf("Cannot decode unknown packet code %d", packet->code);
162  return -1;
163  }
164 
165  packet_ctx.tmp_ctx = talloc(packet, uint8_t);
166 
167  /*
168  * Extract attribute-value pairs
169  */
170  hdr = (radius_packet_t *)packet->data;
171  ptr = hdr->data;
172  packet_length = packet->data_len - RADIUS_HEADER_LENGTH;
173  num_attributes = 0;
174 
175  fr_cursor_init(&cursor, &head);
176 
177  /*
178  * Loop over the attributes, decoding them into VPs.
179  */
180  while (packet_length > 0) {
181  ssize_t my_len;
182 
183  /*
184  * This may return many VPs
185  */
186  my_len = fr_radius_decode_pair(packet, &cursor, dict_radius, ptr, packet_length, &packet_ctx);
187  if (my_len < 0) {
188  fail:
189  talloc_free(packet_ctx.tmp_ctx);
190  fr_pair_list_free(&head);
191  return -1;
192  }
193 
194  /*
195  * This should really be an assertion.
196  */
197  if (my_len == 0) break;
198 
199  /*
200  * Count the ones which were just added
201  */
202  while (fr_cursor_next(&cursor)) num_attributes++;
203 
204  /*
205  * VSA's may not have been counted properly in
206  * fr_radius_packet_ok() above, as it is hard to count
207  * then without using the dictionary. We
208  * therefore enforce the limits here, too.
209  */
210  if ((max_attributes > 0) && (num_attributes > max_attributes)) {
211  char host_ipaddr[INET6_ADDRSTRLEN];
212 
213  fr_strerror_printf("Possible DoS attack from host %s: Too many attributes in request "
214  "(received %d, max %d are allowed)",
215  inet_ntop(packet->socket.inet.src_ipaddr.af,
216  &packet->socket.inet.src_ipaddr.addr,
217  host_ipaddr, sizeof(host_ipaddr)),
218  num_attributes, max_attributes);
219  goto fail;
220  }
221 
222  ptr += my_len;
223  packet_length -= my_len;
224  talloc_free_children(packet_ctx.tmp_ctx);
225  }
226 
227  fr_cursor_init(&out, &packet->vps);
228  fr_cursor_tail(&out); /* Move insertion point to the end of the list */
229  fr_cursor_head(&cursor);
230  fr_cursor_merge(&out, &cursor);
231 
232  /*
233  * Merge information from the outside world into our
234  * random pool.
235  */
236  fr_rand_seed(packet->data, RADIUS_HEADER_LENGTH);
237  talloc_free(packet_ctx.tmp_ctx);
238 
239  return 0;
240 }

243 /** See if the data pointed to by PTR is a valid RADIUS packet.
244  *
245  * Packet is not 'const * const' because we may update data_len, if there's more data
246  * in the UDP packet than in the RADIUS packet.
247  *
248  * @param[in] packet to check.
249  * @param[in] max_attributes to decode.
250  * @param[in] require_ma to require Message-Authenticator.
251  * @param[out] reason if not NULL, will have the failure reason written to where it points.
252  * @return
253  * - True on success.
254  * - False on failure.
255  */
256 bool fr_radius_packet_ok(fr_radius_packet_t *packet, uint32_t max_attributes, bool require_ma, decode_fail_t *reason)
257 {
258  char host_ipaddr[INET6_ADDRSTRLEN];
259 
260  if (!fr_radius_ok(packet->data, &packet->data_len, max_attributes, require_ma, reason)) {
261  FR_DEBUG_STRERROR_PRINTF("Bad packet received from host %s",
262  inet_ntop(packet->socket.inet.src_ipaddr.af, &packet->socket.inet.src_ipaddr.addr,
263  host_ipaddr, sizeof(host_ipaddr)));
264  return false;
265  }
266 
267  /*
268  * Fill RADIUS header fields
269  */
270  packet->code = packet->data[0];
271  packet->id = packet->data[1];
272  memcpy(packet->vector, packet->data + 4, sizeof(packet->vector));
273  return true;
274 }

306  *
307  */
308 int fr_radius_packet_sign(fr_radius_packet_t *packet, fr_radius_packet_t const *original,
309  char const *secret)
310 {
311  int ret;
312  uint8_t const *original_data;
313 
314  if (original) {
315  original_data = original->data;
316  } else {
317  original_data = NULL;
318  }
319 
320  /*
321  * Copy the random vector to the packet. Other packet
322  * codes have the Request Authenticator be the packet
323  * signature.
324  */
325  if ((packet->code == FR_CODE_ACCESS_REQUEST) ||
326  (packet->code == FR_CODE_STATUS_SERVER)) {
327  memcpy(packet->data + 4, packet->vector, sizeof(packet->vector));
328  }
329 
330  ret = fr_radius_sign(packet->data, original_data,
331  (uint8_t const *) secret, talloc_array_length(secret) - 1);
332  if (ret < 0) return ret;
333 
334  memcpy(packet->vector, packet->data + 4, RADIUS_AUTH_VECTOR_LENGTH);
335  return 0;
336 }

339 /** Wrapper for recvfrom, which handles recvfromto, IPv6, and all possible combinations
340  *
341  */
342 static ssize_t rad_recvfrom(int sockfd, fr_radius_packet_t *packet, int flags)
343 {
344  ssize_t data_len;
345 
346  data_len = fr_radius_recv_header(sockfd, &packet->socket.inet.src_ipaddr, &packet->socket.inet.src_port, &packet->code);
347  if (data_len < 0) {
348  if ((errno == EAGAIN) || (errno == EINTR)) return 0;
349  return -1;
350  }
351 
352  if (data_len == 0) return -1; /* invalid packet */
353 
354  packet->data = talloc_array(packet, uint8_t, data_len);
355  if (!packet->data) return -1;
356 
357  packet->data_len = data_len;
358 
359  return udp_recv(sockfd, flags, &packet->socket, packet->data, packet->data_len, &packet->timestamp);
360 }

363 /** Receive UDP client requests, and fill in the basics of a fr_radius_packet_t structure
364  *
365  */
366 fr_radius_packet_t *fr_radius_packet_recv(TALLOC_CTX *ctx, int fd, int flags, uint32_t max_attributes, bool require_ma)
367 {
368  ssize_t data_len;
369  fr_radius_packet_t *packet;
370 
371  /*
372  * Allocate the new request data structure
373  */
374  packet = fr_radius_alloc(ctx, false);
375  if (!packet) {
376  fr_strerror_printf("out of memory");
377  return NULL;
378  }
379 
380  data_len = rad_recvfrom(fd, packet, flags);
381  if (data_len < 0) {
382  FR_DEBUG_STRERROR_PRINTF("Error receiving packet: %s", fr_syserror(errno));
383  fr_radius_packet_free(&packet);
384  return NULL;
385  }
386 
387 #ifdef WITH_VERIFY_PTR
388  /*
389  * Double-check that the fields we want are filled in.
390  */
391  if ((packet->socket.inet.src_ipaddr.af == AF_UNSPEC) ||
392  (packet->socket.inet.src_port == 0) ||
393  (packet->socket.inet.dst_ipaddr.af == AF_UNSPEC) ||
394  (packet->socket.inet.dst_port == 0)) {
395  FR_DEBUG_STRERROR_PRINTF("Error receiving packet: %s", fr_syserror(errno));
396  fr_radius_packet_free(&packet);
397  return NULL;
398  }
399 #endif
400 
401  packet->data_len = data_len; /* unsigned vs signed */
402 
403  /*
404  * If the packet is too big, then rad_recvfrom did NOT
405  * allocate memory. Instead, it just discarded the
406  * packet.
407  */
408  if (packet->data_len > MAX_PACKET_LEN) {
409  FR_DEBUG_STRERROR_PRINTF("Discarding packet: Larger than RFC limitation of 4096 bytes");
410  fr_radius_packet_free(&packet);
411  return NULL;
412  }
413 
414  /*
415  * Read no data. Continue.
416  * This check is AFTER the MAX_PACKET_LEN check above, because
417  * if the packet is larger than MAX_PACKET_LEN, we also have
418  * packet->data == NULL
419  */
420  if ((packet->data_len == 0) || !packet->data) {
421  FR_DEBUG_STRERROR_PRINTF("Empty packet: Socket is not ready");
422  fr_radius_packet_free(&packet);
423  return NULL;
424  }
425 
426  /*
427  * See if it's a well-formed RADIUS packet.
428  */
429  if (!fr_radius_packet_ok(packet, max_attributes, require_ma, NULL)) {
430  fr_radius_packet_free(&packet);
431  return NULL;
432  }
433 
434  /*
435  * Remember which socket we read the packet from.
436  */
437  packet->socket.fd = fd;
438 
439  /*
440  * FIXME: Do even more filtering by only permitting
441  * certain IP's. The problem is that we don't know
442  * how to do this properly for all possible clients...
443  */
444 
445  /*
446  * Explicitely set the VP list to empty.
447  */
448  packet->vps = NULL;
449 
450  return packet;
451 }

453 /** Reply to the request
454  *
455  * Also attach reply attribute value pairs and any user message provided.
456  */
457 int fr_radius_packet_send(fr_radius_packet_t *packet, fr_radius_packet_t const *original,
458  char const *secret)
459 {
460  /*
461  * Maybe it's a fake packet. Don't send it.
462  */
463  if (packet->socket.fd < 0) {
464  return 0;
465  }
466 
467  /*
468  * First time through, allocate room for the packet
469  */
470  if (!packet->data) {
471  /*
472  * Encode the packet.
473  */
474  if (fr_radius_packet_encode(packet, original, secret) < 0) {
475  return -1;
476  }
477 
478  /*
479  * Re-sign it, including updating the
480  * Message-Authenticator.
481  */
482  if (fr_radius_packet_sign(packet, original, secret) < 0) {
483  return -1;
484  }
485 
486  /*
487  * If packet->data points to data, then we print out
488  * the VP list again only for debugging.
489  */
490  }
491 
492  /*
493  * If the socket is TCP, call write(). Calling sendto()
494  * is allowed on some platforms, but it's not nice.
495  */
496  if (packet->socket.proto == IPPROTO_TCP) {
497  ssize_t ret;
498 
499  ret = write(packet->socket.fd, packet->data, packet->data_len);
500  if (ret >= 0) return ret;
501 
502  fr_strerror_printf("sendto failed: %s", fr_syserror(errno));
503  return -1;
504  }
505 
506  /*
507  * And send it on it's way.
508  *
509  * No need to call fr_socket_addr_swap as apparently
510  * the address is already inverted.
511  */
512  return udp_send(&packet->socket, 0, packet->data, packet->data_len);
513 }

534  } else {
535  fr_log(log, L_DBG, file, line, " Code : %u", packet->data[0]);
536  }
537 
538  fr_log(log, L_DBG, file, line, " Id : %u", packet->data[1]);
539  fr_log(log, L_DBG, file, line, " Length : %u", ((packet->data[2] << 8) | (packet->data[3])));

541 
542  if (packet->data_len <= 20) return;
543 
544  for (attr = packet->data + 20, end = packet->data + packet->data_len;
545  attr < end;
546  attr += attr[1]) {

562  offset = 6;
563  }

565 
566  len = attr[1] - offset;
567  if (len > 16) len = 16;
568 
569  for (i = 0; i < len; i++) {
