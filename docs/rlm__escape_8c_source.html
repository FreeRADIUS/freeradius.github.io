49 /** Equivalent to the old safe_characters functionality in rlm_sql but with utf8 support
50  *
51  * Example:
52 @verbatim
53 "%{escape:<img>foo.jpg</img>}" == "=60img=62foo.jpg=60/img=62"
54 @endverbatim
55  *
56  * @ingroup xlat_functions
57  */
58 static ssize_t escape_xlat(UNUSED TALLOC_CTX *ctx, char **out, size_t outlen,
59  void const *mod_inst, UNUSED void const *xlat_inst,
60  UNUSED request_t *request, char const *fmt)
61 {
62  rlm_escape_t const *inst = mod_inst;
63  char const *p = fmt;
64  char *out_p = *out;
65  size_t freespace = outlen;
66  size_t len = talloc_array_length(inst->allowed_chars) - 1;
67 
68  while (p[0]) {
69  int chr_len = 1;
70  int ret = 1; /* -Werror=uninitialized */
71 
72  if (fr_utf8_strchr(&chr_len, inst->allowed_chars, len, p) == NULL) {
73  /*
74  * '=' 1 + ([hex]{2}) * chr_len)
75  */
76  if (freespace <= (size_t)(1 + (chr_len * 3))) break;
77 
78  switch (chr_len) {
79  case 4:
80  ret = snprintf(out_p, freespace, "=%02X=%02X=%02X=%02X",
81  (uint8_t)p[0], (uint8_t)p[1], (uint8_t)p[2], (uint8_t)p[3]);
82  break;
83 
84  case 3:
85  ret = snprintf(out_p, freespace, "=%02X=%02X=%02X",
86  (uint8_t)p[0], (uint8_t)p[1], (uint8_t)p[2]);
87  break;
88 
89  case 2:
90  ret = snprintf(out_p, freespace, "=%02X=%02X", (uint8_t)p[0], (uint8_t)p[1]);
91  break;
92 
93  case 1:
94  ret = snprintf(out_p, freespace, "=%02X", (uint8_t)p[0]);
95  break;
96  }
97 
98  p += chr_len;
99  out_p += ret;
100  freespace -= ret;
101  continue;
102  }
103 
104  /*
105  * Only one byte left.
106  */
107  if (freespace <= 1) break;
108 
109  /*
110  * Allowed character (copy whole mb chars at once)
111  */
112  memcpy(out_p, p, chr_len);
113  out_p += chr_len;
114  p += chr_len;
115  freespace -= chr_len;
116  }
117  *out_p = '\0';
118 
119  return outlen - freespace;
120 }

122 /** Equivalent to the old safe_characters functionality in rlm_sql
123  *
124  * Example:
125 @verbatim
126 "%{unescape:=60img=62foo.jpg=60/img=62}" == "<img>foo.jpg</img>"
127 @endverbatim
128  *
129  * @ingroup xlat_functions
130  */
131 static ssize_t unescape_xlat(UNUSED TALLOC_CTX *ctx, char **out, size_t outlen,
132  UNUSED void const *mod_inst, UNUSED void const *xlat_inst,
133  UNUSED request_t *request, char const *fmt)
134 {
135  char const *p;
136  char *out_p = *out;
137  char *c1, *c2, c3;
138  size_t freespace = outlen;
139 
140  if (outlen <= 1) return 0;
141 
142  p = fmt;
143  while (*p && (--freespace > 0)) {
144  if (*p != '=') {
145  next:
146 
147  *out_p++ = *p++;
148  continue;
149  }
150 
151  /* Is a = char */
152 
153  if (!(c1 = memchr(hextab, tolower(*(p + 1)), 16)) ||
154  !(c2 = memchr(hextab, tolower(*(p + 2)), 16))) goto next;
155  c3 = ((c1 - hextab) << 4) + (c2 - hextab);
156 
157  *out_p++ = c3;
158  p += 3;
159  }
160 
161  *out_p = '\0';
162 
163  return outlen - freespace;
164 }
