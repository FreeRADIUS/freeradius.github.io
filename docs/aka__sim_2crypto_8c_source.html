163 /** Locate the start of the AT_MAC value in the buffer
164  *
165  * @param[out] out The start of the digest portion of the AT_MAC attribute.
166  * @param[in] data to search for the AT_MAC in.
167  * @param[in] data_len size of the data.
168  * @return
169  * - 1 if we couldn't find a MAC.
170  * - 0 if we found and zeroed out the mac field.
171  * - -1 if the field was malformed.
172  */
173 static int fr_aka_sim_find_mac(uint8_t const **out, uint8_t *data, size_t data_len)
174 {
175  uint8_t *p = data, *end = p + data_len;
176  size_t len;
177 
178  *out = NULL;
179 
180  p += 3; /* Skip header */
181  while ((p + 2) < end) {
182  if (p[0] == FR_MAC) {
183  len = p[1] << 2;
184  if ((p + len) > end) {
185  fr_strerror_printf("Malformed AT_MAC: Length (%zu) exceeds buffer (%zu)", len, end - p);
186  return -1;
187  }
188 
189  if (len != AKA_SIM_MAC_SIZE) {
190  fr_strerror_printf("Malformed AT_MAC: Length (%zu) incorrect (%u)",
191  len, AKA_SIM_MAC_SIZE);
192  return -1;
193  }
194  *out = p + 4;
195 
196  return 0;
197  }
198  p += p[1] << 2; /* Advance */
199  }
200 
201  fr_strerror_printf("No MAC attribute found");
202 
203  return 1;
204 }

206 /** Calculate the digest value for a packet
207  *
208  * Run a digest over a fake EAP header, the entire SIM packet and any extra HMAC data,
209  * writing a truncated (16 byte) digest value to out.
210  *
211  * @note The 16 byte digest field in the packet must have either been zeroed out before
212  * this function is called (as it is when encoding data), or zero_mac must be set
213  * to true.
214  *
215  * @note This function uses the EVP_* signing functions. Do not be tempted to swap them
216  * for the HMAC functions, as the EVP interface may be hardware accelerated but
217  * the HMAC interface is purely a software implementation.
218  *
219  * @param[out] out Where to write the digest.
220  * @param[in] eap_packet to extract header values from.
221  * @param[in] zero_mac Assume the mac field is not zeroed (i.e. received packet)
222  * and skip it during mac calculation feeding in 16 zeroed
223  * bytes in its place.
224  * @param[in] md to use to create the HMAC.
225  * @param[in] key to use to sign the packet.
226  * @param[in] key_len Length of the key.
227  * @param[in] hmac_extra data to concatenate with the packet when calculating the HMAC
228  * (may be NULL).
229  * @param[in] hmac_extra_len Length of hmac_extra (may be zero).
230  * @return
231  * - < 0 on failure.
232  * - 0 if there's no MAC attribute to verify.
233  * - > 0 the number of bytes written to out.
234  */
235 ssize_t fr_aka_sim_crypto_sign_packet(uint8_t out[static AKA_SIM_MAC_DIGEST_SIZE],
236  eap_packet_t *eap_packet, bool zero_mac,
237  EVP_MD const *md, uint8_t const *key, size_t const key_len,
238  uint8_t const *hmac_extra, size_t const hmac_extra_len)
239 {
240  EVP_MD_CTX *md_ctx = NULL;
241  EVP_PKEY *pkey;
242 
243  uint8_t digest[SHA256_DIGEST_LENGTH];
244  size_t digest_len = 0;
245  uint8_t const *mac;
246  uint8_t *p = eap_packet->type.data, *end = p + eap_packet->type.length;
247 
248  eap_packet_raw_t eap_hdr;
249  uint16_t packet_len;
250 
251  if (unlikely(!eap_packet)) {
252  fr_strerror_printf("Invalid argument: eap_packet is NULL");
253  return -1;
254  }
255 
256  if (unlikely(!md)) {
257  fr_strerror_printf("Invalid argument: md is NULL");
258  return -1;
259  }
260 
261  if (unlikely(!key) || (key_len == 0)) {
262  fr_strerror_printf("Invalid argument: key is NULL");
263  return -1;
264  }
265 
266  FR_PROTO_HEX_DUMP(key, key_len, "MAC key");
267  pkey = EVP_PKEY_new_mac_key(EVP_PKEY_HMAC, NULL, key, key_len);
268  if (!pkey) {
269  tls_strerror_printf("Failed creating HMAC signing key");
270  error:
271  if (pkey) EVP_PKEY_free(pkey);
272  if (md_ctx) EVP_MD_CTX_destroy(md_ctx);
273  return -1;
274  }
275 
276  md_ctx = EVP_MD_CTX_create();
277  if (!md_ctx) {
278  tls_strerror_printf("Failed creating HMAC ctx");
279  goto error;
280  }
281 
282  if (EVP_DigestSignInit(md_ctx, NULL, md, NULL, pkey) != 1) {
283  tls_strerror_printf("Failed initialising digest");
284  goto error;
285  }
286 
287  /*
288  * The HMAC has to be over the entire packet, which
289  * we don't get access too. So we create a fake EAP
290  * header now, and feed that into the HMAC function.
291  */
292  eap_hdr.code = eap_packet->code;
293  eap_hdr.id = eap_packet->id;
294  packet_len = htons((sizeof(eap_hdr) + eap_packet->type.length) & UINT16_MAX); /* EAP Header + Method + SIM data */
295  memcpy(&eap_hdr.length, &packet_len, sizeof(packet_len));
296  eap_hdr.data[0] = eap_packet->type.num;
297 
298  FR_PROTO_HEX_DUMP((uint8_t *)&eap_hdr, sizeof(eap_hdr), "MAC digest input (eap header)");
299  if (EVP_DigestSignUpdate(md_ctx, &eap_hdr, sizeof(eap_hdr)) != 1) {
300  tls_strerror_printf("Failed digesting EAP data");
301  goto error;
302  }
303 
304  /*
305  * Digest the packet up to the AT_MAC, value, then
306  * digest 16 bytes of zero.
307  */
308  if (zero_mac) {
309  switch (fr_aka_sim_find_mac(&mac, p, end - p)) {
310  case 0:
311  {
312  uint8_t zero[16] = { 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00,
313  0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00 };
314 
315  FR_PROTO_HEX_DUMP(p, mac - p, "MAC digest input");
316 
317  /*
318  * Digest everything up to the hash
319  * part of the AT_MAC, including
320  * AT_MAC header and reserved bytes.
321  */
322  if (EVP_DigestSignUpdate(md_ctx, p, mac - p) != 1) {
323  tls_strerror_printf("Failed digesting packet data (before MAC)");
324  goto error;
325  }
326  p += mac - p;
327 
328 
329  FR_PROTO_HEX_DUMP(zero, sizeof(zero), "MAC digest input");
330  /*
331  * Feed in 16 bytes of zeroes to
332  * simulated the zeroed out Mac.
333  */
334  if (EVP_DigestSignUpdate(md_ctx, zero, sizeof(zero)) != 1) {
335  tls_strerror_printf("Failed digesting zeroed MAC");
336  goto error;
337  }
338  p += sizeof(zero);
339  }
340  break;
341 
342  case 1:
343  return 0;
344 
345  case -1:
346  fr_assert(0); /* Should have been checked by encoder or decoder */
347  goto error;
348  }
349  }
350 
351  if (p < end) {
352  FR_PROTO_HEX_DUMP(p, (end - p), "MAC digest input");
353 
354  /*
355  * Digest the rest of the packet.
356  */
357  if (EVP_DigestSignUpdate(md_ctx, p, end - p) != 1) {
358  tls_strerror_printf("Failed digesting packet data");
359  goto error;
360  }
361  }
362 
363  /*
364  * Digest any HMAC concatenated data
365  *
366  * Some subtypes require the HMAC to be calculated over
367  * a concatenation of packet data, and something extra...
368  */
369  if (hmac_extra) {
370  FR_PROTO_HEX_DUMP(hmac_extra, hmac_extra_len, "MAC digest input (extra)");
371  if (EVP_DigestSignUpdate(md_ctx, hmac_extra, hmac_extra_len) != 1) {
372  tls_strerror_printf("Failed digesting HMAC extra data");
373  goto error;
374  }
375  }
376 
377  if (EVP_DigestSignFinal(md_ctx, digest, &digest_len) != 1) {
378  tls_strerror_printf("Failed finalising digest");
379  goto error;
380  }
381 
382  FR_PROTO_HEX_DUMP(digest, digest_len, "MAC");
383 
384  /*
385  * Truncate by four bytes.
386  */
387  memcpy(out, digest, 16);
388 
389  EVP_PKEY_free(pkey);
390  EVP_MD_CTX_destroy(md_ctx);
391 
392  return 16; /* AT_MAC (1), LEN (1), RESERVED (2) */
393 }

395 /** Key Derivation Function as described in RFC4186 (EAP-SIM) section 7
396  *
397  @verbatim
398  MK = SHA1(Identity|n*Kc| NONCE_MT| Version List| Selected Version)
399  FK = PRF(MK)
400  K_encr = FK[0..127]
401  K_aut = FK[128..255]
402  MSK = FK[256..767]
403  EMSK = FK[768..1279]
404  @endverbatim
405  * @note expects keys to contain a AKA_SIM_VECTOR_GSM.
406  *
407  * @param[in,out] keys Contains the authentication vectors and the buffers
408  * to store the result of the derivation.
409  * @return
410  * - 0 on success.
411  * - -1 on failure.
412  */
413 int fr_aka_sim_crypto_gsm_kdf_0(fr_aka_sim_keys_t *keys)
414 {
415  fr_sha1_ctx context;
416  uint8_t fk[160];
417 
418  uint8_t buf[AKA_SIM_MAX_STRING_LENGTH + sizeof(keys->gsm.nonce_mt) + 2 + sizeof(keys->gsm.version_select)];
419  uint8_t *p;
420 
421  if (!fr_cond_assert(keys->vector_type == AKA_SIM_VECTOR_GSM)) return -1;
422 
423  /*
424  * Our stack buffer should be large enough in
425  * all cases.
426  */

428  (AKA_SIM_VECTOR_GSM_KC_SIZE * 3) +
429  sizeof(keys->gsm.nonce_mt) +
430  keys->gsm.version_list_len +
431  sizeof(keys->gsm.version_select)) <= sizeof(buf))) return -1;
432 
433  p = buf;
434  memcpy(p, keys->identity, keys->identity_len);
435  p += keys->identity_len;
436 
437  memcpy(p, keys->gsm.vector[0].kc, AKA_SIM_VECTOR_GSM_KC_SIZE);
438  p += AKA_SIM_VECTOR_GSM_KC_SIZE;
439 
440  memcpy(p, keys->gsm.vector[1].kc, AKA_SIM_VECTOR_GSM_KC_SIZE);

442

488

490  FR_PROTO_HEX_DUMP(keys->emsk, sizeof(keys->emsk), "K_emsk");
491 
492  return 0;
493 }

573 /** Key Derivation Function (CK', IK') as specified in 3GPP.33.402
574  *
575  @verbatim
576  CK' || IK' = HMAC-SHA-256(Key, S)
577  S = FC || P0 || L0 || P1 || L1 || ... || Pn || Ln
578  Key = CK || IK
579  FC = 0x20
580  P0 = access network identity (3GPP TS 24.302)
581  L0 = length of acceess network identity (2 octets, big endian)
582  P1 = SQN xor AK (if AK is not used, AK is treated as 000..0
583  L1 = 0x00 0x06
584  @endverbatim
585  *
586  * @note expects keys to contain a AKA_SIM_VECTOR_UMTS.
587  *
588  * @param[in,out] keys Contains the authentication vectors and the buffers
589  * to store the result of the derivation.
590  * @return
591  * - 0 on success.
592  * - -1 on failure.
593  */
594 static int ck_ik_prime_derive(fr_aka_sim_keys_t *keys)
595 {
596  uint8_t digest[sizeof(keys->ik_prime) + sizeof(keys->ck_prime)];
597  size_t len;
598 
599  uint8_t sqn_ak_buff[MILENAGE_SQN_SIZE];
600  uint16_t l0, l1;
601 
602  uint8_t k[sizeof(keys->umts.vector.ik) + sizeof(keys->umts.vector.ck)];
603  uint8_t s[sizeof(uint8_t) + AKA_SIM_MAX_STRING_LENGTH + sizeof(l0) + AKA_SIM_SQN_AK_SIZE + sizeof(l1)];
604 
605  uint8_t *p = s;
606

697 /** PRF as described in RFC 5448 (EAP-AKA') section 3.4.1
698  *
699  @verbatim
700  PRF'(K,S) = T1 | T2 | T3 | T4 | ...
701 
702  where:
703  T1 = HMAC-SHA-256 (K, S | 0x01)
704  T2 = HMAC-SHA-256 (K, T1 | S | 0x02)
705  T3 = HMAC-SHA-256 (K, T2 | S | 0x03)
706  T4 = HMAC-SHA-256 (K, T3 | S | 0x04)
707  ...
708  @endverbatim
709  *
710  * PRF' produces as many bits of output as is needed.
711  *
712  * @param[out] out Where to write the output of the PRF.
713  * @param[in] outlen how many bytes need to be generated.
714  * @param[in] key for the PRF (K).
715  * @param[in] key_len Length of key data.
716  * @param[in] in Data to feed into the PRF (S).
717  * @param[in] in_len Length of input data.
718  * @return
719  * - 0 on success.
720  * - -1 on failure.
721  */
722 static int aka_prime_prf(uint8_t *out, size_t outlen,
723  uint8_t const *key, size_t key_len, uint8_t const *in, size_t in_len)
724 {
725  uint8_t *p = out, *end = p + outlen;
726  uint8_t c = 0;
727  uint8_t digest[SHA256_DIGEST_LENGTH];
728  EVP_PKEY *pkey;
729  EVP_MD_CTX *md_ctx = NULL;
730 
731  pkey = EVP_PKEY_new_mac_key(EVP_PKEY_HMAC, NULL, key, key_len);
732  if (!pkey) {
733  tls_strerror_printf("Failed creating HMAC signing key");
734  error:
735  if (pkey) EVP_PKEY_free(pkey);
736  if (md_ctx) EVP_MD_CTX_destroy(md_ctx);
737  return -1;
738  }
739 
740  md_ctx = EVP_MD_CTX_create();
741  if (!md_ctx) {
742  tls_strerror_printf("Failed creating HMAC ctx");
743  goto error;
744  }
745 
746  if (EVP_DigestSignInit(md_ctx, NULL, EVP_sha256(), NULL, pkey) != 1) {
747  tls_strerror_printf("Failed initialising digest");
748  goto error;
749  }
750 
751  while (p < end) {
752  size_t len;
753  size_t copy;
754 
755  c++;
756 
757  if (EVP_DigestSignInit(md_ctx, NULL, EVP_sha256(), NULL, pkey) != 1) goto error;
758  if ((p != out) && EVP_DigestSignUpdate(md_ctx, digest, sizeof(digest)) != 1) goto error;/* Ingest last round */
759  if (EVP_DigestSignUpdate(md_ctx, in, in_len) != 1) goto error; /* Ingest s */
760  if (EVP_DigestSignUpdate(md_ctx, &c, sizeof(c)) != 1) goto error; /* Ingest round number */
761  if (EVP_DigestSignFinal(md_ctx, digest, &len) != 1) goto error; /* Output T(i) */
762 
763  copy = end - p;
764  if (copy > SHA256_DIGEST_LENGTH) copy = SHA256_DIGEST_LENGTH;
765 
766  memcpy(p, digest, copy);
767  p += copy;
768  }
769 
770  EVP_MD_CTX_destroy(md_ctx);;
771  EVP_PKEY_free(pkey);
772 
773  return 0;
774 }
