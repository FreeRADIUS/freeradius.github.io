66 /** Produce a CHAP-Password hash value
67  *
68  * Example:
69 @verbatim
70 "%{chap_password:<password>}" == 0x<id><md5_hash>
71 @endverbatim
72  *
73  * @ingroup xlat_functions
74  */
75 static xlat_action_t xlat_func_chap_password(TALLOC_CTX *ctx, fr_cursor_t *out,
76  request_t *request, UNUSED void const *xlat_inst,
77  UNUSED void *xlat_thread_inst,
78  fr_value_box_t **in)
79 {
80  uint8_t chap_password[1 + RADIUS_CHAP_CHALLENGE_LENGTH];
81  fr_value_box_t *vb;
82  fr_pair_t *challenge;
83  uint8_t const *vector;
84 
85  /*
86  * If there's no input, there's no output
87  */
88  if (!in) return XLAT_ACTION_DONE;
89 
90  if (fr_value_box_list_concat(ctx, *in, in, FR_TYPE_STRING, true) < 0) {
91  RPEDEBUG("Failed concatenating input");
92  return XLAT_ACTION_FAIL;
93  }
94 
95  /*
96  * Use Chap-Challenge pair if present,
97  * Request Authenticator otherwise.
98  */
99  challenge = fr_pair_find_by_da(&request->request_pairs, attr_chap_challenge);
100  if (challenge && (challenge->vp_length == RADIUS_AUTH_VECTOR_LENGTH)) {
101  vector = challenge->vp_octets;
102  } else {
103  vector = request->packet->vector;
104  }
105  fr_radius_encode_chap_password(chap_password, (uint8_t)(fr_rand() & 0xff), vector,
106  (*in)->vb_strvalue, (*in)->vb_length);
107 
108  MEM(vb = fr_value_box_alloc_null(ctx));
109  fr_value_box_memdup(vb, vb, NULL, chap_password, sizeof(chap_password), false);
110  fr_cursor_append(out, vb);
111 
112  return XLAT_ACTION_DONE;
113 }

115 static unlang_action_t CC_HINT(nonnull) mod_authorize(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
116 {
117  fr_pair_t *vp;
118  rlm_chap_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_chap_t);
119 
120  if (fr_pair_find_by_da(&request->control_pairs, attr_auth_type) != NULL) {
121  RDEBUG3("Auth-Type is already set. Not setting 'Auth-Type := %s'", inst->name);
122  RETURN_MODULE_NOOP;
123  }
124 
125  /*
126  * This case means the warnings below won't be printed
127  * unless there's a CHAP-Password in the request.
128  */
129  if (!fr_pair_find_by_da(&request->request_pairs, attr_chap_password)) {
130  RETURN_MODULE_NOOP;
131  }
132 
133  /*
134  * Create the CHAP-Challenge if it wasn't already in the packet.
135  *
136  * This is so that the rest of the code does not need to
137  * understand CHAP.
138  */
139  vp = fr_pair_find_by_da(&request->request_pairs, attr_chap_challenge);
140  if (!vp) {
141  RDEBUG2("Creating &%s from request authenticator", attr_chap_challenge->name);
142 
143  MEM(vp = fr_pair_afrom_da(request->packet, attr_chap_challenge));
144  fr_pair_value_memdup(vp, request->packet->vector, sizeof(request->packet->vector), true);
145  fr_pair_add(&request->request_pairs, vp);
146  }
147 
148  if (!inst->auth_type) {
149  WARN("No 'authenticate %s {...}' section or 'Auth-Type = %s' set. Cannot setup CHAP authentication",
150  inst->name, inst->name);
151  RETURN_MODULE_NOOP;
152  }
153 
154  if (!module_section_type_set(request, attr_auth_type, inst->auth_type)) {
155  RETURN_MODULE_NOOP;
156  }
157 
158  RETURN_MODULE_OK;
159 }
160 
161 /*
162  * Find the named user in this modules database. Create the set
163  * of attribute-value pairs to check and reply with for this user
164  * from the database. The authentication code only needs to check
165  * the password, the rest is done here.
166  */
167 static unlang_action_t CC_HINT(nonnull) mod_authenticate(rlm_rcode_t *p_result, UNUSED module_ctx_t const *mctx, request_t *request)
168 {
169  fr_pair_t *known_good;
170  fr_pair_t *chap, *username;
171  uint8_t pass_str[1 + RADIUS_CHAP_CHALLENGE_LENGTH];
172 
173  int ret;
174 
175  fr_dict_attr_t const *allowed_passwords[] = { attr_cleartext_password };
176  bool ephemeral;
177 
178  fr_pair_t *challenge;
179  uint8_t const *vector;
180 
181  username = fr_pair_find_by_da(&request->request_pairs, attr_user_name);
182  if (!username) {
183  REDEBUG("&User-Name attribute is required for authentication");
184  RETURN_MODULE_INVALID;
185  }
186 
187  chap = fr_pair_find_by_da(&request->request_pairs, attr_chap_password);
188  if (!chap) {
189  REDEBUG("You set '&control.Auth-Type = CHAP' for a request that "
190  "does not contain a CHAP-Password attribute!");
191  RETURN_MODULE_INVALID;
192  }
193 
194  if (chap->vp_length == 0) {
195  REDEBUG("&request.CHAP-Password is empty");
196  RETURN_MODULE_INVALID;
197  }
198 
199  if (chap->vp_length != RADIUS_CHAP_CHALLENGE_LENGTH + 1) {
200  REDEBUG("&request.CHAP-Password has invalid length");
201  RETURN_MODULE_INVALID;
202  }
203 
204  /*
205  * Retrieve the normalised version of
206  * the known_good password, without
207  * mangling the current password attributes
208  * in the request.
209  */
210  known_good = password_find(&ephemeral, request, request,
211  allowed_passwords, NUM_ELEMENTS(allowed_passwords),
212  false);
213  if (!known_good) {
214  REDEBUG("No \"known good\" password found for user");
215  RETURN_MODULE_FAIL;
216  }
217 
218  /*
219  * Output is id + password hash
220  */
221 
222  /*
223  * Use Chap-Challenge pair if present,
224  * Request Authenticator otherwise.
225  */
226  challenge = fr_pair_find_by_da(&request->request_pairs, attr_chap_challenge);
227  if (challenge && (challenge->vp_length == RADIUS_AUTH_VECTOR_LENGTH)) {
228  vector = challenge->vp_octets;
229  } else {
230  vector = request->packet->vector;
231  }
232  fr_radius_encode_chap_password(pass_str, chap->vp_octets[0], vector,
233  known_good->vp_strvalue, known_good->vp_length);
234 
235  /*
236  * The password_find function already emits
237  * a log message about the password attribute contents
238  * so we don't need to duplicate it here.
239  */
240  if (RDEBUG_ENABLED3) {
241  uint8_t const *p;
242  size_t length;
243  fr_pair_t *vp;
244 
245  vp = fr_pair_find_by_da(&request->request_pairs, attr_chap_challenge);
246  if (vp) {
247  RDEBUG2("Using challenge from &request.CHAP-Challenge");
248  p = vp->vp_octets;
249  length = vp->vp_length;
250  } else {
251  RDEBUG2("Using challenge from authenticator field");
252  p = request->packet->vector;
253  length = sizeof(request->packet->vector);
254  }
255 
256  RINDENT();
257  RDEBUG3("CHAP challenge : %pH", fr_box_octets(p, length));
258  RDEBUG3("Client sent : %pH", fr_box_octets(chap->vp_octets + 1, RADIUS_CHAP_CHALLENGE_LENGTH));
259  RDEBUG3("We calculated : %pH", fr_box_octets(pass_str + 1, RADIUS_CHAP_CHALLENGE_LENGTH));
260  REXDENT();
261  }
262 
263  /*
264  * Skip the id field at the beginning of the
265  * password and chap response.
266  */
267  ret = fr_digest_cmp(pass_str + 1, chap->vp_octets + 1, RADIUS_CHAP_CHALLENGE_LENGTH);
268  if (ephemeral) talloc_list_free(&known_good);
269  if (ret != 0) {
270  REDEBUG("Password comparison failed: password is incorrect");
271 
272  RETURN_MODULE_REJECT;
273  }
274 
275  RDEBUG2("CHAP user \"%pV\" authenticated successfully", &username->data);
276 
277  RETURN_MODULE_OK;
278 }
