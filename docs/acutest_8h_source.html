63 /* Macros for testing whether an unit test succeeds or fails. These macros
64  * can be used arbitrarily in functions implementing the unit tests.
65  *
66  * If any condition fails throughout execution of a test, the test fails.
67  *
68  * TEST_CHECK takes only one argument (the condition), TEST_CHECK_ allows
69  * also to specify an error message to print out if the condition fails.
70  * (It expects printf-like format string and its parameters). The macros
71  * return non-zero (condition passes) or 0 (condition fails).
72  *
73  * That can be useful when more conditions should be checked only if some
74  * preceding condition passes, as illustrated in this code snippet:
75  *
76  * SomeStruct* ptr = allocate_some_struct();
77  * if(TEST_CHECK(ptr != NULL)) {
78  * TEST_CHECK(ptr->member1 < 100);
79  * TEST_CHECK(ptr->member2 > 200);
80  * }
81  */
82 #define TEST_CHECK_(cond,...) test_check__((cond), __FILE__, __LINE__, __VA_ARGS__)
83 #define TEST_CHECK(cond) test_check__((cond), __FILE__, __LINE__, "%s", #cond)

86 /* These macros are the same as TEST_CHECK_ and TEST_CHECK except that if the
87  * condition fails, the currently executed unit test is immediately aborted.
88  *
89  * That is done either by calling abort() if the unit test is executed as a
90  * child process; or via longjmp() if the unit test is executed within the
91  * main Acutest process.
92  *
93  * As a side effect of such abortion, your unit tests may cause memory leaks,
94  * unflushed file descriptors, and other fenomena caused by the abortion.
95  *
96  * Therefore you should not use these as a general replacement for TEST_CHECK.
97  * Use it with some caution, especially if your test causes some other side
98  * effects to the outside world (e.g. communicating with some server, inserting
99  * into a database etc.).
100  */
101 #define TEST_ASSERT_(cond,...) \
102  do { \
103  if (!test_check__((cond), __FILE__, __LINE__, __VA_ARGS__)) \
104  test_abort__(); \
105  } while(0)
106 #define TEST_ASSERT(cond) \
107  do { \
108  if (!test_check__((cond), __FILE__, __LINE__, "%s", #cond)) \
109  test_abort__(); \
110  } while(0)

1010  * process who calls test_do_run__(), otherwise it calls test_do_run__()
1011  * directly. */

1013 test_run__(const struct test__* test, int index, int master_index)
1014 {

1016  test_timer_type__ start, end;
1017 
1018  test_current_unit__ = test;
1019  test_current_already_logged__ = 0;
1020  test_timer_get_time__(&start);
1021 
1022  if(!test_no_exec__) {
1023 
1024 #if defined(ACUTEST_UNIX__)
1025 
1026  pid_t pid;
1027  int exit_code;
1028 
1029  /* Make sure the child starts with empty I/O buffers. */
1030  fflush(stdout);
1031  fflush(stderr);
1032 
1033  pid = fork();
1034  if(pid == (pid_t)-1) {
1035  test_error__("Cannot fork. %s [%d]", strerror(errno), errno);
1036  failed = 1;

1038  /* Child: Do the test. */
1039  test_worker__ = 1;

1047  case 0: failed = 0; break; /* test has passed. */
1048  case 1: /* noop */ break; /* "normal" failure. */
1049  default: test_error__("Unexpected exit code [%d]", WEXITSTATUS(exit_code));

1055  case SIGINT: signame = "SIGINT"; break;
1056  case SIGHUP: signame = "SIGHUP"; break;
1057  case SIGQUIT: signame = "SIGQUIT"; break;
1058  case SIGABRT: signame = "SIGABRT"; break;
1059  case SIGKILL: signame = "SIGKILL"; break;
1060  case SIGSEGV: signame = "SIGSEGV"; break;
1061  case SIGILL: signame = "SIGILL"; break;
1062  case SIGTERM: signame = "SIGTERM"; break;

1064  }

1066  } else {
1067  test_error__("Test ended in an unexpected way [%d].", exit_code);

1069  }

1100  }
1101  } else {
1102  test_error__("Cannot create unit test subprocess [%ld].", GetLastError());
1103  failed = 1;

1105 
1106 #else
1107 
1108  /* A platform where we don't know how to run child process. */
1109  failed = (test_do_run__(test, index) != 0);

1111 #endif

1115  failed = (test_do_run__(test, index) != 0);
1116  }

1122  if(failed)

1207

1212  /* Non-option argument. */
1213  ret = callback(TEST_CMDLINE_OPTID_NONE__, argv[i]);
1214  } else if(strcmp(argv[i], "--") == 0) {
1215  /* End of options. All the remaining members are non-option arguments. */
1216  after_doubledash = 1;
1217  } else if(argv[i][0] != '-') {
1218  /* Non-option argument. */
1219  ret = callback(TEST_CMDLINE_OPTID_NONE__, argv[i]);
1220  } else {
1221  for(opt = options; opt->id != 0; opt++) {
1222  if(opt->longname != NULL && strncmp(argv[i], "--", 2) == 0) {
1223  size_t len = strlen(opt->longname);
1224  if(strncmp(argv[i]+2, opt->longname, len) == 0) {
1225  /* Regular long option. */
1226  if(argv[i][2+len] == '\0') {
1227  /* with no argument provided. */
1228  if(!(opt->flags & TEST_CMDLINE_OPTFLAG_REQUIREDARG__))
1229  ret = callback(opt->id, NULL);
1230  else
1231  ret = callback(TEST_CMDLINE_OPTID_MISSINGARG__, argv[i]);
1232  break;
1233  } else if(argv[i][2+len] == '=') {
1234  /* with an argument provided. */

1236  ret = callback(opt->id, argv[i]+2+len+1);

1239  ret = callback(TEST_CMDLINE_OPTID_BOGUSARG__, auxbuf);
1240  }
1241  break;
1242  } else {

1244  }
1245  }

1247  if(argv[i][1] == opt->shortname) {
1248  /* Regular short option. */
1249  if(opt->flags & TEST_CMDLINE_OPTFLAG_REQUIREDARG__) {
1250  if(argv[i][2] != '\0')
1251  ret = callback(opt->id, argv[i]+2);
1252  else if(i+1 < argc)

1255  ret = callback(TEST_CMDLINE_OPTID_MISSINGARG__, argv[i]);
1256  break;
1257  } else {

1259

1261  * grouped together. */

1266  }

1271  if(argv[i][0] != '-') {
1272  /* Non-option argument. */
1273  ret = callback(TEST_CMDLINE_OPTID_NONE__, argv[i]);

1298  i++;
1299  }
