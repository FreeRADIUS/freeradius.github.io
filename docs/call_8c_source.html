33 static unlang_action_t unlang_call_process(rlm_rcode_t *p_result, request_t *request)
34 {
35  unlang_stack_t *stack = request->stack;
36  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
37  unlang_frame_state_call_t *state = talloc_get_type_abort(frame->state, unlang_frame_state_call_t);
38  unlang_t *instruction = frame->instruction;
39  unlang_group_t *g = unlang_generic_to_group(instruction);
40 
41  rlm_rcode_t rcode;
42  unlang_action_t ua;
43 
44  request->request_state = REQUEST_INIT;
45  request->server_cs = state->prev_server_cs; /* So we get correct debug info */
46 
47  /*
48  * Call the process function
49  *
50  * This is a function in the virtual server's state machine.
51  */
52  ua = state->process(&rcode, &(module_ctx_t){ .instance = state->instance }, request);
53  switch (ua) {
54  case UNLANG_ACTION_YIELD:
55  case UNLANG_ACTION_STOP_PROCESSING:
56  case UNLANG_ACTION_PUSHED_CHILD: /* Wants to do more processing */
57  return ua;
58 
59  default:
60  break;
61  }
62 
63  /*
64  * Record the rcode and restore the previous virtual server
65  */
66  *p_result = rcode;
67  request->request_state = state->prev_request_state;
68  request->server_cs = state->prev_server_cs;
69 
70  /*
71  * Push the contents of the call { } section onto the stack.
72  * This gets executed after the server returns.
73  */
74  if (g->children) {
75  if (unlang_interpret_push(request, g->children, frame->result,
76  UNLANG_NEXT_SIBLING, UNLANG_SUB_FRAME) < 0) {
77  *p_result = RLM_MODULE_FAIL;
78  return UNLANG_ACTION_STOP_PROCESSING;
79  }
80  return UNLANG_ACTION_PUSHED_CHILD;
81  };
82 
83  return UNLANG_ACTION_CALCULATE_RESULT;
84 }

86 static unlang_action_t unlang_call_frame_init(rlm_rcode_t *p_result, request_t *request)
87 {
88  unlang_stack_t *stack = request->stack;
89  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
90  unlang_frame_state_call_t *state = talloc_get_type_abort(frame->state, unlang_frame_state_call_t);
91  unlang_t *instruction = frame->instruction;
92 
93  unlang_group_t *g;
94  unlang_call_t *gext;
95  char const *server;
96  fr_dict_enum_t const *type_enum;
97 
98  module_method_t process_p;
99  void *process_inst;
100 
101  /*
102  * Do not check for children here.
103  *
104  * Call shouldn't require children to execute as there
105  * can still be side effects from executing the virtual
106  * server.
107  */
108  g = unlang_generic_to_group(instruction);
109  gext = unlang_group_to_call(g);
110  server = cf_section_name2(gext->server_cs);
111 
112  /*
113  * Push OUR subsection onto the childs stack frame.
114  */
115 
116  /*
117  * Work out the current request type.
118  */
119  type_enum = fr_dict_enum_by_value(gext->attr_packet_type, fr_box_uint32(request->packet->code));
120  if (!type_enum) {
121  REDEBUG("No such value '%d' of attribute 'Packet-Type' for server %s", request->packet->code, server);
122  *p_result = RLM_MODULE_FAIL;
123  return UNLANG_ACTION_CALCULATE_RESULT;
124  }
125 
126  /*
127  * ...and get the processing function
128  * which matches that type in the target
129  * virtual server.
130  */
131  if (virtual_server_get_process_by_name(gext->server_cs, type_enum->name, &process_p, &process_inst) < 0) {
132  REDEBUG("Cannot call virtual server '%s' - %s", server, fr_strerror());
133  *p_result = RLM_MODULE_FAIL;
134  return UNLANG_ACTION_CALCULATE_RESULT;
135  }
136 
137  *state = (unlang_frame_state_call_t){
138  .instance = process_inst,
139  .process = process_p,
140  .prev_request_state = request->request_state,
141  .prev_server_cs = request->server_cs,
142  };
143 
144  return unlang_call_process(p_result, request);
145 }

147 /** Push a call to a virtual server onto the stack for evaluation
148  *
149  * This does the same work as #unlang_call_frame_init.
150  *
151  * @param[in] request The current request.
152  * @param[in] server_cs of the virtual server to run.
153  * @param[in] instance of the state machine.
154  * @param[in] entry_point Where to start in the virtual server state machine.
155  * @param[in] top_frame Set to UNLANG_TOP_FRAME if the interpreter should return.
156  * Set to UNLANG_SUB_FRAME if the interprer should continue.
157  */
158 int unlang_call_push(request_t *request, CONF_SECTION *server_cs,
159  void *instance, module_method_t entry_point, bool top_frame)
160 {
161  unlang_stack_t *stack = request->stack;
162  unlang_stack_frame_t *frame;
163  unlang_frame_state_call_t *state;
164  unlang_call_t *c;
165  char const *name;
166 
167  /*
168  * We need to have a unlang_module_t to push on the
169  * stack. The only sane way to do it is to attach it to
170  * the frame state.
171  */
172  name = cf_section_name2(server_cs);
173  MEM(c = talloc(stack, unlang_call_t)); /* Free at the same time as the state */
174  *c = (unlang_call_t){
175  .group = {
176  .self = {
177  .type = UNLANG_TYPE_CALL,
178  .name = name,
179  .debug_name = name,
180  .actions = {
181  [RLM_MODULE_REJECT] = 0,
182  [RLM_MODULE_FAIL] = MOD_ACTION_RETURN, /* Exit out of nested levels */
183  [RLM_MODULE_OK] = 0,
184  [RLM_MODULE_HANDLED] = 0,
185  [RLM_MODULE_INVALID] = 0,
186  [RLM_MODULE_DISALLOW] = 0,
187  [RLM_MODULE_NOTFOUND] = 0,
188  [RLM_MODULE_NOOP] = 0,
189  [RLM_MODULE_UPDATED] = 0
190  }
191  }
192  }
193  };
194 
195  /*
196  * Push a new call frame onto the stack
197  */
198  if (unlang_interpret_push(request, unlang_call_to_generic(c),
199  RLM_MODULE_UNKNOWN, UNLANG_NEXT_STOP, top_frame) < 0) {
200  talloc_free(c);
201  return -1;
202  }
203 
204  /*
205  * And setup the frame. The memory was
206  * pre-allocated for us by the interpreter.
207  */
208  frame = &stack->frame[stack->depth];
209  state = talloc_get_type_abort(frame->state, unlang_frame_state_call_t);
210  *state = (unlang_frame_state_call_t){
211  .instance = instance,
212  .process = entry_point, /* This mutates */
213  .prev_request_state = request->request_state,
214  .prev_server_cs = request->server_cs
215  };
216  frame->process = unlang_call_process; /* Skip the initialisation */
217  talloc_steal(frame, c); /* Bind our temporary unlang_call_t to the frame */
218 
219  return 0;
220 }
