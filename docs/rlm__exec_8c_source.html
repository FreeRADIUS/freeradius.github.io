68 static size_t rlm_exec_shell_escape(UNUSED request_t *request, char *out, size_t outlen, char const *in,
69  UNUSED void *inst)
70 {
71  char *q, *end;
72  char const *p;
73 
74  q = out;
75  end = out + outlen;
76  p = in;
77 
78  while (*p) {
79  if ((q + 3) >= end) break;
80 
81  if (strchr(special, *p) != NULL) {
82  *(q++) = '\\';
83  }
84  *(q++) = *(p++);
85  }
86 
87  *q = '\0';
88  return q - out;
89 }

92 /** Exec programs from an xlat
93  *
94  * Example:
95 @verbatim
96 "%{exec:/bin/echo hello}" == "hello"
97 @endverbatim
98  *
99  * @ingroup xlat_functions
100  */
101 static ssize_t exec_xlat(UNUSED TALLOC_CTX *ctx, char **out, size_t outlen,
102  void const *mod_inst, UNUSED void const *xlat_inst,
103  request_t *request, char const *fmt)
104 {
105  int result;
106  rlm_exec_t const *inst = mod_inst;
107  fr_pair_t **input_pairs = NULL;
108  char *p;
109 
110  if (!inst->wait) {
111  REDEBUG("'wait' must be enabled to use exec xlat");
112  return -1;
113  }
114 
115  if (inst->input_list) {
116  input_pairs = radius_list(request, inst->input_list);
117  if (!input_pairs) {
118  REDEBUG("Failed to find input pairs for xlat");
119  return -1;
120  }
121  }
122 
123  /*
124  * This function does it's own xlat of the input program
125  * to execute.
126  */
127  result = radius_exec_program(request, *out, outlen, NULL, request, fmt, input_pairs ? *input_pairs : NULL,
128  inst->wait, inst->shell_escape, inst->timeout);
129  if (result != 0) return -1;
130 
131  for (p = *out; *p != '\0'; p++) {
132  if (*p < ' ') *p = ' ';
133  }
134 
135  return strlen(*out);
136 }

138 /*
139  * Do any per-module initialization that is separate to each
140  * configured instance of the module. e.g. set up connections
141  * to external databases, read configuration files, set up
142  * dictionary entries, etc.
143  *
144  * If configuration information is given in the config section
145  * that must be referenced in later calls, store a handle to it
146  * in *instance otherwise put a null pointer there.
147  */
148 static int mod_bootstrap(void *instance, CONF_SECTION *conf)
149 {
150  char const *p;
151  rlm_exec_t *inst = instance;
152 
153  inst->name = cf_section_name2(conf);
154  if (!inst->name) {
155  inst->name = cf_section_name1(conf);
156  }
157 
158  xlat_register_legacy(inst, inst->name, exec_xlat, rlm_exec_shell_escape, NULL, 0, XLAT_DEFAULT_BUF_LEN);
159 
160  if (inst->input) {
161  p = inst->input;
162  p += radius_list_name(&inst->input_list, p, PAIR_LIST_UNKNOWN);
163  if ((inst->input_list == PAIR_LIST_UNKNOWN) || (*p != '\0')) {
164  cf_log_err(conf, "Invalid input list '%s'", inst->input);
165  return -1;
166  }
167  }
168 
169  if (inst->output) {
170  p = inst->output;
171  p += radius_list_name(&inst->output_list, p, PAIR_LIST_UNKNOWN);
172  if ((inst->output_list == PAIR_LIST_UNKNOWN) || (*p != '\0')) {
173  cf_log_err(conf, "Invalid output list '%s'", inst->output);
174  return -1;
175  }
176  }
177 
178  /*
179  * Sanity check the config. If we're told to NOT wait,
180  * then the output pairs must not be defined.
181  */
182  if (!inst->wait && (inst->output != NULL)) {
183  cf_log_err(conf, "Cannot read output pairs if wait = no");
184  return -1;
185  }
186 
187  if (inst->timeout_is_set || !inst->timeout) {
188  /*
189  * Pick the shorter one
190  */
191  inst->timeout = main_config->max_request_time > fr_time_delta_from_sec(EXEC_TIMEOUT) ?
192  fr_time_delta_from_sec(EXEC_TIMEOUT):
193  main_config->max_request_time;
194  }
195  else {
196  if (inst->timeout < fr_time_delta_from_sec(1)) {
197  cf_log_err(conf, "Timeout '%pVs' is too small (minimum: 1s)", fr_box_time_delta(inst->timeout));
198  return -1;
199  }
200 
201  /*
202  * Blocking a request longer than max_request_time isn't going to help anyone.
203  */
204  if (inst->timeout > main_config->max_request_time) {
205  cf_log_err(conf, "Timeout '%pVs' is too large (maximum: %pVs)",
206  fr_box_time_delta(inst->timeout), fr_box_time_delta(main_config->max_request_time));
207  return -1;
208  }
209  }
210 
211  return 0;
212 }

215 /** Instantiate the module
216  *
217  * Creates a new instance of the module reading parameters from a configuration section.
218  *
219  * @param conf to parse.
220  * @param instance configuration data.
221  * @return
222  * - 0 on success.
223  * - < 0 on failure.
224  */
225 static int mod_instantiate(void *instance, CONF_SECTION *conf)
226 {
227  rlm_exec_t *inst = instance;
228  ssize_t slen;
229 
230  if (!inst->program) return 0;
231 
232  slen = tmpl_afrom_substr(inst, &inst->tmpl, &FR_SBUFF_IN(inst->program, strlen(inst->program)),
233  T_BACK_QUOTED_STRING, NULL,
234  &(tmpl_rules_t) {
235  .allow_foreign = true,
236  .allow_unresolved = false,
237  .allow_unknown = false
238  });
239  if (slen <= 0) {
240  char *spaces, *text;
241 
242  fr_canonicalize_error(inst, &spaces, &text, slen, inst->program);
243 
244  cf_log_err(conf, "%s", text);
245  cf_log_perr(conf, "%s^", spaces);
246 
247  talloc_free(spaces);
248  talloc_free(text);
249  return -1;
250  }
251 
252  return 0;
253 }

306 /** Process the exit code returned by one of the exec functions
307  *
308  * @param request Current request.
309  * @param box Output string from exec call.
310  * @param status code returned by exec call.
311  * @return One of the RLM_MODULE_* values.
312  */
313 static rlm_rcode_t rlm_exec_status2rcode(request_t *request, fr_value_box_t *box, int status)
314 {
315  rlm_rcode_t rcode;
316 
317  if (status < 0) return RLM_MODULE_FAIL;
318 
319  /*
320  * Exec'd programs are meant to return exit statuses that correspond
321  * to the standard RLM_MODULE_* + 1.
322  *
323  * This frees up 0, for success where it'd normally be reject.
324  */
325  if (status == 0) {
326  RDEBUG("Program executed successfully");
327 
328  return RLM_MODULE_OK;
329  }
330 
331  if (status > 9) {
332  REDEBUG("Program returned invalid code (greater than max rcode) (%i > 9): %pV",
333  status, box);
334  goto fail;
335  }
336 
337  rcode = status2rcode[status];
338 
339  if (rcode == RLM_MODULE_FAIL) {
340  fail:
341 
342  if (box) log_module_failure_msg(request, "%pV", box);
343 
344  return RLM_MODULE_FAIL;
345  }
346 
347  return rcode;
348 }

350 static unlang_action_t mod_exec_wait_resume(rlm_rcode_t *p_result, module_ctx_t const *mctx,
351  request_t *request, void *rctx)
352 {
353  int status;
354  rlm_exec_ctx_t *m = talloc_get_type_abort(rctx, rlm_exec_ctx_t);
355  rlm_exec_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_exec_t);
356 
357  if (inst->output && m->box) {
358  TALLOC_CTX *ctx;
359  fr_pair_t *vps, **output_pairs;
360 
361  RDEBUG("EXEC GOT -- %pV", m->box);
362 
363  output_pairs = radius_list(request, inst->output_list);
364  fr_assert(output_pairs != NULL);
365 
366  ctx = radius_list_ctx(request, inst->output_list);
367 
368  vps = fr_pair_list_afrom_box(ctx, request->dict, m->box);
369  if (vps) fr_pair_list_move(output_pairs, &vps);
370 
371  m->box = NULL; /* has been consumed */
372  }
373 
374  status = m->status;
375 
376  if (status < 0) {
377  REDEBUG("Program exited with signal %d", -status);
378  RETURN_MODULE_FAIL;
379  }
380 
381  /*
382  * The status rcodes aren't quite the same as the rcode
383  * enumeration.
384  */
385  RETURN_MODULE_RCODE(rlm_exec_status2rcode(request, m->box, status));
386 }

388 /*
389  * Dispatch an async exec method
390  */
391 static unlang_action_t CC_HINT(nonnull) mod_exec_dispatch(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
392 {
393  rlm_exec_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_exec_t);
394  rlm_exec_ctx_t *m;
395  fr_pair_t *env_pairs = NULL;
396  TALLOC_CTX *ctx;
397 
398  if (!inst->tmpl) {
399  RDEBUG("This module requires 'program' to be set.");
400  RETURN_MODULE_FAIL;
401  }
402 
403  /*
404  * Get frame-local talloc ctx
405  */
406  ctx = unlang_interpret_frame_talloc_ctx(request);
407 
408  /*
409  * Do the asynchronous xlat expansion.
410  */
411  if (!inst->wait) {
412  fr_value_box_t *box;
413 
414  MEM(box = talloc_zero(ctx, fr_value_box_t));
415 
416  return unlang_module_yield_to_xlat(request, &box, request, tmpl_xlat(inst->tmpl), mod_exec_nowait_resume, NULL, box);
417  }
418 
419  /*
420  * Decide what input/output the program takes.
421  */
422  if (inst->input) {
423  fr_pair_t **input_pairs;
424 
425  input_pairs = radius_list(request, inst->input_list);
426  if (!input_pairs) RETURN_MODULE_INVALID;
427 
428  env_pairs = *input_pairs;
429  }
430 
431  if (inst->output) {
432  if (!radius_list(request, inst->output_list)) {
433  RETURN_MODULE_INVALID;
434  }
435  }
436 
437  m = talloc_zero(ctx, rlm_exec_ctx_t);
438 
439  return unlang_module_yield_to_tmpl(m, &m->box, &m->status, request, inst->tmpl, env_pairs, mod_exec_wait_resume, NULL, m);
440 }
