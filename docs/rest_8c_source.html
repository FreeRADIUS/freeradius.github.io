291  * @param[out] out Char buffer to write encoded data to.

300 {
301  rlm_rest_request_t *ctx = userdata;

329 /** Encodes fr_pair_t linked list in POST format
330  *
331  * This is a stream function matching the rest_read_t prototype. Multiple
332  * successive calls will return additional encoded fr_pair_ts.
333  * Only complete attribute headers @verbatim '<name>=' @endverbatim and values
334  * will be written to the ptr buffer.
335  *
336  * POST request format is:
337  * @verbatim <attribute0>=<value0>&<attribute1>=<value1>&<attributeN>=<valueN>@endverbatim
338  *
339  * All attributes and values are url encoded. There is currently no support for
340  * nested attributes, or attribute qualifiers.
341  *
342  * Nested attributes may be added in the future using
343  * @verbatim <attribute-outer>:<attribute-inner>@endverbatim
344  * to denotate nesting.
345  *
346  * Requires libcurl for url encoding.
347  *
348  * @see rest_decode_post
349  *
350  * @param[out] out Char buffer to write encoded data to.
351  * @param[in] size Multiply by nmemb to get the length of ptr.
352  * @param[in] nmemb Multiply by size to get the length of ptr.
353  * @param[in] userdata rlm_rest_request_t to keep encoding state between calls.
354  * @return
355  * - Length of data (including NULL) written to ptr.
356  * - 0 if no more data to write.
357  */
358 static size_t rest_encode_post(void *out, size_t size, size_t nmemb, void *userdata)
359 {
360  rlm_rest_request_t *ctx = userdata;
361  request_t *request = ctx->request; /* Used by RDEBUG */
362  fr_pair_t *vp;
363 
364  char *p = out; /* Position in buffer */
365  char *encoded = p; /* Position in buffer of last fully encoded attribute or value */
366  char *escaped; /* Pointer to current URL escaped data */
367 
368  size_t len = 0;
369  ssize_t slen;
370  size_t freespace = (size * nmemb) - 1;
371 
372  /* Allow manual chunking */
373  if ((ctx->chunk) && (ctx->chunk <= freespace)) freespace = (ctx->chunk - 1);
374 
375  if (ctx->state == READ_STATE_END) return 0;
376 
377  /* Post data requires no headers */
378  if (ctx->state == READ_STATE_INIT) ctx->state = READ_STATE_ATTR_BEGIN;
379 
380  while (freespace > 0) {
381  vp = fr_cursor_current(&ctx->cursor);
382  if (!vp) {
383  ctx->state = READ_STATE_END;
384 
385  break;
386  }
387 
388  RDEBUG2("Encoding attribute \"%s\"", vp->da->name);
389 
390  if (ctx->state == READ_STATE_ATTR_BEGIN) {
391  escaped = curl_escape(vp->da->name, strlen(vp->da->name));
392  if (!escaped) {
393  REDEBUG("Failed escaping string \"%s\"", vp->da->name);
394  return 0;
395  }
396 
397  len = strlen(escaped);
398  if (freespace < (1 + len)) {
399  curl_free(escaped);
400  /*
401  * Cleanup for error conditions
402  */
403  no_space:
404  *encoded = '\0';
405 
406  len = encoded - (char *)out;
407 
408  RDEBUG3("POST Data: %pV", fr_box_strvalue_len(out, len));
409 
410  /*
411  * The buffer wasn't big enough to encode a single attribute chunk.
412  */
413  if (len == 0) {
414  REDEBUG("Failed encoding attribute");
415  } else {
416  RDEBUG3("Returning %zd bytes of POST data "
417  "(buffer full or chunk exceeded)", len);
418  }
419 
420  return len;
421  }
422 
423  len = sprintf(p, "%s=", escaped);
424  curl_free(escaped);
425  p += len;
426  freespace -= len;
427 
428  /*
429  * We wrote the attribute header, record progress.
430  */
431  encoded = p;
432  ctx->state = READ_STATE_ATTR_CONT;
433  }
434 
435  /*
436  * Write out single attribute string.
437  */
438  slen = fr_pair_print_value_quoted(&FR_SBUFF_OUT(p, freespace), vp, T_BARE_WORD);
439  if (slen < 0) return 0;
440 
441  RINDENT();
442  RDEBUG3("Length : %zd", (size_t)slen);
443  REXDENT();
444  if (slen > 0) {
445  escaped = curl_escape(p, (size_t)slen);
446  if (!escaped) {
447  REDEBUG("Failed escaping string \"%s\"", vp->da->name);
448  return 0;
449  }
450  len = strlen(escaped);
451 
452  if (freespace < len) {
453  curl_free(escaped);
454  goto no_space;
455  }
456 
457  len = strlcpy(p, escaped, len + 1);
458 
459  curl_free(escaped);
460 
461  RINDENT();
462  RDEBUG3("Value : %s", p);
463  REXDENT();
464 
465  p += len;
466  freespace -= len;
467  }
468 
469  /*
470  * there are more attributes, insert a separator
471  */
472  if (fr_cursor_next(&ctx->cursor)) {
473  if (freespace < 1) goto no_space;
474  *p++ = '&';
475  freespace--;
476  }
477 
478  /*
479  * We wrote one full attribute value pair, record progress.
480  */
481  encoded = p;
482 
483  ctx->state = READ_STATE_ATTR_BEGIN;
484  }
485 
486  *p = '\0';
487 
488  len = p - (char *)out;
489 
490  RDEBUG3("POST Data: %s", (char *)out);
491  RDEBUG3("Returning %zd bytes of POST data", len);
492 
493  return len;
494 }

572 /** Emulates successive libcurl calls to an encoding function
573  *
574  * This function is used when the request will be sent to the HTTP server as one
575  * contiguous entity. A buffer of REST_BODY_ALLOC_CHUNK bytes is allocated and passed
576  * to the stream encoding function.
577  *
578  * If the stream function does not return 0, a new buffer is allocated which is
579  * the size of the previous buffer + REST_BODY_ALLOC_CHUNK bytes, the data from the
580  * previous buffer is copied, and freed, and another call is made to the stream
581  * function, passing a pointer into the new buffer at the end of the previously
582  * written data.
583  *
584  * This process continues until the stream function signals (by returning 0)
585  * that it has no more data to write.
586  *
587  * @param[out] out where the pointer to the alloced buffer should
588  * be written.
589  * @param[in] inst of rlm_rest.
590  * @param[in] func Stream function.
591  * @param[in] limit Maximum buffer size to alloc.
592  * @param[in] userdata rlm_rest_request_t to keep encoding state between calls to
593  * stream function.
594  * @return
595  * - Length of the data written to the buffer (excluding NULL).
596  * - -1 if alloc >= limit.
597  */
598 static ssize_t rest_request_encode_wrapper(char **out, UNUSED rlm_rest_t const *inst,
599  rest_read_t func, size_t limit, void *userdata)
600 {
601  char *buff = NULL;
602  size_t alloc = REST_BODY_ALLOC_CHUNK; /* Size of buffer to alloc */
603  size_t used = 0; /* Size of data written */
604  size_t len = 0;
605 
606  buff = talloc_array(NULL, char, alloc);
607  for (;;) {
608  len = func(buff + used, alloc - used, 1, userdata);
609  used += len;
610  if (!len) {
611  *out = buff;
612  return used;
613  }
614 
615  alloc = alloc * 2;
616  if (alloc > limit) break;
617 
618  MEM(buff = talloc_realloc(NULL, buff, char, alloc));
619  };
620 
621  talloc_free(buff);
622 
623  return -1;
624 }

1225 /** Processes incoming HTTP header data from libcurl.
1226  *
1227  * Processes the status line, and Content-Type headers from the incoming HTTP
1228  * response.
1229  *
1230  * Matches prototype for CURLOPT_HEADERFUNCTION, and will be called directly
1231  * by libcurl.
1232  *
1233  * @param[in] in Char buffer where inbound header data is written.
1234  * @param[in] size Multiply by nmemb to get the length of ptr.
1235  * @param[in] nmemb Multiply by size to get the length of ptr.
1236  * @param[in] userdata rlm_rest_response_t to keep parsing state between calls.
1237  * @return
1238  * - Length of data processed.
1239  * - 0 on error.
1240  */
1241 static size_t rest_response_header(void *in, size_t size, size_t nmemb, void *userdata)
1242 {
1243  rlm_rest_response_t *ctx = userdata;
1244  request_t *request = ctx->request; /* Used by RDEBUG */
1245 
1246  char const *start = (char *)in, *p = start, *end = p + (size * nmemb);
1247  char *q;
1248  size_t len;
1249 
1250  http_body_type_t type;
1251 
1252 #ifndef NDEBUG
1253  if (ctx->instance->fail_header_decode) {
1254  REDEBUG("Forcing header decode failure");
1255  return 0;
1256  }
1257 #endif
1258 
1259  /*
1260  * This seems to be curl's indication there are no more header lines.
1261  */
1262  if (((end - p) == 2) && ((p[0] == '\r') && (p[1] == '\n'))) {
1263  /*
1264  * If we got a 100 Continue, we need to send additional payload data.
1265  * reset the state to WRITE_STATE_INIT, so that when were called again
1266  * we overwrite previous header data with that from the proper header.
1267  */
1268  if (ctx->code == 100) {
1269  RDEBUG2("Continuing...");
1270  ctx->state = WRITE_STATE_INIT;
1271  }
1272 
1273  return (end - start);
1274  }
1275 
1276  switch (ctx->state) {
1277  case WRITE_STATE_INIT:
1278  RDEBUG2("Processing response header");
1279 
1280  /*
1281  * HTTP/<version> <reason_code>[ <reason_phrase>]\r\n
1282  *
1283  * "HTTP/1.1 " (8) + "100 " (4) + "\r\n" (2) = 14
1284  * "HTTP/2 " (8) + "100 " (4) + "\r\n" (2) = 12
1285  */
1286  if ((end - p) < 12) {
1287  REDEBUG("Malformed HTTP header: Status line too short");
1288  malformed:
1289  REDEBUG("Received %zu bytes of invalid header data: %pV",
1290  (end - start), fr_box_strvalue_len(in, (end - start)));
1291  ctx->code = 0;
1292 
1293  /*
1294  * Indicate we parsed the entire line, otherwise
1295  * bad things seem to happen internally with
1296  * libcurl when we try and use it with asynchronous
1297  * I/O handlers.
1298  */
1299  return (end - start);
1300  }
1301  /*
1302  * Check start of header matches...
1303  */
1304  if (strncasecmp("HTTP/", p, 5) != 0) {
1305  REDEBUG("Malformed HTTP header: Missing HTTP version");
1306  goto malformed;
1307  }
1308  p += 5;
1309 
1310  /*
1311  * Skip the version field, next space should mark start of reason_code.
1312  */
1313  q = memchr(p, ' ', (end - p));
1314  if (!q) {
1315  REDEBUG("Malformed HTTP header: Missing reason code");
1316  goto malformed;
1317  }
1318 
1319  p = q;
1320 
1321  /*
1322  * Process reason_code.
1323  *
1324  * " 100" (4) + "\r\n" (2) = 6
1325  */
1326  if ((end - p) < 6) {
1327  REDEBUG("Malformed HTTP header: Reason code too short");
1328  goto malformed;
1329  }
1330  p++;
1331 
1332  /*
1333  * "xxx( |\r)" status code and terminator.
1334  */
1335  if (!isdigit(p[0]) || !isdigit(p[1]) || !isdigit(p[2]) || !((p[3] == ' ') || (p[3] == '\r'))) {
1336  REDEBUG("Malformed HTTP header: Reason code malformed. "
1337  "Expected three digits then space or end of header, got \"%pV\"",
1338  fr_box_strvalue_len(p, 4));
1339  goto malformed;
1340  }
1341 
1342  /*
1343  * Convert status code into an integer value
1344  */
1345  q = NULL;
1346  ctx->code = (int)strtoul(p, &q, 10);
1347  fr_assert(q == (p + 3)); /* We check this above */
1348  p = q;
1349 
1350  /*
1351  * Process reason_phrase (if present).
1352  */
1353  RINDENT();
1354  if (*p == ' ') {
1355  q = memchr(p, '\r', (end - p));
1356  if (!q) goto malformed;
1357  RDEBUG2("Status : %i (%pV)", ctx->code, fr_box_strvalue_len(p, q - p));
1358  } else {
1359  RDEBUG2("Status : %i", ctx->code);
1360  }
1361  REXDENT();
1362 
1363  ctx->state = WRITE_STATE_PARSE_HEADERS;
1364 
1365  break;
1366 
1367  case WRITE_STATE_PARSE_HEADERS:
1368  if (((end - p) >= 14) &&
1369  (strncasecmp("Content-Type: ", p, 14) == 0)) {
1370  p += 14;
1371 
1372  /*
1373  * Check to see if there's a parameter separator.
1374  */
1375  q = memchr(p, ';', (end - p));
1376 
1377  /*
1378  * If there's not, find the end of this header.
1379  */
1380  if (!q) q = memchr(p, '\r', (end - p));
1381 
1382  len = (size_t)(!q ? (end - p) : (q - p));
1383  type = fr_table_value_by_substr(http_content_type_table, p, len, REST_HTTP_BODY_UNKNOWN);
1384 
1385  RINDENT();
1386  RDEBUG2("Type : %s (%pV)", fr_table_str_by_value(http_body_type_table, type, "<INVALID>"),
1387  fr_box_strvalue_len(p, len));
1388  REXDENT();
1389 
1390  /*
1391  * Assume the force_to value has already been validated.
1392  */
1393  if (ctx->force_to != REST_HTTP_BODY_UNKNOWN) {
1394  if (ctx->force_to != ctx->type) {
1395  RDEBUG3("Forcing body type to \"%s\"",
1396  fr_table_str_by_value(http_body_type_table, ctx->force_to, "<INVALID>"));
1397  ctx->type = ctx->force_to;
1398  }
1399  /*
1400  * Figure out if the type is supported by one of the decoders.
1401  */
1402  } else {
1403  ctx->type = http_body_type_supported[type];
1404  switch (ctx->type) {
1405  case REST_HTTP_BODY_UNKNOWN:
1406  RWDEBUG("Couldn't determine type, using the request's type \"%s\".",
1407  fr_table_str_by_value(http_body_type_table, type, "<INVALID>"));
1408  break;
1409 
1410  case REST_HTTP_BODY_UNSUPPORTED:
1411  REDEBUG("Type \"%s\" is currently unsupported",
1412  fr_table_str_by_value(http_body_type_table, type, "<INVALID>"));
1413  break;
1414 
1415  case REST_HTTP_BODY_UNAVAILABLE:
1416  REDEBUG("Type \"%s\" is unavailable, please rebuild this module with the required "
1417  "library", fr_table_str_by_value(http_body_type_table, type, "<INVALID>"));
1418  break;
1419 
1420  case REST_HTTP_BODY_INVALID:
1421  REDEBUG("Type \"%s\" is not a valid web API data markup format",
1422  fr_table_str_by_value(http_body_type_table, type, "<INVALID>"));
1423  break;
1424 
1425  /* supported type */
1426  default:
1427  break;
1428  }
1429  }
1430  }
1431  break;
1432 
1433  default:
1434  break;
1435  }
1436

1438 }

1440 /** Processes incoming HTTP body data from libcurl.
1441  *
1442  * Writes incoming body data to an intermediary buffer for later parsing by
1443  * one of the decode functions.
1444  *
1445  * @param[in] in Char buffer where inbound header data is written
1446  * @param[in] size Multiply by nmemb to get the length of ptr.
1447  * @param[in] nmemb Multiply by size to get the length of ptr.
1448  * @param[in] userdata rlm_rest_response_t to keep parsing state between calls.
1449  * @return
1450  * - Length of data processed.
1451  * - 0 on error.
1452  */
1453 static size_t rest_response_body(void *in, size_t size, size_t nmemb, void *userdata)
1454 {
1455  rlm_rest_response_t *ctx = userdata;
1456  request_t *request = ctx->request; /* Used by RDEBUG */
1457 
1458  char const *start = in, *p = start, *end = p + (size * nmemb);
1459  char *q;
1460 
1461  size_t needed;
1462 
1463  if (start == end) return 0; /* Nothing to process */
1464 
1465 #ifndef NDEBUG
1466  if (ctx->instance->fail_body_decode) {
1467  REDEBUG("Forcing body read failure");
1468  return 0;
1469  }
1470 #endif
1471 
1472  /*
1473  * Any post processing of headers should go here...
1474  */
1475  if (ctx->state == WRITE_STATE_PARSE_HEADERS) ctx->state = WRITE_STATE_PARSE_CONTENT;
1476 
1477  switch (ctx->type) {
1478  case REST_HTTP_BODY_UNSUPPORTED:
1479  case REST_HTTP_BODY_UNAVAILABLE:
1480  case REST_HTTP_BODY_INVALID:
1481  while ((q = memchr(p, '\n', (end - p)))) {
1482  REDEBUG("%pV", fr_box_strvalue_len(p, q - p));
1483  p = q + 1;
1484  }
1485 
1486  if (p != end) REDEBUG("%pV", fr_box_strvalue_len(p, end - p));
1487  break;
1488 
1489  case REST_HTTP_BODY_NONE:
1490  while ((q = memchr(p, '\n', (end - p)))) {
1491  RDEBUG3("%pV", fr_box_strvalue_len(p, q - p));
1492  p = q + 1;
1493  }
1494 
1495  if (p != end) RDEBUG3("%pV", fr_box_strvalue_len(p, end - p));
1496  break;
1497 
1498  default:
1499  {
1500  char *out_p;
1501 
1502  if ((ctx->section->max_body_in > 0) && ((ctx->used + (end - p)) > ctx->section->max_body_in)) {
1503  REDEBUG("Incoming data (%zu bytes) exceeds max_body_in (%zu bytes). "
1504  "Forcing body to type 'invalid'", ctx->used + (end - p), ctx->section->max_body_in);
1505  ctx->type = REST_HTTP_BODY_INVALID;
1506  TALLOC_FREE(ctx->buffer);
1507  break;
1508  }
1509 
1510  needed = ROUND_UP(ctx->used + (end - p), REST_BODY_ALLOC_CHUNK);
1511  if (needed > ctx->alloc) {
1512  MEM(ctx->buffer = talloc_bstr_realloc(NULL, ctx->buffer, needed));
1513  ctx->alloc = needed;
1514  }
1515 
1516  out_p = ctx->buffer + ctx->used;
1517  memcpy(out_p, p, (end - p));
1518  out_p += (end - p);
1519  *out_p = '\0';
1520  ctx->used += (end - p);
1521  }
1522  break;
1523  }
1524 
1525  return (end - start);
1526 }

1723

1725

1731  fr_cursor_t headers;
