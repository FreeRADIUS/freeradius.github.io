170 static int mod_instantiate(UNUSED rlm_sql_config_t const *config, void *instance, UNUSED CONF_SECTION *cs)
171 {
172  rlm_sql_mysql_t *inst = instance;
173  int warnings;
174 
175  warnings = fr_table_value_by_str(server_warnings_table, inst->warnings_str, -1);
176  if (warnings < 0) {
177  ERROR("Invalid warnings value \"%s\", must be yes, no, or auto", inst->warnings_str);
178  return -1;
179  }
180  inst->warnings = (rlm_sql_mysql_warnings)warnings;
181 
182 #if HAVE_TLS_VERIFY_OPTIONS
183  if (inst->tls_check_cert && !inst->tls_required) {
184  WARN("Implicitly setting tls_required = yes, as tls_check_cert = yes");
185  inst->tls_required = true;
186  }
187  if (inst->tls_check_cert_cn) {
188  if (!inst->tls_required) {
189  WARN("Implicitly setting tls_required = yes, as check_cert_cn = yes");
190  inst->tls_required = true;
191  }
192 
193  if (!inst->tls_check_cert) {
194  WARN("Implicitly setting check_cert = yes, as check_cert_cn = yes");
195  inst->tls_check_cert = true;
196  }
197  }
198 #endif
199  return 0;
200 }

220 static sql_rcode_t sql_socket_init(rlm_sql_handle_t *handle, rlm_sql_config_t *config, fr_time_delta_t timeout)
221 {
222  rlm_sql_mysql_conn_t *conn;
223  rlm_sql_mysql_t *inst = config->driver;
224  unsigned int connect_timeout = (unsigned int)fr_time_delta_to_sec(timeout);
225  unsigned long sql_flags;
226 
227  MEM(conn = handle->conn = talloc_zero(handle, rlm_sql_mysql_conn_t));
228  talloc_set_destructor(conn, _sql_socket_destructor);
229 
230  DEBUG("Starting connect to MySQL server");
231 
232  mysql_init(&(conn->db));
233 
234  /*
235  * If any of the TLS options are set, configure TLS
236  *
237  * According to MySQL docs this function always returns 0, so we won't
238  * know if ssl setup succeeded until mysql_real_connect is called below.
239  */
240  if (inst->tls_ca_file || inst->tls_ca_path ||
241  inst->tls_certificate_file || inst->tls_private_key_file) {
242  mysql_ssl_set(&(conn->db), inst->tls_private_key_file, inst->tls_certificate_file,
243  inst->tls_ca_file, inst->tls_ca_path, inst->tls_cipher);
244  }
245 
246 #if HAVE_TLS_OPTIONS
247  {
248  unsigned int ssl_mode = 0;
249  bool ssl_mode_isset = false;
250 
251  if (inst->tls_required) {
252  ssl_mode = SSL_MODE_REQUIRED;
253  ssl_mode_isset = true;
254  }
255 # if HAVE_TLS_VERIFY_OPTIONS
256  if (inst->tls_check_cert) {
257  ssl_mode = SSL_MODE_VERIFY_CA;
258  ssl_mode_isset = true;
259  }
260  if (inst->tls_check_cert_cn) {
261  ssl_mode = SSL_MODE_VERIFY_IDENTITY;
262  ssl_mode_isset = true;
263  }
264 # endif
265  if (ssl_mode_isset) mysql_options(&(conn->db), MYSQL_OPT_SSL_MODE, &ssl_mode);
266  }
267 #endif
268 
269 #if HAVE_CRL_OPTIONS
270  if (inst->tls_crl_file) mysql_options(&(conn->db), MYSQL_OPT_SSL_CRL, inst->tls_crl_file);
271  if (inst->tls_crl_path) mysql_options(&(conn->db), MYSQL_OPT_SSL_CRLPATH, inst->tls_crl_path);
272 #endif
273 
274  mysql_options(&(conn->db), MYSQL_READ_DEFAULT_GROUP, "freeradius");
275 
276  /*
277  * We need to know about connection errors, and are capable
278  * of reconnecting automatically.
279  */
280 #if MYSQL_VERSION_ID >= 50013
281  {
282  bool reconnect = 0;
283  mysql_options(&(conn->db), MYSQL_OPT_RECONNECT, &reconnect);
284  }
285 #endif
286 
287 #if (MYSQL_VERSION_ID >= 50000)
288  mysql_options(&(conn->db), MYSQL_OPT_CONNECT_TIMEOUT, &connect_timeout);
289 
290  if (config->query_timeout) {
291  unsigned int read_timeout = config->query_timeout;
292  unsigned int write_timeout = config->query_timeout;
293 
294  /*
295  * The timeout in seconds for each attempt to read from the server.
296  * There are retries if necessary, so the total effective timeout
297  * value is three times the option value.
298  */
299  if (config->query_timeout >= 3) read_timeout /= 3;
300 
301  /*
302  * The timeout in seconds for each attempt to write to the server.
303  * There is a retry if necessary, so the total effective timeout
304  * value is two times the option value.
305  */
306  if (config->query_timeout >= 2) write_timeout /= 2;
307 
308  /*
309  * Connect timeout is actually connect timeout (according to the
310  * docs) there are no automatic retries.
311  */
312  mysql_options(&(conn->db), MYSQL_OPT_READ_TIMEOUT, &read_timeout);
313  mysql_options(&(conn->db), MYSQL_OPT_WRITE_TIMEOUT, &write_timeout);
314  }
315 #endif
316 
317 #if (MYSQL_VERSION_ID >= 40100)
318  sql_flags = CLIENT_MULTI_RESULTS | CLIENT_FOUND_ROWS;
319 #else
320  sql_flags = CLIENT_FOUND_ROWS;
321 #endif
322 
323 #ifdef CLIENT_MULTI_STATEMENTS
324  sql_flags |= CLIENT_MULTI_STATEMENTS;
325 #endif
326  conn->sock = mysql_real_connect(&(conn->db),
327  config->sql_server,
328  config->sql_login,
329  config->sql_password,
330  config->sql_db,
331  config->sql_port,
332  NULL,
333  sql_flags);
334  if (!conn->sock) {
335  ERROR("Couldn't connect to MySQL server %s@%s:%s", config->sql_login,
336  config->sql_server, config->sql_db);
337  ERROR("MySQL error: %s", mysql_error(&conn->db));
338 
339  conn->sock = NULL;
340  return RLM_SQL_ERROR;
341  }
342 
343  DEBUG2("Connected to database '%s' on %s, server version %s, protocol version %i",
344  config->sql_db, mysql_get_host_info(conn->sock),
345  mysql_get_server_info(conn->sock), mysql_get_proto_info(conn->sock));
346 
347  return RLM_SQL_OK;
348 }

350 /** Analyse the last error that occurred on the socket, and determine an action
351  *
352  * @param server Socket from which to extract the server error. May be NULL.
353  * @param client_errno Error from the client.
354  * @return an action for #rlm_sql_t to take.
355  */
356 static sql_rcode_t sql_check_error(MYSQL *server, int client_errno)
357 {
358  int sql_errno = 0;
359 
360  /*
361  * The client and server error numbers are in the
362  * same numberspace.
363  */
364  if (server) sql_errno = mysql_errno(server);
365  if ((sql_errno == 0) && (client_errno != 0)) sql_errno = client_errno;
366 
367  if (sql_errno > 0) switch (sql_errno) {
368  case CR_SERVER_GONE_ERROR:
369  case CR_SERVER_LOST:
370  case -1:
371  return RLM_SQL_RECONNECT;
372 
373  case CR_OUT_OF_MEMORY:
374  case CR_COMMANDS_OUT_OF_SYNC:
375  case CR_UNKNOWN_ERROR:
376  default:
377  return RLM_SQL_ERROR;
378 
379  /*
380  * Constraints errors that signify a duplicate, or that we might
381  * want to try an alternative query.
382  *
383  * Error constants not found in the 3.23/4.0/4.1 manual page
384  * are checked for.
385  * Other error constants should always be available.
386  */
387  case ER_DUP_UNIQUE: /* Can't write, because of unique constraint, to table '%s'. */
388  case ER_DUP_KEY: /* Can't write; duplicate key in table '%s' */
389 
390  case ER_DUP_ENTRY: /* Duplicate entry '%s' for key %d. */
391  case ER_NO_REFERENCED_ROW: /* Cannot add or update a child row: a foreign key constraint fails */
392  case ER_ROW_IS_REFERENCED: /* Cannot delete or update a parent row: a foreign key constraint fails */
393 #ifdef ER_FOREIGN_DUPLICATE_KEY
394  case ER_FOREIGN_DUPLICATE_KEY: /* Upholding foreign key constraints for table '%s', entry '%s', key %d would lead to a duplicate entry. */
395 #endif
396 #ifdef ER_DUP_ENTRY_WITH_KEY_NAME
397  case ER_DUP_ENTRY_WITH_KEY_NAME: /* Duplicate entry '%s' for key '%s' */
398 #endif
399 #ifdef ER_NO_REFERENCED_ROW_2
400  case ER_NO_REFERENCED_ROW_2:
401 #endif
402 #ifdef ER_ROW_IS_REFERENCED_2
403  case ER_ROW_IS_REFERENCED_2:
404 #endif
405  return RLM_SQL_ALT_QUERY;
406 
407  /*
408  * Constraints errors that signify an invalid query
409  * that can never succeed.
410  */
411  case ER_BAD_NULL_ERROR: /* Column '%s' cannot be null */
412  case ER_NON_UNIQ_ERROR: /* Column '%s' in %s is ambiguous */
413  return RLM_SQL_QUERY_INVALID;
414 
415  }
416 
417  return RLM_SQL_OK;
418 }

420 static sql_rcode_t sql_query(rlm_sql_handle_t *handle, UNUSED rlm_sql_config_t *config, char const *query)
421 {
422  rlm_sql_mysql_conn_t *conn = handle->conn;
423  sql_rcode_t rcode;
424  char const *info;
425 
426  if (!conn->sock) {
427  ERROR("Socket not connected");
428  return RLM_SQL_RECONNECT;
429  }
430 
431  mysql_query(conn->sock, query);
432  rcode = sql_check_error(conn->sock, 0);
433  if (rcode != RLM_SQL_OK) {
434  return rcode;
435  }
436 
437  /* Only returns non-null string for INSERTS */
438  info = mysql_info(conn->sock);
439  if (info) DEBUG2("%s", info);
440 
441  return RLM_SQL_OK;
442 }

444 static sql_rcode_t sql_store_result(rlm_sql_handle_t *handle, UNUSED rlm_sql_config_t *config)
445 {
446  rlm_sql_mysql_conn_t *conn = handle->conn;
447  sql_rcode_t rcode;
448  int ret;
449 
450  if (!conn->sock) {
451  ERROR("Socket not connected");
452  return RLM_SQL_RECONNECT;
453  }
454 
455 retry_store_result:
456  conn->result = mysql_store_result(conn->sock);
457  if (!conn->result) {
458  rcode = sql_check_error(conn->sock, 0);
459  if (rcode != RLM_SQL_OK) return rcode;
460 #if (MYSQL_VERSION_ID >= 40100)
461  ret = mysql_next_result(conn->sock);
462  if (ret == 0) {
463  /* there are more results */
464  goto retry_store_result;
465  } else if (ret > 0) return sql_check_error(NULL, ret);
466  /* ret == -1 signals no more results */
467 #endif
468  }
469  return RLM_SQL_OK;
470 }

526 static sql_rcode_t sql_fields(char const **out[], rlm_sql_handle_t *handle, rlm_sql_config_t *config)
527 {
528  rlm_sql_mysql_conn_t *conn = handle->conn;
529 
530  unsigned int fields, i;
531  MYSQL_FIELD *field_info;
532  char const **names;
533 
534  /*
535  * Use our internal function to abstract out the API call.
536  * Different versions of SQL use different functions,
537  * and some don't like NULL pointers.
538  */
539  fields = sql_num_fields(handle, config);
540  if (fields == 0) return RLM_SQL_ERROR;
541 
542  /*
543  * https://bugs.mysql.com/bug.php?id=32318
544  * Hints that we don't have to free field_info.
545  */
546  field_info = mysql_fetch_fields(conn->result);
547  if (!field_info) return RLM_SQL_ERROR;
548 
549  MEM(names = talloc_array(handle, char const *, fields));
550 
551  for (i = 0; i < fields; i++) names[i] = field_info[i].name;
552  *out = names;
553 
554  return RLM_SQL_OK;
555 }

557 static sql_rcode_t sql_fetch_row(rlm_sql_row_t *out, rlm_sql_handle_t *handle, rlm_sql_config_t *config)
558 {
559  rlm_sql_mysql_conn_t *conn = handle->conn;
560  sql_rcode_t rcode;
561  MYSQL_ROW row;
562  int ret;
563  unsigned int num_fields, i;
564  unsigned long *field_lens;
565 
566  *out = NULL;
567 
568  /*
569  * Check pointer before de-referencing it.
570  */
571  if (!conn->result) return RLM_SQL_RECONNECT;
572 
573  TALLOC_FREE(handle->row); /* Clear previous row set */
574 
575 retry_fetch_row:
576  row = mysql_fetch_row(conn->result);
577  if (!row) {
578  rcode = sql_check_error(conn->sock, 0);
579  if (rcode != RLM_SQL_OK) return rcode;
580 
581 #if (MYSQL_VERSION_ID >= 40100)
582  sql_free_result(handle, config);
583 
584  ret = mysql_next_result(conn->sock);
585  if (ret == 0) {
586  /* there are more results */
587  if ((sql_store_result(handle, config) == 0) && (conn->result != NULL)) {
588  goto retry_fetch_row;
589  }
590  } else if (ret > 0) return sql_check_error(NULL, ret);
591  /* If ret is -1 then there are no more rows */
592 #endif
593  return RLM_SQL_NO_MORE_ROWS;
594  }
595 
596  num_fields = mysql_num_fields(conn->result);
597  if (!num_fields) return RLM_SQL_NO_MORE_ROWS;
598 
599  field_lens = mysql_fetch_lengths(conn->result);
600 
601  MEM(*out = handle->row = talloc_zero_array(handle, char *, num_fields + 1));
602  for (i = 0; i < num_fields; i++) {
603  MEM(handle->row[i] = talloc_bstrndup(handle->row, row[i], field_lens[i]));
604  }
605 
606  return RLM_SQL_OK;
607 }

623  *
624  * MySQL stores a limited number of warnings associated with the last query
625  * executed. These can be very useful in diagnosing issues, or in some cases
626  * working around bugs in MySQL which causes it to return the wrong error.
627  *
628  * @note Caller should free any memory allocated in ctx (talloc_free_children()).
629  *
630  * @param ctx to allocate temporary error buffers in.
631  * @param out Array of sql_log_entrys to fill.
632  * @param outlen Length of out array.
633  * @param handle rlm_sql connection handle.
634  * @param config rlm_sql config.
635  * @return
636  * - Number of errors written to the #sql_log_entry_t array.
637  * - -1 on failure.
638  */
639 static size_t sql_warnings(TALLOC_CTX *ctx, sql_log_entry_t out[], size_t outlen,
640  rlm_sql_handle_t *handle, UNUSED rlm_sql_config_t *config)
641 {
642  rlm_sql_mysql_conn_t *conn = handle->conn;
643 
644  MYSQL_RES *result;
645  MYSQL_ROW row;
646  unsigned int num_fields;
647  size_t i = 0;
648 
649  if (outlen == 0) return 0;
650 
651  /*
652  * Retrieve any warnings associated with the previous query
653  * that were left lingering on the server.
654  */
655  if (mysql_query(conn->sock, "SHOW WARNINGS") != 0) return -1;
656  result = mysql_store_result(conn->sock);
657  if (!result) return -1;
658 
659  /*
660  * Fields should be [0] = Level, [1] = Code, [2] = Message
661  */
662  num_fields = mysql_field_count(conn->sock);
663  if (num_fields < 3) {
664  WARN("Failed retrieving warnings, expected 3 fields got %u", num_fields);
665  mysql_free_result(result);
666 
667  return -1;
668  }
669 
670  while ((row = mysql_fetch_row(result))) {
671  char *msg = NULL;
672  fr_log_type_t type;
673 
674  /*
675  * Translate the MySQL log level into our internal
676  * log levels, so they get colourised correctly.
677  */
678  if (strcasecmp(row[0], "warning") == 0) type = L_WARN;
679  else if (strcasecmp(row[0], "note") == 0) type = L_DBG;
680  else type = L_ERR;
681 
682  msg = talloc_typed_asprintf(ctx, "%s: %s", row[1], row[2]);
683  out[i].type = type;
684  out[i].msg = msg;
685  if (++i == outlen) break;
686  }
687 
688  mysql_free_result(result);
689 
690  return i;
691 }

693 /** Retrieves any errors associated with the connection handle
694  *
695  * @note Caller should free any memory allocated in ctx (talloc_free_children()).
696  *
697  * @param ctx to allocate temporary error buffers in.
698  * @param out Array of sql_log_entrys to fill.
699  * @param outlen Length of out array.
700  * @param handle rlm_sql connection handle.
701  * @param config rlm_sql config.
702  * @return number of errors written to the #sql_log_entry_t array.
703  */
704 static size_t sql_error(TALLOC_CTX *ctx, sql_log_entry_t out[], size_t outlen,
705  rlm_sql_handle_t *handle, rlm_sql_config_t *config)
706 {
707  rlm_sql_mysql_conn_t *conn = handle->conn;
708  rlm_sql_mysql_t *inst = config->driver;
709  char const *error;
710  size_t i = 0;
711 
712  fr_assert(conn && conn->sock);
713  fr_assert(outlen > 0);
714 
715  error = mysql_error(conn->sock);
716 
717  /*
718  * Grab the error now in case it gets cleared on the next operation.
719  */
720  if (error && (error[0] != '\0')) {
721  error = talloc_typed_asprintf(ctx, "ERROR %u (%s): %s", mysql_errno(conn->sock), error,
722  mysql_sqlstate(conn->sock));
723  }
724 
725  /*
726  * Don't attempt to get errors from the server, if the last error
727  * was that the server was unavailable.
728  */
729  if ((outlen > 1) && (sql_check_error(conn->sock, 0) != RLM_SQL_RECONNECT)) {
730  size_t ret;
731  unsigned int msgs;
732 
733  switch (inst->warnings) {
734  case SERVER_WARNINGS_AUTO:
735  /*
736  * Check to see if any warnings can be retrieved from the server.
737  */
738  msgs = mysql_warning_count(conn->sock);
739  if (msgs == 0) {
740  DEBUG3("No additional diagnostic info on server");
741  break;
742  }
743 
744  FALL_THROUGH;
745  case SERVER_WARNINGS_YES:
746  ret = sql_warnings(ctx, out, outlen - 1, handle, config);
747  if (ret > 0) i += ret;
748  break;
749 
750  case SERVER_WARNINGS_NO:
751  break;
752 
753  default:
754  fr_assert(0);
755  }
756  }
757 
758  if (error) {
759  out[i].type = L_ERR;
760  out[i].msg = error;
761  }
762  i++;
763 
764  return i;
765 }

767 /** Finish query
768  *
769  * As a single SQL statement may return multiple results
770  * sets, (for example stored procedures) it is necessary to check
771  * whether more results exist and process them in turn if so.
772  *
773  */
774 static sql_rcode_t sql_finish_query(rlm_sql_handle_t *handle, rlm_sql_config_t *config)
775 {
776 #if (MYSQL_VERSION_ID >= 40100)
777  rlm_sql_mysql_conn_t *conn = handle->conn;
778  int ret;
779  MYSQL_RES *result;
780 
781  /*
782  * If there's no result associated with the
783  * connection handle, assume the first result in the
784  * result set hasn't been retrieved.
785  *
786  * MySQL docs says there's no performance penalty for
787  * calling mysql_store_result for queries which don't
788  * return results.
789  */
790  if (conn->result == NULL) {
791  result = mysql_store_result(conn->sock);
792  if (result) mysql_free_result(result);
793  /*
794  * ...otherwise call sql_free_result to free an
795  * already stored result.
796  */
797  } else {
798  sql_free_result(handle, config); /* sql_free_result sets conn->result to NULL */
799  }
800 
801  /*
802  * Drain any other results associated with the handle
803  *
804  * mysql_next_result advances the result cursor so that
805  * the next call to mysql_store_result will retrieve
806  * the next result from the server.
807  *
808  * Unfortunately this really does appear to be the
809  * only way to return the handle to a consistent state.
810  */
811  while (((ret = mysql_next_result(conn->sock)) == 0) &&
812  (result = mysql_store_result(conn->sock))) {
813  mysql_free_result(result);
814  }
815  if (ret > 0) return sql_check_error(NULL, ret);
816 #endif
817  return RLM_SQL_OK;
818 }
