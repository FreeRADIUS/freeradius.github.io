31 /** Serialize a cache entry as a humanly readable string
32  *
33  * @param ctx to alloc new string in. Should be a talloc pool a little bigger
34  * than the maximum serialized size of the entry.
35  * @param out Where to write pointer to serialized cache entry.
36  * @param c Cache entry to serialize.
37  * @return
38  * - 0 on success.
39  * - -1 on failure.
40  */
41 int cache_serialize(TALLOC_CTX *ctx, char **out, rlm_cache_entry_t const *c)
42 {
43  TALLOC_CTX *value_pool = NULL;
44  char attr[256]; /* Attr name buffer */
45  map_t *map;
46 
47  char *to_store = NULL;
48 
49  to_store = talloc_typed_asprintf(ctx, "Cache-Expires = %pV\nCache-Created = %pV\n",
50  fr_box_date(c->expires), fr_box_date(c->created));
51  if (!to_store) return -1;
52 
53  /*
54  * It's valid to have an empty cache entry (save allocing the pairs pool)
55  */
56  if (!c->maps) goto finish;
57 
58  value_pool = talloc_pool(ctx, 512);
59  if (!value_pool) {
60  error:
61  talloc_free(to_store);
62  talloc_free(value_pool);
63  return -1;
64  }
65 
66  for (map = c->maps; map; map = map->next) {
67  char *value;
68  ssize_t slen;
69 
70  slen = tmpl_print(&FR_SBUFF_OUT(attr, sizeof(attr)), map->lhs, TMPL_ATTR_REF_PREFIX_NO, NULL);
71  if (slen < 0) {
72  fr_strerror_printf("Serialized attribute too long. Must be < " STRINGIFY(sizeof(attr)) " "
73  "bytes, needed %zu additional bytes", (size_t)(slen * -1));
74  goto error;
75  }
76 
77  fr_value_box_aprint(value_pool, &value, tmpl_value(map->rhs), &fr_value_escape_single);
78  if (!value) goto error;
79 
80  to_store = talloc_asprintf_append_buffer(to_store, "%s %s %s\n", attr,
81  fr_table_str_by_value(fr_tokens_table, map->op, "<INVALID>"),
82  value);
83  if (!to_store) goto error;
84  }
85 finish:
86  talloc_free(value_pool);
87  *out = to_store;
88 
89  return 0;
90 }

92 /** Converts a serialized cache entry back into a structure
93  *
94  * @param[in] c Cache entry to populate (should already be allocated)
95  * @param[in] dict to use for unqualified attributes.
96  * @param[in] in String representation of cache entry.
97  * @param[in] inlen Length of string. May be < 0 in which case strlen will be
98  * used to calculate the length of the string.
99  * @return
100  * - 0 on success.
101  * - -1 on failure.
102  */
103 int cache_deserialize(rlm_cache_entry_t *c, fr_dict_t const *dict, char *in, ssize_t inlen)
104 {
105  map_t **last = &c->maps;
106  char *p, *q;
107 
108  if (inlen < 0) inlen = strlen(in);
109 
110  p = in;
111 
112  while (((size_t)(p - in)) < (size_t)inlen) {
113  map_t *map = NULL;
114  tmpl_rules_t parse_rules = {
115  .dict_def = dict,
116  .prefix = TMPL_ATTR_REF_PREFIX_NO
117  };
118 
119  q = strchr(p, '\n');
120  if (!q) break; /* List should also be terminated with a \n */
121  *q = '\0';
122 
123  if (map_afrom_attr_str(c, &map, p, &parse_rules, &parse_rules) < 0) {
124  fr_strerror_printf("Failed parsing pair: %s", p);
125  error:
126  talloc_free(map);
127  return -1;
128  }
129 
130  if (!tmpl_is_attr(map->lhs)) {
131  fr_strerror_printf("Pair left hand side \"%s\" parsed as %s, needed attribute. "
132  "Check local dictionaries", map->lhs->name,
133  fr_table_str_by_value(tmpl_type_table, map->lhs->type, "<INVALID>"));
134  goto error;
135  }
136 
137  if (!tmpl_is_unresolved(map->rhs)) {
138  fr_strerror_printf("Pair right hand side \"%s\" parsed as %s, needed literal. "
139  "Check serialized data quoting", map->rhs->name,
140  fr_table_str_by_value(tmpl_type_table, map->rhs->type, "<INVALID>"));
141  goto error;
142  }
143 
144  /*
145  * Convert literal to a type appropriate for the VP.
146  */
147  if (tmpl_cast_in_place(map->rhs, tmpl_da(map->lhs)->type, tmpl_da(map->lhs)) < 0) goto error;
148 
149  /*
150  * Pull out the special attributes, and set the
151  * relevant cache entry fields.
152  */
153  if (fr_dict_attr_is_top_level(tmpl_da(map->lhs))) switch (tmpl_da(map->lhs)->attr) {
154  case FR_CACHE_CREATED:
155  c->created = tmpl_value(map->rhs)->vb_date;
156  talloc_free(map);
157  goto next;
158 
159  case FR_CACHE_EXPIRES:
160  c->expires = tmpl_value(map->rhs)->vb_date;
161  talloc_free(map);
162  goto next;
163 
164  default:
165  break;
166  }
167 
168  MAP_VERIFY(map);
169 
170  /* It's not a special attribute, add it to the map list */
171  *last = map;
172  last = &(*last)->next;
173 
174  next:
175  p = q + 1;
176  }
177 
178  return 0;
179 }
