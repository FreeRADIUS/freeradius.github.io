80 ssize_t cond_print(fr_sbuff_t *out, fr_cond_t const *in)
81 {
82  fr_sbuff_t our_out = FR_SBUFF_NO_ADVANCE(out);
83  fr_cond_t const *c = in;
84 
85  while (c) {
86  if (c->negate) FR_SBUFF_IN_CHAR_RETURN(&our_out, '!');
87 
88  switch (c->type) {
89  case COND_TYPE_EXISTS:
90  fr_assert(c->data.vpt != NULL);
91  if (c->cast) {
92  FR_SBUFF_IN_SPRINTF_RETURN(&our_out, "<%s>",
93  fr_table_str_by_value(fr_value_box_type_table,
94  c->cast->type, "??"));
95  }
96  FR_SBUFF_RETURN(tmpl_print_quoted, &our_out, c->data.vpt, TMPL_ATTR_REF_PREFIX_YES);
97  break;
98 
99  case COND_TYPE_RCODE:
100  fr_assert(c->data.rcode != RLM_MODULE_UNKNOWN);
101  FR_SBUFF_IN_STRCPY_RETURN(&our_out, fr_table_str_by_value(rcode_table, c->data.rcode, ""));
102  break;
103 
104  case COND_TYPE_MAP:
105  if (c->cast) {
106  FR_SBUFF_IN_SPRINTF_RETURN(&our_out, "<%s>",
107  fr_table_str_by_value(fr_value_box_type_table,
108  c->cast->type, "??"));
109  }
110  FR_SBUFF_RETURN(map_print, &our_out, c->data.map);
111  break;
112 
113  case COND_TYPE_CHILD:
114  FR_SBUFF_IN_CHAR_RETURN(&our_out, '(');
115  FR_SBUFF_RETURN(cond_print, &our_out, c->data.child);
116  FR_SBUFF_IN_CHAR_RETURN(&our_out, ')');
117  break;
118 
119  case COND_TYPE_TRUE:
120  FR_SBUFF_IN_STRCPY_LITERAL_RETURN(&our_out, "true");
121  break;
122 
123  case COND_TYPE_FALSE:
124  FR_SBUFF_IN_STRCPY_LITERAL_RETURN(&our_out, "false");
125  break;
126 
127  default:
128  break;
129  }
130 
131  if (c->next_op == COND_NONE) {
132  fr_assert(c->next == NULL);
133  goto done;
134  }
135 
136  switch (c->next_op) {
137  case COND_AND:
138  FR_SBUFF_IN_STRCPY_LITERAL_RETURN(&our_out, " && ");
139  break;
140 
141  case COND_OR:
142  FR_SBUFF_IN_STRCPY_LITERAL_RETURN(&our_out, " || ");
143  break;
144 
145  default:
146  fr_assert(0);
147  }
148  c = c->next;
149  }
150 
151 done:
152  fr_sbuff_terminate(&our_out);
153  return fr_sbuff_set(out, &our_out);
154 }

157 static bool cond_type_check(fr_cond_t *c, fr_type_t lhs_type)
158 {
159  /*
160  * SOME integer mismatch is OK. If the LHS has a large type,
161  * and the RHS has a small type, it's OK.
162  *
163  * If the LHS has a small type, and the RHS has a large type,
164  * then add a cast to the LHS.
165  */
166  if (lhs_type == FR_TYPE_UINT64) {
167  if ((tmpl_da(c->data.map->rhs)->type == FR_TYPE_UINT32) ||
168  (tmpl_da(c->data.map->rhs)->type == FR_TYPE_UINT16) ||
169  (tmpl_da(c->data.map->rhs)->type == FR_TYPE_UINT8)) {
170  c->cast = NULL;
171  return true;
172  }
173  }
174 
175  if (lhs_type == FR_TYPE_UINT32) {
176  if ((tmpl_da(c->data.map->rhs)->type == FR_TYPE_UINT16) ||
177  (tmpl_da(c->data.map->rhs)->type == FR_TYPE_UINT8)) {
178  c->cast = NULL;
179  return true;
180  }
181 
182  if (tmpl_da(c->data.map->rhs)->type == FR_TYPE_UINT64) {
183  c->cast = tmpl_da(c->data.map->rhs);
184  return true;
185  }
186  }
187 
188  if (lhs_type == FR_TYPE_UINT16) {
189  if (tmpl_da(c->data.map->rhs)->type == FR_TYPE_UINT8) {
190  c->cast = NULL;
191  return true;
192  }
193 
194  if ((tmpl_da(c->data.map->rhs)->type == FR_TYPE_UINT64) ||
195  (tmpl_da(c->data.map->rhs)->type == FR_TYPE_UINT32)) {
196  c->cast = tmpl_da(c->data.map->rhs);
197  return true;
198  }
199  }
200 
201  if (lhs_type == FR_TYPE_UINT8) {
202  if ((tmpl_da(c->data.map->rhs)->type == FR_TYPE_UINT64) ||
203  (tmpl_da(c->data.map->rhs)->type == FR_TYPE_UINT32) ||
204  (tmpl_da(c->data.map->rhs)->type == FR_TYPE_UINT16)) {
205  c->cast = tmpl_da(c->data.map->rhs);
206  return true;
207  }
208  }
209 
210  if ((lhs_type == FR_TYPE_IPV4_PREFIX) &&
211  (tmpl_da(c->data.map->rhs)->type == FR_TYPE_IPV4_ADDR)) {
212  return true;
213  }
214 
215  if ((lhs_type == FR_TYPE_IPV6_PREFIX) &&
216  (tmpl_da(c->data.map->rhs)->type == FR_TYPE_IPV6_ADDR)) {
217  return true;
218  }
219 
220  /*
221  * Same checks as above, but with the types swapped, and
222  * with explicit cast for the interpretor.
223  */
224  if ((lhs_type == FR_TYPE_IPV4_ADDR) &&
225  (tmpl_da(c->data.map->rhs)->type == FR_TYPE_IPV4_PREFIX)) {
226  c->cast = tmpl_da(c->data.map->rhs);
227  return true;
228  }
229 
230  if ((lhs_type == FR_TYPE_IPV6_ADDR) &&
231  (tmpl_da(c->data.map->rhs)->type == FR_TYPE_IPV6_PREFIX)) {
232  c->cast = tmpl_da(c->data.map->rhs);
233  return true;
234  }
235 
236  return false;
237 }

249 static ssize_t cond_check_cast(fr_cond_t *c, char const *start,
250  char const *lhs, char const *rhs)
251 {
252  if (tmpl_is_attr(c->data.map->rhs) &&
253  (c->cast->type != tmpl_da(c->data.map->rhs)->type)) {
254  if (cond_type_check(c, c->cast->type)) {
255  return 1;
256  }
257 
258  fr_strerror_printf("Cannot compare types '%s' (cast) and '%s' (attr)",
259  fr_table_str_by_value(fr_value_box_type_table, c->cast->type, "<INVALID>"),
260  fr_table_str_by_value(fr_value_box_type_table, tmpl_da(c->data.map->rhs)->type, "<INVALID>"));
261  return 0;
262  }
263 
264 #ifdef HAVE_REGEX
265  if (tmpl_contains_regex(c->data.map->rhs)) {
266  fr_strerror_printf("Cannot use cast with regex comparison");
267  return -(rhs - start);
268  }
269 #endif
270 
271  /*
272  * The LHS is a literal which has been cast to a data type.
273  * Cast it to the appropriate data type.
274  */
275  if (tmpl_is_unresolved(c->data.map->lhs) &&
276  (tmpl_cast_in_place(c->data.map->lhs, c->cast->type, c->cast) < 0)) {
277  fr_strerror_printf("Failed to parse field");
278  return -(lhs - start);
279  }
280 
281  /*
282  * The RHS is a literal, and the LHS has been cast to a data
283  * type.
284  */
285  if ((tmpl_is_data(c->data.map->lhs)) &&
286  (tmpl_is_unresolved(c->data.map->rhs)) &&
287  (tmpl_cast_in_place(c->data.map->rhs, c->cast->type, c->cast) < 0)) {
288  fr_strerror_printf("Failed to parse field");
289  return -(rhs - start);
290  }
291 
292  /*
293  * We may be casting incompatible
294  * types. We check this based on
295  * their size.
296  */
297  if (tmpl_is_attr(c->data.map->lhs)) {
298  /*
299  * dst.min == src.min
300  * dst.max == src.max
301  */
302  if ((dict_attr_sizes[c->cast->type][0] == dict_attr_sizes[tmpl_da(c->data.map->lhs)->type][0]) &&
303  (dict_attr_sizes[c->cast->type][1] == dict_attr_sizes[tmpl_da(c->data.map->lhs)->type][1])) {
304  goto cast_ok;
305  }
306 
307  /*
308  * Run-time parsing of strings.
309  * Run-time copying of octets.
310  */
311  if ((tmpl_da(c->data.map->lhs)->type == FR_TYPE_STRING) ||
312  (tmpl_da(c->data.map->lhs)->type == FR_TYPE_OCTETS)) {
313  goto cast_ok;
314  }
315 
316  /*
317  * ifid to uint64 is OK
318  */
319  if ((tmpl_da(c->data.map->lhs)->type == FR_TYPE_IFID) &&
320  (c->cast->type == FR_TYPE_UINT64)) {
321  goto cast_ok;
322  }
323 
324  /*
325  * ipaddr to ipv4prefix is OK
326  */
327  if ((tmpl_da(c->data.map->lhs)->type == FR_TYPE_IPV4_ADDR) &&
328  (c->cast->type == FR_TYPE_IPV4_PREFIX)) {
329  goto cast_ok;
330  }
331 
332  /*
333  * ipv6addr to ipv6prefix is OK
334  */
335  if ((tmpl_da(c->data.map->lhs)->type == FR_TYPE_IPV6_ADDR) &&
336  (c->cast->type == FR_TYPE_IPV6_PREFIX)) {
337  goto cast_ok;
338  }
339 
340  /*
341  * uint64 to ethernet is OK.
342  */
343  if ((tmpl_da(c->data.map->lhs)->type == FR_TYPE_UINT64) &&
344  (c->cast->type == FR_TYPE_ETHERNET)) {
345  goto cast_ok;
346  }
347 
348  /*
349  * dst.max < src.min
350  * dst.min > src.max
351  */
352  if ((dict_attr_sizes[c->cast->type][1] < dict_attr_sizes[tmpl_da(c->data.map->lhs)->type][0]) ||
353  (dict_attr_sizes[c->cast->type][0] > dict_attr_sizes[tmpl_da(c->data.map->lhs)->type][1])) {
354  fr_strerror_printf("Cannot cast to attribute of incompatible size");
355  return 0;
356  }
357  }
358 
359 cast_ok:
360  /*
361  * Casting to a redundant type means we don't need the cast.
362  *
363  * Do this LAST, as the rest of the code above assumes c->cast
364  * is not NULL.
365  */
366  if (tmpl_is_attr(c->data.map->lhs) &&
367  (c->cast->type == tmpl_da(c->data.map->lhs)->type)) {
368  c->cast = NULL;
369  }
370 
371  return 1;
372 }

375  * See if two attribute comparisons are OK.

377 static ssize_t cond_check_attrs(fr_cond_t *c, fr_sbuff_marker_t *m_lhs, fr_sbuff_marker_t *m_rhs)
378 {
379  tmpl_t *attr, *data, *xlat, *unresolved, *xlat_unresolved, *exec, *vpt;
380  tmpl_t *lhs = c->data.map->lhs, *rhs = c->data.map->rhs;
381  fr_token_t op = c->data.map->op;
382 
383 /** True if one operand is of _type_a and the other of _type_b
384  */
385 #define TMPL_OF_TYPE_A_B(_type_a, _type_b) \
386  ((((tmpl_is_##_type_a(lhs) && (_type_a = lhs)) && (tmpl_is_##_type_b(rhs) && (_type_b = rhs)))) || \
387  (((tmpl_is_##_type_a(rhs) && (_type_a = rhs)) && (tmpl_is_##_type_b(lhs) && (_type_b = lhs)))))
388 
389 /** True if one operand is of _type_a and the other is not of _type_a
390  *
391  */
392 #define TMPL_OF_TYPE_A_NOT_A(_type_a, _out_b) \
393  ((((tmpl_is_##_type_a(lhs) && (_type_a = lhs)) && (!tmpl_is_##_type_a(rhs) && (_out_b = rhs)))) || \
394  (((tmpl_is_##_type_a(rhs) && (_type_a = rhs)) && (!tmpl_is_##_type_a(lhs) && (_out_b = lhs)))))
395 
396 /** True if both operands are of _type_a
397  *
398  */
399 #define TMPL_OF_TYPE_A_A(_type_a) \
400  ((tmpl_is_##_type_a(lhs)) && (tmpl_is_##_type_a(rhs)))
401 
402 #define TMPL_RETURN(_vpt) return -((_vpt) == lhs ? fr_sbuff_used(m_lhs) : fr_sbuff_used(m_rhs))
403 
404  /*
405  * Attribute comparison with a box
406  */
407  if (TMPL_OF_TYPE_A_B(attr, data)) {
408  fr_type_t type = tmpl_da(attr)->type;
409 
410  /*
411  * Most of the time the box type takes
412  * precedence, except in the case of a few types
413  * like IP prefixes.
414  */
415  switch (tmpl_value_type(data)) {
416  case FR_TYPE_IPV4_PREFIX:
417  case FR_TYPE_IPV6_PREFIX:
418  type = tmpl_value_type(data);
419  break;
420 
421  default:
422  break;
423  }
424 
425  if (tmpl_cast_in_place(data, type, tmpl_da(attr)) < 0) {
426  fr_strerror_printf_push("Failed casting data to match attribute");
427  return -(data == lhs ? fr_sbuff_used(m_lhs) : fr_sbuff_used(m_rhs));
428  }
429  }
430 
431  /*
432  * Two attributes? They must be of the same type
433  */
434  if (TMPL_OF_TYPE_A_A(attr) && (tmpl_da(lhs)->type != tmpl_da(rhs)->type)) {
435  if (cond_type_check(c, tmpl_da(lhs)->type)) return 1; /* Or be mungeable to the same type */
436 
437  fr_strerror_printf("Cannot compare attributes of type '%s' and '%s'",
438  fr_table_str_by_value(fr_value_box_type_table, tmpl_da(lhs)->type, "<INVALID>"),
439  fr_table_str_by_value(fr_value_box_type_table, tmpl_da(rhs)->type, "<INVALID>"));
440  return 0;
441  }
442 
443  /*
444  * The LHS has been cast to a data type, and the RHS is a
445  * literal. Cast the RHS to the type of the cast.
446  *
447  * FIXME - We should revisit this when RHS casting is supported.
448  */
449  if (c->cast && tmpl_is_unresolved(rhs) &&
450  (tmpl_cast_in_place(rhs, c->cast->type, c->cast) < 0)) {
451  fr_strerror_printf("Failed to parse field");
452  return -fr_sbuff_used(m_rhs);
453  }
454 
455  /*
456  * The LHS is an attribute, and the RHS is a literal. Cast the
457  * RHS to the data type of the LHS.
458  *
459  * Note: There's a hack in here to always parse RHS as the
460  * equivalent prefix type if the LHS is an IP address.
461  *
462  * This allows Framed-IP-Address < 192.168.0.0./24
463  */
464  unresolved = NULL;
465  data = NULL; /* gcc stupidity */
466  if (TMPL_OF_TYPE_A_B(attr, unresolved) || TMPL_OF_TYPE_A_B(attr, data)) {
467  fr_type_t type = tmpl_da(attr)->type;
468 
469  vpt = unresolved ? unresolved : data;
470 
471  /*
472  * Invalid: User-Name == bob
473  * Valid: User-Name == "bob"
474  *
475  * There's no real reason for
476  * this, other than consistency.
477  */
478  if ((unresolved || (data && tmpl_value_type(data) == FR_TYPE_STRING)) &&
479  (tmpl_da(attr)->type == FR_TYPE_STRING) &&
480  (op != T_OP_CMP_TRUE) &&
481  (op != T_OP_CMP_FALSE) &&
482  (vpt->quote == T_BARE_WORD)) {
483  fr_strerror_printf("Comparison value must be a quoted string");
484  TMPL_RETURN(vpt);
485  }
486 
487  switch (tmpl_da(attr)->type) {
488  case FR_TYPE_IPV4_ADDR:
489  if (strchr(vpt->name, '/') != NULL) {
490  type = FR_TYPE_IPV4_PREFIX;
491  c->cast = fr_dict_attr_child_by_num(fr_dict_root(fr_dict_internal()), FR_CAST_BASE + type);
492  }
493  break;
494 
495  case FR_TYPE_IPV6_ADDR:
496  if (strchr(vpt->name, '/') != NULL) {
497  type = FR_TYPE_IPV6_PREFIX;
498  c->cast = fr_dict_attr_child_by_num(fr_dict_root(fr_dict_internal()), FR_CAST_BASE + type);
499  }
500  break;
501 
502  default:
503  break;
504  }
505 
506  /*
507  * Do not pass LHS as enumv if we're casting
508  * as that means there's now a type mismatch between
509  * attr and vpt, which means the enumerations
510  * can never match.
511  */
512  if (tmpl_cast_in_place(vpt, type,
513  c->cast ? NULL : tmpl_da(attr)) < 0) {
514  fr_dict_attr_t const *da = tmpl_da(attr);
515 
516  switch (da->attr) {
517  case FR_AUTH_TYPE:
518  /*
519  * The types for these attributes are dynamically allocated
520  * by module.c, so we can't enforce strictness here.
521  */
522  c->pass2_fixup = PASS2_FIXUP_TYPE;
523  break;
524 
525  default:
526  if (!attr->data.attribute.was_oid) {
527  fr_strerror_printf("Failed to parse value for attribute");
528  TMPL_RETURN(vpt);
529  }
530  /*
531  * Convert the attr to a raw type and
532  * try the cast again.
533  */
534  tmpl_attr_to_raw(attr);
535  if (tmpl_cast_in_place(vpt, tmpl_da(attr)->type,
536  c->cast ? NULL : tmpl_da(attr)) < 0) {
537  fr_strerror_printf("Failed to parse value for attribute");
538  TMPL_RETURN(vpt);
539  }
540  break;
541  }
542  }
543 
544  /*
545  * Stupid WiMAX shit.
546  * Cast the LHS to the
547  * type of the RHS.
548  */
549  if (tmpl_da(attr)->type == FR_TYPE_COMBO_IP_ADDR) {
550  if (tmpl_attr_abstract_to_concrete(attr, tmpl_value_type(vpt)) < 0) {
551  fr_strerror_printf("Cannot find type for attribute");
552  TMPL_RETURN(attr);
553  }
554  }
555  } /* attr to literal comparison */
556 
557  /*
558  * If one side is unresolved, and the other is data,
559  * we can use the data type to attempt a cast on the
560  * unresolved side.
561  */
562  if (TMPL_OF_TYPE_A_B(data, unresolved) &&
563  (tmpl_cast_in_place(unresolved, tmpl_value_type(data), NULL) < 0)) TMPL_RETURN(unresolved);
564 
565  /*
566  * The RHS will turn into... something. Allow for prefixes
567  * there, too.
568  */
569  if (TMPL_OF_TYPE_A_B(attr, xlat_unresolved) || TMPL_OF_TYPE_A_B(attr, xlat) || TMPL_OF_TYPE_A_B(attr, exec)) {
570  if (tmpl_da(attr)->type == FR_TYPE_IPV4_ADDR) {
571  c->cast = fr_dict_attr_child_by_num(fr_dict_root(fr_dict_internal()),
572  FR_CAST_BASE + FR_TYPE_IPV4_PREFIX);
573  }
574 
575  if (tmpl_da(attr)->type == FR_TYPE_IPV6_ADDR) {
576  c->cast = fr_dict_attr_child_by_num(fr_dict_root(fr_dict_internal()),
577  FR_CAST_BASE + FR_TYPE_IPV6_PREFIX);
578  }
579  }

581  /*
582  * If the LHS is a bare word, AND it looks like
583  * an attribute, try to parse it as such.
584  *
585  * This allows LDAP-Group and SQL-Group to work.
586  *
587  * The real fix is to just read the config files,
588  * and do no parsing until after all of the modules
589  * are loaded. But that has issues, too.
590  */
591  if (tmpl_is_unresolved(lhs) && (lhs->quote == T_BARE_WORD)) {
592  int hyphens = 0;
593  bool may_be_attr = true;
594  size_t i;
595  ssize_t attr_slen;
596 
597  /*
598  * Backwards compatibility: Allow Foo-Bar,
599  * e.g. LDAP-Group and SQL-Group.
600  */
601  for (i = 0; i < lhs->len; i++) {
602  if (!fr_dict_attr_allowed_chars[(uint8_t) lhs->name[i]]) {
603  may_be_attr = false;
604  break;
605  }
606 
607  if (lhs->name[i] == '-') {
608  hyphens++;
609  }
610  }
611 
612  if (!hyphens || (hyphens > 3)) may_be_attr = false;
613 
614  if (may_be_attr) {
615  attr_slen = tmpl_afrom_attr_str(c->data.map, NULL, &vpt, fr_sbuff_current(m_lhs),
616  &(tmpl_rules_t){
617  .allow_unknown = true,
618  .allow_unresolved = true
619  });
620  if ((attr_slen > 0) && (vpt->len == lhs->len)) {
621  talloc_free(lhs);
622  c->pass2_fixup = PASS2_FIXUP_ATTR;
623  }
624  }
625  }
626 
627  return 1;

630 static int cond_normalise(TALLOC_CTX *ctx, fr_token_t lhs_type, fr_cond_t **c_out)
631 {
632  fr_cond_t *c = *c_out;
633 
634  /*
635  * Normalize the condition before returning.
636  *
637  * We collapse multiple levels of braces to one. Then
638  * convert maps to literals. Then literals to true/false
639  * statements. Then true/false ||/&& followed by other
640  * conditions to just conditions.
641  *
642  * Order is important. The more complex cases are
643  * converted to simpler ones, from the most complex cases
644  * to the simplest ones.
645  */
646 
647  /*
648  * (FOO) --> FOO
649  * (FOO) ... --> FOO ...
650  */
651  if ((c->type == COND_TYPE_CHILD) && !c->data.child->next) {
652  fr_cond_t *child;
653 
654  child = talloc_steal(ctx, c->data.child);
655  c->data.child = NULL;
656 
657  child->next = talloc_steal(child, c->next);
658  c->next = NULL;
659 
660  child->next_op = c->next_op;
661 
662  /*
663  * Set the negation properly
664  */
665  if ((c->negate && !child->negate) ||
666  (!c->negate && child->negate)) {
667  child->negate = true;
668  } else {
669  child->negate = false;
670  }
671 
672  talloc_free(c);
673  c = child;
674  }
675 
676  /*
677  * (FOO ...) --> FOO ...
678  *
679  * But don't do !(FOO || BAR) --> !FOO || BAR
680  * Because that's different.
681  */
682  if ((c->type == COND_TYPE_CHILD) &&
683  !c->next && !c->negate) {
684  fr_cond_t *child;
685 
686  child = talloc_steal(ctx, c->data.child);
687  c->data.child = NULL;
688 
689  talloc_free(c);
690  c = child;
691  }
692 
693  /*
694  * Convert maps to literals. Convert one form of map to
695  * a standardized form. This doesn't make any
696  * theoretical difference, but it does mean that the
697  * run-time evaluation has fewer cases to check.
698  */
699  if (c->type == COND_TYPE_MAP) do {
700  /*
701  * !FOO !~ BAR --> FOO =~ BAR
702  */
703  if (c->negate && (c->data.map->op == T_OP_REG_NE)) {
704  c->negate = false;
705  c->data.map->op = T_OP_REG_EQ;
706  }
707 
708  /*
709  * FOO !~ BAR --> !FOO =~ BAR
710  */
711  if (!c->negate && (c->data.map->op == T_OP_REG_NE)) {
712  c->negate = true;
713  c->data.map->op = T_OP_REG_EQ;
714  }
715 
716  /*
717  * !FOO != BAR --> FOO == BAR
718  */
719  if (c->negate && (c->data.map->op == T_OP_NE)) {
720  c->negate = false;
721  c->data.map->op = T_OP_CMP_EQ;
722  }
723 
724  /*
725  * This next one catches "LDAP-Group != foo",
726  * which doesn't work as-is, but this hack fixes
727  * it.
728  *
729  * FOO != BAR --> !FOO == BAR
730  */
731  if (!c->negate && (c->data.map->op == T_OP_NE)) {
732  c->negate = true;
733  c->data.map->op = T_OP_CMP_EQ;
734  }
735 
736  /*
737  * FOO =* BAR --> FOO
738  * FOO !* BAR --> !FOO
739  *
740  * FOO may be a string, or a delayed attribute
741  * reference.
742  */
743  if ((c->data.map->op == T_OP_CMP_TRUE) ||
744  (c->data.map->op == T_OP_CMP_FALSE)) {
745  tmpl_t *vpt;
746 
747  vpt = talloc_steal(c, c->data.map->lhs);
748  c->data.map->lhs = NULL;
749 
750  /*
751  * Invert the negation bit.
752  */
753  if (c->data.map->op == T_OP_CMP_FALSE) {
754  c->negate = !c->negate;
755  }
756 
757  TALLOC_FREE(c->data.map);
758 
759  c->type = COND_TYPE_EXISTS;
760  c->data.vpt = vpt;
761  break; /* it's no longer a map */
762  }
763 
764  /*
765  * Both are data (IP address, integer, etc.)
766  *
767  * We can do the evaluation here, so that it
768  * doesn't need to be done at run time
769  */
770  if (tmpl_is_data(c->data.map->lhs) &&
771  tmpl_is_data(c->data.map->rhs)) {
772  int rcode;
773 
774  rcode = cond_eval_map(NULL, 0, c);
775  TALLOC_FREE(c->data.map);
776  c->cast = NULL;
777  if (rcode) {
778  c->type = COND_TYPE_TRUE;
779  } else {
780  c->type = COND_TYPE_FALSE;
781  }
782 
783  break; /* it's no longer a map */
784  }
785 
786  /*
787  * Both are literal strings. They're not parsed
788  * as TMPL_TYPE_DATA because there's no cast to an
789  * attribute.
790  *
791  * We can do the evaluation here, so that it
792  * doesn't need to be done at run time
793  */
794  if (tmpl_is_unresolved(c->data.map->rhs) &&
795  tmpl_is_unresolved(c->data.map->lhs) &&
796  !c->pass2_fixup) {
797  int rcode;
798 
799  fr_assert(c->cast == NULL);
800 
801  rcode = cond_eval_map(NULL, 0, c);
802  if (rcode) {
803  c->type = COND_TYPE_TRUE;
804  } else {
805  DEBUG4("OPTIMIZING (%s %s %s) --> FALSE",
806  c->data.map->lhs->name,
807  fr_table_str_by_value(fr_tokens_table, c->data.map->op, "??"),
808  c->data.map->rhs->name);
809  c->type = COND_TYPE_FALSE;
810  }
811 
812  /*
813  * Free map after using it above.
814  */
815  TALLOC_FREE(c->data.map);
816  break;
817  }
818 
819  /*
820  * <ipaddr>"foo" CMP &Attribute-Name The cast may
821  * not be necessary, and we can re-write it so
822  * that the attribute reference is on the LHS.
823  */
824  if (c->cast &&
825  tmpl_is_attr(c->data.map->rhs) &&
826  (c->cast->type == tmpl_da(c->data.map->rhs)->type) &&
827  !tmpl_is_attr(c->data.map->lhs)) {
828  tmpl_t *tmp;
829 
830  tmp = c->data.map->rhs;
831  c->data.map->rhs = c->data.map->lhs;
832  c->data.map->lhs = tmp;
833 
834  c->cast = NULL;
835 
836  switch (c->data.map->op) {
837  case T_OP_CMP_EQ:
838  /* do nothing */
839  break;
840 
841  case T_OP_LE:
842  c->data.map->op = T_OP_GE;
843  break;
844 
845  case T_OP_LT:
846  c->data.map->op = T_OP_GT;
847  break;
848 
849  case T_OP_GE:
850  c->data.map->op = T_OP_LE;
851  break;
852 
853  case T_OP_GT:
854  c->data.map->op = T_OP_LT;
855  break;
856 
857  default:
858  fr_strerror_printf("Internal sanity check failed 1");
859  return -1;
860  }
861 
862  /*
863  * This must have been parsed into TMPL_TYPE_DATA.
864  */
865  fr_assert(!tmpl_is_unresolved(c->data.map->rhs));
866  }
867 
868  } while (0);
869 
870  /*
871  * Existence checks. We short-circuit static strings,
872  * too.
873  *
874  * FIXME: the data types should be in the template, too.
875  * So that we know where a literal came from.
876  *
877  * "foo" is NOT the same as 'foo' or a bare foo.
878  */
879  if (c->type == COND_TYPE_EXISTS) {
880  switch (c->data.vpt->type) {
881  case TMPL_TYPE_XLAT:
882  case TMPL_TYPE_XLAT_UNRESOLVED:
883  case TMPL_TYPE_ATTR:
884  case TMPL_TYPE_ATTR_UNRESOLVED:
885  case TMPL_TYPE_LIST:
886  case TMPL_TYPE_EXEC:
887  break;
888 
889  /*
890  * 'true' and 'false' are special strings
891  * which mean themselves.
892  *
893  * For integers, 0 is false, all other
894  * integers are true.
895  *
896  * For strings, '' and "" are false.
897  * 'foo' and "foo" are true.
898  *
899  * The str2tmpl function takes care of
900  * marking "%{foo}" as TMPL_TYPE_XLAT_UNRESOLVED, so
901  * the strings here are fixed at compile
902  * time.
903  *
904  * `exec` and "%{...}" are left alone.
905  *
906  * Bare words must be module return
907  * codes.
908  */
909  case TMPL_TYPE_UNRESOLVED:
910  if (!*c->data.vpt->name) {
911  c->type = COND_TYPE_FALSE;
912  TALLOC_FREE(c->data.vpt);
913 
914  } else if ((lhs_type == T_SINGLE_QUOTED_STRING) ||
915  (lhs_type == T_DOUBLE_QUOTED_STRING)) {
916  c->type = COND_TYPE_TRUE;
917  TALLOC_FREE(c->data.vpt);
918 
919  } else if (lhs_type == T_BARE_WORD) {
920  int rcode;
921  bool zeros = true;
922  char const *q;
923 
924  for (q = c->data.vpt->name;
925  *q != '\0';
926  q++) {
927  if (!isdigit((int) *q)) {
928  break;
929  }
930  if (*q != '0') zeros = false;
931  }
932 
933  /*
934  * It's all digits, and therefore
935  * 'false' if zero, and 'true' otherwise.
936  */
937  if (!*q) {
938  if (zeros) {
939  c->type = COND_TYPE_FALSE;
940  } else {
941  c->type = COND_TYPE_TRUE;
942  }
943  TALLOC_FREE(c->data.vpt);
944  break;
945  }
946 
947  /*
948  * Allow &Foo-Bar where Foo-Bar is an attribute
949  * defined by a module.
950  */
951  if (c->pass2_fixup == PASS2_FIXUP_ATTR) {
952  break;
953  }
954 
955  rcode = fr_table_value_by_str(allowed_return_codes, c->data.vpt->name, 0);
956  if (!rcode) {
957  fr_strerror_printf("Expected a module return code");
958  return -1;
959  }
960  }
961 
962  /*
963  * Else lhs_type==T_INVALID, and this
964  * node was made by promoting a child
965  * which had already been normalized.
966  */
967  break;
968 
969  case TMPL_TYPE_DATA:
970  {
971  fr_value_box_t res;
972 
973  if (fr_value_box_cast(NULL, &res, FR_TYPE_BOOL, NULL, tmpl_value(c->data.vpt)) < 0) return -1;
974  c->type = res.vb_bool ? COND_TYPE_TRUE : COND_TYPE_FALSE;
975  TALLOC_FREE(c->data.vpt);
976  }
977  break;
978 
979  default:
980  fr_assert_fail("Internal sanity check failed 2");
981  return -1;
982  }
983  }
984 
985  /*
986  * !TRUE -> FALSE
987  */
988  if (c->type == COND_TYPE_TRUE) {
989  if (c->negate) {
990  c->negate = false;
991  c->type = COND_TYPE_FALSE;
992  }
993  }
994 
995  /*
996  * !FALSE -> TRUE
997  */
998  if (c->type == COND_TYPE_FALSE) {
999  if (c->negate) {
1000  c->negate = false;
1001  c->type = COND_TYPE_TRUE;
1002  }
1003  }
1004 
1005  /*
1006  * true && FOO --> FOO
1007  */
1008  if ((c->type == COND_TYPE_TRUE) &&
1009  (c->next_op == COND_AND)) {
1010  fr_cond_t *next;
1011 
1012  next = talloc_steal(ctx, c->next);
1013  c->next = NULL;
1014 
1015  talloc_free(c);
1016  c = next;
1017  }
1018 
1019  /*
1020  * false && FOO --> false
1021  */
1022  if ((c->type == COND_TYPE_FALSE) &&
1023  (c->next_op == COND_AND)) {
1024 
1025  talloc_free(c->next);
1026  c->next = NULL;
1027  c->next_op = COND_NONE;
1028  }
1029 
1030  /*
1031  * false || FOO --> FOO
1032  */
1033  if ((c->type == COND_TYPE_FALSE) &&
1034  (c->next_op == COND_OR)) {
1035  fr_cond_t *next;
1036 
1037  next = talloc_steal(ctx, c->next);
1038  c->next = NULL;
1039 
1040  talloc_free(c);
1041  c = next;
1042  }
1043 
1044  /*
1045  * true || FOO --> true
1046  */
1047  if ((c->type == COND_TYPE_TRUE) &&
1048  (c->next_op == COND_OR)) {
1049 
1050  talloc_free(c->next);
1051  c->next = NULL;
1052  c->next_op = COND_NONE;
1053  }
1054 
1055  *c_out = c;
1056 
1057  return 0;
1058 }
