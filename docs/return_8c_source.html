30 unlang_action_t unlang_return(rlm_rcode_t *p_result, request_t *request)
31 {
32  unlang_stack_t *stack = request->stack;
33  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
34  unlang_t *instruction = frame->instruction;
35 
36  RDEBUG2("%s", unlang_ops[instruction->type].name);
37 
38  *p_result = frame->result;
39 
40  /*
41  * Stop at the next return point, or if we hit
42  * the a top frame.
43  */
44  return unwind_to_return(stack);
45 }
