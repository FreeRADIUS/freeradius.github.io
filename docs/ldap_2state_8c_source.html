38 /** Move between LDAP connection states
39  *
40  * Bringing up an LDAP connection is quite complex, as we need to do multiple operations
41  * before we can install the main mux/demux functions which do the work of sending
42  * requests to the directory and processing the responses.
43  *
44  * This function moves the connection through different states, setting different I/O
45  * handlers.
46  *
47  * If any of the states
48  */
49 fr_ldap_state_t fr_ldap_state_next(fr_ldap_connection_t *c)
50 {
51 again:
52  switch (c->state) {
53  /*
54  * Start by negotiating TLS, or binding
55  */
56  case FR_LDAP_STATE_INIT:
57  if (c->config->start_tls) {
58  if (fr_ldap_start_tls_async(c, NULL, NULL) < 0) {
59  STATE_TRANSITION(FR_LDAP_STATE_ERROR);
60  goto again;
61  }
62  STATE_TRANSITION(FR_LDAP_STATE_START_TLS);
63  break;
64  }
65  FALL_THROUGH;
66 
67  /*
68  * If we're successful in negotiating TLS,
69  * bind to the server as the credentials
70  * will now be protected.
71  */
72  case FR_LDAP_STATE_START_TLS:
73  STATE_TRANSITION(FR_LDAP_STATE_BIND);
74 
75  /*
76  * SASL uses a different (and more complex) codepath
77  */
78 #ifdef WITH_SASL
79  if (c->config->admin_sasl.mech) {
80  if (fr_ldap_sasl_bind_async(c,
81  c->config->admin_sasl.mech,
82  c->config->admin_identity,
83  c->config->admin_password,
84  c->config->admin_sasl.proxy,
85  c->config->admin_sasl.realm,
86  NULL, NULL) < 0) {
87  STATE_TRANSITION(FR_LDAP_STATE_ERROR);
88  goto again;
89  }
90  break;
91  }
92 #endif
93 
94  /*
95  * Normal binds are just a simple request/response pair
96  */
97  if (fr_ldap_bind_async(c,
98  c->config->admin_identity,
99  c->config->admin_password,
100  NULL, NULL) < 0) {
101  STATE_TRANSITION(FR_LDAP_STATE_ERROR);
102  goto again;
103  }
104  break;
105 
106  /*
107  * After binding install the mux (write) and
108  * demux (read) I/O functions.
109  */
110  case FR_LDAP_STATE_BIND:
111  STATE_TRANSITION(FR_LDAP_STATE_RUN);
112  /*
113  if (fr_ldap_mux_async(c) < 0) {
114  STATE_TRANSITION(FR_LDAP_STATE_ERROR);
115  goto again;
116  }
117  */
118  break;
119 
120  /*
121  * Something went wrong
122  */
123  case FR_LDAP_STATE_RUN: /* There's no next state for run, so this an error */
124  case FR_LDAP_STATE_ERROR:
125  STATE_TRANSITION(FR_LDAP_STATE_INIT);
126  fr_connection_signal_reconnect(c->conn, FR_CONNECTION_FAILED);
127  break;
128  }
129 
130  return c->state;
131 }
