133 static ssize_t mod_read(fr_listen_t *li, void **packet_ctx, fr_time_t *recv_time_p, uint8_t *buffer, size_t buffer_len,
134  size_t *leftover, UNUSED uint32_t *priority, UNUSED bool *is_dup)
135 {
136  proto_dhcpv4_udp_thread_t *thread = talloc_get_type_abort(li->thread_instance, proto_dhcpv4_udp_thread_t);
137  fr_io_address_t *address, **address_p;
138 
139  int flags;
140  ssize_t data_size;
141  size_t packet_len;
142  uint8_t message_type;
143  uint32_t xid, ipaddr;
144  dhcp_packet_t *packet;
145 
146  *leftover = 0; /* always for UDP */
147 
148  /*
149  * Where the addresses should go. This is a special case
150  * for proto_dhcpv4.
151  */
152  address_p = (fr_io_address_t **) packet_ctx;
153  address = *address_p;
154 
155  /*
156  * Tell udp_recv if we're connected or not.
157  */
158  flags = UDP_FLAGS_CONNECTED * (thread->connection != NULL);
159 
160  data_size = udp_recv(thread->sockfd, flags, &address->socket, buffer, buffer_len, recv_time_p);
161  if (data_size < 0) {
162  RATE_LIMIT_GLOBAL(PERROR, "Read error (%zd)", data_size);
163  return data_size;
164  }
165 
166  if (!data_size) {
167  RATE_LIMIT_GLOBAL(WARN, "Got no data - ignoring");
168  return 0;
169  }
170 
171  /*
172  * @todo - make this take "&packet_len", as the DHCPv4
173  * packet may be smaller than the parent UDP packet.
174  */
175  if (!fr_dhcpv4_ok(buffer, data_size, &message_type, &xid)) {
176  RATE_LIMIT_GLOBAL(PWARN, "Invalid packet - ignoring");
177  return 0;
178  }
179 
180  packet_len = data_size;
181 
182  /*
183  * We've seen a server reply to this port, but the giaddr
184  * is *not* our address. Drop it.
185  */
186  packet = (dhcp_packet_t *) buffer;
187  memcpy(&ipaddr, &packet->giaddr, 4);
188  if ((packet->opcode == 2) && (ipaddr != address->socket.inet.dst_ipaddr.addr.v4.s_addr)) {
189  DEBUG2("Ignoring server reply which was not meant for us (was for %pV).",
190  fr_box_ipaddr(address->socket.inet.dst_ipaddr));
191  return 0;
192  }
193 
194  /*
195  * proto_dhcpv4 sets the priority
196  */
197 
198  /*
199  * Print out what we received.
200  */
201  DEBUG2("Received %s XID %08x length %d %s", dhcp_message_types[message_type], xid,
202  (int) packet_len, thread->name);
203 
204  return packet_len;
205 }

208 static ssize_t mod_write(fr_listen_t *li, void *packet_ctx, UNUSED fr_time_t request_time,
209  uint8_t *buffer, size_t buffer_len, UNUSED size_t written)
210 {
211  proto_dhcpv4_udp_t const *inst = talloc_get_type_abort_const(li->app_io_instance, proto_dhcpv4_udp_t);
212  proto_dhcpv4_udp_thread_t *thread = talloc_get_type_abort(li->thread_instance, proto_dhcpv4_udp_thread_t);
213 
214  fr_io_track_t *track = talloc_get_type_abort(packet_ctx, fr_io_track_t);
215  proto_dhcpv4_track_t *request = talloc_get_type_abort(track->packet, proto_dhcpv4_track_t);
216  fr_socket_t socket;
217 
218  int flags;
219  ssize_t data_size;
220 
221  /*
222  * @todo - share a stats interface with the parent? or
223  * put the stats in the listener, so that proto_dhcpv4
224  * can update them, too.. <sigh>
225  */
226  thread->stats.total_responses++;
227 
228  flags = UDP_FLAGS_CONNECTED * (thread->connection != NULL);
229 
230  /*
231  * Swap src/dst IP/port
232  */
233  fr_socket_addr_swap(&socket, &track->address->socket);
234 
235  /*
236  * Figure out which kind of packet we're sending.
237  */
238  if (!thread->connection) {
239  uint8_t const *code, *sid;
240  dhcp_packet_t *packet = (dhcp_packet_t *) buffer;
241 #ifdef WITH_IFINDEX_IPADDR_RESOLUTION
242  fr_ipaddr_t primary;
243 #endif
244 
245  /*
246  * This isn't available in the packet header.
247  */
248  code = fr_dhcpv4_packet_get_option(packet, buffer_len, attr_message_type);
249  if (!code || (code[1] < 1) || (code[2] == 0) || (code[2] > FR_DHCP_LEASE_ACTIVE)) {
250  WARN("Silently discarding reply due to invalid or missing message type");
251  return 0;
252  }
253 
254  /*
255  * Set the source IP we'll use for sending the packets.
256  *
257  * - if src_ipaddr is unicast, use that
258  * - else if socket wasn't bound to *, then use that
259  * - else if we have ifindex, get main IP from that interface and use that.
260  * - else for offer/ack, look at option 54, for Server Identification and use that
261  * - else leave source IP as whatever is already in "socket.inet.src_ipaddr".
262  */
263  if (!fr_ipaddr_is_inaddr_any(&inst->src_ipaddr)) {
264  socket.inet.src_ipaddr = inst->src_ipaddr;
265  } else if (!fr_ipaddr_is_inaddr_any(&inst->ipaddr)) {
266  socket.inet.src_ipaddr = inst->ipaddr;
267 #ifdef WITH_IFINDEX_IPADDR_RESOLUTION
268  } else if ((address->socket.inet.ifindex > 0) &&
269  (fr_ipaddr_from_ifindex(&primary, thread->sockfd, &socket.inet.dst_ipaddr.af,
270  &socket.inet.ifindex) == 0)) {
271  socket.inet.src_ipaddr = primary;
272 #endif
273  } else if (((code[2] == FR_DHCP_OFFER) || (code[2] == FR_DHCP_ACK)) &&
274  ((sid = fr_dhcpv4_packet_get_option(packet, buffer_len, attr_dhcp_server_identifier)) != NULL) &&
275  (sid[1] == 4)) {
276  memcpy(&socket.inet.src_ipaddr.addr.v4.s_addr, sid + 2, 4);
277  }
278 
279  /*
280  * If we're forwarding the Discover or Request,
281  * then we need to figure out where to forward it
282  * to?
283  */
284  if ((code[2] == FR_DHCP_DISCOVER) || (code[2] == FR_DHCP_REQUEST)) {
285  WARN("Silently discarding client request, as we do not know where to send it");
286  return 0;
287  }
288 
289  /*
290  * We have GIADDR in the packet, so send it
291  * there. The packet is FROM our IP address and
292  * port, TO the destination IP address, at the
293  * same (i.e. server) port.
294  *
295  * RFC 2131 page 23
296  *
297  * "If the 'giaddr' field in a DHCP message from
298  * a client is non-zero, the server sends any
299  * return messages to the 'DHCP server' port on
300  * the BOOTP relay agent whose address appears in
301  * 'giaddr'.
302  */
303  if (packet->giaddr != INADDR_ANY) {
304  DEBUG("Reply will be sent to giaddr.");
305  socket.inet.dst_ipaddr.addr.v4.s_addr = packet->giaddr;
306  socket.inet.dst_port = inst->port;
307  socket.inet.src_port = inst->port;
308 
309  /*
310  * Increase the hop count for client
311  * packets sent to the next gateway.
312  */
313  if ((code[2] == FR_DHCP_DISCOVER) ||
314  (code[2] == FR_DHCP_REQUEST)) {
315  packet->opcode = 1; /* client message */
316  packet->hops = request->hops + 1;
317  } else {
318  packet->opcode = 2; /* server message */
319  }
320 
321  goto send_reply;
322  }
323 
324  packet->opcode = 2; /* server message */
325 
326  /*
327  * NAKs are broadcast when there's no giaddr.
328  *
329  * RFC 2131 page 23.
330  *
331  * "In all cases, when 'giaddr' is zero, the server
332  * broadcasts any DHCPNAK messages to 0xffffffff."
333  */
334  if (code[2] == FR_DHCP_NAK) {
335  DEBUG("Reply will be broadcast due to NAK.");
336  socket.inet.dst_ipaddr.addr.v4.s_addr = INADDR_BROADCAST;
337  goto send_reply;
338  }
339 
340  /*
341  * The original packet has CIADDR, so we unicast
342  * the reply there.
343  *
344  * RFC 2131 page 23.
345  *
346  * "If the 'giaddr' field is zero and the
347  * 'ciaddr' field is nonzero, then the server
348  * unicasts DHCPOFFER and DHCPACK messages to the
349  * address in 'ciaddr'."
350  */
351  if (request->ciaddr != INADDR_ANY) {
352  DEBUG("Reply will be unicast to CIADDR from original packet.");
353  memcpy(&socket.inet.dst_ipaddr.addr.v4.s_addr, &request->ciaddr, 4);
354  goto send_reply;
355  }
356 
357  /*
358  * The original packet requested a broadcast
359  * reply, so we broadcast the reply.
360  *
361  * RFC 2131 page 23.
362  *
363  * "If 'giaddr' is zero and 'ciaddr' is zero, and
364  * the broadcast bit is set, then the server
365  * broadcasts DHCPOFFER and DHCPACK messages to
366  * 0xffffffff."
367  */
368  if (request->broadcast) {
369  DEBUG("Reply will be broadcast due to client request.");
370  socket.inet.dst_ipaddr.addr.v4.s_addr = INADDR_BROADCAST;
371  goto send_reply;
372  }
373 
374  /*
375  * The original packet was unicast to us, such as
376  * via a relay. We have a unicast destination
377  * address, so we just use that.
378  *
379  * This extension isn't in the RFC, but we find it useful.
380  */
381  if ((packet->yiaddr == htonl(INADDR_ANY)) &&
382  (socket.inet.dst_ipaddr.addr.v4.s_addr != htonl(INADDR_BROADCAST))) {
383  DEBUG("Reply will be unicast to source IP from original packet.");
384  goto send_reply;
385  }
386 
387  /*
388  * RFC 2131 page 23.
389  *
390  * "If the broadcast bit is not set and 'giaddr'
391  * is zero and 'ciaddr' is zero, then the server
392  * unicasts DHCPOFFER and DHCPACK messages to the
393  * client's hardware address and 'yiaddr'
394  * address."
395  */
396  switch (code[2]) {
397  /*
398  * OFFERs are sent to YIADDR if we
399  * received a unicast packet from YIADDR.
400  * Otherwise, they are unicast to YIADDR
401  * (if we can update ARP), otherwise they
402  * are broadcast.
403  */
404  case FR_DHCP_OFFER:
405  /*
406  * If the packet was unicast from the
407  * client, unicast it back without
408  * updating the ARP table. We presume
409  * that the ARP table has been updated by
410  * the OS, since we received a unicast
411  * packet.
412  *
413  * This check simply makes sure that we
414  * don't needlessly update the ARP table.
415  */
416  if (memcmp(&socket.inet.dst_ipaddr.addr.v4.s_addr, &packet->yiaddr, 4) == 0) {
417  DEBUG("Reply will be unicast to YIADDR.");
418 
419 #ifdef SIOCSARP
420  } else if (inst->broadcast && inst->interface) {
421  uint8_t macaddr[6];
422  uint8_t ipaddr[4];
423 
424  memcpy(&ipaddr, &packet->yiaddr, 4);
425  memcpy(&macaddr, &packet->chaddr, 6);
426 
427  /*
428  * Else the OFFER was broadcast.
429  * This socket is listening for
430  * broadcast packets on a
431  * particular interface. We're
432  * too lazy to write raw UDP
433  * packets, so we update our
434  * local ARP table and then
435  * unicast the reply.
436  */
437  if (fr_arp_entry_add(thread->sockfd, inst->interface, ipaddr, macaddr) == 0) {
438  DEBUG("Reply will be unicast to YIADDR, done ARP table updates.");
439  memcpy(&socket.inet.dst_ipaddr.addr.v4.s_addr, &packet->yiaddr, 4);
440  } else {
441  DEBUG("Failed adding ARP entry. Reply will be broadcast.");
442  socket.inet.dst_ipaddr.addr.v4.s_addr = INADDR_BROADCAST;
443  }
444 
445 #endif
446  } else {
447  DEBUG("Reply will be broadcast as we do not create raw UDP sockets.");
448  socket.inet.dst_ipaddr.addr.v4.s_addr = INADDR_BROADCAST;
449  }
450  break;
451 
452  /*
453  * ACKs are unicast to YIADDR
454  */
455  case FR_DHCP_ACK:
456  DEBUG("Reply will be unicast to YIADDR.");
457  memcpy(&socket.inet.dst_ipaddr.addr.v4.s_addr, &packet->yiaddr, 4);
458  break;
459 
460  default:
461  WARN("Silently discarding reply due to unimplemented message type %d", code[2]);
462  return 0;
463  }
464  }
465 
466 send_reply:
467  /*
468  * proto_dhcpv4 takes care of suppressing do-not-respond, etc.
469  */
470  data_size = udp_send(&socket, flags, buffer, buffer_len);
471 
472  /*
473  * This socket is dead. That's an error...
474  */
475  if (data_size <= 0) return data_size;
476 
477  return data_size;
478 }

500 /** Open a UDP listener for DHCPV4
501  *
502  */
503 static int mod_open(fr_listen_t *li)
504 {
505  proto_dhcpv4_udp_t const *inst = talloc_get_type_abort_const(li->app_io_instance, proto_dhcpv4_udp_t);
506  proto_dhcpv4_udp_thread_t *thread = talloc_get_type_abort(li->thread_instance, proto_dhcpv4_udp_thread_t);
507 
508  int sockfd, rcode;
509  uint16_t port = inst->port;
510 
511  li->fd = sockfd = fr_socket_server_udp(&inst->ipaddr, &port, inst->port_name, true);
512  if (sockfd < 0) {
513  PERROR("Failed opening UDP socket");
514  error:
515  return -1;
516  }
517 
518  li->app_io_addr = fr_socket_addr_alloc_inet_src(li, IPPROTO_UDP, 0, &inst->ipaddr, port);
519 
520  /*
521  * Set SO_REUSEPORT before bind, so that all packets can
522  * listen on the same destination IP address.
523  */
524  if (1) {
525  int on = 1;
526 
527  if (setsockopt(sockfd, SOL_SOCKET, SO_REUSEPORT, &on, sizeof(on)) < 0) {
528  ERROR("Failed to set socket 'reuseport': %s", fr_syserror(errno));
529  close(sockfd);
530  return -1;
531  }
532  }
533 
534  if (inst->broadcast) {
535  int on = 1;
536 
537  if (setsockopt(sockfd, SOL_SOCKET, SO_BROADCAST, &on, sizeof(on)) < 0) {
538  ERROR("Failed to set broadcast option: %s", fr_syserror(errno));
539  close(sockfd);
540  return -1;
541  }
542  }
543 
544  rcode = fr_socket_bind(sockfd, &inst->ipaddr, &port, inst->interface);
545  if (rcode < 0) {
546  close(sockfd);
547  PERROR("Failed binding socket");
548  goto error;
549  }
550 
551  thread->sockfd = sockfd;
552 
553  fr_assert((cf_parent(inst->cs) != NULL) && (cf_parent(cf_parent(inst->cs)) != NULL)); /* listen { ... } */
554 
555  thread->name = fr_app_io_socket_name(thread, &proto_dhcpv4_udp,
556  NULL, 0,
557  &inst->ipaddr, inst->port,
558  inst->interface);
559 
560  return 0;
561 }

584 static void *mod_track_create(TALLOC_CTX *ctx, uint8_t const *packet, size_t packet_len)
585 {
586  proto_dhcpv4_track_t *track;
587  dhcp_packet_t const *dhcp = (dhcp_packet_t const *) packet;
588  uint8_t const *option;
589 
590  option = fr_dhcpv4_packet_get_option(dhcp, packet_len, attr_message_type);
591  if (!option || (option[1] == 0)) {
592  DEBUG("No %s in the packet - ignoring", attr_message_type->name);
593  return NULL;
594  }
595 
596  track = talloc_zero(ctx, proto_dhcpv4_track_t);
597  if (!track) return NULL;
598 
599  memcpy(&track->xid, &dhcp->xid, sizeof(track->xid));
600 
601  track->message_type = option[2];
602 
603  /*
604  * Track most packets by chaddr. For lease queries, that
605  * field can be the client address being queried, not the
606  * address of the system which sent the packet. So
607  * instead for lease queries, we use giaddr, which MUST
608  * exist according to RFC 4388 Section 6.3
609  */
610  if (option[2] != FR_DHCP_LEASE_QUERY) {
611  if (dhcp->hlen == 6) memcpy(&track->chaddr, &dhcp->chaddr, 6);
612  }
613 
614  track->broadcast = ((dhcp->flags & FR_DHCP_FLAGS_VALUE_BROADCAST) != 0);
615  track->hops = dhcp->hops;
616  memcpy(&track->ciaddr, &dhcp->ciaddr, sizeof(track->ciaddr));
617  memcpy(&track->giaddr, &dhcp->giaddr, sizeof(track->giaddr));
618 
619  return track;
620 }

622 static int mod_compare(UNUSED void const *instance, UNUSED void *thread_instance, UNUSED RADCLIENT *client,
623  void const *one, void const *two)
624 {
625  int ret;
626  proto_dhcpv4_track_t const *a = one;
627  proto_dhcpv4_track_t const *b = two;
628 
629  /*
630  * The tree is ordered by XIDs, which are (hopefully)
631  * pseudo-randomly distributed.
632  */
633  ret = memcmp(&a->xid, &b->xid, sizeof(a->xid));
634  if (ret != 0) return ret;
635 
636  /*
637  * Hardware addresses should also be randomly distributed.
638  */
639  ret = memcmp(&a->chaddr, &b->chaddr, sizeof(a->chaddr));
640  if (ret != 0) return ret;
641 
642  /*
643  * Compare giaddr, but not ciaddr
644  */
645  ret = memcmp(&a->giaddr, &b->giaddr, sizeof(a->giaddr));
646  if (ret != 0) return ret;
647 
648  return (a->message_type < b->message_type) - (a->message_type > b->message_type);
649 }

659 static int mod_bootstrap(void *instance, CONF_SECTION *cs)
660 {
661  proto_dhcpv4_udp_t *inst = talloc_get_type_abort(instance, proto_dhcpv4_udp_t);
662  size_t num;
663  CONF_ITEM *ci;
664  CONF_SECTION *server_cs;
665  RADCLIENT *client;
666 
667  inst->cs = cs;
668 
669  /*
670  * Complain if no "ipaddr" is set.
671  */
672  if (inst->ipaddr.af == AF_UNSPEC) {
673  cf_log_err(cs, "No 'ipaddr' was specified in the 'udp' section");
674  return -1;
675  }
676 
677  if (inst->ipaddr.af != AF_INET) {
678  cf_log_err(cs, "DHCPv4 transport cannot use IPv6 for 'ipaddr'");
679  return -1;
680  }
681 
682  /*
683  * If src_ipaddr is defined, it must be of the same address family as "ipaddr"
684  */
685  if ((inst->src_ipaddr.af != AF_UNSPEC) &&
686  (inst->src_ipaddr.af != inst->ipaddr.af)) {
687  cf_log_err(cs, "Both 'ipaddr' and 'src_ipaddr' must be from the same address family");
688  return -1;
689  }
690 
691  /*
692  * Set src_ipaddr to INADDR_NONE if not otherwise specified
693  */
694  if (inst->src_ipaddr.af == AF_UNSPEC) {
695  memset(&inst->src_ipaddr, 0, sizeof(inst->src_ipaddr));
696  inst->src_ipaddr.af = AF_INET;
697  }
698 
699  if (inst->recv_buff_is_set) {
700  FR_INTEGER_BOUND_CHECK("recv_buff", inst->recv_buff, >=, 32);
701  FR_INTEGER_BOUND_CHECK("recv_buff", inst->recv_buff, <=, INT_MAX);
702  }
703 
704  FR_INTEGER_BOUND_CHECK("max_packet_size", inst->max_packet_size, >=, MIN_PACKET_SIZE);
705  FR_INTEGER_BOUND_CHECK("max_packet_size", inst->max_packet_size, <=, 65536);
706 
707  if (!inst->port) {
708  struct servent *s;
709 
710  if (!inst->port_name) {
711  cf_log_err(cs, "No 'port' was specified in the 'udp' section");
712  return -1;
713  }
714 
715  s = getservbyname(inst->port_name, "udp");
716  if (!s) {
717  cf_log_err(cs, "Unknown value for 'port_name = %s", inst->port_name);
718  return -1;
719  }
720 
721  inst->port = ntohl(s->s_port);
722  }
723 
724 #ifdef SIOCSARP
725  /*
726  * If we're listening for broadcast requests, we MUST
727  */
728  if (inst->broadcast && !inst->interface) {
729  cf_log_warn(cs, "You SHOULD set 'interface' if you have set 'broadcast = yes'.");
730  cf_log_warn(cs, "All replies will be broadcast, as ARP updates require 'interface' to be set.");
731  }
732 #endif
733 
734  /*
735  * Parse and create the trie for dynamic clients, even if
736  * there's no dynamic clients.
737  */
738  num = talloc_array_length(inst->allow);
739  if (!num) {
740  if (inst->dynamic_clients) {
741  cf_log_err(cs, "The 'allow' subsection MUST contain at least one 'network' entry when 'dynamic_clients = true'.");
742  return -1;
743  }
744  } else {
745  inst->trie = fr_master_io_network(inst, inst->ipaddr.af, inst->allow, inst->deny);
746  if (!inst->trie) {
747  cf_log_perr(cs, "Failed creating list of networks");
748  return -1;
749  }
750  }
751 
752  ci = cf_parent(inst->cs); /* listen { ... } */
753  fr_assert(ci != NULL);
754  ci = cf_parent(ci);
755  fr_assert(ci != NULL);
756 
757  server_cs = cf_item_to_section(ci);
758 
759  /*
760  * Look up local clients, if they exist.
761  *
762  * @todo - ensure that we only parse clients which are
763  * for IPPROTO_UDP, and don't require a "secret".
764  */
765  if (cf_section_find_next(server_cs, NULL, "client", CF_IDENT_ANY)) {
766  inst->clients = client_list_parse_section(server_cs, IPPROTO_UDP, false);
767  if (!inst->clients) {
768  cf_log_err(cs, "Failed creating local clients");
769  return -1;
770  }
771  }
772 
773  /*
774  * Create a fake client.
775  */
776  client = inst->default_client = talloc_zero(inst, RADCLIENT);
777  if (!inst->default_client) return 0;
778 
779  client->ipaddr.af = AF_INET;
780  client->ipaddr.addr.v4.s_addr = htonl(INADDR_NONE);
781  client->src_ipaddr = client->ipaddr;
782 
783  client->longname = client->shortname = client->secret = talloc_strdup(client, "default");
784  client->nas_type = talloc_strdup(client, "other");
785 
786  return 0;
787 }
