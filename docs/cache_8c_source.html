78 /** Execute the virtual server configured to perform cache actions
79  *
80  * @param[out] p_result from the virtual server.
81  * @param[in] request The current request.
82  * @param[in] action to execute.
83  */
84 unlang_action_t fr_tls_cache_process(rlm_rcode_t *p_result, request_t *request, CONF_SECTION *action)
85 {
86  rlm_rcode_t rcode;
87  CONF_SECTION *server_cs;
88  char const *module;
89  char const *component;
90 
91  if (!action) RETURN_MODULE_NOOP;
92 
93  /*
94  * Save the current status of the request.
95  */
96  server_cs = request->server_cs;
97  module = request->module;
98  component = request->component;
99  request->server_cs = cf_item_to_section(cf_parent(action));
100  request->module = NULL;
101 
102  /*
103  * Run it through the appropriate virtual server.
104  * FIXME - This is still blocking.
105  */
106  rcode = unlang_interpret_synchronous(request, action, RLM_MODULE_NOOP, true);
107 
108  /*
109  * Restore the original status of the request.
110  */
111  request->server_cs = server_cs;
112  request->module = module;
113  request->component = component;
114 
115  RETURN_MODULE_RCODE(rcode);
116 }

118 /** Pre-compile unlang cache sections and store pointers to them
119  *
120  * @param[out] actions Structure to hold pointers to sections.
121  * @param[in] server_cs to lookup sections in.
122  * @return
123  * - -1 on failure.
124  * - 0 on success.
125  */
126 int fr_tls_cache_compile(fr_tls_cache_t *actions, CONF_SECTION *server_cs)
127 {
128  bool found = false;
129 
130  if (!fr_cond_assert(server_cs)) return -1;
131 
132 #define CACHE_SECTION(_out, _verb, _name) \
133 do { \
134  CONF_SECTION *_tmp; \
135  _tmp = cf_section_find(server_cs, _verb, _name); \
136  if (_tmp) { \
137  if (unlang_compile(_tmp, MOD_AUTHORIZE, NULL, NULL) < 0) return -1; \
138  found = true; \
139  } \
140  if (actions) _out = _tmp; \
141 } while (0)
142 
143  CACHE_SECTION(actions->load, "load", "tls-session");
144  CACHE_SECTION(actions->store, "store", "tls-session");
145  CACHE_SECTION(actions->clear, "clear", "tls-session");
146 
147  /*
148  * Warn if we couldn't find any sections.
149  */
150  if (actions && !found) cf_log_warn(server_cs, "No TLS session cache sections found in "
151  "virtual server \"%s\"", cf_section_name2(server_cs));
152 
153  return 0;
154 }

169 /** Write a newly created session data to the tls_session structure
170  *
171  * @note If you hit an assert in this function, it was likely called twice, which shouldn't happen
172  * so blame OpenSSL.
173  *
174  * @param[in] ssl session state.
175  * @param[in] sess to serialise and write to the cache.
176  * @return 0. What we return is not used by OpenSSL to indicate success or failure,
177  * but to indicate whether it should free its copy of the session data.
178  */
179 static int fr_tls_cache_serialize(SSL *ssl, SSL_SESSION *sess)
180 {
181  request_t *request;
182  fr_tls_session_t *tls_session;
183  size_t len, rcode;
184 
185  uint8_t *p, *data = NULL;
186 
187  uint8_t const *key;
188  size_t key_len;
189 
190  request = SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_REQUEST);
191  tls_session = talloc_get_type_abort(SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_TLS_SESSION), fr_tls_session_t);
192 
193  /*
194  * This functions should only be called once during the lifetime
195  * of the tls_session, as the fields aren't re-populated on
196  * resumption.
197  */
198  fr_assert(!tls_session->session_id);
199  fr_assert(!tls_session->session_blob);
200 
201  key_len = fr_tls_cache_id(&key, sess);
202  if (key_len == 0) {
203  REDEBUG("Session ID buffer to small");
204  return 0;
205  }
206 
207  /* find out what length data we need */
208  len = i2d_SSL_SESSION(sess, NULL);
209  if (len < 1) {
210  /* something went wrong */
211  RWDEBUG("Session serialisation failed, couldn't determine required buffer length");
212  return 0;
213  }
214 
215  /* alloc and convert to ASN.1 */
216  data = talloc_array(tls_session, uint8_t, len);
217  if (!data) {
218  RWDEBUG("Session serialisation failed, couldn't allocate buffer (%zd bytes)", len);
219  return 0;
220  }
221 
222  /* openssl mutates &p */
223  p = data;
224  rcode = i2d_SSL_SESSION(sess, &p);
225  if (rcode != len) {
226  RWDEBUG("Session serialisation failed");
227  talloc_free(data);
228  return 0;
229  }
230 
231  /*
232  * Store the session blob and session id for writing
233  * later, once all the authentication phases have completed.
234  */
235  tls_session->session_id = talloc_memdup(tls_session, key, key_len);
236  if (!tls_session->session_id) {
237  talloc_free(data);
238  return 0;
239  }
240  tls_session->session_blob = data;
241 
242  return 0;
243 }

245 /** Call the specified virtual server to write session data to the cache
246  *
247  * @note Should be called after all authentication methods have completed.
248  *
249  * We do this here (instead of in fr_tls_cache_serialize),
250  * because we only want to write session data to the cache if all phases were successful.
251  *
252  * If we wrote out the cache data earlier, and the server exited whilst the session was in
253  * progress, the supplicant could resume the session (and get access) even if phase2
254  * never completed.
255  *
256  * @param[in] request The current request.
257  * @param[in] tls_session to write to the cache.
258  * @return
259  * - 1 noop.
260  * - 0 success.
261  * - -1 failed writing cached session.
262  */
263 int fr_tls_cache_write(request_t *request, fr_tls_session_t *tls_session)
264 {
265  fr_tls_conf_t *conf;
266  int ret = 0;
267  fr_pair_t *vp;
268  rlm_rcode_t rcode;
269 
270  conf = SSL_get_ex_data(tls_session->ssl, FR_TLS_EX_INDEX_CONF);
271 
272  if (!tls_session->session_blob || !tls_session->session_id) {
273  RDEBUG2("No session data available to cache");
274  return 1;
275  }
276 
277  if (fr_tls_cache_session_id_to_vp(request, tls_session->session_id,
278  talloc_array_length(tls_session->session_id)) < 0) {
279  RWDEBUG("Failed adding session key to the request");
280  return -1;
281  }
282 
283  /*
284  * Put the SSL data into an attribute.
285  */
286  MEM(vp = fr_pair_afrom_da(request->state_ctx, attr_tls_session_data));
287  fr_pair_value_memdup(vp, tls_session->session_blob, talloc_array_length(tls_session->session_blob), false);
288  RINDENT();
289  RDEBUG2("&session-state.%pP", vp);
290  REXDENT();
291  fr_pair_add(&request->state, vp);
292 
293  /*
294  * Call the virtual server to write the session
295  */
296  fr_tls_cache_process(&rcode, request, conf->session_cache.store);
297  switch (rcode) {
298  case RLM_MODULE_OK:
299  case RLM_MODULE_UPDATED:
300  break;
301 
302  default:
303  RWDEBUG("Failed storing session data");
304  ret = -1;
305  break;
306  }
307 
308  /*
309  * Ensure that the session data can't be used by anyone else.
310  */
311  fr_pair_delete_by_da(&request->state, attr_tls_session_data);
312 
313  return ret;
314 }

316 /** Read session data from the cache
317  *
318  * @param[in] ssl session state.
319  * @param[in] key to retrieve session data for.
320  * @param[in] key_len The length of the key.
321  * @param[out] copy Indicates whether OpenSSL should increment the reference
322  * count on SSL_SESSION to prevent it being automatically freed. We always
323  * set this to 0.
324  * @return
325  * - Deserialised session data on success.
326  * - NULL on error.
327  */
328 static SSL_SESSION *fr_tls_cache_read(SSL *ssl,
329 #if OPENSSL_VERSION_NUMBER >= 0x10100000L
330  unsigned char const *key,
331 #else
332  unsigned char *key,
333 #endif
334  int key_len, int *copy)
335 {
336  fr_tls_conf_t *conf;
337  request_t *request;
338  unsigned char const **p;
339  uint8_t const *q;
340  fr_pair_t *vp;
341  SSL_SESSION *sess;
342  rlm_rcode_t rcode;
343 
344  request = SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_REQUEST);
345  conf = SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_CONF);
346 
347  if (fr_tls_cache_session_id_to_vp(request, key, key_len) < 0) {
348  RWDEBUG("Failed adding session key to the request");
349  return NULL;
350  }
351 
352  *copy = 0;
353 
354  /*
355  * Call the virtual server to read the session
356  */
357  fr_tls_cache_process(&rcode, request, conf->session_cache.load);
358  switch (rcode) {
359  case RLM_MODULE_OK:
360  case RLM_MODULE_UPDATED:
361  break;
362 
363  default:
364  RWDEBUG("Failed acquiring session data");
365  return NULL;
366  }
367 
368  vp = fr_pair_find_by_da(&request->state, attr_tls_session_data);
369  if (!vp) {
370  RWDEBUG("No cached session found");
371  return NULL;
372  }
373 
374  q = vp->vp_octets; /* openssl will mutate q, so we can't use vp_octets directly */
375  p = (unsigned char const **)&q;
376 
377  sess = d2i_SSL_SESSION(NULL, p, vp->vp_length);
378  if (!sess) {
379  RWDEBUG("Failed loading persisted session: %s", ERR_error_string(ERR_get_error(), NULL));
380  return NULL;
381  }
382  RDEBUG3("Read %zu bytes of session data. Session deserialized successfully", vp->vp_length);
383 
384  /*
385  * OpenSSL's API is very inconsistent.
386  *
387  * We need to set external data here, so it can be
388  * retrieved in fr_tls_cache_delete.
389  *
390  * ex_data is not serialised in i2d_SSL_SESSION
391  * so we don't have to bother unsetting it.
392  */
393  SSL_SESSION_set_ex_data(sess, FR_TLS_EX_INDEX_TLS_SESSION, SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_TLS_SESSION));
394 
395  /*
396  * SSL_set_session increases the reference count
397  * on the session, so when OpenSSL attempts to
398  * free it, when setting our returned session
399  * it becomes a noop.
400  *
401  * Spent many hours trying to find a better place
402  * to do validation than this, but it seems
403  * like this is the only way.
404  */
405  SSL_set_session(ssl, sess);
406  if (fr_tls_validate_client_cert_chain(ssl) != 1) {
407  RWDEBUG("Validation failed, forcefully expiring resumed session");
408  SSL_SESSION_set_timeout(sess, 0);
409  }
410 
411  /*
412  * Ensure that the session data can't be used by anyone else.
413  */
414  fr_pair_delete_by_da(&request->state, attr_tls_session_data);
415 
416  return sess;
417 }

420  *
421  * @param[in] ctx Current ssl context.
422  * @param[in] sess to be deleted.
423  */
424 static void fr_tls_cache_delete(SSL_CTX *ctx, SSL_SESSION *sess)
425 {
426  fr_tls_conf_t *conf;
427  fr_tls_session_t *tls_session;
428  request_t *request;
429  uint8_t const *key;
430  ssize_t key_len;
431 
432  rlm_rcode_t rcode;
433 
434  conf = talloc_get_type_abort(SSL_CTX_get_app_data(ctx), fr_tls_conf_t);
435  tls_session = talloc_get_type_abort(SSL_SESSION_get_ex_data(sess, FR_TLS_EX_INDEX_TLS_SESSION), fr_tls_session_t);
436  request = talloc_get_type_abort(SSL_get_ex_data(tls_session->ssl, FR_TLS_EX_INDEX_REQUEST), request_t);
437 
438  /*
439  * Free any previously stored session blobs or data
440  */
441  TALLOC_FREE(tls_session->session_id);
442  TALLOC_FREE(tls_session->session_blob);
443 
444  key_len = fr_tls_cache_id(&key, sess);
445  if (key_len < 0) {
446  RWDEBUG("Session ID buffer too small");
447  error:
448  talloc_free(request);
449  return;
450  }
451 
452  if (fr_tls_cache_session_id_to_vp(request, key, (size_t)key_len) < 0) {
453  RWDEBUG("Failed adding session key to the request");
454  goto error;
455  }
456 
457  /*
458  * Call the virtual server to delete the session
459  */
460  fr_tls_cache_process(&rcode, request, conf->session_cache.clear);
461  switch (rcode) {
462  case RLM_MODULE_OK:
463  case RLM_MODULE_UPDATED:
464  case RLM_MODULE_NOTFOUND:
465  case RLM_MODULE_NOOP:
466  break;
467 
468  default:
469  RWDEBUG("Failed deleting session data");
470  goto error;
471  }
472 }

493 /** Prevent a TLS session from being resumed in future
494  *
495  * @note In OpenSSL > 1.1.0 this should not be called directly, but passed as a callback to
496  * SSL_CTX_set_not_resumable_session_callback.
497  *
498  * @param ssl The current OpenSSL session.
499  * @param is_forward_secure Whether the cipher is forward secure, pass -1 if unknown.
500  * @return
501  * - 0 on success.
502  * - 1 if enabling session resumption was disabled for this session.
503  */
504 int fr_tls_cache_disable_cb(SSL *ssl,
505 #if OPENSSL_VERSION_NUMBER < 0x10100000L
506  UNUSED
507 #endif
508  int is_forward_secure)
509 {
510  request_t *request;
511 
512  fr_tls_session_t *session;
513  fr_pair_t *vp;
514 
515  session = talloc_get_type_abort(SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_TLS_SESSION), fr_tls_session_t);
516  request = talloc_get_type_abort(SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_REQUEST), request_t);
517 
518 #if OPENSSL_VERSION_NUMBER >= 0x10100000L
519  {
520  fr_tls_conf_t *conf;
521 
522  conf = talloc_get_type_abort(SSL_get_ex_data(ssl, FR_TLS_EX_INDEX_CONF), fr_tls_conf_t);
523  if (conf->session_cache_require_extms && (SSL_get_extms_support(session->ssl) != 1)) {
524  RDEBUG2("Client does not support the Extended Master Secret extension, "
525  "disabling session resumption");
526  goto disable;
527  }
528 
529  if (conf->session_cache_require_pfs && !is_forward_secure) {
530  RDEBUG2("Cipher suite is not forward secure, disabling session resumption");
531  goto disable;
532  }
533  }
534 #endif
535 
536  /*
537  * If there's no session resumption, delete the entry
538  * from the cache. This means either it's disabled
539  * globally for this SSL context, OR we were told to
540  * disable it for this user.
541  *
542  * This also means you can't turn it on just for one
543  * user.
544  */
545  if (!session->allow_session_resumption) goto disable;
546 
547  vp = fr_pair_find_by_da(&request->control_pairs, attr_allow_session_resumption);
548  if (vp && (vp->vp_uint32 == 0)) {
549  RDEBUG2("&control.Allow-Session-Resumption == no, disabling session resumption");
550  disable:
551  SSL_CTX_remove_session(session->ctx, session->session);
552  session->allow_session_resumption = false;
553  return 1;
554  }
555 
556  return 0;
557 }

559 /** Sets callbacks on a SSL_CTX to enable/disable session resumption
560  *
561  * @param ctx to modify.
562  * @param enabled Whether session caching should be enabled.
563  * @param lifetime The maximum period a cached session remains
564  * valid for.
565  */
566 void fr_tls_cache_init(SSL_CTX *ctx, bool enabled, uint32_t lifetime)
567 {
568  if (!enabled) {
569 #if OPENSSL_VERSION_NUMBER >= 0x10101000L
570  /*
571  * This controls the number of stateful or stateless tickets
572  * generated with TLS 1.3. In OpenSSL 1.1.0 it's also
573  * required to disable sending session tickets,
574  * SSL_SESS_CACHE_OFF is not good enough.
575  */
576  SSL_CTX_set_num_tickets(ctx, 0);
577 #endif
578  SSL_CTX_set_session_cache_mode(ctx, SSL_SESS_CACHE_OFF);
579  return;
580  }
581 
582  SSL_CTX_sess_set_new_cb(ctx, fr_tls_cache_serialize);
583  SSL_CTX_sess_set_get_cb(ctx, fr_tls_cache_read);
584  SSL_CTX_sess_set_remove_cb(ctx, fr_tls_cache_delete);
585  SSL_CTX_set_quiet_shutdown(ctx, 1);
586 
587  SSL_CTX_set_session_cache_mode(ctx, SSL_SESS_CACHE_SERVER | SSL_SESS_CACHE_NO_INTERNAL);
588  SSL_CTX_set_timeout(ctx, lifetime);
589 
590 #if OPENSSL_VERSION_NUMBER >= 0x10101000L
591  SSL_CTX_set_num_tickets(ctx, 1);
592 #endif
593 #if OPENSSL_VERSION_NUMBER >= 0x10100000L
594  SSL_CTX_set_not_resumable_session_callback(ctx, fr_tls_cache_disable_cb);
595 #endif
596 }
