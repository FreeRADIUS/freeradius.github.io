69 /** Apply a list of modifications on one or more fr_pair_t lists.
70  *
71  * @param[in] request The current request.
72  * @param[out] p_result The rcode indicating what the result
73  * of the operation was.
74  * @return
75  * - UNLANG_ACTION_CALCULATE_RESULT changes were applied.
76  * - UNLANG_ACTION_PUSHED_CHILD async execution of an expansion is required.
77  */
78 static unlang_action_t list_mod_apply(rlm_rcode_t *p_result, request_t *request)
79 {
80  unlang_stack_t *stack = request->stack;
81  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
82  unlang_frame_state_update_t *update_state = frame->state;
83  vp_list_mod_t const *vlm;
84 
85  /*
86  * No modifications...
87  */
88  if (!update_state->vlm_head) {
89  RDEBUG2("Nothing to update");
90  goto done;
91  }
92 
93  /*
94  * Apply the list of modifications. This should not fail
95  * except on memory allocation error.
96  */
97  for (vlm = update_state->vlm_head;
98  vlm;
99  vlm = vlm->next) {
100  int ret;
101 
102  ret = map_list_mod_apply(request, vlm);
103  if (!fr_cond_assert(ret == 0)) {
104  TALLOC_FREE(frame->state);
105 
106  *p_result = RLM_MODULE_FAIL;
107  return UNLANG_ACTION_CALCULATE_RESULT;
108  }
109  }
110 
111 done:
112  *p_result = RLM_MODULE_NOOP;
113  return UNLANG_ACTION_CALCULATE_RESULT;
114 }

116 /** Create a list of modifications to apply to one or more fr_pair_t lists
117  *
118  * @param[in] request The current request.
119  * @param[out] p_result The rcode indicating what the result
120  * of the operation was.
121  * @return
122  * - UNLANG_ACTION_CALCULATE_RESULT changes were applied.
123  * - UNLANG_ACTION_PUSHED_CHILD async execution of an expansion is required.
124  */
125 static unlang_action_t list_mod_create(rlm_rcode_t *p_result, request_t *request)
126 {
127  unlang_stack_t *stack = request->stack;
128  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
129  unlang_frame_state_update_t *update_state = frame->state;
130  map_t *map;
131 
132  /*
133  * Iterate over the maps producing a set of modifications to apply.
134  */
135  for (map = fr_cursor_current(&update_state->maps);
136  map;
137  map = fr_cursor_next(&update_state->maps)) {
138  switch (update_state->state) {
139  case UNLANG_UPDATE_MAP_INIT:
140  update_state->state = UNLANG_UPDATE_MAP_EXPANDED_LHS;
141 
142  fr_assert(!update_state->lhs_result); /* Should have been consumed */
143  fr_assert(!update_state->rhs_result); /* Should have been consumed */
144 
145  switch (map->lhs->type) {
146  default:
147  break;
148 
149  case TMPL_TYPE_EXEC:
150  if (unlang_tmpl_push(update_state, &update_state->lhs_result,
151  request, map->lhs, NULL, NULL) < 0) {
152  *p_result = RLM_MODULE_FAIL;
153  return UNLANG_ACTION_STOP_PROCESSING;
154  }
155  return UNLANG_ACTION_PUSHED_CHILD;
156 
157  case TMPL_TYPE_XLAT:
158  if (unlang_xlat_push(update_state, &update_state->lhs_result,
159  request, tmpl_xlat(map->lhs), false) < 0) {
160  *p_result = RLM_MODULE_FAIL;
161  return UNLANG_ACTION_STOP_PROCESSING;
162  }
163  return UNLANG_ACTION_PUSHED_CHILD;
164 
165  case TMPL_TYPE_REGEX_XLAT_UNRESOLVED:
166  case TMPL_TYPE_REGEX:
167  case TMPL_TYPE_REGEX_UNCOMPILED:
168  case TMPL_TYPE_REGEX_XLAT:
169  case TMPL_TYPE_XLAT_UNRESOLVED:
170  fr_assert(0);
171  error:
172  TALLOC_FREE(frame->state);
173 
174  *p_result = RLM_MODULE_FAIL;
175 
176  return UNLANG_ACTION_CALCULATE_RESULT;
177  }
178  FALL_THROUGH;
179 
180  case UNLANG_UPDATE_MAP_EXPANDED_LHS:
181  /*
182  * map_to_list_mod() already concatentates the LHS, so we don't need to do it here.
183  */
184  if (!map->rhs) goto next;
185 
186  update_state->state = UNLANG_UPDATE_MAP_EXPANDED_RHS;
187 
188  switch (map->rhs->type) {
189  default:
190  break;
191 
192  case TMPL_TYPE_EXEC:
193  if (unlang_tmpl_push(update_state, &update_state->rhs_result,
194  request, map->rhs, NULL, NULL) < 0) {
195  *p_result = RLM_MODULE_FAIL;
196  return UNLANG_ACTION_STOP_PROCESSING;
197  }
198  return UNLANG_ACTION_PUSHED_CHILD;
199 
200  case TMPL_TYPE_XLAT:
201  if (unlang_xlat_push(update_state, &update_state->rhs_result,
202  request, tmpl_xlat(map->rhs), false) < 0) {
203  *p_result = RLM_MODULE_FAIL;
204  return UNLANG_ACTION_STOP_PROCESSING;
205  }
206  return UNLANG_ACTION_PUSHED_CHILD;
207 
208  case TMPL_TYPE_REGEX:
209  case TMPL_TYPE_REGEX_UNCOMPILED:
210  case TMPL_TYPE_REGEX_XLAT:
211  case TMPL_TYPE_REGEX_XLAT_UNRESOLVED:
212  case TMPL_TYPE_XLAT_UNRESOLVED:
213  fr_assert(0);
214  goto error;
215  }
216  FALL_THROUGH;
217 
218  case UNLANG_UPDATE_MAP_EXPANDED_RHS:
219  /*
220  * Concat the top level results together
221  */
222  if (update_state->rhs_result &&
223  (fr_value_box_list_concat(update_state, update_state->rhs_result, &update_state->rhs_result,
224  FR_TYPE_STRING, true) < 0)) {
225  RPEDEBUG("Failed concatenating RHS expansion results");
226  goto error;
227  }
228 
229  if (map_to_list_mod(update_state, update_state->vlm_next,
230  request, map,
231  &update_state->lhs_result, &update_state->rhs_result) < 0) goto error;
232 
233  talloc_list_free(&update_state->rhs_result);
234 
235  next:
236  update_state->state = UNLANG_UPDATE_MAP_INIT;
237  talloc_list_free(&update_state->lhs_result);
238 
239  /*
240  * Wind to the end...
241  */
242  while (*update_state->vlm_next) update_state->vlm_next = &(*(update_state->vlm_next))->next;
243  break;
244  }
245  };
246 
247  return list_mod_apply(p_result, request);
248 }

251 /** Execute an update block
252  *
253  * Update blocks execute in two phases, first there's an evaluation phase where
254  * each input map is evaluated, outputting one or more modification maps. The modification
255  * maps detail a change that should be made to a list in the current request.
256  * The request is not modified during this phase.
257  *
258  * The second phase applies those modification maps to the current request.
259  * This re-enables the atomic functionality of update blocks provided in v2.x.x.
260  * If one map fails in the evaluation phase, no more maps are processed, and the current
261  * result is discarded.
262  */
263 static unlang_action_t unlang_update_state_init(rlm_rcode_t *p_result, request_t *request)
264 {
265  unlang_stack_t *stack = request->stack;
266  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
267  unlang_t *instruction = frame->instruction;
268 
269  unlang_group_t *g = unlang_generic_to_group(instruction);
270  unlang_map_t *gext = unlang_group_to_map(g);
271  unlang_frame_state_update_t *update_state;
272 
273  /*
274  * Initialise the frame state
275  */
276  MEM(frame->state = update_state = talloc_zero_pooled_object(stack, unlang_frame_state_update_t,
277  (sizeof(map_t) +
278  (sizeof(tmpl_t) * 2) + 128),
279  g->num_children)); /* 128 is for string buffers */
280 
281  fr_cursor_init(&update_state->maps, &gext->map);
282  update_state->vlm_next = &update_state->vlm_head;
283  repeatable_set(frame);
284 
285  /*
286  * Call list_mod_create
287  */
288  frame->process = list_mod_create;
289  return list_mod_create(p_result, request);
290 }

293 static unlang_action_t map_proc_apply(rlm_rcode_t *p_result, request_t *request)
294 {
295  unlang_stack_t *stack = request->stack;
296  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
297  unlang_t *instruction = frame->instruction;
298 
299  unlang_group_t *g = unlang_generic_to_group(instruction);
300  unlang_map_t *gext = unlang_group_to_map(g);
301 
302  map_proc_inst_t *inst = gext->proc_inst;
303  unlang_frame_state_map_proc_t *map_proc_state = talloc_get_type_abort(frame->state, unlang_frame_state_map_proc_t);
304 
305  RDEBUG2("MAP %s \"%pM\"", inst->proc->name, map_proc_state->src_result);
306 
307  /*
308  * FIXME - We don't yet support async LHS/RHS expansions for map procs
309  */
310 #ifndef NDEBUG
311  if (map_proc_state->src_result) talloc_list_get_type_abort(map_proc_state->src_result, fr_value_box_t);
312 #endif
313  *p_result = map_proc(request, gext->proc_inst, &map_proc_state->src_result);
314 #ifndef NDEBUG
315  if (map_proc_state->src_result) talloc_list_get_type_abort(map_proc_state->src_result, fr_value_box_t);
316 #endif
317 
318  return *p_result == RLM_MODULE_YIELD ? UNLANG_ACTION_YIELD : UNLANG_ACTION_CALCULATE_RESULT;
319 }

321 static unlang_action_t unlang_map_state_init(rlm_rcode_t *p_result, request_t *request)
322 {
323  unlang_stack_t *stack = request->stack;
324  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
325  unlang_t *instruction = frame->instruction;
326  unlang_group_t *g = unlang_generic_to_group(instruction);
327  unlang_map_t *gext = unlang_group_to_map(g);
328  map_proc_inst_t *inst = gext->proc_inst;
329  unlang_frame_state_map_proc_t *map_proc_state = talloc_get_type_abort(frame->state, unlang_frame_state_map_proc_t);
330 
331  /*
332  * Initialise the frame state
333  */
334  repeatable_set(frame);
335 
336  /*
337  * Set this BEFORE doing anything else, as we will be
338  * called again after unlang_xlat_push() returns.
339  */
340  frame->process = map_proc_apply;
341 
342  /*
343  * Expand the map source
344  */
345  if (inst->src) switch (inst->src->type) {
346  default:
347  if (tmpl_aexpand(frame->state, &map_proc_state->src_result,
348  request, inst->src, NULL, NULL) < 0) {
349  REDEBUG("Failed expanding map src");
350  error:
351  *p_result = RLM_MODULE_FAIL;
352  return UNLANG_ACTION_CALCULATE_RESULT;
353  }
354  break;
355 
356  case TMPL_TYPE_EXEC:
357  if (unlang_tmpl_push(map_proc_state, &map_proc_state->src_result,
358  request, inst->src, NULL, NULL) < 0) {
359  *p_result = RLM_MODULE_FAIL;
360  return UNLANG_ACTION_STOP_PROCESSING;
361  }
362  return UNLANG_ACTION_PUSHED_CHILD;
363 
364  case TMPL_TYPE_XLAT:
365  if (unlang_xlat_push(map_proc_state, &map_proc_state->src_result,
366  request, tmpl_xlat(inst->src), false) < 0) {
367  *p_result = RLM_MODULE_FAIL;
368  return UNLANG_ACTION_STOP_PROCESSING;
369  }
370  return UNLANG_ACTION_PUSHED_CHILD;
371 
372 
373  case TMPL_TYPE_REGEX:
374  case TMPL_TYPE_REGEX_UNCOMPILED:
375  case TMPL_TYPE_REGEX_XLAT:
376  case TMPL_TYPE_REGEX_XLAT_UNRESOLVED:
377  case TMPL_TYPE_XLAT_UNRESOLVED:
378  fr_assert(0);
379  goto error;
380  }
381 
382  return map_proc_apply(p_result, request);
383 }
