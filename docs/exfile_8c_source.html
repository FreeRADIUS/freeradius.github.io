60 /** Send an exfile trigger.
61  *
62  * @param[in] ef to send trigger for.
63  * @param[in] request The current request.
64  * @param[in] entry for the file that the event occurred on.
65  * @param[in] name_suffix trigger name suffix.
66  */
67 static inline void exfile_trigger_exec(exfile_t *ef, request_t *request, exfile_entry_t *entry, char const *name_suffix)
68 {
69  char name[128];
70  fr_pair_t *vp, *args;
71  fr_dict_attr_t const *da;
72  fr_cursor_t cursor;
73 
74  fr_assert(ef != NULL);
75  fr_assert(name_suffix != NULL);
76 
77  if (!ef->trigger_prefix) return;
78 
79  da = fr_dict_attr_child_by_num(fr_dict_root(fr_dict_internal()), FR_EXFILE_NAME);
80  if (!da) {
81  ROPTIONAL(RERROR, ERROR, "Incomplete internal dictionary: Missing definition for \"Exfile-Name\"");
82  return;
83  }
84 
85  args = ef->trigger_args;
86  fr_cursor_init(&cursor, &args);
87 
88  MEM(vp = fr_pair_afrom_da(NULL, da));
89  fr_pair_value_strdup(vp, entry->filename);
90 
91  fr_cursor_prepend(&cursor, vp);
92 
93  snprintf(name, sizeof(name), "%s.%s", ef->trigger_prefix, name_suffix);
94  trigger_exec(request, ef->conf, name, false, args);
95 
96  talloc_free(vp);
97 }

100 static void exfile_cleanup_entry(exfile_t *ef, request_t *request, exfile_entry_t *entry)
101 {
102  if (entry->fd >= 0) close(entry->fd);
103 
104  entry->hash = 0;
105  entry->fd = -1;
106 
107  /*
108  * Issue close trigger *after* we've closed the fd
109  */
110  exfile_trigger_exec(ef, request, entry, "close");
111 
112  /*
113  * Trigger still needs access to filename to populate Exfile-Name
114  */
115  TALLOC_FREE(entry->filename);
116 }

119 static int _exfile_free(exfile_t *ef)
120 {
121  uint32_t i;
122 
123  pthread_mutex_lock(&ef->mutex);
124 
125  for (i = 0; i < ef->max_entries; i++) {
126  if (!ef->entries[i].filename) continue;
127 
128  exfile_cleanup_entry(ef, NULL, &ef->entries[i]);
129  }
130 
131  pthread_mutex_unlock(&ef->mutex);
132  pthread_mutex_destroy(&ef->mutex);
133 
134  return 0;

137 /** Initialize a way for multiple threads to log to one or more files.
138  *
139  * @param ctx The talloc context
140  * @param max_entries Max file descriptors to cache, and manage locks for.
141  * @param max_idle Maximum time a file descriptor can be idle before it's closed.
142  * @param locking whether or not to lock the files.
143  * @return
144  * - new context.
145  * - NULL on error.
146  */
147 exfile_t *exfile_init(TALLOC_CTX *ctx, uint32_t max_entries, uint32_t max_idle, bool locking)
148 {
149  exfile_t *ef;
150 
151  ef = talloc_zero(NULL, exfile_t);
152  if (!ef) return NULL;
153 
154  talloc_link_ctx(ctx, ef);
155 
156  ef->max_entries = max_entries;
157  ef->max_idle = max_idle;
158  ef->locking = locking;
159 
160  /*
161  * If we're not locking the files, just return the
162  * handle. Each call to exfile_open() will just open a
163  * new file descriptor.
164  */
165  if (!ef->locking) return ef;
166 
167  ef->entries = talloc_zero_array(ef, exfile_entry_t, max_entries);
168  if (!ef->entries) {
169  talloc_free(ef);
170  return NULL;
171  }
172 
173  if (pthread_mutex_init(&ef->mutex, NULL) != 0) {
174  talloc_free(ef);
175  return NULL;
176  }
177 
178  talloc_set_destructor(ef, _exfile_free);
179 
180  return ef;
181 }

209 /*
210  * Try to open the file. It it doesn't exist, try to
211  * create it's parent directories.
212  */
213 static int exfile_open_mkdir(exfile_t *ef, char const *filename, mode_t permissions)
214 {
215  int fd;
216 
217  fd = open(filename, O_RDWR | O_CREAT, permissions);
218  if (fd < 0) {
219  mode_t dirperm;
220  char *p, *dir;
221 
222  /*
223  * Maybe the directory doesn't exist. Try to
224  * create it.
225  */
226  dir = talloc_typed_strdup(ef, filename);
227  if (!dir) return -1;
228  p = strrchr(dir, FR_DIR_SEP);
229  if (!p) {
230  fr_strerror_printf("No '/' in '%s'", filename);
231  talloc_free(dir);
232  return -1;
233  }
234  *p = '\0';
235 
236  /*
237  * Ensure that the 'x' bit is set, so that we can
238  * read the directory.
239  */
240  dirperm = permissions;
241  if ((dirperm & 0600) != 0) dirperm |= 0100;
242  if ((dirperm & 0060) != 0) dirperm |= 0010;
243  if ((dirperm & 0006) != 0) dirperm |= 0001;
244 
245  if (fr_mkdir(NULL, dir, -1, dirperm, NULL, NULL) < 0) {
246  fr_strerror_printf("Failed to create directory %s: %s", dir, fr_syserror(errno));
247  talloc_free(dir);
248  return -1;
249  }
250  talloc_free(dir);
251 
252  fd = open(filename, O_RDWR | O_CREAT, permissions);
253  if (fd < 0) {
254  fr_strerror_printf("Failed to open file %s: %s", filename, fr_syserror(errno));
255  return -1;
256  }
257  }
258 
259  return fd;
260 }

263 /** Open a new log file, or maybe an existing one.
264  *
265  * When multithreaded, the FD is locked via a mutex. This way we're
266  * sure that no other thread is writing to the file.
267  *
268  * @param ef The logfile context returned from exfile_init().
269  * @param request The current request.
270  * @param filename the file to open.
271  * @param permissions to use.
272  * @return
273  * - FD used to write to the file.
274  * - -1 on failure.
275  */
276 int exfile_open(exfile_t *ef, request_t *request, char const *filename, mode_t permissions)
277 {
278  int i, tries, unused = -1, found = -1, oldest = -1;
279  bool do_cleanup = false;
280  uint32_t hash;
281  time_t now;
282  struct stat st;
283 
284  if (!ef || !filename) return -1;
285 
286  /*
287  * No locking: just return a new FD.
288  */
289  if (!ef->locking) {
290  found = exfile_open_mkdir(ef, filename, permissions);
291  if (found < 0) return -1;
292 
293  (void) lseek(found, 0, SEEK_END);
294  return found;
295  }
296 
297  /*
298  * It's faster to do hash comparisons of a string than
299  * full string comparisons.
300  */
301  hash = fr_hash_string(filename);
302  now = time(NULL);
303  unused = -1;
304 
305  pthread_mutex_lock(&ef->mutex);
306 
307  if (now > (ef->last_cleaned + 1)) do_cleanup = true;
308 
309  /*
310  * Find the matching entry, or an unused one.
311  *
312  * Also track which entry is the oldest, in case there
313  * are no unused entries.
314  */
315  for (i = 0; i < (int) ef->max_entries; i++) {
316  if (!ef->entries[i].filename) {
317  if (unused < 0) unused = i;
318  continue;
319  }
320 
321  if ((oldest < 0) ||
322  (ef->entries[i].last_used < ef->entries[oldest].last_used)) {
323  oldest = i;
324  }
325 
326  /*
327  * Hash comparisons are fast. String comparisons are slow.
328  *
329  * But we still need to do string comparisons if
330  * the hash matches, because 1/2^16 filenames
331  * will result in a hash collision. And that's
332  * enough filenames in a long-running server to
333  * ensure that it happens.
334  */
335  if ((found < 0) &&
336  (ef->entries[i].hash == hash) &&
337  (strcmp(ef->entries[i].filename, filename) == 0)) {
338  found = i;
339 
340  /*
341  * If we're not cleaning up, stop now.
342  */
343  if (!do_cleanup) break;
344 
345  /*
346  * If we are cleaning up, then clean up
347  * entries OTHER than the one we found,
348  * do so now.
349  */
350  } else if (do_cleanup) {
351  if ((ef->entries[i].last_used + ef->max_idle) >= now) continue;
352 
353  exfile_cleanup_entry(ef, request, &ef->entries[i]);
354  }
355  }
356 
357  if (do_cleanup) ef->last_cleaned = now;
358 
359  /*
360  * We found an existing entry, return that.
361  */
362  if (found >= 0) {
363  i = found;
364 
365  /*
366  * Stat the *filename*, not the file we opened.
367  * If that's not the file we opened, then go back
368  * and re-open the file.
369  */
370  if (stat(ef->entries[i].filename, &st) < 0) {
371  goto reopen;
372  }
373 
374  if ((st.st_dev != ef->entries[i].st_dev) ||
375  (st.st_ino != ef->entries[i].st_ino)) {
376  close(ef->entries[i].fd);
377  goto reopen;
378  }
379 
380  goto try_lock;
381  }
382 
383  /*
384  * There are no unused entries, free the oldest one.
385  */
386  if (unused < 0) {
387  exfile_cleanup_entry(ef, request, &ef->entries[oldest]);
388  unused = oldest;
389  }
390 
391  /*
392  * Create a new entry.
393  */
394  i = unused;
395 
396  ef->entries[i].hash = hash;
397  ef->entries[i].filename = talloc_typed_strdup(ef->entries, filename);
398  ef->entries[i].fd = -1;
399 
400 reopen:
401  ef->entries[i].fd = exfile_open_mkdir(ef, filename, permissions);
402  if (ef->entries[i].fd < 0) goto error;
403 
404  exfile_trigger_exec(ef, request, &ef->entries[i], "open");
405 
406 try_lock:
407  /*
408  * Lock from the start of the file.
409  */
410  if (lseek(ef->entries[i].fd, 0, SEEK_SET) < 0) {
411  fr_strerror_printf("Failed to seek in file %s: %s", filename, fr_syserror(errno));
412 
413  error:
414  exfile_cleanup_entry(ef, request, &ef->entries[i]);
415  pthread_mutex_unlock(&(ef->mutex));
416  return -1;
417  }
418 
419  /*
420  * Try to lock it. If we can't lock it, it's because
421  * some reader has re-named the file to "foo.work" and
422  * locked it. So, we close the current file, re-open it,
423  * and try again/
424  */
425  for (tries = 0; tries < MAX_TRY_LOCK; tries++) {
426  if (rad_lockfd_nonblock(ef->entries[i].fd, 0) >= 0) break;
427 
428  if (errno != EAGAIN) {
429  fr_strerror_printf("Failed to lock file %s: %s", filename, fr_syserror(errno));
430  goto error;
431  }
432 
433  close(ef->entries[i].fd);
434  ef->entries[i].fd = open(filename, O_RDWR | O_CREAT, permissions);
435  if (ef->entries[i].fd < 0) {
436  fr_strerror_printf("Failed to open file %s: %s", filename, fr_syserror(errno));
437  goto error;
438  }
439  }
440 
441  if (tries >= MAX_TRY_LOCK) {
442  fr_strerror_printf("Failed to lock file %s: too many tries", filename);
443  goto error;
444  }
445 
446  /*
447  * Maybe someone deleted the file while we were waiting
448  * for the lock. If so, re-open it.
449  */
450  if (fstat(ef->entries[i].fd, &st) < 0) {
451  fr_strerror_printf("Failed to stat file %s: %s", filename, fr_syserror(errno));
452  goto reopen;
453  }
454 
455  if (st.st_nlink == 0) {
456  close(ef->entries[i].fd);
457  goto reopen;
458  }
459 
460  /*
461  * Remember which device and inode this file is
462  * for.
463  */
464  ef->entries[i].st_dev = st.st_dev;
465  ef->entries[i].st_ino = st.st_ino;
466 
467  /*
468  * Sometimes the file permissions are changed externally.
469  * just be sure to update the permission if necessary.
470  */
471  if ((st.st_mode & ~S_IFMT) != permissions) {
472  char str_need[10], oct_need[5];
473  char str_have[10], oct_have[5];
474 
475  rad_mode_to_oct(oct_need, permissions);
476  rad_mode_to_str(str_need, permissions);
477 
478  rad_mode_to_oct(oct_have, st.st_mode & ~S_IFMT);
479  rad_mode_to_str(str_have, st.st_mode & ~S_IFMT);
480 
481  WARN("File %s permissions are %s (%s) not %s (%s))", filename,
482  oct_have, str_have, oct_need, str_need);
483 
484  if (((st.st_mode | permissions) != st.st_mode) &&
485  (fchmod(ef->entries[i].fd, (st.st_mode & ~S_IFMT) | permissions) < 0)) {
486  rad_mode_to_oct(oct_need, (st.st_mode & ~S_IFMT) | permissions);
487  rad_mode_to_str(str_need, (st.st_mode & ~S_IFMT) | permissions);
488 
489  WARN("Failed resetting file %s permissions to %s (%s): %s",
490  filename, oct_need, str_need, fr_syserror(errno));
491  }
492  }
493 
494  /*
495  * Seek to the end of the file before returning the FD to
496  * the caller.
497  */
498  (void) lseek(ef->entries[i].fd, 0, SEEK_END);
499 
500  /*
501  * Return holding the mutex for the entry.
502  */
503  ef->entries[i].last_used = now;
504 
505  exfile_trigger_exec(ef, request, &ef->entries[i], "reserve");
506 
507  /* coverity[missing_unlock] */
508  return ef->entries[i].fd;
509 }

511 /** Close the log file. Really just return it to the pool.
512  *
513  * When multithreaded, the FD is locked via a mutex. This way we're sure that no other thread is
514  * writing to the file. This function will unlock the mutex, so that other threads can write to
515  * the file.
516  *
517  * @param ef The logfile context returned from #exfile_init.
518  * @param request The current request.
519  * @param fd the FD to close (i.e. return to the pool).
520  * @return
521  * - 0 on success.
522  * - -1 on failure.
523  */
524 int exfile_close(exfile_t *ef, request_t *request, int fd)
525 {
526  uint32_t i;
527 
528  /*
529  * No locking: just close the file.
530  */
531  if (!ef->locking) {
532  close(fd);
533  return 0;
534  }
535 
536  /*
537  * Unlock the bytes that we had previously locked.
538  */
539  for (i = 0; i < ef->max_entries; i++) {
540  if (ef->entries[i].fd != fd) continue;
541 
542  (void) lseek(ef->entries[i].fd, 0, SEEK_SET);
543  (void) rad_unlockfd(ef->entries[i].fd, 0);
544  pthread_mutex_unlock(&(ef->mutex));
545 
546  exfile_trigger_exec(ef, request, &ef->entries[i], "release");
547  return 0;
548  }
549 
550  pthread_mutex_unlock(&(ef->mutex));
551 
552  fr_strerror_printf("Attempt to unlock file which is not tracked");
553  return -1;
554 }
