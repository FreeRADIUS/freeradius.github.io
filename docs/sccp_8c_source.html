61 static int sigtran_txn_cmp(void const *a, void const *b)
62 {
63  sigtran_transaction_t const *a_tx = a; /* May be stack allocated */
64  sigtran_transaction_t const *b_tx = b; /* May be stack allocated */
65 
66  if (a_tx->ctx.otid > b_tx->ctx.otid) return +1;
67  if (a_tx->ctx.otid < b_tx->ctx.otid) return -1;
68 
69  if (a_tx->ctx.invoke_id > b_tx->ctx.invoke_id) return +1;
70  if (a_tx->ctx.invoke_id < b_tx->ctx.invoke_id) return -1;
71 
72  return 0;
73 }

219 static int sigtran_tcap_incoming(struct msgb *msg, UNUSED unsigned int length, UNUSED void *ctx)
220 {
221  sigtran_vector_t *vec;
222  uint8_t *tcap = msg->l3h;
223  uint8_t *p, *end;
224  size_t len = (size_t)msgb_l3len(msg);
225 
226  sigtran_transaction_t find, *found;
227  sigtran_transaction_t *txn;
228 
229  sigtran_map_send_auth_info_req_t *req;
230  sigtran_map_send_auth_info_res_t *res;
231 
232  struct osmo_fd *ofd;
233  sigtran_vector_t **last;
234 
235  memset(&find, 0, sizeof(find));
236 
237 // sigtran_conn_t *conn = talloc_get_type_abort(ctx, sigtran_conn_t);
238 
239  DEBUG3("Got %zu bytes of L4 data", (size_t)msgb_l3len(msg));
240 // log_request_hex(L_DBG, L_DBG_LVL_3, request, msg->l3h, (size_t)msgb_l3len(msg));
241 
242  find.ctx.otid = *(msg->l3h + 0x5);
243 
244  // find.ctx.invoke_id = *(msg->l3h + 0x34);
245  find.ctx.invoke_id = 1; /* Always 1 for now... */
246  DEBUG2("Received response with DTID %u Invoke ID %u", find.ctx.otid, find.ctx.invoke_id);
247 
248  /*
249  * Lookup the transaction in our tree of outstanding transactions
250  */
251  found = rbtree_finddata(txn_tree, &find);
252  if (!found) {
253  /*
254  * Not an error, could be a retransmission
255  */
256  ERROR("No outstanding transaction with DTID %u Invoke ID %u", find.ctx.otid, find.ctx.invoke_id);
257  return 0;
258  }
259  if (!rbtree_deletebydata(txn_tree, found)) { /* Remove the outstanding transaction */
260  ERROR("Failed removing transaction");
261  fr_assert(0);
262  }
263 
264  txn = talloc_get_type_abort(found, sigtran_transaction_t);
265  req = talloc_get_type_abort(txn->request.data, sigtran_map_send_auth_info_req_t);
266  ofd = txn->ctx.ofd;
267  osmo_timer_del(&txn->ctx.timer); /* Remove the timeout timer */
268 
269  MEM(res = talloc_zero(txn, sigtran_map_send_auth_info_res_t));
270  txn->response.type = SIGTRAN_RESPONSE_OK;
271  txn->response.data = res;
272  last = &res->vector; /* Head of vector list */
273 
274 #define sigtran_memdup(_x) \
275  do { \
276  p++; \
277  DEBUG4("Start 0x%02x len %u", (unsigned int)(tcap - p), p[0]); \
278  if (p[0] >= (len - (p - tcap))) { \
279  ERROR("Invalid length %u specified for vector component", p[0]); \
280  return -1; \
281  } \
282  vec->_x = talloc_memdup(vec, p + 1, p[0]); \
283  talloc_set_type(vec->_x, uint8_t); \
284  p += p[0] + 1; \
285  } while (0)
286 
287  end = tcap + msgb_l3len(msg);
288 
289  /*
290  * And now pretend to parse the response by looking at
291  * fixed offsets in the response data...
292  *
293  * Umm.. fixme?
294  */
295  if (req->version == 2) {
296  p = tcap + 0x40;
297  while (p < end) {
298  if ((p[0] != 0x30) || (p[1] != 0x22)) {
299  DEBUG4("Breaking out of parsing loop at %x", (uint32_t)(p - tcap));
300  break;
301  }
302  p += 2;
303 
304  MEM(vec = talloc_zero(res, sigtran_vector_t));
305  vec->type = SIGTRAN_VECTOR_TYPE_SIM_TRIPLETS;
306 
307  sigtran_memdup(sim.rand);
308  sigtran_memdup(sim.sres);
309  sigtran_memdup(sim.kc);
310 
311  *last = vec;
312  last = &(vec->next);
313  }
314  } else if (req->version == 3) {
315  p = tcap + 0x40; /* fixed offset for now */
316 
317  MEM(vec = talloc_zero(res, sigtran_vector_t));
318  vec->type = SIGTRAN_VECTOR_TYPE_UMTS_QUINTUPLETS;
319  sigtran_memdup(umts.rand);
320  sigtran_memdup(umts.xres);
321  sigtran_memdup(umts.ck);
322  sigtran_memdup(umts.ik);
323  sigtran_memdup(umts.authn);
324 
325  *last = vec;
326  }
327 
328  if (sigtran_event_submit(ofd, txn) < 0) {
329  ERROR("Failed informing event client of result: %s", fr_syserror(errno));
330  return -1;
331  }
332 
333  return 0;
334 }
