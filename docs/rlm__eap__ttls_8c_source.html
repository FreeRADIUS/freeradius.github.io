129 /*
130  * Do authentication, by letting EAP-TLS do most of the work.
131  */
132 static unlang_action_t mod_process(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
133 {
134  rlm_eap_ttls_t *inst = talloc_get_type_abort(mctx->instance, rlm_eap_ttls_t);
135 
136  eap_tls_status_t status;
137  eap_session_t *eap_session = eap_session_get(request->parent);
138  eap_tls_session_t *eap_tls_session = talloc_get_type_abort(eap_session->opaque, eap_tls_session_t);
139  fr_tls_session_t *tls_session = eap_tls_session->tls_session;
140 
141  ttls_tunnel_t *tunnel = NULL;
142  static char keying_prf_label[] = "ttls keying material";
143 
144  if (tls_session->opaque) tunnel = talloc_get_type_abort(tls_session->opaque, ttls_tunnel_t);
145 
146  eap_tls_session->include_length = inst->include_length;
147 
148  /*
149  * Process TLS layer until done.
150  */
151  status = eap_tls_process(request, eap_session);
152  if ((status == EAP_TLS_INVALID) || (status == EAP_TLS_FAIL)) {
153  REDEBUG("[eap-tls process] = %s", fr_table_str_by_value(eap_tls_status_table, status, "<INVALID>"));
154  } else {
155  RDEBUG2("[eap-tls process] = %s", fr_table_str_by_value(eap_tls_status_table, status, "<INVALID>"));
156  }
157 
158  switch (status) {
159  /*
160  * EAP-TLS handshake was successful, tell the
161  * client to keep talking.
162  *
163  * If this was EAP-TLS, we would just return
164  * an EAP-TLS-Success packet here.
165  */
166  case EAP_TLS_ESTABLISHED:
167  if (SSL_session_reused(tls_session->ssl)) {
168  RDEBUG2("Skipping Phase2 due to session resumption");
169  goto do_keys;
170  }
171 
172  if (tunnel && tunnel->authenticated) {
173 
174  do_keys:
175  /*
176  * Success: Automatically return MPPE keys.
177  */
178  if (eap_tls_success(request, eap_session,
179  keying_prf_label, sizeof(keying_prf_label) - 1,
180  NULL, 0) < 0) RETURN_MODULE_FAIL;
181  RETURN_MODULE_OK;
182  } else {
183  eap_tls_request(request, eap_session);
184  }
185  RETURN_MODULE_OK;
186 
187  /*
188  * The TLS code is still working on the TLS
189  * exchange, and it's a valid TLS request.
190  * do nothing.
191  */
192  case EAP_TLS_HANDLED:
193  RETURN_MODULE_HANDLED;
194 
195  /*
196  * Handshake is done, proceed with decoding tunneled
197  * data.
198  */
199  case EAP_TLS_RECORD_RECV_COMPLETE:
200  break;
201 
202  /*
203  * Anything else: fail.
204  */
205  default:
206  RETURN_MODULE_FAIL;
207  }
208 
209  /*
210  * Session is established, proceed with decoding
211  * tunneled data.
212  */
213  RDEBUG2("Session established. Decoding Diameter attributes");
214 
215  /*
216  * We may need TTLS data associated with the session, so
217  * allocate it here, if it wasn't already alloacted.
218  */
219  if (!tls_session->opaque) tls_session->opaque = ttls_alloc(tls_session, inst);
220 
221  /*
222  * Process the TTLS portion of the request.
223  */
224  switch (eap_ttls_process(request, eap_session, tls_session)) {
225  case FR_CODE_ACCESS_REJECT:
226  eap_tls_fail(request, eap_session);
227  RETURN_MODULE_REJECT;
228 
229  /*
230  * Access-Challenge, continue tunneled conversation.
231  */
232  case FR_CODE_ACCESS_CHALLENGE:
233  eap_tls_request(request, eap_session);
234  RETURN_MODULE_OK;
235 
236  /*
237  * Success: Automatically return MPPE keys.
238  */
239  case FR_CODE_ACCESS_ACCEPT:
240  goto do_keys;
241 
242  /*
243  * No response packet, MUST be proxying it.
244  * The main EAP module will take care of discovering
245  * that the request now has a "proxy" packet, and
246  * will proxy it, rather than returning an EAP packet.
247  */
248  case FR_CODE_STATUS_CLIENT:
249  RETURN_MODULE_OK;
250 
251  default:
252  break;
253  }
254 
255  /*
256  * Something we don't understand: Reject it.
257  */
258  eap_tls_fail(request, eap_session);
259  RETURN_MODULE_INVALID;
260 }

262 /*
263  * Send an initial eap-tls request to the peer, using the libeap functions.
264  */
265 static unlang_action_t mod_session_init(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
266 {
267  rlm_eap_ttls_t *inst = talloc_get_type_abort(mctx->instance, rlm_eap_ttls_t);
268  eap_session_t *eap_session = eap_session_get(request->parent);
269 
270  eap_tls_session_t *eap_tls_session;
271  fr_pair_t *vp;
272  bool client_cert;
273 
274  eap_session->tls = true;
275 
276  /*
277  * EAP-TLS-Require-Client-Cert attribute will override
278  * the require_client_cert configuration option.
279  */
280  vp = fr_pair_find_by_da(&request->control_pairs, attr_eap_tls_require_client_cert);
281  if (vp) {
282  client_cert = vp->vp_uint32 ? true : false;
283  } else {
284  client_cert = inst->req_client_cert;
285  }
286 
287  eap_session->opaque = eap_tls_session = eap_tls_session_init(request, eap_session, inst->tls_conf, client_cert);
288  if (!eap_tls_session) RETURN_MODULE_FAIL;
289 
290  /*
291  * TLS session initialization is over. Now handle TLS
292  * related handshaking or application data.
293  */
294  if (eap_tls_start(request, eap_session) < 0) {
295  talloc_free(eap_tls_session);
296  RETURN_MODULE_FAIL;
297  }
298 
299  eap_session->process = mod_process;
300 
301  RETURN_MODULE_OK;
302 }

307 static int mod_instantiate(void *instance, CONF_SECTION *cs)
308 {
309  rlm_eap_ttls_t *inst = talloc_get_type_abort(instance, rlm_eap_ttls_t);
310 
311  if (!virtual_server_find(inst->virtual_server)) {
312  cf_log_err_by_child(cs, "virtual_server", "Unknown virtual server '%s'", inst->virtual_server);
313  return -1;
314  }
315 
316  /*
317  * Read tls configuration, either from group given by 'tls'
318  * option, or from the eap-tls configuration.
319  */
320  inst->tls_conf = eap_tls_conf_parse(cs, "tls");
321  if (!inst->tls_conf) {
322  ERROR("Failed initializing SSL context");
323  return -1;
324  }
325 
326  return 0;
327 }
