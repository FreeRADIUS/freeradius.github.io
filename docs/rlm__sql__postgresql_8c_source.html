152 static sql_rcode_t sql_classify_error(rlm_sql_postgres_t *inst, ExecStatusType status, PGresult const *result)
153 {
154  char const *error_code;
155  char const *error_msg;
156  sql_state_entry_t const *entry;
157 
158  error_code = PQresultErrorField(result, PG_DIAG_SQLSTATE);
159  if (!error_code) {
160  switch (status){
161  /*
162  * Successful completion of a command returning no data.
163  */
164  case PGRES_COMMAND_OK:
165  #ifdef HAVE_PGRES_SINGLE_TUPLE
166  case PGRES_SINGLE_TUPLE:
167  #endif
168  case PGRES_TUPLES_OK:
169  #ifdef HAVE_PGRES_COPY_BOTH
170  case PGRES_COPY_BOTH:
171  #endif
172  case PGRES_COPY_OUT:
173  case PGRES_COPY_IN:
174  error_code = "00000";
175  break;
176 
177  case PGRES_EMPTY_QUERY: /* Shouldn't happen */
178  error_code = "42000";
179  break;
180 
181  case PGRES_BAD_RESPONSE:
182  case PGRES_NONFATAL_ERROR:
183  case PGRES_FATAL_ERROR:
184  ERROR("libpq provided no error code");
185  return RLM_SQL_ERROR;
186  }
187  }
188 
189  entry = sql_state_entry_find(inst->states, error_code);
190  if (!entry) {
191  ERROR("Can't classify: %s", error_code);
192  return RLM_SQL_ERROR;
193  }
194 
195  DEBUG2("sqlstate %s matched %s: %s (%s)", error_code,
196  entry->sql_state, entry->meaning, fr_table_str_by_value(sql_rcode_table, entry->rcode, "<DEFAULT>"));
197 
198  /*
199  * WARNING error class.
200  */
201  if ((entry->sql_state[0] == '0') && (entry->sql_state[1] == '1')) {
202  error_msg = PQresultErrorField(result, PG_DIAG_MESSAGE_PRIMARY);
203  if (error_msg) WARN("%s", error_msg);
204  }
205 
206  return entry->rcode;
207 }

211  ERROR("Error occurred, no more information available, rebuild with newer libpq");

257 static CC_HINT(nonnull) sql_rcode_t sql_query(rlm_sql_handle_t *handle, rlm_sql_config_t *config,
258  char const *query)
259 {
260  rlm_sql_postgres_conn_t *conn = handle->conn;
261  rlm_sql_postgres_t *inst = config->driver;
262  fr_time_delta_t timeout = fr_time_delta_from_sec(config->query_timeout);
263  fr_time_t start;
264  int sockfd;
265  PGresult *tmp_result;
266  int numfields = 0;
267  ExecStatusType status;
268 
269  if (!conn->db) {
270  ERROR("Socket not connected");
271  return RLM_SQL_RECONNECT;
272  }
273 
274  sockfd = PQsocket(conn->db);
275  if (sockfd < 0) {
276  ERROR("Unable to obtain socket: %s", PQerrorMessage(conn->db));
277  return RLM_SQL_RECONNECT;
278  }
279 
280  if (!PQsendQuery(conn->db, query)) {
281  ERROR("Failed to send query: %s", PQerrorMessage(conn->db));
282  return RLM_SQL_RECONNECT;
283  }
284 
285  /*
286  * We try to avoid blocking by waiting until the driver indicates that
287  * the result is ready or our timeout expires
288  */
289  start = fr_time();
290  while (PQisBusy(conn->db)) {
291  int r;
292  fd_set read_fd;
293  fr_time_delta_t elapsed = 0;
294 
295  FD_ZERO(&read_fd);
296  FD_SET(sockfd, &read_fd);
297 
298  if (config->query_timeout) {
299  elapsed = fr_time() - start;
300  if (elapsed >= timeout) goto too_long;
301  }
302 
303  r = select(sockfd + 1, &read_fd, NULL, NULL, config->query_timeout ? &fr_time_delta_to_timeval(timeout - elapsed) : NULL);
304  if (r == 0) {
305  too_long:
306  ERROR("Socket read timeout after %d seconds", config->query_timeout);
307  return RLM_SQL_RECONNECT;
308  }
309  if (r < 0) {
310  if (errno == EINTR) continue;
311  ERROR("Failed in select: %s", fr_syserror(errno));
312  return RLM_SQL_RECONNECT;
313  }
314  if (!PQconsumeInput(conn->db)) {
315  ERROR("Failed reading input: %s", PQerrorMessage(conn->db));
316  return RLM_SQL_RECONNECT;
317  }
318  }
319 
320  /*
321  * Returns a PGresult pointer or possibly a null pointer.
322  * A non-null pointer will generally be returned except in
323  * out-of-memory conditions or serious errors such as inability
324  * to send the command to the server. If a null pointer is
325  * returned, it should be treated like a PGRES_FATAL_ERROR
326  * result.
327  */
328  conn->result = PQgetResult(conn->db);
329 
330  /* Discard results for appended queries */
331  while ((tmp_result = PQgetResult(conn->db)) != NULL)
332  PQclear(tmp_result);
333 
334  /*
335  * As this error COULD be a connection error OR an out-of-memory
336  * condition return value WILL be wrong SOME of the time
337  * regardless! Pick your poison...
338  */
339  if (!conn->result) {
340  ERROR("Failed getting query result: %s", PQerrorMessage(conn->db));
341  return RLM_SQL_RECONNECT;
342  }
343 
344  status = PQresultStatus(conn->result);
345  switch (status){
346  /*
347  * Successful completion of a command returning no data.
348  */
349  case PGRES_COMMAND_OK:
350  /*
351  * Affected_rows function only returns the number of affected rows of a command
352  * returning no data...
353  */
354  conn->affected_rows = affected_rows(conn->result);
355  DEBUG2("query affected rows = %i", conn->affected_rows);
356  break;
357  /*
358  * Successful completion of a command returning data (such as a SELECT or SHOW).
359  */
360 #ifdef HAVE_PGRES_SINGLE_TUPLE
361  case PGRES_SINGLE_TUPLE:
362 #endif
363  case PGRES_TUPLES_OK:
364  conn->cur_row = 0;
365  conn->affected_rows = PQntuples(conn->result);
366  numfields = PQnfields(conn->result); /*Check row storing functions..*/
367  DEBUG2("query returned rows = %i, fields = %i", conn->affected_rows, numfields);
368  break;
369 
370 #ifdef HAVE_PGRES_COPY_BOTH
371  case PGRES_COPY_BOTH:
372 #endif
373  case PGRES_COPY_OUT:
374  case PGRES_COPY_IN:
375  DEBUG2("Data transfer started");
376  break;
377 
378  /*
379  * Weird.. this shouldn't happen.
380  */
381  case PGRES_EMPTY_QUERY:
382  case PGRES_BAD_RESPONSE: /* The server's response was not understood */
383  case PGRES_NONFATAL_ERROR:
384  case PGRES_FATAL_ERROR:
385  break;
386  }
387 
388  return sql_classify_error(inst, status, conn->result);;
389 }

414 static sql_rcode_t sql_fetch_row(rlm_sql_row_t *out, rlm_sql_handle_t *handle, UNUSED rlm_sql_config_t *config)
415 {
416 
417  int records, i, len;
418  rlm_sql_postgres_conn_t *conn = handle->conn;
419 
420  *out = NULL;
421  handle->row = NULL;
422 
423  if (conn->cur_row >= PQntuples(conn->result)) return RLM_SQL_NO_MORE_ROWS;
424 
425  free_result_row(conn);
426 
427  records = PQnfields(conn->result);
428  conn->num_fields = records;
429 
430  if ((PQntuples(conn->result) > 0) && (records > 0)) {
431  conn->row = talloc_zero_array(conn, char *, records + 1);
432  for (i = 0; i < records; i++) {
433  len = PQgetlength(conn->result, conn->cur_row, i);
434  conn->row[i] = talloc_array(conn->row, char, len + 1);
435  strlcpy(conn->row[i], PQgetvalue(conn->result, conn->cur_row, i), len + 1);
436  }
437  conn->cur_row++;
438  *out = handle->row = conn->row;
439 
440  return RLM_SQL_OK;
441  }
442 
443  return RLM_SQL_NO_MORE_ROWS;
444 }

486  size_t i = 0;
487 
488  fr_assert(outlen > 0);
489

491  while ((q = strchr(p, '\n'))) {
492  out[i].type = L_ERR;
493  out[i].msg = talloc_typed_asprintf(ctx, "%.*s", (int) (q - p), p);
494  p = q + 1;
495  if (++i == outlen) return outlen;
496  }
497  if (*p != '\0') {
498  out[i].type = L_ERR;
499  out[i].msg = p;
500  i++;
501  }
502 
503  return i;

513 static size_t sql_escape_func(request_t *request, char *out, size_t outlen, char const *in, void *arg)
514 {
515  size_t inlen, ret;
516  rlm_sql_handle_t *handle = talloc_get_type_abort(arg, rlm_sql_handle_t);
517  rlm_sql_postgres_conn_t *conn = handle->conn;
518  int err;
519 
520  /* Check for potential buffer overflow */
521  inlen = strlen(in);
522  if ((inlen * 2 + 1) > outlen) return 0;
523  /* Prevent integer overflow */
524  if ((inlen * 2 + 1) <= inlen) return 0;
525 
526  ret = PQescapeStringConn(conn->db, out, in, inlen, &err);
527  if (err) {
528  REDEBUG("Error escaping string \"%s\": %s", in, PQerrorMessage(conn->db));
529  return 0;
530  }
531 
532  return ret;
533 }

535 static int mod_instantiate(rlm_sql_config_t const *config, void *instance, CONF_SECTION *conf)
536 {
537  rlm_sql_postgres_t *inst = instance;
538  char application_name[NAMEDATALEN];
539  char *db_string;
540 
541  /*
542  * Allow the user to set their own, or disable it
543  */
544  if (inst->send_application_name) {
545  CONF_SECTION *cs;
546  char const *name;
547 
548  cs = cf_item_to_section(cf_parent(conf));
549 
550  name = cf_section_name2(cs);
551  if (!name) name = cf_section_name1(cs);
552 
553  snprintf(application_name, sizeof(application_name),
554  "FreeRADIUS " RADIUSD_VERSION_STRING " - %s (%s)", main_config->name, name);
555  }
556 
557  /*
558  * Old style database name
559  *
560  * Append options if they were set in the config
561  */
562  if (!strchr(config->sql_db, '=')) {
563  db_string = talloc_typed_asprintf(inst, "dbname='%s'", config->sql_db);
564 
565  if (config->sql_server[0] != '\0') {
566  db_string = talloc_asprintf_append(db_string, " host='%s'", config->sql_server);
567  }
568 
569  if (config->sql_port) {
570  db_string = talloc_asprintf_append(db_string, " port=%i", config->sql_port);
571  }
572 
573  if (config->sql_login[0] != '\0') {
574  db_string = talloc_asprintf_append(db_string, " user='%s'", config->sql_login);
575  }
576 
577  if (config->sql_password[0] != '\0') {
578  db_string = talloc_asprintf_append(db_string, " password='%s'", config->sql_password);
579  }
580 
581  if (config->query_timeout) {
582  db_string = talloc_asprintf_append(db_string, " connect_timeout=%d", config->query_timeout);
583  }
584 
585  if (inst->send_application_name) {
586  db_string = talloc_asprintf_append(db_string, " application_name='%s'", application_name);
587  }
588 
589  /*
590  * New style parameter string
591  *
592  * Only append options when not already present
593  */
594  } else {
595  db_string = talloc_typed_strdup(inst, config->sql_db);
596 
597  if ((config->sql_server[0] != '\0') && !strstr(db_string, "host=")) {
598  db_string = talloc_asprintf_append(db_string, " host='%s'", config->sql_server);
599  }
600 
601  if (config->sql_port && !strstr(db_string, "port=")) {
602  db_string = talloc_asprintf_append(db_string, " port=%i", config->sql_port);
603  }
604 
605  if ((config->sql_login[0] != '\0') && !strstr(db_string, "user=")) {
606  db_string = talloc_asprintf_append(db_string, " user='%s'", config->sql_login);
607  }
608 
609  if ((config->sql_password[0] != '\0') && !strstr(db_string, "password=")) {
610  db_string = talloc_asprintf_append(db_string, " password='%s'", config->sql_password);
611  }
612 
613  if ((config->query_timeout) && !strstr(db_string, "connect_timeout=")) {
614  db_string = talloc_asprintf_append(db_string, " connect_timeout=%d", config->query_timeout);
615  }
616 
617  if (inst->send_application_name && !strstr(db_string, "application_name=")) {
618  db_string = talloc_asprintf_append(db_string, " application_name='%s'", application_name);
619  }
620  }
621  inst->db_string = db_string;
622 
623  inst->states = sql_state_trie_alloc(inst);
624 
625  /*
626  * Load in the PostgreSQL specific sqlstates
627  */
628  if (sql_state_entries_from_table(inst->states, sql_state_table) < 0) return -1;
629 
630  /*
631  * Load in overrides from the driver's configuration section
632  */
633  {
634  CONF_SECTION *cs;
635 
636  cs = cf_section_find(conf, "states", NULL);
637  if (cs && (sql_state_entries_from_cs(inst->states, cs) < 0)) return -1;
638  }
639 
640  return 0;
641 }
