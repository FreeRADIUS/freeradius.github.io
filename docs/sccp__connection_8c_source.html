67 static void sccp_outgoing_state(struct sccp_connection *sccp, int old_state)
68 {
69  sigtran_conn_t *conn = talloc_get_type_abort(sccp->ctx_data, sigtran_conn_t);
70 
71  if (sccp->connection_state >= SCCP_CONNECTION_STATE_RELEASE_COMPLETE) {
72  if (conn->sccp) {
73  LOGP(DMSC, LOGL_ERROR, "ERROR: The lchan is still associated\n.");
74  }
75  free_queued(conn);
76  sccp_connection_free(sccp);
77  conn->sccp = NULL;
78  } else if (sccp->connection_state == SCCP_CONNECTION_STATE_ESTABLISHED) {
79  LOGP(DMSC, LOGL_DEBUG, "Connection established: %p\n", sccp);
80 
81  osmo_timer_del(&conn->sccp->sccp_cc_timeout);
82  osmo_timer_schedule(&conn->sccp->sccp_it_timeout, SCCP_IT_TIMER, 0);

125 int sigtran_sccp_enque(struct osmo_bsc_sccp_con *conn, struct msgb *msg)
126 {
127  struct sccp_connection *sccp = conn->sccp;
128 
129  if (sccp->connection_state > SCCP_CONNECTION_STATE_ESTABLISHED) {
130  LOGP(DMSC, LOGL_ERROR, "Connection closing, dropping packet on: %p\n", sccp);
131  msgb_free(msg);
132  } else if (sccp->connection_state == SCCP_CONNECTION_STATE_ESTABLISHED
133  && conn->sccp_queue_size == 0) {
134  sccp_connection_write(sccp, msg);
135  msgb_free(msg);
136  } else if (conn->sccp_queue_size > 10) {
137  LOGP(DMSC, LOGL_ERROR, "Connection closing, dropping packet on: %p\n", sccp);
138  msgb_free(msg);
139  } else {
140  LOGP(DMSC, LOGL_DEBUG, "Queueing packet on %p. Queue size: %d\n", sccp, conn->sccp_queue_size);
141  conn->sccp_queue_size += 1;
142  msgb_enqueue(&conn->sccp_queue, msg);
143  }
144 
145  return 0;
146 }

159 int sigtran_sccp_conn_patch(sigtran_conn_t *conn)
160 {
161  struct sccp_connection *sccp;
162 
163  sccp = sccp_connection_socket();
164  if (!sccp) {
165  LOGP(DMSC, LOGL_ERROR, "Failed to allocate memory.\n");
166  return -1;
167  }
168 
169  conn->sccp = talloc_zero(conn, sigtran_sccp_conn_t);
170  talloc_set_destructor(conn->sccp, _sigtran_sccp_conn_free);
171 
172  /* callbacks */
173  sccp->state_cb = sccp_outgoing_state;
174  sccp->data_cb = sccp_outgoing_data;
175  sccp->data_ctx = conn;
176 
177  /* prepare the timers */
178  conn->sccp->sccp_it_timeout.cb = sccp_it_timeout;
179  conn->sccp->sccp_it_timeout.data = bsc_con;
180  conn->scpp->sccp_cc_timeout.cb = sccp_cc_timeout;
181  conn->sccp->sccp_cc_timeout.data = bsc_con;
182 
183  INIT_LLIST_HEAD(&conn->sccp->sccp_queue);
184 
185  conn->sccp->sccp = sccp;
186 
187  return 0;
188 }
