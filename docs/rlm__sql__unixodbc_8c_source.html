66 static sql_rcode_t sql_socket_init(rlm_sql_handle_t *handle, rlm_sql_config_t *config,
67  fr_time_delta_t timeout)
68 {
69  rlm_sql_unixodbc_conn_t *conn;
70  long err_handle;
71  uint32_t timeout_ms = fr_time_delta_to_msec(timeout);
72 
73  MEM(conn = handle->conn = talloc_zero(handle, rlm_sql_unixodbc_conn_t));
74  talloc_set_destructor(conn, _sql_socket_destructor);
75 
76  /* 1. Allocate environment handle and register version */
77  err_handle = SQLAllocHandle(SQL_HANDLE_ENV, SQL_NULL_HANDLE, &conn->env);
78  if (sql_check_error(err_handle, handle, config)) {
79  ERROR("Can't allocate environment handle");
80  return RLM_SQL_ERROR;
81  }
82 
83  err_handle = SQLSetEnvAttr(conn->env, SQL_ATTR_ODBC_VERSION, (void*)SQL_OV_ODBC3, 0);
84  if (sql_check_error(err_handle, handle, config)) {
85  ERROR("Can't register ODBC version");
86  return RLM_SQL_ERROR;
87  }
88 
89  /* 2. Allocate connection handle */
90  err_handle = SQLAllocHandle(SQL_HANDLE_DBC, conn->env, &conn->dbc);
91  if (sql_check_error(err_handle, handle, config)) {
92  ERROR("Can't allocate connection handle");
93  return RLM_SQL_ERROR;
94  }
95 
96  /* Set the connection timeout */
97  SQLSetConnectAttr(conn->dbc, SQL_ATTR_LOGIN_TIMEOUT, &timeout_ms, SQL_IS_UINTEGER);
98 
99  /* 3. Connect to the datasource */
100  {
101  SQLCHAR *odbc_server, *odbc_login, *odbc_password;
102 
103  memcpy(&odbc_server, &config->sql_server, sizeof(odbc_server));
104  memcpy(&odbc_login, &config->sql_login, sizeof(odbc_login));
105  memcpy(&odbc_password, &config->sql_password, sizeof(odbc_password));
106  err_handle = SQLConnect(conn->dbc,
107  odbc_server, strlen(config->sql_server),
108  odbc_login, strlen(config->sql_login),
109  odbc_password, strlen(config->sql_password));
110  }
111 
112  if (sql_check_error(err_handle, handle, config)) {
113  ERROR("Connection failed");
114  return RLM_SQL_ERROR;
115  }
116 
117  /* 4. Allocate the stmt */
118  err_handle = SQLAllocHandle(SQL_HANDLE_STMT, conn->dbc, &conn->stmt);
119  if (sql_check_error(err_handle, handle, config)) {
120  ERROR("Can't allocate the stmt");
121  return RLM_SQL_ERROR;
122  }
123 
124  return RLM_SQL_OK;
125 }

149 static sql_rcode_t sql_select_query(rlm_sql_handle_t *handle, rlm_sql_config_t *config, char const *query)
150 {
151  rlm_sql_unixodbc_conn_t *conn = handle->conn;
152  SQLINTEGER i;
153  SQLLEN len;
154  int colcount;
155  int state;
156 
157  /* Only state = 0 means success */
158  if ((state = sql_query(handle, config, query))) {
159  return state;
160  }
161 
162  colcount = sql_num_fields(handle, config);
163  if (colcount < 0) {
164  return RLM_SQL_ERROR;
165  }
166 
167  /* Reserving memory for result */
168  conn->row = talloc_zero_array(conn, char *, colcount + 1); /* Space for pointers */
169 
170  for (i = 1; i <= colcount; i++) {
171  len = 0;
172  SQLColAttributes(conn->stmt, ((SQLUSMALLINT) i), SQL_DESC_LENGTH, NULL, 0, NULL, &len);
173  conn->row[i - 1] = talloc_array(conn->row, char, ++len);
174  SQLBindCol(conn->stmt, i, SQL_C_CHAR, (SQLCHAR *)conn->row[i - 1], len, NULL);
175  }
176 
177  return RLM_SQL_OK;
178 }

192 static sql_rcode_t sql_fields(char const **out[], rlm_sql_handle_t *handle, UNUSED rlm_sql_config_t *config)
193 {
194  rlm_sql_unixodbc_conn_t *conn = handle->conn;
195 
196  SQLSMALLINT fields, len, i;
197 
198  char const **names;
199  char field[128];
200 
201  SQLNumResultCols(conn->stmt, &fields);
202  if (fields == 0) return RLM_SQL_ERROR;
203 
204  MEM(names = talloc_array(handle, char const *, fields));
205 
206  for (i = 0; i < fields; i++) {
207  char *p;
208 
209  switch (SQLColAttribute(conn->stmt, i, SQL_DESC_BASE_COLUMN_NAME,
210  field, sizeof(field), &len, NULL)) {
211  case SQL_INVALID_HANDLE:
212  case SQL_ERROR:
213  ERROR("Failed retrieving field name at index %i", i);
214  talloc_free(names);
215  return RLM_SQL_ERROR;
216 
217  default:
218  break;
219  }
220 
221  MEM(p = talloc_array(names, char, (size_t)len + 1));
222  strlcpy(p, field, (size_t)len + 1);
223  names[i] = p;
224  }
225  *out = names;
226 
227  return RLM_SQL_OK;
228 }

260  * SQLExecute() with the same or different values in the

324 /** Checks the error code to determine if the connection needs to be re-esttablished
325  *
326  * @param error_handle Return code from a failed unixodbc call.
327  * @param handle rlm_sql connection handle.
328  * @param config rlm_sql config.
329  * @return
330  * - #RLM_SQL_OK on success.
331  * - #RLM_SQL_RECONNECT if reconnect is needed.
332  * - #RLM_SQL_ERROR on error.
333  */
334 static sql_rcode_t sql_check_error(long error_handle, rlm_sql_handle_t *handle, UNUSED rlm_sql_config_t *config)
335 {
336  SQLCHAR state[256];
337  SQLCHAR error[256];
338  SQLINTEGER errornum = 0;
339  SQLSMALLINT length = 255;
340  int res = -1;
341 
342  rlm_sql_unixodbc_conn_t *conn = handle->conn;
343 
344  if (SQL_SUCCEEDED(error_handle)) return 0; /* on success, just return 0 */
345 
346  error[0] = state[0] = '\0';
347 
348  SQLError(conn->env, conn->dbc, conn->stmt, state, &errornum,
349  error, sizeof(error), &length);
350 
351  if (state[0] == '0') {
352  switch (state[1]) {
353  /* SQLSTATE 01 class contains info and warning messages */
354  case '1':
355  INFO("%s %s", state, error);
356  FALL_THROUGH;
357  case '0': /* SQLSTATE 00 class means success */
358  res = RLM_SQL_OK;
359  break;
360 
361  /* SQLSTATE 08 class describes various connection errors */
362  case '8':
363  ERROR("SQL down %s %s", state, error);
364  res = RLM_SQL_RECONNECT;
365  break;
366 
367  /* any other SQLSTATE means error */
368  default:
369  ERROR("%s %s", state, error);
370  res = RLM_SQL_ERROR;
371  break;
372  }
373  } else {
374  ERROR("%s %s", state, error);
375  }
376 
377  return res;
378 }
