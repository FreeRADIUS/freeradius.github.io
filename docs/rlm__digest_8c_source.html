87 static unlang_action_t CC_HINT(nonnull) mod_authorize(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
88 {
89  rlm_digest_t const *inst = talloc_get_type_abort(mctx->instance, rlm_digest_t);
90  fr_cursor_t cursor;
91  fr_pair_t *vp;
92 
93  /*
94  * Find the first attribute which is parented by Digest-Attributes.
95  */
96  for (vp = fr_cursor_init(&cursor, &request->request_pairs);
97  vp;
98  vp = fr_cursor_next(&cursor)) {
99  if (vp->da->parent == attr_digest_attributes) break;
100  }
101  /*
102  * For simplicity, we allow the caller to omit things
103  * that they don't care about.
104  */
105  if (!vp) RETURN_MODULE_NOOP;
106 
107  if (!inst->auth_type) {
108  WARN("No 'authenticate %s {...}' section or 'Auth-Type = %s' set. Cannot setup Digest authentication",
109  inst->name, inst->name);
110  RETURN_MODULE_NOOP;
111  }
112 
113  /*
114  * Everything's OK, add a digest authentication type.
115  */
116  if (!module_section_type_set(request, attr_auth_type, inst->auth_type)) RETURN_MODULE_NOOP;
117 
118  RETURN_MODULE_OK;
119 }

124 static unlang_action_t CC_HINT(nonnull) mod_authenticate(rlm_rcode_t *p_result, UNUSED module_ctx_t const *mctx, request_t *request)
125 {
126  size_t a1_len, a2_len, kd_len;
127  uint8_t a1[(FR_MAX_STRING_LEN + 1) * 5]; /* can be 5 attributes */
128  uint8_t a2[(FR_MAX_STRING_LEN + 1) * 3]; /* can be 3 attributes */
129  uint8_t kd[(FR_MAX_STRING_LEN + 1) * 5];
130  uint8_t hash[16]; /* MD5 output */
131  fr_pair_t *vp, *passwd, *algo;
132  fr_pair_t *qop, *nonce;
133 
134  /*
135  * We require access to the plain-text password, or to the
136  * Digest-HA1 parameter.
137  */
138  passwd = fr_pair_find_by_da(&request->control_pairs, attr_digest_ha1);
139  if (passwd) {
140  if (passwd->vp_length != 32) {
141  REDEBUG("Digest-HA1 has invalid length, authentication failed");
142  RETURN_MODULE_INVALID;
143  }
144  } else {
145  passwd = fr_pair_find_by_da(&request->control_pairs, attr_cleartext_password);
146  }
147  if (!passwd) {
148  REDEBUG("Cleartext-Password or Digest-HA1 is required for authentication");
149  RETURN_MODULE_INVALID;
150  }
151 
152  /*
153  * We require access to the Digest-Nonce-Value
154  */
155  nonce = fr_pair_find_by_da(&request->request_pairs, attr_digest_nonce);
156  if (!nonce) {
157  REDEBUG("No Digest-Nonce: Cannot perform Digest authentication");
158  RETURN_MODULE_INVALID;
159  }
160 
161  /*
162  * A1 = Digest-User-Name ":" Realm ":" Password
163  */
164  vp = fr_pair_find_by_da(&request->request_pairs, attr_digest_user_name);
165  if (!vp) {
166  REDEBUG("No Digest-User-Name: Cannot perform Digest authentication");
167  RETURN_MODULE_INVALID;
168  }
169  memcpy(&a1[0], vp->vp_octets, vp->vp_length);
170  a1_len = vp->vp_length;
171 
172  a1[a1_len] = ':';
173  a1_len++;
174 
175  vp = fr_pair_find_by_da(&request->request_pairs, attr_digest_realm);
176  if (!vp) {
177  REDEBUG("No Digest-Realm: Cannot perform Digest authentication");
178  RETURN_MODULE_INVALID;
179  }
180  memcpy(&a1[a1_len], vp->vp_octets, vp->vp_length);
181  a1_len += vp->vp_length;
182 
183  a1[a1_len] = ':';
184  a1_len++;
185 
186  if (passwd->da == attr_cleartext_password) {
187  memcpy(&a1[a1_len], passwd->vp_octets, passwd->vp_length);
188  a1_len += passwd->vp_length;
189  a1[a1_len] = '\0';
190  RDEBUG2("A1 = %s", a1);
191  } else {
192  a1[a1_len] = '\0';
193  RDEBUG2("A1 = %s (using Digest-HA1)", a1);
194  a1_len = 16;
195  }
196 
197  /*
198  * See which variant we calculate.
199  * Assume MD5 if no Digest-Algorithm attribute received
200  */
201  algo = fr_pair_find_by_da(&request->request_pairs, attr_digest_algorithm);
202  if ((!algo) ||
203  (strcasecmp(algo->vp_strvalue, "MD5") == 0)) {
204  /*
205  * Set A1 to Digest-HA1 if no User-Password found
206  */
207  if (passwd->da == attr_digest_ha1) {
208  if (fr_hex2bin(NULL, &FR_DBUFF_TMP(&a1[0], sizeof(a1)),
209  &FR_SBUFF_IN(passwd->vp_strvalue, passwd->vp_length), false) != 16) {
210  RDEBUG2("Invalid text in Digest-HA1");
211  RETURN_MODULE_INVALID;
212  }
213  }
214 
215  } else if (strcasecmp(algo->vp_strvalue, "MD5-sess") == 0) {
216  /*
217  * K1 = H(A1) : Digest-Nonce ... : H(A2)
218  *
219  * If we find Digest-HA1, we assume it contains
220  * H(A1).
221  */
222  if (passwd->da == attr_cleartext_password) {
223  fr_md5_calc(hash, &a1[0], a1_len);
224  fr_bin2hex(&FR_SBUFF_OUT((char *) &a1[0], 32 + 1), &FR_DBUFF_TMP(hash, 16), SIZE_MAX);
225  } else { /* MUST be Digest-HA1 */
226  memcpy(&a1[0], passwd->vp_strvalue, 32);
227  }
228  a1_len = 32;
229 
230  a1[a1_len] = ':';
231  a1_len++;
232 
233  /*
234  * Tack on the Digest-Nonce. Length must be even
235  */
236  if ((nonce->vp_length & 1) != 0) {
237  REDEBUG("Received Digest-Nonce hex string with invalid length: Cannot perform Digest authentication");
238  RETURN_MODULE_INVALID;
239  }
240  memcpy(&a1[a1_len], nonce->vp_octets, nonce->vp_length);
241  a1_len += nonce->vp_length;
242 
243  a1[a1_len] = ':';
244  a1_len++;
245 
246  vp = fr_pair_find_by_da(&request->request_pairs, attr_digest_cnonce);
247  if (!vp) {
248  REDEBUG("No Digest-CNonce: Cannot perform Digest authentication");
249  RETURN_MODULE_INVALID;
250  }
251 
252  /*
253  * Digest-CNonce length must be even
254  */
255  if ((vp->vp_length & 1) != 0) {
256  REDEBUG("Received Digest-CNonce hex string with invalid length: Cannot perform Digest authentication");
257  RETURN_MODULE_INVALID;
258  }
259  memcpy(&a1[a1_len], vp->vp_octets, vp->vp_length);
260  a1_len += vp->vp_length;
261 
262  } else if (strcasecmp(algo->vp_strvalue, "MD5") != 0) {
263  /*
264  * We check for "MD5-sess" and "MD5".
265  * Anything else is an error.
266  */
267  REDEBUG("%pP - Unknown Digest-Algorithm: Cannot perform Digest authentication", vp);
268  RETURN_MODULE_INVALID;
269  }
270 
271  /*
272  * A2 = Digest-Method ":" Digest-URI
273  */
274  vp = fr_pair_find_by_da(&request->request_pairs, attr_digest_method);
275  if (!vp) {
276  REDEBUG("No Digest-Method: Cannot perform Digest authentication");
277  RETURN_MODULE_INVALID;
278  }
279  memcpy(&a2[0], vp->vp_octets, vp->vp_length);
280  a2_len = vp->vp_length;
281 
282  a2[a2_len] = ':';
283  a2_len++;
284 
285  vp = fr_pair_find_by_da(&request->request_pairs, attr_digest_uri);
286  if (!vp) {
287  REDEBUG("No Digest-URI: Cannot perform Digest authentication");
288  RETURN_MODULE_INVALID;
289  }
290  memcpy(&a2[a2_len], vp->vp_octets, vp->vp_length);
291  a2_len += vp->vp_length;
292 
293  /*
294  * QOP is "auth-int", tack on ": Digest-Body-Digest"
295  */
296  qop = fr_pair_find_by_da(&request->request_pairs, attr_digest_qop);
297  if (qop) {
298  if (strcasecmp(qop->vp_strvalue, "auth-int") == 0) {
299  fr_pair_t *body;
300 
301  /*
302  * Add in Digest-Body-Digest
303  */
304  a2[a2_len] = ':';
305  a2_len++;
306 
307  /*
308  * Must be a hex representation of an MD5 digest.
309  */
310  body = fr_pair_find_by_da(&request->request_pairs, attr_digest_body_digest);
311  if (!body) {
312  REDEBUG("No Digest-Body-Digest: Cannot perform Digest authentication");
313  RETURN_MODULE_INVALID;
314  }
315 
316  if ((a2_len + body->vp_length) > sizeof(a2)) {
317  REDEBUG("Digest-Body-Digest is too long");
318  RETURN_MODULE_INVALID;
319  }
320 
321  memcpy(a2 + a2_len, body->vp_octets, body->vp_length);
322  a2_len += body->vp_length;
323 
324  } else if (strcasecmp(qop->vp_strvalue, "auth") != 0) {
325  REDEBUG("%pP - Unknown value: Cannot perform Digest authentication", qop);
326  RETURN_MODULE_INVALID;
327  }
328  }
329 
330  a2[a2_len] = '\0';
331  RDEBUG2("A2 = %s", a2);
332 
333  /*
334  * KD = H(A1) : Digest-Nonce ... : H(A2).
335  * Compute MD5 if Digest-Algorithm == "MD5-Sess",
336  * or if we found a User-Password.
337  */
338  if (((algo != NULL) && (strcasecmp(algo->vp_strvalue, "MD5-Sess") == 0)) ||
339  (passwd->da == attr_cleartext_password)) {
340  a1[a1_len] = '\0';
341  fr_md5_calc(&hash[0], &a1[0], a1_len);
342  } else {
343  memcpy(&hash[0], &a1[0], a1_len);
344  }
345  fr_bin2hex(&FR_SBUFF_OUT((char *) kd, (sizeof(hash) * 2) + 1), &FR_DBUFF_TMP(hash, sizeof(hash)), SIZE_MAX);
346 
347  RHEXDUMP_INLINE3(hash, sizeof(hash), "H(A1)");
348 
349  kd_len = 32;
350 
351  kd[kd_len] = ':';
352  kd_len++;
353 
354  memcpy(&kd[kd_len], nonce->vp_octets, nonce->vp_length);
355  kd_len += nonce->vp_length;
356 
357  /*
358  * No QOP defined. Do RFC 2069 compatibility.
359  */
360  if (!qop) {
361  /*
362  * Do nothing here.
363  */
364 
365  } else { /* Digest-QOP MUST be "auth" or "auth-int" */
366  /*
367  * Tack on ":" Digest-Nonce-Count ":" Digest-CNonce
368  * ":" Digest-QOP
369  */
370  kd[kd_len] = ':';
371  kd_len++;
372 
373  vp = fr_pair_find_by_da(&request->request_pairs, attr_digest_nonce_count);
374  if (!vp) {
375  REDEBUG("No Digest-Nonce-Count: Cannot perform Digest authentication");
376  RETURN_MODULE_INVALID;
377  }
378  memcpy(&kd[kd_len], vp->vp_octets, vp->vp_length);
379  kd_len += vp->vp_length;
380 
381  kd[kd_len] = ':';
382  kd_len++;
383 
384  vp = fr_pair_find_by_da(&request->request_pairs, attr_digest_cnonce);
385  if (!vp) {
386  REDEBUG("No Digest-CNonce: Cannot perform Digest authentication");
387  RETURN_MODULE_INVALID;
388  }
389  memcpy(&kd[kd_len], vp->vp_octets, vp->vp_length);
390  kd_len += vp->vp_length;
391 
392  kd[kd_len] = ':';
393  kd_len++;
394 
395  memcpy(&kd[kd_len], qop->vp_octets, qop->vp_length);
396  kd_len += qop->vp_length;
397  }
398 
399  /*
400  * Tack on ":" H(A2)
401  */
402  kd[kd_len] = ':';
403  kd_len++;
404 
405  fr_md5_calc(&hash[0], &a2[0], a2_len);
406 
407  fr_bin2hex(&FR_SBUFF_OUT((char *) kd + kd_len, (sizeof(hash) * 2) + 1), &FR_DBUFF_TMP(hash, sizeof(hash)), SIZE_MAX);
408 
409  RHEXDUMP_INLINE3(hash, sizeof(hash), "H(A2)");
410 
411  kd_len += 32;
412 
413  kd[kd_len] = 0;
414 
415  RDEBUG2("KD = %s\n", &kd[0]);
416 
417  /*
418  * Take the hash of KD.
419  */
420  fr_md5_calc(&hash[0], &kd[0], kd_len);
421  memcpy(&kd[0], &hash[0], 16);
422 
423  /*
424  * Get the binary value of Digest-Response
425  */
426  vp = fr_pair_find_by_da(&request->request_pairs, attr_digest_response);
427  if (!vp) {
428  REDEBUG("No Digest-Response attribute in the request. Cannot perform digest authentication");
429  RETURN_MODULE_INVALID;
430  }
431 
432  if (fr_hex2bin(NULL, &FR_DBUFF_TMP(&hash[0], sizeof(hash)),
433  &FR_SBUFF_IN(vp->vp_strvalue, vp->vp_length), false) != (ssize_t)(vp->vp_length >> 1)) {
434  RDEBUG2("Invalid text in Digest-Response");
435  RETURN_MODULE_INVALID;
436  }
437 
438  RDEBUG3("Comparing hashes, received: %pV, calculated: %pH", &vp->data, fr_box_octets(kd, 16));
439 
440  /*
441  * And finally, compare the digest in the packet with KD.
442  */
443  if (memcmp(&kd[0], &hash[0], 16) == 0) RETURN_MODULE_OK;
444 
445  REDEBUG("FAILED authentication");
446  RETURN_MODULE_REJECT;
447 }
