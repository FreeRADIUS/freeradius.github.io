97 /** Prints information to the debug log on the current timeout settings
98  *
99  * There are so many different timers in LDAP it's often hard to debug
100  * issues with them, hence the need for this function.
101  */
102 void fr_ldap_timeout_debug(request_t *request, fr_ldap_connection_t const *conn,
103  fr_time_delta_t timeout, char const *prefix)
104 {
105  struct timeval *net = NULL, *client = NULL;
106  int server = 0;
107  fr_ldap_config_t const *handle_config = conn->config;
108 
109  if (request) RINDENT();
110 
111 #ifdef LDAP_OPT_NETWORK_TIMEOUT
112  if (ldap_get_option(conn->handle, LDAP_OPT_NETWORK_TIMEOUT, &net) != LDAP_OPT_SUCCESS) {
113  ROPTIONAL(REDEBUG, ERROR, "Failed getting LDAP_OPT_NETWORK_TIMEOUT");
114  }
115 #endif
116 
117 #ifdef LDAP_OPT_TIMEOUT
118  if (ldap_get_option(conn->handle, LDAP_OPT_TIMEOUT, &client) != LDAP_OPT_SUCCESS) {
119  ROPTIONAL(REDEBUG, ERROR, "Failed getting LDAP_OPT_TIMEOUT");
120  }
121 #endif
122 
123  if (ldap_get_option(conn->handle, LDAP_OPT_TIMELIMIT, &server) != LDAP_OPT_SUCCESS) {
124  ROPTIONAL(REDEBUG, ERROR, "Failed getting LDAP_OPT_TIMELIMIT");
125  }
126 
127  ROPTIONAL(RDEBUG4, DEBUG4, "%s: Timeout settings", prefix);
128 
129  if (timeout) {
130  ROPTIONAL(RDEBUG4, DEBUG4, "Client side result timeout (ovr): %pVs",
131  fr_box_time_delta(timeout));
132  } else {
133  ROPTIONAL(RDEBUG4, DEBUG4, "Client side result timeout (ovr): unset");
134  }
135 
136 #ifdef LDAP_OPT_TIMEOUT
137  if (client && (client->tv_sec != -1)) {
138  ROPTIONAL(RDEBUG4, DEBUG4, "Client side result timeout (dfl): %pVs",
139  fr_box_time_delta(fr_time_delta_from_timeval(client)));
140 
141  } else {
142  ROPTIONAL(RDEBUG4, DEBUG4, "Client side result timeout (dfl): unset");
143  }
144 #endif
145 
146 #ifdef LDAP_OPT_NETWORK_TIMEOUT
147  if (net && (net->tv_sec != -1)) {
148  ROPTIONAL(RDEBUG4, DEBUG4, "Client side network I/O timeout : %pVs",
149  fr_box_time_delta(fr_time_delta_from_timeval(net)));
150  } else {
151  ROPTIONAL(RDEBUG4, DEBUG4, "Client side network I/O timeout : unset");
152 
153  }
154 #endif
155  ROPTIONAL(RDEBUG4, DEBUG4, "Server side result timeout : %i", server);
156  if (request) REXDENT();
157 
158  free(net);
159  free(client);
160 }

178 /** Perform basic parsing of multiple types of messages, checking for error conditions
179  *
180  * @note Error messages should be retrieved with fr_strerror() and fr_strerror_pop()
181  *
182  * @param[out] ctrls Server ctrls returned to the client. May be NULL if not required.
183  * Must be freed with ldap_free_ctrls.
184  * @param[in] conn the message was received on.
185  * @param[in] msg we're parsing.
186  * @param[in] dn if processing the result from a search request.
187  * @return One of the LDAP_PROC_* (#fr_ldap_rcode_t) values.
188  */
189 fr_ldap_rcode_t fr_ldap_error_check(LDAPControl ***ctrls, fr_ldap_connection_t const *conn, LDAPMessage *msg, char const *dn)
190 {
191  fr_ldap_rcode_t status = LDAP_PROC_SUCCESS;
192 
193  int msg_type;
194  int lib_errno = LDAP_SUCCESS; /* errno returned by the library */
195  int srv_errno = LDAP_SUCCESS; /* errno in the result message */
196 
197  char *part_dn = NULL; /* Partial DN match */
198  char *srv_err = NULL; /* Server's extended error message */
199 
200  ssize_t len;
201 
202  if (ctrls) *ctrls = NULL;
203 
204  if (!msg) {
205  ldap_get_option(conn->handle, LDAP_OPT_ERROR_NUMBER, &lib_errno);
206  if (lib_errno != LDAP_SUCCESS) goto process_error;
207 
208  fr_strerror_printf("No result available");
209  return LDAP_PROC_NO_RESULT;
210  }
211 
212  msg_type = ldap_msgtype(msg);
213  switch (msg_type) {
214  /*
215  * Parse the result and check for errors sent by the server
216  */
217  case LDAP_RES_SEARCH_RESULT: /* The result of a search */
218  case LDAP_RES_BIND: /* The result of a bind operation */
219  case LDAP_RES_EXTENDED:
220  lib_errno = ldap_parse_result(conn->handle, msg,
221  &srv_errno, &part_dn, &srv_err,
222  NULL, ctrls, 0);
223  break;
224 
225  /*
226  * These are messages containing objects so unless they're
227  * malformed they can't contain errors.
228  */
229  case LDAP_RES_SEARCH_ENTRY:
230  if (ctrls) lib_errno = ldap_get_entry_controls(conn->handle, msg, ctrls);
231  break;
232 
233  /*
234  * An intermediate message updating us on the result of an operation
235  */
236  case LDAP_RES_INTERMEDIATE:
237  lib_errno = ldap_parse_intermediate(conn->handle, msg, NULL, NULL, ctrls, 0);
238  break;
239 
240  /*
241  * Can't extract any more useful information.
242  */
243  default:
244  return LDAP_PROC_SUCCESS;
245  }
246 
247  /*
248  * Stupid messy API
249  */
250  if (lib_errno != LDAP_SUCCESS) {
251  fr_assert(!ctrls || !*ctrls);
252  ldap_get_option(conn->handle, LDAP_OPT_ERROR_NUMBER, &lib_errno);
253  }
254 
255 process_error:
256  if ((lib_errno == LDAP_SUCCESS) && (srv_errno != LDAP_SUCCESS)) {
257  lib_errno = srv_errno;
258  } else if ((lib_errno != LDAP_SUCCESS) && (srv_errno == LDAP_SUCCESS)) {
259  srv_errno = lib_errno;
260  }
261 
262  switch (lib_errno) {
263  case LDAP_SUCCESS:
264  fr_strerror_printf("Success");
265  break;
266 
267  case LDAP_SASL_BIND_IN_PROGRESS:
268  fr_strerror_printf("Continuing");
269  status = LDAP_PROC_CONTINUE;
270  break;
271 
272  case LDAP_NO_SUCH_OBJECT:
273  fr_strerror_printf("The specified DN wasn't found");
274  status = LDAP_PROC_BAD_DN;
275 
276  /*
277  * Build our own internal diagnostic string
278  */
279  if (dn && part_dn) {
280  char *spaces;
281  char *text;
282 
283  len = fr_ldap_common_dn(dn, part_dn);
284  if (len < 0) break;
285 
286  fr_canonicalize_error(NULL, &spaces, &text, -len, dn);
287  fr_strerror_printf_push("%s", text);
288  fr_strerror_printf_push("%s^ %s", spaces, "match stopped here");
289 
290  talloc_free(spaces);
291  talloc_free(text);
292  }
293  goto error_string;
294 
295  case LDAP_INSUFFICIENT_ACCESS:
296  fr_strerror_printf("Insufficient access. Check the identity and password configuration directives");
297  status = LDAP_PROC_NOT_PERMITTED;
298  break;
299 
300  case LDAP_UNWILLING_TO_PERFORM:
301  fr_strerror_printf("Server was unwilling to perform");
302  status = LDAP_PROC_NOT_PERMITTED;
303  break;
304 
305  case LDAP_FILTER_ERROR:
306  fr_strerror_printf("Bad search filter");
307  status = LDAP_PROC_ERROR;
308  break;
309 
310  case LDAP_TIMEOUT:
311  fr_strerror_printf("Timed out while waiting for server to respond");
312  status = LDAP_PROC_TIMEOUT;
313  break;
314 
315  case LDAP_TIMELIMIT_EXCEEDED:
316  fr_strerror_printf("Time limit exceeded");
317  status = LDAP_PROC_TIMEOUT;
318  break;
319 
320  case LDAP_BUSY:
321  case LDAP_UNAVAILABLE:
322  case LDAP_SERVER_DOWN:
323  status = LDAP_PROC_BAD_CONN;
324  goto error_string;
325 
326  case LDAP_INVALID_CREDENTIALS:
327  case LDAP_CONSTRAINT_VIOLATION:
328  status = LDAP_PROC_REJECT;
329  goto error_string;
330 
331  case LDAP_OPERATIONS_ERROR:
332  fr_strerror_printf("Please set 'chase_referrals=yes' and 'rebind=yes'. "
333  "See the ldap module configuration for details");
334  FALL_THROUGH;
335 
336  default:
337  status = LDAP_PROC_ERROR;
338 
339  error_string:
340  if (lib_errno == srv_errno) {
341  fr_strerror_printf("lib error: %s (%u)", ldap_err2string(lib_errno), lib_errno);
342  } else {
343  fr_strerror_printf("lib error: %s (%u), srv error: %s (%u)",
344  ldap_err2string(lib_errno), lib_errno,
345  ldap_err2string(srv_errno), srv_errno);
346  }
347 
348  if (srv_err) fr_strerror_printf_push("Server said: %s", srv_err);
349 
350  break;
351  }
352 
353  /*
354  * Cleanup memory
355  */
356  if (srv_err) ldap_memfree(srv_err);
357  if (part_dn) ldap_memfree(part_dn);
358 
359  return status;
360 }

362 /** Parse response from LDAP server dealing with any errors
363  *
364  * Should be called after an LDAP operation. Will check result of operation and if
365  * it was successful, then attempt to retrieve and parse the result. Will also produce
366  * extended error output including any messages the server sent, and information about
367  * partial DN matches.
368  *
369  * @note Error messages should be retrieved with fr_strerror() and fr_strerror_pop()
370  *
371  * @param[out] result Where to write result, if NULL result will be freed. If not NULL caller
372  * must free with ldap_msgfree().
373  * @param[out] ctrls Server ctrls returned to the client. May be NULL if not required.
374  * Must be freed with ldap_free_ctrls.
375  * @param[in] conn Current connection.
376  * @param[in] msgid returned from last operation.
377  * Special values are:
378  * - LDAP_RES_ANY - Retrieve any received messages useful for multiplexing.
379  * - LDAP_RES_UNSOLICITED - Any unsolicited message.
380  * @param[in] all How many messages to retrieve:
381  * - LDAP_MSG_ONE - Retrieve the first message matching msgid (waiting if one is not available).
382  * - LDAP_MSG_ALL - Retrieve all received messages matching msgid (waiting if none are available).
383  * - LDAP_MSG_RECEIVED - Retrieve all received messages.
384  * @param[in] dn Last search or bind DN. May be NULL.
385  * @param[in] timeout Override the default result timeout.
386  *
387  * @return One of the LDAP_PROC_* (#fr_ldap_rcode_t) values.
388  */
389 fr_ldap_rcode_t fr_ldap_result(LDAPMessage **result, LDAPControl ***ctrls,
390  fr_ldap_connection_t const *conn, int msgid, int all,
391  char const *dn,
392  fr_time_delta_t timeout)
393 {
394  fr_ldap_rcode_t status = LDAP_PROC_SUCCESS;
395  int lib_errno;
396  fr_time_delta_t our_timeout = timeout;
397 
398  LDAPMessage *tmp_msg = NULL, *msg; /* Temporary message pointer storage if we weren't provided with one */
399  LDAPMessage **result_p = result;
400 
401  if (result) *result = NULL;
402  if (ctrls) *ctrls = NULL;
403 
404  /*
405  * We always need the result, but our caller may not
406  */
407  if (!result) result_p = &tmp_msg;
408 
409  /*
410  * Check if there was an error sending the request
411  */
412  ldap_get_option(conn->handle, LDAP_OPT_ERROR_NUMBER, &lib_errno);
413  if (lib_errno != LDAP_SUCCESS) return fr_ldap_error_check(NULL, conn, NULL, dn);
414 
415  if (!timeout) our_timeout = conn->config->res_timeout;
416 
417  /*
418  * Now retrieve the result and check for errors
419  * ldap_result returns -1 on failure, and 0 on timeout
420  */
421  lib_errno = ldap_result(conn->handle, msgid, all, &fr_time_delta_to_timeval(our_timeout), result_p);
422  switch (lib_errno) {
423  case 0:
424  lib_errno = LDAP_TIMEOUT;
425  fr_strerror_printf("timeout waiting for result");
426  return LDAP_PROC_TIMEOUT;
427 
428  case -1:
429  return fr_ldap_error_check(NULL, conn, NULL, dn);
430 
431  default:
432  break;
433  }
434 
435  for (msg = ldap_first_message(conn->handle, *result_p);
436  msg;
437  msg = ldap_next_message(conn->handle, msg)) {
438  status = fr_ldap_error_check(ctrls, conn, msg, dn);
439  if (status != LDAP_PROC_SUCCESS) break;
440  }
441 
442  if (*result_p && ((status < 0) || !result)) {
443  ldap_msgfree(*result_p);
444  *result_p = NULL;
445  }
446 
447  return status;
448 }

450 /** Bind to the LDAP directory as a user
451  *
452  * Performs a simple bind to the LDAP directory, and handles any errors that occur.
453  *
454  * @param[in] request Current request, this may be NULL, in which case all
455  * debug logging is done with log.
456  * @param[in,out] pconn to use. May change as this function calls functions
457  * which auto re-connect.
458  * @param[in] dn of the user, may be NULL to bind anonymously.
459  * @param[in] password of the user, may be NULL if no password is specified.
460  * @param[in] sasl mechanism to use for bind, and additional parameters.
461  * @param[in] timeout Maximum time bind is allowed to take.
462  * @param[in] serverctrls Only used for SASL binds. May be NULL.
463  * @param[in] clientctrls Search controls for sasl_bind.
464  * Only used for SASL binds. May be NULL.
465  * @return One of the LDAP_PROC_* (#fr_ldap_rcode_t) values.
466  */
467 fr_ldap_rcode_t fr_ldap_bind(request_t *request,
468  fr_ldap_connection_t **pconn,
469  char const *dn, char const *password,
470 #ifdef WITH_SASL
471  fr_ldap_sasl_t const *sasl,
472 #else
473  NDEBUG_UNUSED fr_ldap_sasl_t const *sasl,
474 #endif
475  fr_time_delta_t timeout,
476  LDAPControl **serverctrls, LDAPControl **clientctrls)
477 {
478  fr_ldap_rcode_t status = LDAP_PROC_ERROR;
479  fr_ldap_config_t const *handle_config = (*pconn)->config;
480 
481  int msgid = -1;
482 
483  fr_assert(*pconn && (*pconn)->handle);
484 
485 #ifndef WITH_SASL
486  fr_assert(!sasl || !sasl->mech);
487 #endif
488 
489  if (DEBUG_ENABLED4 || (request && RDEBUG_ENABLED4)) {
490  fr_ldap_timeout_debug(request, *pconn, timeout, __FUNCTION__);
491  }
492 
493  /*
494  * Bind as anonymous user
495  */
496  if (!dn) dn = "";
497 
498 #ifdef WITH_SASL
499  if (sasl && sasl->mech) {
500  status = fr_ldap_sasl_interactive(request, *pconn, dn, password, sasl,
501  serverctrls, clientctrls, timeout);
502  } else
503 #endif
504  {
505  int ret;
506  struct berval cred;
507 
508  if (password) {
509  memcpy(&cred.bv_val, &password, sizeof(cred.bv_val));
510  cred.bv_len = talloc_array_length(password) - 1;
511  } else {
512  cred.bv_val = NULL;
513  cred.bv_len = 0;
514  }
515 
516  /*
517  * Yes, confusingly named. This is the simple version
518  * of the SASL bind function that should always be
519  * available.
520  */
521  ret = ldap_sasl_bind((*pconn)->handle, dn, LDAP_SASL_SIMPLE, &cred,
522  serverctrls, clientctrls, &msgid);
523 
524  /* We got a valid message ID */
525  if ((ret == 0) && (msgid >= 0)) ROPTIONAL(RDEBUG2, DEBUG2, "Waiting for bind result...");
526 
527  status = fr_ldap_result(NULL, NULL, *pconn, msgid, 0, dn, 0);
528  }
529 
530  switch (status) {
531  case LDAP_PROC_SUCCESS:
532  ROPTIONAL(RDEBUG2, DEBUG2, "Bind successful");
533  break;
534 
535  case LDAP_PROC_NOT_PERMITTED:
536  ROPTIONAL(RPEDEBUG, PERROR, "Bind as \"%s\" to \"%s\" not permitted",
537  *dn ? dn : "(anonymous)", handle_config->server);
538  break;
539 
540  default:
541  ROPTIONAL(RPEDEBUG, PERROR, "Bind as \"%s\" to \"%s\" failed",
542  *dn ? dn : "(anonymous)", handle_config->server);
543  break;
544  }
545 
546  return status; /* caller closes the connection */
547 }

549 /** Search for something in the LDAP directory
550  *
551  * Binds as the administrative user and performs a search, dealing with any errors.
552  *
553  * @param[out] result Where to store the result. Must be freed with ldap_msgfree
554  * if LDAP_PROC_SUCCESS is returned.
555  * May be NULL in which case result will be automatically freed after use.
556  * @param[in] request Current request.
557  * @param[in,out] pconn to use. May change as this function calls functions which auto re-connect.
558  * @param[in] dn to use as base for the search.
559  * @param[in] scope to use (LDAP_SCOPE_BASE, LDAP_SCOPE_ONE, LDAP_SCOPE_SUB).
560  * @param[in] filter to use, should be pre-escaped.
561  * @param[in] attrs to retrieve.
562  * @param[in] serverctrls Search controls to pass to the server. May be NULL.
563  * @param[in] clientctrls Search controls for ldap_search. May be NULL.
564  * @return One of the LDAP_PROC_* (#fr_ldap_rcode_t) values.
565  */
566 fr_ldap_rcode_t fr_ldap_search(LDAPMessage **result, request_t *request,
567  fr_ldap_connection_t **pconn,
568  char const *dn, int scope, char const *filter, char const * const *attrs,
569  LDAPControl **serverctrls, LDAPControl **clientctrls)
570 {
571  fr_ldap_rcode_t status = LDAP_PROC_ERROR;
572  LDAPMessage *our_result = NULL;
573 
574  fr_ldap_config_t const *handle_config = (*pconn)->config;
575 
576  int msgid; // Message id returned by
577  // ldap_search_ext.
578 
579  int count = 0; // Number of results we got.
580 
581  struct timeval tv; // Holds timeout values.
582 
583  LDAPControl *our_serverctrls[LDAP_MAX_CONTROLS];
584  LDAPControl *our_clientctrls[LDAP_MAX_CONTROLS];
585 
586  fr_ldap_control_merge(our_serverctrls, our_clientctrls,
587  NUM_ELEMENTS(our_serverctrls),
588  NUM_ELEMENTS(our_clientctrls),
589  *pconn, serverctrls, clientctrls);
590 
591  fr_assert(*pconn && (*pconn)->handle);
592 
593  if (DEBUG_ENABLED4 || (request && RDEBUG_ENABLED4)) {
594  fr_ldap_timeout_debug(request, *pconn, 0, __FUNCTION__);
595  }
596 
597  /*
598  * OpenLDAP library doesn't declare attrs array as const, but
599  * it really should be *sigh*.
600  */
601  char **search_attrs;
602  memcpy(&search_attrs, &attrs, sizeof(attrs));
603 
604  /*
605  * Do all searches as the admin user.
606  */
