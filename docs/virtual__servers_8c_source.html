322 /** Define a values for Auth-Type attributes by the sections present in a virtual-server
323  *
324  * The ident2 value of any sections found will be converted into values of the specified da.
325  *
326  * @param[in] server_cs The virtual server containing the sections.
327  * @param[in] subcs_name of the subsection to search for.
328  * @param[in] da to add enumeration values for.
329  * @return
330  * - 0 all values added successfully.
331  * - -1 an error occurred.
332  */
333 int virtual_server_section_attribute_define(CONF_SECTION *server_cs, char const *subcs_name, fr_dict_attr_t const *da)
334 {
335  int rcode = 0;
336  CONF_SECTION *subcs = NULL;
337 
338  fr_assert(strcmp(cf_section_name1(server_cs), "server") == 0);
339 
340  while ((subcs = cf_section_find_next(server_cs, subcs, subcs_name, CF_IDENT_ANY))) {
341  char const *name2;
342  fr_dict_enum_t *dv;
343 
344  name2 = cf_section_name2(subcs);
345  if (!name2) {
346  cf_log_err(subcs, "Invalid '%s { ... }' section, it must have a name", subcs_name);
347  return -1;
348  }
349 
350  /*
351  * If the value already exists, don't
352  * create it again.
353  */
354  dv = fr_dict_enum_by_name(da, name2, -1);
355  if (dv) continue;
356 
357  cf_log_debug(subcs, "Creating %s = %s", da->name, name2);
358 
359  /*
360  * Create a new unique value with a meaningless
361  * number. You can't look at it from outside of
362  * this code, so it doesn't matter. The only
363  * requirement is that it's unique.
364  */
365  if (fr_dict_attr_enum_add_name_next(fr_dict_attr_unconst(da), name2) < 0) {
366  PERROR("Failed adding section value");
367  return -1;
368  }
369 
370  rcode = 1;
371  }
372 
373  return rcode;
374 }

411 /** Compare listeners by app_io_addr
412  *
413  * Only works for IP addresses, and will blow up on file names
414  */
415 static int listen_addr_cmp(void const *one, void const *two)
416 {
417  fr_listen_t const *a = one;
418  fr_listen_t const *b = two;
419  fr_ipaddr_t aip, bip;
420  int rcode;
421 
422  /*
423  * The caller must ensure that the address field is set.
424  */
425  if (!a->app_io_addr && !b->app_io_addr) return 0;
426  if (!a->app_io_addr && b->app_io_addr) return -1;
427  if (a->app_io_addr && !b->app_io_addr) return +1;
428 
429  /*
430  * UDP vs TCP
431  */
432  rcode = a->app_io_addr->proto - b->app_io_addr->proto;
433  if (rcode != 0) return rcode;
434 
435  /*
436  * Check ports.
437  */
438  rcode = a->app_io_addr->inet.src_port - b->app_io_addr->inet.src_port;
439  if (rcode != 0) return rcode;
440 
441  /*
442  * Don't call fr_ipaddr_cmp(), as we need to do our own
443  * checks here. We have various wildcard checks which
444  * aren't globally applicable.
445  */
446 
447  /*
448  * Different address families.
449  */
450  rcode = a->app_io_addr->inet.src_ipaddr.af - b->app_io_addr->inet.src_ipaddr.af;
451  if (rcode != 0) return rcode;
452 
453  /*
454  * If both are bound to interfaces, AND the interfaces
455  * are different, then there is no conflict.
456  */
457  if (a->app_io_addr->inet.src_ipaddr.scope_id && b->app_io_addr->inet.src_ipaddr.scope_id) {
458  rcode = a->app_io_addr->inet.src_ipaddr.scope_id - b->app_io_addr->inet.src_ipaddr.scope_id;
459  if (rcode != 0) return rcode;
460  }
461 
462  rcode = a->app_io_addr->inet.src_ipaddr.prefix - b->app_io_addr->inet.src_ipaddr.prefix;
463  aip = a->app_io_addr->inet.src_ipaddr;
464  bip = b->app_io_addr->inet.src_ipaddr;
465 
466  /*
467  * Mask out the longer prefix to match the shorter
468  * prefix.
469  */
470  if (rcode < 0) {
471  fr_ipaddr_mask(&bip, a->app_io_addr->inet.src_ipaddr.prefix);
472 
473  } else if (rcode > 0) {
474  fr_ipaddr_mask(&aip, b->app_io_addr->inet.src_ipaddr.prefix);
475 
476  }
477 
478  return fr_ipaddr_cmp(&aip, &bip);
479 }

507 /** Instantiate all the virtual servers
508  *
509  * @return
510  * - 0 on success.
511  * - -1 on failure.
512  */
513 int virtual_servers_instantiate(void)
514 {
515  size_t i, server_cnt = virtual_servers ? talloc_array_length(virtual_servers) : 0;
516  rbtree_t *vns_tree = cf_data_value(cf_data_find(virtual_server_root, rbtree_t, "vns_tree"));
517 
518  fr_assert(virtual_servers);
519 
520  DEBUG2("#### Instantiating listeners ####");
521 
522  if (fr_command_register_hook(NULL, NULL, virtual_server_root, cmd_table) < 0) {
523  PERROR("Failed registering radmin commands for virtual servers");
524  return -1;
525  }
526 
527  for (i = 0; i < server_cnt; i++) {
528  fr_virtual_listen_t **listener;
529  size_t j, listen_cnt;
530  CONF_ITEM *ci = NULL;
531  CONF_SECTION *server_cs = virtual_servers[i]->server_cs;
532  CONF_PAIR *ns;
533 
534  listener = virtual_servers[i]->listener;
535  listen_cnt = talloc_array_length(listener);
536 
537  DEBUG("Compiling policies in server %s { ... }", cf_section_name2(server_cs));
538 
539  ns = cf_pair_find(server_cs, "namespace");
540  if (!ns) {
541  cf_log_err(server_cs, "Missing \"namespace\" config item");
542  return -1;
543  }
544 
545  /*
546  * If the dictionary was already loaded, e.g. by
547  * listen_on_read(), then don't do anything else.
548  */
549  if (!cf_data_find(server_cs, virtual_server_dict_t, "dictionary")) {
550  fr_dict_t *dict = NULL;
551  fr_virtual_namespace_t *found = NULL;
552 
553  if (vns_tree) {
554  found = rbtree_finddata(vns_tree, &(fr_virtual_namespace_t){ .namespace = cf_pair_value(ns) });
555  if (found) {
556  virtual_server_dict_set(server_cs, found->dict, true);
557 
558  if ((found->func(server_cs) < 0)) {
559  PERROR("Failed compiling %s", cf_section_name2(server_cs));
560  return -1;
561  }
562  }
563  }
564 
565  /*
566  * Maybe it's a valid namespace, but we
567  * just don't have a dictionary for it.
568  * Load the dictionary now.
569  */
570  if (!found) {
571  char const *value = cf_pair_value(ns);
572 
573  if (fr_dict_protocol_afrom_file(&dict, value, NULL) == 0) {
574  virtual_server_dict_set(server_cs, dict, true);
575  }
576  }
577  }
578 
579  for (j = 0; j < listen_cnt; j++) {
580  fr_virtual_listen_t *listen = listener[j];
581 
582  fr_assert(listen != NULL);
583  fr_assert(listen->proto_module != NULL);
584  fr_assert(listen->app != NULL);
585 
586  if (listen->app->instantiate &&
587  listen->app->instantiate(listen->proto_module->data, listen->proto_module->conf) < 0) {
588  cf_log_err(listen->proto_module->conf, "Could not load virtual server \"%s\".",
589  cf_section_name2(server_cs));
590  return -1;
591  }
592  }
593 
594  if (fr_app_process_instantiate(server_cs) < 0) return -1;
595 
596  /*
597  * Print out warnings for unused "recv" and
598  * "send" sections.
599  */
600  while ((ci = cf_item_next(server_cs, ci))) {
601  char const *name;
602  CONF_SECTION *subcs;
603 
604  if (!cf_item_is_section(ci)) continue;
605 
606  subcs = cf_item_to_section(ci);
607  name = cf_section_name1(subcs);
608 
609  /*
610  * Skip known "other" sections
611  */
612  if ((strcmp(name, "listen") == 0) || (strcmp(name, "client") == 0)) continue;
613 
614  /*
615  * For every other section, warn if it hasn't
616  * been compiled.
617  */
618  if (!cf_data_find(subcs, unlang_group_t, NULL)) {
619  char const *name2;
620

1376

1378 
1379  /*
1380  * The sections are in trees, so this isn't as bad as it
1381  * looks. It's not O(n^2), but O(n logn). But it could
1382  * still be improved.

1385  int rcode;

1387

1389  * We are looking for a specific subsection.

1391  * found.

1395 
1396  subcs = cf_section_find(server, list[i].name, list[i].name2);
1397  if (!subcs) {

1399  list[i].name, list[i].name2);
1400  continue;
1401  }
1402 
1403  /*
1404  * Duplicate sections are forbidden.
1405  */
1406  bad = cf_section_find_next(server, subcs, list[i].name, list[i].name2);
1407  if (bad) {
1408  forbidden:
1409  cf_log_err(bad, "Duplicate sections are forbidden.");
1410  cf_log_err(subcs, "Previous definition occurs here.");
1411  return -1;
1412  }
1413 
1414  rcode = unlang_compile(subcs, list[i].component, rules, &instruction);
1415  if (rcode < 0) return -1;
1416

1418  * Cache the CONF_SECTION which was found.
1419  */
1420  if (uctx) {
1421  if (list[i].offset > 0) {
1422  *(CONF_SECTION **) (((uint8_t *) uctx) + list[i].offset) = subcs;
1423  }
1424  if (list[i].instruction > 0) {
1425  *(void **) (((uint8_t *) uctx) + list[i].instruction) = instruction;
1426  }
1427  }
1428 
1429  found++;
1430  continue;
1431  }
1432 
1433  /*
1434  * Reset this so that we start from the beginning
1435  * again, instead of starting from the last "send
1436  * foo" block.
1437  */
1438  subcs = NULL;
1439 
1440  /*
1441  * Find all subsections with the given first name
1442  * and compile them.
1443  */
1444  while ((subcs = cf_section_find_next(server, subcs, list[i].name, CF_IDENT_ANY))) {
1445  char const *name2;
1446 
1447  name2 = cf_section_name2(subcs);

1451  }
1452

1454  * Duplicate sections are forbidden.
1455  */
1456  bad = cf_section_find_next(server, subcs, list[i].name, name2);
1457  if (bad) goto forbidden;
1458 
1459  rcode = unlang_compile(subcs, list[i].component, rules, NULL);
1460  if (rcode < 0) return -1;
1461 
1462  /*
1463  * Note that we don't store the
1464  * CONF_SECTION here, as it's a wildcard.
1465  *
1466  * @todo - count number of subsections
1467  * and store them in an array?
1468  */
1469  found++;
1470  }
1471  }
