73 static int sql_snprint_error(char *out, size_t outlen, rlm_sql_handle_t *handle, UNUSED rlm_sql_config_t *config)
74 {
75  sb4 errcode = 0;
76  rlm_sql_oracle_conn_t *conn = handle->conn;
77 
78  fr_assert(conn);
79 
80  out[0] = '\0';
81 
82  OCIErrorGet((dvoid *) conn->error, 1, (OraText *) NULL, &errcode, (OraText *) out,
83  outlen, OCI_HTYPE_ERROR);
84  if (!errcode) return -1;
85 
86  return 0;
87 }

89 /** Retrieves any errors associated with the connection handle

100 static size_t sql_error(TALLOC_CTX *ctx, sql_log_entry_t out[], NDEBUG_UNUSED size_t outlen,
101  rlm_sql_handle_t *handle, rlm_sql_config_t *config)
102 {
103  char errbuff[512];
104  int ret;
105 
106  fr_assert(outlen > 0);
107 
108  ret = sql_snprint_error(errbuff, sizeof(errbuff), handle, config);
109  if (ret < 0) return 0;
110 
111  out[0].type = L_ERR;
112  out[0].msg = talloc_strdup(ctx, errbuff);
113 
114  return 1;

141 static sql_rcode_t sql_socket_init(rlm_sql_handle_t *handle, rlm_sql_config_t *config,
142  UNUSED fr_time_delta_t timeout)
143 {
144  char errbuff[512];
145 
146  rlm_sql_oracle_conn_t *conn;
147 
148  MEM(conn = handle->conn = talloc_zero(handle, rlm_sql_oracle_conn_t));
149  talloc_set_destructor(conn, _sql_socket_destructor);
150 
151  /*
152  * Initialises the oracle environment
153  */
154  if (OCIEnvCreate(&conn->env, OCI_DEFAULT | OCI_THREADED, NULL, NULL, NULL, NULL, 0, NULL)) {
155  ERROR("Couldn't init Oracle OCI environment (OCIEnvCreate())");
156 
157  return RLM_SQL_ERROR;
158  }
159 
160  /*
161  * Allocates an error handle
162  */
163  if (OCIHandleAlloc((dvoid *)conn->env, (dvoid **)&conn->error, OCI_HTYPE_ERROR, 0, NULL)) {
164  ERROR("Couldn't init Oracle ERROR handle (OCIHandleAlloc())");
165 
166  return RLM_SQL_ERROR;
167  }
168 
169  /*
170  * Allocate handles for select and update queries
171  */
172  if (OCIHandleAlloc((dvoid *)conn->env, (dvoid **)&conn->query, OCI_HTYPE_STMT, 0, NULL)) {
173  ERROR("Couldn't init Oracle query handles: %s",
174  (sql_snprint_error(errbuff, sizeof(errbuff), handle, config) == 0) ? errbuff : "unknown");
175 
176  return RLM_SQL_ERROR;
177  }
178 
179  /*
180  * Login to the oracle server
181  */
182  if (OCILogon(conn->env, conn->error, &conn->ctx,
183  (OraText const *)config->sql_login, strlen(config->sql_login),
184  (OraText const *)config->sql_password, strlen(config->sql_password),
185  (OraText const *)config->sql_db, strlen(config->sql_db))) {
186  ERROR("Oracle logon failed: '%s'",
187  (sql_snprint_error(errbuff, sizeof(errbuff), handle, config) == 0) ? errbuff : "unknown");
188 
189  return RLM_SQL_ERROR;
190  }
191 
192  return RLM_SQL_OK;
193 }

207 static sql_rcode_t sql_fields(char const **out[], rlm_sql_handle_t *handle, rlm_sql_config_t *config)
208 {
209  rlm_sql_oracle_conn_t *conn = handle->conn;
210  int fields, i, status;
211  char const **names;
212  OCIParam *param;
213 
214  fields = sql_num_fields(handle, config);
215  if (fields <= 0) return RLM_SQL_ERROR;
216 
217  MEM(names = talloc_array(handle, char const *, fields));
218 
219  for (i = 0; i < fields; i++) {
220  OraText *pcol_name = NULL;
221  ub4 pcol_size = 0;
222 
223  status = OCIParamGet(conn->query, OCI_HTYPE_STMT, conn->error, (dvoid **)&param, i + 1);
224  if (status != OCI_SUCCESS) {
225  ERROR("OCIParamGet(OCI_HTYPE_STMT) failed in sql_fields()");
226  error:
227  talloc_free(names);
228 
229  return RLM_SQL_ERROR;
230  }
231 
232  status = OCIAttrGet((dvoid **)param, OCI_DTYPE_PARAM, &pcol_name, &pcol_size,
233  OCI_ATTR_NAME, conn->error);
234  if (status != OCI_SUCCESS) {
235  ERROR("OCIParamGet(OCI_ATTR_NAME) failed in sql_fields()");
236 
237  goto error;
238  }
239 
240  names[i] = (char const *)pcol_name;
241  }
242 
243  *out = names;
244 
245  return RLM_SQL_OK;
246 }

248 static sql_rcode_t sql_query(rlm_sql_handle_t *handle, rlm_sql_config_t *config, char const *query)
249 {
250  int status;
251  rlm_sql_oracle_conn_t *conn = handle->conn;
252 
253  OraText *oracle_query;
254 
255  memcpy(&oracle_query, &query, sizeof(oracle_query));
256 
257  if (!conn->ctx) {
258  ERROR("Socket not connected");
259 
260  return RLM_SQL_RECONNECT;
261  }
262 
263  if (OCIStmtPrepare(conn->query, conn->error, oracle_query, strlen(query),
264  OCI_NTV_SYNTAX, OCI_DEFAULT)) {
265  ERROR("prepare failed in sql_query");
266 
267  return RLM_SQL_ERROR;
268  }
269 
270  status = OCIStmtExecute(conn->ctx, conn->query, conn->error, 1, 0,
271  NULL, NULL, OCI_COMMIT_ON_SUCCESS);
272 
273  if (status == OCI_SUCCESS) return RLM_SQL_OK;
274  if (status == OCI_ERROR) {
275  ERROR("execute query failed in sql_query");
276 
277  return sql_check_reconnect(handle, config);
278  }
279 
280  return RLM_SQL_ERROR;
281 }

283 static sql_rcode_t sql_select_query(rlm_sql_handle_t *handle, rlm_sql_config_t *config, char const *query)
284 {
285  int status;
286  char **row;
287 
288  int i;
289  OCIParam *param;
290  OCIDefine *define;
291 
292  ub2 dtype;
293  ub2 dsize;
294 
295  sb2 *ind;
296 
297  OraText *oracle_query;
298 
299  rlm_sql_oracle_conn_t *conn = handle->conn;
300 
301  memcpy(&oracle_query, &query, sizeof(oracle_query));
302 
303  if (OCIStmtPrepare(conn->query, conn->error, oracle_query, strlen(query), OCI_NTV_SYNTAX,
304  OCI_DEFAULT)) {
305  ERROR("prepare failed in sql_select_query");
306 
307  return RLM_SQL_ERROR;
308  }
309 
310  /*
311  * Retrieve a single row
312  */
313  status = OCIStmtExecute(conn->ctx, conn->query, conn->error, 0, 0, NULL, NULL, OCI_DEFAULT);
314  if (status == OCI_NO_DATA) return RLM_SQL_OK;
315  if (status != OCI_SUCCESS) {
316  ERROR("query failed in sql_select_query");
317 
318  return sql_check_reconnect(handle, config);
319  }
320 
321  /*
322  * We only need to do this once per result set, because
323  * the number of columns won't change.
324  */
325  if (conn->col_count == 0) {
326  conn->col_count = sql_num_fields(handle, config);
327 
328  if (conn->col_count == 0) return RLM_SQL_ERROR;
329  }
330 
331  MEM(row = talloc_zero_array(conn, char*, conn->col_count + 1));
332  MEM(ind = talloc_zero_array(row, sb2, conn->col_count + 1));
333 
334  for (i = 0; i < conn->col_count; i++) {
335  status = OCIParamGet(conn->query, OCI_HTYPE_STMT, conn->error, (dvoid **)&param, i + 1);
336  if (status != OCI_SUCCESS) {
337  ERROR("OCIParamGet() failed in sql_select_query");
338 
339  goto error;
340  }
341 
342  status = OCIAttrGet((dvoid*)param, OCI_DTYPE_PARAM, (dvoid*)&dtype, NULL, OCI_ATTR_DATA_TYPE,
343  conn->error);
344  if (status != OCI_SUCCESS) {
345  ERROR("OCIAttrGet() failed in sql_select_query");
346 
347  goto error;
348  }
349 
350  dsize = MAX_DATASTR_LEN;
351 
352  /*
353  * Use the retrieved length of dname to allocate an output buffer, and then define the output
354  * variable (but only for char/string type columns).
355  */
356  switch (dtype) {
357 #ifdef SQLT_AFC
358  case SQLT_AFC: /* ansii fixed char */
359 #endif
360 #ifdef SQLT_AFV
361  case SQLT_AFV: /* ansii var char */
362 #endif
363  case SQLT_VCS: /* var char */
364  case SQLT_CHR: /* char */
365  case SQLT_STR: /* string */
366  status = OCIAttrGet((dvoid *)param, OCI_DTYPE_PARAM, (dvoid *)&dsize, NULL,
367  OCI_ATTR_DATA_SIZE, conn->error);
368  if (status != OCI_SUCCESS) {
369  ERROR("OCIAttrGet() failed in sql_select_query");
370 
371  goto error;
372  }
373 
374  MEM(row[i] = talloc_zero_array(row, char, dsize + 1));
375 
376  break;
377  case SQLT_DAT:
378  case SQLT_INT:
379  case SQLT_UIN:
380  case SQLT_FLT:
381  case SQLT_PDN:
382  case SQLT_BIN:
383  case SQLT_NUM:
384  MEM(row[i] = talloc_zero_array(row, char, dsize + 1));
385 
386  break;
387  default:
388  dsize = 0;
389  row[i] = NULL;
390  break;
391  }
392 
393  ind[i] = 0;
394 
395  /*
396  * Grab the actual row value and write it to the buffer we allocated.
397  */
398  status = OCIDefineByPos(conn->query, &define, conn->error, i + 1, (ub1 *)row[i], dsize + 1, SQLT_STR,
399  (dvoid *)&ind[i], NULL, NULL, OCI_DEFAULT);
400 
401  if (status != OCI_SUCCESS) {
402  ERROR("OCIDefineByPos() failed in sql_select_query");
403  goto error;
404  }
405  }
406 
407  conn->row = row;
408  conn->ind = ind;
409 
410  return RLM_SQL_OK;
411 
412  error:
413  talloc_free(row);
414 
415  return RLM_SQL_ERROR;
416 }

429 static sql_rcode_t sql_fetch_row(rlm_sql_row_t *out, rlm_sql_handle_t *handle, rlm_sql_config_t *config)
430 {
431  int status;
432  rlm_sql_oracle_conn_t *conn = handle->conn;
433 
434  *out = NULL;
435 
436  if (!conn->ctx) {
437  ERROR("Socket not connected");
438 
439  return RLM_SQL_RECONNECT;
440  }
441 
442  handle->row = NULL;
443 
444  status = OCIStmtFetch(conn->query, conn->error, 1, OCI_FETCH_NEXT, OCI_DEFAULT);
445  if (status == OCI_SUCCESS) {
446  *out = handle->row = conn->row;
447 
448  return RLM_SQL_OK;
449  }
450 
451  if (status == OCI_NO_DATA) {
452  handle->row = 0;
453 
454  return RLM_SQL_NO_MORE_ROWS;
455  }
456 
457  if (status == OCI_ERROR) {
458  ERROR("fetch failed in sql_fetch_row");
459  return sql_check_reconnect(handle, config);
460  }
461 
462  return RLM_SQL_ERROR;
463 }
