397  *
398  * @param[in] a the first file descriptor handle.
399  * @param[in] b the second file descriptor handle.

489 /** Build a new evset based on function pointers present
490  *
491  * @note The contents of active functions may be inconsistent if this function errors. But the
492  * only time that will occur is if the caller passed invalid arguments.
493  *
494  * @param[out] out_kev where to write the evset.
495  * @param[in] outlen length of output buffer.
496  * @param[out] active The set of function pointers with active filters.
497  * @param[in] ef event to insert.
498  * @param[in] new Functions to map to filters.
499  * @param[in] prev Previous set of functions mapped to filters.
500  * @return
501  * - >= 0 the number of changes written to out.
502  * - < 0 an error ocurred.
503  */
504 static ssize_t fr_event_build_evset(struct kevent out_kev[], size_t outlen, fr_event_funcs_t *active,
505  fr_event_fd_t *ef,
506  fr_event_funcs_t const *new, fr_event_funcs_t const *prev)
507 {
508  struct kevent *out = out_kev, *end = out + outlen;
509  fr_event_func_map_t const *map;
510  struct kevent add[10], *add_p = add;
511  size_t i;
512 
513  EVENT_DEBUG("Building new evset for FD %i (new %p, prev %p)", ef->fd, new, prev);
514 
515  /*
516  * Iterate over the function map, setting/unsetting
517  * filters and filter flags.
518  */
519  for (map = ef->map; map->name; map++) {
520  bool has_current_func = false;
521  bool has_prev_func = false;
522  uint32_t current_fflags = 0;
523  uint32_t prev_fflags = 0;
524 
525  do {
526  fr_event_fd_cb_t prev_func;
527  fr_event_fd_cb_t new_func;
528 
529  /*
530  * If the previous value was the 'noop'
531  * callback, it's identical to being unset.
532  */
533  prev_func = *(fr_event_fd_cb_t const *)((uint8_t const *)prev + map->offset);
534  if (prev_func && (prev_func != fr_event_fd_noop)) {
535  EVENT_DEBUG("\t%s prev set (%p)", map->name, prev_func);
536  prev_fflags |= map->fflags;
537  has_prev_func = true;
538  } else {
539  EVENT_DEBUG("\t%s prev unset", map->name);
540  }
541 
542  new_func = *(fr_event_fd_cb_t const *)((uint8_t const *)new + map->offset);
543  if (new_func) {
544  EVENT_DEBUG("\t%s curr set (%p)", map->name, new_func);
545  current_fflags |= map->fflags;
546  has_current_func = true;
547 
548  /*
549  * Check the filter will work for the
550  * type of file descriptor specified.
551  */
552  if (!(map->type & ef->type)) {
553  fr_strerror_printf("kevent %s (%s), can't be applied to fd of type %s",
554  map->name,
555  fr_table_str_by_value(kevent_filter_table, map->filter, "<INVALID>"),
556  fr_table_str_by_value(fr_event_fd_type_table,
557  map->type, "<INVALID>"));
558  return -1;
559  }
560 
561  /*
562  * Mark this filter function as active
563  */
564  memcpy((uint8_t *)active + map->offset, (uint8_t const *)new + map->offset,
565  sizeof(fr_event_fd_cb_t));
566  } else {
567  EVENT_DEBUG("\t%s curr unset", map->name);
568 
569  /*
570  * Mark this filter function as inactive
571  * by setting it to the 'noop' callback.
572  */
573  *((fr_event_fd_cb_t *)((uint8_t *)active + map->offset)) = fr_event_fd_noop;
574  }
575 
576  if (!(map + 1)->coalesce) break;
577  map++;
578  } while (1);
579 
580  if (out > end) {
581  fr_strerror_printf("Out of memory to store kevent filters");
582  return -1;
583  }
584 
585  /*
586  * Upsert if we add a function or change the flags.
587  */
588  if (has_current_func &&
589  (!has_prev_func || (current_fflags != prev_fflags))) {
590  if ((size_t)(add_p - add) >= (NUM_ELEMENTS(add))) {
591  fr_strerror_printf("Out of memory to store kevent EV_ADD filters");
592  return -1;
593  }
594  EVENT_DEBUG("\tEV_SET EV_ADD filter %s (%i), flags %i, fflags %i",
595  fr_table_str_by_value(kevent_filter_table, map->filter, "<INVALID>"),
596  map->filter, map->flags, current_fflags);
597  EV_SET(add_p++, ef->fd, map->filter, map->flags, current_fflags, 0, ef);
598 
599  /*
600  * Delete if we remove a function.
601  */
602  } else if (!has_current_func && has_prev_func) {
603  EVENT_DEBUG("\tEV_SET EV_DELETE filter %s (%i), flags %i, fflags %i",
604  fr_table_str_by_value(kevent_filter_table, map->filter, "<INVALID>"),
605  map->filter, EV_DELETE, 0, 0);
606  EV_SET(out++, ef->fd, map->filter, EV_DELETE, 0, 0, ef);
607  }
608  }
609 
610  /*
611  * kevent is fine with adds/deletes in the same operation
612  * on the same file descriptor, but libkqueue doesn't do
613  * any kind of coalescing or ordering so you get an EEXIST
614  * error.
615  */
616  for (i = 0; i < (size_t)(add_p - add); i++) memcpy(out++, &add[i], sizeof(*out));
617 
618  return out - out_kev;
619 }

627  * @param[in] fd to discover the type of.
628  * @return
629  * - 0 on success.
630  * - -1 on failure.
631  */
632 static int fr_event_fd_type_set(fr_event_fd_t *ef, int fd)
633 {
634  socklen_t opt_len = sizeof(ef->sock_type);
635 
636  /*
637  * It's a socket or PCAP socket
638  */
639  if (getsockopt(fd, SOL_SOCKET, SO_TYPE, &ef->sock_type, &opt_len) == 0) {
640 #ifdef SO_GET_FILTER
641  opt_len = 0;
642  if (unlikely(getsockopt(fd, SOL_SOCKET, SO_ATTACH_FILTER, NULL, &opt_len) < 0)) {
643  fr_strerror_printf("Failed determining PF status: %s", fr_syserror(errno));
644  return -1;
645  }
646  if (opt_len) {
647  ef->type = FR_EVENT_FD_PCAP;
648  } else
649 #endif
650  {
651  ef->type = FR_EVENT_FD_SOCKET;
652  }
653 
654  /*
655  * It's a file or directory
656  */
657  } else {
658  struct stat buf;
659 
660  if (errno != ENOTSOCK) {
661  fr_strerror_printf("Failed retrieving socket type: %s", fr_syserror(errno));
662  return -1;
663  }
664 
665  if (fstat(fd, &buf) < 0) {
666  fr_strerror_printf("Failed calling stat() on file: %s", fr_syserror(errno));
667  return -1;
668  }
669 
670  if (S_ISDIR(buf.st_mode)) {
671  ef->type = FR_EVENT_FD_DIRECTORY;
672  } else {
673  ef->type = FR_EVENT_FD_FILE;
674  }
675  }
676  ef->fd = fd;

678  return 0;

681 /** Remove a file descriptor from the event loop and rbtree but don't explicitly free it
682  *
683  *
684  * @param[in] ef to remove.
685  * @return
686  * - 0 on success.
687  * - -1 on error;
688  */
689 static int _event_fd_delete(fr_event_fd_t *ef)
690 {
691  struct kevent evset[10];
692  int count = 0;
693  fr_event_list_t *el = ef->el;
694  fr_event_funcs_t funcs;
695 
696  /*
697  * Already been removed from the various trees and
698  * the event loop.
699  */
700  if (ef->is_registered) {
701  memset(&funcs, 0, sizeof(funcs));
702 
703  fr_assert(ef->armour == 0);
704 
705  /*
706  * If this fails, it's a pretty catastrophic error.
707  */
708  count = fr_event_build_evset(evset, sizeof(evset)/sizeof(*evset), &ef->active, ef, &funcs, &ef->active);
709  if (count > 0) {
710  int ret;
711 
712  /*
713  * If this fails, assert on debug builds.
714  */
715  ret = kevent(el->kq, evset, count, NULL, 0, NULL);
716  if (!fr_cond_assert_msg(ret >= 0,
717  "FD %i was closed without being removed from the KQ: %s",
718  ef->fd, fr_syserror(errno))) {
719  return -1; /* Prevent the free, and leave the fd in the trees */
720  }
721  }
722 
723  rbtree_deletebydata(el->fds, ef);
724  ef->is_registered = false;
725  }
726 
727  /*
728  * Insert into the deferred free list, event will be
729  * freed later.
730  */
731  if (el->in_handler) {
732  /*
733  * Don't allow the same event to be
734  * inserted into the free list multiple
735  * times.
736  *
737  * This can happen if the same ef is
738  * delivered by multiple filters, i.e.
739  * if EVFILT_READ and EVFILT_WRITE
740  * were both high, and both handlers
741  * attempted to delete the event
742  * we'd need to prevent the event being
743  * inserted into the free list multiple
744  * times.
745  */
746  if (!ef->in_fd_to_free) {
747  ef->next = el->fd_to_free; /* Link into the deferred free list */
748  el->fd_to_free = ef;
749  ef->in_fd_to_free = true;
750  }
751  return -1; /* Will be freed later */
752  }
753 
754  return 0;
755 }

802 /** Suspend/resume a subset of filters
803  *
804  * This function trades producing useful errors for speed.
805  *
806  * An example of suspending the read filter for an FD would be:
807  @code {.c}
808  static fr_event_update_t pause_read[] = {
809  FR_EVENT_SUSPEND(fr_event_io_func_t, read),
810  { 0 }
811  }
812 
813  fr_event_filter_update(el, fd, FR_EVENT_FILTER_IO, pause_read);
814  @endcode
815  *
816  * @param[in] el to update descriptor in.
817  * @param[in] fd to update filters for.
818  * @param[in] filter The type of filter to update.
819  * @param[in] updates An array of updates to toggle filters on/off without removing
820  * the callback function.
821  */
822 int _fr_event_filter_update(NDEBUG_LOCATION_ARGS
823  fr_event_list_t *el, int fd, fr_event_filter_t filter, fr_event_update_t updates[])
824 {
825  fr_event_fd_t *ef;
826  size_t i;
827  fr_event_funcs_t curr_active, curr_stored;
828  struct kevent evset[10];
829  int count = 0;
830 
831  ef = rbtree_finddata(el->fds, &(fr_event_fd_t){ .fd = fd, .filter = filter });
832  if (unlikely(!ef)) {
833  fr_strerror_printf("No events are registered for fd %i", fd);
834  return -1;
835  }
836 
837 #ifndef NDEBUG
838  ef->file = file;
839  ef->line = line;
840 #endif
841 
842  /*
843  * Cheapest way of ensuring this function can error without
844  * leaving everything in an inconsistent state.
845  */
846  memcpy(&curr_active, &ef->active, sizeof(curr_active));
847  memcpy(&curr_stored, &ef->stored, sizeof(curr_stored));
848 
849  /*
850  * Apply modifications to our copies of the active/stored array.
851  */
852  for (i = 0; updates[i].op; i++) {
853  switch (updates[i].op) {
854  default:
855  case FR_EVENT_OP_SUSPEND:
856  fr_assert(ef->armour == 0); /* can't suspect protected FDs */
857  memcpy((uint8_t *)&ef->stored + updates[i].offset,
858  (uint8_t *)&ef->active + updates[i].offset, sizeof(fr_event_fd_cb_t));
859  memset((uint8_t *)&ef->active + updates[i].offset, 0, sizeof(fr_event_fd_cb_t));
860  break;
861 
862  case FR_EVENT_OP_RESUME:
863  memcpy((uint8_t *)&ef->active + updates[i].offset,
864  (uint8_t *)&ef->stored + updates[i].offset, sizeof(fr_event_fd_cb_t));
865  memset((uint8_t *)&ef->stored + updates[i].offset, 0, sizeof(fr_event_fd_cb_t));
866  break;
867  }
868  }
869 
870  count = fr_event_build_evset(evset, sizeof(evset)/sizeof(*evset), &ef->active,
871  ef, &ef->active, &curr_active);
872  if (unlikely(count < 0)) {
873  error:
874  memcpy(&ef->active, &curr_active, sizeof(curr_active));
875  memcpy(&ef->stored, &curr_stored, sizeof(curr_stored));
876  return -1;
877  }
878 
879  if (count && unlikely(kevent(el->kq, evset, count, NULL, 0, NULL) < 0)) {
880  fr_strerror_printf("Failed updating filters for FD %i: %s", ef->fd, fr_syserror(errno));
881  goto error;
882  }
883 
884  return 0;
885 }

887 /** Insert a filter for the specified fd
888  *
889  * @param[in] ctx to bind lifetime of the event to.
890  * @param[out] ef_out Previously allocated ef, or NULL.
891  * @param[in] el to insert fd callback into.
892  * @param[in] fd to install filters for.
893  * @param[in] filter one of the #fr_event_filter_t values.
894  * @param[in] funcs Structure containing callback functions. If a function pointer
895  * is set, the equivalent kevent filter will be installed.
896  * @param[in] error function to call when an error occurs on the fd.
897  * @param[in] uctx to pass to handler.
898  */
899 int _fr_event_filter_insert(NDEBUG_LOCATION_ARGS
900  TALLOC_CTX *ctx, fr_event_fd_t **ef_out,
901  fr_event_list_t *el, int fd,
902  fr_event_filter_t filter,
903  void *funcs, fr_event_error_cb_t error,
904  void *uctx)
905 {
906  ssize_t count;
907  fr_event_fd_t *ef;
908  fr_event_funcs_t active;
909  struct kevent evset[10];
910 
911  if (unlikely(!el)) {
912  fr_strerror_printf("Invalid argument: NULL event list");
913  return -1;
914  }
915 
916  if (unlikely(fd < 0)) {
917  fr_strerror_printf("Invalid arguments: Bad FD %i", fd);
918  return -1;
919  }
920 
921  if (unlikely(el->exit)) {
922  fr_strerror_printf("Event loop exiting");
923  return -1;
924  }
925 
926  if (!ef_out || !*ef_out) {
927  ef = rbtree_finddata(el->fds, &(fr_event_fd_t){ .fd = fd, .filter = filter });
928  } else {
929  ef = *ef_out;
930  fr_assert((fd < 0) || (ef->fd == fd));
931  }
932 
933  /*
934  * Need to free the event to change the talloc link.
935  *
936  * This is generally bad. If you hit this
937  * code path you probably screwed up somewhere.
938  */
939  if (unlikely(ef && (ef->linked_ctx != ctx))) TALLOC_FREE(ef);
940 
941  /*
942  * No pre-existing event. Allocate an entry
943  * for insertion into the rbtree.
944  */
945  if (!ef) {
946  ef = talloc_zero(el, fr_event_fd_t);
947  if (unlikely(!ef)) {
948  fr_strerror_printf("Out of memory");
949  return -1;
950  }
951  talloc_set_destructor(ef, _event_fd_delete);
952 
953  /*
954  * Bind the lifetime of the event to the specified
955  * talloc ctx. If the talloc ctx is freed, the
956  * event will also be freed.
957  */
958  if (ctx != el) talloc_link_ctx(ctx, ef);
959  ef->linked_ctx = ctx;
960  ef->el = el;

962  /*
963  * Determine what type of file descriptor
964  * this is.
965  */
966  if (fr_event_fd_type_set(ef, fd) < 0) {
967  free:
968  talloc_free(ef);
969  return -1;
970  }
971 
972  switch (filter) {
973  case FR_EVENT_FILTER_IO:
974  ef->map = io_func_map;
975  break;
976 
977  case FR_EVENT_FILTER_VNODE:
978  ef->map = vnode_func_map;
979  break;
980 
981  default:
982  fr_strerror_printf("Filter %i not supported", filter);
983  goto free;
984  }
985 
986  count = fr_event_build_evset(evset, sizeof(evset)/sizeof(*evset), &ef->active, ef, funcs, &ef->active);
987  if (count < 0) goto free;
988  if (count && (unlikely(kevent(el->kq, evset, count, NULL, 0, NULL) < 0))) {
989  fr_strerror_printf("Failed inserting filters for FD %i: %s", fd, fr_syserror(errno));
990  goto free;
991  }
992 
993  ef->filter = filter;
994  rbtree_insert(el->fds, ef);
995  ef->is_registered = true;
996 
997  /*
998  * Pre-existing event, update the filters and
999  * functions associated with the file descriptor.
1000  */
1001  } else {
1002  fr_assert(ef->is_registered == true);
1003 
1004  /*
1005  * Take a copy of the current set of active
1006  * functions, so we can error out in a
1007  * consistent state.
1008  */
1009  memcpy(&active, &ef->active, sizeof(ef->active));
1010 
1011  fr_assert((ef->armour == 0) || ef->active.io.read);
1012 
1013  count = fr_event_build_evset(evset, sizeof(evset)/sizeof(*evset), &ef->active, ef, funcs, &ef->active);
1014  if (count < 0) {
1015  error:
1016  memcpy(&ef->active, &active, sizeof(ef->active));
1017  return -1;
1018  }
1019  if (count && (unlikely(kevent(el->kq, evset, count, NULL, 0, NULL) < 0))) {
1020  fr_strerror_printf("Failed modifying filters for FD %i: %s", fd, fr_syserror(errno));
1021  goto error;
1022  }
1023 
1024  /*
1025  * Clear any previously suspended functions
1026  */
1027  memset(&ef->stored, 0, sizeof(ef->stored));
1028  }
1029 
1030 #ifndef NDEBUG
1031  ef->file = file;
1032  ef->line = line;
1033 #endif
1034  ef->error = error;
1035  ef->uctx = uctx;
1036 
1037  if (ef_out) *ef_out = ef;
1038 
1039  return 0;
1040 }

1216  * @param[in] when we should run the event.
1217  * @param[in] callback function to execute if the event fires.
1218  * @param[in] uctx user data to pass to the event.

1235  fr_strerror_printf("Invalid arguments: NULL callback");

1251  * and allocating a new one. This is to reduce memory
1252  * churn for repeat events.

1254  if (!*ev_p) {

1719 /** Gather outstanding timer and file descriptor events
1720  *
1721  * @param[in] el to process events for.
1722  * @param[in] now The current time.
1723  * @param[in] wait if true, block on the kevent() call until a timer or file descriptor event occurs.
1724  * @return
1725  * - <0 error, or the event loop is exiting
1726  * - the number of outstanding I/O events, +1 if at least one timer will fire.
1727  */
1728 int fr_event_corral(fr_event_list_t *el, fr_time_t now, bool wait)
1729 {
1730  fr_time_t when, *wake;
1731  struct timespec ts_when, *ts_wake;
1732  fr_event_pre_t *pre;
1733  int num_fd_events;
1734  bool timer_event_ready = false;
1735  fr_event_timer_t *ev;
1736 #ifdef LOCAL_PID
1737  fr_event_pid_t *pid;
1738  fr_heap_iter_t iter;
1739  int num_pid_events;
1740 #endif
1741 
1742 
1743  el->num_fd_events = 0;
1744 
1745  if (el->will_exit || el->exit) {
1746  el->exit = el->will_exit;
1747 
1748  fr_strerror_printf("Event loop exiting");
1749  return -1;
1750  }
1751 
1752  /*
1753  * By default we wait for 0ns, which means returning
1754  * immediately from kevent().
1755  */
1756  when = 0;
1757  wake = &when;
1758  el->now = now;
1759 
1760  /*
1761  * See when we have to wake up. Either now, if the timer
1762  * events are in the past. Or, we wait for a future
1763  * timer event.
1764  */
1765  ev = fr_heap_peek(el->times);
1766  if (ev) {
1767  if (ev->when <= el->now) {
1768  timer_event_ready = true;
1769 
1770  } else if (wait) {
1771  when = ev->when - el->now;
1772 
1773  } /* else we're not waiting, leave "when == 0" */
1774 
1775  } else if (wait) {
1776  /*
1777  * We're asked to wait, but there's no timer
1778  * event. We can then sleep forever.
1779  */
1780  wake = NULL;
1781  }
1782 
1783  /*
1784  * Run the status callbacks. It may tell us that the
1785  * application has more work to do, in which case we
1786  * re-set the timeout to be instant.
1787  *
1788  * We only run these callbacks if the caller is otherwise
1789  * idle.
1790  */
1791  if (wait) {
1792  for (pre = fr_dlist_head(&el->pre_callbacks);
1793  pre != NULL;
1794  pre = fr_dlist_next(&el->pre_callbacks, pre)) {
1795  if (pre->callback(pre->uctx, wake ? *wake : 0) > 0) {
1796  wake = &when;
1797  when = 0;
1798  }
1799  }
1800  }
1801 
1802  /*
1803  * Wake is the delta between el->now
1804  * (the event loops view of the current time)
1805  * and when the event should occur.
1806  */
1807  if (wake) {
1808  ts_when = fr_time_delta_to_timespec(when);
1809  ts_wake = &ts_when;
1810  } else {
1811  ts_wake = NULL;
1812  }
1813 
1814  /*
1815  * Populate el->events with the list of I/O events
1816  * that occurred since this function was last called
1817  * or wait for the next timer event.
1818  */
1819 #ifndef LOCAL_PID
1820  num_fd_events = kevent(el->kq, NULL, 0, el->events, FR_EV_BATCH_FDS, ts_wake);
1821 
1822  /*
1823  * Interrupt is different from timeout / FD events.
1824  */
1825  if (unlikely(num_fd_events < 0)) {
1826  if (errno == EINTR) {
1827  return 0;
1828  } else {
1829  fr_strerror_printf("Failed calling kevent: %s", fr_syserror(errno));
1830  return -1;
1831  }
1832  }
1833 
1834 #else /* LOCAL_PID */
1835  num_pid_events = 0;
1836 
1837  /*
1838  * Brute-force wait for all open PIDs
1839  */
1840  for (pid = fr_heap_iter_init(el->pids, &iter);
1841  pid != NULL;
1842  pid = fr_heap_iter_next(el->pids, &iter)) {
1843  int status;
1844 
1845  if (waitpid(pid->pid, &status, WNOHANG) != pid->pid) continue;
1846 
1847  /*
1848  * Synthesize a kevent for the exit status.
1849  */
1850  el->events[num_pid_events].filter = EVFILT_PROC;
1851  el->events[num_pid_events].ident = pid->pid;
1852  el->events[num_pid_events].flags = 0;
1853  el->events[num_pid_events].fflags = NOTE_EXIT;
1854  el->events[num_pid_events].data = status;
1855  el->events[num_pid_events].udata = pid;
1856 
1857  num_pid_events++;
1858 
1859  /*
1860  * Limit the number of PID events to no
1861  * more than half of events. This should
1862  * be more than enough for most exec
1863  * scenarios. And, it allows us to still
1864  * service FD events.
1865  */
1866  if (num_pid_events >= (FR_EV_BATCH_FDS / 2)) break;
1867  }
1868 
1869  /*
1870  * Wake up NOW, in order to service the PID events.
1871  */
1872  if (num_pid_events > 0) {
1873  ts_when.tv_sec = 0;
1874  ts_when.tv_nsec = 0;
1875  ts_wake = &ts_when;
1876  }
1877 
1878  num_fd_events = kevent(el->kq, NULL, 0, &el->events[num_pid_events], FR_EV_BATCH_FDS - num_pid_events, ts_wake);
1879 
1880  /*
1881  * Interrupt is different from timeout / FD events.
1882  */
1883  if (unlikely(num_fd_events < 0)) {
1884  if (errno != EINTR) {
1885  fr_strerror_printf("Failed calling kevent: %s", fr_syserror(errno));
1886  return -1;
1887  }
1888 
1889  if (num_pid_events == 0) {
1890  return 0;
1891  }
1892 
1893  num_fd_events = 0; /* still service the PID events */
1894  }
1895 
1896  num_fd_events += num_pid_events;
1897 #endif /* LOCAL_PID */
1898 
1899  el->num_fd_events = num_fd_events;
1900 
1901  EVENT_DEBUG("%s - kevent returned %u FD events", __FUNCTION__, el->num_fd_events);
1902 
1903  /*
1904  * If there are no FD events, we must have woken up from a timer
1905  */
1906  if (!num_fd_events) {
1907  el->now += when;
1908  if (wait) timer_event_ready = true;
1909  }
1910  /*
1911  * The caller doesn't really care what the value of the
1912  * return code is. Just that it's greater than zero if
1913  * events needs servicing.
1914  *
1915  * num_fd_events > 0 - if kevent() returns FD events
1916  * timer_event_ready > 0 - if there were timers ready BEFORE or AFTER calling kevent()
1917  */
1918  return num_fd_events + timer_event_ready;
1919 }

1923  * @param[in] el containing events to service.
1924  */
1925 void fr_event_service(fr_event_list_t *el)
1926 {
1927  int i;
1928  fr_event_post_t *post;
1929  fr_time_t when;
1930  fr_event_timer_t *ev;
1931 
1932  if (unlikely(el->exit)) return;
1933 
1934  EVENT_DEBUG("%s - Servicing %u FD events", __FUNCTION__, el->num_fd_events);
1935 
1936  /*
1937  * Run all of the file descriptor events.
1938  */
1939  el->in_handler = true;
1940  for (i = 0; i < el->num_fd_events; i++) {
1941  fr_event_fd_t *ef;
1942  int fd_errno = 0;
1943  int flags = el->events[i].flags;
1944 
1945  /*
1946  * Process any user events
1947  */
1948  switch (el->events[i].filter) {

1952 
1953  /*
