65 /** Wrap fopencookie seek function
66  *
67  */
68 static fpos_t _seek(void *c, fpos_t offset, int whence)
69 {
70  fr_funopen_cookie_t *oc = talloc_get_type_abort(c, fr_funopen_cookie_t);
71  int ret;
72 
73  /*
74  * fopencookie seek cb should return
75  * and update offset to be the new position,
76  * or not zero and leave offset untouched.
77  *
78  * funopen seek cb should return -1 for an error
79  * or the new offset on success.
80  */
81  ret = oc->io_funcs.seek(oc->cookie, (off64_t *)&offset, whence);
82  if (ret != 0) return -1;
83 
84  return offset;
85 }

87 /** Wrap fopencookie close function and free our fr_funopen_cookie_t
88  *
89  */
90 static int _close(void *c)
91 {
92  fr_funopen_cookie_t *oc = talloc_get_type_abort(c, fr_funopen_cookie_t);
93  int ret = oc->io_funcs.close ? oc->io_funcs.close(oc->cookie) : 0;
94 
95  talloc_free(oc);

100 FILE *fopencookie(void *cookie, const char *mode, cookie_io_functions_t io_funcs)
101 {
102  fr_funopen_cookie_t *oc;
103  int need = 0;
104  FILE *f;
105  char const *p;
106 
107  /*
108  * Process mode string as described by `man fopen`
109  */
110  for (p = mode; *p != '\0'; p++) {
111  switch (*p) {
112  case 'r':
113  need |= NEED_READ;
114 
115  if (p[1] == 'b') p++; /* Skip binary */
116  if (p[1] == '+') {
117  p++;
118  need |= NEED_WRITE;
119  continue;
120  }
121  continue;
122 
123  case 'w':
124  need |= NEED_WRITE;
125 
126  if (p[1] == 'b') p++; /* Skip binary */
127  if (p[1] == '+') {
128  p++;
129  need |= NEED_READ;
130  continue;
131  }
132  continue;
133 
134  case 'a':
135  need |= (NEED_SEEK | NEED_WRITE);
136 
137  if (p[1] == 'b') p++; /* Skip binary */
138  if (p[1] == '+') {
139  p++;
140  need |= NEED_READ;
141  continue;
142  }
143  continue;
144 
145  /*
146  * 'b' is also allowed as the last char
147  */
148  case 'b':
149  if (p[1] != '\0') {
150  invalid_arg:
151  errno = EINVAL;
152  return NULL;
153  }
154  continue; /* Loop will exit next iteration */
155 
156  default:
157  goto invalid_arg;
158  }
159  }
160 
161  if ((need & NEED_READ) && !io_funcs.read) goto invalid_arg;
162  if ((need & NEED_WRITE) && !io_funcs.write) goto invalid_arg;
163  if ((need & NEED_SEEK) && !io_funcs.seek) goto invalid_arg;
164 
165  oc = talloc_zero(NULL, fr_funopen_cookie_t);
166  if (!oc) {
167  errno = ENOMEM;
168  return NULL;
169  }
170 
171  oc->io_funcs = io_funcs;
172  oc->cookie = cookie;
173 
174  f = funopen(oc,
175  oc->io_funcs.read ? _read : NULL,
176  oc->io_funcs.write ? _write : NULL,
177  oc->io_funcs.seek ? _seek : NULL,
178  _close);
179  if (!f) {
180  talloc_free(oc);
181  return NULL;
182  }
183 
184  if (need & NEED_SEEK) {
185  if (fseek(f, 0L, SEEK_END) < 0) {
186  fclose(f);
187  talloc_free(oc);
188  return NULL;
189  }
190  }
191 
192  return f;
193 }
