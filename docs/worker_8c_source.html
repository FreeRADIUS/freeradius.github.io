17  /**
18  * $Id: 66d64c2b1dedca1252fed07a4da4b5c9352563bf $
19  *
20  * @brief Worker thread functions.
21  * @file io/worker.c
22  *
23  * The "worker" thread is the one responsible for the bulk of the
24  * work done when processing a request. Workers are spawned by the
25  * scheduler, and create a kqueue (KQ) and control-plane
26  * Atomic Queue (AQ) for control-plane communication.
27  *
28  * When a network thread discovers that it needs more workers, it
29  * asks the scheduler for a KQ/AQ combination. The network thread
30  * then creates a channel dedicated to that worker, and sends the
31  * channel to the worker in a "new channel" message. The worker
32  * receives the channel, and sends an ACK back to the network thread.
33  *
34  * The network thread then sends the worker new packets, which the
35  * worker receives and processes.
36  *
37  * When a packet is decoded, it is put into the "runnable" heap, and
38  * also into the "time_order" heap. The main loop fr_worker() then
39  * pulls new requests off of this heap and runs them. The
40  * worker_check_timeouts() function also checks the tail of the
41  * "time_order" heap, and ages out requests which have been active
42  * for "too long".
43  *
44  * A request may return one of RLM_MODULE_YIELD,
45  * RLM_MODULE_OK, or RLM_MODULE_HANDLED. If a request is
46  * yielded, it is placed onto the yielded list in the worker
47  * "tracking" data structure.
48  *
49  * @copyright 2016 Alan DeKok (aland@freeradius.org)
50  */

153 /** Handle a control plane message sent to the worker via a channel
154  *
155  * @param[in] ctx the worker
156  * @param[in] data the message
157  * @param[in] data_size size of the data
158  * @param[in] now the current time
159  */
160 static void worker_channel_callback(void *ctx, void const *data, size_t data_size, fr_time_t now)
161 {
162  int i;
163  bool ok, was_sleeping;
164  fr_channel_t *ch;
165  fr_message_set_t *ms;
166  fr_channel_event_t ce;
167  fr_worker_t *worker = ctx;
168 
169  was_sleeping = worker->was_sleeping;
170  worker->was_sleeping = false;
171 
172  /*
173  * We were woken up by a signal to do something. We're
174  * not sleeping.
175  */
176  ce = fr_channel_service_message(now, &ch, data, data_size);
177  DEBUG3("Channel %s",
178  fr_table_str_by_value(channel_signals, ce, "<INVALID>"));
179  switch (ce) {
180  case FR_CHANNEL_ERROR:
181  return;
182 
183  case FR_CHANNEL_EMPTY:
184  return;
185 
186  case FR_CHANNEL_NOOP:
187  return;
188 
189  case FR_CHANNEL_DATA_READY_REQUESTOR:
190  fr_assert(0 == 1);
191  break;
192 
193  case FR_CHANNEL_DATA_READY_RESPONDER:
194  fr_assert(ch != NULL);
195 
196  if (!fr_channel_recv_request(ch)) {
197  worker->was_sleeping = was_sleeping;
198 
199  } else while (fr_channel_recv_request(ch));
200  break;
201 
202  case FR_CHANNEL_OPEN:
203  fr_assert(ch != NULL);
204 
205  ok = false;
206  for (i = 0; i < worker->config.max_channels; i++) {
207  fr_assert(worker->channel[i] != ch);
208 
209  if (worker->channel[i] != NULL) continue;
210 
211  worker->channel[i] = ch;
212  DEBUG3("Received channel %p into array entry %d", ch, i);
213 
214  ms = fr_message_set_create(worker, worker->config.message_set_size,
215  sizeof(fr_channel_data_t),
216  worker->config.ring_buffer_size);
217  fr_assert(ms != NULL);
218  fr_channel_responder_uctx_add(ch, ms);
219 
220  worker->num_channels++;
221  ok = true;
222  break;
223  }
224 
225  fr_cond_assert(ok);
226  break;
227 
228  case FR_CHANNEL_CLOSE:
229  fr_assert(ch != NULL);
230 
231  ok = false;
232 
233  /*
234  * Locate the signalling channel in the list
235  * of channels.
236  */
237  for (i = 0; i < worker->config.max_channels; i++) {
238  if (!worker->channel[i]) continue;
239 
240  if (worker->channel[i] != ch) continue;
241 
242  ms = fr_channel_responder_uctx_get(ch);
243 
244  fr_channel_responder_ack_close(ch);
245  fr_assert(ms != NULL);
246  fr_message_set_gc(ms);
247  talloc_free(ms);
248 
249  worker->channel[i] = NULL;
250  fr_assert(worker->num_channels > 0);
251  worker->num_channels--;
252  ok = true;
253  break;
254  }
255 
256  fr_cond_assert(ok);
257 
258  /*
259  * Our last input channel closed,
260  * time to die.
261  */
262  if (worker->num_channels == 0) worker_exit(worker);
263  break;
264  }
265 }

268 /** Send a NAK to the network thread
269  *
270  * The network thread believes that a worker is running a request until that request has been NAK'd.
271  * We typically NAK requests when they've been hanging around in the worker's backlog too long,
272  * or there was an error executing the request.
273  *
274  * @param[in] worker the worker
275  * @param[in] cd the message to NAK
276  * @param[in] now when the message is NAKd
277  */
278 static void worker_nak(fr_worker_t *worker, fr_channel_data_t *cd, fr_time_t now)
279 {
280  size_t size;
281  fr_channel_data_t *reply;
282  fr_channel_t *ch;
283  fr_message_set_t *ms;
284  fr_listen_t *listen;
285 
286  worker->num_naks++;
287 
288  /*
289  * Cache the outbound channel. We'll need it later.
290  */
291  ch = cd->channel.ch;
292  listen = cd->listen;
293 
294  /*
295  * If the channel has been closed, but we haven't
296  * been informed, that is extremely bad.
297  *
298  * Try to continue working... but we'll likely
299  * leak memory or SEGV soon.
300  */
301  if (!fr_cond_assert_msg(fr_channel_active(ch), "Wanted to send NAK but channel has been closed")) {
302  fr_message_done(&cd->m);
303  return;
304  }
305 
306  ms = fr_channel_responder_uctx_get(ch);
307  fr_assert(ms != NULL);
308 
309  size = listen->app_io->default_reply_size;
310  if (!size) size = listen->app_io->default_message_size;
311 
312  /*
313  * Allocate a default message size.
314  */
315  reply = (fr_channel_data_t *) fr_message_reserve(ms, size);
316  fr_assert(reply != NULL);
317 
318  /*
319  * Encode a NAK
320  */
321  if (listen->app_io->nak) {
322  size = listen->app_io->nak(listen, cd->packet_ctx, cd->m.data,
323  cd->m.data_size, reply->m.data, reply->m.rb_size);
324  } else {
325  size = 1; /* rely on them to figure it the heck out */
326  }
327 
328  (void) fr_message_alloc(ms, &reply->m, size);
329 
330  /*
331  * Fill in the NAK.
332  */
333  reply->m.when = now;
334  reply->reply.cpu_time = worker->tracking.running_total;
335  reply->reply.processing_time = 10; /* @todo - set to something better? */
336  reply->reply.request_time = cd->request.recv_time;
337 
338  reply->listen = cd->listen;
339  reply->packet_ctx = cd->packet_ctx;
340 
341  /*
342  * Mark the original message as done.
343  */
344  fr_message_done(&cd->m);
345 
346  /*
347  * Send the reply, which also polls the request queue.
348  */
349  if (fr_channel_send_reply(ch, reply) < 0) {
350  DEBUG2("Failed sending reply to channel");
351  }
352 
353  worker->stats.out++;
354 }

359 /** Send a response packet to the network side
360  *
361  * @param[in] worker This worker.
362  * @param[in] request we're sending a reply for.
363  * @param[in] size The maximum size of the reply data
364  * @param[in] now The current time
365  */
366 static void worker_send_reply(fr_worker_t *worker, request_t *request, size_t size, fr_time_t now)
367 {
368  fr_channel_data_t *reply;
369  fr_channel_t *ch;
370  fr_message_set_t *ms;
371 
372  REQUEST_VERIFY(request);
373 
374  /*
375  * If we're sending a reply, then it's no longer runnable.
376  */
377  fr_assert(request->runnable_id < 0);
378 
379  /*
380  * If it's a fake request, or we're exiting, don't send a
381  * real reply. Just toss the request.
382  */
383  if (request->async->fake || worker->exiting) {
384  fr_time_tracking_end(&worker->predicted, &request->async->tracking, now);
385  goto finished;
386  }
387 
388  /*
389  * Allocate and send the reply.
390  */
391  ch = request->async->channel;
392  fr_assert(ch != NULL);
393 
394  /*
395  * If the channel has been closed, but we haven't
396  * been informed, that is extremely bad.
397  *
398  * Try to continue working... but we'll likely
399  * leak memory or SEGV soon.
400  */
401  if (!fr_cond_assert_msg(fr_channel_active(ch), "Wanted to send reply but channel has been closed")) {
402  return;
403  }
404 
405  ms = fr_channel_responder_uctx_get(ch);
406  fr_assert(ms != NULL);
407 
408  reply = (fr_channel_data_t *) fr_message_reserve(ms, size);
409  fr_assert(reply != NULL);
410 
411  /*
412  * Encode it, if required.
413  */
414  if (size) {
415  ssize_t slen = 0;
416  fr_listen_t const *listen = request->async->listen;
417 
418  if (listen->app->encode) {
419  slen = listen->app->encode(listen->app_instance, request,
420  reply->m.data, reply->m.rb_size);
421  } else if (listen->app_io->encode) {
422  slen = listen->app_io->encode(listen->app_io_instance, request,
423  reply->m.data, reply->m.rb_size);
424  }
425  if (slen < 0) {
426  RPERROR("Failed encoding request");
427  *reply->m.data = 0;
428  slen = 1;
429  }
430 
431  /*
432  * Shrink the buffer to the actual packet size.
433  *
434  * This will ALWAYS return the same message as we put in.
435  */
436  fr_assert((size_t) slen <= reply->m.rb_size);
437  (void) fr_message_alloc(ms, &reply->m, slen);
438  }
439 
440  /*
441  * The request is done. Track that.
442  */
443  fr_time_tracking_end(&worker->predicted, &request->async->tracking, now);
444  fr_assert(worker->num_active > 0);
445  worker->num_active--;
446 
447  /*
448  * Fill in the rest of the fields in the channel message.
449  *
450  * sequence / ack will be filled in by fr_channel_send_reply()
451  */
452  reply->m.when = now;
453  reply->reply.cpu_time = worker->tracking.running_total;
454  reply->reply.processing_time = request->async->tracking.running_total;
455  reply->reply.request_time = request->async->recv_time;
456 
457  reply->listen = request->async->listen;
458  reply->packet_ctx = request->async->packet_ctx;
459 
460  /*
461  * Update the various timers.
462  */
463  fr_time_elapsed_update(&worker->cpu_time, now, now + reply->reply.processing_time);
464  fr_time_elapsed_update(&worker->wall_clock, reply->reply.request_time, now);
465 
466  RDEBUG("Finished request");
467 
468  /*
469  * Send the reply, which also polls the request queue.
470  */
471  if (fr_channel_send_reply(ch, reply) < 0) {
472  /*
473  * Should only happen if the TO_REQUESTOR
474  * channel is full, or it's not yet active.
475  *
476  * Not much we can do except complain
477  * loudly and cleanup the request.
478  */
479  RPERROR("Failed sending reply to network thread");
480  }
481 
482  worker->stats.out++;
483 
484  /*
485  * @todo Use a talloc pool for the request. Clean it up,
486  * and insert it back into a slab allocator.
487  */
488 finished:
489  if (request->time_order_id >= 0) (void) fr_heap_extract(worker->time_order, request);
490  if (request->runnable_id >= 0) (void) fr_heap_extract(worker->runnable, request);
491 
492  fr_assert(request->time_order_id < 0);
493  fr_assert(request->runnable_id < 0);
494 
495 #ifndef NDEBUG
496  request->async->el = NULL;
497  request->async->process = NULL;
498  request->async->channel = NULL;
499  request->async->packet_ctx = NULL;
500  request->async->listen = NULL;
501 #endif
502 
503  DEBUG3("Freeing request %s", request->name);
504  talloc_free(request);
505 }

508 /** Signal the unlang interpreter that it needs to stop running the request
509  *
510  * Signalling is a synchronous operation. Whatever I/O requests the request
511  * is currently performing are immediately cancelled, and all the frames are
512  * popped off the unlang stack.
513  *
514  * Modules and unlang keywords explicitly register signal handlers to deal
515  * with their yield points being cancelled/interrupted via this function.
516  *
517  * @param[in] worker running the request.
518  * @param[in] request to cancel.
519  * @param[in] now The current time.
520  */
521 static void worker_stop_request(fr_worker_t *worker, request_t *request, fr_time_t now)
522 {
523  if (request->async->tracking.state == FR_TIME_TRACKING_YIELDED) {
524  fr_time_tracking_resume(&request->async->tracking, now);
525  }
526 
527  unlang_interpret_signal(request, FR_SIGNAL_CANCEL);
528 
529  /*
530  * The request is ALWAYS in the time_order list.
531  *
532  * It MAY be in the runnable list if it has yielded
533  * but is ready to be resumed.
534  *
535  * MAY be in the dedup list, but if
536  * app_io->track_duplicates is true, i.e. we have
537  * been requested by an application via its public
538  * interface to track duplicate requests.
539  */
540  if (request->time_order_id >= 0) (void) fr_heap_extract(worker->time_order, request);
541  if (request->runnable_id >= 0) (void) fr_heap_extract(worker->runnable, request);
542  if (request->async->listen && request->async->listen->track_duplicates) rbtree_deletebydata(worker->dedup, request);
543 
544 #ifndef NDEBUG
545  request->async->process = NULL;
546 #endif
547 }

549 /** Enforce max_request_time
550  *
551  * Run periodically, and tries to clean up requests which were received by the network
552  * thread more than max_request_time seconds ago. In the interest of not adding a
553  * timer for every packet, the requests are given a 1 second leeway.
554  *
555  * @param[in] el the worker's event list
556  * @param[in] when the current time
557  * @param[in] uctx the fr_worker_t.
558  */
559 static void worker_max_request_time(UNUSED fr_event_list_t *el, UNUSED fr_time_t when, void *uctx)
560 {
561  fr_time_t now = fr_time();
562  request_t *request;
563  fr_worker_t *worker = talloc_get_type_abort(uctx, fr_worker_t);
564 
565  /*
566  * Look at the oldest requests, and see if they need to
567  * be deleted.
568  */
569  while ((request = fr_heap_peek_tail(worker->time_order)) != NULL) {
570  fr_time_t cleanup;
571 
572  REQUEST_VERIFY(request);
573 
574  cleanup = request->async->recv_time;
575  cleanup += worker->config.max_request_time;
576  if (cleanup > now) break;
577 
578  /*
579  * Waiting too long, delete it.
580  */
581  REDEBUG("Request has reached max_request_time - signalling it to stop");
582  worker_stop_request(worker, request, now);
583 
584  /*
585  * Tell the network side that this request is
586  * done. Also make sure that each time stamp is
587  * unique.
588  */
589  worker_send_reply(worker, request, 1, now);
590  }
591 
592  /*
593  * Reset the max request timer.
594  */
595  worker_max_request_timer(worker);
596 }

598 /** See when we next need to service the time_order heap for "too old" packets
599  *
600  * Inserts a timer into the event list will will trigger when the packet that
601  * was received longest ago, would be older than max_request_time.
602  */
603 static void worker_max_request_timer(fr_worker_t *worker)
604 {
605  fr_time_t cleanup;
606  request_t *request;
607 
608  /*
609  * No more requests, delete the timer.
610  */
611  request = fr_heap_peek_tail(worker->time_order);
612  if (!request) return;
613 
614  cleanup = request->async->recv_time;
615  cleanup += worker->config.max_request_time;
616 
617  DEBUG2("Resetting cleanup timer to +%pV", fr_box_time_delta(worker->config.max_request_time));
618  if (fr_event_timer_at(worker, worker->el, &worker->ev_cleanup,
619  cleanup, worker_max_request_time, worker) < 0) {
620  ERROR("Failed inserting max_request_time timer");
621  }
622 }

628 static char *itoa_internal(TALLOC_CTX *ctx, uint64_t number)
629 {

633

668 /** Start time tracking for a request, and mark it as runnable.
669  *
670  */
671 static void worker_request_time_tracking_start(fr_worker_t *worker, request_t *request, fr_time_t now)
672 {
673  /*
674  * New requests are inserted into the time order heap in
675  * strict time priority. Once they are in the list, they
676  * are only removed when the request is done / free'd.
677  */
678  fr_assert(request->time_order_id < 0);
679  (void) fr_heap_insert(worker->time_order, request);
680 
681  /*
682  * Bootstrap the async state machine with the initial
683  * state of the request.
684  */
685  fr_time_tracking_start(&worker->tracking, &request->async->tracking, now);
686  fr_time_tracking_yield(&request->async->tracking, now);
687  worker->num_active++;
688 
689  fr_assert(request->runnable_id < 0);
690  (void) fr_heap_insert(worker->runnable, request);
691

695 static void worker_request_bootstrap(fr_worker_t *worker, fr_channel_data_t *cd, fr_time_t now)
696 {
697  bool is_dup;
698  int ret = -1;
699  request_t *request;
700  TALLOC_CTX *ctx;
701  fr_listen_t const *listen;
702 
703  if (fr_heap_num_elements(worker->time_order) >= (uint32_t) worker->config.max_requests) goto nak;
704 
705  ctx = request = request_alloc(NULL);
706  if (!request) goto nak;
707 
708  worker_request_init(worker, request, now);
709 
710  request->packet->timestamp = cd->request.recv_time; /* Legacy - Remove once everything looks at request->async */
711 
712  /*
713  * Receive a message to the worker queue, and decode it
714  * to a request.
715  */
716  fr_assert(cd->listen != NULL);
717  request->server_cs = cd->listen->server_cs;
718 
719  /*
720  * Update the transport-specific fields.
721  */
722  request->async->channel = cd->channel.ch;
723 
724  request->async->recv_time = cd->request.recv_time;
725 
726 
727  request->async->listen = cd->listen;
728  request->async->packet_ctx = cd->packet_ctx;
729  listen = request->async->listen;
730 
731  /*
732  * Now that the "request" structure has been initialized, go decode the packet.
733  *
734  * Note that this also sets the "async process" function.
735  */
736  if (listen->app->decode) {
737  ret = listen->app->decode(listen->app_instance, request, cd->m.data, cd->m.data_size);
738  } else if (listen->app_io->decode) {
739  ret = listen->app_io->decode(listen->app_io_instance, request, cd->m.data, cd->m.data_size);
740  }
741 
742  if (ret < 0) {
743  talloc_free(ctx);
744 nak:
745  worker_nak(worker, cd, now);
746  return;
747  }
748 
749  /*
750  * Call the main protocol handler to set the right async
751  * process function.
752  */
753  listen->app->entry_point_set(listen->app_instance, request);
754 
755  if (!request->async->process) {
756  RERROR("Protocol failed to set 'process' function");
757  worker_nak(worker, cd, now);
758  return;
759  }
760 
761  /*
762  * We're done with this message.
763  */
764  is_dup = cd->request.is_dup;
765  fr_message_done(&cd->m);
766 
767  /*
768  * Look for conflicting / duplicate packets, but only if
769  * requested to do so.
770  */
771  if (request->async->listen->track_duplicates) {
772  request_t *old;
773 
774  old = rbtree_finddata(worker->dedup, request);
775  if (!old) {
776  /*
777  * Ignore duplicate packets where we've
778  * already sent the reply.
779  */
780  if (is_dup) {
781  RDEBUG("Got duplicate packet notice after we had sent a reply - ignoring");
782  fr_channel_null_reply(request->async->channel);
783  talloc_free(request);
784  return;
785  }
786  goto insert_new;
787  }
788 
789  fr_assert(old->async->listen == request->async->listen);
790  fr_assert(old->async->channel == request->async->channel);
791 
792  /*
793  * There's a new packet. Do we keep the old one,
794  * or the new one? This decision is made by
795  * checking the recv_time, which is a
796  * nanosecond-resolution timer. If the time is
797  * identical, then the new packet is the same as
798  * the old one.
799  *
800  * If the new packet is a duplicate of the old
801  * one, then we can just discard the new one. We
802  * have to tell the channel that we've "eaten"
803  * this reply, so the sequence number should
804  * increase.
805  *
806  * @todo - fix the channel code to do queue
807  * depth, and not sequence / ack.
808  */
809  if (old->async->recv_time == request->async->recv_time) {
810  RWARN("Discarding duplicate of request (%"PRIu64")", old->number);
811 
812  fr_channel_null_reply(request->async->channel);
813  talloc_free(request);
814 
815  /*
816  * Signal there's a dup, and ignore the
817  * return code. We don't bother replying
818  * here, as an FD event or timer will
819  * wake up the request, and cause it to
820  * continue.
821  *
822  * @todo - the old request is NOT
823  * running, but is yielded. It MAY clean
824  * itself up, or do something...
825  */
826  unlang_interpret_signal(old, FR_SIGNAL_DUP);
827  worker->stats.dup++;
828  return;
829  }
830 
831  /*
832  * Stop the old request, and decrement the number
833  * of active requests.
834  */
835  RWARN("Got conflicting packet for request (%" PRIu64 "), telling old request to stop", old->number);
836 
837  worker_stop_request(worker, old, now);
838  fr_assert(worker->num_active > 0);
839  worker->num_active--;
840  worker->stats.dropped++;
841  talloc_free(old);
842 
843  insert_new:
844  (void) rbtree_insert(worker->dedup, request);
845  }
846 
847  worker_request_time_tracking_start(worker, request, now);
848 }

851 /** Run a request
852  *
853  * Until it either yields, or is done.
854  *
855  * This function is also responsible for sending replies, and
856  * cleaning up the request.
857  *
858  * @param[in] worker the worker
859  * @param[in] start the current time
860  */
861 static void worker_run_request(fr_worker_t *worker, fr_time_t start)
862 {
863  ssize_t size = 0;
864  rlm_rcode_t final;
865  request_t *request;
866  fr_time_t now;
867 
868  WORKER_VERIFY;
869 
870  now = start;
871 
872 redo:
873  request = fr_heap_pop(worker->runnable);
874  if (!request) return;
875 
876  REQUEST_VERIFY(request);
877  fr_assert(request->runnable_id < 0);
878  fr_time_tracking_resume(&request->async->tracking, now);
879 
880  fr_assert(request->parent == NULL);
881  fr_assert(request->async->process != NULL);
882  fr_assert(request->async->fake || (request->async->listen != NULL));
883  fr_assert(request->runnable_id < 0); /* removed from the runnable heap */
884 
885  RDEBUG("Running request");
886 
887  /*
888  * For real requests, if the channel is gone, just stop
889  * the request and free it.
890  */
891  if (!request->async->fake && !fr_channel_active(request->async->channel)) {
892  worker_stop_request(worker, request, now);
893  talloc_free(request);
894  return;
895  }
896 
897  /*
898  * Everything else, run the request.
899  */
900  request->async->process(&final, &(module_ctx_t){ .instance = request->async->process_inst }, request);
901 
902  /*
903  * Figure out what to do next.
904  */
905  switch (final) {
906  case RLM_MODULE_HANDLED:
907  /*
908  * Done: don't send a reply.
909  */
910  break;
911 
912  case RLM_MODULE_FAIL:
913  default:
914  /*
915  * Something went wrong. It's done, but we don't send a reply.
916  */
917  break;
918 
919  case RLM_MODULE_YIELD:
920  now = fr_time();
921  fr_time_tracking_yield(&request->async->tracking, now);
922  goto keep_going;
923 
924  case RLM_MODULE_OK:
925  /*
926  * Don't reply to internally generated request.
927  */
928  if (request->parent || request->async->fake) break;
929 
930  size = request->async->listen->app_io->default_reply_size;
931  if (!size) size = request->async->listen->app_io->default_message_size;
932  break;
933  }
934 
935  RDEBUG("Done request");
936 
937  /*
938  * Only real packets are in the dedup tree. And even
939  * then, only some of the time.
940  */
941  if (!request->async->fake && request->async->listen->track_duplicates) {
942  (void) rbtree_deletebydata(worker->dedup, request);
943  }
944 
945  now = fr_time();
946  worker_send_reply(worker, request, size, now);
947  now = fr_time();
948 
949 keep_going:
950  /*
951  * If this request ran for less than 0.1ms, then go run
952  * another request. This change means that the worker
953  * checks the event loop only 10K times per second, which
954  * should be good enough.
955  */
956  if ((now - start) > (NSEC / 100000)) {
957  return;
958  }
959 
960  goto redo;
961 }

1002 /** Destroy a worker
1003  *
1004  * The input channels are signaled, and local messages are cleaned up.
1005  *
1006  * This should be called to _EXPLICITLY_ destroy a worker, when some fatal
1007  * error has occurred on the worker side, and we need to destroy it.
1008  *
1009  * We signal all pending requests in the backlog to stop, and tell the
1010  * network side that it should not send us any more requests.
1011  *
1012  * @param[in] worker the worker to destroy.
1013  */
1014 void fr_worker_destroy(fr_worker_t *worker)
1015 {

1017  request_t *request;
1018  fr_time_t now = fr_time();
1019 
1020 // WORKER_VERIFY;
1021 
1022  /*
1023  * Destroy all of the active requests. These are ones
1024  * which are still waiting for timers or file descriptor
1025  * events.
1026  */
1027  count = 0;
1028  while ((request = fr_heap_peek(worker->time_order)) != NULL) {
1029  if (count < 10) {
1030  DEBUG("Worker is exiting - telling request %s to stop", request->name);
1031  count++;
1032  }
1033  worker_stop_request(worker, request, now);
1034  talloc_free(request);
1035  }
1036  fr_assert(fr_heap_num_elements(worker->runnable) == 0);

1039  * Signal the channels that we're closing.
1040  *
1041  * The other end owns the channel, and will take care of
1042  * popping messages in the TO_RESPONDER queue, and marking
1043  * them FR_MESSAGE_DONE. It will ignore the messages in
1044  * the TO_REQUESTOR queue, as we own those. They will be
1045  * automatically freed when our talloc context is freed.
1046  */
1047  for (i = 0; i < worker->config.max_channels; i++) {
1048  if (!worker->channel[i]) continue;
1049 
1050  fr_channel_responder_ack_close(worker->channel[i]);
1051  }
1052 
1053  thread_local_worker = NULL;
1054  talloc_free(worker);
1055 }

1161  *
1162  * @param[in] worker the worker data structure to manage
1163  */
1164 void fr_worker(fr_worker_t *worker)
1165 {

1179  if (wait_for_event) {
