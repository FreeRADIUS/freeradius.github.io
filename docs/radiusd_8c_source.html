88 /** Configure talloc debugging features
89  *
90  * @param[in] config The main config.
91  * @return
92  * - 1 on config conflict.
93  * - 0 on success.
94  * - -1 on error.
95  */
96 static int talloc_config_set(main_config_t *config)
97 {
98  if (config->spawn_workers) {
99  if (config->talloc_memory_limit || config->talloc_memory_report) {
100  fr_strerror_printf("talloc_memory_limit and talloc_memory_report "
101  "require single threaded mode (-s | -X)");
102  return 1;
103  }
104  return 0;
105  }
106 
107  if (!config->talloc_memory_limit && !config->talloc_memory_report) {
108  talloc_disable_null_tracking();
109  return 0;
110  }
111 
112  talloc_enable_null_tracking();
113 
114  if (config->talloc_memory_limit) {
115  TALLOC_CTX *null_child = talloc_new(NULL);
116  TALLOC_CTX *null_ctx = talloc_parent(null_child);
117 
118  talloc_free(null_child);
119 
120  if (talloc_set_memlimit(null_ctx, config->talloc_memory_limit) < 0) {
121  fr_strerror_printf("Failed applying memory limit");
122  return -1;
123  }
124  }
125 
126  return 0;
127 }

172 #ifndef NDEBUG
173 /** Encourage the server to exit after a period of time
174  *
175  * @param[in] el The main loop.
176  * @param[in] now Current time. Should be 0, when adding the event.
177  * @param[in] uctx Pointer to a fr_time_delta_t indicating how long
178  * the server should run before exit.
179  */
180 static void fr_exit_after(fr_event_list_t *el, fr_time_t now, void *uctx)
181 {
182  static fr_event_timer_t const *ev;
183 
184  fr_time_delta_t exit_after = *(fr_time_delta_t *)uctx;
185 
186  if (now == 0) {
187  if (fr_event_timer_in(el, el, &ev, exit_after, fr_exit_after, uctx) < 0) {
188  PERROR("Failed inserting exit event");
189  }
190  return;
191  }
192 
193  main_loop_signal_raise(RADIUS_SIGNAL_SELF_TERM);
194 }

236  /*
237  * We probably don't want to free the talloc global_ctx context
238  * directly, so we'll allocate a new context beneath it, and
239  * free that before any leak reports.
240  */
241  {
242  char *env;
243 
244  /*
245  * If a FR_GLOBAL_POOL value is provided and
246  * is of a valid size, we pre-allocate a global
247  * memory pool, and mprotect() it once we're done
248  * parsing the global config.
249  *
250  * This lets us catch stray writes into global
251  * memory.
252  */
253  env = getenv("FR_GLOBAL_POOL");
254  if (env) {
255  if (fr_size_from_str(&pool_size, env) < 0) {
256  fr_perror("Invalid pool size string \"%s\"", env);
257  EXIT_WITH_FAILURE;
258  }
259 
260  /*
261  * Pre-allocate a global memory pool for the static
262  * config to exist in. We mprotect() this later to
263  * catch any stray writes.
264  */
265  global_ctx = talloc_page_aligned_pool(talloc_autofree_context(),
266  &pool_page_start, &pool_page_end, pool_size);
267  do_mprotect = true;
268  } else {
269  global_ctx = talloc_new(talloc_autofree_context());
270  do_mprotect = false;
271  }
272 
273  if (!global_ctx) {
274  fprintf(stderr, "Failed allocating global context\n");
275  EXIT_WITH_FAILURE;
276  }
277  }
278 
279  /*
280  * Allocate the main config structure.
281  * It's allocated so we can hang talloced buffers off it.
282  */
283  config = main_config_alloc(global_ctx);
284  if (!config) {
285  fprintf(stderr, "Failed allocating main config");
286  EXIT_WITH_FAILURE;
287  }
288 
289  /*
290  * Set some default values
291  */
292  program = strrchr(argv[0], FR_DIR_SEP);
293  if (!program) {
294  program = argv[0];
295  } else {
296  program++;
297  }
298  main_config_name_set_default(config, program, false);
299 
300  config->daemonize = true;
301  config->spawn_workers = true;
302 
303 #ifdef OSFC2
304  set_auth_parameters(argc, argv);
305 #endif
306 
307 #ifdef WIN32
308  {
309  WSADATA wsaData;
310  if (WSAStartup(MAKEWORD(2, 0), &wsaData)) {
311  fprintf(stderr, "%s: Unable to initialize socket library.\n", config->name);
312  EXIT_WITH_FAILURE;
313  }
314  }
315 #endif
316 
317  fr_debug_lvl = 0;
318  fr_time_start();
319 
320  /*
321  * Don't put output anywhere until we get told a little
322  * more.
323  */
324  default_log.dst = L_DST_NULL;
325  default_log.fd = -1;
326  default_log.print_level = true;
327 
328  /*
329  * Set the panic action and enable other debugging facilities
330  */
331  if (fr_fault_setup(global_ctx, getenv("PANIC_ACTION"), argv[0]) < 0) {
332  fr_perror("Failed installing fault handlers... continuing");
333  }
334 
335  /* Process the options. */
336  while ((c = getopt(argc, argv, "Cd:D:e:fhi:l:L:Mn:p:PrstTvxX")) != -1) switch (c) {
337  case 'C':
338  check_config = true;
339  config->spawn_workers = false;
340  config->daemonize = false;
341  break;
342 
343  case 'd':
344  main_config_raddb_dir_set(config, optarg);
345  raddb_dir_set = true;
346  break;
347 
348  case 'D':
349  main_config_dict_dir_set(config, optarg);
350  break;
351 
352 #ifndef NDEBUG
353  case 'e':
354  exit_after = (fr_time_delta_t)atoi(optarg) * NSEC;
355  break;
356 #endif
357 
358  case 'f':
359  config->daemonize = false;
360  break;
361 
362  case 'h':
363  usage(config, EXIT_SUCCESS);
364  break;
365 
366  case 'l':
367  if (strcmp(optarg, "stdout") == 0) goto do_stdout;
368 
369  config->log_file = talloc_typed_strdup(global_ctx, optarg);
370  default_log.file = talloc_typed_strdup(global_ctx, optarg);
371  default_log.dst = L_DST_FILES;
372  default_log.fd = open(config->log_file, O_WRONLY | O_APPEND | O_CREAT, 0640);
373  if (default_log.fd < 0) {
374  fprintf(stderr, "%s: Failed to open log file %s: %s\n",
375  config->name, config->log_file, fr_syserror(errno));
376  EXIT_WITH_FAILURE;
377  }
378  break;
379 
380  case 'L':
381  {
382  size_t limit;
383 
384  if (fr_size_from_str(&limit, optarg) < 0) {
385  fr_perror("%s: Invalid memory limit", config->name);
386  EXIT_WITH_FAILURE;
387  }
388 
389  if ((limit > (((size_t)((1024 * 1024) * 1024)) * 16) || (limit < ((1024 * 1024) * 10)))) {
390  fprintf(stderr, "%s: Memory limit must be between 10M-16G\n", config->name);
391  EXIT_WITH_FAILURE;
392  }
393 
394  config->talloc_memory_limit = limit;
395  }
396  break;
397 
398  case 'n':
399  main_config_name_set_default(config, optarg, true);
400  break;
401 
402  case 'M':
403  config->talloc_memory_report = true;
404  break;
405 
406  case 'P': /* Force the PID to be written, even in -f mode */
407  config->write_pid = true;
408  break;
409 
410  case 'r': /* internal radmin-style control interface */
411  config->spawn_workers = false;
412  config->daemonize = false;
413  radmin = true;
414  break;
415 
416  case 's': /* Single process mode */
417  config->spawn_workers = false;
418  config->daemonize = false;
419  break;
420 
421  case 't': /* no child threads */
422  config->spawn_workers = false;
423  break;
424 
425  case 'T': /* enable timestamps */
426  default_log.timestamp = L_TIMESTAMP_ON;
427  break;
428 
429  case 'v':
430  display_version = true;
431  break;
432 
433  case 'X':
434  config->spawn_workers = false;
435  config->daemonize = false;
436  fr_debug_lvl += 2;
437  do_stdout:
438  default_log.dst = L_DST_STDOUT;
439  default_log.fd = STDOUT_FILENO;
440  break;
441 
442  case 'x':
443  fr_debug_lvl++;
444  break;
445 
446  default:
447  usage(config, EXIT_FAILURE);
448  break;
449  }

452  * Allow the configuration directory to be set from an
453  * environment variable. This allows tests to change the
454  * configuration directory without changing the scripts
455  * being executed.
456  */
457  if (!raddb_dir_set) {
458  char const *raddb_dir = getenv("FREERADIUS_CONFIG_DIR");
459 
460  if (raddb_dir) main_config_raddb_dir_set(config, raddb_dir);
461  }
462 
463  config->debug_level = fr_debug_lvl;
464 
465  /*
466  * Mismatch between the binary and the libraries it depends on.
467  */
468  if (fr_check_lib_magic(RADIUSD_MAGIC_NUMBER) < 0) {
469  fr_perror("%s", config->name);
470  EXIT_WITH_FAILURE;
471  }
472 
473  if (rad_check_lib_magic(RADIUSD_MAGIC_NUMBER) < 0) EXIT_WITH_FAILURE;
474 
475 
476 #ifdef HAVE_OPENSSL_CRYPTO_H
477  /*
478  * Mismatch between build time OpenSSL and linked SSL, better to die
479  * here than segfault later.
480  */
481  if (ssl_check_consistency() < 0) EXIT_WITH_FAILURE;
482 
483  /*
484  * Initialising OpenSSL once, here, is safer than having individual modules do it.
485  * Must be called before display_version to ensure relevant engines are loaded.
486  *
487  * fr_openssl_init() must be called before *ANY* OpenSSL functions are used, which is why
488  * it's called so early.
489  */
490  if (fr_openssl_init() < 0) EXIT_WITH_FAILURE;
491 #endif
492 
493  /*
494  * According to the talloc peeps, no two threads may modify any part of
495  * a ctx tree with a common root without synchronisation.
496  *
497  * So we can't run with a null context and threads.
498  */
499  if (talloc_config_set(config) != 0) {
500  fr_perror("%s", config->name);
501  EXIT_WITH_FAILURE;
502  }
503 
504  /*
505  * Better here, so it doesn't matter whether we get passed -xv or -vx.
506  */
507  if (display_version) {
508  if (fr_debug_lvl == 0) fr_debug_lvl = 1;
509  default_log.dst = L_DST_STDOUT;
510  default_log.fd = STDOUT_FILENO;
511 
512  INFO("%s - %s", config->name, radiusd_version);
513  dependency_version_print();
514  EXIT_WITH_SUCCESS;
515  }
516 
517  if (fr_debug_lvl) dependency_version_print();
518 
519  /*
520  * Under linux CAP_SYS_PTRACE is usually only available before setuid/setguid,
521  * so we need to check whether we can attach before calling those functions
522  * (in main_config_init()).
523  */
524  fr_debug_state_store();
525 
526  /*
527  * Write the PID always if we're running as a daemon.
528  */
529  if (config->daemonize) config->write_pid = true;
530 
531  /*
532  * Initialize the DL infrastructure, which is used by the
533  * config file parser. Note that we pass an empty path
534  * here, as we haven't yet read the configuration file.
535  */
536  dl_modules = dl_module_loader_init(NULL);
537  if (!dl_modules) {
538  fr_perror("%s", program);
539  EXIT_WITH_FAILURE;
540  }
541 
542  /*
543  * Initialise the top level dictionary hashes which hold
544  * the protocols.
545  */
546  if (!fr_dict_global_ctx_init(global_ctx, config->dict_dir)) {
547  fr_perror("%s", program);
548  EXIT_WITH_FAILURE;
549  }
550 
551 #ifdef HAVE_OPENSSL_CRYPTO_H
552  if (fr_tls_dict_init() < 0) {
553  fr_perror("%s", program);
554  EXIT_WITH_FAILURE;
555  }
556 #endif
557 
558  /*
559  * Read the configuration files, BEFORE doing anything else.
560  */
561  if (main_config_init(config) < 0) EXIT_WITH_FAILURE;
562 
563  if (modules_init() < 0) {
564  fr_perror("%s", program);
565  EXIT_WITH_FAILURE;
566  }
567 
568  if (virtual_servers_init(config->root_cs) < 0) {
569  fr_perror("%s", program);
570  EXIT_WITH_FAILURE;
571  }
572 
573  /*
574  * Set panic_action from the main config if one wasn't specified in the
575  * environment.
576  */
577  if (config->panic_action && !getenv("PANIC_ACTION") &&
578  (fr_fault_setup(global_ctx, config->panic_action, argv[0]) < 0)) {
579  fr_perror("radiusd - Failed configuring panic action: %s", config->name);
580  EXIT_WITH_FAILURE;
581  }
582

584  * This is very useful in figuring out why the panic_action didn't fire.

587 
588  /*
589  * Call this again now we've loaded the configuration. Yes I know...
590  */
591  if (talloc_config_set(config) < 0) {
592  fr_perror("%s", config->name);
593  EXIT_WITH_FAILURE;
594  }

597  * Check for vulnerabilities in the version of libssl were linked against.
598  */
599 #if defined(HAVE_OPENSSL_CRYPTO_H) && defined(ENABLE_OPENSSL_VERSION_CHECK)
600  if (fr_openssl_version_check(config->allow_vulnerable_openssl) < 0) EXIT_WITH_FAILURE;
601 #endif
602

604  /*
605  * Toggle OpenSSL FIPS mode
606  */
607  if (config->openssl_fips_mode_is_set) {
608  if (FIPS_mode_set(config->openssl_fips_mode ? 1 : 0) == 0) {
609  fr_tls_log_error(NULL, "Failed %s OpenSSL FIPS mode",
610  config->openssl_fips_mode ? "enabling" : "disabling");
611  EXIT_WITH_FAILURE;
612  }
613  }
614 #endif
615 
616  /*
617  * The systemd watchdog enablement must be checked before we
618  * daemonize, but the watchdog notifications can come from any
619  * process.
620  */
621 #ifdef HAVE_SYSTEMD_WATCHDOG
622  if (!check_config) main_loop_set_sd_watchdog_interval();
623 #else
624  /*
625  * If the default systemd unit file is used, but the server wasn't
626  * built with support for systemd, the status returned by systemctl
627  * will stay permanently as "activating".
628  *
629  * We detect this condition and warn about it here, using the
630  * presence of the NOTIFY_SOCKET envrionmental variable to determine
631  * whether we're running under systemd.
632  */
633  if (getenv("NOTIFY_SOCKET")) INFO("Built without support for systemd watchdog, but running under systemd");
634 #endif
635 
636  /*
637  * Don't allow radmin when checking the config.
638  */
639  if (check_config) radmin = false;
640 
641  if (fr_radmin_start(config, radmin) < 0) EXIT_WITH_FAILURE;
642 
643  /*
644  * Disconnect from session
645  */
646  if (config->daemonize) {
647  pid_t pid;
648  int devnull;
649 
650  /*
651  * Really weird things happen if we leave stdin open and call things like
652  * system() later.
653  */
654  devnull = open("/dev/null", O_RDWR);
655  if (devnull < 0) {
656  ERROR("Failed opening /dev/null: %s", fr_syserror(errno));
657  EXIT_WITH_FAILURE;
658  }
659  dup2(devnull, STDIN_FILENO);
660 
661  close(devnull);
662 
663  if (pipe(from_child) != 0) {
664  ERROR("Couldn't open pipe for child status: %s", fr_syserror(errno));
665  EXIT_WITH_FAILURE;
666  }
667 
668  pid = fork();
669  if (pid < 0) {
670  ERROR("Couldn't fork: %s", fr_syserror(errno));
671  EXIT_WITH_FAILURE;
672  }
673 
674  /*
675  * The parent exits, so the child can run in the background.
676  *
677  * As the child can still encounter an error during initialisation
678  * we do a blocking read on a pipe between it and the parent.
679  *
680  * Just before entering the event loop the child will send a success
681  * or failure message to the parent, via the pipe.
682  */
683  if (pid > 0) {
684  uint8_t child_ret;
685  int stat_loc;
686 
687  /* So the pipe is correctly widowed if the child exits */
688  close(from_child[1]);
689 
690  /*
691  * The child writes a 0x01 byte on success, and closes
692  * the pipe on error.
693  */
694  if ((read(from_child[0], &child_ret, 1) < 0)) child_ret = 0;
695 
696  /* For cleanliness... */
697  close(from_child[0]);
698 
699  /* Don't turn children into zombies */
700  if (child_ret == 0) {
701  waitpid(pid, &stat_loc, WNOHANG);
702  EXIT_WITH_FAILURE;
703  }
704 
705 #ifdef HAVE_SYSTEMD
706  /*
707  * Update the systemd MAINPID to be our child,
708  * as the parent is about to exit.
709  */
710  sd_notifyf(0, "MAINPID=%lu", (unsigned long)pid);
711 #endif
712 
713  goto cleanup;
714  }
715 
716  /* so the pipe is correctly widowed if the parent exits?! */
717  close(from_child[0]);
718 #ifdef HAVE_SETSID
719  setsid();
720 #endif
721  }
722 
723  /*
724  * Ensure that we're using the CORRECT pid after forking, NOT the one
725  * we started with.
726  */
727  radius_pid = getpid();
728 
729  /*
730  * Initialise the interpreter, registering operations.
731  */
732  if (unlang_init() < 0) return -1;
733 
734  if (server_init(config->root_cs) < 0) EXIT_WITH_FAILURE;
735 
736  /*
737  * Everything seems to have loaded OK, exit gracefully.
738  */
739  if (check_config) {
740  DEBUG("Configuration appears to be OK");
741  goto cleanup;
742  }
743 
744  /*
745  * Initialise the SNMP stats structures
746  */
747  if (fr_snmp_init() < 0) {
748  PERROR("Failed initialising SNMP");
749  EXIT_WITH_FAILURE;
750  }
751 
752  /*
753  * Initialize the global event loop which handles things like
754  * systemd.
755  *
756  * This has to be done post-fork in case we're using kqueue, where the
757  * queue isn't inherited by the child process.
758  */
759  if (main_loop_init() < 0) {
760  PERROR("Failed initialising main event loop");
761  EXIT_WITH_FAILURE;
762  }
763 
764  /*
765  * Redirect stderr/stdout as appropriate.
766  */
767  if (log_global_init(&default_log, config->daemonize) < 0) EXIT_WITH_FAILURE;
768 
769  /*
770  * Start the network / worker threads.
771  */
772  {
773  fr_event_list_t *el = NULL;
774  fr_schedule_config_t *schedule;
775 
776  schedule = talloc_zero(global_ctx, fr_schedule_config_t);
777  schedule->max_workers = config->max_workers;
778  schedule->max_networks = config->max_networks;
779  schedule->stats_interval = config->stats_interval;
780 
781  schedule->network.max_outstanding = config->max_requests;
782  schedule->worker.max_requests = config->max_requests;
783  schedule->worker.max_request_time = config->max_request_time;
784 
785  /*
786  * Single server mode: use the global event list.
787  * Otherwise, each network thread will create
788  * its own event list.
789  */
790  if (!config->spawn_workers) {
791  el = main_loop_event_list();
792  }
793 
794  sc = fr_schedule_create(NULL, el, &default_log, fr_debug_lvl,
795  thread_instantiate, thread_detach, schedule);
796  if (!sc) {
797  PERROR("Failed starting the scheduler");
798  EXIT_WITH_FAILURE;
799  }
800 
801  /*
802  * Tell the virtual servers to open their sockets.
803  */
804  if (virtual_servers_open(sc) < 0) EXIT_WITH_FAILURE;
805  }
806 
807 #ifndef NDEBUG
808  {
809  size_t size;
810 
811  size = talloc_total_size(config->root_cs);
812 
813  if (talloc_set_memlimit(config->root_cs, size)) {
814  PERROR("Failed setting memory limit for global configuration");
815  } else {
816  DEBUG3("Memory limit for global configuration is set to %zd bytes", size);
817  }
818  }
819 #endif
820 
821  /*
822  * At this point, no one has any business *ever* going
823  * back to root uid.
824  */
825  rad_suid_down_permanent();
826 
827  /*
828  * Dropping down may change the RLIMIT_CORE value, so
829  * reset it back to what to should be here.
830  */
831  fr_reset_dumpable();
832 
833  /*
834  * If we're debugging, then a CTRL-C will cause the server to die
835  * immediately. Use SIGTERM to shut down the server cleanly in
836  * that case.
837  */
838  if (fr_set_signal(SIGINT, sig_fatal) < 0) {
839  set_signal_error:
840  PERROR("Failed installing signal handler");
841  EXIT_WITH_FAILURE;
842  }
843 
844 #ifdef SIGQUIT
845  if (fr_set_signal(SIGQUIT, sig_fatal) < 0) goto set_signal_error;
846 #endif
847 
848  /*
849  * Now that we've set everything up, we can install the signal
850  * handlers. Before this, if we get any signal, we don't know
851  * what to do, so we might as well do the default, and die.
852  */
853 #ifdef SIGPIPE
854  signal(SIGPIPE, SIG_IGN);
855 #endif
856 
857  if (fr_set_signal(SIGHUP, sig_hup) < 0) goto set_signal_error;
858  if (fr_set_signal(SIGTERM, sig_fatal) < 0) goto set_signal_error;
859 
860 #ifdef WITH_STATS
861  radius_stats_init(0);
862 #endif
863 
864  /*
865  * Write the PID after we've forked, so that we write the correct one.
866  */
867  if (config->write_pid) {
868  FILE *fp;
869 
870  fp = fopen(config->pid_file, "w");
871  if (fp != NULL) {
872  /*
873  * @fixme What about following symlinks,
874  * and having it over-write a normal file?
875  */
876  fprintf(fp, "%d\n", (int) radius_pid);
877  fclose(fp);
878  } else {
879  ERROR("Failed creating PID file %s: %s", config->pid_file, fr_syserror(errno));
880  EXIT_WITH_FAILURE;
881  }
882  }
883 
884  trigger_exec(NULL, NULL, "server.start", false, NULL);
885 
886  /*
887  * Inform the parent (who should still be waiting) that the rest of
888  * initialisation went OK, and that it should exit with a 0 status.
889  * If we don't get this far, then we just close the pipe on exit, and the
890  * parent gets a read failure.
891  */
892  if (config->daemonize) {
893  if (write(from_child[1], "\001", 1) < 0) {
894  WARN("Failed informing parent of successful start: %s",
895  fr_syserror(errno));
896  }
897  close(from_child[1]);
898  }
899 
900  /*
901  * Clear the libfreeradius error buffer.
902  */
903  fr_strerror();
904 
905  /*
906  * Prevent anything from modifying the dictionaries
907  * they're now immutable.
908  */
909  fr_dict_global_read_only();
910 
911  /*
912  * Protect global memory - If something attempts
913  * to write to this memory we get a SIGBUS.
914  */
915  if (do_mprotect) {
916  if (mprotect(pool_page_start, (uintptr_t)pool_page_end - (uintptr_t)pool_page_start, PROT_READ) < 0) {
917  PERROR("Protecting global memory failed: %s", fr_syserror(errno));
918  EXIT_WITH_FAILURE;
919  }
920  DEBUG("Global memory protected");
921  }
922 
923  fr_time_sync_event(main_loop_event_list(), fr_time(), NULL);
924 #ifndef NDEBUG
925  if (exit_after > 0) fr_exit_after(main_loop_event_list(), 0, &exit_after);
926 #endif
927  /*
928  * Process requests until HUP or exit.
929  */
930  INFO("Ready to process requests"); /* we were actually ready a while ago, but oh well */
931  while ((status = main_loop_start()) == 0x80) {
932 #ifdef WITH_STATS
933  radius_stats_init(1);
934 #endif
935  main_config_hup(config);
936  }
937 
938  /*
939  * Ignore the TERM signal: we're about to die.
940  */
941  signal(SIGTERM, SIG_IGN);
942 
943  /*
944  * Unprotect global memory
945  */
946  if (do_mprotect) {
947  if (mprotect(pool_page_start, (uintptr_t)pool_page_end - (uintptr_t)pool_page_start,
948  PROT_READ | PROT_WRITE) < 0) {
949  PERROR("Unprotecting global memory failed: %s", fr_syserror(errno));
950  EXIT_WITH_FAILURE;
951  }
952  DEBUG("Global memory unprotected");
953  }
954 
955  if (status < 0) {
956  PERROR("Exiting due to internal error");
957  ret = EXIT_FAILURE;
958  } else {
959  INFO("Exiting normally");
960  ret = EXIT_SUCCESS;
961  }
962 
963  fr_radmin_stop();
964 
965  /*
966  * Fire signal and stop triggers after ignoring SIGTERM, so handlers are
967  * not killed with the rest of the process group, below.
968  */
969  if (status == 2) trigger_exec(NULL, NULL, "server.signal.term", true, NULL);
970  trigger_exec(NULL, NULL, "server.stop", false, NULL);
971 
972  /*
973  * Stop the scheduler, this signals the network and worker threads
974  * to exit gracefully. fr_schedule_destroy only returns once all
975  * threads have been joined.
976  */
977  (void) fr_schedule_destroy(&sc);
978 
979  /*
980  * We're exiting, so we can delete the PID file.
981  * (If it doesn't exist, we can ignore the error returned by unlink)
982  */
983  if (config->daemonize) unlink(config->pid_file);
984 
985  /*
986  * Free memory in an explicit and consistent order
987  *
988  * We could let everything be freed by the global_ctx
989  * context, but in some cases there are odd interactions
990  * with destructors that may cause double frees and
991  * SEGVs.
992  */
993  if (!config->spawn_workers) {
994  fr_event_list_t *el;
995 
996  el = main_loop_event_list();
997  fr_event_loop_exit(el, 1);
998  }
999 
1000  main_loop_free(); /* Free the requests */
1001 
1002  /*
1003  * Send a TERM signal to all associated processes
1004  * (including us, which gets ignored.)
1005  *
1006  * This _shouldn't_ be needed, but may help with
1007  * processes created by the exec code or triggers.
1008  */
1009  if (config->spawn_workers) {
1010  INFO("All threads have exited, sending SIGTERM to remaining children");
1011  kill(-radius_pid, SIGTERM);
1012  }
1013 cleanup:
1014  /*
1015  * This may not have been done earlier if we're
1016  * exiting due to a startup error.
1017  */
1018  (void) fr_schedule_destroy(&sc);
1019 
1020  /*
1021  * Frees request specific logging resources which is OK
1022  * because all the requests will have been stopped.
1023  */
1024  log_global_free();
1025 
1026  fr_snmp_free();
1027 
1028  server_free();
1029 
1030  /*
1031  * Free any resources used by the unlang interpreter.
1032  */
1033  unlang_free();
1034 
1035 #ifdef HAVE_OPENSSL_CRYPTO_H
1036  fr_openssl_free(); /* Cleanup any memory alloced by OpenSSL and placed into globals */
1037 #endif
1038 
1039  if (config) talloc_memory_report = config->talloc_memory_report; /* Grab this before we free the config */
1040 
1041  /*
1042  * And now nothing should be left anywhere except the
1043  * parsed configuration items.
1044  */
1045  main_config_free(&config);
1046 
1047  /*
1048  * Free the modules that we loaded.
1049  */
1050  if (dl_modules) talloc_free(dl_modules);
1051 
1052  /*
1053  * Cleanup everything else
1054  */
1055  talloc_free(global_ctx);
1056 
1057  /*
1058  * Anything not cleaned up by the above is allocated in
1059  * the NULL top level context, and is likely leaked memory.
1060  */
1061  if (talloc_memory_report) fr_log_talloc_report(NULL);
1062 
1063  /*
1064  * If we're running under LSAN, try and SUID back up so
1065  * we don't inteferere with the onexit() handler.
1066  */
1067  if (!rad_suid_is_down_permanent() && (fr_get_lsan_state() == 1)) rad_suid_up();
1068  fr_strerror(); /* clear error buffer */
1069 
1070  return ret;
1071 }
