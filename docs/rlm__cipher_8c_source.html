319  *
320  * @param[in] ctx UNUSED. Although the EVP_PKEY struct will be allocated
321  * with talloc, we need to call the specialised free
322  * function anyway.
323  * @param[out] out Where to write the EVP_PKEY * representing the

332 static int cipher_rsa_private_key_file_load(TALLOC_CTX *ctx, void *out, UNUSED void *parent,
333  CONF_ITEM *ci, UNUSED CONF_PARSER const *rule)
334 {
335  FILE *fp;
336  char const *filename;
337  cipher_rsa_t *rsa_inst = talloc_get_type_abort(ctx, cipher_rsa_t); /* Yeah this is a bit hacky */
338  EVP_PKEY *pkey;
339  void *pass;
340  int pkey_type;
341 
342  filename = cf_pair_value(cf_item_to_pair(ci));
343 
344  fp = fopen(filename, "r");
345  if (!fp) {
346  cf_log_err(ci, "Failed opening private_key file \"%s\": %s", filename, fr_syserror(errno));
347  return -1;
348  }
349 
350  memcpy(&pass, &rsa_inst->private_key_password, sizeof(pass));
351 
352  pkey = PEM_read_PrivateKey(fp, (EVP_PKEY **)out, _get_private_key_password, pass);
353  fclose(fp);
354 
355  if (!pkey) {
356  tls_strerror_printf(NULL);
357  cf_log_perr(ci, "Error loading private certificate file \"%s\"", filename);
358 
359  return -1;
360  }
361 
362  pkey_type = EVP_PKEY_type(EVP_PKEY_id(pkey));
363  if (pkey_type != EVP_PKEY_RSA) {
364  cf_log_err(ci, "Expected certificate to contain %s private key, got %s private key",
365  fr_table_str_by_value(pkey_types, EVP_PKEY_RSA, OBJ_nid2sn(pkey_type)),
366  fr_table_str_by_value(pkey_types, pkey_type, OBJ_nid2sn(pkey_type)));
367 
368  EVP_PKEY_free(pkey);
369  return -1;
370  }
371 
372  talloc_set_type(pkey, EVP_PKEY);
373  (void)talloc_steal(ctx, pkey); /* Bind lifetime to config */
374  talloc_set_destructor(pkey, _evp_pkey_free); /* Free pkey correctly on chunk free */
375 
376  return 0;
377 }

580  fr_assert((size_t)digest_len == talloc_array_length(xt->digest_buff));

583  * Then sign the digest
584  */

587  return XLAT_ACTION_FAIL;
588  }

592  if (EVP_PKEY_sign(xt->evp_sign_ctx, sig, &sig_len, xt->digest_buff, (size_t)digest_len) <= 0) {
593  fr_tls_log_error(request, "Failed signing message digest");
594  talloc_free(vb);
595  return XLAT_ACTION_FAIL;
596  }
597  MEM(fr_value_box_mem_realloc(vb, NULL, vb, sig_len) == 0);
598  fr_cursor_append(out, vb);
599

666 /** Verify input data
667  *
668  * Arguments are @verbatim(<signature>, <plaintext>...)@endverbatim
669  *
670 @verbatim
671 %{<inst>_verify:<signature> <plaintext>...}
672 @endverbatim
673  *
674  * If multiple arguments are provided (after @verbatim<signature>@endverbatim)
675  * they will be concatenated.
676  *
677  * @ingroup xlat_functions
678  */
679 static xlat_action_t cipher_rsa_verify_xlat(TALLOC_CTX *ctx, fr_cursor_t *out,
680  request_t *request, void const *xlat_inst, void *xlat_thread_inst,
681  fr_value_box_t **in)
682 {
683  rlm_cipher_t const *inst = talloc_get_type_abort_const(*((void const * const *)xlat_inst),
684  rlm_cipher_t);
685  rlm_cipher_rsa_thread_inst_t *xt = talloc_get_type_abort(*((void **)xlat_thread_inst),
686  rlm_cipher_rsa_thread_inst_t);
687 
688  uint8_t const *sig;
689  size_t sig_len;
690 
691  char const *msg;
692  size_t msg_len;
693 
694  unsigned int digest_len = 0;
695 
696  fr_value_box_t *vb;
697 
698  if (!*in) {
699  REDEBUG("verification requires two or more arguments (<signature>, <message>...)");
700  return XLAT_ACTION_FAIL;
701  }
702 
703  /*
704  * Check we have at least two boxed values
705  */
706  if (!(*in)->next) {
707  REDEBUG("Missing message data arg or message data was (null)");
708  return XLAT_ACTION_FAIL;
709  }
710 
711  /*
712  * Don't auto-cast to octets if the signature
713  * isn't already in that form.
714  * It could be hexits or base64 or some other encoding.
715  */
716  if ((*in)->type != FR_TYPE_OCTETS) {
717  REDEBUG("Signature argument wrong type, expected %s, got %s. "
718  "Use %%{base64_decode:<text>} or %%{hex_decode:<text>} if signature is armoured",
719  fr_table_str_by_value(fr_value_box_type_table, FR_TYPE_OCTETS, "?Unknown?"),
720  fr_table_str_by_value(fr_value_box_type_table, (*in)->type, "?Unknown?"));
721  return XLAT_ACTION_FAIL;
722  }
723  sig = (*in)->vb_octets;
724  sig_len = (*in)->vb_length;
725 
726  /*
727  * Concat (...) args to get message data
728  */
729  if (fr_value_box_list_concat(ctx, (*in)->next, &((*in)->next), FR_TYPE_STRING, true) < 0) {
730  REDEBUG("Failed concatenating arguments to form plaintext");
731  return XLAT_ACTION_FAIL;
732  }
733  msg = (*in)->next->vb_strvalue;
734  msg_len = (*in)->next->vb_length;
735 
736  /*
737  * The argument separator also gets rolled into
738  * the concatenate buffer... We should probably
739  * figure out a cleaner way of doing this.
740  */
741  if (*msg != ' ') {
742  REDEBUG("Expected whitespace argument separator");
743  return XLAT_ACTION_FAIL;
744  }
745  msg++;
746  msg_len--;
747 
748  if (msg_len == 0) {
749  REDEBUG("Zero length message data");
750  return XLAT_ACTION_FAIL;
751  }
752 
753  /*
754  * First produce a digest of the message
755  */
756  if (unlikely(EVP_DigestInit_ex(xt->evp_md_ctx, inst->rsa->sig_digest, NULL) <= 0)) {
757  fr_tls_log_error(request, "Failed initialising message digest");
758  return XLAT_ACTION_FAIL;
759  }
760 
761  if (EVP_DigestUpdate(xt->evp_md_ctx, msg, msg_len) <= 0) {
762  fr_tls_log_error(request, "Failed ingesting message");
763  return XLAT_ACTION_FAIL;
764  }
765 
766  if (EVP_DigestFinal_ex(xt->evp_md_ctx, xt->digest_buff, &digest_len) <= 0) {
767  fr_tls_log_error(request, "Failed finalising message digest");
768  return XLAT_ACTION_FAIL;
769  }
770  fr_assert((size_t)digest_len == talloc_array_length(xt->digest_buff));
771 
772  /*
773  * Now check the signature matches what we expected
774  */
775  switch (EVP_PKEY_verify(xt->evp_verify_ctx, sig, sig_len, xt->digest_buff, (size_t)digest_len)) {
776  case 1: /* success (signature valid) */
777  MEM(vb = fr_value_box_alloc(ctx, FR_TYPE_BOOL, NULL, false));
778  vb->vb_bool = true;
779  fr_cursor_append(out, vb);
780  break;
781 
782  case 0: /* failure (signature not valid) */
783  MEM(vb = fr_value_box_alloc(ctx, FR_TYPE_BOOL, NULL, false));
784  vb->vb_bool = false;
785  fr_cursor_append(out, vb);
786  break;
787 
788  default:
789  fr_tls_log_error(request, "Failed validating signature");
790  return XLAT_ACTION_FAIL;
791  }
792 
793  return XLAT_ACTION_DONE;
794 }

902 /** Pre-initialises the EVP_PKEY_CTX necessary for performing RSA encryption/decryption/sign/verify
903  *
904  * If reference counting is used for EVP_PKEY structs, should also prevent any mutex contention
905  * associated with incrementing/decrementing those references.
906  *
907  * xlat functions MUST NOT interleave PKEY operations with yields
908  *
909  * @return 0.
910  */
911 static int cipher_rsa_thread_instantiate(UNUSED CONF_SECTION const *conf, void *instance,
912  UNUSED fr_event_list_t *el, void *thread)
913 {
914  rlm_cipher_t const *inst = talloc_get_type_abort(instance, rlm_cipher_t);
915  rlm_cipher_rsa_thread_inst_t *ti = thread;
916 
917  /*
918  * Pre-allocate different contexts for the different operations
919  * The OpenSSL docs say this is fine, and it reduces the potential
920  * for SEGVs and other random errors due to trying to change the
921  * configuration of a context multiple times.
922  */
923  if (inst->rsa->certificate_file) {
924  /*
925  * Alloc encrypt
926  */
927  ti->evp_encrypt_ctx = EVP_PKEY_CTX_new(inst->rsa->certificate_file, NULL);
928  if (!ti->evp_encrypt_ctx) {
929  tls_strerror_printf(NULL);
930  PERROR("%s: Failed allocating encrypt EVP_PKEY_CTX", __FUNCTION__);
931  return -1;
932  }
933  talloc_set_type(ti->evp_encrypt_ctx, EVP_PKEY_CTX);
934  ti->evp_encrypt_ctx = talloc_steal(ti, ti->evp_encrypt_ctx); /* Bind lifetime to instance */
935  talloc_set_destructor(ti->evp_encrypt_ctx, _evp_pkey_ctx_free); /* Free ctx correctly on chunk free */
936 
937  /*
938  * Configure encrypt
939  */
940  if (unlikely(EVP_PKEY_encrypt_init(ti->evp_encrypt_ctx) <= 0)) {
941  tls_strerror_printf(NULL);
942  PERROR("%s: Failed initialising encrypt EVP_PKEY_CTX", __FUNCTION__);
943  return XLAT_ACTION_FAIL;
944  }
945  if (unlikely(cipher_rsa_padding_params_set(ti->evp_encrypt_ctx, inst->rsa) < 0)) {
946  ERROR("%s: Failed setting padding for encrypt EVP_PKEY_CTX", __FUNCTION__);
947  return -1;
948  }
949 
950  /*
951  * Alloc verify
952  */
953  ti->evp_verify_ctx = EVP_PKEY_CTX_new(inst->rsa->certificate_file, NULL);
954  if (!ti->evp_verify_ctx) {
955  tls_strerror_printf(NULL);
956  PERROR("%s: Failed allocating verify EVP_PKEY_CTX", __FUNCTION__);
957  return -1;
958  }
959  talloc_set_type(ti->evp_verify_ctx, EVP_PKEY_CTX);
960  ti->evp_verify_ctx = talloc_steal(ti, ti->evp_verify_ctx); /* Bind lifetime to instance */
961  talloc_set_destructor(ti->evp_verify_ctx, _evp_pkey_ctx_free); /* Free ctx correctly on chunk free */
962 
963  /*
964  * Configure verify
965  */
966  if (unlikely(EVP_PKEY_verify_init(ti->evp_verify_ctx) <= 0)) {
967  tls_strerror_printf(NULL);
968  PERROR("%s: Failed initialising verify EVP_PKEY_CTX", __FUNCTION__);
969  return XLAT_ACTION_FAIL;
970  }
971 
972  /*
973  * OAEP not valid for signing or verification
974  */
975  if (inst->rsa->padding != RSA_PKCS1_OAEP_PADDING) {
976  if (unlikely(cipher_rsa_padding_params_set(ti->evp_verify_ctx, inst->rsa) < 0)) {
977  ERROR("%s: Failed setting padding for verify EVP_PKEY_CTX", __FUNCTION__);
978  return -1;
979  }
980  }
981 
982  if (unlikely(EVP_PKEY_CTX_set_signature_md(ti->evp_verify_ctx, inst->rsa->sig_digest)) <= 0) {
983  tls_strerror_printf(NULL);
984  PERROR("%s: Failed setting signature digest type", __FUNCTION__);
985  return XLAT_ACTION_FAIL;
986  }
987  }
988 
989  if (inst->rsa->private_key_file) {
990  /*
991  * Alloc decrypt
992  */
993  ti->evp_decrypt_ctx = EVP_PKEY_CTX_new(inst->rsa->private_key_file, NULL);
994  if (!ti->evp_decrypt_ctx) {
995  tls_strerror_printf(NULL);
996  PERROR("%s: Failed allocating decrypt EVP_PKEY_CTX", __FUNCTION__);
997  return -1;
998  }
999  talloc_set_type(ti->evp_decrypt_ctx, EVP_PKEY_CTX);
1000  ti->evp_decrypt_ctx = talloc_steal(ti, ti->evp_decrypt_ctx); /* Bind lifetime to instance */
1001  talloc_set_destructor(ti->evp_decrypt_ctx, _evp_pkey_ctx_free); /* Free ctx correctly on chunk free */

1004  * Configure decrypt
1005  */

1007  tls_strerror_printf(NULL);
1008  PERROR("%s: Failed initialising decrypt EVP_PKEY_CTX", __FUNCTION__);
1009  return XLAT_ACTION_FAIL;
1010  }
1011  if (unlikely(cipher_rsa_padding_params_set(ti->evp_decrypt_ctx, inst->rsa) < 0)) {
1012  ERROR("%s: Failed setting padding for decrypt EVP_PKEY_CTX", __FUNCTION__);

1014  }
1015 
1016  /*
1017  * Alloc sign
1018  */
1019  ti->evp_sign_ctx = EVP_PKEY_CTX_new(inst->rsa->private_key_file, NULL);

1030  * Configure sign
1031  */

1033  tls_strerror_printf(NULL);
1034  PERROR("%s: Failed initialising sign EVP_PKEY_CTX", __FUNCTION__);
1035  return XLAT_ACTION_FAIL;
1036  }
1037

1039  * OAEP not valid for signing or verification

1045  }
1046  }
1047

1049  tls_strerror_printf(NULL);
1050  PERROR("%s: Failed setting signature digest type", __FUNCTION__);
1051  return XLAT_ACTION_FAIL;
1052  }
1053 
1054  /*
1055  * Alloc digest ctx for signing and verification
1056  */
1057  ti->evp_md_ctx = EVP_MD_CTX_create();

1059  tls_strerror_printf(NULL);
1060  PERROR("%s: Failed allocating EVP_MD_CTX", __FUNCTION__);
1061  return -1;
1062  }
1063  talloc_set_type(ti->evp_md_ctx, EVP_MD_CTX);
1064  ti->evp_md_ctx = talloc_steal(ti, ti->evp_md_ctx); /* Bind lifetime to instance */
1065  talloc_set_destructor(ti->evp_md_ctx, _evp_md_ctx_free); /* Free ctx correctly on chunk free */
1066  MEM(ti->digest_buff = talloc_array(ti, uint8_t, EVP_MD_size(inst->rsa->sig_digest)));
1067  }
1068 
1069  return 0;
1070 }
