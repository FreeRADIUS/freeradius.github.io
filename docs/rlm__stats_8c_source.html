98 static void coalesce(uint64_t final_stats[FR_RADIUS_MAX_PACKET_CODE], rlm_stats_thread_t *t,
99  size_t tree_offset, rlm_stats_data_t *mydata)
100 {
101  rlm_stats_data_t *stats;
102  rlm_stats_thread_t *other;
103  rbtree_t **tree;
104  uint64_t local_stats[FR_RADIUS_MAX_PACKET_CODE];
105 
106  tree = (rbtree_t **) (((uint8_t *) t) + tree_offset);
107 
108  /*
109  * Bootstrap with my statistics, where we don't need a
110  * lock.
111  */
112  stats = rbtree_finddata(*tree, mydata);
113  if (!stats) {
114  memset(final_stats, 0, sizeof(uint64_t) * FR_RADIUS_MAX_PACKET_CODE);
115  } else {
116  memcpy(final_stats, stats->stats, sizeof(stats->stats));
117  }
118 
119  /*
120  * Loop over all of the other thread instances, locking
121  * them, and adding their statistics in.
122  */
123  for (other = fr_dlist_head(&t->inst->list);
124  other != NULL;
125  other = fr_dlist_next(&t->inst->list, other)) {
126  int i;
127 
128  if (other == t) continue;
129 
130  tree = (rbtree_t **) (((uint8_t *) other) + tree_offset);
131  stats = rbtree_finddata(*tree, mydata);
132  if (!stats) {
133  continue;
134  }
135  memcpy(&local_stats, stats->stats, sizeof(stats->stats));
136 
137  for (i = 0; i < FR_RADIUS_MAX_PACKET_CODE; i++) {
138  final_stats[i] += local_stats[i];
139  }
140  }
141 }

144 /*
145  * Do the statistics
146  */
147 static unlang_action_t CC_HINT(nonnull) mod_stats(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
148 {
149  rlm_stats_t *inst = talloc_get_type_abort(mctx->instance, rlm_stats_t);
150  rlm_stats_thread_t *t = talloc_get_type_abort(mctx->thread, rlm_stats_thread_t);
151  int i;
152  uint32_t stats_type;
153 
154 
155  fr_pair_t *vp;
156  rlm_stats_data_t mydata, *stats;
157  fr_cursor_t cursor;
158  char buffer[64];
159  uint64_t local_stats[NUM_ELEMENTS(inst->stats)];
160 
161  /*
162  * Increment counters only in "send foo" sections.
163  *
164  * i.e. only when we have a reply to send.
165  */
166  if (request->request_state == REQUEST_SEND) {
167  int src_code, dst_code;
168 
169  src_code = request->packet->code;
170  if (src_code >= FR_RADIUS_MAX_PACKET_CODE) src_code = 0;
171 
172  dst_code = request->reply->code;
173  if (dst_code >= FR_RADIUS_MAX_PACKET_CODE) dst_code = 0;
174 
175  t->stats[src_code]++;
176  t->stats[dst_code]++;
177 
178  /*
179  * Update source statistics
180  */
181  mydata.ipaddr = request->packet->socket.inet.src_ipaddr;
182  stats = rbtree_finddata(t->src, &mydata);
183  if (!stats) {
184  MEM(stats = talloc_zero(t, rlm_stats_data_t));
185 
186  stats->ipaddr = request->packet->socket.inet.src_ipaddr;
187  stats->created = request->async->recv_time;
188 
189  (void) rbtree_insert(t->src, stats);
190  }
191 
192  stats->last_packet = request->async->recv_time;
193  stats->stats[src_code]++;
194  stats->stats[dst_code]++;
195 
196  /*
197  * Update destination statistics
198  */
199  mydata.ipaddr = request->packet->socket.inet.dst_ipaddr;
200  stats = rbtree_finddata(t->dst, &mydata);
201  if (!stats) {
202  MEM(stats = talloc_zero(t, rlm_stats_data_t));
203 
204  stats->ipaddr = request->packet->socket.inet.dst_ipaddr;
205  stats->created = request->async->recv_time;
206 
207  (void) rbtree_insert(t->dst, stats);
208  }
209 
210  stats->last_packet = request->async->recv_time;
211  stats->stats[src_code]++;
212  stats->stats[dst_code]++;
213 
214  /*
215  * @todo - periodically clean up old entries.
216  */
217 
218  if ((t->last_global_update + NSEC) > request->async->recv_time) {
219  RETURN_MODULE_UPDATED;
220  }
221 
222  t->last_global_update = request->async->recv_time;
223 
224  pthread_mutex_lock(&inst->mutex);
225  for (i = 0; i < FR_RADIUS_MAX_PACKET_CODE; i++) {
226  inst->stats[i] += t->stats[i];
227  t->stats[i] = 0;
228  }
229  pthread_mutex_unlock(&inst->mutex);
230 
231  RETURN_MODULE_UPDATED;
232  }
233 
234  /*
235  * Ignore "authenticate" and anything other than Status-Server
236  */
237  if ((request->request_state != REQUEST_RECV) ||
238  (request->packet->code != FR_CODE_STATUS_SERVER)) {
239  RETURN_MODULE_NOOP;
240  }
241 
242  vp = fr_pair_find_by_da(&request->request_pairs, attr_freeradius_stats4_type);
243  if (!vp) {
244  stats_type = FR_FREERADIUS_STATS4_TYPE_VALUE_GLOBAL;
245  } else {
246  stats_type = vp->vp_uint32;
247  }
248 
249  /*
250  * Create attributes based on the statistics.
251  */
252  fr_cursor_init(&cursor, &request->reply_pairs);
253 
254  MEM(pair_update_reply(&vp, attr_freeradius_stats4_type) >= 0);
255  vp->vp_uint32 = stats_type;
256 
257  switch (stats_type) {
258  case FR_FREERADIUS_STATS4_TYPE_VALUE_GLOBAL: /* global */
259  /*
260  * Merge our stats with the global stats, and then copy
261  * the global stats to a thread-local variable.
262  *
263  * The copy helps minimize mutex contention.
264  */
265  pthread_mutex_lock(&inst->mutex);
266  for (i = 0; i < FR_RADIUS_MAX_PACKET_CODE; i++) {
267  inst->stats[i] += t->stats[i];
268  t->stats[i] = 0;
269  }
270  memcpy(&local_stats, inst->stats, sizeof(inst->stats));
271  pthread_mutex_unlock(&inst->mutex);
272  vp = NULL;
273  break;
274 
275  case FR_FREERADIUS_STATS4_TYPE_VALUE_CLIENT: /* src */
276  vp = fr_pair_find_by_da(&request->request_pairs, attr_freeradius_stats4_ipv4_address);
277  if (!vp) vp = fr_pair_find_by_da(&request->request_pairs, attr_freeradius_stats4_ipv6_address);
278  if (!vp) RETURN_MODULE_NOOP;
279 
280  mydata.ipaddr = vp->vp_ip;
281  coalesce(local_stats, t, offsetof(rlm_stats_thread_t, src), &mydata);
282  break;
283 
284  case FR_FREERADIUS_STATS4_TYPE_VALUE_LISTENER: /* dst */
285  vp = fr_pair_find_by_da(&request->request_pairs, attr_freeradius_stats4_ipv4_address);
286  if (!vp) vp = fr_pair_find_by_da(&request->request_pairs, attr_freeradius_stats4_ipv6_address);
287  if (!vp) RETURN_MODULE_NOOP;
288 
289  mydata.ipaddr = vp->vp_ip;
290  coalesce(local_stats, t, offsetof(rlm_stats_thread_t, dst), &mydata);
291  break;
292 
293  default:
294  REDEBUG("Invalid value '%d' for FreeRADIUS-Stats4-type", stats_type);
295  RETURN_MODULE_FAIL;
296  }
297 
298  if (vp ) {
299  vp = fr_pair_copy(request->reply, vp);
300  if (vp) {
301  fr_cursor_append(&cursor, vp);
302  (void) fr_cursor_tail(&cursor);
303  }
304  }
305 
306  strcpy(buffer, "FreeRADIUS-Stats4-");
307 
308  for (i = 0; i < FR_RADIUS_MAX_PACKET_CODE; i++) {
309  fr_dict_attr_t const *da;
310 
311  if (!local_stats[i]) continue;
312 
313  strlcpy(buffer + 18, fr_packet_codes[i], sizeof(buffer) - 18);
314  da = fr_dict_attr_by_name(dict_radius, buffer);
315  if (!da) continue;
316 
317  MEM(vp = fr_pair_afrom_da(request->reply, da));
318  vp->vp_uint64 = local_stats[i];
319 
320  fr_cursor_append(&cursor, vp);
321  (void) fr_cursor_tail(&cursor);
322  }
323 
324  RETURN_MODULE_OK;
325 }

353  pthread_mutex_unlock(&inst->mutex);
