90 /** Implements the default iterator to encode pairs belonging to a specific dictionary that are not internal
91  *
92  * @param[in,out] prev The fr_pair_t before curr. Will be updated to point to the
93  * pair before the one returned, or the last pair in the list
94  * if no matching pairs found.
95  * @param[in] to_eval The fr_pair_t after cursor->current. Will be checked to
96  * see if it matches the specified fr_dict_t.
97  * @param[in] uctx The fr_dict_t to search for.
98  * @return
99  * - Next matching fr_pair_t.
100  * - NULL if not more matching fr_pair_ts could be found.
101  */
102 void *fr_proto_next_encodable(void **prev, void *to_eval, void *uctx)
103 {
104  fr_pair_t *c, *p;
105  fr_dict_t *dict = talloc_get_type_abort(uctx, fr_dict_t);
106 
107  if (!to_eval) return NULL;
108 
109  for (p = *prev, c = to_eval; c; p = c, c = c->next) {
110  VP_VERIFY(c);
111  if ((c->da->dict == dict) && (!c->da->flags.internal)) break;
112  }
113 
114  *prev = p;
115 
116  return c;
117 }

119 /** Build a complete DA stack from the da back to the root
120  *
121  * @param[out] stack to populate.
122  * @param[in] da to build the stack for.
123  */
124 void fr_proto_da_stack_build(fr_da_stack_t *stack, fr_dict_attr_t const *da)
125 {
126  fr_dict_attr_t const **cached;
127 
128  if (!da) return;
129 
130  /*
131  * See if we have a cached da stack available
132  */
133  cached = fr_dict_attr_da_stack(da);
134  if (cached) {
135  /*
136  * da->da_stack[0] is dict->root
137  */
138  memcpy(&stack->da[0], &cached[1], sizeof(stack->da[0]) * da->depth);
139 
140  } else {
141  fr_dict_attr_t const *da_p, **da_o;
142 
143  /*
144  * Unknown attributes don't have a da->da_stack.
145  */
146  da_p = da;
147  da_o = stack->da + (da->depth - 1);
148 
149  while (da_o >= stack->da) {
150  *da_o-- = da_p;
151  da_p = da_p->parent;
152  }
153  }
154 
155  stack->depth = da->depth;
156  stack->da[stack->depth] = NULL;
157 }

159 /** Complete the DA stack for a child attribute
160  *
161  * @param[out] stack to populate.
162  * @param[in] parent to populate from.
163  * @param[in] da to populate to.
164  */
165 void fr_proto_da_stack_build_partial(fr_da_stack_t *stack, fr_dict_attr_t const *parent, fr_dict_attr_t const *da)
166 {
167  fr_dict_attr_t const *da_p, **da_q, **da_o;
168 
169 #ifndef NDEBUG
170  if (!fr_cond_assert(fr_dict_attr_common_parent(parent, da, true) == parent)) {
171  fr_strerror_printf("Expected \"%s\" to be a descendent of \"%s\" but it isn't",
172  da->name, parent->name);
173  return;
174  }
175 #endif
176 
177  da_p = da;
178  da_q = stack->da + (parent->depth - 1);
179  da_o = stack->da + (da->depth - 1);
180 
181  while (da_o >= da_q) {
182  *da_o-- = da_p;
183  da_p = da_p->parent;
184  }
185 
186  stack->depth = da->depth;
187  stack->da[stack->depth] = NULL;
188 }
