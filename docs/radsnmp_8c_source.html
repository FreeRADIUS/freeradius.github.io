180 /** Builds attribute representing OID string and adds 'index' attributes where required
181  *
182  * Will convert an OID string in the format @verbatim .1.2.3.4.5.0 @endverbatim
183  * into a pair with a #fr_dict_attr_t of the dictionary attribute matching the OID
184  * string, as evaluated from the specified parent.
185  *
186  * If an OID component does not match a child of a previous OID component, but a child
187  * with attribute number 0 exists, and a child with attribute number 1 also exists,
188  * the child with attribute number 0 will be used as an 'index' pair, and will be
189  * created with the value of the non matching OID component.
190  *
191  * Parsing will then resume using the child with attribute number 1.
192  *
193  * This allows traversals of SNMP tables to be represented by the sequence of pairs
194  * and allows the full range of entry indexes which would not be possible if we represented
195  * table index numbers as TLV attributes.
196  *
197  * @param[in] ctx to allocate new pairs in.
198  * @param[in] conf radsnmp config.
199  * @param[in] cursor to add pairs to.
200  * @param[in] oid string to evaluate.
201  * @param[in] type SNMP value type.
202  * @param[in] value to assign to OID attribute (SET operations only).
203  * @return
204  * - >0 on success (how much of the OID string we parsed).
205  * - <=0 on failure (where format error occurred).
206  */
207 static ssize_t radsnmp_pair_from_oid(TALLOC_CTX *ctx, radsnmp_conf_t *conf, fr_cursor_t *cursor,
208  char const *oid, int type, char const *value)
209 {
210  ssize_t slen;
211  char const *p = oid;
212  unsigned int attr;
213  fr_dict_attr_t const *index_attr, *da;
214  fr_dict_attr_t const *parent = conf->snmp_root;
215  fr_pair_t *vp;
216  int ret;
217 
218  if (!oid) return 0;
219 
220  fr_cursor_tail(cursor);
221 
222  /*
223  * Trim first.
224  */
225  if (p[0] == '.') p++;
226 
227  /*
228  * Support for indexes. If we can't find an attribute
229  * matching a child at a given level in the OID tree,
230  * look for attribute 0 (type integer) at that level.
231  * We use this to represent the index instead.
232  */
233  for (;;) {
234  unsigned int num = 0;
235 
236  slen = fr_dict_attr_by_oid(conf->dict, &parent, &attr, p);
237  if (slen > 0) break;
238  p += -(slen);
239 
240  if (fr_dict_oid_component(&num, &p) < 0) break; /* Just advances the pointer */
241  assert(attr == num);
242  p++;
243 
244  /*
245  * Check for an index attribute
246  */
247  index_attr = fr_dict_attr_child_by_num(parent, 0);
248  if (!index_attr) {
249  fr_strerror_printf("Unknown OID component: No index attribute at this level");
250  break;
251  }
252 
253  if (index_attr->type != FR_TYPE_UINT32) {
254  fr_strerror_printf("Index is not a \"integer\"");
255  break;
256  }
257 
258  /*
259  * By convention SNMP entries are at .1
260  */
261  parent = fr_dict_attr_child_by_num(parent, 1);
262  if (!parent) {
263  fr_strerror_printf("Unknown OID component: No entry attribute at this level");
264  break;
265  }
266 
267  /*
268  * Entry must be a TLV type
269  */
270  if (parent->type != FR_TYPE_TLV) {
271  fr_strerror_printf("Entry is not \"tlv\"");
272  break;
273  }
274 
275  /*
276  * We've skipped over the index attribute, and
277  * the index number should be available in attr.
278  */
279  MEM(vp = fr_pair_afrom_da(ctx, index_attr));
280  vp->vp_uint32 = attr;
281 
282  fr_cursor_append(cursor, vp);
283  }
284 
285  /*
286  * We errored out processing the OID.
287  */
288  if (slen <= 0) {
289  error:
290  fr_cursor_free_list(cursor);
291  return slen;
292  }
293 
294  fr_strerror(); /* Clear pending errors */
295 
296  /*
297  * SNMP requests the leaf under the OID with .0.
298  */
299  if (attr != 0) {
300  da = fr_dict_attr_child_by_num(parent, attr);
301  if (!da) {
302  fr_strerror_printf("Unknown leaf attribute %i", attr);
303  return -(slen);
304  }
305  } else {
306  da = parent;
307  }
308 
309  MEM(vp = fr_pair_afrom_da(ctx, da));
310 
311  /*
312  * fr_pair_ts with no value need a 1 byte value buffer.
313  */
314  if (!value) {
315  switch (da->type) {
316  /*
317  * We can blame the authors of RFC 6929 for
318  * this hack. Apparently the presence or absence
319  * of an attribute isn't considered a useful means
320  * of conveying information, so empty TLVs are
321  * disallowed.
322  */
323  case FR_TYPE_TLV:
324  fr_pair_to_unknown(vp);
325  FALL_THROUGH;
326 
327  case FR_TYPE_OCTETS:
328  fr_pair_value_memdup(vp, (uint8_t const *)"\0", 1, true);
329  break;
330 
331  case FR_TYPE_STRING:
332  fr_pair_value_bstrndup(vp, "\0", 1, true);
333  break;
334 
335  /*
336  * Fine to leave other values zeroed out.
337  */
338  default:
339  break;
340  }
341 
342  fr_cursor_append(cursor, vp);
343  return slen;
344  }
345 
346  if (da->type == FR_TYPE_TLV) {
347  fr_strerror_printf("TLVs cannot hold values");
348  return -(slen);
349  }
350 
351  ret = fr_pair_value_from_str(vp, value, -1, '\0', true);
352  if (ret < 0) {
353  slen = -(slen);
354  goto error;
355  }
356 
357  MEM(vp = fr_pair_afrom_da(ctx, attr_freeradius_snmp_type));
358  vp->vp_uint32 = type;
359 
360  fr_cursor_append(cursor, vp);
361 
362  return slen;
363 }

365 /** Write the result of a get or getnext operation back to net-snmp
366  *
367  * Returns three lines of output per attribute:
368  * - OID string
369  * - type
370  * - value
371  *
372  * Index attributes (num 0) must be in order of depth (shallowest first).
373  *
374  * If no attributes were written, will write "NONE\n" to inform net-snmp
375  * that no value was available at the specified OID.
376  *
377  * @param fd to write to.
378  * @param root of the SNMP portion of the main dictionary.
379  * @param type attribute.
380  * @param head of list of attributes to convert and write.
381  * @return
382  * - >=0 on success (the number of varbind responses written).
383  * - -1 on failure.
384  */
385 static int radsnmp_get_response(int fd,
386  fr_dict_attr_t const *root, fr_dict_attr_t const *type,
387  fr_pair_t *head)
388 {
389  fr_cursor_t cursor;
390  fr_pair_t *vp, *type_vp;
391  fr_dict_attr_t const *parent = root;
392  unsigned int written = 0;
393 
394  ssize_t slen;
395  size_t len;
396 
397  char type_buff[32]; /* type */
398  size_t type_len = 0;
399  char oid_buff[256];
400  char value_buff[128];
401  char *p = oid_buff, *end = p + sizeof(oid_buff);
402 
403  struct iovec io_vector[6];
404 
405  char newline[] = "\n";
406 
407  type_buff[0] = '\0';
408 
409  /*
410  * Print first part of OID string.
411  */
412  slen = snprintf(oid_buff, sizeof(oid_buff), "%u.", parent->attr);
413  if (is_truncated((size_t)slen, sizeof(oid_buff))) {
414  oob:
415  fr_strerror_printf("OID Buffer too small");
416  return -1;
417  }
418  p += slen;
419 
420  /*
421  * @fixme, this is very dependent on ordering
422  *
423  * This code should be reworked when we have proper
424  * attribute grouping to coalesce all related index
425  * attributes under a single request OID.
426  */
427  for (vp = fr_cursor_init(&cursor, &head);
428  vp;
429  vp = fr_cursor_next(&cursor)) {
430  fr_dict_attr_t const *common;
431  /*
432  * We only care about TLV attributes beneath our root
433  */
434  if (!fr_dict_attr_common_parent(root, vp->da, true)) continue;
435 
436  /*
437  * Sanity checks to ensure we're processing attributes
438  * in the right order.
439  */
440  common = fr_dict_attr_common_parent(parent, vp->da, true);
441  if (!common) {
442  fr_strerror_printf("Out of order index attributes. \"%s\" is not a child of \"%s\"",
443  vp->da->name, parent->name);
444  return -1;
445  }
446 
447  /*
448  * Index attribute
449  */
450  if (vp->da->attr == 0) {
451  /*
452  * Print OID from last index/root up to the parent of
453  * the index attribute.
454  */
455  slen = fr_dict_print_attr_oid(&FR_SBUFF_OUT(p, end), parent, vp->da->parent);
456  if (slen <= 0) return -1;
457 
458  if (vp->vp_type != FR_TYPE_UINT32) {
459  fr_strerror_printf("Index attribute \"%s\" is not of type \"integer\"", vp->da->name);
460  return -1;
461  }
462 
463  if (slen >= (end - p)) goto oob;
464  p += slen;
465 
466  /*
467  * Add the value of the index attribute as the next
468  * OID component.
469  */
470  len = snprintf(p, end - p, ".%i.", vp->vp_uint32);
471  if (is_truncated(len, end - p)) goto oob;
472 
473  p += len;
474 
475  /*
476  * Set the parent to be the attribute representing
477  * the entry.
478  */
479  parent = fr_dict_attr_child_by_num(vp->da->parent, 1);
480  continue;
481  }
482 
483  /*
484  * Actual TLV attribute
485  */
486  slen = fr_dict_print_attr_oid(&FR_SBUFF_OUT(p, end), parent, vp->da);
487  if (slen < 0) return -1;
488 
489  /*
490  * Next attribute should be the type
491  */
492  type_vp = fr_cursor_next(&cursor);
493  if (!type_vp || (type_vp->da != type)) {
494  fr_strerror_printf("No %s found in response, or occurred out of order", type->name);
495  return -1;
496  }
497  slen = fr_pair_print_value_quoted(&FR_SBUFF_OUT(type_buff, sizeof(type_buff)), type_vp, T_BARE_WORD);
498  if (slen < 0) return -1;
499 
500  type_len = (size_t)slen;
501 
502  /*
503  * Build up the vector
504  *
505  * This represents output for a single varbind attribute
506  */
507  io_vector[0].iov_base = oid_buff;
508  io_vector[0].iov_len = strlen(oid_buff);
509  io_vector[1].iov_base = newline;
510  io_vector[1].iov_len = 1;
511  io_vector[2].iov_base = type_buff;
512  io_vector[2].iov_len = type_len;
513  io_vector[3].iov_base = newline;
514  io_vector[3].iov_len = 1;
515 
516  switch (vp->vp_type) {
517  case FR_TYPE_OCTETS:
518  memcpy(&io_vector[4].iov_base, &vp->vp_strvalue, sizeof(io_vector[4].iov_base));
519  io_vector[4].iov_len = vp->vp_length;
520  break;
521 
522  case FR_TYPE_STRING:
523  memcpy(&io_vector[4].iov_base, &vp->vp_strvalue, sizeof(io_vector[4].iov_base));
524  io_vector[4].iov_len = vp->vp_length;
525  break;
526 
527  default:
528  /*
529  * We call fr_value_box_print with a NULL da pointer
530  * because we always need return integer values not
531  * value aliases.
532  */
533  slen = fr_value_box_print(&FR_SBUFF_OUT(value_buff, sizeof(value_buff)), &vp->data, NULL);
534  if (slen < 0) {
535  fr_strerror_printf("Insufficient fixed value buffer");
536  return -1;
537  }
538  io_vector[4].iov_base = value_buff;
539  io_vector[4].iov_len = (size_t)slen;
540  break;
541  }
542  io_vector[5].iov_base = newline;
543  io_vector[5].iov_len = 1;
544 
545  DEBUG2("said: %s", (char *)io_vector[0].iov_base);
546  DEBUG2("said: %s", (char *)io_vector[2].iov_base);
547  DEBUG2("said: %s", (char *)io_vector[4].iov_base);
548 
549  if (writev(fd, io_vector, NUM_ELEMENTS(io_vector)) < 0) {
550  fr_strerror_printf("Failed writing varbind result: %s", fr_syserror(errno));
551  return -1;
552  }
553 
554  /*
555  * Reset in case we're encoding multiple values
556  */
557  parent = root;
558  p = oid_buff;
559  type_buff[0] = '\0';
560  written++;
561  }
562 
563  if (!written && (write(fd, "NONE\n", 5)) < 0) {
564  fr_strerror_printf("Failed writing get response: %s", fr_syserror(errno));
565  return -1;
566  }
567 
568  return written;
569 }

571 /** Write the result of a set operation back to net-snmp
572  *
573  * Writes "DONE\n" on success, or an error as described in man snmpd.conf
574  * on error.
575  *
576  * @param fd to write to.
577  * @param error attribute.
578  * @param head of list of attributes to convert and write.
579  * @return
580  * - 0 on success.
581  * - -1 on failure.
582  */
583 static int radsnmp_set_response(int fd, fr_dict_attr_t const *error, fr_pair_t *head)
584 {
585  fr_pair_t *vp;
586  char buffer[64];
587  ssize_t slen;
588  struct iovec io_vector[2];
589  char newline[] = "\n";
590 
591  vp = fr_pair_find_by_da(&head, error);
592  if (!vp) {
593  if (write(fd, "DONE\n", 5) < 0) {
594  fr_strerror_printf("Failed writing set response: %s", fr_syserror(errno));
595  return -1;
596  }
597  return 0;
598  }
599 
600  slen = fr_pair_print_value_quoted(&FR_SBUFF_OUT(buffer, sizeof(buffer)), vp, T_BARE_WORD);
601  if (slen < 0) {
602  assert(0);
603  return -1;
604  }
605 
606  io_vector[0].iov_base = buffer;
607  io_vector[0].iov_len = (size_t)slen;
608  io_vector[1].iov_base = newline;
609  io_vector[1].iov_len = 1;
610 
611  DEBUG2("said: %s", buffer);
612 
613  if (writev(fd, io_vector, NUM_ELEMENTS(io_vector)) < 0) {
614  fr_strerror_printf("Failed writing set response: %s", fr_syserror(errno));
615  return -1;
616  }
617 
618  return 0;
619 }

624 
625 #define NEXT_LINE(_line, _buffer) \
