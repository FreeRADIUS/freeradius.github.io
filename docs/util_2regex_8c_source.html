119 /** Thread local init for pcre2
120  *
121  */
122 static int fr_pcre2_tls_init(void)
123 {
124  fr_pcre2_tls_t *tls;
125 
126  if (unlikely(fr_pcre2_tls != NULL)) return 0;
127 
128  fr_pcre2_tls = tls = talloc_zero(NULL, fr_pcre2_tls_t);
129  if (!tls) return -1;
130  talloc_set_destructor(tls, _pcre2_tls_free);
131 
132  tls->gcontext = pcre2_general_context_create(_pcre2_talloc, _pcre2_talloc_free, NULL);
133  if (!tls->gcontext) {
134  fr_strerror_printf("Failed allocating general context");
135  return -1;
136  }
137 
138  tls->ccontext = pcre2_compile_context_create(tls->gcontext);
139  if (!tls->ccontext) {
140  fr_strerror_printf("Failed allocating compile context");
141  error:
142  fr_pcre2_tls = NULL;
143  _pcre2_tls_free(tls);
144  return -1;
145  }
146 
147  tls->mcontext = pcre2_match_context_create(tls->gcontext);
148  if (!tls->mcontext) {
149  fr_strerror_printf("Failed allocating match context");
150  goto error;
151  }
152 
153 #ifdef PCRE2_CONFIG_JIT
154  pcre2_config(PCRE2_CONFIG_JIT, &tls->do_jit);
155  if (tls->do_jit) {
156  tls->jit_stack = pcre2_jit_stack_create(FR_PCRE_JIT_STACK_MIN, FR_PCRE_JIT_STACK_MAX, tls->gcontext);
157  if (!tls->jit_stack) {
158  fr_strerror_printf("Failed allocating JIT stack");
159  goto error;
160  }
161  pcre2_jit_stack_assign(tls->mcontext, NULL, tls->jit_stack);
162  }
163 #endif
164 
165  /*
166  * Free on thread exit
167  */
168  fr_thread_local_set_destructor(fr_pcre2_tls, _pcre2_tls_free_on_exit, tls);
169  fr_pcre2_tls = tls; /* Assign to thread local storage */
170 
171  return 0;
172 }

187 /** Wrapper around pcre2_compile
188  *
189  * Allows the rest of the code to do compilations using one function signature.
190  *
191  * @note Compiled expression must be freed with talloc_free.
192  *
193  * @param[out] out Where to write out a pointer to the structure containing
194  * the compiled expression.
195  * @param[in] pattern to compile.
196  * @param[in] len of pattern.
197  * @param[in] flags controlling matching. May be NULL.
198  * @param[in] subcaptures Whether to compile the regular expression to store subcapture
199  * data.
200  * @param[in] runtime If false run the pattern through the PCRE JIT (if available)
201  * to convert it to machine code. This trades startup time (longer)
202  * for runtime performance (better).
203  * @return
204  * - >= 1 on success.
205  * - <= 0 on error. Negative value is offset of parse error.
206  */
207 ssize_t regex_compile(TALLOC_CTX *ctx, regex_t **out, char const *pattern, size_t len,
208  fr_regex_flags_t const *flags, bool subcaptures, bool runtime)
209 {
210  int ret;
211  PCRE2_SIZE offset;
212  uint32_t cflags = 0;
213  regex_t *preg;
214 
215  /*
216  * Check inputs
217  */
218  *out = NULL;
219 
220  /*
221  * Thread local initialisation
222  */
223  if (unlikely(!fr_pcre2_tls) && (fr_pcre2_tls_init() < 0)) return -1;
224 
225  if (len == 0) {
226  fr_strerror_printf("Empty expression");
227  return 0;
228  }
229 
230  /*
231  * Options
232  */
233  if (flags) {
234  /* flags->global implemented by substitution function */
235  if (flags->ignore_case) cflags |= PCRE2_CASELESS;
236  if (flags->multiline) cflags |= PCRE2_MULTILINE;
237  if (flags->dot_all) cflags |= PCRE2_DOTALL;
238  if (flags->unicode) cflags |= PCRE2_UTF;
239  if (flags->extended) cflags |= PCRE2_EXTENDED;
240  }
241 
242  if (!subcaptures) cflags |= PCRE2_NO_AUTO_CAPTURE;
243 
244  preg = talloc_zero(ctx, regex_t);
245  talloc_set_destructor(preg, _regex_free);
246 
247  preg->compiled = pcre2_compile((PCRE2_SPTR8)pattern, len,
248  cflags, &ret, &offset, fr_pcre2_tls->ccontext);
249  if (!preg->compiled) {
250  PCRE2_UCHAR errbuff[128];
251 
252  pcre2_get_error_message(ret, errbuff, sizeof(errbuff));
253  fr_strerror_printf("%s", (char *)errbuff);
254  talloc_free(preg);
255 
256  return -(ssize_t)offset;
257  }
258 
259  if (!runtime) {
260  preg->precompiled = true;
261 
262 #ifdef PCRE2_CONFIG_JIT
263  /*
264  * This is expensive, so only do it for
265  * expressions that are going to be
266  * evaluated repeatedly.
267  */
268  if (fr_pcre2_tls->do_jit) {
269  ret = pcre2_jit_compile(preg->compiled, PCRE2_JIT_COMPLETE);
270  if (ret < 0) {
271  PCRE2_UCHAR errbuff[128];
272 
273  pcre2_get_error_message(ret, errbuff, sizeof(errbuff));
274  fr_strerror_printf("Pattern JIT failed: %s", (char *)errbuff);
275  talloc_free(preg);
276 
277  return 0;
278  }
279  preg->jitd = true;
280  }
281 #endif
282  }
283 
284  *out = preg;
285 
286  return len;
287 }

289 /** Wrapper around pcre2_exec
290  *
291  * @param[in] preg The compiled expression.
292  * @param[in] subject to match.
293  * @param[in] len Length of subject.
294  * @param[in] regmatch Array of match pointers.
295  * @return
296  * - -1 on failure.
297  * - 0 on no match.
298  * - 1 on match.
299  */
300 int regex_exec(regex_t *preg, char const *subject, size_t len, fr_regmatch_t *regmatch)
301 {
302  int ret;
303  uint32_t options = 0;
304 
305  char *our_subject = NULL;
306  bool dup_subject = true;
307  pcre2_match_data *match_data;
308 
309  /*
310  * Thread local initialisation
311  */
312  if (unlikely(!fr_pcre2_tls) && (fr_pcre2_tls_init() < 0)) return -1;
313 
314  if (regmatch) {
315 #ifdef PCRE2_COPY_MATCHED_SUBJECT
316  /*
317  * This is apparently only supported for pcre2_match
318  * NOT pcre2_jit_match.
319  */
320 # ifdef PCRE2_CONFIG_JIT
321  if (!preg->jitd) {
322 # endif
323  dup_subject = false;
324 
325  /*
326  * If PCRE2_COPY_MATCHED_SUBJECT is available
327  * and set as an options flag, pcre2_match will
328  * strdup the subject string if pcre2_match is
329  * successful and store a pointer to it in the
330  * regmatch struct.
331  *
332  * The lifetime of the string memory will be
333  * bound to the regmatch struct. This is more
334  * efficient that doing it ourselves, as the
335  * strdup only occurs if the subject matches.
336  */
337  options |= PCRE2_COPY_MATCHED_SUBJECT;
338 # ifdef PCRE2_CONFIG_JIT
339  }
340 # endif
341 #endif
342  if (dup_subject) {
343  /*
344  * We have to dup and operate on the duplicate
345  * of the subject, because pcre2_jit_match and
346  * pcre2_match store a pointer to the subject
347  * in the regmatch structure.
348  */
349  subject = our_subject = talloc_bstrndup(regmatch, subject, len);
350  if (!subject) {
351  fr_strerror_printf("Out of memory");
352  return -1;
353  }
354 #ifndef NDEBUG
355  regmatch->subject = subject; /* Stored only for tracking memory issues */
356 #endif
357  }
358  }
359 
360  /*
361  * If we weren't given match data we
362  * need to alloc it else pcre2_match
363  * fails when passed NULL match data.
364  */
365  if (!regmatch) {
366  match_data = pcre2_match_data_create_from_pattern(preg->compiled, fr_pcre2_tls->gcontext);
367  if (!match_data) {
368  fr_strerror_printf("Failed allocating temporary match data");
369  return -1;
370  }
371  } else {
372  match_data = regmatch->match_data;
373  }
374 
375 #ifdef PCRE2_CONFIG_JIT
376  if (preg->jitd) {
377  ret = pcre2_jit_match(preg->compiled, (PCRE2_SPTR8)subject, len, 0, options,
378  match_data, fr_pcre2_tls->mcontext);
379  } else
380 #endif
381  {
382  ret = pcre2_match(preg->compiled, (PCRE2_SPTR8)subject, len, 0, options,
383  match_data, fr_pcre2_tls->mcontext);
384  }
385  if (!regmatch) pcre2_match_data_free(match_data);
386  if (ret < 0) {
387  PCRE2_UCHAR errbuff[128];
388 
389  if (dup_subject) talloc_free(our_subject);
390 
391  if (ret == PCRE2_ERROR_NOMATCH) {
392  if (regmatch) regmatch->used = 0;
393  return 0;
394  }
395 
396  pcre2_get_error_message(ret, errbuff, sizeof(errbuff));
397  fr_strerror_printf("regex evaluation failed with code (%i): %s", ret, errbuff);
398 
399  return -1;
400  }
401 
402  if (regmatch) regmatch->used = ret;
403 
404  return 1;
405 }

407 /** Wrapper around pcre2_substitute
408  *
409  * @param[in] ctx to allocate output string in.
410  * @param[out] out Output string with replacements performed.
411  * @param[in] max_out Maximum length of output buffer. If this is 0 then
412  * the output length is unlimited.
413  * @param[in] preg The compiled expression.
414  * @param[in] flags that affect matching.
415  * @param[in] subject to perform replacements on.
416  * @param[in] subject_len the length of the subject.
417  * @param[in] replacement replacement string containing substitution
418  * markers.
419  * @param[in] replacement_len Length of the replacement string.
420  * @param[in] regmatch Array of match pointers.
421  * @return
422  * - >= 0 the length of the output string.
423  * - < 0 on error.
424  */
425 int regex_substitute(TALLOC_CTX *ctx, char **out, size_t max_out, regex_t *preg, fr_regex_flags_t *flags,
426  char const *subject, size_t subject_len,
427  char const *replacement, size_t replacement_len,
428  fr_regmatch_t *regmatch)
429 {
430  int ret;
431  uint32_t options = 0;
432  size_t buff_len, actual_len;
433  char *buff;
434 
435 #ifndef PCRE2_COPY_MATCHED_SUBJECT
436  char *our_subject = NULL;
437 #endif
438 
439  /*
440  * Thread local initialisation
441  */
442  if (unlikely(!fr_pcre2_tls) && (fr_pcre2_tls_init() < 0)) return -1;
443 
444  /*
445  * Internally pcre2_substitute just calls pcre2_match to
446  * generate the match data, so the same hack as the
447  * regex_exec function above is required.
448  */
449  if (regmatch) {
450 #ifndef PCRE2_COPY_MATCHED_SUBJECT
451  /*
452  * We have to dup and operate on the duplicate
453  * of the subject, because pcre2_jit_match and
454  * pcre2_match store a pointer to the subject
455  * in the regmatch structure.
456  */
457  subject = our_subject = talloc_bstrndup(regmatch, subject, subject_len);
458  if (!subject) {
459  fr_strerror_printf("Out of memory");
460  return -1;
461  }
462 #else
463  /*
464  * If PCRE2_COPY_MATCHED_SUBJECT is available
465  * and set as an options flag, pcre2_match will
466  * strdup the subject string if pcre2_match is
467  * successful and store a pointer to it in the
468  * regmatch struct.
469  *
470  * The lifetime of the string memory will be
471  * bound to the regmatch struct. This is more
472  * efficient that doing it ourselves, as the
473  * strdup only occurs if the subject matches.
474  */
475  options |= PCRE2_COPY_MATCHED_SUBJECT;
476 #endif
477  }
478 
479  /*
480  * Guess (badly) what the length of the output buffer should be
481  */
482  actual_len = buff_len = subject_len + 1; /* +1 for the \0 */
483  buff = talloc_array(ctx, char, buff_len);
484  if (!buff) {
485 #ifndef PCRE2_COPY_MATCHED_SUBJECT
486  talloc_free(our_subject);
487 #endif
488  fr_strerror_printf("Out of memory");
489  return -1;
490  }
491 
492  options |= PCRE2_SUBSTITUTE_OVERFLOW_LENGTH;
493  if (flags->global) options |= PCRE2_SUBSTITUTE_GLOBAL;
494 
495 again:
496  /*
497  * actual_len input value should be the size of the
498  * buffer including space for '\0'.
499  * If input buffer is too small, then actual_len will be set
500  * to the buffer space needed including space for '\0'.
501  * If input buffer is the correct size, then actual_len
502  * will be set to the size of the string written to buff
503  * without the terminating '\0'.
504  */
505  ret = pcre2_substitute(preg->compiled,
506  (PCRE2_SPTR8)subject, (PCRE2_SIZE)subject_len, 0,
507  options, regmatch ? regmatch->match_data : NULL, fr_pcre2_tls->mcontext,
508  (PCRE2_UCHAR const *)replacement, replacement_len, (PCRE2_UCHAR *)buff, &actual_len);
509 
510  if (ret < 0) {
511  PCRE2_UCHAR errbuff[128];
512 
513 #ifndef PCRE2_COPY_MATCHED_SUBJECT
514  talloc_free(our_subject);
515 #endif
516  talloc_free(buff);
517 
518  if (ret == PCRE2_ERROR_NOMEMORY) {
519  if ((max_out > 0) && (actual_len > max_out)) {
520  fr_strerror_printf("String length with substitutions (%zu) "
521  "exceeds max string length (%zu)", actual_len - 1, max_out - 1);
522  return -1;
523  }
524 
525  /*
526  * Check that actual_len != buff_len as that'd be
527  * an actual error.
528  */
529  if (actual_len == buff_len) {
530  fr_strerror_printf("libpcre2 out of memory");
531  return -1;
532  }
533 
534  talloc_free(buff);
535  buff_len = actual_len; /* The length we get passed back includes the \0 */
536  buff = talloc_array(ctx, char, buff_len);
537  goto again;
538  }
539 
540  if (ret == PCRE2_ERROR_NOMATCH) {
541  if (regmatch) regmatch->used = 0;
542  return 0;
543  }
544 
545  pcre2_get_error_message(ret, errbuff, sizeof(errbuff));
546  fr_strerror_printf("regex evaluation failed with code (%i): %s", ret, errbuff);
547  return -1;
548  }
549 
550  /*
551  * Trim the replacement buffer to the correct length
552  *
553  * buff_len includes \0.
554  * ...and as pcre2_substitute just succeeded actual_len does not include \0.
555  */
556  if (actual_len < (buff_len - 1)) {
557  buff = talloc_bstr_realloc(ctx, buff, actual_len);
558  if (!buff) {
559  fr_strerror_printf("reallocing pcre2_substitute result buffer failed");
560  return -1;
561  }
562  }
563 
564  if (regmatch) regmatch->used = ret;
565  *out = buff;
566 
567  return 1;
568 }
