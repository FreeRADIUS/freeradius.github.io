63 /** Convert CONFIG_PAIR (which may contain refs) to map_t.
64  *
65  * Treats the left operand as an attribute reference
66  * @verbatim<request>.<list>.<attribute>@endverbatim
67  *
68  * Treatment of left operand depends on quotation, barewords are treated as
69  * attribute references, double quoted values are treated as expandable strings,
70  * single quoted values are treated as literal strings.
71  *
72  * Return must be freed with talloc_free
73  *
74  * @param[in] ctx for talloc.
75  * @param[in] out Where to write the pointer to the new #map_t.
76  * @param[in] cp to convert to map.
77  * @param[in] lhs_rules rules for parsing LHS attribute references.
78  * @param[in] rhs_rules rules for parsing RHS attribute references.
79  * @return
80  * - #map_t if successful.
81  * - NULL on error.
82  */
83 int map_afrom_cp(TALLOC_CTX *ctx, map_t **out, CONF_PAIR *cp,
84  tmpl_rules_t const *lhs_rules, tmpl_rules_t const *rhs_rules)
85 {
86  map_t *map;
87  char const *attr, *value, *marker_subject;
88  ssize_t slen;
89  fr_token_t type;
90 
91  *out = NULL;
92 
93  if (!cp) return -1;
94 
95  MEM(map = talloc_zero(ctx, map_t));
96  map->op = cf_pair_operator(cp);
97  map->ci = cf_pair_to_item(cp);
98 
99  attr = cf_pair_attr(cp);
100  value = cf_pair_value(cp);
101  if (!value) {
102  cf_log_err(cp, "Missing attribute value");
103  goto error;
104  }
105 
106  /*
107  * LHS may be an expansion (that expands to an attribute reference)
108  * or an attribute reference. Quoting determines which it is.
109  */
110  type = cf_pair_attr_quote(cp);
111  switch (type) {
112  case T_DOUBLE_QUOTED_STRING:
113  case T_BACK_QUOTED_STRING:
114  slen = tmpl_afrom_substr(ctx, &map->lhs,
115  &FR_SBUFF_IN(attr, talloc_array_length(attr) - 1),
116  type,
117  tmpl_parse_rules_unquoted[type], /* We're not searching for quotes */
118  lhs_rules);
119  if (slen <= 0) {
120  char *spaces, *text;
121 
122  marker_subject = attr;
123  marker:
124  fr_canonicalize_error(ctx, &spaces, &text, slen, marker_subject);
125  cf_log_err(cp, "%s", text);
126  cf_log_perr(cp, "%s^", spaces);
127 
128  talloc_free(spaces);
129  talloc_free(text);
130  goto error;
131  }
132  break;
133 
134  default:
135  slen = tmpl_afrom_attr_str(ctx, NULL, &map->lhs, attr, lhs_rules);
136  if (slen <= 0) {
137  cf_log_err(cp, "Failed parsing attribute reference");
138  marker_subject = attr;
139  goto marker;
140  }
141 
142  if (tmpl_is_attr(map->lhs) && tmpl_attr_unknown_add(map->lhs) < 0) {
143  cf_log_perr(cp, "Failed creating attribute %s", map->lhs->name);
144  goto error;
145  }
146  break;
147  }
148 
149  /*
150  * RHS might be an attribute reference.
151  */
152  type = cf_pair_value_quote(cp);
153 
154  slen = tmpl_afrom_substr(map, &map->rhs,
155  &FR_SBUFF_IN(value, strlen(value)),
156  type,
157  tmpl_parse_rules_unquoted[type], /* We're not searching for quotes */
158  rhs_rules);
159  if (slen < 0) {
160  marker_subject = value;
161  goto marker;
162  }
163 
164  if (!map->rhs) {
165  cf_log_perr(cp, "Failed parsing RHS");
166  goto error;
167  }
168 
169  if (tmpl_is_attr(map->rhs) && (tmpl_attr_unknown_add(map->rhs) < 0)) {
170  cf_log_perr(cp, "Failed creating attribute %s", map->rhs->name);
171  goto error;
172  }
173 
174  /*
175  * We cannot assign a count to an attribute. That must
176  * be done in an xlat.
177  */
178  if (tmpl_is_attr(map->rhs) &&
179  (tmpl_num(map->rhs) == NUM_COUNT)) {
180  cf_log_err(cp, "Cannot assign from a count");
181  goto error;
182  }
183 
184  MAP_VERIFY(map);
185 
186  *out = map;
187 
188  return 0;
189 
190 error:
191  talloc_free(map);
192  return -1;
193 }

254 /** Parse sbuff into (which may contain refs) to map_t.
255  *
256  * Treats the left operand as an attribute reference
257  * @verbatim<request>.<list>.<attribute>@endverbatim
258  *
259  * Treatment of left operand depends on quotation, barewords are treated as
260  * attribute references, double quoted values are treated as expandable strings,
261  * single quoted values are treated as literal strings.
262  *
263  * The op_table should be #cond_cmp_op_table for check items, and
264  * #map_assignment_op_table for reply items.
265  *
266  * Return must be freed with talloc_free
267  *
268  * @param[in] ctx for talloc.
269  * @param[in] out Where to write the pointer to the new #map_t.
270  * @param[in] in to convert to map.
271  * @param[in] op_table for lhs OP rhs
272  * @param[in] op_table_len length of op_table
273  * @param[in] lhs_rules rules for parsing LHS attribute references.
274  * @param[in] rhs_rules rules for parsing RHS attribute references.
275  * @return
276  * - <0 on error
277  * - 0 on success
278  */
279 int map_afrom_sbuff(TALLOC_CTX *ctx, map_t **out, fr_sbuff_t *in,
280  fr_table_num_sorted_t const *op_table, size_t op_table_len,
281  tmpl_rules_t const *lhs_rules, tmpl_rules_t const *rhs_rules)
282 {
283  char quote;
284  ssize_t slen;
285  fr_token_t token;
286  tmpl_t *tmpl;
287  map_t *map;
288  fr_sbuff_t sbuff = FR_SBUFF_NO_ADVANCE(in);
289 
290  *out = NULL;
291  MEM(map = talloc_zero(ctx, map_t));
292 
293  slen = fr_sbuff_adv_past_whitespace(&sbuff, SIZE_MAX);
294  if (slen < 0) return -1;
295 
296  if (!fr_sbuff_remaining(in)) return 0;
297 
298  quote = *fr_sbuff_current(&sbuff);
299  switch (quote) {
300  case '"':
301  token = T_DOUBLE_QUOTED_STRING;
302  goto parse_lhs;
303 
304  case '\'':
305  token = T_SINGLE_QUOTED_STRING;
306  goto parse_lhs;
307 
308  case '`':
309  token = T_BACK_QUOTED_STRING;
310  parse_lhs:
311  fr_sbuff_advance(&sbuff, 1); /* no need to check */
312 
313  slen = tmpl_afrom_substr(map, &map->lhs, &sbuff, token,
314  tmpl_parse_rules_quoted[token], lhs_rules);
315  break;
316 
317  default:
318  token = T_BARE_WORD;
319 
320  slen = tmpl_afrom_attr_substr(map, NULL, &map->lhs, &sbuff,
321  &map_parse_rules_bareword_quoted, lhs_rules);
322  break;
323  }
324 
325  if (slen <= 0) {
326  error:
327  talloc_free(map);
328  return -1;
329  }
330 
331  /*
332  * Check for, and skip, the trailing quote if we had a leading quote.
333  */
334  if (token != T_BARE_WORD) {
335  if (!fr_sbuff_is_char(&sbuff, quote)) {
336  fr_strerror_printf("Unexpected end of quoted string");
337  return -1;
338  }
339 
340  fr_sbuff_advance(&sbuff, 1);
341  }
342 
343  slen = fr_sbuff_adv_past_whitespace(&sbuff, SIZE_MAX);
344  if (slen < 0) {
345  fr_strerror_printf("Unexpected end of string after parsing left side");
346  goto error;
347  }
348 
349  /*
350  * Parse operator.
351  */
352  fr_sbuff_out_by_longest_prefix(&slen, &map->op, op_table, &sbuff, 0);
353  if (slen == 0) {
354  fr_strerror_printf("Invalid operator");
355  goto error;
356  }
357 
358  slen = fr_sbuff_adv_past_whitespace(&sbuff, SIZE_MAX);
359  if (slen < 0) {
360  fr_strerror_printf("Unexpected end of string after operator");
361  goto error;
362  }
363 
364  /*
365  * Copy LHS code above, except parsing in RHS, with some
366  * minor modifications.
367  */
368  quote = *fr_sbuff_current(&sbuff);
369  switch (quote) {
370  case '"':
371  token = T_DOUBLE_QUOTED_STRING;
372  goto parse_rhs;
373 
374  case '\'':
375  token = T_SINGLE_QUOTED_STRING;
376  goto parse_rhs;
377 
378  case '`':
379  token = T_BACK_QUOTED_STRING;
380  parse_rhs:
381  fr_sbuff_advance(&sbuff, 1); /* no need to check */
382 
383  slen = tmpl_afrom_substr(map, &map->rhs, &sbuff, token,
384  tmpl_parse_rules_quoted[token], rhs_rules);
385  break;
386 
387  default:
388  token = T_BARE_WORD;
389 
390  slen = tmpl_afrom_substr(map, &tmpl, &sbuff, token,
391  tmpl_parse_rules_quoted[token], rhs_rules);
392  break;
393  }
394  if (slen <= 0) goto error;
395 
396  /*
397  * Check for, and skip, the trailing quote if we had a leading quote.
398  */
399  if (token != T_BARE_WORD) {
400  if (!fr_sbuff_is_char(&sbuff, quote)) {
401  fr_strerror_printf("Unexpected end of quoted string");
402  return -1;
403  }
404 
405  fr_sbuff_advance(&sbuff, 1);
406  }
407 
408  fr_sbuff_set(in, &sbuff);
409  MAP_VERIFY(map);
410  *out = map;
411 
412  return 0;
413 }

416 /** Convert an 'update' config section into an attribute map.
417  *
418  * Uses 'name2' of section to set default request and lists.
419  *
420  * @param[in] ctx for talloc.
421  * @param[out] out Where to store the allocated map.
422  * @param[in] cs the update section
423  * @param[in] lhs_rules rules for parsing LHS attribute references.
424  * @param[in] rhs_rules rules for parsing RHS attribute references.
425  * @param[in] validate map using this callback (may be NULL).
426  * @param[in] uctx to pass to callback.
427  * @param[in] max number of mappings to process.
428  * @return
429  * - 0 on success.
430  * - -1 on failure.
431  */
432 int map_afrom_cs(TALLOC_CTX *ctx, map_t **out, CONF_SECTION *cs,
433  tmpl_rules_t const *lhs_rules, tmpl_rules_t const *rhs_rules,
434  map_validate_t validate, void *uctx,
435  unsigned int max)
436 {
437  char const *cs_list, *p;
438 
439  CONF_ITEM *ci;
440  CONF_PAIR *cp;
441 
442  unsigned int total = 0;
443  map_t **tail, *map;
444  TALLOC_CTX *parent;
445 
446  tmpl_rules_t our_lhs_rules = *lhs_rules; /* Mutable copy of the destination */
447 
448  *out = NULL;
449  tail = out;
450 
451  /*
452  * The first map has ctx as the parent.
453  * The rest have the previous map as the parent.
454  */
455  parent = ctx;
456 
457  ci = cf_section_to_item(cs);
458 
459  /*
460  * Check the destination list for "update" sections.
461  */
462  cs_list = p = cf_section_name2(cs);
463  if (cs_list && (strcmp(cf_section_name1(cs), "update") == 0)) {
464  p += radius_request_name(&our_lhs_rules.request_def, p, REQUEST_CURRENT);
465  if (our_lhs_rules.request_def == REQUEST_UNKNOWN) {
466  cf_log_err(ci, "Default request specified in mapping section is invalid");
467  return -1;
468  }
469 
470  our_lhs_rules.list_def = fr_table_value_by_str(pair_list_table, p, PAIR_LIST_UNKNOWN);
471  if (our_lhs_rules.list_def == PAIR_LIST_UNKNOWN) {
472  cf_log_err(ci, "Default list \"%s\" specified in mapping section is invalid", p);
473  return -1;
474  }
475  }
476 
477  for (ci = cf_item_next(cs, NULL);
478  ci != NULL;
479  ci = cf_item_next(cs, ci)) {
480  if (total++ == max) {
481  cf_log_err(ci, "Map size exceeded");
482  error:
483  TALLOC_FREE(*out);
484  return -1;
485  }
486 
487  /*
488  * If we have a subsection, AND the name2 is an
489  * assignment operator, THEN we allow sub-maps.
490  */
491  if (cf_item_is_section(ci)) {
492  CONF_SECTION *subcs;
493  fr_token_t token;
494  ssize_t slen;
495  bool qualifiers = our_lhs_rules.disallow_qualifiers;
496 
497  subcs = cf_item_to_section(ci);
498  token = cf_section_name2_quote(subcs);
499 
500  if (!fr_assignment_op[token]) {
501  cf_log_err(ci, "Invalid operator '%s'", fr_tokens[token]);
502  goto error;
503  }
504 
505  MEM(map = map_alloc(parent));
506  map->op = token;
507  map->ci = ci;
508 
509  /*
510  * The LHS MUST be an attribute name.
511  * map_afrom_cp() allows for dynamic
512  * names, but for simplicity we forbid
513  * them for now. Once the functionality
514  * is tested and used, we can allow that.
515  */
516  slen = tmpl_afrom_attr_str(ctx, NULL, &map->lhs, cf_section_name1(subcs), &our_lhs_rules);
517  if (slen <= 0) {
518  cf_log_err(ci, "Failed parsing attribute reference");
519  talloc_free(map);
520  goto error; /* re-do "goto marker" stuff to print out spaces ? */
521  }
522 
523  /*
524  * The LHS MUST be an attribute reference
525  * for now.
526  */
527  if (!tmpl_is_attr(map->lhs)) {
528  cf_log_err(ci, "Left side of group '%s' is NOT an attribute reference",
529  map->lhs->name);
530  talloc_free(map);
531  goto error; /* re-do "goto marker" stuff to print out spaces ? */
532  }
533 
534  if (tmpl_da(map->lhs)->flags.is_unknown) {
535  cf_log_err(ci, "Unknown attribute '%s'", map->lhs->name);
536  talloc_free(map);
537  goto error; /* re-do "goto marker" stuff to print out spaces ? */
538  }
539 
540  /*
541  * Disallow list qualifiers for the child
542  * templates. The syntax requires that
543  * the child attributes go into the
544  * parent one.
545  */
546  our_lhs_rules.disallow_qualifiers = true;
547 
548  /*
549  * The leaf reference of the outer section
550  * is used as the parsing context of the
551  * inner section.
552  */
553  our_lhs_rules.attr_parent = tmpl_da(map->lhs);
554  our_lhs_rules.prefix = TMPL_ATTR_REF_PREFIX_NO;
555 
556  /*
557  * This prints out any relevant error
558  * messages. We MAY want to print out
559  * additional ones, but that might get
560  * complex and confusing.
561  */
562  if (map_afrom_cs(map, &map->child, cf_item_to_section(ci),
563  &our_lhs_rules, rhs_rules, validate, uctx, max) < 0) {
564  talloc_free(map);
565  goto error;
566  }
567 
568  our_lhs_rules.disallow_qualifiers = qualifiers;
569  MAP_VERIFY(map);
570  goto next;
571  }
572 
573  if (!cf_item_is_pair(ci)) {
574  cf_log_err(ci, "Entry is not in \"attribute = value\" format");
575  goto error;
576  }
577 
578  cp = cf_item_to_pair(ci);
579  fr_assert(cp != NULL);
580 
581  if (map_afrom_cp(parent, &map, cp, &our_lhs_rules, rhs_rules) < 0) {
582  cf_log_err(ci, "Failed creating map from '%s = %s'",
583  cf_pair_attr(cp), cf_pair_value(cp));
584  goto error;
585  }
586 
587  MAP_VERIFY(map);
588 
589  /*
590  * Check the types in the map are valid
591  */
592  if (validate && (validate(map, uctx) < 0)) goto error;
593 
594  next:
595  parent = *tail = map;
596  tail = &(map->next);
597  }
598 
599  return 0;
600 
601 }

754 static void map_sort_split(map_t *source, map_t **front, map_t **back)
755 {
756  map_t *fast;
757  map_t *slow;
758 
759  /*
760  * Stopping condition - no more elements left to split
761  */
762  if (!source || !source->next) {
763  *front = source;
764  *back = NULL;
765 
766  return;
767  }
768 
769  /*
770  * Fast advances twice as fast as slow, so when it gets to the end,
771  * slow will point to the middle of the linked list.
772  */
773  slow = source;
774  fast = source->next;
775 
776  while (fast) {
777  fast = fast->next;
778  if (fast) {
779  slow = slow->next;
780  fast = fast->next;
781  }
782  }
783 
784  *front = source;
785  *back = slow->next;
786  slow->next = NULL;
787 }

789 static map_t *map_sort_merge(map_t *a, map_t *b, fr_cmp_t cmp)
790 {
791  map_t *result = NULL;
792 
793  if (!a) return b;
794  if (!b) return a;
795 
796  /*
797  * Compare things in the maps
798  */
799  if (cmp(a, b) <= 0) {
800  result = a;
801  result->next = map_sort_merge(a->next, b, cmp);
802  } else {
803  result = b;
804  result->next = map_sort_merge(a, b->next, cmp);
805  }
806 
807  return result;
808 }

810 /** Sort a linked list of #map_t using merge sort
811  *
812  * @param[in,out] maps List of #map_t to sort.
813  * @param[in] cmp to sort with
814  */
815 void map_sort(map_t **maps, fr_cmp_t cmp)
816 {
817  map_t *head = *maps;
818  map_t *a;
819  map_t *b;
820 
821  /*
822  * If there's 0-1 elements it must already be sorted.
823  */
824  if (!head || !head->next) {
825  return;
826  }
827 
828  map_sort_split(head, &a, &b); /* Split into sublists */
829  map_sort(&a, cmp); /* Traverse left */
830  map_sort(&b, cmp); /* Traverse right */
831 
832  /*
833  * merge the two sorted lists together
834  */
835  *maps = map_sort_merge(a, b, cmp);
