66 static int mod_decode(UNUSED void const *instance, request_t *request, uint8_t *const data, size_t data_len)
67 {
68 // proto_arp_t const *inst = talloc_get_type_abort_const(instance, proto_arp_t);
69  fr_arp_packet_t const *arp;
70 
71  /*
72  * Set the request dictionary so that we can do
73  * generic->protocol attribute conversions as
74  * the request runs through the server.
75  */
76  request->dict = dict_arp;
77 
78  if (fr_arp_decode(request->packet, data, data_len, &request->request_pairs) < 0) {
79  RPEDEBUG("Failed decoding packet");
80  return -1;
81  }
82 
83  arp = (fr_arp_packet_t const *) data;
84  request->packet->code = (arp->op[0] << 8) | arp->op[1];
85  fr_assert(request->packet->code < FR_ARP_MAX_PACKET_CODE);
86 
87  request->packet->data = talloc_memdup(request->packet, data, data_len);
88  request->packet->data_len = data_len;
89 
90  request->config = main_config;
91  REQUEST_VERIFY(request);
92 
93  if (RDEBUG_ENABLED) {
94  RDEBUG("Received ARP %s via socket %s",
95  fr_arp_packet_codes[request->packet->code],
96  request->async->listen->name);
97 
98  log_request_pair_list(L_DBG_LVL_1, request, request->request_pairs, "");
99  }
100 
101  return 0;
102 }

106 static ssize_t mod_encode(void const *instance, request_t *request, uint8_t *buffer, size_t buffer_len)
107 {
108  ssize_t slen;
109  proto_arp_t const *inst = talloc_get_type_abort_const(instance, proto_arp_t);
110  fr_arp_packet_t *arp;
111 
112  /*
113  * The packet timed out. Tell the network side that the packet is dead.
114  */
115  if (buffer_len == 1) {
116  *buffer = true;
117  return 1;
118  }
119 
120  /*
121  * "Do not respond"
122  */
123  if (!inst->active ||
124  (request->reply->code == FR_ARP_CODE_DO_NOT_RESPOND) ||
125  (request->reply->code == 0) || (request->reply->code >= FR_ARP_MAX_PACKET_CODE)) {
126  *buffer = false;
127  return 1;
128  }
129 
130  slen = fr_arp_encode(&FR_DBUFF_TMP(buffer, buffer_len), request->packet->data, request->reply_pairs);
131  if (slen <= 0) {
132  RPEDEBUG("Failed encoding reply");
133  return -1;
134  }
135  fr_assert(slen == FR_ARP_PACKET_SIZE);
136 
137  arp = (fr_arp_packet_t *) buffer;
138  fr_assert(request->packet->data_len == FR_ARP_PACKET_SIZE);
139 
140  if (memcmp(arp->sha, zeros, sizeof(arp->sha)) == 0) {
141  RDEBUG("WARNING: Sender-Hardware-Address of zeros will likely cause problems");
142  }
143 
144  if (RDEBUG_ENABLED) {
145  RDEBUG("Sending %d via socket %s",
146  request->reply->code,
147  request->async->listen->name);
148 
149  log_request_pair_list(L_DBG_LVL_1, request, request->reply_pairs, "");
150  }
151 
152  return slen;
153 }

179 /** Open listen sockets/connect to external event source
180  *
181  * @param[in] instance Ctx data for this application.
182  * @param[in] sc to add our file descriptor to.
183  * @param[in] conf Listen section parsed to give us instance.
184  * @return
185  * - 0 on success.
186  * - -1 on failure.
187  */
188 static int mod_open(void *instance, fr_schedule_t *sc, UNUSED CONF_SECTION *conf)
189 {
190  fr_listen_t *li;
191  proto_arp_t *inst = talloc_get_type_abort(instance, proto_arp_t);
192 
193  /*
194  * Build the #fr_listen_t. This describes the complete
195  * path, data takes from the socket to the decoder and
196  * back again.
197  */
198  li = talloc_zero(inst, fr_listen_t);
199  talloc_set_destructor(li, fr_io_listen_free);
200 
201  li->app = &proto_arp;
202  li->app_instance = instance;
203  li->server_cs = inst->server_cs;
204 
205  /*
206  * Set configurable parameters for message ring buffer.
207  */
208  li->default_message_size = FR_ARP_PACKET_SIZE;
209  li->num_messages = inst->num_messages;
210 
211  li->app_io = inst->app_io;
212  li->app_io_instance = inst->app_io_instance;
213  if (li->app_io->thread_inst_size) {
214  li->thread_instance = talloc_zero_array(NULL, uint8_t, li->app_io->thread_inst_size);
215  talloc_set_name(li->thread_instance, "proto_%s_thread_t", inst->app_io->name);
216  }
217 
218  /*
219  * Open the raw socket.
220  */
221  if (inst->app_io->open(li) < 0) {
222  talloc_free(li);
223  return -1;
224  }
225  fr_assert(li->fd >= 0);
226 
227  li->name = inst->app_io->get_name(li);
228 
229  /*
230  * Watch the directory for changes.
231  */
232  if (!fr_schedule_listen_add(sc, li)) {
233  talloc_free(li);
234  return -1;
235  }
236 
237  inst->listen = li; /* Probably won't need it, but doesn't hurt */
238  inst->sc = sc;
239 
240  return 0;
241 }

243 /** Instantiate the application
244  *
245  * Instantiate I/O and type submodules.
246  *
247  * @param[in] instance Ctx data for this application.
248  * @param[in] conf Listen section parsed to give us instance.
249  * @return
250  * - 0 on success.
251  * - -1 on failure.
252  */
253 static int mod_instantiate(void *instance, CONF_SECTION *conf)
254 {
255  proto_arp_t *inst = talloc_get_type_abort(instance, proto_arp_t);
256  fr_app_worker_t const *app_process;
257 
258  /*
259  * Instantiate the I/O module. But DON'T instantiate the
260  * work submodule. We leave that until later.
261  */
262  if (inst->app_io->instantiate &&
263  (inst->app_io->instantiate(inst->app_io_instance,
264  inst->app_io_conf) < 0)) {
265  cf_log_err(conf, "Instantiation failed for \"%s\"", inst->app_io->name);
266  return -1;
267  }
268 
269  if (!inst->num_messages) inst->num_messages = 256;
270 
271  FR_INTEGER_BOUND_CHECK("num_messages", inst->num_messages, >=, 32);
272  FR_INTEGER_BOUND_CHECK("num_messages", inst->num_messages, <=, 65535);
273 
274  /*
275  * Instantiate the ARP processor
276  */
277  app_process = (fr_app_worker_t const *)inst->app_process->module->common;
278  if (app_process->instantiate && (app_process->instantiate(inst->app_process->data,
279  inst->app_process->conf) < 0)) {
280  cf_log_err(conf, "Instantiation failed for \"%s\"", app_process->name);
281  return -1;
282  }
283 
284  /*
285  * Compile the processing sections.
286  */
287  if (app_process->compile_list) {
288  tmpl_rules_t parse_rules;
289 
290  memset(&parse_rules, 0, sizeof(parse_rules));
291  parse_rules.dict_def = dict_arp;
292 
293  if (virtual_server_compile_sections(inst->server_cs, app_process->compile_list, &parse_rules, inst->app_process->data) < 0) {
294  return -1;
295  }
296  }
297 
298  return 0;
299 }

302 /** Bootstrap the application
303  *
304  * Bootstrap I/O and type submodules.
305  *
306  * @param[in] instance Ctx data for this application.
307  * @param[in] conf Listen section parsed to give us instance.
308  * @return
309  * - 0 on success.
310  * - -1 on failure.
311  */
312 static int mod_bootstrap(void *instance, CONF_SECTION *conf)
313 {
314  proto_arp_t *inst = talloc_get_type_abort(instance, proto_arp_t);
315  dl_module_inst_t *parent_inst;
316  fr_app_worker_t const *app_process;
317 
318  /*
319  * Ensure that the server CONF_SECTION is always set.
320  */
321  inst->server_cs = cf_item_to_section(cf_parent(conf));
322  inst->cs = conf;
323 
324  parent_inst = cf_data_value(cf_data_find(inst->cs, dl_module_inst_t, "proto_arp"));
325  fr_assert(parent_inst);
326 
327  if (dl_module_instance(inst->cs, &inst->io_submodule, inst->cs,
328  parent_inst, "ethernet", DL_MODULE_TYPE_SUBMODULE) < 0) {
329  cf_log_perr(inst->cs, "Failed to load proto_arp_ethernet");
330  return -1;
331  }
332 
333  /*
334  * Parent dl_module_inst_t added in virtual_servers.c (listen_parse)
335  */
336  if (dl_module_instance(inst->cs, &inst->app_process, inst->cs,
337  parent_inst, "process", DL_MODULE_TYPE_SUBMODULE) < 0) {
338  cf_log_perr(inst->cs, "Failed to load proto_arp_process");
339  return -1;
340  }
341 
342  /*
343  * Bootstrap the I/O module
344  */
345  inst->app_io = (fr_app_io_t const *) inst->io_submodule->module->common;
346  inst->app_io_instance = inst->io_submodule->data;
347  inst->app_io_conf = conf;
348 
349  if (inst->app_io->bootstrap && (inst->app_io->bootstrap(inst->app_io_instance,
350  inst->app_io_conf) < 0)) {
351  cf_log_err(inst->app_io_conf, "Bootstrap failed for \"%s\"", inst->app_io->name);
352  return -1;
353  }
354 
355  /*
356  * Bootstrap the ARP processor
357  */
358  app_process = (fr_app_worker_t const *)(inst->app_process->module->common);
359  if (app_process->bootstrap && (app_process->bootstrap(inst->app_process->data,
360  inst->app_process->conf) < 0)) {
361  cf_log_err(conf, "Bootstrap failed for \"%s\"", app_process->name);
362  return -1;
363  }
364 
365  /*
366  * Register the processing sections with the
367  * module manager.
368  *
369  * We have to do this ourselves because we don't call
370  * fr_app_process_bootstrap(). That function requires a
371  * "type = ..." configuration, which isn't used for ARP.
372  */
373  if (app_process->compile_list) {
374  int j;
375  virtual_server_compile_t const *list = app_process->compile_list;
376 
377  for (j = 0; list[j].name != NULL; j++) {
378  if (list[j].name == CF_IDENT_ANY) continue;
379 
380  if (virtual_server_section_register(&list[j]) < 0) {
381  cf_log_err(conf, "Failed registering section name for %s",
382  app_process->name);
383  return -1;
384  }
385  }
386  }
387 
388  return 0;
389 }
