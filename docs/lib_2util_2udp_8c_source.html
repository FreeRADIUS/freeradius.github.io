33 /** Send a packet via a UDP socket.
34  *
35  * @param[in] socket we're reading from.
36  * @param[in] flags to pass to send(), or sendto()
37  * @param[in] data to data to send
38  * @param[in] data_len length of data to send
39  * @return
40  * - 0 on success.
41  * - -1 on failure.
42  */
43 int udp_send(fr_socket_t const *socket, int flags, void *data, size_t data_len)
44 {
45  int ret;
46 
47  if (unlikely(socket->proto != IPPROTO_UDP)) {
48  fr_strerror_printf("Invalid proto type %u", socket->proto);
49  return -1;
50  }
51 
52  if (flags & UDP_FLAGS_CONNECTED) {
53  ret = (send(socket->fd, data, data_len, 0) == (ssize_t)data_len) ? 0 : -1;
54  } else {
55  struct sockaddr_storage dst, src;
56  socklen_t sizeof_dst, sizeof_src;
57 
58  if (fr_ipaddr_to_sockaddr(&dst, &sizeof_dst,
59  &socket->inet.dst_ipaddr, socket->inet.dst_port) < 0) return -1;
60  if (fr_ipaddr_to_sockaddr(&src, &sizeof_src,
61  &socket->inet.src_ipaddr, socket->inet.src_port) < 0) return -1;
62 
63  ret = sendfromto(socket->fd, data, data_len, 0,
64  socket->inet.ifindex,
65  (struct sockaddr *)&src, sizeof_src,
66  (struct sockaddr *)&dst, sizeof_dst);
67  }
68 
69  if (ret < 0) fr_strerror_printf("udp_send failed: %s", fr_syserror(errno));
70 
71  return ret;
72 }

90 /** Peek at the header of a UDP packet.
91  *
92  * @param[in] sockfd we're reading from.
93  * @param[out] data pointer where data will be written
94  * @param[in] data_len length of data to read
95  * @param[in] flags for things
96  * @param[out] src_ipaddr of the packet.
97  * @param[out] src_port of the packet.
98  */
99 ssize_t udp_recv_peek(int sockfd, void *data, size_t data_len, int flags, fr_ipaddr_t *src_ipaddr, uint16_t *src_port)
100 {
101  ssize_t peeked;
102  struct sockaddr_storage src;
103  socklen_t sizeof_src = sizeof(src);
104 
105  if (!src_ipaddr || ((flags & UDP_FLAGS_CONNECTED) != 0)) {
106  peeked = recv(sockfd, data, data_len, MSG_PEEK);
107  if (peeked < 0) {
108  if ((errno == EAGAIN) || (errno == EINTR)) return 0;
109  return -1;
110  }
111 
112  return peeked;
113  }
114 
115  peeked = recvfrom(sockfd, data, data_len, MSG_PEEK, (struct sockaddr *)&src, &sizeof_src);
116  if (peeked < 0) {
117  if ((errno == EAGAIN) || (errno == EINTR)) return 0;
118  return -1;
119  }
120 
121  /*
122  * Convert AF. If unknown, discard packet.
123  */
124  if (fr_ipaddr_from_sockaddr(src_ipaddr, src_port, &src, sizeof_src) < 0) {
125  FR_DEBUG_STRERROR_PRINTF("Unknown address family");
126  (void) udp_recv_discard(sockfd);
127 
128  return -1;
129  }
130 
131  return peeked;
132 }

135 /** Read a UDP packet
136  *
137  * @param[in] sockfd we're reading from.
138  * @param[in] flags for things
139  * @param[out] socket_out Information about the src/dst address of the packet
140  * and the interface it was received on.
141  * @param[out] data pointer where data will be written
142  * @param[in] data_len length of data to read
143  * @param[out] when the packet was received.
144  * @return
145  * - > 0 on success (number of bytes read).
146  * - < 0 on failure.
147  */
148 ssize_t udp_recv(int sockfd, int flags,
149  fr_socket_t *socket_out, void *data, size_t data_len, fr_time_t *when)
150 {
151  int sock_flags = 0;
152  struct sockaddr_storage src;
153  struct sockaddr_storage dst;
154  socklen_t sizeof_src = sizeof(src);
155  socklen_t sizeof_dst = sizeof(dst);
156  ssize_t slen;
157 
158  if ((flags & UDP_FLAGS_PEEK) != 0) sock_flags |= MSG_PEEK;
159 
160  if (when) *when = 0;
161 
162  /*
163  * Always initialise the output socket structure
164  */
165  *socket_out = (fr_socket_t){
166  .fd = sockfd,
167  .proto = IPPROTO_UDP
168  };
169 
170  /*
171  * Connected sockets already know src/dst IP/port
172  */
173  if ((flags & UDP_FLAGS_CONNECTED) != 0) {
174  slen = recv(sockfd, data, data_len, sock_flags);
175  if (slen <= 0) goto done;
176 
177  goto done;
178  }
179 
180  /*
181  * Receive the packet. The OS will discard any data in the
182  * packet after "len" bytes.
183  */
184  slen = recvfromto(sockfd, data, data_len, sock_flags,
185  &socket_out->inet.ifindex,
186  (struct sockaddr *)&src, &sizeof_src,
187  (struct sockaddr *)&dst, &sizeof_dst,
188  when);
189  if (slen <= 0) goto done;
190 
191  if (fr_ipaddr_from_sockaddr(&socket_out->inet.src_ipaddr, &socket_out->inet.src_port, &src, sizeof_src) < 0) {
192  fr_strerror_printf_push("Failed converting src sockaddr to ipaddr");
193  return -1;
194  }
195  if (fr_ipaddr_from_sockaddr(&socket_out->inet.dst_ipaddr, &socket_out->inet.dst_port, &dst, sizeof_dst) < 0) {
196  fr_strerror_printf_push("Failed converting dst sockaddr to ipaddr");
197  return -1;
198  }
199 
200 done:
201  if (slen < 0) {
202  if ((errno == EWOULDBLOCK) || (errno == EAGAIN)) return 0;
203 
204  fr_strerror_printf("Failed reading socket: %s", fr_syserror(errno));
205  return slen;
206  }
207 
208  /*
209  * We didn't get it from the kernel
210  * so use our own time source.
211  */
212  if (when && !*when) *when = fr_time();
213 
214  return slen;
215 }
