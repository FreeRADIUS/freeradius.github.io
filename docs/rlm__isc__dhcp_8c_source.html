208 /** Refills the read buffer with one line from the file.
209  *
210  * This function also takes care of suppressing blank lines, and
211  * lines which only contain comments.
212  */
213 static int refill(rlm_isc_dhcp_tokenizer_t *state)
214 {
215  char *p;
216 
217  if (state->eof) return 0;
218 
219  /*
220  * We've run out of data to parse, reset to the start of
221  * the buffer.
222  */
223  if (!*state->ptr) state->ptr = state->buffer;
224 
225 redo:
226  state->lineno++;
227  state->line = state->ptr;
228 
229  if (!fgets(state->ptr, state->bufsize - (state->ptr - state->buffer), state->fp)) {
230  if (feof(state->fp)) {
231  state->eof = true;
232  return 0;
233  }
234 
235  return -1;
236  }
237 
238  /*
239  * Skip leading spaces
240  */
241  p = state->ptr;
242  fr_skip_whitespace(p);
243 
244  /*
245  * The line is all spaces, OR we've hit a comment. Go
246  * get more data.
247  */
248  if (!*p || (*p == '#')) goto redo;
249 
250  /*
251  * Point to the first non-space data.
252  */
253  state->ptr = p;
254 
255  return 1;
256 }

258 static int skip_spaces(rlm_isc_dhcp_tokenizer_t *state, char *p)
259 {
260  state->ptr = p;
261  char *start = p;
262 
263  fr_skip_whitespace(state->ptr);
264 
265  /*
266  * If we ran out of text on this line, re-fill the
267  * buffer. Note that refill() also takes care of
268  * suppressing blank lines and comments. refill() also
269  * takes care of skipping leading spaces, too.
270  */
271  if (!state->eof && !*state->ptr) {
272  int ret;
273 
274  state->ptr = start;
275 
276  ret = refill(state);
277  if (ret < 0) return -1;
278  }
279 
280  /*
281  * Set the semicolon flag as a "peek
282  * ahead", so that the various other
283  * parsers don't need to check it.
284  */
285  if (*state->ptr == ';') state->saw_semicolon = true;
286 
287  return 0;
288 }

292 /*
293  * ISC's double quoted strings allow all kinds of extra magic, so
294  * we re-implement string parsing yet again.
295  */
296 static int read_string(rlm_isc_dhcp_tokenizer_t *state)
297 {
298  char *p = state->ptr + 1;
299  char *q = state->string;
300 
301  while (true) {
302  if (!*p) {
303  fr_strerror_printf("unterminated string");
304  return -1;
305  }
306 
307  if (*p == '"') {
308  p++;
309  if (isspace((int) *p)) {
310  if (skip_spaces(state, p) < 0) return -1;
311  break;
312  }
313  }
314 
315  if ((size_t) (q - state->string) >= sizeof(state->string)) {
316  fr_strerror_printf("string is too long");
317  return -1;
318  }
319 
320  if (*p != '\\') {
321  *(q++) = *(p++);
322  continue;
323  }
324 
325  // @todo - all of ISC's string escapes, e.g. \x...
326  }
327 
328  *q = '\0';
329 
330  state->token = state->string;
331  state->token_len = (q - state->string);
332  return 1;
333 }

336 /** Reads one token into state->token
337  *
338  * Note that this function *destroys* the input buffer. So if
339  * you need to read two tokens, you have to save the first one
340  * somewhere *outside* of the input buffer.
341  */
342 static int read_token(rlm_isc_dhcp_tokenizer_t *state, fr_token_t hint, int semicolon, bool allow_rcbrace)
343 {
344  char *p;
345 
346 redo:
347  /*
348  * If the buffer is empty, re-fill it.
349  */
350  if (!*state->ptr) {
351  int ret;
352 
353  ret = refill(state);
354  if (ret < 0) return ret;
355 
356  if (ret == 0) {
357  if (!state->allow_eof) {
358  fr_strerror_printf("Unexpected EOF");
359  return -1;
360  }
361 
362  return 0;
363  }
364  }
365 
366  /*
367  * The previous token may have ended on a space
368  * or semi-colon. We skip those characters
369  * before looking for the next token.
370  */
371  while (isspace((int) *state->ptr) || (*state->ptr == ';') || (*state->ptr == ',')) state->ptr++;
372 
373  if (!*state->ptr) goto redo;
374 
375  /*
376  * Start looking for the next token from where we left
377  * off last time.
378  */
379  state->token = state->ptr;
380  state->saw_semicolon = false;
381 
382  /*
383  * Special-case quoted strings.
384  */
385  if (state->token[0] == '"') {
386  if (hint != T_DOUBLE_QUOTED_STRING) {
387  fr_strerror_printf("Unexpected '\"'");
388  return -1;
389  }
390 
391  return read_string(state);
392  }
393 
394  for (p = state->token; *p != '\0'; p++) {
395  /*
396  * "end of word" character. It might be allowed
397  * here, or it might not be.
398  */
399  if (*p == ';') {
400  if (semicolon == NO_SEMICOLON) {
401  fr_strerror_printf("unexpected ';'");
402  return -1;
403  }
404 
405  state->ptr = p;
406  state->saw_semicolon = true;
407  break;
408  }
409 
410  /*
411  * For lists of things and code definitions.
412  */
413  if (*p == ',') {
414  state->ptr = p;
415  break;
416  }
417 
418  /*
419  * Allow braces / equal as single character
420  * tokens if they're the first character we saw.
421  * Otherwise, the characters are "end of word"
422  * markers/
423  */
424  if ((*p == '{') || (*p == '}') || (*p == '=')) {
425  if (p == state->token) p++;
426 
427  state->ptr = p;
428  break;
429  }
430 
431  /*
432  * If we find a comment, we ignore everything
433  * until the end of the line.
434  */
435  if (*p == '#') {
436  *p = '\0';
437  state->ptr = p;
438 
439  /*
440  * Nothing here, get more text
441  */
442  if (state->token == state->ptr) goto redo;
443  break;
444  }
445 
446  /*
447  * Whitespace, we're done.
448  */
449  if (isspace((int) *p)) {
450  if (skip_spaces(state, p) < 0) return -1;
451  break;
452  }
453  }
454 
455  /*
456  * Protect the rest of the code from buffer overflows.
457  */
458  state->token_len = p - state->token;
459 
460  if (state->token_len == 0) {
461  fr_strerror_printf("FUCK");
462  return -1;
463  }
464 
465  if (state->token_len >= 256) {
466  fr_strerror_printf("token too large");
467  return -1;
468  }
469 
470  /*
471  * Double-check the token against what we were expected
472  * to read.
473  */
474  if (hint == T_LCBRACE) {
475  if (*state->token != '{') {
476  fr_strerror_printf("missing '{'");
477  return -1;
478  }
479 
480  if ((size_t) state->braces >= (sizeof(spaces) - 1)) {
481  fr_strerror_printf("sections are nested too deep");
482  return -1;
483  }
484 
485  state->braces++;
486  return 1;
487  }
488 
489  if (hint == T_RCBRACE) {
490  if (*state->token != '}') {
491  fr_strerror_printf("missing '}'");
492  return -1;
493  }
494 
495  state->braces--;
496  return 1;
497  }
498 
499  /*
500  * If we're inside of a section, we may also allow
501  * right-brace as the first keyword. In that case, it's
502  * the end of the enclosing section.
503  */
504  if (*state->token == '}') {
505  if (!allow_rcbrace) {
506  fr_strerror_printf("unexpected '}'");
507  return -1;
508  }
509 
510  state->braces--;
511  return 1;
512  }
513 
514  /*
515  * Don't return left brace if we were looking for a
516  * something else.
517  */
518  if ((hint == T_BARE_WORD) || (hint == T_DOUBLE_QUOTED_STRING)) {
519  if (*state->token == '{') {
520  fr_strerror_printf("unexpected '{'");
521  return -1;
522  }
523  }
524 
525 
526  return 1;
527 }

529 /** Recursively match subwords inside of a command string.
530  *
531  */
532 static int match_subword(rlm_isc_dhcp_tokenizer_t *state, char const *cmd, rlm_isc_dhcp_info_t *info)
533 {
534  int ret, type;
535  int semicolon = NO_SEMICOLON;
536  bool multi = false;
537  char *p;
538  char const *q;
539  char const *next;
540  char type_name[64];
541 
542  fr_skip_whitespace(cmd);
543 
544  if (!*cmd) return -1; /* internal error */
545 
546  /*
547  * Remember the next command.
548  */
549  next = q = cmd;
550  while (*next && !isspace((int) *next) && (*next != ',')) next++;
551  if (!*next) semicolon = YES_SEMICOLON;
552 
553  /*
554  * Matching an in-line word.
555  */
556  if (islower((int) *q)) {
557  ret = read_token(state, T_BARE_WORD, semicolon, false);
558  if (ret <= 0) return -1;
559 
560  /*
561  * Look for a verbatim word.
562  */
563  for (p = state->token; p < (state->token + state->token_len); p++, q++) {
564  if (*p != *q) {
565  fail:
566  fr_strerror_printf("Expected '%.*s', got unknown text '%.*s'",
567  state->token_len, state->token,
568  (int) (next - cmd), cmd);
569  return -1;
570  }
571  }
572 
573  /*
574  * Matched all of 'q', we're done.
575  */
576  if (!*q) {
577  DDEBUG("... WORD %.*s ", state->token_len, state->token);
578  return 1;
579  }
580 
581  /*
582  * Matched all of this word in 'q', but there are
583  * more words after this one.. Recurse.
584  */
585  if (isspace((int) *q)) {
586  return match_subword(state, next, info);
587  }
588 
589  /*
590  * Matched all of 'p', but there's more 'q'. Fail.
591  *
592  * e.g. got "foo", but expected "food".
593  */
594  goto fail;
595  }
596 
597  /*
598  * SECTION must be the last thing in the command
599  */
600  if (strcmp(q, "SECTION") == 0) {
601  if (q[7] != '\0') return -1; /* internal error */
602 
603  ret = read_token(state, T_LCBRACE, NO_SEMICOLON, false);
604  if (ret <= 0) return ret;
605 
606  ret = parse_section(state, info);
607  if (ret < 0) return ret;
608 
609  /*
610  * Empty sections are allowed.
611  */
612  return 2; /* SECTION */
613  }
614 
615  /*
616  * Uppercase words are INTEGER or STRING or IPADDR, which
617  * are FreeRADIUS data types.
618  *
619  * We copy the name here because some options allow for
620  * multiple fields.
621  */
622  p = type_name;
623  while (*q && !isspace((int) *q) && (*q != ',')) {
624  if ((p - type_name) >= (int) sizeof(type_name)) return -1; /* internal error */
625  *(p++) = tolower((int) *(q++));
626  }
627  *p = '\0';
628 
629  /*
630  * "fixed-address IPADDR," means it can take multiple IP
631  * addresses.
632  *
633  * @todo - pre-parse the field and save the strings
634  * somewhere, so that we can create info->argv of the
635  * right size. Or, just create an array of 2 by default,
636  * and then double it every time we run out... a little
637  * more work, but it doesn't involve further mangling the
638  * parser.
639  *
640  * We could likely just manually parse state->ptr, look
641  * until ';' or '\0', and count the words. That would
642  * work 99% of the time.
643  *
644  * @todo - We should also note that the ISC default is to
645  * allow hostnames, in which case it will add all IPs
646  * associated with that hostname, while we will add only
647  * one. That could likely be fixed, too.
648  */
649  if (*q == ',') {
650  if (q[1]) return -1; /* internal error */
651  multi = true;
652  semicolon = MAYBE_SEMICOLON;
653  }
654 
655  type = fr_table_value_by_str(fr_value_box_type_table, type_name, -1);
656  if (type < 0) {
657  fr_strerror_printf("unknown data type '%.*s'",
658  (int) (next - cmd), cmd);
659  return -1; /* internal error */
660  }
661 
662 redo_multi:
663  /*
664  * We were asked to parse a data type, so instead allow
665  * just about anything.
666  *
667  * @todo - if we get fancy, dynamically expand this, too.
668  * ISC doesn't support it, but we can.
669  */
670  ret = read_token(state, T_DOUBLE_QUOTED_STRING, semicolon, false);
671  if (ret <= 0) return ret;
672 
673  DDEBUG("... DATA %.*s ", state->token_len, state->token);
674 
675  /*
676  * BOOLs in ISC are "true", "false", or "ignore".
677  *
678  * Isn't that smart? "ignore" means "ignore this option
679  * as if it was commented out". So we do that.
680  *
681  * I sure wish I was smart enough to allow 3 values for a
682  * boolean data type.
683  */
684  if ((type == FR_TYPE_BOOL) && (state->token_len == 6) &&
685  (strcmp(state->token, "ignore") == 0)) {
686  talloc_free(info);
687  return 2;
688  }
689 
690  /*
691  * Parse the data to its final form.
692  */
693  info->argv[info->argc] = talloc_zero(info, fr_value_box_t);
694 
695  ret = fr_value_box_from_str(info, info->argv[info->argc], (fr_type_t *) &type, NULL,
696  state->token, state->token_len, 0, false);
697  if (ret < 0) return ret;
698 
699  info->argc++;
700 
701  if (multi) {
702  if (state->saw_semicolon) return 1;
703 
704  if (info->argc >= info->cmd->max_argc) {
705  fr_strerror_printf("Too many arguments (%d > %d) for command '%s'",
706  info->argc, info->cmd->max_argc, info->cmd->name);
707  return -1;
708  }
709 
710  goto redo_multi;
711  }

847  *
848  * "new-name" can also be SPACE.NAME
849  *
850  */
851 static int parse_option_definition(rlm_isc_dhcp_info_t *parent, rlm_isc_dhcp_tokenizer_t *state,
852  char *name)
853 {
854  int ret;
855  char *p;
856  fr_type_t type;
857  fr_dict_attr_t const *da, *root;
858  fr_value_box_t box;
859  fr_dict_attr_flags_t flags;
860 
861  p = strchr(name, '.');
862  if (p) {
863  fr_strerror_printf("cannot (yet) define options in spaces");
864  error:
865  talloc_free(name);
866  return -1;
867  }
868 
869  if (parent != state->inst->head) {
870  fr_strerror_printf("option definitions cannot be scoped");
871  goto error;
872  }
873 
874  /*
875  * Grab the integer code value.
876  */
877  ret = read_token(state, T_BARE_WORD, NO_SEMICOLON, false);
878  if (ret <= 0) {
879  error_ret:
880  talloc_free(name);
881  return ret;
882  }
883 
884  type = FR_TYPE_UINT32;
885  ret = fr_value_box_from_str(NULL, &box, &type, NULL,
886  state->token, state->token_len, 0, false);
887  if (ret < 0) goto error;
888 
889  /*
890  * Look for '='
891  */
892  ret = read_token(state, T_BARE_WORD, NO_SEMICOLON, false);
893  if (ret <= 0) goto error_ret;
894 
895  if ((state->token_len != 1) || (state->token[0] != '=')) {
896  fr_strerror_printf("expected '=' after code definition got '%.*s'", state->token_len, state->token);
897  goto error;
898  }
899 
900  memset(&flags, 0, sizeof(flags));
901 
902  /*
903  * Data type is:
904  *
905  * TYPE
906  * array of TYPE
907  * { TYPE, ... }
908  *
909  * Note that it also supports
910  *
911  * array of { TYPE, ... }
912  */
913  ret = read_token(state, T_BARE_WORD, MAYBE_SEMICOLON, false);
914  if (ret <= 0) goto error_ret;
915 
916 
917  if ((state->token_len == 5) && (memcmp(state->token, "array", 5) == 0)) {
918  flags.array = 1;
919 
920  ret = read_token(state, T_BARE_WORD, NO_SEMICOLON, false);
921  if (ret <= 0) goto error_ret;
922 
923  if (! ((state->token_len == 2) && (memcmp(state->token, "of", 2) == 0))) {
924  fr_strerror_printf("expected 'array of', not 'array %.*s'",
925  state->token_len, state->token);
926  goto error;
927  }
928 
929  /*
930  * Grab the next token. For now, it MUST have a semicolon
931  */
932  ret = read_token(state, T_BARE_WORD, YES_SEMICOLON, false);
933  if (ret <= 0) goto error_ret;
934  }
935 
936  if ((state->token_len == 1) && (state->token[0] == '{')) {
937  fr_strerror_printf("records are not supported in option definition");
938  goto error;
939  }
940 
941  /*
942  * This check is needed only because we have
943  * MAYBE_SEMICOLON above. That's in order to allow
944  * "array of.." statements to product an *array* error,
945  * not a *semicolon* error.
946  */
947  if (!state->saw_semicolon) {
948  fr_strerror_printf("expected ';'");
949  goto error;
950  }
951 
952  type = isc2fr_type(state);
953  if (type == FR_TYPE_INVALID) goto error;
954 
955  /*
956  * Now that we've parsed everything, look up the name.
957  * We forbid conflicts, but silently allow duplicates.
958  */
959  da = fr_dict_attr_by_name(dict_dhcpv4, name);
960  if (da &&
961  ((da->attr != box.vb_uint32) || (da->type != type))) {
962  fr_strerror_printf("cannot add different code / type for a pre-existing name '%s'", name);
963  goto error;
964  }
965 
966  /*
967  * And look it up by code, too.
968  *
969  * We allow multiple attributes of the same code / type,
970  * but with different names.
971  */
972  root = fr_dict_root(dict_dhcpv4);
973  da = fr_dict_attr_child_by_num(root, box.vb_uint32);
974  if (da && (da->type != type)) {
975  fr_strerror_printf("cannot add different type for a pre-existing code %d", box.vb_uint32);
976  goto error;
977  }
978 
979  /*
980  * Add it in. Note that this function adds it by name
981  * and by code. So we don't *necessarily* have to do the
982  * name/code checks above. But doing so allows us to
983  * have better error messages.
984  */
985  ret = fr_dict_attr_add(fr_dict_unconst(dict_dhcpv4), root, name, box.vb_uint32, type, &flags);
986  talloc_free(name);
987  if (ret < 0) return ret;
988 
989  /*
990  * Caller doesn't need to do anything else with the thing
991  * we just parsed.
992  */
993  return 2;
994 }

1067 /** Parse "option" command
1068  *
1069  * In any sane system, commands which do different things should
1070  * have different names. In this syntax, it's all miracles and
1071  * unicorns.
1072  *
1073  * option NAME VALUE ;
1074  * option new-name code new-code = definition ;
1075  *
1076  * option space name [ [ code width number ] [ length width number ] [ hash size number ] ] ;
1077  */
1078 static int parse_options(rlm_isc_dhcp_info_t *parent, rlm_isc_dhcp_tokenizer_t *state)
1079 {
1080  int ret, argc = 0;
1081  char *argv[2];
1082  char name[FR_DICT_ATTR_MAX_NAME_LEN + 5];
1083 
1084  /*
1085  * Since read_token() mashes the input buffer, we have to save the tokens somewhere.
1086  */
1087  while (!state->saw_semicolon) {
1088  ret = read_token(state, T_BARE_WORD, MAYBE_SEMICOLON, false);
1089  if (ret < 0) return ret;
1090 
1091  argv[argc++] = talloc_strndup(parent, state->token, state->token_len);
1092 
1093  if (argc == 2) break;
1094  }
1095 
1096  /*
1097  * Must have at least two arguments.
1098  */
1099  if (argc < 2) {
1100  fr_strerror_printf("unexpected ';'");
1101  return -1;
1102  }
1103 
1104  /*
1105  * Define an option space.
1106  */
1107  if (strcmp(argv[0], "space") == 0) {
1108  talloc_free(argv[0]);

1110  }
1111 
1112  /*
1113  * Look up the name. If the option is defined, then
1114  * parse the following options according to the data
1115  * type. Which MAY be a "struct" data type, or an
1116  * "array" data type.
1117  */
1118  if (state->saw_semicolon || (state->ptr[0] == ',')) {
1119  fr_dict_attr_t const *da;
1120

1122  if (da) {
1123  talloc_free(argv[0]);
1124  return parse_option(parent, state, da, argv[1]);
1125  }
1126 
1127  /*
1128  * @todo - nuke this extra step once we have dictionary.isc defined.
1129  */

1131  strlcpy(name + 5, argv[0], sizeof(name) - 5);
1132

1135  talloc_free(argv[0]);
1136  return parse_option(parent, state, da, argv[1]);

1142  *
1143  * It must be "option NAME code NUMBER = DEFINITION"
1144  */

1146  fr_strerror_printf("unknown option '%s'", argv[0]);
1147  talloc_free(argv[0]);
1148  talloc_free(argv[1]);
1149  return -1;
1150  }
1151 
1152  talloc_free(argv[1]);
1153  return parse_option_definition(parent, state, argv[0]);

1157 static int match_keyword(rlm_isc_dhcp_info_t *parent, rlm_isc_dhcp_tokenizer_t *state, rlm_isc_dhcp_cmd_t const *tokens, int num_tokens)
1158 {
1159  int start, end, half;
1160  int semicolon;
1161  int ret;
1162  char const *q = NULL;
1163  rlm_isc_dhcp_info_t *info;
1164 
1165  start = 0;
1166  end = num_tokens - 1;
1167  half = -1;
1168 
1169  /*
1170  * There are no super-short commands.
1171  */
1172  if (state->token_len < 4) goto unknown;
1173 
1174  /*
1175  * Walk over the input token, doing a binary search on
1176  * the token list.
1177  */
1178  while (start <= end) {
1179  half = (start + end) / 2;
1180 
1181  /*
1182  * Skips a function call, and is better for 99%
1183  * of the situations. Since there are no 1 or 2
1184  * character keywords, this always works.
1185  */
1186  ret = state->token[0] - tokens[half].name[0];
1187  if (ret != 0) goto recurse;
1188 
1189  ret = state->token[1] - tokens[half].name[1];
1190  if (ret != 0) goto recurse;
1191 
1192  ret = state->token[2] - tokens[half].name[2];
1193  if (ret != 0) goto recurse;
1194 
1195  /*
1196  * Compare all of the strings.
1197  */
1198  ret = strncmp(state->token, tokens[half].name, state->token_len);
1199 
1200  /*
1201  * Exact match. But maybe we have "foo" input,
1202  * and "food" command?
1203  */
1204  if (ret == 0) {
1205  char c = tokens[half].name[state->token_len];
1206 
1207  /*
1208  * The token exactly matches the command.
1209  */
1210  if (!c || isspace((int) c)) {
1211  q = &(tokens[half].name[state->token_len]);
1212  break;
1213  }
1214 
1215  /*
1216  * The token is "foo", but the command is
1217  * "food". Go search the lower half of
1218  * the command table.
1219  */
1220  ret = -1;
1221  }
1222 
1223  recurse:
1224  /*
1225  * Token is smaller than the command we checked,
1226  * go check the lower half of the table.
1227  */
1228  if (ret < 0) {
1229  end = half - 1;
1230  } else {
1231  start = half + 1;
1232  }
1233  }
1234 
1235  /*
1236  * Nothing matched, it's a failure.
1237  */
1238  if (!q) {
1239  unknown:
1240  fr_strerror_printf("unknown command '%.*s'", state->token_len, state->token);
1241  return -1;
1242  }
1243 
1244  fr_assert(half >= 0);
1245 
1246  /*
1247  * "option" has multiple parse possibilities, so we treat
1248  * it specially.
1249  */
1250  if (tokens[half].type == ISC_OPTION) {
1251  return parse_options(parent, state);
1252  }
1253 
1254  /*
1255  * Print out more warnings / errors in pedantic mode.
1256  */
1257  if (state->inst->pedantic && !tokens[half].parse) {
1258  if (tokens[half].type == ISC_INVALID) {
1259  ERROR("Command '%.*s' is not supported.",
1260  state->token_len, state->token);
1261  return -1;
1262  }
1263 
1264  /*
1265  * Print out WARNING messages only in debug mode.
1266  * We don't need to spam the main log file every
1267  * time the server starts.
1268  */
1269  if (DEBUG_ENABLED) {
1270  if (tokens[half].type == ISC_NOOP) {
1271  WARN("Command '%.*s' is not yet implemented.",
1272  state->token_len, state->token);
1273  }
1274 
1275  if (tokens[half].type == ISC_IGNORE) {
1276  WARN("Ignoring command '%.*s'. It is not relevant.",
1277  state->token_len, state->token);
1278  }
1279  }
1280  }
1281 
1282  semicolon = YES_SEMICOLON; /* default to always requiring this */
1283 
1284  DDEBUG("... TOKEN %.*s ", state->token_len, state->token);
1285 
1286  info = talloc_zero(parent, rlm_isc_dhcp_info_t);
1287  if (tokens[half].max_argc) {
1288  info->argv = talloc_zero_array(info, fr_value_box_t *, tokens[half].max_argc);
1289  }
1290 
1291  /*
1292  * Remember which command we parsed.
1293  */
1294  info->parent = parent;
1295  info->cmd = &tokens[half];
1296  info->last = &(info->child);
1297 
1298  /*
1299  * There's more to this command,
1300  * go parse that, too.
1301  */
1302  if (isspace((int) *q)) {
1303  if (state->saw_semicolon) goto unexpected;
1304 
1305  ret = match_subword(state, q, info);
1306  if (ret <= 0) return ret;
1307 
1308  /*
1309  * SUBSECTION must be at the end
1310  */
1311  if (ret == 2) semicolon = NO_SEMICOLON;
1312  }
1313 
1314  /*
1315  * *q must be empty at this point.
1316  */
1317  if ((semicolon == NO_SEMICOLON) && state->saw_semicolon) {
1318  unexpected:
1319  fr_strerror_printf("unexpected ';'");
1320  talloc_free(info);
1321  return -1;
1322  }
1323 
1324  if ((semicolon == YES_SEMICOLON) && !state->saw_semicolon) {
1325  fr_strerror_printf("missing ';'");
1326  talloc_free(info);
1327  return -1;
1328  }
1329 
1330  // @todo - print out the thing we parsed
1331 
1332  /*
1333  * Call the "parse" function which should do
1334  * validation, etc.
1335  */
1336  if (tokens[half].parse) {
1337  ret = tokens[half].parse(state, info);
1338  if (ret <= 0) {
1339  talloc_free(info);
1340  return ret;
1341  }
1342 
1343  /*
1344  * The parse function took care of
1345  * remembering the "info" structure. So
1346  * we don't add it to the parent list.
1347  *
1348  * This process ensures that for some
1349  * things (e.g. hosts and subnets), we
1350  * have have O(1) lookups instead of
1351  * O(N).
1352  *
1353  * It also means that the *rest* of the
1354  * commands we parse are in a relatively
1355  * tiny list, which makes the O(N)
1356  * processing of it fairly minor.
1357  */
1358  if (ret == 2) return 1;
1359  }
1360 
1361  /*
1362  * Add the parsed structure to the tail of the
1363  * current list. Note that this portion adds
1364  * only ONE command at a time.
1365  */
1366  *(parent->last) = info;
1367  parent->last = &(info->next);
1368 
1369  /*
1370  * It's a match, and it's OK.
1371  */
1372  return 1;
1373 }

1529 /** subnet IPADDR netmask MASK { ... }
1530  *
1531  */
1532 static int parse_subnet(rlm_isc_dhcp_tokenizer_t *state, rlm_isc_dhcp_info_t *info)
1533 {
1534  rlm_isc_dhcp_info_t *parent;
1535  int ret, bits;
1536  uint32_t netmask = info->argv[1]->vb_ipv4addr;
1537 
1538  /*
1539  * Check if argv[1] is a valid netmask
1540  */
1541  if (!(netmask & (~netmask >> 1))) {
1542  fr_strerror_printf("invalid netmask '%pV'", info->argv[1]);
1543  return -1;
1544  }
1545 
1546  /*
1547  * 192.168.2.1/16 is wrong.
1548  */
1549  if ((info->argv[0]->vb_ipv4addr & netmask) != info->argv[0]->vb_ipv4addr) {
1550  fr_strerror_printf("subnet '%pV' does not match netmask '%pV'", info->argv[0], info->argv[1]);
1551  return -1;
1552  }
1553 
1554  /*
1555  * Get number of bits set in netmask.
1556  */
1557  netmask = netmask - ((netmask >> 1) & 0x55555555);
1558  netmask = (netmask & 0x33333333) + ((netmask >> 2) & 0x33333333);
1559  netmask = (netmask + (netmask >> 4)) & 0x0F0F0F0F;
1560  netmask = netmask + (netmask >> 8);
1561  netmask = netmask + (netmask >> 16);
1562  bits = netmask & 0x0000003F;
1563 
1564  parent = info->parent;
1565  if (parent->subnets) {
1566  rlm_isc_dhcp_info_t *old;
1567 
1568  /*
1569  * Duplicate or overlapping "subnet" entries aren't allowd.
1570  */
1571  old = fr_trie_lookup(parent->subnets, &(info->argv[0]->vb_ipv4addr), bits);
1572  if (old) {
1573  fr_strerror_printf("subnet %pV netmask %pV' overlaps with existing subnet", info->argv[0], info->argv[1]);
1574  return -1;
1575 
1576  }
1577  } else {
1578  parent->subnets = fr_trie_alloc(parent);
1579  if (!parent->subnets) return -1;
1580  }
1581 
1582  /*
1583  * Add the subnet to the *parents* trie. That way when
1584  * we apply the parent, we can look up the subnet in its
1585  * trie. And avoid the O(N) issue of having thousands of
1586  * "subnet" entries in the parent->child list.
1587  */
1588 
1589  ret = fr_trie_insert(parent->subnets, &(info->argv[0]->vb_ipv4addr), bits, info);
1590  if (ret < 0) {
1591  fr_strerror_printf("Failed inserting 'subnet %pV netmask %pV' into trie",
1592  info->argv[0], info->argv[1]);
1593  return -1;
1594  }
1595 
1596  /*
1597  * @todo - if there's no 'option subnet-mask', add one
1598  * from the netmask given here. If there is an 'option
1599  * subnet-mask', then assume that the admin knows what
1600  * he's doing, and don't add one.
1601  */
1602 
1603  IDEBUG("%.*s subnet %pV netmask %pV { ... }", state->braces, spaces, info->argv[0], info->argv[1]);
1604 
1605  /*
1606  * We've remembered the subnet in the parent trie.
1607  * There's no need to add it to the child list here.
1608  */
1609  return 2;
1610 }

1805 /** Apply all rules *except* fixed IP
1806  *
1807  */
1808 static int apply(rlm_isc_dhcp_t const *inst, request_t *request, rlm_isc_dhcp_info_t *head)
1809 {
1810  int ret, child_ret;
1811  rlm_isc_dhcp_info_t *info;
1812  fr_pair_t *yiaddr;
1813 
1814  ret = 0;
1815  yiaddr = fr_pair_find_by_da(&request->reply_pairs, attr_your_ip_address);
1816 
1817  /*
1818  * First, apply any "host" options
1819  */
1820  if (head->hosts_by_ether) {
1821  rlm_isc_dhcp_info_t *host = NULL;
1822 
1823  host = get_host(request, head->hosts_by_ether, head->hosts_by_uid);
1824  if (!host) goto subnet;
1825 
1826  /*
1827  * Apply any options in the "host" section.
1828  */
1829  child_ret = apply(inst, request, host);
1830  if (child_ret < 0) return child_ret;
1831  if (child_ret == 1) ret = 1;
1832  }
1833 
1834 subnet:
1835  /*
1836  * Look in the trie for matching subnets, and apply any
1837  * subnets that match.
1838  */
1839  if (head->subnets && yiaddr) {
1840  info = fr_trie_lookup(head->subnets, &yiaddr->vp_ipv4addr, 32);
1841  if (!info) goto recurse;
1842 
1843  child_ret = apply(inst, request, info);
1844  if (child_ret < 0) return child_ret;
1845  if (child_ret == 1) ret = 1;
1846  }
1847 
1848 recurse:
1849  for (info = head->child; info != NULL; info = info->next) {
1850  if (!info->cmd) return -1; /* internal error */
1851 
1852  if (!info->cmd->apply) continue;
1853 
1854  child_ret = info->cmd->apply(inst, request, info);
1855  if (child_ret < 0) return child_ret;
1856  if (child_ret == 0) continue;
1857 
1858  ret = 1;
1859  }
1860 
1861  /*
1862  * Now that our children have added options, see if we
1863  * can add some, too.
1864  */
1865  if (head->options) {
1866  fr_pair_t *vp = NULL;
1867  fr_cursor_t option_cursor;
1868  fr_cursor_t reply_cursor;
1869 
1870  (void) fr_cursor_init(&reply_cursor, &request->reply_pairs);
1871  (void) fr_cursor_tail(&reply_cursor);
1872 
1873  /*
1874  * Walk over the input list, adding the options
1875  * only if they don't already exist in the reply.
1876  *
1877  * Yes, we know that this is O(R*P*D), complexity
1878  * is (reply VPs * option VPs * depth of options).
1879  *
1880  * Unless we make the code a lot smarter, this is
1881  * the best we can do. Since there are likely
1882  * only a few options (i.e. less than 100), this
1883  * is deemed to be OK.
1884  *
1885  * In order to fix this, we would need to sort
1886  * all of the options first, sort the reply VPs,
1887  * then walk over the reply VPs, and look at each
1888  * option list in turn, seeing if there are
1889  * options that match. This would likely be
1890  * faster.
1891  */
1892  for (vp = fr_cursor_init(&option_cursor, &head->options);
1893  vp != NULL;
1894  vp = fr_cursor_next(&option_cursor)) {
1895  fr_pair_t *reply;
1896 
1897  reply = fr_pair_find_by_da(&request->reply_pairs, vp->da);
1898  if (reply) continue;
1899 
1900  /*
1901  * Copy all of the same options to the
1902  * reply.
1903  */
1904  while (vp) {
1905  fr_pair_t *next, *copy;
1906 
1907  copy = fr_pair_copy(request->reply, vp);
1908  if (!copy) return -1;
1909 
1910  fr_cursor_append(&reply_cursor, copy);
1911  (void) fr_cursor_tail(&reply_cursor);
1912 
1913  next = fr_cursor_next_peek(&option_cursor);
1914  if (!next) break;
1915  if (next->da != vp->da) break;
1916 
1917  vp = fr_cursor_next(&option_cursor);
1918  }
1919  }
1920 
1921  /*
1922  * We applied some options.
1923  */
1924  ret = 1;
1925  }
1926 
1927  return ret;
1928 }

2065 {

2070  * We allow "group" inside of "group". But we don't
2071  * allow other sections to nest.
2072  */
2073  if (info->cmd->type != ISC_GROUP) {
2074  rlm_isc_dhcp_info_t *parent;
2075

2077  char const *q;
2078 
2079  if (!parent->cmd) break; /* top level */
2080 
2081  if (parent->cmd != info->cmd) continue;

2084  * Be gentle to the end user
2085  */
2086  q = parent->cmd->name;
2087  fr_skip_not_whitespace(q);
2088 
2089  fr_strerror_printf("cannot nest '%.*s' statements",
2090  (int) (q - parent->cmd->name), parent->cmd->name);

2092  }
2093  }

2097

2099  ret = read_token(state, T_BARE_WORD, YES_SEMICOLON, true);
2100  if (ret < 0) return ret;
2101  if (ret == 0) break;
2102 
2103  /*

2105  */
2106  if (*state->token == '}') break;

2109  if (ret < 0) return ret;

2111

2113  }
