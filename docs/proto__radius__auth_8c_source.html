153 /*
154  * Return a short string showing the terminal server, port
155  * and calling station ID.
156  */
157 static char *auth_name(char *buf, size_t buflen, request_t *request)
158 {
159  fr_pair_t *cli;
160  fr_pair_t *pair;
161  uint32_t port = 0; /* RFC 2865 NAS-Port is 4 bytes */
162  char const *tls = "";
163 
164  cli = fr_pair_find_by_da(&request->request_pairs, attr_calling_station_id);
165 
166  pair = fr_pair_find_by_da(&request->request_pairs, attr_nas_port);
167  if (pair != NULL) port = pair->vp_uint32;
168 
169  if (request->packet->socket.inet.dst_port == 0) tls = " via proxy to virtual server";
170 
171  snprintf(buf, buflen, "from client %.128s port %u%s%.128s%s",
172  request->client->shortname, port,
173  (cli ? " cli " : ""), (cli ? cli->vp_strvalue : ""),
174  tls);
175 
176  return buf;
177 }

179 /*
180  * Make sure user/pass are clean and then create an attribute
181  * which contains the log message.
182  */
183 static void CC_HINT(format (printf, 4, 5)) auth_message(proto_radius_auth_t const *inst,
184  request_t *request, bool goodpass, char const *fmt, ...)
185 {
186  va_list ap;
187 
188  bool logit;
189  char const *extra_msg = NULL;
190 
191  char password_buff[128];
192  char const *password_str = NULL;
193 
194  char buf[1024];
195  char extra[1024];
196  char *p;
197  char *msg;
198  fr_pair_t *username = NULL;
199  fr_pair_t *password = NULL;
200 
201  /*
202  * No logs? Then no logs.
203  */
204  if (!inst->log_auth) return;
205 
206  /*
207  * Get the correct username based on the configured value
208  */
209  if (!inst->log_stripped_names) {
210  username = fr_pair_find_by_da(&request->request_pairs, attr_user_name);
211  } else {
212  username = fr_pair_find_by_da(&request->request_pairs, attr_stripped_user_name);
213  if (!username) username = fr_pair_find_by_da(&request->request_pairs, attr_user_name);
214  }
215 
216  /*
217  * Clean up the password
218  */
219  if (inst->log_auth_badpass || inst->log_auth_goodpass) {
220  password = fr_pair_find_by_da(&request->request_pairs, attr_user_password);
221  if (!password) {
222  fr_pair_t *auth_type;
223 
224  auth_type = fr_pair_find_by_da(&request->control_pairs, attr_auth_type);
225  if (auth_type) {
226  snprintf(password_buff, sizeof(password_buff), "<via Auth-Type = %s>",
227  fr_dict_enum_name_by_value(auth_type->da, &auth_type->data));
228  password_str = password_buff;
229  } else {
230  password_str = "<no User-Password attribute>";
231  }
232  } else if (fr_pair_find_by_da(&request->request_pairs, attr_chap_password)) {
233  password_str = "<CHAP-Password>";
234  }
235  }
236 
237  if (goodpass) {
238  logit = inst->log_auth_goodpass;
239  extra_msg = inst->auth_goodpass_msg;
240  } else {
241  logit = inst->log_auth_badpass;
242  extra_msg = inst->auth_badpass_msg;
243  }
244 
245  if (extra_msg) {
246  extra[0] = ' ';
247  p = extra + 1;
248  if (xlat_eval(p, sizeof(extra) - 1, request, extra_msg, NULL, NULL) < 0) return;
249  } else {
250  *extra = '\0';
251  }
252 
253  /*
254  * Expand the input message
255  */
256  va_start(ap, fmt);
257  msg = fr_vasprintf(request, fmt, ap);
258  va_end(ap);
259 
260  RAUTH("%s: [%pV%s%pV] (%s)%s",
261  msg,
262  username ? &username->data : fr_box_strvalue("<no User-Name attribute>"),
263  logit ? "/" : "",
264  logit ? (password_str ? fr_box_strvalue(password_str) : &password->data) : fr_box_strvalue(""),
265  auth_name(buf, sizeof(buf), request),
266  extra);
267 
268  talloc_free(msg);
269 }

271 static unlang_action_t mod_process(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
272 {
273  proto_radius_auth_t const *inst = talloc_get_type_abort_const(mctx->instance, proto_radius_auth_t);
274  fr_pair_t *vp, *auth_type;
275  rlm_rcode_t rcode;
276  CONF_SECTION *unlang;
277  fr_dict_enum_t const *dv = NULL;
278  fr_cursor_t cursor;
279  void *instruction;
280 
281  REQUEST_VERIFY(request);
282 
283  switch (request->request_state) {
284  case REQUEST_INIT:
285  if (request->parent && RDEBUG_ENABLED) {
286  RDEBUG("Received %s ID %i", fr_packet_codes[request->packet->code], request->packet->id);
287  log_request_pair_list(L_DBG_LVL_1, request, request->request_pairs, "");
288  }
289 
290  request->component = "radius";
291 
292  if (!inst->unlang_access_request) {
293  REDEBUG("Failed to find 'recv Access-Request' section");
294  request->reply->code = FR_CODE_ACCESS_REJECT;
295  goto setup_send;
296  }
297 
298  /*
299  * Grab the VPS and data associated with the State attribute.
300  */
301  if (!request->parent) fr_state_to_request(inst->state_tree, request);
302 
303  /*
304  * Push the conf section into the unlang stack.
305  */
306  RDEBUG("Running 'recv Access-Request' from file %s", cf_filename(inst->recv_access_request));
307  if (unlang_interpret_push_instruction(request, inst->unlang_access_request,
308  RLM_MODULE_REJECT, UNLANG_TOP_FRAME) < 0) {
309  RETURN_MODULE_FAIL;
310  }
311 
312  request->request_state = REQUEST_RECV;
313  FALL_THROUGH;
314 
315  case REQUEST_RECV:
316  rcode = unlang_interpret(request);
317 
318  if (request->master_state == REQUEST_STOP_PROCESSING) {
319  *p_result = RLM_MODULE_HANDLED;
320  return UNLANG_ACTION_STOP_PROCESSING;
321  }
322 
323  if (rcode == RLM_MODULE_YIELD) return UNLANG_ACTION_YIELD;
324 
325  switch (rcode) {
326  case RLM_MODULE_NOOP:
327  case RLM_MODULE_NOTFOUND:
328  case RLM_MODULE_OK:
329  case RLM_MODULE_UPDATED:
330  break;
331 
332  case RLM_MODULE_HANDLED:
333  goto setup_send;
334 
335  case RLM_MODULE_FAIL:
336  case RLM_MODULE_INVALID:
337  case RLM_MODULE_REJECT:
338  case RLM_MODULE_DISALLOW:
339  default:
340  if ((vp = fr_pair_find_by_da(&request->request_pairs,
341  attr_module_failure_message)) != NULL) {
342  auth_message(inst, request, false, "Invalid user (%pV)", &vp->data);
343  } else {
344  auth_message(inst, request, false, "Invalid user");
345  }
346 
347  request->reply->code = FR_CODE_ACCESS_REJECT;
348  goto setup_send;
349  }
350 
351  /*
352  * Find Auth-Type, and complain if they have too many.
353  */
354  auth_type = NULL;
355  for (vp = fr_cursor_iter_by_da_init(&cursor, &request->control_pairs, attr_auth_type);
356  vp;
357  vp = fr_cursor_next(&cursor)) {
358  if (!auth_type) {
359  auth_type = vp;
360  continue;
361  }
362 
363  RWDEBUG("Ignoring extra %pP", vp);
364  }
365 
366  /*
367  * No Auth-Type, force it to reject.
368  */
369  if (!auth_type) {
370  /*
371  * Handle Service-Type = Authorize-Only
372  *
373  * If they want to reject the request,
374  * the "recv Access-Request" section
375  * should have returned reject.
376  */
377  vp = fr_pair_find_by_da(&request->request_pairs, attr_service_type);
378  if (vp && (vp->vp_uint32 == FR_SERVICE_TYPE_VALUE_AUTHORIZE_ONLY)) {
379  RDEBUG("Skipping authenticate as we have found %pP", vp);
380  request->reply->code = FR_CODE_ACCESS_ACCEPT;
381  goto setup_send;
382  }
383 
384  /*
385  * Allow for over-ride of reply code.
386  */
387  vp = fr_pair_find_by_da(&request->reply_pairs, attr_packet_type);
388  if (vp) {
389  request->reply->code = vp->vp_uint32;
390  goto setup_send;
391  }
392 
393  REDEBUG2("No Auth-Type available: rejecting the user.");
394  request->reply->code = FR_CODE_ACCESS_REJECT;
395  goto setup_send;
396  }
397 
398  /*
399  * Handle hard-coded Accept and Reject.
400  */
401  if (auth_type->vp_uint32 == FR_AUTH_TYPE_VALUE_ACCEPT) {
402  RDEBUG2("%pP, allowing user", auth_type);
403  request->reply->code = FR_CODE_ACCESS_ACCEPT;
404  goto setup_send;
405  }
406 
407  if (auth_type->vp_uint32 == FR_AUTH_TYPE_VALUE_REJECT) {
408  RDEBUG2("%pP, rejecting user", auth_type);
409  request->reply->code = FR_CODE_ACCESS_REJECT;
410  goto setup_send;
411  }
412 
413  /*
414  * Find the appropriate Auth-Type by name.
415  */
416  vp = auth_type;
417  dv = fr_dict_enum_by_value(vp->da, &vp->data);
418  if (!dv) {
419  REDEBUG2("Unknown Auth-Type %d found: rejecting the user", vp->vp_uint32);
420  request->reply->code = FR_CODE_ACCESS_REJECT;
421  goto setup_send;
422  }
423 
424  unlang = cf_section_find(request->server_cs, "authenticate", dv->name);
425  if (!unlang) {
426  REDEBUG2("No 'authenticate %s' section found: rejecting the user", dv->name);
427  request->reply->code = FR_CODE_ACCESS_REJECT;
428  goto setup_send;
429  }
430 
431  RDEBUG("Running 'authenticate %s' from file %s", cf_section_name2(unlang), cf_filename(unlang));
432  if (unlang_interpret_push_section(request, unlang, RLM_MODULE_NOTFOUND, UNLANG_TOP_FRAME) < 0) {
433  RETURN_MODULE_FAIL;
434  }
435 
436  request->request_state = REQUEST_PROCESS;
437  FALL_THROUGH;
438 
439  case REQUEST_PROCESS:
440  rcode = unlang_interpret(request);
441 
442  if (request->master_state == REQUEST_STOP_PROCESSING) {
443  RETURN_MODULE_HANDLED;
444  }
445 
446  if (rcode == RLM_MODULE_YIELD) return UNLANG_ACTION_YIELD;
447 
448  switch (rcode) {
449  /*
450  * An authentication module FAIL
451  * return code, or any return code that
452  * is not expected from authentication,
453  * is the same as an explicit REJECT!
454  */
455  case RLM_MODULE_FAIL:
456  case RLM_MODULE_INVALID:
457  case RLM_MODULE_NOOP:
458  case RLM_MODULE_NOTFOUND:
459  case RLM_MODULE_REJECT:
460  case RLM_MODULE_UPDATED:
461  case RLM_MODULE_DISALLOW:
462  default:
463  RDEBUG2("Failed to authenticate the user");
464  request->reply->code = FR_CODE_ACCESS_REJECT;
465 
466  /*
467  * Maybe the shared secret is wrong?
468  */
469  vp = fr_pair_find_by_da(&request->request_pairs, attr_user_password);
470  if (vp) {
471  if (RDEBUG_ENABLED2) {
472  uint8_t const *p;
473 
474  p = (uint8_t const *) vp->vp_strvalue;
475  while (*p) {
476  int size;
477 
478  size = fr_utf8_char(p, -1);
479  if (!size) {
480  RWDEBUG("Unprintable characters in the password. "
481  "Double-check the shared secret on the server "
482  "and the NAS!");
483  break;
484  }
485  p += size;
486  }
487  }
488  }
489  goto setup_send;
490 
491  case RLM_MODULE_OK:
492  if (!request->reply->code) request->reply->code = FR_CODE_ACCESS_ACCEPT;
493  break;
494 
495  case RLM_MODULE_HANDLED:
496  goto setup_send;
497  }
498 
499  /*
500  * Allow for over-ride of reply code.
501  */
502  vp = fr_pair_find_by_da(&request->reply_pairs, attr_packet_type);
503  if (vp) request->reply->code = vp->vp_uint32;
504 
505  setup_send:
506  if (!request->reply->code) {
507  vp = fr_pair_find_by_da(&request->reply_pairs, attr_packet_type);
508  if (vp) {
509  request->reply->code = vp->vp_uint32;
510  } else {
511  RDEBUG("No reply code was set. Forcing to Access-Reject");
512  request->reply->code = FR_CODE_ACCESS_REJECT;
513  }
514  }
515 
516  rerun_nak:
517  /*
518  * Access-Challenge packets require a State. If
519  * there is none, create one here. This is so
520  * that the State attribute is accessible in the
521  * "send Access-Challenge" section.
522  */
523  if ((request->reply->code == FR_CODE_ACCESS_CHALLENGE) &&
524  !(vp = fr_pair_find_by_da(&request->reply_pairs, attr_state))) {
525  uint8_t buffer[16];
526 
527  fr_rand_buffer(buffer, sizeof(buffer));
528 
529  MEM(pair_update_reply(&vp, attr_state) >= 0);
530  fr_pair_value_memdup(vp, buffer, sizeof(buffer), false);
531  }
532 
533  switch (request->reply->code) {
534  case FR_CODE_ACCESS_ACCEPT:
535  unlang = inst->send_access_accept;
536  instruction = inst->unlang_access_accept;
537  break;
538 
539  case FR_CODE_ACCESS_REJECT:
540  unlang = inst->send_access_reject;
541  instruction = inst->unlang_access_reject;
542  break;
543 
544  case FR_CODE_ACCESS_CHALLENGE:
545  unlang = inst->send_access_challenge;
546  instruction = inst->unlang_access_challenge;
547  break;
548 
549  default:
550  request->reply->code = FR_CODE_DO_NOT_RESPOND;
551  FALL_THROUGH;
552 
553  case FR_CODE_DO_NOT_RESPOND:
554  unlang = inst->send_do_not_respond;
555  instruction = inst->unlang_do_not_respond;
556  break;
557 
558  case FR_CODE_PROTOCOL_ERROR:
559  unlang = inst->send_protocol_error;
560  instruction = inst->unlang_protocol_error;
561  break;
562 
563  }
564 
565  if (!instruction) goto send_reply;
566 
567  RDEBUG("Running 'send %s' from file %s", cf_section_name2(unlang), cf_filename(unlang));
568  if (unlang_interpret_push_instruction(request, instruction, RLM_MODULE_NOOP, UNLANG_TOP_FRAME) < 0) {
569  RETURN_MODULE_FAIL;
570  }
571 
572  request->request_state = REQUEST_SEND;
573  FALL_THROUGH;
574 
575  case REQUEST_SEND:
576  rcode = unlang_interpret(request);
577 
578  if (request->master_state == REQUEST_STOP_PROCESSING) {
579  *p_result = RLM_MODULE_HANDLED;
580  return UNLANG_ACTION_STOP_PROCESSING;
581  }
582 
583  if (rcode == RLM_MODULE_YIELD) return UNLANG_ACTION_YIELD;
584 
585  switch (rcode) {
586  case RLM_MODULE_FAIL:
587  case RLM_MODULE_INVALID:
588  case RLM_MODULE_REJECT:
589  case RLM_MODULE_DISALLOW:
590  default:
591  /*
592  * If we over-ride an ACK with a NAK, run
593  * the NAK section.
594  */
595  if (request->reply->code != FR_CODE_ACCESS_REJECT) {
596  dv = fr_dict_enum_by_value(attr_packet_type, fr_box_uint32(request->reply->code));
597 
598  RWDEBUG("Failed running 'send %s', trying 'send Access-Reject'", dv ? dv->name : "???" );

601

608

610  }
611 
612  /*
613  * Else it was already an Access-Reject.
614  */
615  break;
616 
617  case RLM_MODULE_HANDLED:
618  case RLM_MODULE_NOOP:
619  case RLM_MODULE_NOTFOUND:
620  case RLM_MODULE_OK:
621  case RLM_MODULE_UPDATED:
622  /* reply code is already set */
623  break;
624  }
625 
626  send_reply:
627  request->reply->timestamp = fr_time();
628 
629  /*
630  * Save session-state list for Access-Challenge from a NAS.
631  * discard it for everything else.
632  */
633  if (!request->parent) {
634  if (request->reply->code == FR_CODE_ACCESS_CHALLENGE) {
635  /*
636  * We can't create a valid response
637  */
638  if (fr_request_to_state(inst->state_tree, request) < 0) {
639  request->reply->code = FR_CODE_DO_NOT_RESPOND;
640  RETURN_MODULE_OK;
641  }
642  } else {
643  fr_state_discard(inst->state_tree, request);
644  }
645  }
646 
647  /*
648  * Check for "do not respond".
649  */
650  if (request->reply->code == FR_CODE_DO_NOT_RESPOND) {
651  RDEBUG("Not sending reply to client.");
652  break;
653  }
654 
655  /*
656  * Write login OK message when we actually know
657  * we're sending an accept.
658  */
659  if (request->reply->code == FR_CODE_ACCESS_ACCEPT) {
660  vp = fr_pair_find_by_da(&request->request_pairs, attr_module_success_message);
661  if (vp){
662  auth_message(inst, request, true, "Login OK (%pV)", &vp->data);
663  } else {
664  auth_message(inst, request, true, "Login OK");
665  }
666  } else if (request->reply->code == FR_CODE_ACCESS_REJECT) {
667  vp = fr_pair_find_by_da(&request->request_pairs, attr_module_failure_message);
668  if (vp) {
669  auth_message(inst, request, false, "Login incorrect (%pV)", &vp->data);
670  } else {
671  auth_message(inst, request, false, "Login incorrect");
672  }
673  }
674  if (request->parent && RDEBUG_ENABLED) {
675  RDEBUG("Sending %s ID %i", fr_packet_codes[request->reply->code], request->reply->id);
676  log_request_pair_list(L_DBG_LVL_1, request, request->reply_pairs, "");
677  }
678  break;
679 
680  default:
681  RETURN_MODULE_FAIL;
682  }
683 
684  RETURN_MODULE_OK;
685 }
