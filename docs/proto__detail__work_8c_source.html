147 static ssize_t mod_read(fr_listen_t *li, void **packet_ctx, fr_time_t *recv_time_p, uint8_t *buffer, size_t buffer_len, size_t *leftover, uint32_t *priority, UNUSED bool *is_dup)
148 {
149  proto_detail_work_t const *inst = talloc_get_type_abort_const(li->app_io_instance, proto_detail_work_t);
150  proto_detail_work_thread_t *thread = talloc_get_type_abort(li->thread_instance, proto_detail_work_thread_t);
151 
152  ssize_t data_size;
153  size_t packet_len;
154  fr_detail_entry_t *track;
155  uint8_t *partial, *end, *next, *p, *record_end;
156  uint8_t *stopped_search;
157  off_t done_offset;
158 
159  fr_assert(*leftover < buffer_len);
160  fr_assert(thread->fd >= 0);
161 
162  MPRINT("AT COUNT %d offset %ld", thread->count, (long) thread->read_offset);
163 
164  /*
165  * Process retransmissions before anything else in the
166  * file.
167  */
168  track = fr_dlist_head(&thread->list);
169  if (track) {
170  fr_dlist_remove(&thread->list, track);
171 
172  /*
173  * Don't over-write "leftover" bytes!
174  */
175  if (*leftover) {
176  fr_assert(thread->leftover == 0);
177  if (!thread->leftover_buffer) MEM(thread->leftover_buffer = talloc_array(thread, uint8_t, buffer_len));
178 
179  memcpy(thread->leftover_buffer, buffer, *leftover);
180  thread->leftover = *leftover;
181  *leftover = 0;
182  }
183 
184  fr_assert(buffer_len >= track->packet_len);
185  memcpy(buffer, track->packet, track->packet_len);
186 
187  DEBUG("Retrying packet %d (retransmission %u)", track->id, track->retry.count);
188  *packet_ctx = track;
189  *recv_time_p = track->timestamp;
190  *priority = inst->parent->priority;
191  return track->packet_len;
192  }
193 
194  /*
195  * If we decide that we're closing, ignore everything
196  * else in the file. Someone extended the file on us
197  * without locking it first. So too bad for them.
198  */
199  if (thread->closing) {
200  if (inst->track_progress) thread->read_offset = lseek(thread->fd, 0, SEEK_END);
201  return 0;
202  }
203 
204  /*
205  * Once a socket is ready, the network side tries to read
206  * many packets. So if we want to stop it from reading,
207  * we have to check this ourselves.
208  */
209  if (thread->outstanding >= inst->max_outstanding) {
210  fr_assert(thread->paused);
211  return 0;
212  }
213 
214  /*
215  * If we've cached leftover data from the ring buffer,
216  * copy it back.
217  */
218  if (thread->leftover) {
219  fr_assert(*leftover == 0);
220  fr_assert(thread->leftover < buffer_len);
221 
222  memcpy(buffer, thread->leftover_buffer, thread->leftover);
223  *leftover = thread->leftover;
224  thread->leftover = 0;
225  }
226 
227  /*
228  * Seek to the current read offset.
229  */
230  (void) lseek(thread->fd, thread->read_offset, SEEK_SET);
231 
232  /*
233  * There will be "leftover" bytes left over in the buffer
234  * from any previous read. At the start of the file,
235  * "leftover" will be zero.
236  */
237  partial = buffer + *leftover;
238 
239  MPRINT("READ leftover %zd", *leftover);
240 
241  /*
242  * Try to read as much data as possible.
243  */
244  if (!thread->eof) {
245  size_t room;
246 
247  room = buffer_len - *leftover;
248 
249  data_size = read(thread->fd, partial, room);
250  if (data_size < 0) {
251  ERROR("proto_detail (%s): Failed reading file %s: %s",
252  thread->name, thread->filename_work, fr_syserror(errno));
253  return -1;
254  }
255 
256  MPRINT("GOT %zd bytes", data_size);
257 
258  /*
259  * Remember the read offset, and whether we got EOF.
260  */
261  thread->read_offset = lseek(thread->fd, 0, SEEK_CUR);
262 
263  /*
264  * Only set EOF if there's no more data in the buffer to manage.
265  */
266  thread->eof = (data_size == 0) || (thread->read_offset == thread->file_size) || ((size_t) data_size < room);
267  if (thread->eof) {
268  MPRINT("Set EOF data_size %ld vs room %ld", data_size, room);
269  MPRINT("Set EOF read %ld vs file %ld", (long) thread->read_offset, (long) thread->file_size);
270  }
271  end = partial + data_size;
272 
273  } else {
274  MPRINT("READ UNTIL EOF");
275  /*
276  * We didn't read any more data from the file,
277  * but there should be data left in the buffer.
278  */
279  fr_assert(*leftover > 0);
280  end = buffer + *leftover;
281  }
282 
283 redo:
284  next = NULL;
285  stopped_search = end;
286 
287  /*
288  * Look for "end of record" marker, starting from the
289  * beginning of the buffer.
290  *
291  * Note that all of the data MUST be printable, and raw
292  * LFs are forbidden in attribute contents.
293  */
294  fr_assert((buffer + thread->last_search) <= end);
295 
296  MPRINT("Starting search from offset %ld", thread->last_search);
297 
298  p = buffer + thread->last_search;
299  while (p < end) {
300  if (p[0] != '\n') {
301  p++;
302  continue;
303  }
304  if ((p + 1) == end) {
305  /*
306  * Remember the last LF, so if the next
307  * read starts with a LF, we can find the
308  * end of record marker.
309  */
310  stopped_search = p;
311  break; /* no more data */
312  }
313 
314  if (p[1] == '\n') {
315  p[0] = '\0';
316  p[1] = '\0';
317  next = p + 2;
318  stopped_search = next;
319  break;
320  }
321 
322  /*
323  * If we're not at EOF, and we're not at end of
324  * record, every line MUST have a leading tab.
325  */
326  if (p[1] != '\t') {
327  ERROR("proto_detail (%s): Malformed line found at offset %zu in file %s",
328  thread->name, (size_t)((p - buffer) + thread->header_offset),
329  thread->filename_work);
330  return -1;
331  }
332 
333  /*
334  * Smash the \n with zero, so that each line can
335  * be parsed individually.
336  */
337  p[0] = '\0';
338 
339  /*
340  * Skip the \n\t
341  */
342  if ((p + 2) >= end) {
343  stopped_search = p;
344  break;
345  }
346 
347  p += 2;
348 
349  /*
350  * Skip attribute name
351  */
352  while ((p < end) && !isspace(*p)) p++;
353 
354  /*
355  * Not enough room for " = ", skip this sanity
356  * check, and just search for a \n on the next
357  * round through the loop.
358  */
359  if ((end - p) < 3) {
360  stopped_search = p;
361  break;
362  }
363 
364  /*
365  * Check for " = ". If the line doesn't contain
366  * this, it's malformed.
367  */
368  if (memcmp(p, " = ", 3) != 0) {
369  ERROR("proto_detail (%s): Malformed line found at offset %zu: %.*s of file %s",
370  thread->name,
371  (size_t)((p - buffer) + thread->header_offset), (int) (end - p), p,
372  thread->filename_work);
373  return -1;
374  }
375 
376  /*
377  * Skip the " = ", and go back to the top of the
378  * loop where we check for the next \n.
379  */
380  p += 3;
381  }
382 
383  thread->last_search = (stopped_search - buffer);
384 
385  /*
386  * If there is a next record, remember how large this
387  * record is, and update "leftover" bytes.
388  */
389  if (next) {
390  packet_len = next - buffer;
391  *leftover = end - next;
392 
393  MPRINT("FOUND next at %zd, leftover is %zd", packet_len, *leftover);
394 
395  } else if (!thread->eof) {
396  if ((size_t) (end - buffer) == buffer_len) {
397  ERROR("proto_detail (%s): Too large entry (>%d bytes) found at offset %zu: %.*s of file %s",
398  thread->name, (int) buffer_len,
399  (size_t)((p - buffer) + thread->header_offset), (int) (end - p), p,
400  thread->filename_work);
401  return -1;
402  }
403 
404  /*
405  * We're not at EOF, and there is no "next"
406  * entry. Remember all of the leftover data in
407  * the buffer, and ask the caller to call us when
408  * there's more data.
409  */
410  *leftover = end - buffer;
411  MPRINT("Not at EOF, and no next. Leftover is %zd", *leftover);
412  return 0;
413 
414  } else {
415  /*
416  * Else we're at EOF, it's OK to not have an "end
417  * of record" marker. We just eat all of the
418  * remaining data.
419  */
420  packet_len = end - buffer;
421  *leftover = 0;
422 
423  MPRINT("NO end of record, but at EOF, found %zd leftover is 0", packet_len);
424  }
425 
426  /*
427  * Too big? Ignore it.
428  *
429  * @todo - skip the record, using memmove() etc.
430  */
431  if (packet_len > inst->parent->max_packet_size) {
432  DEBUG("Ignoring 'too large' entry at offset %zu of %s",
433  (size_t) thread->header_offset, thread->filename_work);
434  DEBUG("Entry size %lu is greater than allowed maximum %u",
435  packet_len, inst->parent->max_packet_size);
436  skip_record:
437  MPRINT("Skipping record");
438  if (next) {
439  memmove(buffer, next, (end - next));
440  data_size = (end - next);
441  *leftover = 0;
442  end = buffer + data_size;
443  thread->last_search = 0;
444 
445  /*
446  * No more data, we're done.
447  */
448  if (end == buffer) return 0;
449  goto redo;
450  }
451 
452  fr_assert(*leftover == 0);
453  goto done;
454  }
455 
456  /*
457  * Search for the "Timestamp" attribute. We overload
458  * that to track which entries have been used.
459  */
460  record_end = buffer + packet_len;
461  p = buffer;
462  done_offset = 0;
463 
464  while (p < record_end) {
465  if (*p != '\0') {
466  p++;
467  continue;
468  }
469 
470  p++;
471  if (p == record_end) break;
472 
473  if (((record_end - p) >= 5) &&
474  (memcmp(p, "\tDone", 5) == 0)) {
475  goto skip_record;
476  }
477 
478  if (((record_end - p) > 10) &&
479  (memcmp(p, "\tTimestamp", 10) == 0)) {
480  p++;
481  done_offset = thread->header_offset + (p - buffer);
482  }
483  }
484 
485  /*
486  * Allocate the tracking entry.
487  */
488  track = talloc_zero(thread, fr_detail_entry_t);
489  track->parent = thread;
490  track->timestamp = fr_time();
491  track->id = thread->count++;
492 
493  track->done_offset = done_offset;
494  if (inst->retransmit) {
495  track->packet = talloc_memdup(track, buffer, packet_len);
496  track->packet_len = packet_len;
497  }
498 
499  /*
500  * We've read one more packet.
501  */
502  thread->header_offset += packet_len;
503 
504  *packet_ctx = track;
505  *recv_time_p = track->timestamp;
506  *priority = inst->parent->priority;
507 
508 done:
509  /*
510  * If we're at EOF, mark us as "closing".
511  */
512  if (thread->eof) {
513  fr_assert(!thread->closing);
514  thread->closing = (*leftover == 0);
515  MPRINT("AT EOF, BUT CLOSING %d", thread->closing);
516  }
517 
518  thread->outstanding++;
519 
520  /*
521  * Pause reading until such time as we need more packets.
522  */
523  if (!thread->paused && (thread->outstanding >= inst->max_outstanding)) {
524  (void) fr_event_filter_update(thread->el, thread->fd, FR_EVENT_FILTER_IO, pause_read);
525  thread->paused = true;
526 
527  /*
528  * Back up so that read() knows there's more data.
529  */
530  if (*leftover) (void) lseek(thread->fd, thread->read_offset - 1, SEEK_SET);
531  }
532 
533  /*
534  * Next time, start searching from the start of the
535  * buffer.
536  */
537  thread->last_search = 0;
538 
539  MPRINT("Returning NUM %u - %.*s", thread->outstanding, (int) packet_len, buffer);
540  return packet_len;
541 }

544 static void work_retransmit(UNUSED fr_event_list_t *el, UNUSED fr_time_t now, void *uctx)
545 {
546  fr_detail_entry_t *track = talloc_get_type_abort(uctx, fr_detail_entry_t);
547  proto_detail_work_thread_t *thread = track->parent;
548 
549  DEBUG("%s - retransmitting packet %d", thread->name, track->id);
550  track->retry.count++;
551 
552  fr_dlist_insert_tail(&thread->list, track);
553 
554  if (thread->paused && (thread->outstanding < thread->inst->max_outstanding)) {
555  (void) fr_event_filter_update(thread->el, thread->fd, FR_EVENT_FILTER_IO, resume_read);
556  thread->paused = false;
557  }
558 
559  fr_assert(thread->fd >= 0);
560 
561  /*
562  * Seek to the START of the file, so that the FD will
563  * always return ready.
564  *
565  * The mod_read() function will take care of seeking to
566  * the correct read offset.
567  */
568  (void) lseek(thread->fd, 0, SEEK_SET);
569 
570 #ifdef __linux__
571  fr_network_listen_read(thread->nr, thread->listen);
572 #endif
573 }

575 static ssize_t mod_write(fr_listen_t *li, void *packet_ctx, UNUSED fr_time_t request_time,
576  uint8_t *buffer, size_t buffer_len, UNUSED size_t written)
577 {
578  proto_detail_work_t const *inst = talloc_get_type_abort_const(li->app_io_instance, proto_detail_work_t);
579  proto_detail_work_thread_t *thread = talloc_get_type_abort(li->thread_instance, proto_detail_work_thread_t);
580  fr_detail_entry_t *track = packet_ctx;
581 
582  if (buffer_len < 1) return -1;
583 
584  fr_assert(thread->outstanding > 0);
585  fr_assert(thread->fd >= 0);
586 
587  if (!buffer[0]) {
588  if (track->retry.start == 0) {
589  fr_retry_init(&track->retry, fr_time(), &inst->retry_config);
590  } else {
591  fr_retry_state_t state;
592 
593  state = fr_retry_next(&track->retry, fr_time());
594  if (state == FR_RETRY_MRC) {
595  DEBUG("%s - packet %d failed after %u retransmissions",
596  thread->name, track->id, track->retry.count);
597  goto fail;
598 
599  }
600 
601  if (state == FR_RETRY_MRD) {
602  DEBUG("%s - packet %d failed after %u seconds",
603  thread->name, track->id,
604  (unsigned int) fr_time_delta_to_sec(inst->retry_config.mrd));
605  goto fail;
606  }
607  }
608 
609  DEBUG("%s - packet %d failed during processing. Will retransmit in %d.%06ds",
610  thread->name, track->id, (int) (track->retry.rt / NSEC), (int) ((track->retry.rt % NSEC) / 1000));
611 
612  if (fr_event_timer_at(thread, thread->el, &track->ev,
613  track->retry.next, work_retransmit, track) < 0) {
614  ERROR("%s - Failed inserting retransmission timeout", thread->name);
615  fail:
616  if (inst->track_progress && (track->done_offset > 0)) goto mark_done;
617  goto free_track;
618  }
619 
620  if (!thread->paused && (thread->outstanding >= inst->max_outstanding)) {
621  (void) fr_event_filter_update(thread->el, thread->fd, FR_EVENT_FILTER_IO, pause_read);
622  thread->paused = true;
623  }
624 
625  return 1;
626 
627  } else if (inst->track_progress && (track->done_offset > 0)) {
628  mark_done:
629  /*
630  * Seek to the entry, mark it as done, and then seek to
631  * the point in the file where we were reading from.
632  */
633  (void) lseek(thread->fd, track->done_offset, SEEK_SET);
634  if (write(thread->fd, "Done", 4) < 0) {
635  ERROR("%s - Failed marking entry as done: %s", thread->name, fr_syserror(errno));
636  }
637  (void) lseek(thread->fd, thread->read_offset, SEEK_SET);
638  }
639 
640 free_track:
641  thread->outstanding--;
642 
643  /*
644  * If we need to read some more packet, let's do so.
645  */
646  if (thread->paused && (thread->outstanding < inst->max_outstanding)) {
647  (void) fr_event_filter_update(thread->el, thread->fd, FR_EVENT_FILTER_IO, resume_read);
648  thread->paused = false;
649 
650  /*
651  * And seek to the start of the file, so that the
652  * reader gets activated again. The reader will
653  * lseek() to the read offset, so this seek is fine.
654  */
655  (void) lseek(thread->fd, 0, SEEK_SET);
656  }
657 
658  /*
659  * @todo - add a used / free pool for these
660  */
661  talloc_free(track);
662 
663  /*
664  * Close the socket if we're at EOF, and there are no
665  * outstanding replies to deal with.
666  */
667  if (thread->closing && !thread->outstanding) {
668  MPRINT("WRITE ASKED TO CLOSE");
669  return 0;
670  }
671 
672  MPRINT("WRITE RETURN B %ld", buffer_len);
673  return buffer_len;
674 }

677  *
678  */
679 static int mod_open(fr_listen_t *li)
680 {
681  proto_detail_work_t const *inst = talloc_get_type_abort_const(li->app_io_instance, proto_detail_work_t);
682  proto_detail_work_thread_t *thread = talloc_get_type_abort(li->thread_instance, proto_detail_work_thread_t);
683 
684  fr_dlist_init(&thread->list, fr_detail_entry_t, entry);
685 
686  /*
687  * Open the file if we haven't already been given one.
688  */
689  if (thread->fd < 0) {
690  thread->filename_work = talloc_strdup(inst, inst->filename_work);
691 
692  li->fd = thread->fd = open(thread->filename_work, inst->mode);
693  if (thread->fd < 0) {
694  cf_log_err(inst->cs, "Failed opening %s: %s", thread->filename_work, fr_syserror(errno));
695  return -1;
696  }
697  }
698 
699  /*
700  * If we're tracking progress, learn where the EOF is.
701  */
702  if (inst->track_progress) {
703  struct stat buf;
704 
705  if (fstat(thread->fd, &buf) < 0) {
706  cf_log_err(inst->cs, "Failed examining %s: %s", thread->filename_work, fr_syserror(errno));
707  return -1;
708  }
709 
710  thread->file_size = buf.st_size;
711  } else {
712  /*
713  * Avoid triggering erroneous EOF.
714  */
715  thread->file_size = 1;
716  }
717 
718  fr_assert(thread->name == NULL);
719  fr_assert(thread->filename_work != NULL);
720  thread->name = talloc_typed_asprintf(thread, "detail_work from filename %s", thread->filename_work);
721 
722  return 0;
723 }

726 static int mod_close_internal(proto_detail_work_thread_t *thread)
727 {
728  /*
729  * One less worker... we check for "0" because of the
730  * hacks in proto_detail which let us start up with
731  * "transport = work" for debugging purposes.
732  */
733  if (thread->file_parent) {
734  pthread_mutex_lock(&thread->file_parent->worker_mutex);
735  if (thread->file_parent->num_workers > 0) thread->file_parent->num_workers--;
736  pthread_mutex_unlock(&thread->file_parent->worker_mutex);
737  }
738 
739  DEBUG("Closing and deleting detail worker file %s", thread->name);
740 
741 #ifdef NOTE_REVOKE
742  fr_event_fd_delete(thread->el, thread->fd, FR_EVENT_FILTER_VNODE);
743 #endif
744  fr_event_fd_delete(thread->el, thread->fd, FR_EVENT_FILTER_IO);
745 
746  unlink(thread->filename_work);
747 
748  close(thread->fd);
749  thread->fd = -1;
750 
751  /*
752  * If we've been spawned from proto_detail_file, clean
753  * ourselves up, including our listener.
754  */
755  if (thread->listen) {
756  talloc_free(thread->listen);
757  }
758 
759  return 0;
760 }
