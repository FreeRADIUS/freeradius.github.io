37 static ssize_t op_to_token(fr_token_t *token, char const *op, size_t oplen)
38 {
39  char const *p = op;
40 
41  if (!token || !op || !oplen) return 0;
42 
43  switch (*p) {
44  default:
45  fr_strerror_printf("Invalid text. Expected comparison operator");
46  return -(p - op);
47 
48  case '!':
49  if (oplen < 2) goto invalid_operator;
50 
51  if (p[1] == '=') {
52  *token = T_OP_NE;
53  p += 2;
54 
55 #ifdef HAVE_REGEX
56  } else if (p[1] == '~') {
57  *token = T_OP_REG_NE;
58  p += 2;
59 #endif
60 
61  } else if (p[1] == '*') {
62  *token = T_OP_CMP_FALSE;
63  p += 2;
64 
65  } else {
66  invalid_operator:
67  fr_strerror_printf("Invalid operator");
68  return -(p - op);
69  }
70  break;
71 
72  case '=':
73  /*
74  * Bare '=' is allowed.
75  */
76  if (oplen == 1) {
77  *token = T_OP_EQ;
78  p++;
79  break;
80  }
81 
82  if (oplen < 2) goto invalid_operator;
83 
84  if (p[1] == '=') {
85  *token = T_OP_CMP_EQ;
86  p += 2;
87 
88 #ifdef HAVE_REGEX
89  } else if (p[1] == '~') {
90  *token = T_OP_REG_EQ;
91  p += 2;
92 #endif
93 
94  } else if (p[1] == '*') {
95  *token = T_OP_CMP_TRUE;
96  p += 2;
97 
98  } else {
99  /*
100  * Ignore whatever is after the '=' sign.
101  */
102  *token = T_OP_EQ;
103  p++;
104  }
105  break;
106 
107  case '<':
108  if ((oplen > 1) && (p[1] == '=')) {
109  *token = T_OP_LE;
110  p += 2;
111 
112  } else {
113  *token = T_OP_LT;
114  p++;
115  }
116  break;
117 
118  case '>':
119  if ((oplen > 1) && (p[1] == '=')) {
120  *token = T_OP_GE;
121  p += 2;
122 
123  } else {
124  *token = T_OP_GT;
125  p++;
126  }
127  break;
128  }
129 
130  return p - op;
131 }

172 /** Allocate one fr_pair_t from a string, and add it to the pair_ctx cursor.
173  *
174  * @param[in,out] pair_ctx the parsing context
175  * @param[in] start Where to create relative error offsets in relation to.
176  * @param in String to parse
177  * @param inlen length of string to parse
178  * @return
179  * - <= 0 on error (offset as negative integer)
180  * - > 0 on success (number of bytes parsed).
181  */
182 static ssize_t fr_pair_afrom_str(fr_pair_ctx_t *pair_ctx, char const *start, char const *in, size_t inlen)
183 {
184  char const *end = in + inlen;
185  char const *p = in;
186  ssize_t slen;
187  fr_dict_attr_t const *da;
188  char quote;
189  char const *value;
190  size_t value_len;
191  fr_pair_t *vp;
192  fr_token_t op;
193 
194  slen = fr_dict_attr_by_name_substr(NULL, &da, pair_ctx->parent->dict, &FR_SBUFF_IN(p, end));
195  if (slen <= 0) return slen - (in - start);
196 
197  if (da->parent != pair_ctx->parent) {
198  fr_strerror_printf("Unexpected attribute %s is not a child of %s",
199  da->name, pair_ctx->parent->name);
200  return -(in - start);
201  }
202 
203  p = in + slen;
204  if (p >= end) {
205  fr_strerror_printf("Attribute name overflows the input buffer");
206  return -(in - start);
207  }
208 
209  while ((isspace((int) *p)) && (p < end)) p++;
210 
211  if (p >= end) {
212  fr_strerror_printf("No operator found in the input buffer");
213  return -(p - start);
214  }
215 
216  /*
217  * For now, the only allowed operator is equals.
218  */
219  slen = op_to_token(&op, p, (end - p));
220  if (slen <= 0) {
221  fr_strerror_printf("Syntax error: expected '='");
222  return slen - -(p - start);
223  }
224  p += slen;
225 
226  while ((isspace((int) *p)) && (p < end)) p++;
227 
228  if (p >= end) {
229  fr_strerror_printf("No value found in the input buffer");
230  return -(p - start);
231  }
232 
233  if (*p == '`') {
234  fr_strerror_printf("Invalid string quotation");
235  return -(p - start);
236  }
237 
238  if ((*p == '"') || (*p == '\'')) {
239  quote = *p;
240  value = p + 1;
241 
242  slen = fr_skip_string(p, end);
243  if (slen <= 0) return slen - (p - start);
244  p += slen;
245  value_len = slen - 2; /* account for two "" */
246 
247  } else {
248  quote = 0;
249  value = p;
250 
251  /*
252  * Skip bare words, but end at comma or end-of-buffer.
253  */
254  while (!isspace((int) *p) && (*p != ',') && (p < end)) p++;
255 
256  value_len = p - value;
257  }
258 
259  if (p > end) {
260  fr_strerror_printf("Value overflows the input buffer");
261  return -(p - start);
262  }
263 
264  vp = fr_pair_afrom_fields(pair_ctx->ctx, da, op, value, value_len, quote);
265  if (!vp) return -(in - start);
266 
267  fr_cursor_append(pair_ctx->cursor, vp);
268 
269  return p - start;
270 }

272 /** Set a new DA context based on the input string
273  *
274  * @param[in,out] pair_ctx the parsing context
275  * @param[in] in String to parse
276  * @param[in] inlen length of string to parse
277  * @return
278  * - <= 0 on error (offset as negative integer)
279  * - > 0 on success (number of bytes parsed).
280  *
281  * pair_ctx->da is set to the new parsing context.
282  *
283  * @todo - allow for child contexts, so that we can parse TLVs into vp->vp_children.
284  * This change requires nested cursors, but not necessarily nested contexts.
285  * We probably want to have a `fr_dlist_t` of pair_ctx, and we always
286  * operate on the last one. When we change contexts to an attributes parent,
287  * we also change pair_ctx to a parent context. This is like the da stack,
288  * but with child cursors, too.
289  *
290  * We also want to emulate the previous behavior of group attributes based
291  * on parent, and increasing child_num. i.e. if we're parsing a series of
292  * "attr-foo = bar", then we watch the parent context, and create a new
293  * parent VP if this child has a SMALLER attribute number than the previous
294  * child. This allows the previous configurations to "just work".
295  */
296 static ssize_t fr_pair_ctx_set(fr_pair_ctx_t *pair_ctx, char const *in, size_t inlen)
297 {
298  char const *end = in + inlen;
299  char const *p = in;
300  ssize_t slen;
301  fr_dict_attr_t const *da;
302  fr_dict_attr_t const *parent = pair_ctx->parent;
303 
304  /*
305  * Parse the attribute name.
306  */
307  while (p < end) {
308  slen = fr_dict_attr_by_name_substr(NULL, &da, parent->dict, &FR_SBUFF_IN(p, end));
309  if (slen <= 0) return slen - (p - in);
310 
311  if (da->parent != parent) {
312  fr_strerror_printf("Unexpected attribute %s is not a child of %s",
313  da->name, parent->name);
314  return -(p - in);
315  }
316 
317  if ((p + slen) > end) {
318  fr_strerror_printf("Attribute name overflows the input buffer");
319  return -(p - in);
320  }
321 
322  /*
323  * Check for ending conditions.
324  */
325  p += slen;
326  parent = da;
327 
328  if ((p >= end) || (*p == ',')) {
329  break;
330  }
331 
332  /*
333  * We now MUST have FOO.BAR
334  */
335  if (*p != '.') {
336  fr_strerror_printf("Unexpected text after attribute");
337  return -(p - in);
338  }
339  p++;
340  }
341 
342  pair_ctx->parent = parent;
343  return p - in;
344 }

347 /** Parse a pair context from a string.
348  *
349  * @param pair_ctx the parsing context
350  * @param in String to parse
351  * @param inlen length of string to parse
352  * @return
353  * - <= 0 on error (offset as negative integer)
354  * - > 0 on success (number of bytes parsed).
355  *
356  * This function will parse fr_pair_ts, or context changes, up to
357  * end of string, or a trailing ','. The caller is responsible for
358  * parsing the comma.
359  *
360  * It accepts the following syntax:
361  *
362  * - Attribute = value
363  * * reset to a new top-level context according to the parent of Attribute
364  * * parse the attribute and the value
365  *
366  * - .Attribute = value
367  * * parse the attribute and the value in the CURRENT top-level context
368  *
369  * - .Attribute
370  * * reset to a new context according to Attribute, relative to the current context
371  *
372  * - ..Attribute
373  * * reset to a new context according to Attribute, relative to the current context
374  * * more '.' will walk back up the context tree.
375  *
376  * - Attribute
377  * * reset to a new top-level context according to Attribute
378  *
379  */
380 ssize_t fr_pair_ctx_afrom_str(fr_pair_ctx_t *pair_ctx, char const *in, size_t inlen)
381 {
382  char const *end = in + inlen;
383  char const *p = in;
384  ssize_t slen;
385  fr_dict_attr_t const *da;
386 
387  while (isspace((int) *p) && (p < end)) p++;
388  if (p >= end) return end - in;
389 
390  /*
391  * There may be one or more leading '.'
392  */
393  if (*p == '.') {
394 
395  /*
396  * .ATTRIBUTE = VALUE
397  */
398  if (p[1] != '.') {
399  p++;
400  return fr_pair_afrom_str(pair_ctx, in, p, end - p);
401  }
402 
403  /*
404  * '.' is our parent.
405  */
406  da = pair_ctx->parent;
407 
408  /*
409  * Loop until we find the end of the '.', resetting parent each time.
410  */
411  while (p < end) {
412  if (*p == '.') {
413  pair_ctx->parent = da;
414  da = da->parent;
415  p++;
416  continue;
417  }
418 
419  /*
420  * Comma is the end of a reference.
421  */
422  if (*p == ',') {
423  return p - in;
424  }
425 
426  /*
427  * .FOO, must be a reference.
428  */
429  break;
430  }
431 
432  if (p == end) return p - in;
433 
434  } else {
435  /*
436  * No leading '.', the reference MUST be from the root.
437  */
438  pair_ctx->parent = fr_dict_root(pair_ctx->parent->dict);
439 
440  /*
441  * We allow a leaf OR a reference here.
442  */
443  slen = fr_dict_attr_by_name_substr(NULL, &da, pair_ctx->parent->dict, &FR_SBUFF_IN(p, end));
444  if (slen <= 0) return slen - (p - in);
445 
446  /*
447  * Structural types do not have values. So a
448  * bare "Foo-Group" string MUST be changing the
449  * reference to that da.
450  */
451  switch (da->type) {
452  case FR_TYPE_GROUP:
453  case FR_TYPE_TLV:
454  case FR_TYPE_STRUCT:
455  pair_ctx->parent = da;
456  p += slen;
457 
458  if ((p >= end) || (*p == ',')) return p - in;
459 
460  if (*p != '.') {
461  fr_strerror_printf("Unexpected text");
462  return - (p - in);
463  }
464 
465  p++;
466  break;
467 
468  default:
469  /*
470  * Non-structural types MUST have values.
471  * So change the parent to its parent,
472  * and parse the leaf pair.
473  */
474  pair_ctx->parent = da->parent;
475  return fr_pair_afrom_str(pair_ctx, in, p, end - in);
476  }
477  }
478 
479  /*
480  * Set the new context based on the attribute
481  */
482  slen = fr_pair_ctx_set(pair_ctx, p, end - p);
483  if (slen <= 0) return slen - (p - in);
484 
485  p += slen;
486  return p - in;
487 }
