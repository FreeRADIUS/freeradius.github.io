96 /** Process a subrequest until it either detaches, or is done.
97  *
98  */
99 static unlang_action_t unlang_subrequest_process(rlm_rcode_t *p_result, request_t *request)
100 {
101  unlang_stack_t *stack = request->stack;
102  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
103  unlang_frame_state_subrequest_t *state = talloc_get_type_abort(frame->state, unlang_frame_state_subrequest_t);
104  request_t *child = talloc_get_type_abort(state->child, request_t);
105  unlang_group_t *g = unlang_generic_to_group(frame->instruction);
106  rlm_rcode_t rcode;
107 
108  fr_assert(child != NULL);
109 
110  rcode = unlang_interpret(child);
111  if (rcode != RLM_MODULE_YIELD) {
112  if (!fr_cond_assert(rcode < NUM_ELEMENTS(frame->instruction->actions))) return UNLANG_ACTION_STOP_PROCESSING;
113 
114  if (state->session.enable) fr_state_store_in_parent(child,
115  state->session.unique_ptr,
116  state->session.unique_int);
117 
118  /*
119  * Copy the subrequest's reply list into
120  * the parent, creating any n
121  */
122 
123  calculate_result:
124  if (child->reply) {
125  unlang_subrequest_t *gext;
126 
127  /*
128  * Copy reply attributes into the specified
129  * destination.
130  */
131  gext = unlang_group_to_subrequest(g);
132  if (gext->dst) {
133  tmpl_attr_extent_t *extent = NULL;
134  fr_dlist_head_t leaf;
135  fr_dlist_head_t interior;
136 
137  fr_dlist_talloc_init(&leaf, tmpl_attr_extent_t, entry);
138  fr_dlist_talloc_init(&interior, tmpl_attr_extent_t, entry);
139 
140  /*
141  * Find out what we need to build and build it
142  */
143  if ((tmpl_extents_find(state, &leaf, &interior, request, gext->dst) < 0) ||
144  (tmpl_extents_build_to_leaf(&leaf, &interior, gext->dst) < 0)) {
145  RPDEBUG("Discarding subrequest attributes - Failed allocating groups");
146  fr_dlist_talloc_free(&leaf);
147  fr_dlist_talloc_free(&interior);
148  goto done;
149  }
150  while ((extent = fr_dlist_tail(&leaf))) {
151  fr_pair_list_copy(extent->list_ctx, extent->list, &child->reply_pairs);
152  fr_dlist_talloc_free_tail(&leaf);
153  }
154  }
155  }
156 
157  done:
158  if (state->free_child) {
159  unlang_subrequest_free(&child);
160  state->child = NULL;
161  frame->signal = NULL;
162  }
163 
164  /*
165  * Pass the result back to the module
166  * that created the subrequest, or
167  * use it to modify the current section
168  * rcode.
169  */
170  if (state->p_result) *state->p_result = rcode;
171 
172  *p_result = rcode;
173 
174  return UNLANG_ACTION_CALCULATE_RESULT;
175  }
176 
177  /*
178  * The child has yielded, BUT has also detached itself.
179  * We know this because it has reached into our state,
180  * and removed itself as a child. We therefore just keep
181  * running, and don't return yield.
182  */
183  if (!state->child) {
184  rcode = RLM_MODULE_NOOP;
185  goto calculate_result;
186  }
187 
188  /*
189  * Else the child yielded, so we have to yield. Set up
190  * the resume frame and continue.
191  */
192  return UNLANG_ACTION_YIELD;
193 }

197 {
198  unlang_stack_t *stack = request->stack;
199  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
200  unlang_frame_state_subrequest_t *state = talloc_get_type_abort(frame->state, unlang_frame_state_subrequest_t);
201  request_t *child = state->child;
202 
203  /*
204  * Restore state from the parent to the
205  * subrequest.
206  */
207  if (state->session.enable) fr_state_restore_to_child(child,
208  state->session.unique_ptr,
209  state->session.unique_int);
210 
211  RDEBUG2("Creating subrequest (%s)", child->name);
212  log_request_pair_list(L_DBG_LVL_1, request, child->request_pairs, NULL);
213 
214  frame->process = unlang_subrequest_process;
215  return unlang_subrequest_process(p_result, request);
216 }

219 static unlang_action_t unlang_subrequest_state_init(rlm_rcode_t *p_result, request_t *request)
220 {
221  unlang_stack_t *stack = request->stack;
222  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
223  unlang_t *instruction = frame->instruction;
224  unlang_frame_state_subrequest_t *state = talloc_get_type_abort(frame->state, unlang_frame_state_subrequest_t);
225  request_t *child;
226 
227  rlm_rcode_t rcode;
228  fr_pair_t *vp;
229 
230  unlang_group_t *g;
231  unlang_subrequest_t *gext;
232 
233  /*
234  * Initialize the state
235  */
236  g = unlang_generic_to_group(instruction);
237  if (!g->num_children) {
238  *p_result = RLM_MODULE_NOOP;
239  return UNLANG_ACTION_CALCULATE_RESULT;
240  }
241 
242  gext = unlang_group_to_subrequest(g);
243  child = state->child = unlang_io_subrequest_alloc(request, gext->dict, UNLANG_DETACHABLE);
244  if (!child) {
245  fail:
246  rcode = RLM_MODULE_FAIL;
247 
248  /*
249  * Pass the result back to the module
250  * that created the subrequest, or
251  * use it to modify the current section
252  * rcode.
253  */
254  if (state->p_result) *state->p_result = rcode;
255 
256  *p_result = rcode;
257  return UNLANG_ACTION_CALCULATE_RESULT;
258  }
259  /*
260  * Set the packet type.
261  */
262  MEM(vp = fr_pair_afrom_da(child->packet, gext->attr_packet_type));
263 
264  if (gext->type_enum) {
265  child->packet->code = vp->vp_uint32 = gext->type_enum->value->vb_uint32;
266  } else {
267  fr_dict_enum_t const *type_enum;
268  fr_pair_t *attr;
269 
270  if (tmpl_find_vp(&attr, request, gext->vpt) < 0) {
271  RDEBUG("Failed finding attribute %s", gext->vpt->name);
272  goto fail;
273  }
274 
275  if (tmpl_da(gext->vpt)->type == FR_TYPE_STRING) {
276  type_enum = fr_dict_enum_by_name(gext->attr_packet_type, attr->vp_strvalue, attr->vp_length);
277  if (!type_enum) {
278  RDEBUG("Unknown Packet-Type %pV", &attr->data);
279  goto fail;
280  }
281 
282  child->packet->code = vp->vp_uint32 = type_enum->value->vb_uint32;
283  } else {
284  fr_value_box_t box;
285 
286  fr_value_box_init(&box, FR_TYPE_UINT32, NULL, false);
287  if (fr_value_box_cast(request, &box, FR_TYPE_UINT32, NULL, &attr->data) < 0) {
288  RDEBUG("Failed casting value from %pV to data type uint32", &attr->data);
289  goto fail;
290  }
291 
292  /*
293  * Check that the value is known to the server.
294  *
295  * If it isn't known, then there's no
296  * "recv foo" section for it and we can't
297  * do anything with this packet.
298  */
299  type_enum = fr_dict_enum_by_value(gext->attr_packet_type, &box);
300  if (!type_enum) {
301  RDEBUG("Invalid value %pV for Packet-Type", &box);
302  goto fail;
303  }
304 
305  child->packet->code = vp->vp_uint32 = box.vb_uint32;
306  }
307 
308  }
309  fr_pair_add(&child->request_pairs, vp);
310 
311  if (gext->src) {
312  if (tmpl_is_list(gext->src)) {
313  if (tmpl_copy_pairs(child->packet, &child->request_pairs, request, gext->src) < -1) {
314  RPEDEBUG("Failed copying source attributes into subrequest");
315  goto fail;
316  }
317  } else {
318  if (tmpl_copy_pair_children(child->packet, &child->request_pairs, request, gext->src) < -1) {
319  RPEDEBUG("Failed copying source attributes into subrequest");
320  goto fail;
321  }
322  }
323  }
324 
325  /*
326  * Push the first instruction the child's
327  * going to run.
328  */
329  if (unlang_interpret_push(child, g->children, frame->result,
330  UNLANG_NEXT_SIBLING, UNLANG_TOP_FRAME) < 0) goto fail;
331 
332  /*
333  * Probably not a great idea to set state->p_result to
334  * p_result, as it could be a pointer to an rlm_rcode_t
335  * somewhere on the stack which could be invalidated
336  * between unlang_subrequest being called and
337  * unlang_subrequest_resume being called.
338  *
339  * ...so we just set it to NULL and interpret
340  * that as use the p_result that was passed
341  * in to the currently executing function.
342  */
343  state->p_result = NULL;
344  state->free_child = true;
345  state->detachable = true;
346 
347  /*
348  * Store/restore session information in the subrequest
349  * keyed off the exact subrequest keyword.
350  */
351  state->session.enable = true;
352  state->session.unique_ptr = instruction;
353  state->session.unique_int = 0;
354 
355  frame->process = unlang_subrequest_start;
356  return unlang_subrequest_start(p_result, request);
357 }

359 /** Initialize a detached child
360  *
361  * Detach it from the parent, set up it's lifetime, and mark it as
362  * runnable.
363  */
364 int unlang_detached_child_init(request_t *request)
365 {
366  fr_pair_t *vp;
367 
368  if (request_detach(request, false) < 0) {
369  ERROR("Failed detaching child");
370  return -1;
371  }
372 
373  /*
374  * Set Request Lifetime
375  */
376  vp = fr_pair_find_by_da(&request->control_pairs, attr_request_lifetime);
377  if (!vp || (vp->vp_uint32 > 0)) {
378  fr_time_delta_t when = 0;
379  const fr_event_timer_t **ev_p;
380 
381  if (!vp) {
382  when += fr_time_delta_from_sec(30); /* default to 30s if not set */
383 
384  } else if (vp->vp_uint32 > 3600) {
385  RWDEBUG("Request-Timeout can be no more than 3600 seconds");
386  when += fr_time_delta_from_sec(3600);
387 
388  } else if (vp->vp_uint32 < 5) {
389  RWDEBUG("Request-Timeout can be no less than 5 seconds");
390  when += fr_time_delta_from_sec(5);
391 
392  } else {
393  when += fr_time_delta_from_sec(vp->vp_uint32);
394  }
395 
396  ev_p = talloc_size(request, sizeof(*ev_p));
397  memset(ev_p, 0, sizeof(*ev_p));
398 
399  (void) fr_event_timer_in(request, request->el, ev_p, when, unlang_max_request_time, request);
400  }
401 
402  /*
403  * Mark the child as runnable.
404  */
405  fr_assert(request->parent == NULL);
406  if (fr_heap_insert(request->backlog, request) < 0) {
407  RPERROR("Failed inserting ourself into the backlog.");
408  return -1;
409  }
410 
411  return 0;
412 }

414 static unlang_action_t unlang_detach(rlm_rcode_t *p_result, request_t *request)
415 {
416  unlang_stack_t *stack = request->stack;
417  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
418  unlang_frame_state_detach_t *state = talloc_get_type_abort(frame->state, unlang_frame_state_detach_t);
419 
420  unlang_frame_state_subrequest_t *parent_state;
421 
422  /*
423  * We've detached, yielded, and now are continuing
424  * processing. There's nothing more to do, so just
425  * continue.
426  */
427  if (!request->parent) {
428  *p_result = state->rcode;
429  return UNLANG_ACTION_CALCULATE_RESULT;
430  }
431 
432  /*
433  * First time through, detach from the parent.
434  */
435  RDEBUG2("detach");
436 
437  fr_assert(request->parent != NULL);
438 
439  /*
440  * Get the PARENT's stack.
441  */
442  stack = request->parent->stack;
443  frame = &stack->frame[stack->depth];
444  parent_state = talloc_get_type_abort(frame->state, unlang_frame_state_subrequest_t);
445 
446  if (!parent_state->detachable) {
447  RWDEBUG("Ignoring 'detach' as the request is not detachable");
448  return UNLANG_ACTION_CALCULATE_RESULT;
449  }
450 
451  /*
452  * If we can't detach the child OR we can't insert it
453  * into the backlog, stop processing it.
454  */
455  if (unlang_detached_child_init(request) < 0) {
456  return UNLANG_ACTION_STOP_PROCESSING;
457  }
458 
459  /*
460  * The parent frame no longer has a child, and therefore
461  * can't be signaled.
462  */
463  fr_assert(parent_state->child == request);
464  parent_state->child = NULL;
465  frame->signal = NULL;
466 
467  /*
468  * Pass through whatever the previous instruction had as
469  * the result.
470  */
471  state->rcode = *p_result;
472 
473  /*
474  * Yield to the parent, who will discover that there's no
475  * child, and return.
476  */
477  return UNLANG_ACTION_YIELD;
478 }

493 /** Push a pre-existing child back onto the stack as a subrequest
494  *
495  * The child *MUST* have been allocated with unlang_io_subrequest_alloc, or something
496  * that calls it.
497  * Instruction *MUST* belong to the same virtual server as is set in the child.
498  *
499  * After the child is no longer required it *MUST* be freed with #unlang_subrequest_free.
500  * It's not enough to free it with talloc_free.
501  *
502  * @param[in] out Where to write the result of the subrequest.
503  * @param[in] child to push.
504  * @param[in] session control values. Whether we restore/store session info.
505  * @param[in] top_frame Set to UNLANG_TOP_FRAME if the interpreter should return.
506  * Set to UNLANG_SUB_FRAME if the interprer should continue.
507  * @return
508  * - 0 on success.
509  * - -1 on failure.
510  */
511 int unlang_subrequest_push(rlm_rcode_t *out, request_t *child,
512  unlang_subrequest_session_t const *session, bool top_frame)
513 {
514  unlang_stack_t *stack = child->parent->stack;
515  unlang_stack_frame_t *frame;
516  unlang_frame_state_subrequest_t *state;
517 
518  /*
519  * Push a new subrequest frame onto the stack
520  */
521  if (unlang_interpret_push(child->parent, &subrequest_instruction->self,
522  RLM_MODULE_UNKNOWN, UNLANG_NEXT_STOP, top_frame) < 0) return -1;
523 
524  frame = &stack->frame[stack->depth];
525 
526  /*
527  * Allocate a state for the subrequest
528  * This lets us override the normal request
529  * the subrequest instruction would alloc.
530  */
531  state = talloc_get_type_abort(frame->state, unlang_frame_state_subrequest_t);
532  state->p_result = out;
533  state->child = child;
534  state->free_child = false;
535  state->detachable = false;
536  if (session) state->session = *session;
537 
538  frame->process = unlang_subrequest_start;
539 
540  return 0;
541 }
