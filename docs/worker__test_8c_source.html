195 static void master_process(void)
196 {
197  bool running, signaled_close;
198  int rcode, i, num_events, which_worker;
199  int num_outstanding, num_messages;
200  int num_replies;
201  fr_message_set_t *ms;
202  TALLOC_CTX *ctx;
203  fr_channel_t *ch;
204  fr_channel_event_t ce;
205  pthread_attr_t attr;
206  fr_schedule_worker_t *sw;
207  fr_listen_t listen = { .app_io = &app_io };
208  struct kevent events[MAX_KEVENTS];
209 
210  MEM(ctx = talloc_init_const("master"));
211 
212  ms = fr_message_set_create(ctx, MAX_MESSAGES, sizeof(fr_channel_data_t), MAX_MESSAGES * 1024);
213  if (!ms) {
214  fprintf(stderr, "Failed creating message set\n");
215  fr_exit_now(EXIT_FAILURE);
216  }
217 
218  MPRINT1("Master started.\n");
219 
220  /*
221  * Create the worker threads.
222  */
223  (void) pthread_attr_init(&attr);
224  (void) pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_JOINABLE);
225 
226  for (i = 0; i < num_workers; i++) {
227  workers[i].id = i;
228  (void) pthread_create(&workers[i].pthread_id, &attr, worker_thread, &workers[i]);
229  }
230 
231  MPRINT1("Master created %d workers.\n", num_workers);
232 
233  /*
234  * Busy loop because that's fine for the test
235  */
236  num_outstanding = 0;
237  while (num_outstanding < num_workers) {
238  for (i = 0; i < num_workers; i++) {
239  if (!workers[i].worker) continue;
240  if (workers[i].ch != NULL) continue;
241 
242  /*
243  * Create the channel and signal the
244  * worker that it is open
245  */
246  MPRINT1("Master creating channel to worker %d.\n", num_workers);
247  workers[i].ch = fr_worker_channel_create(workers[i].worker, ctx, control_master);
248  fr_assert(workers[i].ch != NULL);
249 
250  (void) fr_channel_master_ctx_add(workers[i].ch, &workers[i]);
251 
252  num_outstanding++;
253  }
254  }
255 
256  MPRINT1("Master created all channels.\n");
257 
258  /*
259  * Bootstrap the queue with messages.
260  */
261  num_replies = num_outstanding = num_messages = 0;
262  which_worker = 0;
263 
264  running = true;
265  signaled_close = false;
266 
267  while (running) {
268  fr_time_t now;
269  int num_to_send;
270  fr_channel_data_t *cd, *reply;
271 
272  /*
273  * Ensure we have outstanding messages.
274  */
275  if (num_messages >= max_messages) {
276  MPRINT1("Master DONE sending\n");
277  goto check_close;
278  }
279 
280  num_to_send = max_outstanding - num_outstanding;
281  if ((num_messages + num_to_send) > max_messages) {
282  num_to_send = max_messages - num_messages;
283  }
284  MPRINT1("Master sending %d messages\n", num_to_send);
285 
286  for (i = 0; i < num_to_send; i++) {
287  cd = (fr_channel_data_t *) fr_message_alloc(ms, NULL, 100);
288  fr_assert(cd != NULL);
289 
290  num_outstanding++;
291  num_messages++;
292 
293  cd->m.when = fr_time();
294 
295  cd->priority = 0;
296  cd->listen = &listen;
297 
298  if (touch_memory) {
299  size_t j, k;
300 
301  for (j = k = 0; j < cd->m.data_size; j++) {
302  k += cd->m.data[j];
303  }
304 
305  cd->m.data[4] = k;
306  }
307 
308  memcpy(cd->m.data, &num_messages, sizeof(num_messages));
309 
310  MPRINT1("Master sent message %d to worker %d\n", num_messages, which_worker);
311  rcode = fr_channel_send_request(workers[which_worker].ch, cd, &reply);
312  if (rcode < 0) {
313  fprintf(stderr, "Failed sending request: %s\n", fr_syserror(errno));
314  }
315  which_worker++;
316  if (which_worker >= num_workers) which_worker = 0;
317 
318  fr_assert(rcode == 0);
319  if (reply) {
320  num_replies++;
321  num_outstanding--;
322  MPRINT1("Master got reply %d, outstanding=%d, %d/%d sent.\n",
323  num_replies, num_outstanding, num_messages, max_messages);
324  fr_message_done(&reply->m);
325  }
326  }
327 
328  /*
329  * Signal close only when done.
330  */
331 check_close:
332  if (!signaled_close && (num_messages >= max_messages) && (num_outstanding == 0)) {
333  MPRINT1("Master signaling workers to exit.\n");
334 
335  for (i = 0; i < num_workers; i++) {
336  if (!quiet) {
337  printf("Worker %d\n", i);
338  fr_worker_debug(workers[i].worker, stdout);
339  }
340 
341  rcode = fr_channel_signal_responder_close(workers[i].ch);
342  MPRINT1("Master asked exit for worker %d.\n", workers[i].id);
343  if (rcode < 0) {
344  fprintf(stderr, "Failed signaling close %d: %s\n", i, fr_syserror(errno));
345  fr_exit_now(EXIT_FAILURE);
346  }
347  }
348  signaled_close = true;
349  }
350 
351  MPRINT1("Master waiting on events.\n");
352  fr_assert(num_messages <= max_messages);
353 
354  num_events = kevent(kq_master, NULL, 0, events, MAX_KEVENTS, NULL);
355  MPRINT1("Master kevent returned %d\n", num_events);
356 
357  if (num_events < 0) {
358  if (errno == EINTR) continue;
359 
360  fprintf(stderr, "Failed waiting for kevent: %s\n", fr_syserror(errno));
361  fr_exit_now(EXIT_FAILURE);
362  }
363 
364  if (num_events == 0) continue;
365 
366  /*
367  * Service the events.
368  *
369  * @todo this should NOT take a channel pointer
370  */
371  for (i = 0; i < num_events; i++) {
372  (void) fr_channel_service_kevent(workers[0].ch, control_master, &events[i]);
373  }
374 
375  now = fr_time();
376 
377  MPRINT1("Master servicing control-plane\n");
378 
379  while (true) {
380  uint32_t id;
381  size_t data_size;
382  char data[256];
383 
384  data_size = fr_control_message_pop(aq_master, &id, data, sizeof(data));
385  if (!data_size) break;
386 
387  fr_assert(id == FR_CONTROL_ID_CHANNEL);
388 
389  ce = fr_channel_service_message(now, &ch, data, data_size);
390  MPRINT1("Master got channel event %d\n", ce);
391 
392  switch (ce) {
393  case FR_CHANNEL_DATA_READY_REQUESTOR:
394  MPRINT1("Master got data ready signal\n");
395 
396  reply = fr_channel_recv_reply(ch);
397  if (!reply) {
398  MPRINT1("Master SIGNAL WITH NO DATA!\n");
399  continue;
400  }
401 
402  do {
403  num_replies++;
404  num_outstanding--;
405  MPRINT1("Master got reply %d, outstanding=%d, %d/%d sent.\n",
406  num_replies, num_outstanding, num_messages, max_messages);
407  fr_message_done(&reply->m);
408  } while ((reply = fr_channel_recv_reply(ch)) != NULL);
409  break;
410 
411  case FR_CHANNEL_CLOSE:
412  sw = fr_channel_master_ctx_get(ch);
413  fr_assert(sw != NULL);
414 
415  MPRINT1("Master received close signal for worker %d\n", sw->id);
416  fr_assert(signaled_close == true);
417 
418  (void) pthread_kill(sw->pthread_id, SIGTERM);
419  running = false;
420  break;
421 
422  case FR_CHANNEL_NOOP:
423  break;
424 
425  default:
426  fprintf(stderr, "Master got unexpected CE %d\n", ce);
427 
428  /*
429  * Not written yet!
430  */
431  fr_assert(0 == 1);
432  break;
433  } /* switch over signal returned */
434  } /* drain the control plane */
435  } /* loop until told to exit */
436 
437  MPRINT1("Master exiting.\n");
438 
439  fr_time_t last_checked = fr_time();
440 
441  /*
442  * Busy-wait for the workers to exit;
443  */
444  do {
445  fr_time_t now = fr_time();
446 
447  num_outstanding = num_workers;
448 
449  for (i = 0; i < num_workers; i++) {
450  if (!workers[i].worker) num_outstanding--;
451  }
452 
453  if ((now - last_checked) > (NSEC / 10)) {
454  MPRINT1("still num_outstanding %d\n", num_outstanding);
455  }
456 
457  } while (num_outstanding > 0);
458 
459  /*
460  * Force all messages to be garbage collected
461  */
462  MPRINT2("GC\n");
463  fr_message_set_gc(ms);
464 
465  if (debug_lvl > 1) fr_message_set_debug(ms, stdout);
466 
467  /*
468  * After the garbage collection, all messages marked "done" MUST also be marked "free".
469  */
470  rcode = fr_message_set_messages_used(ms);
471  MPRINT2("Master messages used = %d\n", rcode);
472  fr_assert(rcode == 0);
473 
474  talloc_free(ctx);
475 
476 }
