93 #ifdef SIOCSARP
94 /** Forcibly add an ARP entry so we can send unicast packets to hosts that don't have IP addresses yet
95  *
96  * @param[in] fd to add arp entry on.
97  * @param[in] interface to add arp entry on.
98  * @param[in] ipaddr to insert into ARP table.
99  * @param[in] macaddr to insert into ARP table.
100  * @return
101  * - 0 on success.
102  * - -1 on failure.
103  */
104 int fr_arp_entry_add(int fd, char const *interface, uint8_t ipaddr[static 4], uint8_t macaddr[static 6])
105 {
106  struct sockaddr_in *sin;
107  struct arpreq req;
108 
109  if (!interface) {
110  fr_strerror_printf("No interface specified. Cannot update ARP table");
111  return -1;
112  }
113 
114  memset(&req, 0, sizeof(req));
115  sin = (struct sockaddr_in *) &req.arp_pa;
116  sin->sin_family = AF_INET;
117  memcpy(&sin->sin_addr.s_addr, ipaddr, 4);
118 
119  strlcpy(req.arp_dev, interface, sizeof(req.arp_dev));
120 
121  memcpy(&req.arp_ha.sa_data, macaddr, 6);
122 
123  req.arp_flags = ATF_COM;
124  if (ioctl(fd, SIOCSARP, &req) < 0) {
125  fr_strerror_printf("Failed to add entry in ARP cache: %s (%d)", fr_syserror(errno), errno);
126  return -1;
127  }
128 
129  return 0;
130 }
131 #else
132 int fr_arp_entry_add(UNUSED int fd, UNUSED char const *interface,

134 {
135  fr_strerror_printf("Adding ARP entry is unsupported on this system");

141 /** Encode VPS into a raw ARP packet.
142  *
143  */
144 ssize_t fr_arp_encode(fr_dbuff_t *dbuff, uint8_t const *original, fr_pair_t *vps)
145 {
146  ssize_t slen;
147  fr_pair_t *vp;
148  fr_arp_packet_t *arp;
149  fr_cursor_t cursor;
150  fr_da_stack_t da_stack;
151  fr_dbuff_t work_dbuff = FR_DBUFF_NO_ADVANCE(dbuff);
152 
153  if (!vps) {
154  fr_strerror_printf("Cannot encode empty packet");
155  return -1;
156  }
157 
158  /*
159  * Find the first attribute which is parented by ARP-Packet.
160  */
161  for (vp = fr_cursor_init(&cursor, &vps);
162  vp;
163  vp = fr_cursor_next(&cursor)) {
164  if (vp->da->parent == attr_arp_packet) break;
165  }
166 
167  /*
168  * For simplicity, we allow the caller to omit things
169  * that they don't care about.
170  */
171  if (!vp) {
172  fr_strerror_printf("No ARP attributes in the attribute list");
173  return -1;
174  }
175 
176  fr_proto_da_stack_build(&da_stack, attr_arp_packet);
177  FR_PROTO_STACK_PRINT(&da_stack, 0);
178 
179  /*
180  * Call the struct encoder to do the actual work.
181  */
182  slen = fr_struct_to_network_dbuff(&work_dbuff, &da_stack, 0, &cursor, NULL, NULL);
183  if (slen <= 0) return slen;
184 
185  if (slen != FR_ARP_PACKET_SIZE) return slen;
186 
187  /*
188  * Hard-code fields which can be omitted.
189  */
190  arp = (fr_arp_packet_t *)fr_dbuff_start(&work_dbuff);
191 
192  if ((arp->htype[0] == 0) && (arp->htype[1] == 0)) arp->htype[1] = FR_HARDWARE_FORMAT_VALUE_ETHERNET;
193  if ((arp->ptype[0] == 0) && (arp->ptype[1] == 0)) arp->ptype[0] = (FR_PROTOCOL_FORMAT_VALUE_IPV4 >> 8); /* 0x0800 */
194  if (arp->hlen == 0) arp->hlen = 6;
195  if (arp->plen == 0) arp->plen = 4;
196 
197  /*
198  * The reply generally swaps Sender / Target addresses,
199  * BUT fills out the Sender-Hardware-Address with the
200  * queried MAC address. Ensure that the admin doesn't
201  * have to fill out all of the fields.
202  */
203  if (original) {
204  fr_arp_packet_t const *our_original = (fr_arp_packet_t const *) original;
205 
206 #define COPY(_a, _b) do { \
207  if (memcmp(arp->_a, zeros, sizeof(arp->_a)) == 0) { \
208  memcpy(arp->_a, our_original->_b, sizeof(arp->_a)); \
209  } \
210  } while (0)
211 
212  COPY(spa, tpa); /* answer is about the asked-for target */
213  COPY(tha, sha); /* answer is sent to the requestor hardware address */
214  COPY(tpa, spa); /* answer is sent to the requestor protocol address */
215  }
216 
217  return fr_dbuff_set(dbuff, &work_dbuff);;
218 }

220 /** Decode a raw ARP packet into VPs
221  *
222  */
223 ssize_t fr_arp_decode(TALLOC_CTX *ctx, uint8_t const *packet, size_t packet_len, fr_pair_list_t *list)
224 {
225  fr_arp_packet_t const *arp;
226  fr_cursor_t cursor;
227  fr_dict_attr_t const *child;
228 
229  if (packet_len < FR_ARP_PACKET_SIZE) {
230  fr_strerror_printf("Packet is too small (%d) to be ARP", (int) packet_len);
231  return -1;
232  }
233 
234  /*
235  * Check that the fields have the desired values.
236  */
237  arp = (fr_arp_packet_t const *) packet;
238  if ((arp->htype[0] != 0) || (arp->htype[1] != 1)) {
239  fr_strerror_printf("Hardware-Format != Ethernet");
240  return -1;
241  }
242 
243  if ((arp->ptype[0] != 8) || (arp->ptype[1] != 0)) {
244  fr_strerror_printf("Protocol-Format != IPv4");
245  return -1;
246  }
247 
248  if (arp->hlen != 6) {
249  fr_strerror_printf("Hardware-Length != 6");
250  return -1;
251  }
252 
253  if (arp->plen != 4) {
254  fr_strerror_printf("Protocol-Length != 4");
255  return -1;
256  }
257 
258  /*
259  * If the packet is too long, we discard any extra data.
260  */
261  fr_pair_list_init(list);
262  fr_cursor_init(&cursor, list);
263  return fr_struct_from_network(ctx, &cursor, attr_arp_packet, packet, FR_ARP_PACKET_SIZE, &child, NULL, NULL);
264 }
