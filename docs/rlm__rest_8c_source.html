182 static int rlm_rest_perform(rlm_rest_t const *instance, rlm_rest_thread_t *t,
183  rlm_rest_section_t const *section, fr_curl_io_request_t *randle,
184  request_t *request, char const *username, char const *password)
185 {
186  ssize_t uri_len;
187  char *uri = NULL;
188  int ret;
189

191 
192  /*
193  * Build xlat'd URI, this allows REST servers to be specified by
194  * request attributes.
195  */
196  uri_len = rest_uri_build(&uri, instance, request, section->uri);
197  if (uri_len <= 0) return -1;
198 
199  RDEBUG2("Sending HTTP %s to \"%s\"", fr_table_str_by_value(http_method_table, section->method, NULL), uri);
200

202  * Configure various CURL options, and initialise the read/write
203  * context data.
204  */

206  uri, username, password);

209

211  * Send the CURL request, pre-parse headers, aggregate incoming
212  * HTTP body data into a single contiguous buffer.
213  */

215  if (ret < 0) return -1;
216 
217  return 0;

220 static xlat_action_t rest_xlat_resume(TALLOC_CTX *ctx, fr_cursor_t *out,
221  request_t *request, UNUSED void const *xlat_inst, void *xlat_thread_inst,
222  UNUSED fr_value_box_t **in, void *rctx)
223 {
224  rest_xlat_thread_inst_t *xti = talloc_get_type_abort(xlat_thread_inst, rest_xlat_thread_inst_t);
225  rlm_rest_t const *mod_inst = xti->inst;
226  rlm_rest_thread_t *t = xti->t;
227 
228  rlm_rest_xlat_rctx_t *our_rctx = talloc_get_type_abort(rctx, rlm_rest_xlat_rctx_t);
229  int hcode;
230  ssize_t len;
231  char const *body;
232  xlat_action_t xa = XLAT_ACTION_DONE;
233 
234  fr_curl_io_request_t *handle = talloc_get_type_abort(our_rctx->handle, fr_curl_io_request_t);
235  rlm_rest_section_t *section = &our_rctx->section;
236 
237  if (section->tls.extract_cert_attrs) fr_curl_response_certinfo(request, handle);
238 
239  if (rlm_rest_status_update(request, handle) < 0) {
240  xa = XLAT_ACTION_FAIL;
241  goto finish;
242  }
243 
244  hcode = rest_get_handle_code(handle);
245  switch (hcode) {
246  case 404:
247  case 410:
248  case 403:
249  case 401:
250  {
251  xa = XLAT_ACTION_FAIL;
252 error:
253  rest_response_error(request, handle);
254  goto finish;
255  }
256  case 204:
257  goto finish;
258 
259  default:
260  /*
261  * Attempt to parse content if there was any.
262  */
263  if ((hcode >= 200) && (hcode < 300)) {
264  break;
265  } else if (hcode < 500) {
266  xa = XLAT_ACTION_FAIL;
267  goto error;
268  } else {
269  xa = XLAT_ACTION_FAIL;
270  goto error;
271  }
272  }
273 
274  len = rest_get_handle_data(&body, handle);
275  if (len > 0) {
276  fr_value_box_t *vb;
277 
278  MEM(vb = fr_value_box_alloc_null(ctx));
279  fr_value_box_bstrndup(vb, vb, NULL, body, len, true);
280  fr_cursor_insert(out, vb);
281  }
282 
283 finish:
284  rest_request_cleanup(mod_inst, handle);
285 
286  fr_pool_connection_release(t->pool, request, handle);
287 
288  talloc_free(our_rctx);
289 
290  return xa;
291 }

293 /** Simple xlat to read text data from a URL
294  *
295  * Example:
296 @verbatim
297 %{rest:http://example.com/}
298 @endverbatim
299  *
300  * @ingroup xlat_functions
301  */
302 static xlat_action_t rest_xlat(TALLOC_CTX *ctx, UNUSED fr_cursor_t *out,
303  request_t *request, UNUSED void const *xlat_inst, void *xlat_thread_inst,
304  fr_value_box_t **in)
305 {
306  rest_xlat_thread_inst_t *xti = talloc_get_type_abort(xlat_thread_inst, rest_xlat_thread_inst_t);
307  rlm_rest_t const *mod_inst = xti->inst;
308  rlm_rest_thread_t *t = xti->t;
309 
310  fr_curl_io_request_t *randle = NULL;
311  ssize_t len;
312  int ret;
313  char *uri = NULL;
314  char const *p = NULL, *q;
315  http_method_t method;
316 
317  /* There are no configurable parameters other than the URI */
318  rlm_rest_xlat_rctx_t *rctx;
319  rlm_rest_section_t *section;
320 
321  if (!*in) {
322  REDEBUG("Got empty URL string");
323  return XLAT_ACTION_FAIL;
324  }
325 
326  if (fr_value_box_list_concat(ctx, *in, in, FR_TYPE_STRING, true) < 0) {
327  REDEBUG("Failed concatenating arguments into URL string");
328  return XLAT_ACTION_FAIL;
329  }
330  p = (*in)->vb_strvalue;
331 
332  MEM(rctx = talloc(request, rlm_rest_xlat_rctx_t));
333  section = &rctx->section;
334 
335  /*
336  * Section gets modified, so we need our own copy.
337  */
338  memcpy(&rctx->section, &mod_inst->xlat, sizeof(*section));
339 
340  RDEBUG2("Expanding URI components");
341 
342  /*
343  * Extract the method from the start of the format string (if there is one)
344  */
345  method = fr_table_value_by_substr(http_method_table, p, -1, REST_HTTP_METHOD_UNKNOWN);
346  if (method != REST_HTTP_METHOD_UNKNOWN) {
347  section->method = method;
348  p += http_method_table[method].name.len;
349  /*
350  * If the method is unknown, it's either a URL or a verb
351  */
352  } else {
353  for (q = p; (*q != ' ') && (*q != '\0') && isalpha(*q); q++);
354 
355  /*
356  * If the first non-alpha char was a space,
357  * then assume this is a verb.
358  */
359  if ((*q == ' ') && (q != p)) {
360  section->method = REST_HTTP_METHOD_CUSTOM;
361  MEM(section->method_str = talloc_bstrndup(rctx, p, q - p));
362  p = q;
363  } else {
364  section->method = REST_HTTP_METHOD_GET;
365  }
366  }
367 
368  /*
369  * Trim whitespace
370  */
371  fr_skip_whitespace(p);
372 
373  randle = rctx->handle = fr_pool_connection_get(t->pool, request);
374  if (!randle) return XLAT_ACTION_FAIL;
375 
376  /*
377  * Unescape parts of xlat'd URI, this allows REST servers to be specified by
378  * request attributes.
379  */
380  len = rest_uri_host_unescape(&uri, mod_inst, request, randle, p);
381  if (len <= 0) {
382  error:
383  rest_request_cleanup(mod_inst, randle);
384  fr_pool_connection_release(t->pool, request, randle);
385  talloc_free(section);
386 
387  return XLAT_ACTION_FAIL;
388  }
389 
390  /*
391  * Extract freeform body data (url can't contain spaces)
392  */
393  q = strchr(p, ' ');
394  if (q && (*++q != '\0')) {
395  section->body = REST_HTTP_BODY_CUSTOM_LITERAL;
396  section->data = q;
397  }
398 
399  RDEBUG2("Sending HTTP %s to \"%s\"",
400  (section->method == REST_HTTP_METHOD_CUSTOM) ?
401  section->method_str : fr_table_str_by_value(http_method_table, section->method, NULL),
402  uri);
403 
404  /*
405  * Configure various CURL options, and initialise the read/write
406  * context data.
407  *
408  * @todo We could extract the User-Name and password from the URL string.
409  */
410  ret = rest_request_config(mod_inst, t, section, request,
411  randle, section->method, section->body, uri, NULL, NULL);
412  talloc_free(uri);
413  if (ret < 0) goto error;
414 
415  /*
416  * Send the CURL request, pre-parse headers, aggregate incoming
417  * HTTP body data into a single contiguous buffer.
418  *
419  * @fixme need to pass in thread to all xlat functions
420  */
421  ret = fr_curl_io_request_enqueue(t->mhandle, request, randle);
422  if (ret < 0) goto error;
423 
424  return unlang_xlat_yield(request, rest_xlat_resume, rest_io_xlat_signal, rctx);
425 }

427 static unlang_action_t mod_authorize_result(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request, void *rctx)
428 {
429  rlm_rest_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_rest_t);
430  rlm_rest_thread_t *t = talloc_get_type_abort(mctx->thread, rlm_rest_thread_t);
431  rlm_rest_section_t const *section = &inst->authenticate;
432  fr_curl_io_request_t *handle = talloc_get_type_abort(rctx, fr_curl_io_request_t);
433 
434  int hcode;
435  rlm_rcode_t rcode = RLM_MODULE_OK;
436  int ret;
437 
438  if (section->tls.extract_cert_attrs) fr_curl_response_certinfo(request, handle);
439 
440  if (rlm_rest_status_update(request, handle) < 0) {
441  rcode = RLM_MODULE_FAIL;
442  goto finish;
443  }
444 
445  hcode = rest_get_handle_code(handle);
446  switch (hcode) {
447  case 404:
448  case 410:
449  rcode = RLM_MODULE_NOTFOUND;
450  break;
451 
452  case 403:
453  rcode = RLM_MODULE_DISALLOW;
454  break;
455 
456  case 401:
457  /*
458  * Attempt to parse content if there was any.
459  */
460  ret = rest_response_decode(inst, section, request, handle);
461  if (ret < 0) {
462  rcode = RLM_MODULE_FAIL;
463  break;
464  }
465 
466  rcode = RLM_MODULE_REJECT;
467  break;
468 
469  case 204:
470  rcode = RLM_MODULE_OK;
471  break;
472 
473  default:
474  /*
475  * Attempt to parse content if there was any.
476  */
477  if ((hcode >= 200) && (hcode < 300)) {
478  ret = rest_response_decode(inst, section, request, handle);
479  if (ret < 0) rcode = RLM_MODULE_FAIL;
480  else if (ret == 0) rcode = RLM_MODULE_OK;
481  else rcode = RLM_MODULE_UPDATED;
482  break;
483  } else if (hcode < 500) {
484  rcode = RLM_MODULE_INVALID;
485  } else {
486  rcode = RLM_MODULE_FAIL;
487  }
488  }
489 
490  switch (rcode) {
491  case RLM_MODULE_INVALID:
492  case RLM_MODULE_FAIL:
493  case RLM_MODULE_DISALLOW:
494  rest_response_error(request, handle);
495  break;
496 
497  default:
498  rest_response_debug(request, handle);
499  break;
500  }
501 
502 finish:
503  rest_request_cleanup(inst, handle);
504 
505  fr_pool_connection_release(t->pool, request, handle);
506 
507  RETURN_MODULE_RCODE(rcode);
508 }
