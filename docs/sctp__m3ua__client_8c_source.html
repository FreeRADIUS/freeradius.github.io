46 static struct xua_msg *xua_from_part(struct xua_msg_part *part)
47 {
48  size_t len = part->len;
49  uint8_t *data = part->dat;
50  struct xua_parameter_hdr *par;
51 
52  struct xua_msg *msg;
53  uint16_t pos, par_len, padding;
54  int rc;
55 
56  msg = xua_msg_alloc();
57  if (!msg)
58  return NULL;
59  pos = 0;
60 
61  while (pos + sizeof(*par) < len) {
62  par = (struct xua_parameter_hdr *) &data[pos];
63  par_len = ntohs(par->len);
64 
65  if (pos + par_len > len || par_len < 4)
66  goto fail;
67 
68  rc = xua_msg_add_data(msg, ntohs(par->tag), par_len - 4, par->data);
69  if (rc != 0)
70  goto fail;
71 
72  pos += par_len;
73 
74  /* move over the padding */
75  padding = (4 - (par_len % 4)) & 0x3;
76  pos += padding;
77  }
78 
79  /* TODO: parse */
80  return msg;
81 
82 fail:
83  LOGP(DINP, LOGL_ERROR, "Failed to parse.\n");
84  xua_msg_free(msg);
85  return NULL;
86 }

113 static int m3ua_setnonblocking(int fd)
114 {
115  int flags;
116 
117  flags = fcntl(fd, F_GETFL, NULL);
118  if (flags < 0) {
119  LOGP(DINP, LOGL_ERROR, "Failed getting socket flags whilst setting O_NONBLOCK.\n");
120  return -1;
121  }
122 
123  flags |= O_NONBLOCK;
124  if (fcntl(fd, F_SETFL, flags) < 0) {
125  LOGP(DINP, LOGL_ERROR, "Failed setting O_NONBLOCK\n");
126  return -1;
127  }
128 
129  return flags;
130 }

132 static int m3ua_setblocking(int fd)
133 {
134  int flags;
135 
136  flags = fcntl(fd, F_GETFL, NULL);
137  if (flags < 0) {
138  LOGP(DINP, LOGL_ERROR, "Failed getting socket flags whilst clearing O_NONBLOCK.\n");
139  return -1;

244 static int m3ua_conn_handle(struct mtp_m3ua_client_link *link,
245  struct msgb *msg, struct sctp_sndrcvinfo *info)
246 {
247  struct xua_msg *m3ua;
248  m3ua = xua_from_msg(M3UA_VERSION, msg->len, msg->data);
249  if (!m3ua) {
250  LOGP(DINP, LOGL_ERROR, "Failed to parse the message.\n");
251  return -1;
252  }
253 
254  switch (m3ua->hdr.msg_class) {
255  case M3UA_CLS_MGMT:
256  m3ua_handle_mgmt(link, m3ua);
257  break;
258  case M3UA_CLS_ASPSM:
259  m3ua_handle_aspsm(link, m3ua);
260  break;
261  case M3UA_CLS_ASPTM:
262  m3ua_handle_asptm(link, m3ua);
263  break;
264  case M3UA_CLS_TRANS:
265  m3ua_handle_trans(link, m3ua);
266  break;
267  case M3UA_CLS_RKM:
268  m3ua_handle_reg_rsp(link, m3ua);
269  break;
270  default:
271  LOGP(DINP, LOGL_ERROR, "Unhandled msg_class %d\n",
272  m3ua->hdr.msg_class);
273  break;
274  }
275 
276  xua_msg_free(m3ua);
277  return 0;
278 }

281 {
282  size_t ret;
283  char strerrbuf[256];
284  struct sctp_sndrcvinfo info;

295  ofd->fd, ret, strerror_r(errno, strerrbuf, sizeof(strerrbuf)));
296  } else {
297  LOGP(DINP, LOGL_DEBUG, "Wrote %u bytes to fd %i\n", msgb_l2len(msg), ofd->fd);
298  }
299 
300  return 0;

303 static int m3ua_conn_send(struct mtp_m3ua_client_link *link,
304  struct xua_msg *m3ua,
305  struct sctp_sndrcvinfo *info)
306 {
307  struct msgb *msg;
308  msg = xua_to_msg(M3UA_VERSION, m3ua);
309  if (!msg)
310  return -1;
311 
312  /* save the OOB data in front of the message */
313  msg->l2h = msg->data;
314  msgb_push(msg, sizeof(*info));
315  memcpy(msg->data, info, sizeof(*info));
316 
317  if (osmo_wqueue_enqueue(&link->queue, msg) != 0) {
318  LOGP(DINP, LOGL_ERROR, "Failed to enqueue.\n");
319  rate_ctr_inc(&link->base->ctrg->ctr[MTP_LNK_DRP]);
320  rate_ctr_inc(&link->base->set->ctrg->ctr[MTP_LSET_TOTA_DRP_MSG]);
321  msgb_free(msg);
322  return -1;
323  }
324 
325  return 0;
326 }

328 static int m3ua_conn_read(struct osmo_fd *fd)
329 {
330  struct sockaddr_in addr;
331  struct sctp_sndrcvinfo info;
332  socklen_t len = sizeof(addr);
333  struct mtp_m3ua_client_link *link = fd->data;
334  struct msgb *msg;
335  int rc;
336 
337  msg = msgb_alloc(2048, "m3ua buffer");
338  if (!msg) {
339  LOGP(DINP, LOGL_ERROR, "Failed to allocate buffer.\n");
340  fail_link(link);
341  return -1;
342  }
343 
344  memset(&info, 0, sizeof(info));
345  memset(&addr, 0, sizeof(addr));
346  rc = sctp_recvmsg(fd->fd, msg->data, msg->data_len,
347  (struct sockaddr *) &addr, &len, &info, NULL);
348  if (rc <= 0) {
349  LOGP(DINP, LOGL_ERROR, "Failed to read: %d errno: %d\n",
350  rc, errno);
351  msgb_free(msg);
352  fail_link(link);
353  return -1;
354  }
355 
356  if (ntohl(info.sinfo_ppid) != SCTP_PPID_M3UA) {
357  LOGP(DINP, LOGL_ERROR, "Only M3UA is allowed on this socket: %d\n",
358  ntohl(info.sinfo_ppid));
359  msgb_free(msg);
360  return -1;
361  }
362 
363  msgb_put(msg, rc);
364  LOGP(DINP, LOGL_DEBUG, "Read %d on stream: %d ssn: %d assoc: %d\n",
365  rc, info.sinfo_stream, info.sinfo_ssn, info.sinfo_assoc_id);
366  m3ua_conn_handle(link, msg, &info);
367  msgb_free(msg);
368  return 0;
369 }

371 static int m3ua_sctp_assoc_complete(struct osmo_fd *ofd, unsigned int what)
372 {
373  struct mtp_m3ua_client_link *link = ofd->data;
374  int ret, err;
375  socklen_t len = sizeof(err);
376 
377  osmo_fd_unregister(ofd); /* Remove our connect callback */
378 
379  ret = getsockopt(ofd->fd, SOL_SOCKET, SO_ERROR, &err, &len);
380  if (ret < 0) {
381  LOGP(DINP, LOGL_ERROR, "Failed getting socket error: %s (%i).\n", strerror(errno), errno);
382  error:
383  close(ofd->fd);
384  ofd->fd = -1;
385  fail_link(link);
386  return -1;
387  }
388 
389  if (err != 0) {
390  LOGP(DINP, LOGL_ERROR, "SCTP association failed: %s (%i).\n", strerror(err), errno);
391  goto error;
392  }
393 
394  LOGP(DINP, LOGL_NOTICE, "SCTP association established\n");
395 
396 // if (m3ua_setblocking(ofd->fd) < 0) goto error;
397 
398  link->queue.bfd.fd = ofd->fd;
399  link->queue.bfd.data = link;
400  link->queue.bfd.when = BSC_FD_READ;
401  link->queue.read_cb = m3ua_conn_read;
402  link->queue.write_cb = m3ua_conn_write;
403 
404  if (osmo_fd_register(&link->queue.bfd) != 0) {
405  LOGP(DINP, LOGL_ERROR, "Failed to register fd\n");
406  goto error;
407  }
408 
409  /* reset route state */
410  llist_splice_init(&link->routes_active, &link->routes);
411  llist_splice_init(&link->routes_failed, &link->routes);
412 
413  LOGP(DINP, LOGL_NOTICE, "Sending ASPUP\n");
414  m3ua_send_aspup(link);
415  schedule_aspup_t_ack(link);
416 
417  return 0;
418 }

420 static void m3ua_start(void *data)
421 {
422  int sctp, ret, on = 1;
423  struct mtp_m3ua_client_link *link = data;
424  struct sctp_event_subscribe events;
425 
426  sctp = socket(PF_INET, SOCK_STREAM, IPPROTO_SCTP);
427  if (!sctp) {
428  LOGP(DINP, LOGL_ERROR, "Failed to create socket.\n");
429  return fail_link(link);
430  }
431 
432  if (setsockopt(sctp, SOL_SOCKET, SO_REUSEADDR, &on, sizeof(on)) < 0) {
433  LOGP(DINP, LOGL_ERROR, "Failed setting reuseaddr: %s (%i).\n", strerror(errno), errno);
434  error:
435  close(sctp);
436  return fail_link(link);
437  }
438 
439  memset(&events, 0, sizeof(events));
440  events.sctp_data_io_event = 1;
441  events.sctp_association_event = 1;
442  ret = setsockopt(sctp, IPPROTO_SCTP, SCTP_EVENTS, &events, sizeof(events));
443  if (ret != 0) {
444  LOGP(DINP, LOGL_ERROR, "Failed to enable SCTP Events. Closing socket.\n");
445  goto error;
446  }
447 
448  if (bind(sctp, (struct sockaddr *) &link->local, sizeof(link->local)) != 0) {
449  LOGP(DINP, LOGL_ERROR, "Failed binding local side of SCTP association: %s (%i).\n", strerror(errno), errno);
450  goto error;
451  }
452 
453  if (m3ua_setnonblocking(sctp) < 0) goto error;
454 
455  LOGP(DINP, LOGL_NOTICE, "Initialising SCTP association\n");
456 
457  ret = connect(sctp, (struct sockaddr *) &link->remote, sizeof(link->remote));
458  if ((ret != 0) && (errno != EINPROGRESS)) {
459  LOGP(DINP, LOGL_ERROR, "Failed creating SCTP association: %s (%i).\n", strerror(errno), errno);
460  goto error;
461  }
462 
463  link->connect.fd = sctp;
464  link->connect.data = link;
465  link->connect.when = BSC_FD_WRITE | BSC_FD_READ | BSC_FD_EXCEPT;
466  link->connect.cb = m3ua_sctp_assoc_complete;
467 
468  if (osmo_fd_register(&link->connect) != 0) {
469  LOGP(DINP, LOGL_ERROR, "Failed to register fd\n");
470  goto error;
471  }
472 }

474 static int m3ua_write(struct mtp_link *mtp_link, struct msgb *msg)
475 {
476  struct mtp_m3ua_client_link *link = mtp_link->data;
477  struct sctp_sndrcvinfo info;
478  struct xua_msg *m3ua;
479  struct mtp_level_3_hdr *mtp_hdr;
480  struct m3ua_protocol_data proto_data;
481  uint8_t *proto_start;
482  uint32_t netappear;
483 
484  if (!link->asptm_active) {
485  LOGP(DINP, LOGL_ERROR, "ASP not ready for %d/%s of %d/%s.\n",
486  mtp_link->nr, mtp_link->name, mtp_link->set->nr,
487  mtp_link->set->name);
488  goto clean;
489  }
490 
491  /*
492  * TODO.. we could enhance the structure of mtp_link to
493  * have function pointers for operations like SLTM instead
494  * of doing what we do here.
495  * The entire m3ua episode (code + reading the spec) had a
496  * budget of < 2 man days so the amount of architecture changes
497  * we can do.
498  */
499 
500  /* TODO.. need to terminate MTPL3 locally... */
501 
502  /* TODO.. extract MTP information.. */
503  mtp_hdr = (struct mtp_level_3_hdr *) msg->l2h;
504  switch (mtp_hdr->ser_ind) {
505  case MTP_SI_MNT_SNM_MSG:
506  case MTP_SI_MNT_REG_MSG:
507  LOGP(DINP, LOGL_ERROR,
508  "Dropping SNM/REG message %d\n", mtp_hdr->ser_ind);
509  goto clean;
510  break;
511  case MTP_SI_MNT_ISUP:
512  case MTP_SI_MNT_SCCP:
513  default:
514  memset(&proto_data, 0, sizeof(proto_data));
515  proto_data.opc = htonl(__hack_opc);
516  proto_data.dpc = htonl(__hack_dpc);
517  proto_data.sls = MTP_LINK_SLS(mtp_hdr->addr);
518  proto_data.si = mtp_hdr->ser_ind;
519  proto_data.ni = mtp_link->set->ni;
520 
521  msg->l3h = mtp_hdr->data;
522  msgb_pull_to_l3(msg);
523 
524  netappear = htonl(1);
525 
526  proto_start = msgb_push(msg, sizeof(proto_data));
527  memcpy(proto_start, &proto_data, sizeof(proto_data));
528  break;
529  };
530 
531  m3ua = xua_msg_alloc();
532  if (!m3ua)
533  goto clean;
534 
535  mtp_handle_pcap(mtp_link, NET_OUT, msg->data, msg->len);
536 
537  m3ua->hdr.msg_class = M3UA_CLS_TRANS;
538  m3ua->hdr.msg_type = M3UA_TRANS_DATA;
539 
540  /*
541  * Modify the data...to create a true protocol data..
542  */
543  xua_msg_add_data(m3ua, M3UA_TAG_NET_APPEAR, sizeof(netappear), (uint8_t *)&netappear);
544  xua_msg_add_data(m3ua, M3UA_TAG_PROTO_DATA, msg->len, msg->data);
545 
546  memset(&info, 0, sizeof(info));
547  info.sinfo_stream = 1;
548  info.sinfo_assoc_id = 1;
549  info.sinfo_ppid = htonl(SCTP_PPID_M3UA);
550 
551  m3ua_conn_send(link, m3ua, &info);
552  xua_msg_free(m3ua);
553 
554 clean:
555  msgb_free(msg);
556  return 0;
557 }

908 {

910

915  struct xua_msg *result;
916

919  uint32_t routing_ctx;
920 
921  struct llist_head *entry;
922 
923  if (param->tag != M3UA_TAG_REG_RESULT) {
924  LOGP(DINP, LOGL_NOTICE, "Invalid param %i in REG_RSP, skipping...\n", param->tag);
925  continue;

927 
928  result = xua_from_part(param);
929  if (!result) {
930  LOGP(DINP, LOGL_ERROR, "Decoding REG_RESULT failed.\n");
931  continue;
932  }
933 
934  param = xua_msg_find_tag(result, M3UA_TAG_LOCAL_ROUT_KEY_IDENT);
935  if (!param) {
936  LOGP(DINP, LOGL_ERROR, "No LOCAL_ROUT_KEY_IDENT in REG_RSP result\n");
937  goto next;
938  }
939  if (param->len != 4) {
940  LOGP(DINP, LOGL_ERROR, "Bad length for LOCAL_ROUT_KEY_IDENT in REG_RSP result\n");
941  goto next;
942  }
943  memcpy(&local_rk_identifier, &param->dat[0], 4);
944 
945  llist_for_each(entry, &link->routes) {
946  route = llist_entry(entry, struct mtp_m3ua_reg_req, list);
947  if (route->local_rk_identifier == local_rk_identifier) break;
948  }
949 
950  /* Check if it's in our outstanding list */
951  if (!route) {
952  LOGP(DINP, LOGL_NOTICE, "No outstanding REG_REQ matching REG_RSP %u",
953  local_rk_identifier);
954  continue;
955  }
956 
957  param = xua_msg_find_tag(result, MUA_TAG_STATUS);
958  if (!param) {
959  LOGP(DINP, LOGL_ERROR, "No REG_STATUS in REG_RSP result.\n");
960  goto next;
961  }
962  if (param->len != 4) {
963  LOGP(DINP, LOGL_ERROR, "Bad length for REG_STATUS in REG_RSP result.\n");
964  goto next;
965  }
966  memcpy(&registration_status, &param->dat[0], 4);
967 
968  param = xua_msg_find_tag(result, MUA_TAG_ROUTING_CTX);
969  if (!param) {
970  LOGP(DINP, LOGL_ERROR, "No ROUTING_CTX in REG_RSP result.\n");
971  goto next;
972  }
973  if (param->len != 4) {
974  LOGP(DINP, LOGL_ERROR, "Bad length for ROUTING_CTX in REG_RSP result.\n");
975  goto next;
976  }
977  memcpy(&routing_ctx, &param->dat[0], 4);
978 
979  osmo_timer_del(&route->reg_rsp_timer); /* disarm timer */
980 
981  if (registration_status != 0) {
982  LOGP(DINP, LOGL_NOTICE, "REG_REQ RK ID %u failed. Registration status %i.\n",
983  local_rk_identifier, registration_status);
984 
985  llist_move_tail(entry, &link->routes_failed);
986  goto next;
987  }
988 
989  LOGP(DINP, LOGL_INFO, "REG_REQ RK ID %u succeeded.\n",
990  local_rk_identifier);
991 
992  route->reg_routing_context = routing_ctx;
993  llist_move_tail(entry, &link->routes_active);
994 
995  next:
996  xua_msg_free(result);
997  }
998 
999  /* No outstanding routes to process */
1000  if (llist_empty(&link->routes)) {
1001  LOGP(DINP, LOGL_NOTICE, "All REG_REQ complete.. sending ASPAC\n");
1002 
1003  m3ua_send_aspac(link);
1004  schedule_aspac_t_ack(link);
1005  }
1006  break;
1007  default:
1008  LOGP(DINP, LOGL_ERROR, "Unhandled msg_type %d\n", m3ua->hdr.msg_type);
1009  break;
1010  }
1011 }
