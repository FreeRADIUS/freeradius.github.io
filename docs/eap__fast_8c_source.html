244 static int eap_fast_verify(request_t *request, fr_tls_session_t *tls_session, uint8_t const *data, unsigned int data_len)
245 {
246  uint16_t attr;
247  uint16_t length;
248  unsigned int remaining = data_len;
249  int total = 0;
250  int num[EAP_FAST_TLV_MAX] = {0};
251  eap_fast_tunnel_t *t = talloc_get_type_abort(tls_session->opaque, eap_fast_tunnel_t);
252  uint32_t present = 0;
253 
254  fr_assert(sizeof(present) * 8 > EAP_FAST_TLV_MAX);
255 
256  while (remaining > 0) {
257  if (remaining < 4) {
258  RDEBUG2("EAP-FAST TLV is too small (%u) to contain a EAP-FAST TLV header", remaining);
259  return 0;
260  }
261 
262  memcpy(&attr, data, sizeof(attr));
263  attr = ntohs(attr) & EAP_FAST_TLV_TYPE;
264 
265  if ((attr == attr_eap_fast_result->attr) ||
266  (attr == attr_eap_fast_nak->attr) ||
267  (attr == attr_eap_fast_error->attr) ||
268  (attr == attr_eap_fast_vendor_specific->attr) ||
269  (attr == attr_eap_fast_eap_payload->attr) ||
270  (attr == attr_eap_fast_intermediate_result->attr) ||
271  (attr == attr_eap_fast_pac_tlv->attr) ||
272  (attr == attr_eap_fast_crypto_binding->attr)) {
273  num[attr]++;
274  present |= 1 << attr;
275 
276  if (num[attr_eap_fast_eap_payload->attr] > 1) {
277  REDEBUG("Too many EAP-Payload TLVs");
278 unexpected:
279  for (int i = 0; i < EAP_FAST_TLV_MAX; i++) {
280  if (present & (1 << i)) RDEBUG2(" - attribute %d is present", i);
281  }
282  eap_fast_send_error(tls_session, EAP_FAST_ERR_UNEXPECTED_TLV);
283  return 0;
284  }
285 
286  if (num[attr_eap_fast_intermediate_result->attr] > 1) {
287  REDEBUG("Too many Intermediate-Result TLVs");
288  goto unexpected;
289  }
290  } else {
291  if ((data[0] & 0x80) != 0) {
292  REDEBUG("Unknown mandatory TLV %02x", attr);
293  goto unexpected;
294  }
295 
296  num[0]++;
297  }
298 
299  total++;
300 
301  memcpy(&length, data + 2, sizeof(length));
302  length = ntohs(length);
303 
304  data += 4;
305  remaining -= 4;
306 
307  if (length > remaining) {
308  RDEBUG2("EAP-FAST TLV %u is longer than room remaining in the packet (%u > %u).", attr,
309  length, remaining);
310  return 0;
311  }
312 
313  /*
314  * If the rest of the TLVs are larger than
315  * this attribute, continue.
316  *
317  * Otherwise, if the attribute over-flows the end
318  * of the TLCs, die.
319  */
320  if (remaining < length) {
321  RDEBUG2("EAP-FAST TLV overflows packet!");
322  return 0;
323  }
324 
325  /*
326  * If there's an error, we bail out of the
327  * authentication process before allocating
328  * memory.
329  */
330  if ((attr == attr_eap_fast_intermediate_result->attr) || (attr == attr_eap_fast_result->attr)) {
331  uint16_t status;
332 
333  if (length < 2) {
334  REDEBUG("EAP-FAST TLV %u is too short. Expected 2, got %d", attr, length);
335  return 0;
336  }
337 
338  memcpy(&status, data, 2);
339  status = ntohs(status);
340 
341  if (status == EAP_FAST_TLV_RESULT_FAILURE) {
342  REDEBUG("EAP-FAST TLV %u indicates failure. Rejecting request", attr);
343  return 0;
344  }
345 
346  if (status != EAP_FAST_TLV_RESULT_SUCCESS) {
347  REDEBUG("EAP-FAST TLV %u contains unknown value. Rejecting request", attr);
348  goto unexpected;
349  }
350  }
351 
352  /*
353  * remaining > length, continue.
354  */
355  remaining -= length;
356  data += length;
357  }
358 
359  /*
360  * Check if the peer mixed & matched TLVs.
361  */
362  if ((num[attr_eap_fast_nak->attr] > 0) && (num[attr_eap_fast_nak->attr] != total)) {
363  REDEBUG("NAK TLV sent with non-NAK TLVs. Rejecting request");
364  goto unexpected;
365  }
366 
367  if (num[attr_eap_fast_intermediate_result->attr] > 0) {
368  REDEBUG("NAK TLV sent with non-NAK TLVs. Rejecting request");
369  goto unexpected;
370  }
371 
372  /*
373  * Check mandatory or not mandatory TLVs.
374  */
375  switch (t->stage) {
376  case EAP_FAST_TLS_SESSION_HANDSHAKE:
377  if (present) {
378  REDEBUG("Unexpected TLVs in TLS Session Handshake stage");
379  goto unexpected;
380  }
381  break;
382  case EAP_FAST_AUTHENTICATION:
383  if (present != (uint32_t)(1 << attr_eap_fast_eap_payload->attr)) {
384  REDEBUG("Unexpected TLVs in authentication stage");
385  goto unexpected;
386  }
387  break;
388  case EAP_FAST_CRYPTOBIND_CHECK:
389  {
390  uint32_t bits = (t->result_final)
391  ? 1 << attr_eap_fast_result->attr
392  : 1 << attr_eap_fast_intermediate_result->attr;
393  if (present & ~(bits | (1 << attr_eap_fast_crypto_binding->attr) | (1 << attr_eap_fast_pac_tlv->attr))) {
394  REDEBUG("Unexpected TLVs in cryptobind checking stage");
395  goto unexpected;
396  }
397  break;
398  }
399  case EAP_FAST_PROVISIONING:
400  if (present & ~((1 << attr_eap_fast_pac_tlv->attr) | (1 << attr_eap_fast_result->attr))) {
401  REDEBUG("Unexpected TLVs in provisioning stage");
402  goto unexpected;
403  }
404  break;
405  case EAP_FAST_COMPLETE:
406  if (present) {
407  REDEBUG("Unexpected TLVs in complete stage");
408  goto unexpected;
409  }
410  break;
411  default:
412  REDEBUG("Unexpected stage %d", t->stage);
413  return 0;
414  }
415 
416  /*
417  * We got this far. It looks OK.
418  */
419  return 1;
420 }

426 ssize_t eap_fast_decode_pair(TALLOC_CTX *ctx, fr_cursor_t *cursor, fr_dict_attr_t const *parent,
427  uint8_t const *data, size_t data_len,
428  void *decoder_ctx)
429 {

431  uint8_t const *p = data, *end = p + data_len;
432 
433  /*
434  * Decode the TLVs
435  */
436  while (p < end) {
437  ssize_t ret;
438  uint16_t attr;
439  uint16_t len;
440  fr_pair_t *vp;
441 
442  attr = fr_net_to_uint16(p) & EAP_FAST_TLV_TYPE;
443  p += 2;
444  len = fr_net_to_uint16(p);
445  p += 2;
446 
447  da = fr_dict_attr_child_by_num(parent, attr);
448  if (!da) {
449  MEM(vp = fr_pair_afrom_child_num(ctx, parent, attr));

452  p += (size_t) eap_fast_decode_pair(ctx, cursor, parent, p, len, decoder_ctx);

454 
455  } else {
456  MEM(vp = fr_pair_afrom_da(ctx, da));
457  }
458 
459  ret = fr_value_box_from_network(vp, &vp->data, vp->vp_type, vp->da, p, len, true);

461  fr_pair_to_unknown(vp);
462  fr_pair_value_memdup(vp, p, len, true);
463  }
464  fr_cursor_append(cursor, vp);
465  p += len;
466  }
467 
468  return p - data;
469 }

484 
485  /*
486  * If the response packet was Access-Accept, then
487  * we're OK. If not, die horribly.
488  *
489  * FIXME: EAP-Messages can only start with 'identity',
490  * NOT 'eap start', so we should check for that....

501  */

509  wrong_length:
510  REDEBUG("Found %s with incorrect length. Expected %u, got %zu",
511  vp->da->name, RADIUS_CHAP_CHALLENGE_LENGTH, vp->vp_length);
512  rcode = RLM_MODULE_INVALID;

517  break;
518

797 static FR_CODE eap_fast_process_tlvs(request_t *request, eap_session_t *eap_session,
798  fr_tls_session_t *tls_session, fr_pair_t *fast_vps)
799 {
800  eap_fast_tunnel_t *t = talloc_get_type_abort(tls_session->opaque, eap_fast_tunnel_t);
801  fr_pair_t *vp;
802  fr_cursor_t cursor;
803  eap_tlv_crypto_binding_tlv_t my_binding, *binding = NULL;
804 
805  memset(&my_binding, 0, sizeof(my_binding));
806 
807  for (vp = fr_cursor_init(&cursor, &fast_vps);
808  vp;

814  if (code == FR_CODE_ACCESS_ACCEPT) t->stage = EAP_FAST_CRYPTOBIND_CHECK;
815  } else if ((vp->da == attr_eap_fast_result) ||
816  (vp->da == attr_eap_fast_intermediate_result)) {
817  code = FR_CODE_ACCESS_ACCEPT;
818  t->stage = EAP_FAST_PROVISIONING;

822  }
823  } else if (vp->da->parent == attr_eap_fast_crypto_binding) {

828  */

831  binding->reserved = vp->vp_uint8;
832  break;
833  case 2: /* FR_EAP_FAST_CRYPTO_BINDING_VERSION */
834  binding->version = vp->vp_uint8;

836  case 3: /* FR_EAP_FAST_CRYPTO_BINDING_RECV_VERSION */

844  memcpy(binding->nonce, vp->vp_octets, vp->vp_length);
845  }
846  break;
847  case 6: /* FR_EAP_FAST_CRYPTO_BINDING_COMPOUND_MAC */
848  if (vp->vp_length >= sizeof(binding->compound_mac)) {
849  memcpy(binding->compound_mac, vp->vp_octets, sizeof(binding->compound_mac));
850  }
851  break;
852  }

854  } else if (vp->da->parent == attr_eap_fast_pac_tlv) {
855  if (vp->da == attr_eap_fast_pac_acknowledge) {

859  t->pac.expired = false;
860  t->stage = EAP_FAST_COMPLETE;
861  }
862  } else if (vp->da == attr_eap_fast_pac_info_pac_type) {

866  }

868  continue;
869  } else {
870  RDEBUG2("ignoring unknown EAP-FAST-PAC-TLV %pP", vp);
871  continue;
872  }
873  } else {
874  RDEBUG2("ignoring non-EAP-FAST TLV %pP", vp);
875  continue;
876  }
877

879  }
880

882  FR_CODE code = eap_fast_crypto_binding(request, eap_session, tls_session, binding);
883  if (code == FR_CODE_ACCESS_ACCEPT) {
884  t->stage = EAP_FAST_PROVISIONING;
885  }
886  return code;
887  }
888 
889  return FR_CODE_ACCESS_ACCEPT;
890 }

896 FR_CODE eap_fast_process(request_t *request, eap_session_t *eap_session, fr_tls_session_t *tls_session)
897 {
898  FR_CODE code;
899  fr_pair_t *fast_vps = NULL;
900  fr_cursor_t cursor;
901  uint8_t const *data;
902  size_t data_len;
903  eap_fast_tunnel_t *t;

906  * Just look at the buffer directly, without doing
907  * record_to_buff.
908  */

912

914 
915  /*
916  * See if the tunneled data is well formed.
917  */
918  if (!eap_fast_verify(request, tls_session, data, data_len)) return FR_CODE_ACCESS_REJECT;
919 
920  if (t->stage == EAP_FAST_TLS_SESSION_HANDSHAKE) {
921  fr_assert(t->mode == EAP_FAST_UNKNOWN);
922 
923  char buf[256];
924  if (strstr(SSL_CIPHER_description(SSL_get_current_cipher(tls_session->ssl),
925  buf, sizeof(buf)), "Au=None")) {
926  /* FIXME enforce MSCHAPv2 - RFC 5422 section 3.2.2 */
927  RDEBUG2("Using anonymous provisioning");
928  t->mode = EAP_FAST_PROVISIONING_ANON;
929  t->pac.send = true;
930  } else {
931  if (SSL_session_reused(tls_session->ssl)) {
932  RDEBUG2("Session Resumed from PAC");
933  t->mode = EAP_FAST_NORMAL_AUTH;
934  } else {
935  RDEBUG2("Using authenticated provisioning");
936  t->mode = EAP_FAST_PROVISIONING_AUTH;
937  }
938 
939  if (!t->pac.expires || t->pac.expired || t->pac.expires - time(NULL) < t->pac_lifetime * 0.6) {
940  t->pac.send = true;
941  }
942  }
943 
944  eap_fast_init_keys(request, tls_session);
945 
946  eap_fast_send_identity_request(request, tls_session, eap_session);
947 
948  t->stage = EAP_FAST_AUTHENTICATION;
949  return FR_CODE_ACCESS_CHALLENGE;
950  }
951 
952  fr_cursor_init(&cursor, &fast_vps);
953  if (eap_fast_decode_pair(request, &cursor, attr_eap_fast_tlv,
954  data, data_len, NULL) < 0) return FR_CODE_ACCESS_REJECT;
955 
956  RDEBUG2("Got Tunneled FAST TLVs");
957  log_request_pair_list(L_DBG_LVL_1, request, fast_vps, NULL);
958  code = eap_fast_process_tlvs(request, eap_session, tls_session, fast_vps);
959  fr_pair_list_free(&fast_vps);
960 
961  if (code == FR_CODE_ACCESS_REJECT) return FR_CODE_ACCESS_REJECT;
962 
963  switch (t->stage) {
964  case EAP_FAST_AUTHENTICATION:
965  code = FR_CODE_ACCESS_CHALLENGE;
966  break;
967 
968  case EAP_FAST_CRYPTOBIND_CHECK:
969  {
970  if (t->mode != EAP_FAST_PROVISIONING_ANON && !t->pac.send)
971  t->result_final = true;
972 
973  eap_fast_append_result(tls_session, code);
974 
975  eap_fast_update_icmk(request, tls_session, (uint8_t *)&t->isk);
976  eap_fast_append_crypto_binding(request, tls_session);
977 
978  code = FR_CODE_ACCESS_CHALLENGE;
979  break;
980  }
981  case EAP_FAST_PROVISIONING:
982  t->result_final = true;
983 
984  eap_fast_append_result(tls_session, code);
985 
986  if (t->pac.send) {
987  RDEBUG2("Peer requires new PAC");
988  eap_fast_send_pac_tunnel(request, tls_session);
989  code = FR_CODE_ACCESS_CHALLENGE;
990  break;
991  }
992 
993  t->stage = EAP_FAST_COMPLETE;
994  FALL_THROUGH;
995 
996  case EAP_FAST_COMPLETE:
997  /*
998  * RFC 5422 section 3.5 - Network Access after EAP-FAST Provisioning
999  */
1000  if (t->pac.type && t->pac.expired) {
1001  REDEBUG("Rejecting expired PAC.");
1002  code = FR_CODE_ACCESS_REJECT;
1003  break;
1004  }
1005 
1006  if (t->mode == EAP_FAST_PROVISIONING_ANON) {
1007  REDEBUG("Rejecting unauthenticated provisioning");
1008  code = FR_CODE_ACCESS_REJECT;
1009  break;
1010  }
1011 
1012  /*
1013  * eap_crypto_mppe_keys() is unsuitable for EAP-FAST as Cisco decided
1014  * it would be a great idea to flip the recv/send keys around
1015  */
1016  #define EAPTLS_MPPE_KEY_LEN 32
1017  eap_add_reply(request, attr_ms_mppe_recv_key, t->msk, EAPTLS_MPPE_KEY_LEN);
1018  eap_add_reply(request, attr_ms_mppe_send_key, &t->msk[EAPTLS_MPPE_KEY_LEN], EAPTLS_MPPE_KEY_LEN);
1019  eap_add_reply(request, attr_eap_msk, t->msk, EAP_FAST_KEY_LEN);
1020  eap_add_reply(request, attr_eap_emsk, t->emsk, EAP_EMSK_LEN);
1021 
1022  break;
1023 
1024  default:
1025  RERROR("Internal sanity check failed in EAP-FAST at %d", t->stage);
1026  code = FR_CODE_ACCESS_REJECT;
1027  }
1028 
1029  return code;
1030 }
