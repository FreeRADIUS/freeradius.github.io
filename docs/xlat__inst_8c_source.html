44  *

75 /** Destructor for xlat_thread_inst_t
76  *
77  * Calls thread_detach method if provided by xlat expansion
78  *
79  * @note This cannot be converted to a talloc destructor,
80  * as we need to call thread_detach *before* any of the children
81  * of the talloc ctx are freed.
82  */
83 static int _xlat_thread_inst_detach(xlat_thread_inst_t *thread_inst)
84 {
85  fr_assert(thread_inst->node->type == XLAT_FUNC);
86 
87  if (thread_inst->node->call.func->thread_detach) {
88  (void) thread_inst->node->call.func->thread_detach(thread_inst->data, thread_inst->node->call.func->thread_uctx);
89  }

91  return 0;

106 /** Create thread instances where needed
107  *
108  * @param[in] ctx to allocate thread instance data in.
109  * @param[in] inst to allocate thread-instance data for.
110  * @return
111  * - 0 on success. The node/thread specific data will be inserted
112  * into xlat_thread_inst_tree.
113  * - -1 on failure.
114  */
115 static xlat_thread_inst_t *xlat_thread_inst_alloc(TALLOC_CTX *ctx, xlat_inst_t *inst)
116 {
117  xlat_thread_inst_t *thread_inst = NULL;
118 
119  (void)talloc_get_type_abort(inst, xlat_inst_t);
120 
121  if (inst->node->call.func->thread_inst_size) {
122  MEM(thread_inst = talloc_zero_pooled_object(ctx, xlat_thread_inst_t,
123  1, inst->node->call.func->thread_inst_size));
124  } else {
125  MEM(thread_inst = talloc_zero(ctx, xlat_thread_inst_t));
126  }
127 
128  thread_inst->node = inst->node;
129 
130  fr_assert(inst->node->type == XLAT_FUNC);
131  fr_assert(!inst->node->call.thread_inst); /* May be missing inst, but this is OK */
132 
133  talloc_set_destructor(thread_inst, _xlat_thread_inst_detach);
134  if (inst->node->call.func->thread_inst_size) {
135  MEM(thread_inst->data = talloc_zero_array(thread_inst, uint8_t, inst->node->call.func->thread_inst_size));
136 
137  /*
138  * This is expensive, only do it if we might
139  * might be using it.
140  */
141 #ifndef TALLOC_GET_TYPE_ABORT_NOOP
142  talloc_set_name_const(thread_inst->data, inst->node->call.func->thread_inst_type);
143 #endif
144  }
145 
146  DEBUG4("Worker alloced xlat thread instance (%p/%p)", thread_inst, thread_inst->data);
147 
148  return thread_inst;
149 }

151 /** Destructor for xlat_inst_t
152  *
153  * Calls detach method if provided by xlat expansion
154  *
155  * @note This cannot be converted to a talloc destructor,
156  * as we need to call thread_detach *before* any of the children
157  * of the talloc ctx are freed.
158  */
159 static int _xlat_inst_detach(xlat_inst_t *inst)
160 {
161  (void)talloc_get_type_abort_const(inst->node, xlat_exp_t);
162  fr_assert(inst->node->type == XLAT_FUNC);
163 
164  /*
165  * Remove permanent data from the instance tree.
166  */
167  if (!inst->node->call.ephemeral) {
168  rbtree_deletebydata(xlat_inst_tree, inst);
169  if (rbtree_num_elements(xlat_inst_tree) == 0) TALLOC_FREE(xlat_inst_tree);
170  }
171 
172  if (inst->node->call.func->detach) (void) inst->node->call.func->detach(inst->data, inst->node->call.func->uctx);
173 
174  return 0;
175 }

186 /** Allocate instance data for an xlat expansion
187  *
188  * @param[in] node to allocate instance data for.
189  */
190 static xlat_inst_t *xlat_inst_alloc(xlat_exp_t *node)
191 {
192  xlat_inst_t *inst = NULL;
193 
194  (void)talloc_get_type_abort(node, xlat_exp_t);
195 
196  fr_assert(xlat_inst_tree); /* xlat_inst_init must have been called */
197  fr_assert(node->type == XLAT_FUNC);
198  fr_assert(!node->call.inst);
199 
200  if (node->call.func->inst_size) {
201  MEM(inst = talloc_zero_pooled_object(node, xlat_inst_t, 1, node->call.func->inst_size));
202  } else {
203  MEM(inst = talloc_zero(node, xlat_inst_t));
204  }
205 
206  inst->node = node;
207 
208  /*
209  * Instance data is freed when the
210  * node is freed.
211  */
212  talloc_set_destructor(inst, _xlat_inst_detach);
213  if (node->call.func->inst_size) {
214  MEM(inst->data = talloc_zero_array(inst, uint8_t, node->call.func->inst_size));
215 
216  /*
217  * This is expensive, only do it if we might
218  * might be using it.
219  */
220 #ifndef TALLOC_GET_TYPE_ABORT_NOOP
221  talloc_set_name_const(inst->data, node->call.func->inst_type);
222 #endif
223  }
224 
225  return inst;
226 }

228 /** Callback for creating "ephemeral" instance data for a #xlat_exp_t
229  *
230  * @note Epehemeral xlats must not be shared between requests.
231  *
232  * @param[in] node to create "ephemeral" instance data for.
233  * @param[in] uctx UNUSED.
234  * @return
235  * - 0 if instantiation functions were successful.
236  * - -1 if either instantiation function failed.
237  */
238 static int _xlat_instantiate_ephemeral_walker(xlat_exp_t *node, UNUSED void *uctx)
239 {
240  fr_assert(!node->call.inst && !node->call.thread_inst);
241 
242  node->call.inst = xlat_inst_alloc(node);
243  if (!node->call.inst) return -1;
244 
245  /*
246  * Instantiate immediately unlike permanent XLATs
247  * Where it's a separate phase.
248  */
249  if (node->call.func->instantiate &&
250  (node->call.func->instantiate(node->call.inst->data, node, node->call.func->uctx) < 0)) {
251  error:
252  TALLOC_FREE(node->call.inst);
253  return -1;
254  }
255 
256  /*
257  * Create a thread instance too.
258  */
259  node->call.thread_inst = xlat_thread_inst_alloc(node, node->call.inst);
260  if (!node->call.thread_inst) goto error;
261 
262  if (node->call.func->thread_instantiate &&
263  node->call.func->thread_instantiate(node->call.inst, node->call.thread_inst->data,
264  node, node->call.func->thread_uctx) < 0) goto error;
265 
266  /*
267  * Mark this up as an ephemeral node, so the destructors
268  * don't search for it in the xlat_inst_tree.
269  */
270  node->call.ephemeral = true;
271 
272  return 0;
273 }

289 static int _xlat_thread_instantiate(void *data, void *uctx)
290 {
291  xlat_thread_inst_t *thread_inst;
292  xlat_inst_t *inst = talloc_get_type_abort(data, xlat_inst_t);
293 
294  thread_inst = xlat_thread_inst_alloc(uctx, data);
295  if (!thread_inst) return -1;
296 
297  DEBUG3("Instantiating xlat \"%s\" node %p, instance %p, new thread instance %p",
298  inst->node->call.func->name, inst->node, inst, thread_inst);
299 
300  if (inst->node->call.func->thread_instantiate) {
301  int ret;
302 
303  ret = inst->node->call.func->thread_instantiate(inst->data, thread_inst->data,
304  inst->node, inst->node->call.func->thread_uctx);
305  if (ret < 0) {
306  talloc_free(thread_inst);
307  return -1;
308  }
309  }
310 
311  rbtree_insert(xlat_thread_inst_tree, thread_inst);
312 
313  return 0;
314 }

316 /** Retrieve xlat/thread specific instance data
317  *
318  * @param[in] node to find thread specific data for.
319  * @return

321  * - NULL if the xlat has no thread instance data (should not happen).
322  */
323 xlat_thread_inst_t *xlat_thread_instance_find(xlat_exp_t const *node)
324 {
325  xlat_thread_inst_t *found;
326 
327  fr_assert(xlat_thread_inst_tree);
328  fr_assert(node->type == XLAT_FUNC);
329 
330  if (node->call.ephemeral) return node->call.thread_inst;
331 
332  found = rbtree_finddata(xlat_thread_inst_tree, &(xlat_thread_inst_t){ .node = node });
333  fr_assert(found);
334

338 /** Create thread specific instance tree and create thread instances
339  *
340  * This should be called directly after the modules_thread_instantiate() function.
341  *
342  * Memory will be freed automatically when the thread exits.
343  *
344  * @param[in] ctx to bind instance tree lifetime to. Must not be
345  * shared between multiple threads.
346  * @return
347  * - 0 on success.
348  * - -1 on failure.
349  */
350 int xlat_thread_instantiate(TALLOC_CTX *ctx)
351 {
352  int ret;
353 
354  if (!xlat_inst_tree) return 0;
355 
356  if (!xlat_thread_inst_tree) {
357  MEM(xlat_thread_inst_tree = rbtree_talloc_alloc(ctx, _xlat_thread_inst_cmp,
358  xlat_thread_inst_t, _xlat_thread_inst_free, 0));
359  }
360 
361  /*
362  * Walk the inst tree, creating thread specific instances.
363  */
364  ret = rbtree_walk(xlat_inst_tree, RBTREE_PRE_ORDER, _xlat_thread_instantiate, xlat_thread_inst_tree);
365  if (ret < 0) {
366  TALLOC_FREE(xlat_thread_inst_tree);
367  return -1;
368  }
369 
370  return 0;
371 }

423 /** Callback for creating "permanent" instance data for a #xlat_exp_t
424  *
425  * This function records the #xlat_exp_t requiring instantiation but does
426  * not call the instantiation function. This is to allow for a clear separation
427  * between the module instantiation phase and the xlat instantiation phase.
428  *
429  * @param[in] node to create "permanent" instance data for.
430  * @return
431  * - 0 if instantiation functions were successful.
432  * - -1 if either instantiation function failed.
433  */
434 int xlat_bootstrap_func(xlat_exp_t *node)
435 {
436  bool ret;
437 
438  fr_assert(node->type == XLAT_FUNC);
439  fr_assert(!node->call.inst && !node->call.thread_inst);
440 
441  node->call.inst = xlat_inst_alloc(node);
442  if (!node->call.inst) return -1;
443 
444  DEBUG3("Instantiating xlat \"%s\" node %p, new instance %p", node->call.func->name, node, node->call.inst);
445 
446  ret = rbtree_insert(xlat_inst_tree, node->call.inst);
447  if (!fr_cond_assert(ret)) {
448  TALLOC_FREE(node->call.inst);
449  return -1;
450  }
451 
452  return 0;
453 }

460 /** Create instance data for "permanent" xlats
461  *
462  * @note This must only be used for xlats created during startup.
463  * IF THIS IS CALLED FOR XLATS TOKENIZED AT RUNTIME YOU WILL LEAK LARGE AMOUNTS OF MEMORY.
464  * USE xlat_instantiate_request() INSTEAD.
465  *
466  * @param[in] root of xlat tree to create instance data for.
