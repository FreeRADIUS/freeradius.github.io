31 {
32  fr_sbuff_t our_out = FR_SBUFF_NO_ADVANCE(out);
33 
34 #define FLAG_SET(_flag) if (flags->_flag) FR_SBUFF_IN_STRCPY_LITERAL_RETURN(&our_out, STRINGIFY(_flag)",")
35 
36  FLAG_SET(is_root);
37  FLAG_SET(is_unknown);
38  FLAG_SET(is_raw);
39  FLAG_SET(internal);
40  FLAG_SET(array);
41  FLAG_SET(has_value);
42  FLAG_SET(virtual);
43 
44  if (dict && !flags->extra && flags->subtype) {
45  FR_SBUFF_IN_STRCPY_RETURN(&our_out, fr_table_str_by_value(dict->subtype_table, flags->subtype, "?"));
46  }
47 
48  if (flags->length) FR_SBUFF_IN_SPRINTF_RETURN(&our_out, "length=%i,", flags->length);

52  FR_SBUFF_IN_STRCPY_LITERAL_RETURN(&our_out, "key,");
53  break;
54 
55  case FLAG_LENGTH_UINT16:
56  FR_SBUFF_IN_STRCPY_LITERAL_RETURN(&our_out, "length=uint16,");
57  break;
58 
59  default:
60  break;
61  }
62  }
63 
64  /*
65  * Print out the date precision.
66  */
67  if ((type == FR_TYPE_DATE) || (type == FR_TYPE_TIME_DELTA)) {
68  FR_SBUFF_IN_STRCPY_RETURN(&our_out,
69  fr_table_str_by_value(date_precision_table, flags->type_size, "?"));
70  }
71 
72  fr_sbuff_in_trim(&our_out, ',');
73 
74  return fr_sbuff_set(out, &our_out);

77 /** Build the da_stack for the specified DA and encode the path in OID form
78  *
79  * @param[out] out Where to write the OID.
80  * @param[in] ancestor If not NULL, only print OID portion between ancestor and da.
81  * @param[in] da to print OID string for.
82  * @return
83  * - >0 The number of bytes written to the buffer.
84  * - <= 0 The number of bytes we would have needed to write the
85  * next OID component.
86  */
87 ssize_t fr_dict_print_attr_oid(fr_sbuff_t *out, fr_dict_attr_t const *ancestor, fr_dict_attr_t const *da)
88 {
89  int i;
90  int depth = 0;
91  fr_da_stack_t da_stack;
92  fr_sbuff_t our_out = FR_SBUFF_NO_ADVANCE(out);
93 
94  /*
95  * If the ancestor and the DA match, there's
96  * no OID string to print.
97  */
98  if ((ancestor == da) || (da->depth == 0)) return 0;
99 
100  fr_proto_da_stack_build(&da_stack, da);
101 
102  if (ancestor) {
103  if (da_stack.da[ancestor->depth - 1] != ancestor) {
104  fr_strerror_printf("Attribute \"%s\" is not a descendent of \"%s\"", da->name, ancestor->name);
105  return -1;
106  }
107  depth = ancestor->depth;
108  }
109 
110  /*
111  * We don't print the ancestor, we print the OID
112  * between it and the da.
113  */
114  FR_SBUFF_IN_SPRINTF_RETURN(&our_out, "%u", da_stack.da[depth]->attr);
115  for (i = depth + 1; i < (int)da->depth; i++) FR_SBUFF_IN_SPRINTF_RETURN(&our_out, ".%u", da_stack.da[i]->attr);
116 
117  return fr_sbuff_set(out, &our_out);
118 }

127 {

139  name = "TLV";
140  break;
141 
142  case FR_TYPE_VENDOR:
143  name = "VENDOR";
144  break;
145 
146  case FR_TYPE_STRUCT:
147  name = "STRUCT";
148  break;
149 
150  case FR_TYPE_GROUP:
151  name = "GROUP";
152  break;
153 
154  default:
155  if (da->parent && da->parent->type == FR_TYPE_STRUCT) {
156  name = "MEMBER";
157  break;
158  }
159 
160  name = "ATTRIBUTE";
161  break;
162  }
163 
164  printf("%u%.*s%s \"%s\" vendor: %x (%u), num: %x (%u), type: %s, flags: %s\n", da->depth, depth,
165  "\t\t\t\t\t\t\t\t\t\t\t\t\t\t\t\t\t\t", name, da->name,
166  fr_dict_vendor_num_by_da(da), fr_dict_vendor_num_by_da(da), da->attr, da->attr,
167  fr_table_str_by_value(fr_value_box_type_table, da->type, "?Unknown?"), ctx->buff);
168 
169  return 0;
170 }
