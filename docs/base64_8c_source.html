183 /** Base 64 encode binary data
184  *
185  * Base64 encode IN array of size INLEN into OUT array of size OUTLEN.
186  *
187  * @param[out] out Where to write Base64 string.
188  * @param[in] outlen size of buffer including NULL byte.
189  * @param[in] in Data to encode.
190  * @param[in] inlen Length of data to encode.
191  * @return
192  * - Amount of data we wrote to the buffer.
193  * - -1 if output buffer was too small.
194  */
195 size_t fr_base64_encode(char *out, size_t outlen, uint8_t const *in, size_t inlen)
196 {
197  char *p = out;
198  size_t need = FR_BASE64_ENC_LENGTH(inlen) + 1;
199 
200  if (outlen < need) {
201  fr_strerror_printf("Output buffer too small, exected %zu bytes, got %zu bytes", need, outlen);
202  *out = '\0';
203  return -1;
204  }
205 
206  while (inlen) {
207  *p++ = fr_base64_str[(in[0] >> 2) & 0x3f];
208  *p++ = fr_base64_str[((in[0] << 4) + (--inlen ? in[1] >> 4 : 0)) & 0x3f];
209  *p++ = (inlen ? fr_base64_str[((in[1] << 2) + (--inlen ? in[2] >> 6 : 0)) & 0x3f] : '=');
210  *p++ = inlen ? fr_base64_str[in[2] & 0x3f] : '=';
211 
212  if (inlen) inlen--;
213  if (inlen) in += 3;
214  }
215 
216  p[0] = '\0';
217 
218  return p - out;
219 }

235 /* Decode base64 encoded input array.
236  *
237  * Decode base64 encoded input array IN of length INLEN to output array OUT that
238  * can hold *OUTLEN bytes. Return true if decoding was successful, i.e.
239  * if the input was valid base64 data, -1 otherwise.
240  *
241  * If *OUTLEN is too small, as many bytes as possible will be written to OUT.
242  * On return, *OUTLEN holds the length of decoded bytes in OUT.
243  *
244  * Note that as soon as any non-alphabet characters are encountered,
245  * decoding is stopped and -1 is returned.
246  *
247  * This means that, when applicable, you must remove any line terminators
248  * that is part of the data stream before calling this function.
249  *
250  * @param[out] out Where to write the decoded data.
251  * @param[in] outlen The length of the output buffer.
252  * @param[in] in Base64 string to decode.
253  * @param[in] inlen length of Base64 string.
254  * @return
255  * - <= 0 on failure. The offset where the decoding error occurred as a negative integer.
256  * - Length of decoded data.
257  */
258 ssize_t fr_base64_decode(uint8_t *out, size_t outlen, char const *in, size_t inlen)
259 {
260  uint8_t *out_p = out;
261  uint8_t *out_end = out + outlen;
262  char const *p = in, *q;
263  char const *end = p + inlen;
264 
265  /*
266  * Process complete 24bit quanta
267  */
268  while ((end - p) >= 4) {
269  if (!fr_is_base64(p[0]) || !fr_is_base64(p[1]) || !fr_is_base64(p[2]) || !fr_is_base64(p[3])) break;
270 
271  /*
272  * Check we have enough bytes to write out
273  * the 24bit quantum.
274  */
275  if ((out_end - out_p) <= 3) {
276  oob:
277  fr_strerror_printf("Output buffer too small, needed at least %zu bytes", outlen + 1);
278  return p - end;
279  }
280 
281  *out_p++ = ((fr_base64_sextet[us(p[0])] << 2) | (fr_base64_sextet[us(p[1])] >> 4));
282  *out_p++ = ((fr_base64_sextet[us(p[1])] << 4) & 0xf0) | (fr_base64_sextet[us(p[2])] >> 2);
283  *out_p++ = ((fr_base64_sextet[us(p[2])] << 6) & 0xc0) | fr_base64_sextet[us(p[3])];
284 
285  p += 4; /* 32bit input -> 24bit output */
286  }
287 
288  q = p;
289 
290  /*
291  * Find the first non-base64 char
292  */
293  while ((q < end) && fr_is_base64(*q)) q++;
294 
295  switch (q - p) {
296  case 0: /* Final quantum is 24 bits */
297  break;
298 
299  case 2: /* Final quantum is 8 bits */
300  if ((out_end - out_p) < 1) goto oob;
301  *out_p++ = ((fr_base64_sextet[us(p[0])] << 2) | (fr_base64_sextet[us(p[1])] >> 4));
302  p += 2;
303  break;
304 
305  case 3: /* Final quantum is 16 bits */
306  if ((out_end - out_p) < 2) goto oob;
307  *out_p++ = ((fr_base64_sextet[us(p[0])] << 2) | (fr_base64_sextet[us(p[1])] >> 4));
308  *out_p++ = ((fr_base64_sextet[us(p[1])] << 4) & 0xf0) | (fr_base64_sextet[us(p[2])] >> 2);
309  p += 3;
310  break;
311 
312  default:
313  fr_strerror_printf("Invalid base64 padding data");
314  return p - end;
315  }
316 
317  while (p < end) {
318  if (*p != '=') {
319  fr_strerror_printf("Found non-padding char '%c' at end of base64 string", *p);
320  return p - end;
321  }
322  p++;
323  }
324 
325  return out_p - out;
326 }
