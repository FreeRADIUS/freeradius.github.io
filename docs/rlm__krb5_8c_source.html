83 static int mod_instantiate(void *instance, CONF_SECTION *conf)
84 {
85  rlm_krb5_t *inst = instance;
86  krb5_error_code ret;
87 #ifndef HEIMDAL_KRB5
88  krb5_keytab keytab;
89  char keytab_name[200];
90  char *princ_name;
91 #endif
92 
93 #ifdef HEIMDAL_KRB5
94  DEBUG("Using Heimdal Kerberos library");
95 #else
96  DEBUG("Using MIT Kerberos library");
97 #endif
98 
99 
100  if (!krb5_is_thread_safe()) {
101 /*
102  * rlm_krb5 was built as threadsafe
103  */
104 #ifdef KRB5_IS_THREAD_SAFE
105  ERROR("Build time libkrb5 was threadsafe, but run time library claims not to be");
106  ERROR("Modify runtime linker path (LD_LIBRARY_PATH on most systems), to prefer threadsafe libkrb5");
107  return -1;
108 /*
109  * rlm_krb5 was not built as threadsafe
110  */
111 #else
112  fr_log(&default_log, L_WARN, __FILE__, __LINE__,
113  "libkrb5 is not threadsafe, recompile it with thread support enabled ("
114 # ifdef HEIMDAL_KRB5
115  "--enable-pthread-support"
116 # else
117  "--disable-thread-support=no"
118 # endif
119  ")");
120  WARN("rlm_krb5 will run in single threaded mode, performance may be degraded");
121  } else {
122  WARN("Build time libkrb5 was not threadsafe, but run time library claims to be");
123  WARN("Reconfigure and recompile rlm_krb5 to enable thread support");
124 #endif
125  }
126 
127  inst->name = cf_section_name2(conf);
128  if (!inst->name) inst->name = cf_section_name1(conf);
129 
130  ret = krb5_init_context(&inst->context);
131  if (ret) {
132  ERROR("Context initialisation failed: %s", rlm_krb5_error(inst, NULL, ret));
133 
134  return -1;
135  }
136 
137  /*
138  * Split service principal into service and host components
139  * they're needed to build the server principal in MIT,
140  * and to set the validation service in Heimdal.
141  */
142  if (inst->service_princ) {
143  size_t len;
144  /* Service principal appears to contain a host component */
145  inst->hostname = strchr(inst->service_princ, '/');
146  if (inst->hostname) {
147  len = (inst->hostname - inst->service_princ);
148  inst->hostname++;
149  } else {
150  len = strlen(inst->service_princ);
151  }
152 
153  if (len) {
154  inst->service = talloc_array(inst, char, (len + 1));
155  strlcpy(inst->service, inst->service_princ, len + 1);
156  }
157  }
158 
159 #ifdef HEIMDAL_KRB5
160  if (inst->hostname) DEBUG("Ignoring hostname component of service principal \"%s\", not "
161  "needed/supported by Heimdal", inst->hostname);
162 #else
163 
164  /*
165  * Convert the service principal string to a krb5 principal.
166  */
167  ret = krb5_sname_to_principal(inst->context, inst->hostname, inst->service, KRB5_NT_SRV_HST, &(inst->server));
168  if (ret) {
169  ERROR("Failed parsing service principal: %s", rlm_krb5_error(inst, inst->context, ret));
170 
171  return -1;
172  }
173 
174  ret = krb5_unparse_name(inst->context, inst->server, &princ_name);
175  if (ret) {
176  /* Uh? */
177  ERROR("Failed constructing service principal string: %s", rlm_krb5_error(inst, inst->context, ret));
178 
179  return -1;
180  }
181 
182  /*
183  * Not necessarily the same as the config item
184  */
185  DEBUG("Using service principal \"%s\"", princ_name);
186  krb5_free_unparsed_name(inst->context, princ_name);
187 
188  /*
189  * Setup options for getting credentials and verifying them
190  */
191  ret = krb5_get_init_creds_opt_alloc(inst->context, &(inst->gic_options)); /* For some reason the 'init' version
192  of this function is deprecated */
193  if (ret) {
194  ERROR("Couldn't allocate initial credential options: %s", rlm_krb5_error(inst, inst->context, ret));
195 
196  return -1;
197  }
198 
199  /*
200  * Perform basic checks on the keytab
201  */
202  ret = inst->keytabname ?
203  krb5_kt_resolve(inst->context, inst->keytabname, &keytab) :
204  krb5_kt_default(inst->context, &keytab);
205  if (ret) {
206  ERROR("Resolving keytab failed: %s", rlm_krb5_error(inst, inst->context, ret));
207 
208  return -1;
209  }
210 
211  ret = krb5_kt_get_name(inst->context, keytab, keytab_name, sizeof(keytab_name));
212  krb5_kt_close(inst->context, keytab);
213  if (ret) {
214  ERROR("Can't retrieve keytab name: %s", rlm_krb5_error(inst, inst->context, ret));
215 
216  return -1;
217  }
218 
219  DEBUG("Using keytab \"%s\"", keytab_name);
220 
221  MEM(inst->vic_options = talloc_zero(inst, krb5_verify_init_creds_opt));
222  krb5_verify_init_creds_opt_init(inst->vic_options);
223 #endif
224 
225 #ifdef KRB5_IS_THREAD_SAFE
226  /*
227  * Initialize the socket pool.
228  */
229  inst->pool = module_connection_pool_init(conf, inst, krb5_mod_conn_create, NULL, NULL, NULL, NULL);
230  if (!inst->pool) return -1;
231 #else
232  inst->conn = krb5_mod_conn_create(inst, inst, 0);
233  if (!inst->conn) return -1;
234 #endif
235  return 0;
236 }

238 /** Common function for transforming a User-Name string into a principal.
239  *
240  * @param[out] client Where to write the client principal.
241  * @param[in] inst of rlm_krb5.
242  * @param[in] request Current request.
243  * @param[in] context Kerberos context.
244  */
245 static rlm_rcode_t krb5_parse_user(krb5_principal *client, KRB5_UNUSED rlm_krb5_t const *inst, request_t *request,
246  krb5_context context)
247 {
248  krb5_error_code ret;
249  char *princ_name;
250  fr_pair_t *username;
251 
252  username = fr_pair_find_by_da(&request->request_pairs, attr_user_name);
253 
254  /*
255  * We can only authenticate user requests which HAVE
256  * a User-Name attribute.
257  */
258  if (!username) {
259  REDEBUG("Attribute \"User-Name\" is required for authentication");
260  return RLM_MODULE_FAIL;
261  }
262 
263  ret = krb5_parse_name(context, username->vp_strvalue, client);
264  if (ret) {
265  REDEBUG("Failed parsing username as principal: %s", rlm_krb5_error(inst, context, ret));
266 
267  return RLM_MODULE_FAIL;
268  }
269 
270  krb5_unparse_name(context, *client, &princ_name);
271  RDEBUG2("Using client principal \"%s\"", princ_name);
272 #ifdef HEIMDAL_KRB5
273  free(princ_name);
274 #else
275  krb5_free_unparsed_name(context, princ_name);
276 #endif
277  return RLM_MODULE_OK;
278 }

280 /** Log error message and return appropriate rcode
281  *
282  * Translate kerberos error codes into return codes.
283  * @param inst of rlm_krb5.
284  * @param request Current request.
285  * @param ret code from kerberos.
286  * @param conn used in the last operation.
287  */
288 static rlm_rcode_t krb5_process_error(rlm_krb5_t const *inst, request_t *request, rlm_krb5_handle_t *conn, int ret)
289 {
290  fr_assert(ret != 0);
291 
292  if (!fr_cond_assert(inst)) return RLM_MODULE_FAIL;
293  if (!fr_cond_assert(conn)) return RLM_MODULE_FAIL;; /* Silences warnings */
294 
295  switch (ret) {
296  case KRB5_LIBOS_BADPWDMATCH:
297  case KRB5KRB_AP_ERR_BAD_INTEGRITY:
298  REDEBUG("Provided password was incorrect (%i): %s", ret, rlm_krb5_error(inst, conn->context, ret));
299  return RLM_MODULE_REJECT;
300 
301  case KRB5KDC_ERR_KEY_EXP:
302  case KRB5KDC_ERR_CLIENT_REVOKED:
303  case KRB5KDC_ERR_SERVICE_REVOKED:
304  REDEBUG("Account has been locked out (%i): %s", ret, rlm_krb5_error(inst, conn->context, ret));
305  return RLM_MODULE_DISALLOW;
306 
307  case KRB5KDC_ERR_C_PRINCIPAL_UNKNOWN:
308  RDEBUG2("User not found (%i): %s", ret, rlm_krb5_error(inst, conn->context, ret));
309  return RLM_MODULE_NOTFOUND;
310 
311  default:
312  REDEBUG("Error verifying credentials (%i): %s", ret, rlm_krb5_error(inst, conn->context, ret));
313  return RLM_MODULE_FAIL;
314  }
315 }

322 static unlang_action_t CC_HINT(nonnull) mod_authenticate(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
323 {
324  rlm_krb5_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_krb5_t);
325  rlm_rcode_t rcode;
326  krb5_error_code ret;
327  rlm_krb5_handle_t *conn;
328  krb5_principal client = NULL;
329  fr_pair_t *password;
330 
331  password = fr_pair_find_by_da(&request->request_pairs, attr_user_password);
332 
333  if (!password) {
334  REDEBUG("Attribute \"User-Password\" is required for authentication");
335  RETURN_MODULE_INVALID;
336  }
337 
338  /*
339  * Make sure the supplied password isn't empty
340  */
341  if (password->vp_length == 0) {
342  REDEBUG("User-Password must not be empty");
343  RETURN_MODULE_INVALID;
344  }
345 
346  /*
347  * Log the password
348  */
349  if (RDEBUG_ENABLED3) {
350  RDEBUG("Login attempt with password \"%pV\"", &password->data);
351  } else {
352  RDEBUG2("Login attempt with password");
353  }
354 
355 # ifdef KRB5_IS_THREAD_SAFE
356  conn = fr_pool_connection_get(inst->pool, request);
357  if (!conn) RETURN_MODULE_FAIL;
358 # else
359  conn = inst->conn;
360 # endif
361 
362  rcode = krb5_parse_user(&client, inst, request, conn->context);
363  if (rcode != RLM_MODULE_OK) goto cleanup;
364 
365  /*
366  * Verify the user, using the options we set in instantiate
367  */
368  ret = krb5_verify_user_opt(conn->context, client, password->vp_strvalue, &conn->options);
369  if (ret) {
370  rcode = krb5_process_error(inst, request, conn, ret);
371  goto cleanup;
372  }
373 
374  /*
375  * krb5_verify_user_opt adds the credentials to the ccache
376  * we specified with krb5_verify_opt_set_ccache.
377  *
378  * To make sure we don't accumulate thousands of sets of
379  * credentials, remove them again here.
380  *
381  * @todo This should definitely be optional, which means writing code for the MIT
382  * variant as well.
383  */
384  {
385  krb5_cc_cursor cursor;
386  krb5_creds cred;
387 
388  krb5_cc_start_seq_get(conn->context, conn->ccache, &cursor);
389  for (ret = krb5_cc_next_cred(conn->context, conn->ccache, &cursor, &cred);
390  ret == 0;
391  ret = krb5_cc_next_cred(conn->context, conn->ccache, &cursor, &cred)) {
392  krb5_cc_remove_cred(conn->context, conn->ccache, 0, &cred);
393  }
394  krb5_cc_end_seq_get(conn->context, conn->ccache, &cursor);
395  }
396 
397 cleanup:
398  if (client) {
399  krb5_free_principal(conn->context, client);
400  }
401 
402 # ifdef KRB5_IS_THREAD_SAFE
403  fr_pool_connection_release(inst->pool, request, conn);
404 # endif
405  RETURN_MODULE_RCODE(rcode);
406 }

413 static unlang_action_t CC_HINT(nonnull) mod_authenticate(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
414 {
415  rlm_krb5_t const *inst = talloc_get_type_abort_const(mctx->instance, rlm_krb5_t);
416  rlm_rcode_t rcode;
417  krb5_error_code ret;
418 
419  rlm_krb5_handle_t *conn;
420 
421  krb5_principal client = NULL; /* actually a pointer value */
422  krb5_creds init_creds;
423  char *nonconst_password; /* compiler warnings */
424  fr_pair_t *password;
425 
426  password = fr_pair_find_by_da(&request->request_pairs, attr_user_password);
427 
428  if (!password) {
429  REDEBUG("Attribute \"User-Password\" is required for authentication");
430  RETURN_MODULE_INVALID;
431  }
432 
433  /*
434  * Make sure the supplied password isn't empty
435  */
436  if (password->vp_length == 0) {
437  REDEBUG("User-Password must not be empty");
438  RETURN_MODULE_INVALID;
439  }
440 
441  /*
442  * Log the password
443  */
444  if (RDEBUG_ENABLED3) {
445  RDEBUG("Login attempt with password \"%pV\"", &password->data);
446  } else {
447  RDEBUG2("Login attempt with password");
448  }
449 
450 # ifdef KRB5_IS_THREAD_SAFE
451  conn = fr_pool_connection_get(inst->pool, request);
452  if (!conn) RETURN_MODULE_FAIL;
453 # else
454  conn = inst->conn;
455 # endif
456 
457  /*
458  * Zero out local storage
459  */
460  memset(&init_creds, 0, sizeof(init_creds));
461 
462  /*
463  * Check we have all the required VPs, and convert the username
464  * into a principal.
465  */
466  rcode = krb5_parse_user(&client, inst, request, conn->context);
467  if (rcode != RLM_MODULE_OK) goto cleanup;
468 
469  /*
470  * Retrieve the TGT from the TGS/KDC and check we can decrypt it.
471  */
472  memcpy(&nonconst_password, &password->vp_strvalue, sizeof(nonconst_password));
473  RDEBUG2("Retrieving and decrypting TGT");
474  ret = krb5_get_init_creds_password(conn->context, &init_creds, client, nonconst_password,
475  NULL, NULL, 0, NULL, inst->gic_options);
476  if (ret) {
477  rcode = krb5_process_error(inst, request, conn, ret);
478  goto cleanup;
479  }
480 
481  RDEBUG2("Attempting to authenticate against service principal");
482  ret = krb5_verify_init_creds(conn->context, &init_creds, inst->server, conn->keytab, NULL, inst->vic_options);
483  if (ret) rcode = krb5_process_error(inst, request, conn, ret);
484 
485 cleanup:
486  if (client) krb5_free_principal(conn->context, client);
487  krb5_free_cred_contents(conn->context, &init_creds);
488 
489 # ifdef KRB5_IS_THREAD_SAFE
490  fr_pool_connection_release(inst->pool, request, conn);
491 # endif
492  RETURN_MODULE_RCODE(rcode);
493 }
