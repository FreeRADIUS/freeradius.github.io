55 /** Takes the object DN and BER encodes the data into the BER value which is used as part of the request
56  *
57  @verbatim
58  RequestBer contents:
59  clientVersion INTEGER
60  targetObjectDN OCTET STRING
61  @endverbatim
62  *
63  * @param[out] request_bv where to write the request BER value (must be freed with ber_bvfree).
64  * @param[in] dn to query for.
65  * @return
66  * - 0 on success.
67  * - < 0 on error.
68  */
69 static int ber_encode_request_data(char const *dn, struct berval **request_bv)
70 {
71  int err = 0;
72  int rc = 0;
73  BerElement *request_ber = NULL;
74 
75  if (!dn || !*dn) {
76  err = NMAS_E_INVALID_PARAMETER;
77  goto finish;
78  }
79 
80  /* Allocate a BerElement for the request parameters.*/
81  if ((request_ber = ber_alloc()) == NULL) {
82  err = NMAS_E_FRAG_FAILURE;
83  goto finish;
84  }
85 
86  rc = ber_printf(request_ber, "{io}", NMAS_LDAP_EXT_VERSION, dn, strlen(dn) + 1);
87  if (rc < 0) {
88  err = NMAS_E_FRAG_FAILURE;
89  goto finish;
90  }
91 
92  /*
93  * Convert the BER we just built to a berval that we'll
94  * send with the extended request.
95  */
96  if (ber_flatten(request_ber, request_bv) < 0) {
97  err = NMAS_E_FRAG_FAILURE;
98  goto finish;
99  }
100 
101 finish:
102  if (request_ber) ber_free(request_ber, 1);
103 
104  return err;
105 }

107 /** Converts the reply into server version and a return code
108  *
109  * This function takes the reply BER Value and decodes the NMAS server version and return code and if a non
110  * null retData buffer was supplied, tries to decode the the return data and length.
111  *
112  @verbatim
113  ResponseBer contents:
114  server_version INTEGER
115  error INTEGER
116  data OCTET STRING
117  @endverbatim
118  *
119  * @param[in] reply_bv reply data from extended request.
120  * @param[out] server_version that responded.
121  * @param[out] out data.
122  * @param[out] outlen Length of data written to out.
123  * @return
124  * - 0 on success.
125  * - < 0 on error.
126  */
127 static int ber_decode_login_data(struct berval *reply_bv, int *server_version, void *out, size_t *outlen)
128 {
129  int rc = 0;
130  int err = 0;
131  BerElement *reply_ber = NULL;
132 
133  fr_assert(out != NULL);
134  fr_assert(outlen != NULL);
135 
136  if ((reply_ber = ber_init(reply_bv)) == NULL) {
137  err = NMAS_E_SYSTEM_RESOURCES;
138  goto finish;
139  }
140 
141  rc = ber_scanf(reply_ber, "{iis}", server_version, &err, out, outlen);
142  if (rc == -1) {
143  err = NMAS_E_FRAG_FAILURE;
144  goto finish;
145  }
146 
147 finish:
148 
149  if (reply_ber) ber_free(reply_ber, 1);
150 
151  return err;
152 }

154 /** Attempt to retrieve the universal password from Novell eDirectory
155  *
156  * @param[in] ld LDAP handle.
157  * @param[in] dn of user we want to retrieve the password for.
158  * @param[out] password Where to write the retrieved password.
159  * @param[out] passlen Length of data written to the password buffer.
160  * @return
161  * - 0 on success.
162  * - < 0 on failure.
163  */
164 int fr_ldap_edir_get_password(LDAP *ld, char const *dn, char *password, size_t *passlen)
165 {
166  int err = 0;
167  struct berval *request_bv = NULL;
168  char *reply_oid = NULL;
169  struct berval *reply_bv = NULL;
170  int server_version;
171  size_t bufsize;
172  char buffer[256];
173 
174  /* Validate parameters. */
175  if (!dn || !*dn || !passlen || !ld) {
176  return NMAS_E_INVALID_PARAMETER;
177  }
178 
179  err = ber_encode_request_data(dn, &request_bv);
180  if (err) goto finish;
181 
182  /* Call the ldap_extended_operation (synchronously) */
183  err = ldap_extended_operation_s(ld, NMASLDAP_GET_PASSWORD_REQUEST, request_bv, NULL, NULL, &reply_oid, &reply_bv);
184  if (err) goto finish;
185 
186  /* Make sure there is a return OID */
187  if (!reply_oid) {
188  err = NMAS_E_NOT_SUPPORTED;
189  goto finish;
190  }
191 
192  /* Is this what we were expecting to get back. */
193  if (strcmp(reply_oid, NMASLDAP_GET_PASSWORD_RESPONSE) != 0) {
194  err = NMAS_E_NOT_SUPPORTED;
195  goto finish;
196  }
197 
198  /* Do we have a good returned berval? */
199  if (!reply_bv) {
200  /*
201  * No; returned berval means we experienced a rather
202  * drastic error. Return operations error.
203  */
204  err = NMAS_E_SYSTEM_RESOURCES;
205  goto finish;
206  }
207 
208  bufsize = sizeof(buffer);
209  err = ber_decode_login_data(reply_bv, &server_version, buffer, &bufsize);
210  if (err) goto finish;
211 
212  if (server_version != NMAS_LDAP_EXT_VERSION) {
213  err = NMAS_E_INVALID_VERSION;
214  goto finish;
215  }
216 
217  if (bufsize > *passlen) {
218  err = NMAS_E_BUFFER_OVERFLOW;
219  goto finish;
220  }
221 
222  memcpy(password, buffer, bufsize);
223  password[bufsize] = '\0';
224  *passlen = bufsize;
225 
226 finish:
227  if (reply_bv) {
228  ber_bvfree(reply_bv);
229  }
230 
231  /* Free the return OID string if one was returned. */
232  if (reply_oid) {
233  ldap_memfree(reply_oid);
234  }
235 
236  /* Free memory allocated while building the request ber and berval. */
237  if (request_bv) {
238  ber_bvfree(request_bv);
239  }
240 
241  return err;
242 }
