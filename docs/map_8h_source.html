59 /** Value pair map
60  *
61  * Value pair maps contain a pair of templates, that describe a src attribute
62  * or value, and a destination attribute.
63  *
64  * Neither src or dst need to be an FR attribute, and their type can be inferred
65  * from whether map->da is NULL (not FR).
66  *
67  * @see tmpl_t
68  */
69 struct vp_map_s {
70  tmpl_t *lhs; //!< Typically describes the attribute to add, modify or compare.
71  tmpl_t *rhs; //!< Typically describes a literal value or a src attribute
72  ///< to copy or compare.
73 
74  fr_token_t op; //!< The operator that controls insertion of the dst attribute.
75  fr_type_t cast; //!< Cast value to this type.
76 
77  CONF_ITEM *ci; //!< Config item that the map was created from. Mainly used for
78  //!< logging validation errors.
79 
80  map_t *child; //!< a child map. If it exists, `rhs` MUST be NULL
81  map_t *next; //!< The next valuepair map.
82 };
