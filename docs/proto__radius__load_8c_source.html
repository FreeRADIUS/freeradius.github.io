117 static ssize_t mod_read(fr_listen_t *li, void **packet_ctx, fr_time_t *recv_time_p, uint8_t *buffer, size_t buffer_len, size_t *leftover, UNUSED uint32_t *priority, UNUSED bool *is_dup)
118 {
119  proto_radius_load_t const *inst = talloc_get_type_abort_const(li->app_io_instance, proto_radius_load_t);
120  proto_radius_load_thread_t *thread = talloc_get_type_abort(li->thread_instance, proto_radius_load_thread_t);
121  fr_io_address_t *address, **address_p;
122 
123  size_t packet_len;
124 
125  if (thread->done) return -1;
126 
127  *leftover = 0; /* always for load generation */
128 
129  /*
130  * Where the addresses should go. This is a special case
131  * for proto_radius.
132  */
133  address_p = (fr_io_address_t **) packet_ctx;
134  address = *address_p;
135 
136  memset(address, 0, sizeof(*address));
137  address->socket.inet.src_ipaddr.af = AF_INET;
138  address->socket.inet.dst_ipaddr.af = AF_INET;
139  address->radclient = inst->client;
140 
141  *recv_time_p = thread->recv_time;
142 
143  if (buffer_len < inst->packet_len) {
144  DEBUG2("proto_radius_load read buffer is too small for input packet");
145  return 0;
146  }
147 
148  memcpy(buffer, inst->packet, inst->packet_len);
149  packet_len = inst->packet_len;
150 
151  // @todo - try for some variation in the packet
152 
153  /*
154  * The packet is always OK for RADIUS.
155  */
156 
157  /*
158  * proto_radius sets the priority
159  */
160 
161  /*
162  * Print out what we received.
163  */
164  DEBUG2("proto_radius_load - Received %s ID %d length %d %s",
165  fr_packet_codes[buffer[0]], buffer[1],
166  (int) packet_len, thread->name);
167 
168  return packet_len;
169 }

172 static ssize_t mod_write(fr_listen_t *li, UNUSED void *packet_ctx, fr_time_t request_time,
173  UNUSED uint8_t *buffer, size_t buffer_len, UNUSED size_t written)
174 {
175  proto_radius_load_thread_t *thread = talloc_get_type_abort(li->thread_instance, proto_radius_load_thread_t);
176  fr_load_reply_t state;
177 
178  /*
179  * @todo - share a stats interface with the parent? or
180  * put the stats in the listener, so that proto_radius
181  * can update them, too.. <sigh>
182  */
183  thread->stats.total_responses++;
184 
185  /*
186  * Tell the load generatopr subsystem that we have a
187  * reply. Then if the load test is done, exit the
188  * server.
189  */
190  state = fr_load_generator_have_reply(thread->l, request_time);
191  if (state == FR_LOAD_DONE) {
192  if (!thread->inst->repeat) {
193  thread->done = true;
194  } else {
195  (void) fr_load_generator_stop(thread->l); /* ensure l->ev is gone */
196  (void) fr_load_generator_start(thread->l);
197  }
198  }
199 
200  return buffer_len;
201 }

208 {

220 
221  memset(&ipaddr, 0, sizeof(ipaddr));
222  ipaddr.af = AF_INET;
223  li->app_io_addr = fr_socket_addr_alloc_inet_src(li, IPPROTO_UDP, 0, &ipaddr, 0);
224 
225  fr_assert((cf_parent(inst->cs) != NULL) && (cf_parent(cf_parent(inst->cs)) != NULL)); /* listen { ... } */
226 
227  thread->name = talloc_typed_asprintf(thread, "radius_load from filename %s", inst->filename ? inst->filename : "none");
228  thread->parent = talloc_parent(li);
229 
230  return 0;

369 static int mod_instantiate(void *instance, CONF_SECTION *cs)
370 {
371  proto_radius_load_t *inst = talloc_get_type_abort(instance, proto_radius_load_t);
372  RADCLIENT *client;
373 
374  bool done;
375  fr_pair_t *vp, *vps = NULL;
376  ssize_t packet_len;
377  int code = FR_CODE_ACCESS_REQUEST;
378 
379  inst->client = client = talloc_zero(inst, RADCLIENT);
380  if (!inst->client) return 0;
381 
382  client->ipaddr.af = AF_INET;
383  client->src_ipaddr = client->ipaddr;
384 
385  client->longname = client->shortname = inst->filename;
386  client->secret = talloc_strdup(client, "testing123");
387  client->nas_type = talloc_strdup(client, "load");
388  client->use_connected = false;
389 
390  if (inst->filename) {
391  FILE *fp;
392 
393  fp = fopen(inst->filename, "r");
394  if (!fp) {
395  cf_log_err(cs, "Failed reading %s - %s",
396  inst->filename, fr_syserror(errno));
397  return -1;
398  }
399 
400  if (fr_pair_list_afrom_file(inst, dict_radius, &vps, fp, &done) < 0) {
401  cf_log_perr(cs, "Failed reading %s", inst->filename);
402  fclose(fp);
403  return -1;
404  }
405 
406  fclose(fp);
407  }
408 
409  MEM(inst->packet = talloc_zero_array(inst, uint8_t, inst->max_packet_size));
410 
411  vp = fr_pair_find_by_da(&vps, attr_packet_type);
412  if (vp) code = vp->vp_uint32;
413 
414  /*
415  * Encode the packet.
416  */
417  packet_len = fr_radius_encode(inst->packet, inst->max_packet_size, NULL,
418  client->secret, talloc_array_length(client->secret),
419  code, 0, vps);
420  if (packet_len <= 0) {
421  cf_log_perr(cs, "Failed encoding packet from %s",
422  inst->filename);
423  return -1;
424  }
425 
426  inst->packet_len = packet_len;
427 
428  return 0;
429 }
