75 /*
76  * EAP packet format to be sent over the wire
77  *
78  * i.e. code+id+length+data where data = null/type+typedata
79  * based on code.
80  *
81  * INPUT to function is reply->code
82  * reply->id
83  * reply->type - setup with data
84  *
85  * OUTPUT reply->packet is setup with wire format, and will
86  * be allocated to the right size.
87  *
88  */
89 static int eap_wireformat(eap_packet_t *reply)
90 {
91  eap_packet_raw_t *header;
92  uint16_t total_length = 0;
93 
94  if (!reply) return 0;
95 
96  /*
97  * If reply->packet is set, then the wire format
98  * has already been calculated, just succeed.
99  */
100  if(reply->packet != NULL) return 0;
101 
102  total_length = EAP_HEADER_LEN;
103  if (reply->code < 3) {
104  total_length += 1/* EAP Method */;
105  if (reply->type.data && reply->type.length > 0) {
106  total_length += reply->type.length;
107  }
108  }
109 
110  reply->packet = talloc_array(reply, uint8_t, total_length);
111  header = (eap_packet_raw_t *)reply->packet;
112  if (!header) {
113  return -1;
114  }
115 
116  header->code = (reply->code & 0xFF);
117  header->id = (reply->id & 0xFF);
118 
119  total_length = htons(total_length);
120  memcpy(header->length, &total_length, sizeof(total_length));
121 
122  /*
123  * Request and Response packets are special.
124  */
125  if ((reply->code == FR_EAP_CODE_REQUEST) ||
126  (reply->code == FR_EAP_CODE_RESPONSE)) {
127  header->data[0] = (reply->type.num & 0xFF);
128 
129  /*
130  * Here since we cannot know the typedata format and length
131  *
132  * Type_data is expected to be wired by each EAP-Type
133  *
134  * Zero length/No typedata is supported as long as
135  * type is defined
136  */
137  if (reply->type.data && reply->type.length > 0) {
138  memcpy(&header->data[1], reply->type.data, reply->type.length);
139  talloc_free(reply->type.data);
140  reply->type.data = reply->packet + EAP_HEADER_LEN + 1/*EAPtype*/;
141  }
142  }
143 
144  return 0;
145 }

147 /*
148  * compose EAP reply packet in EAP-Message attr of RADIUS.
149  *
150  * Set the RADIUS reply codes based on EAP request codes. Append
151  * any additional VPs to RADIUS reply
152  */
153 rlm_rcode_t eap_compose(eap_session_t *eap_session)
154 {
155  fr_pair_t *vp;
156  eap_packet_raw_t *eap_packet;
157  request_t *request;
158  eap_round_t *eap_round;
159  eap_packet_t *reply;
160  int rcode;
161 
162  eap_session = talloc_get_type_abort(eap_session, eap_session_t);
163  request = talloc_get_type_abort(eap_session->request, request_t);
164  eap_round = talloc_get_type_abort(eap_session->this_round, eap_round_t);
165  reply = talloc_get_type_abort(eap_round->request, eap_packet_t);
166 
167  /*
168  * The Id for the EAP packet to the NAS wasn't set.
169  * Do so now.
170  */
171  if (!eap_round->set_request_id) {
172  /*
173  * Id serves to suppport request/response
174  * retransmission in the EAP layer and as such
175  * must be different for 'adjacent' packets
176  * except in case of success/failure-replies.
177  *
178  * RFC2716 (EAP-TLS) requires this to be
179  * incremented, RFC2284 only makes the above-
180  * mentioned restriction.
181  */
182  reply->id = eap_session->this_round->response->id;
183 
184  switch (reply->code) {
185  /*
186  * The Id is a simple "ack" for success
187  * and failure.
188  *
189  * RFC 3748 section 4.2 says
190  *
191  * ... The Identifier field MUST match
192  * the Identifier field of the Response
193  * packet that it is sent in response
194  * to.
195  */
196  case FR_EAP_CODE_SUCCESS:
197  case FR_EAP_CODE_FAILURE:
198  break;
199 
200  /*
201  * We've sent a response to their
202  * request, the Id is incremented.
203  */
204  default:
205  ++reply->id;
206  }
207  }
208 
209  /*
210  * For Request & Response packets, set the EAP sub-type,
211  * if the EAP sub-module didn't already set it.
212  *
213  * This allows the TLS module to be "morphic", and means
214  * that the TTLS and PEAP modules can call it to do most
215  * of their dirty work.
216  */
217  if (((eap_round->request->code == FR_EAP_CODE_REQUEST) ||
218  (eap_round->request->code == FR_EAP_CODE_RESPONSE)) &&
219  (eap_round->request->type.num == 0)) {
220  fr_assert(eap_session->type >= FR_EAP_METHOD_MD5);
221  fr_assert(eap_session->type < FR_EAP_METHOD_MAX);
222 
223  eap_round->request->type.num = eap_session->type;
224  }
225 
226  if (eap_wireformat(reply) < 0) return RLM_MODULE_INVALID;
227 
228  eap_packet = (eap_packet_raw_t *)reply->packet;
229 
230  MEM(pair_add_reply(&vp, attr_eap_message) >= 0);
231  vp->vp_length = eap_packet->length[0] * 256 + eap_packet->length[1];
232  vp->vp_octets = talloc_steal(vp, reply->packet);
233  reply->packet = NULL;
234 
235  /*
236  * EAP-Message is always associated with
237  * Message-Authenticator but not vice-versa.
238  *
239  * Don't add a Message-Authenticator if
240  * it's already there.
241  */
242  vp = fr_pair_find_by_da(&request->reply_pairs, attr_message_authenticator);
243  if (!vp) {
244  static uint8_t auth_vector[RADIUS_AUTH_VECTOR_LENGTH] = { 0x00 };
245 
246  MEM(pair_add_reply(&vp, attr_message_authenticator) >= 0);
247  fr_pair_value_memdup(vp, auth_vector, sizeof(auth_vector), false);
248  }
249 
250  /* Set request reply code, but only if it's not already set. */
251  rcode = RLM_MODULE_OK;
252  if (!request->reply->code) switch (reply->code) {
253  case FR_EAP_CODE_RESPONSE:
254  request->reply->code = FR_CODE_ACCESS_ACCEPT;
255  rcode = RLM_MODULE_HANDLED;
256  break;
257 
258  case FR_EAP_CODE_SUCCESS:
259  request->reply->code = FR_CODE_ACCESS_ACCEPT;
260  rcode = RLM_MODULE_OK;
261  break;
262 
263  case FR_EAP_CODE_FAILURE:
264  request->reply->code = FR_CODE_ACCESS_REJECT;
265  rcode = RLM_MODULE_REJECT;
266  break;
267 
268  case FR_EAP_CODE_REQUEST:
269  request->reply->code = FR_CODE_ACCESS_CHALLENGE;
270  rcode = RLM_MODULE_HANDLED;
271  break;
272 
273  default:
274  /*
275  * When we're pulling MS-CHAPv2 out of EAP-MS-CHAPv2,
276  * we do so WITHOUT setting a reply code, as the
277  * request is being proxied.
278  */
279  if (request->options & RAD_REQUEST_OPTION_PROXY_EAP) return RLM_MODULE_HANDLED;
280 
281  /* Should never enter here */
282  REDEBUG("Reply code %d is unknown, rejecting the request", reply->code);
283  request->reply->code = FR_CODE_ACCESS_REJECT;
284  reply->code = FR_EAP_CODE_FAILURE;
285  rcode = RLM_MODULE_REJECT;
286  break;
287  }
288 
289  RDEBUG2("Sending EAP %s (code %i) ID %d length %i",
290  eap_codes[eap_packet->code], eap_packet->code, reply->id,
291  eap_packet->length[0] * 256 + eap_packet->length[1]);
292 
293  return rcode;
294 }

297  * Radius criteria, EAP-Message is invalid without Message-Authenticator
298  * For EAP_START, send Access-Challenge with EAP Identity request.
299  */
300 rlm_rcode_t eap_start(request_t *request, rlm_eap_method_t const methods[], bool ignore_unknown_types)
301 {
302  fr_pair_t *vp;
303  fr_pair_t *eap_msg;
304 
305  eap_msg = fr_pair_find_by_da(&request->request_pairs, attr_eap_message);
306  if (!eap_msg) {
307  RDEBUG2("No EAP-Message, not doing EAP");
308  return RLM_MODULE_NOOP;
309  }
310 
311  /*
312  * Look for EAP-Type = None (FreeRADIUS specific attribute)
313  * this allows you to NOT do EAP for some users.
314  */
315  vp = fr_pair_find_by_da(&request->request_pairs, attr_eap_type);
316  if (vp && vp->vp_uint32 == 0) {
317  RDEBUG2("Found EAP-Message, but EAP-Type = None, so we're not doing EAP");
318  return RLM_MODULE_NOOP;
319  }
320 
321  /*
322  * http://www.freeradius.org/rfc/rfc2869.html#EAP-Message
323  *
324  * Checks for Message-Authenticator are handled by fr_radius_packet_recv().
325  */
326 
327  /*
328  * Check the length before de-referencing the contents.
329  *
330  * Lengths of zero are required by the RFC for EAP-Start,
331  * but we've never seen them in practice.
332  *
333  * Lengths of two are what we see in practice as
334  * EAP-Starts.
335  */
336  if ((eap_msg->vp_length == 0) || (eap_msg->vp_length == 2)) {
337  uint8_t *p;
338 
339  RDEBUG2("Got EAP_START message");
340 
341  MEM(pair_add_reply(&vp, attr_eap_message) >= 0);
342 
343  /*
344  * Manually create an EAP Identity request
345  */
346  MEM(fr_pair_value_mem_alloc(vp, &p, 5, false) == 0);
347  p[0] = FR_EAP_CODE_REQUEST;
348  p[1] = 0; /* ID */
349  p[2] = 0;
350  p[3] = 5; /* length */
351  p[4] = FR_EAP_METHOD_IDENTITY;
352 
353  return RLM_MODULE_HANDLED;
354  } /* end of handling EAP-Start */
355 
356  /*
357  * Supplicants don't usually send EAP-Failures to the
358  * server, but they're not forbidden from doing so.
359  * This behaviour was observed with a Spirent Avalanche test server.
360  */
361  if ((eap_msg->vp_length == EAP_HEADER_LEN) && (eap_msg->vp_octets[0] == FR_EAP_CODE_FAILURE)) {
362  REDEBUG("Peer sent EAP %s (code %i) ID %d length %zu",
363  eap_codes[eap_msg->vp_octets[0]],
364  eap_msg->vp_octets[0],
365  eap_msg->vp_octets[1],
366  eap_msg->vp_length);
367  return RLM_MODULE_FAIL;
368  /*
369  * The EAP packet header is 4 bytes, plus one byte of
370  * EAP sub-type. Short packets are discarded.
371  */
372  } else if (eap_msg->vp_length < (EAP_HEADER_LEN + 1)) {
373  RDEBUG2("Ignoring EAP-Message which is too short to be meaningful");
374  return RLM_MODULE_FAIL;
375  }
376 
377  /*
378  * Create an EAP-Type containing the EAP-type
379  * from the packet.
380  */
381  MEM(pair_add_request(&vp, attr_eap_type) >= 0);
382  vp->vp_uint32 = eap_msg->vp_octets[4];
383 
384  /*
385  * From now on, we're supposed to be handling the
386  * EAP packet. We better understand it...
387  */
388 
389  /*
390  * We're allowed only a few codes. Request, Response,
391  * Success, or Failure.
392  */
393  if ((eap_msg->vp_octets[0] == 0) ||
394  (eap_msg->vp_octets[0] >= FR_EAP_CODE_MAX)) {
395  RDEBUG2("Peer sent EAP packet with unknown code %i", eap_msg->vp_octets[0]);
396  } else {
397  RDEBUG2("Peer sent EAP %s (code %i) ID %d length %zu",
398  eap_codes[eap_msg->vp_octets[0]],
399  eap_msg->vp_octets[0],
400  eap_msg->vp_octets[1],
401  eap_msg->vp_length);
402  }
403 
404  /*
405  * We handle request and responses. The only other defined
406  * codes are success and fail. The client SHOULD NOT be
407  * sending success/fail packets to us, as it doesn't make
408  * sense.
409  */
410  if ((eap_msg->vp_octets[0] != FR_EAP_CODE_REQUEST) &&
411  (eap_msg->vp_octets[0] != FR_EAP_CODE_RESPONSE)) {
412  RDEBUG2("Ignoring EAP packet which we don't know how to handle");
413  return RLM_MODULE_FAIL;
414  }
415 
416  /*
417  * We've been told to ignore unknown EAP types, AND it's
418  * an unknown type. Return "NOOP", which will cause the
419  * mod_authorize() to return NOOP.
420  *
421  * EAP-Identity, Notification, and NAK are all handled
422  * internally, so they never have eap_sessions.
423  */
424  if ((eap_msg->vp_octets[4] >= FR_EAP_METHOD_MD5) &&
425  ignore_unknown_types &&
426  ((eap_msg->vp_octets[4] == 0) ||
427  (eap_msg->vp_octets[4] >= FR_EAP_METHOD_MAX) ||
428  (!methods[eap_msg->vp_octets[4]].submodule))) {
429  RDEBUG2("Ignoring Unknown EAP type");
430  return RLM_MODULE_NOOP;
431  }
432 
433  /*
434  * They're NAKing the EAP type we wanted to use, and
435  * asking for one which we don't support.
436  *
437  * NAK is code + id + length1 + length + NAK
438  * + requested EAP type(s).
439  *
440  * We know at this point that we can't handle the
441  * request. We could either return an EAP-Fail here, but
442  * it's not too critical.
443  *
444  * By returning "noop", we can ensure that authorize()
445  * returns NOOP, and another module may choose to proxy
446  * the request.
447  */
448  if ((eap_msg->vp_octets[4] == FR_EAP_METHOD_NAK) &&
449  (eap_msg->vp_length >= (EAP_HEADER_LEN + 2)) &&
450  ignore_unknown_types &&
451  ((eap_msg->vp_octets[5] == 0) ||
452  (eap_msg->vp_octets[5] >= FR_EAP_METHOD_MAX) ||
453  (!methods[eap_msg->vp_octets[5]].submodule))) {
454  RDEBUG2("Ignoring NAK with request for unknown EAP type");
455  return RLM_MODULE_NOOP;
456  }
457 
458  if ((eap_msg->vp_octets[4] == FR_EAP_METHOD_TTLS) ||
459  (eap_msg->vp_octets[4] == FR_EAP_METHOD_PEAP)) {
460  RDEBUG2("Continuing tunnel setup");
461  return RLM_MODULE_OK;
462  }
463  /*
464  * We return ok in response to EAP identity
465  * This means we can write:
466  *
467  * eap {
468  * ok = return
469  * }
470  * ldap
471  * sql
472  *
473  * ...in the inner-tunnel, to avoid expensive and unnecessary SQL/LDAP lookups
474  */
475  if (eap_msg->vp_octets[4] == FR_EAP_METHOD_IDENTITY) {
476  RDEBUG2("Peer sent EAP-Identity. Returning 'ok' so we can short-circuit the rest of authorize");
477  return RLM_MODULE_OK;
478  }
479 
480  /*
481  * Later EAP messages are longer than the 'start'
482  * message, so if everything is OK, this function returns
483  * 'no start found', so that the rest of the EAP code can
484  * use the State attribute to match this EAP-Message to
485  * an ongoing conversation.
486  */
487  RDEBUG2("Continuing on-going EAP conversation");
488 
489  return RLM_MODULE_NOTFOUND;
490 }

554 /*
555  * Create our Request-Response data structure with the eap packet
556  */
557 eap_round_t *eap_round_build(eap_session_t *eap_session, eap_packet_raw_t **eap_packet_p)
558 {
559  eap_round_t *eap_round = NULL;
560  int typelen;
561  eap_packet_raw_t *eap_packet = *eap_packet_p;
562  uint16_t len;
563 
564  eap_round = eap_round_alloc(eap_session);
565  if (eap_round == NULL) return NULL;
566 
567  eap_round->response->packet = (uint8_t *)eap_packet;
568  (void) talloc_steal(eap_round, eap_packet);
569  eap_round->response->code = eap_packet->code;
570  eap_round->response->id = eap_packet->id;
571  eap_round->response->type.num = eap_packet->data[0];
572 
573  memcpy(&len, eap_packet->length, sizeof(uint16_t));
574  len = ntohs(len);
575  eap_round->response->length = len;
576 
577  /*
578  * We've eaten the eap packet into the eap_round.
579  */
580  *eap_packet_p = NULL;
581 
582  /*
583  * First 5 bytes in eap, are code + id + length(2) + type.
584  *
585  * The rest is type-specific data. We skip type while
586  * getting typedata from data.
587  */
588  typelen = len - 5/*code + id + length + type */;
589  if (typelen > 0) {
590  /*
591  * Since the packet contains the complete
592  * eap_packet, typedata will be a ptr in packet
593  * to its typedata
594  */
595  eap_round->response->type.data = eap_round->response->packet + 5/*code+id+length+type*/;
596  eap_round->response->type.length = typelen;
597  } else {
598  eap_round->response->type.length = 0;
599  eap_round->response->type.data = NULL;
600  }
601 
602  return eap_round;
603 }
