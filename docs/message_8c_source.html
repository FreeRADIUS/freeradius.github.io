43 /** A Message set, composed of message headers and ring buffer data.
44  *
45  * A message set is intended to send short-lived messages. The
46  * message headers are fixed in size, and allocated from an array
47  * which is treated like a circular buffer. Message bodies (i.e. raw
48  * packets) are variable in size, and live in a separate ring buffer.
49  *
50  * The message set starts off with a small array of message headers,
51  * and a small ring buffer. If an array/buffer fills up, a new one
52  * is allocated at double the size of the previous one.
53  *
54  * The array / buffers are themselves kept in fixed-size arrays, of
55  * MSG_ARRAY_SIZE. The reason is that the memory for fr_message_set_t
56  * should be contiguous, and not in a linked list scattered in
57  * memory.
58  *
59  * The originator allocates a message, and sends it to a recipient.
60  * The recipient (usually in another thread) uses the message, and
61  * marks it as FR_MESSAGE_DONE. The originator then asynchronously
62  * cleans up the message.
63  *
64  * This asynchronous cleanup is done via self-clocking. If there is
65  * no need to clean up the messages, it isn't done. Only when we run
66  * out of space to store messages (or packets) is the cleanup done.
67  *
68  * This cleanup latency ensures that we don't have cache line
69  * bouncing, where the originator sends the message, and then while
70  * the recipieent is reading it... thrashes the cache line with
71  * checks for "are you done? Are you done?"
72  *
73  * If there are more than one used entry in either array, we then try
74  * to coalesce the buffers on cleanup. If we discover that one array
75  * is empty, we discard it, and move the used array entries into it's
76  * place.
77  *
78  * This process ensures that we don't have too many buffers in
79  * progress. It is better to have a few large buffers than many
80  * small ones.
81  *
82  * MSG_ARRAY_SIZE is defined to be large (16 doublings) to allow for
83  * the edge case where messages are stuck for long periods of time.
84  *
85  * With an initial message array size of 64, this gives us room for
86  * 2M packets, if *all* of the mr_array entries have packets stuck in
87  * them that aren't cleaned up for extended periods of time.
88  *
89  * @todo Add a flag for UDP-style protocols, where we can put the
90  * message into the ring buffer. This helps with locality of
91  * reference, and removes the need to track two separate things.
92  */
93 struct fr_message_set_s {
94  int mr_current; //!< current used message ring entry
95  int mr_max; //!< max used message ring entry
96 
97  size_t message_size; //!< size of the callers message, including fr_message_t
98 
99  int mr_cleaned; //!< where we last cleaned
100 
101  int rb_current; //!< current used ring buffer entry
102  int rb_max; //!< max used ring buffer entry
103 
104  size_t max_allocation; //!< maximum allocation size
105 
106  int allocated;
107  int freed;
108 
109  fr_ring_buffer_t *mr_array[MSG_ARRAY_SIZE]; //!< array of message arrays
110 
111  fr_ring_buffer_t *rb_array[MSG_ARRAY_SIZE]; //!< array of ring buffers
112 };

115 /** Create a message set
116  *
117  * @param[in] ctx the context for talloc
118  * @param[in] num_messages size of the initial message array. MUST be a power of 2.
119  * @param[in] message_size the size of each message, INCLUDING fr_message_t, which MUST be at the start of the struct
120  * @param[in] ring_buffer_size of the ring buffer. MUST be a power of 2.
121  * @return
122  * - NULL on error
123  * - newly allocated fr_message_set_t on success
124  */
125 fr_message_set_t *fr_message_set_create(TALLOC_CTX *ctx, int num_messages, size_t message_size, size_t ring_buffer_size)
126 {
127  fr_message_set_t *ms;
128 
129  /*
130  * Too small, or not a power of 2.
131  */
132  if (num_messages < 8) num_messages = 8;
133 
134  if ((num_messages & (num_messages - 1)) != 0) {
135  fr_strerror_printf("Number of messages must be a power of 2");
136  return NULL;
137  }
138 
139  if (message_size < sizeof(fr_message_t)) {
140  fr_strerror_printf("Message size must be at least %zd", sizeof(fr_message_t));
141  return NULL;
142  }
143 
144  if (message_size > 1024) {
145  fr_strerror_printf("Message size must be no larger than 1024");
146  return NULL;
147  }
148 
149  ms = talloc_zero(ctx, fr_message_set_t);
150  if (!ms) {
151  fr_strerror_printf("Failed allocating memory");
152  return NULL;
153  }
154 
155  message_size += 15;
156  message_size &= ~(size_t) 15;
157  ms->message_size = message_size;
158 
159  ms->rb_array[0] = fr_ring_buffer_create(ms, ring_buffer_size);
160  if (!ms->rb_array[0]) {
161  talloc_free(ms);
162  return NULL;
163  }
164  ms->rb_max = 0;
165 
166  ms->mr_array[0] = fr_ring_buffer_create(ms, num_messages * message_size);
167  if (!ms->mr_array[0]) {
168  talloc_free(ms);
169  return NULL;
170  }
171 
172  ms->max_allocation = ring_buffer_size / 2;
173 
174  return ms;
175 }

178 /** Mark a message as done
179  *
180  * Note that this call is usually done from a thread OTHER than the
181  * originator of the message. As such, the message is NOT actually
182  * freed. Instead, it is just marked as freed.
183  *
184  * @param[in] m the message to make as done.
185  * @return
186  * - <0 on error
187  * - 0 on success
188  */
189 int fr_message_done(fr_message_t *m)
190 {
191  fr_assert(m->status != FR_MESSAGE_FREE);
192  fr_assert(m->status != FR_MESSAGE_DONE);
193 
194  /*
195  * Mark a message as freed. The originator will take
196  * care of cleaning it up.
197  */
198  if (m->status == FR_MESSAGE_USED) {
199  m->status = FR_MESSAGE_DONE;
200  return 0;
201  }
202 
203  /*
204  * This message was localized, so we can free it via
205  * talloc.
206  */
207  if (m->status == FR_MESSAGE_LOCALIZED) {
208  talloc_free(m);
209  return 0;
210  }
211 
212  /*
213  * A catastrophic error.
214  */
215  fr_assert(0 == 1);
216 
217  fr_strerror_printf("Failed marking message as done");
218  return -1;
219 }

222 /** Localize a message by copying it to local storage
223  *
224  * This function "localizes" a message by copying it to local
225  * storage. In the case where the recipient of a message has to sit
226  * on it for a while, that blocks the originator from cleaning up the
227  * message. The recipient can then copy the message to local
228  * storage, so that the originator can clean it up.
229  *
230  * The localized message is marked as FR_MESSAGE_LOCALIZED, so that
231  * the recipient can call the normal fr_message_done() function to
232  * free it.
233  *
234  * @param[in] ctx the talloc context to use for localization
235  * @param[in] m the message to be localized
236  * @param[in] message_size the size of the message, including the fr_message_t
237  * @return
238  * - NULL on allocation errror
239  * - a newly localized message
240  */
241 fr_message_t *fr_message_localize(TALLOC_CTX *ctx, fr_message_t *m, size_t message_size)
242 {
243  fr_message_t *l;
244 
245  if (m->status != FR_MESSAGE_USED) {
246  fr_strerror_printf("Cannot localize message unless it is in use");
247  return NULL;
248  }
249 
250  if (message_size <= sizeof(fr_message_t)) {
251  fr_strerror_printf("Message size is too small");
252  return NULL;
253  }
254 
255  l = talloc_memdup(ctx, m, message_size);
256  if (!l) {
257  nomem:
258  fr_strerror_printf("Failed allocating memory");
259  return NULL;
260  }
261 
262  l->data = NULL;
263 
264  if (l->data_size) {
265  l->data = talloc_memdup(l, m->data, l->data_size);
266  if (!l->data) {
267  talloc_free(l);
268  goto nomem;
269  }
270  }
271 
272  l->status = FR_MESSAGE_LOCALIZED;
273 
274  /*
275  * After this change, "m" should not be used for
276  * anything.
277  */
278  m->status = FR_MESSAGE_DONE;
279 
280  /*
281  * Now clean up the other fields of the newly localized
282  * message.
283  */
284  l->rb = NULL;
285  l->rb_size = 0;
286 
287  return l;
288 }

291 /** Clean up messages in a message ring.
292  *
293  * Find the oldest messages which are marked FR_MESSAGE_DONE,
294  * and mark them FR_MESSAGE_FREE.
295  *
296  * FIXME: If we care, track which ring buffer is in use, and how
297  * many contiguous chunks we can free. Then, free the chunks at
298  * once, instead of piecemeal. Realistically tho... this will
299  * probably make little difference.
300  *
301  * @param[in] ms the message set
302  * @param[in] mr the message ring
303  * @param[in] max_to_clean maximum number of messages to clean at a time.
304  */
305 static int fr_message_ring_gc(fr_message_set_t *ms, fr_ring_buffer_t *mr, int max_to_clean)
306 {
307  int messages_cleaned = 0;
308  size_t size;
309  fr_message_t *m;
310 
311  while (true) {
312  (void) fr_ring_buffer_start(mr, (uint8_t **) &m, &size);
313  if (size == 0) break;
314 
315  fr_assert(m != NULL);
316  fr_assert(size >= ms->message_size);
317 
318 
319  fr_assert(m->status != FR_MESSAGE_FREE);
320  if (m->status != FR_MESSAGE_DONE) break;
321 
322  messages_cleaned++;
323  m->status = FR_MESSAGE_FREE;
324  ms->freed++;
325 
326  if (m->rb) {
327  (void) fr_ring_buffer_free(m->rb, m->rb_size);
328 #ifndef NDEBUG
329  memset(m, 0, ms->message_size);
330 #endif
331  }
332 
333  fr_ring_buffer_free(mr, ms->message_size);
334 
335  if (messages_cleaned >= max_to_clean) break;
336  }
337 
338  MPRINT("CLEANED %d (%p) left\n", messages_cleaned, mr);
339  return messages_cleaned;
340 }

343 /** Garbage collect "done" messages.
344  *
345  * Called only from the originating thread. We also clean a limited
346  * number of messages at a time, so that we don't have sudden latency
347  * spikes when cleaning 1M messages.
348  *
349  * @param[in] ms the message set
350  * @param[in] max_to_clean the maximum number of messages to clean
351  */
352 static void fr_message_gc(fr_message_set_t *ms, int max_to_clean)
353 {
354  int i;
355  int arrays_freed, arrays_used, empty_slot;
356  int largest_free_slot;
357  int total_cleaned;
358  size_t largest_free_size;
359 
360  /*
361  * Clean up "done" messages.
362  */
363  total_cleaned = 0;
364 
365  /*
366  * Garbage collect the smaller buffers first.
367  */
368  for (i = 0; i <= ms->mr_max; i++) {
369  int cleaned;
370 
371  cleaned = fr_message_ring_gc(ms, ms->mr_array[i], max_to_clean - total_cleaned);
372  total_cleaned += cleaned;
373  fr_assert(total_cleaned <= max_to_clean);
374 
375  /*
376  * Stop when we've reached our GC limit.
377  */
378  if (total_cleaned == max_to_clean) break;
379  }
380 
381  /*
382  * Couldn't GC anything. Don't do more work.
383  */
384  if (total_cleaned == 0) return;
385 
386  arrays_freed = 0;
387  arrays_used = 0;
388 
389  /*
390  * Keep the two largest message buffers (used or not),
391  * and free all smaller ones which are empty.
392  */
393  for (i = ms->mr_max; i >= 0; i--) {
394  fr_assert(ms->mr_array[i] != NULL);
395 
396  if (arrays_used < 2) {
397  MPRINT("\tleaving entry %d alone\n", i);
398  arrays_used++;
399  continue;
400  }
401 
402  /*
403  * If the message ring buffer is empty, check if
404  * we should perhaps delete it.
405  */
406  if (fr_ring_buffer_used(ms->mr_array[i]) == 0) {
407  MPRINT("\tfreeing entry %d\n", i);
408  TALLOC_FREE(ms->mr_array[i]);
409  arrays_freed++;
410  continue;
411  }
412 
413  MPRINT("\tstill in use entry %d\n", i);
414  }
415 
416  /*
417  * Some entries have been freed. We need to coalesce the
418  * remaining entries.
419  */
420  if (arrays_freed) {
421  MPRINT("TRYING TO PACK from %d free arrays out of %d\n", arrays_freed, ms->rb_max + 1);
422 
423  empty_slot = -1;
424 
425  /*
426  * Pack the rb array by moving used entries to
427  * the bottom of the array.
428  */
429  for (i = 0; i <= ms->mr_max; i++) {
430  int j;
431 
432  /*
433  * Skip over empty entries, but set
434  * "empty_slot" to the first empty on we
435  * found.
436  */
437  if (!ms->mr_array[i]) {
438  if (empty_slot < 0) empty_slot = i;
439 
440  continue;
441  }
442 
443  /*
444  * This array entry is used, but there is
445  * no empty slot to put it into. Ignore
446  * it, and continue
447  */
448  if (empty_slot < 0) continue;
449 
450  fr_assert(ms->mr_array[empty_slot] == NULL);
451 
452  ms->mr_array[empty_slot] = ms->mr_array[i];
453  ms->mr_array[i] = NULL;
454 
455  /*
456  * Find the next empty slot which is
457  * greater than the one we just used.
458  */
459  for (j = empty_slot + 1; j <= i; j++) {
460  if (!ms->mr_array[j]) {
461  empty_slot = j;
462  break;
463  }
464  }
465  }
466 
467  /*
468  * Lower max, and set current to the largest
469  * array, whether or not it's used.
470  */
471  ms->mr_max -= arrays_freed;
472  ms->mr_current = ms->mr_max;
473 
474 #ifndef NDEBUG
475  MPRINT("NUM RB ARRAYS NOW %d\n", ms->mr_max + 1);
476  for (i = 0; i <= ms->mr_max; i++) {
477  MPRINT("\t%d %p\n", i, ms->mr_array[i]);
478  fr_assert(ms->mr_array[i] != NULL);
479  }
480 #endif
481  }
482 
483  /*
484  * And now we do the same thing for the ring buffers.
485  * Except that freeing the messages above also cleaned up
486  * the contents of each ring buffer, so all we need to do
487  * is find the largest empty ring buffer.
488  *
489  * We do this by keeping the two largest ring buffers
490  * (used or not), and then freeing all smaller ones which
491  * are empty.
492  */
493  arrays_used = 0;
494  arrays_freed = 0;
495  MPRINT("TRYING TO FREE ARRAYS %d\n", ms->rb_max);
496  for (i = ms->rb_max; i >= 0; i--) {
497  fr_assert(ms->rb_array[i] != NULL);
498 
499  if (arrays_used < 2) {
500  MPRINT("\tleaving entry %d alone\n", i);
501  arrays_used++;
502  continue;
503  }
504 
505  if (fr_ring_buffer_used(ms->rb_array[i]) == 0) {
506  MPRINT("\tfreeing entry %d\n", i);
507  TALLOC_FREE(ms->rb_array[i]);
508  arrays_freed++;
509  continue;
510  }
511 
512  MPRINT("\tstill in use entry %d\n", i);
513  }
514 
515  /*
516  * Pack the array entries back down.
517  */
518  if (arrays_freed > 0) {
519  MPRINT("TRYING TO PACK from %d free arrays out of %d\n", arrays_freed, ms->rb_max + 1);
520 
521  empty_slot = -1;
522 
523  /*
524  * Pack the rb array by moving used entries to
525  * the bottom of the array.
526  */
527  for (i = 0; i <= ms->rb_max; i++) {
528  int j;
529 
530  /*
531  * Skip over empty entries, but set
532  * "empty_slot" to the first empty on we
533  * found.
534  */
535  if (!ms->rb_array[i]) {
536  if (empty_slot < 0) empty_slot = i;
537 
538  continue;
539  }
540 
541  /*
542  * This array entry is used, but there is
543  * no empty slot to put it into. Ignore
544  * it, and continue
545  */
546  if (empty_slot < 0) continue;
547 
548  fr_assert(ms->rb_array[empty_slot] == NULL);
549 
550  ms->rb_array[empty_slot] = ms->rb_array[i];
551  ms->rb_array[i] = NULL;
552 
553  /*
554  * Find the next empty slot which is
555  * greater than the one we just used.
556  */
557  for (j = empty_slot + 1; j <= i; j++) {
558  if (!ms->rb_array[j]) {
559  empty_slot = j;
560  break;
561  }
562  }
563  }
564 
565  /*
566  * Lower max, and set current to the largest
567  * array, whether or not it's used.
568  */
569  ms->rb_max -= arrays_freed;
570  ms->rb_current = ms->rb_max;
571 
572 #ifndef NDEBUG
573  MPRINT("NUM RB ARRAYS NOW %d\n", ms->rb_max + 1);
574  for (i = 0; i <= ms->rb_max; i++) {
575  MPRINT("\t%d %p\n", i, ms->rb_array[i]);
576  fr_assert(ms->rb_array[i] != NULL);
577  }
578 #endif
579  }
580 
581  /*
582  * Set the current ring buffer to the one with the
583  * largest free space in it.
584  *
585  * This is different from the allocation strategy for
586  * messages.
587  */
588  if (!fr_cond_assert(ms->rb_array[ms->rb_max] != NULL)) return;
589 
590  largest_free_slot = ms->rb_max;
591  largest_free_size = (fr_ring_buffer_size(ms->rb_array[ms->rb_max]) -
592  fr_ring_buffer_used(ms->rb_array[ms->rb_max]));
593 
594  for (i = 0; i < ms->rb_max; i++) {
595  size_t free_size;
596 
597  fr_assert(ms->rb_array[i] != NULL);
598 
599  free_size = (fr_ring_buffer_size(ms->rb_array[i]) -
600  fr_ring_buffer_used(ms->rb_array[i]));
601  if (largest_free_size < free_size) {
602  largest_free_slot = i;
603  largest_free_size = free_size;
604  }
605  }
606 
607  ms->rb_current = largest_free_slot;
608  fr_assert(ms->rb_current >= 0);
609  fr_assert(ms->rb_current <= ms->rb_max);
610 }

612 /** Allocate a message from a message ring.
613  *
614  * The newly allocated message is zeroed.
615  *
616  * @param[in] ms the message set
617  * @param[in] mr the message ring to allocate from
618  * @param[in] clean whether to clean the message ring
619  * @return
620  * - NULL on failed allocation
621  * - fr_message_t* on successful allocation.
622  */
623 static fr_message_t *fr_message_ring_alloc(fr_message_set_t *ms, fr_ring_buffer_t *mr, bool clean)
624 {
625  fr_message_t *m;
626 
627  /*
628  * We're at the start of a buffer with data, and there's
629  * no room. Do a quick check to see if we can free up
630  * the oldest entry. If not, return.
631  *
632  * Otherwise, fall through to allocating a entry, of
633  * which there must now be at least one free one.
634  *
635  * This check results in a small amount of cache line
636  * thrashing. But if the buffer is full, it's likely
637  * that the oldest entry can be freed. If not, we have a
638  * small amount of cache thrashing, which should be
639  * extremely rare.
640  */
641  if (clean) {
642  if (fr_message_ring_gc(ms, mr, 4) == 0) {
643  fr_strerror_printf("No free memory after GC attempt");
644  return NULL;
645  }
646 
647  /*
648  * Else we cleaned up some entries in this array.
649  * Go allocate a message.
650  */
651  }
652 
653  /*
654  * Grab a new message from the underlying ring buffer.
655  */
656  m = (fr_message_t *) fr_ring_buffer_alloc(mr, ms->message_size);
657  if (!m) return NULL;
658 
659 #ifndef NDEBUG
660  memset(m, 0, ms->message_size);
661 #endif
662  m->status = FR_MESSAGE_USED;
663  return m;
664 }

666 /** Allocate a fr_message_t, WITHOUT a ring buffer.
667  *
668  * @param[in] ms the message set
669  * @param[out] p_cleaned a flag to indicate if we cleaned the message array
670  * @return
671  * - NULL on error
672  * - fr_message_t* on success
673  */
674 static fr_message_t *fr_message_get_message(fr_message_set_t *ms, bool *p_cleaned)
675 {
676  int i;
677  fr_message_t *m;
678  fr_ring_buffer_t *mr;
679 
680  ms->allocated++;
681  *p_cleaned = false;
682 
683  /*
684  * Grab the current message array. In the general case,
685  * there's room, so we grab a message and go find a ring
686  * buffer.
687  */
688  mr = ms->mr_array[ms->mr_current];
689  m = (fr_message_t *) fr_ring_buffer_alloc(mr, ms->message_size);
690  if (m) {
691  memset(m, 0, ms->message_size);
692  m->status = FR_MESSAGE_USED;
693  MPRINT("ALLOC normal\n");
694  return m;
695  }
696 
697  MPRINT("CLEANING UP (%zd - %zd = %zd)\n", ms->allocated, ms->freed,
698  ms->allocated - ms->freed);
699 
700  /*
701  * Else the buffer is full. Do a global cleanup.
702  */
703  fr_message_gc(ms, 128);
704  *p_cleaned = true;
705 
706  /*
707  * If we're lucky, the cleanup has given us a new
708  * "current" buffer, which is empty. If so, use it.
709  */
710  mr = ms->mr_array[ms->mr_current];
711  m = fr_message_ring_alloc(ms, mr, true);
712  if (m) {
713  MPRINT("ALLOC after cleanup\n");
714  return m;
715  }
716 
717  /*
718  * We've tried two allocations, and both failed. Brute
719  * force over all arrays, trying to allocate one
720  * somewhere... anywhere. We start from the largest

1026  *
1027  * This function allocates a previously reserved message, and then
1028  * reserves a new message.
1029  *
1030  * The application should call fr_message_reserve() with a large
1031  * buffer, and then read data into the buffer. If the buffer
1032  * contains multiple packets, the application should call
1033  * fr_message_alloc_reserve() repeatedly to allocate the full
1034  * packets, while reserving room for the partial packet.
1035  *
1036  * When the application is determines that there is only one full
1037  * packet, and one partial packet in the buffer, it should call this
1038  * function with actual_packet_size, and a large reserve_size. The
1039  * partial packet will be reserved. If the ring buffer is full, the
1040  * partial packet will be copied to a new ring buffer.
1041  *
1042  * When the application determines that there are multiple full
1043  * packets in the buffer, it should call this function with
1044  * actual_packet_size for each buffer, and reserve_size which
1045  * reserves all of the data in the buffer. i.e. the full packets +
1046  * partial packets, which should start off as the original
1047  * reserve_size.

1050  * while decreasing reserve_size by each actual_packet_size that was
1051  * allocated. Once there is only one full and a partial packet in
1052  * the buffer, it should use a large reserve_size, as above.

1055  * reserve_size, at the cost of substantially more memcpy()s.
