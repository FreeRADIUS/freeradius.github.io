83 /** Extract the MCC and MCN from the 3GPP domain
84  *
85  * 3GPP Root NAI domain format wlan.mnc<MNC>.mcc<MCC>.3gppnetwork.org.
86  *
87  * @param[out] mnc Mobile network code.
88  * @param[out] mcc Mobile country code.
89  * @param[in] domain to parse.
90  * @param[in] domain_len Length of the domain component.
91  * @return
92  * - number of bytes parsed.
93  * - <= 0 on error - The negative offset of where parsing failed.
94  */
95 ssize_t fr_aka_sim_3gpp_root_nai_domain_mcc_mnc(uint16_t *mnc, uint16_t *mcc,
96  char const *domain, size_t domain_len)
97 {
98  char const *p = domain, *end = p + domain_len;
99  char *q;
100  unsigned long num;
101 
102  if (((p + 8) < end) || (CRYPTO_memcmp(p, "wlan.mnc", 8) != 0)) return -1;
103  p += 8;
104 
105  if (((p + 3) < end)) {
106  fr_strerror_printf("Missing MNC component");
107  return (domain - p);
108  }
109  num = strtoul(p, &q, 10);
110  if (*q != '.') {
111  fr_strerror_printf("Invalid MCN component");
112  return (domain - q);
113  }
114  *mnc = (uint16_t)num;
115  p = q + 1;
116 
117  if (((p + 3) < end) || (CRYPTO_memcmp(p, "mcc", 3) != 0)) {
118  fr_strerror_printf("Missing MCC component");
119  return (domain - p);
120  }
121  num = strtoul(p, &q, 10);
122  if (*q != '.') {
123  fr_strerror_printf("Invalid MCC component");
124  return (domain - q);
125  }
126  *mcc = (uint16_t)num;
127 
128  p = q + 1;
129  if (((p + 15) < end) || (CRYPTO_memcmp(p, "3gppnetwork.org", 15) != 0)) {
130  fr_strerror_printf("Missing 3gppnetwork.org suffix");
131  return (domain - p);
132  }
133  p += 15;
134 
135  if (p != end) {
136  fr_strerror_printf("Trailing garbage");
137  return (domain - p);
138  }
139 
140  return p - domain;
141 }

143 /** Determine what type of ID was provided in the initial identity response
144  *
145  * @param[out] hint Whether this is a hint to do EAP-SIM or EAP-AKA[']:
146  * - AKA_SIM_METHOD_HINT_AKA_PRIME this ID was generated during an EAP-AKA' exchange
147  * or the supplicant hints it wants to perform EAP-AKA'.
148  * - AKA_SIM_METHOD_HINT_AKA this ID was generated during an EAP-AKA exchange
149  * or the supplicant hints it wants to perform EAP-AKA.
150  * - AKA_SIM_METHOD_HINT_SIM this IS was generated during an EAP-SIM exchange
151  * or the supplicant hints it wants to perform EAP-SIM.
152  * - AKA_SIM_METHOD_HINT_UNKNOWN we don't know what type of authentication generated
153  * this ID or which one to start.
154  * @param[out] type What type of identity this is:
155  * - AKA_SIM_ID_TYPE_PERMANENT if the ID is an IMSI.
156  * - AKA_SIM_ID_TYPE_PSEUDONYM if the ID is a freeform pseudonym.
157  * - AKA_SIM_ID_TYPE_FASTAUTH if the ID is a fastauth identity.
158  * - AKA_SIM_ID_TYPE_UNKNOWN if we can't determine what sort of ID this is.
159  * @param[in] id the NAI string provided.
160  * @param[in] id_len the length of the NAI string.
161  * @return
162  * - 0 on success.
163  * - -1 on failure.
164  */
165 int fr_aka_sim_id_type(fr_aka_sim_id_type_t *type, fr_aka_sim_method_hint_t *hint, char const *id, size_t id_len)
166 {
167  size_t i;
168 
169  if (id_len < 1) {
170  *hint = AKA_SIM_METHOD_HINT_UNKNOWN;
171  *type = AKA_SIM_ID_TYPE_UNKNOWN;
172  fr_strerror_printf("ID length too short");
173  return -1;
174  }
175 
176  /*
177  * Just get the length of the ID between the start
178  * and the first '@'. Returns the length of the
179  * entire string if no '@' found, so works with
180  * full NAI strings and Stripped-User-Name.
181  */
182  id_len = fr_aka_sim_id_user_len(id, id_len);
183 
184  /*
185  * Permanent ID format check
186  */
187  switch (id[0]) {
188  case ID_TAG_SIM_PERMANENT:
189  case ID_TAG_AKA_PERMANENT:
190  case ID_TAG_AKA_PRIME_PERMANENT:
191  if (id_len > 16) {
192  fr_strerror_printf("IMSI too long, expected <= 16 bytes got %zu bytes", id_len);
193  goto bad_format;
194  }
195 
196  for (i = 1; i < id_len; i++) {
197  if (!isdigit(id[i])) {
198  fr_strerror_printf("Invalid digit '%pV' in IMSI \"%pV\"",
199  fr_box_strvalue_len(&id[i], 1),
200  fr_box_strvalue_len(id, id_len));
201  goto bad_format;
202  }
203  }
204 
205  switch (id[0]) {
206  case ID_TAG_SIM_PERMANENT:
207  *hint = AKA_SIM_METHOD_HINT_SIM;
208  *type = AKA_SIM_ID_TYPE_PERMANENT; /* All digits */
209  return 0;
210 
211  case ID_TAG_AKA_PERMANENT:
212  *hint = AKA_SIM_METHOD_HINT_AKA;
213  *type = AKA_SIM_ID_TYPE_PERMANENT; /* All digits */
214  return 0;
215 
216  case ID_TAG_AKA_PRIME_PERMANENT:
217  *hint = AKA_SIM_METHOD_HINT_AKA_PRIME;
218  *type = AKA_SIM_ID_TYPE_PERMANENT; /* All Digits */
219  return 0;
220 
221  default:
222  break;
223  }
224  break;
225 
226  default:
227  break;
228  }
229 
230 bad_format:
231  /*
232  * Pseudonym
233  */
234  switch (id[0]) {
235  case ID_TAG_SIM_PSEUDONYM:
236  *hint = AKA_SIM_METHOD_HINT_SIM;
237  *type = AKA_SIM_ID_TYPE_PSEUDONYM;
238  return 0;
239 
240  case ID_TAG_AKA_PSEUDONYM:
241  *hint = AKA_SIM_METHOD_HINT_AKA;
242  *type = AKA_SIM_ID_TYPE_PSEUDONYM;
243  return 0;
244 
245  case ID_TAG_AKA_PRIME_PSEUDONYM:
246  *hint = AKA_SIM_METHOD_HINT_AKA_PRIME;
247  *type = AKA_SIM_ID_TYPE_PSEUDONYM;
248  return 0;
249 
250  /*
251  * Fast reauth identity
252  */
253  case ID_TAG_SIM_FASTAUTH:
254  *hint = AKA_SIM_METHOD_HINT_SIM;
255  *type = AKA_SIM_ID_TYPE_FASTAUTH;
256  return 0;
257 
258  case ID_TAG_AKA_FASTAUTH:
259  *hint = AKA_SIM_METHOD_HINT_AKA;
260  *type = AKA_SIM_ID_TYPE_FASTAUTH;
261  return 0;
262 
263  case ID_TAG_AKA_PRIME_FASTAUTH:
264  *hint = AKA_SIM_METHOD_HINT_AKA_PRIME;
265  *type = AKA_SIM_ID_TYPE_FASTAUTH;
266  return 0;
267 
268  case ID_TAG_SIM_PERMANENT:
269  *hint = AKA_SIM_METHOD_HINT_UNKNOWN;
270  *type = AKA_SIM_ID_TYPE_UNKNOWN;
271  fr_strerror_printf_push("Got SIM-Permanent-ID tag, but identity is not a permanent ID");
272  return -1;
273 
274  case ID_TAG_AKA_PERMANENT:
275  *hint = AKA_SIM_METHOD_HINT_UNKNOWN;
276  *type = AKA_SIM_ID_TYPE_UNKNOWN;
277  fr_strerror_printf_push("Got AKA-Permanent-ID tag, but identity is not a permanent ID");
278  return -1;
279 
280  default:
281  *hint = AKA_SIM_METHOD_HINT_UNKNOWN;
282  *type = AKA_SIM_ID_TYPE_UNKNOWN;
283  fr_strerror_printf_push("Unrecognised tag '%pV'", fr_box_strvalue_len(id, 1));
284  return -1;
285  }
286 }

325 /** Create a 3gpp pseudonym from a permanent ID
326  *
327  * @param[out] out Where to write the resulting pseudonym, must be a buffer of
328  * exactly AKA_SIM_3GPP_PSEUDONYM_LEN + 1 bytes.
329  * @param[in] imsi Permanent ID to derive pseudonym from. Note: If the IMSI is less than
330  * 15 digits it will be rpadded with zeros.
331  * @param[in] imsi_len Length of the IMSI. Must be between 1-15.
332  * @param[in] tag Tag value to prepend to the pseudonym. This field is 6 bits (0-63).
333  * @param[in] key_ind Key indicator (or key index), the key number used to produce
334  * the encr ID. There may be up to 16 keys in use at any one
335  * time. This field is 4 bits (0-15).
336  * @param[in] key as described by the 'Security aspects of non-3GPP accesses' document.
337  * Must be 128 bits (16 bytes).
338  * @return
339  * - 0 on success.
340  * - -1 if any of the parameters were invalid.
341  */
342 int fr_aka_sim_id_3gpp_pseudonym_encrypt(char out[AKA_SIM_3GPP_PSEUDONYM_LEN + 1],
343  char const *imsi, size_t imsi_len,
344  uint8_t tag, uint8_t key_ind, uint8_t const key[16])
345 {
346  uint8_t padded[16]; /* Random (8 bytes) + Compressed (8 bytes) */
347  uint8_t encr[16]; /* aes_ecb(padded) */
348  size_t encr_len, len = 0;
349 
350  char *out_p = out;
351 
352  char const *p = imsi, *end = p + imsi_len;
353  uint8_t *u_p, *u_end;
354  uint32_t rand[2];
355  uint8_t *compressed = padded + sizeof(rand); /* Part of padded which contains the compressed IMSI */
356 
357  EVP_CIPHER_CTX *evp_ctx;
358 
359  if (unlikely(key_ind > 15)) { /* 4 bits */
360  fr_strerror_printf("Invalid key indicator value, expected value between 0-15, got %u", key_ind);
361  return -1;
362  }
363  if (unlikely(tag > 63)) { /* 6 bits */
364  fr_strerror_printf("Invalid tag value, expected value between 0-63, got %u", tag);
365  return -1;
366  }
367 
368  /*
369  * Technically the IMSI number is between 14-15, but this
370  * encryption scheme only works for 15 char IMSIs.
371  */
372  if (unlikely(imsi_len != AKA_SIM_IMSI_MAX_LEN)) {
373  fr_strerror_printf("Invalid ID len, expected length of 15, got %zu", imsi_len);
374  return -1;
375  }
376  if (unlikely(!key)) {
377  fr_strerror_printf("Provided key was NULL");
378  return -1;
379  }
380 
381  memset(padded, 0, sizeof(padded)); /* So we don't output garbage if imsi_len < 15 */
382 
383  /*
384  * ID is an odd length (15).
385  */
386  *compressed++ = (0xf0 | (*p++ - '0'));
387 
388  /*
389  * Because we know the remaining IMSI length
390  * is a multiple of two, we can consume it
391  * 2 bytes at a time (omnomnom).
392  */
393  while (p < end) {
394  if (unlikely(!isdigit((char)p[0]) || !isdigit((char)p[1]))) {
395  fr_strerror_printf("IMSI contains invalid character");
396  return -1;
397  }
398 
399  *compressed++ = ((p[0] - '0') << 4) | (p[1] - '0');
400  p += 2;
401  }
402 
403  /*
404  * Add 8 bytes of random data to pad out the
405  * compressed IMSI to a multiple of 16 (needed
406  * to help secure AES-ECB).
407  */
408  rand[0] = fr_rand();
409  rand[1] = fr_rand();
410 
411  memcpy(padded, (uint8_t *)rand, sizeof(rand));
412 
413  /*
414  * Now we have to encrypt the padded IMSI with AES-ECB
415  */
416  evp_ctx = EVP_CIPHER_CTX_new();
417  if (!evp_ctx) {
418  tls_strerror_printf("Failed allocating EVP context");
419  return -1;
420  }
421 
422  if (unlikely(EVP_EncryptInit_ex(evp_ctx, EVP_aes_128_ecb(), NULL, key, NULL) != 1)) {
423  tls_strerror_printf("Failed initialising AES-128-ECB context");
424  error:
425  EVP_CIPHER_CTX_free(evp_ctx);
426  return -1;
427  }
428 
429  /*
430  * By default OpenSSL will try and pad out a 16 byte
431  * plaintext to 32 bytes so that it's detectable that
432  * there was padding.
433  *
434  * In this case we know the length of the plaintext
435  * we're trying to recover, so we explicitly tell
436  * OpenSSL not to pad here, and not to expected padding
437  * when decrypting.
438  */
439  EVP_CIPHER_CTX_set_padding(evp_ctx, 0);
440  if (unlikely(EVP_EncryptUpdate(evp_ctx, encr, (int *)&len, padded, sizeof(padded)) != 1)) {
441  tls_strerror_printf("Failed encrypting padded IMSI");
442  goto error;
443  }
444  encr_len = len;
445 
446  if (unlikely(EVP_EncryptFinal_ex(evp_ctx, encr + len, (int *)&len) != 1)) {
447  tls_strerror_printf("Failed finalising encrypted IMSI");
448  goto error;
449  }
450  encr_len += len;
451 
452  /*
453  * Ciphertext should be same length as plaintext.
454  */
455  if (unlikely(encr_len != sizeof(padded))) {
456  fr_strerror_printf("Invalid ciphertext length, expected %zu, got %zu", sizeof(padded), encr_len);
457  goto error;
458  }
459 
460  EVP_CIPHER_CTX_free(evp_ctx);
461 
462  /*
463  * Now encode the entire output as base64.
464  */
465  u_p = encr;
466  u_end = u_p + encr_len;
467 
468  /*
469  * Consume tag (6 bits) + key_ind (4 bits) + encr[0] (8 bits) = 18 bits (or 3 bytes of b64)
470  */
471  *out_p++ = fr_base64_str[tag & 0x3f]; /* 6 bits tag */
472  *out_p++ = fr_base64_str[((key_ind & 0x0f) << 2) | ((u_p[0] & 0xc0) >> 6)]; /* 4 bits key_ind + 2 high bits encr[0] */
473  *out_p++ = fr_base64_str[u_p[0] & 0x3f]; /* 6 low bits of encr[0] */
474  u_p++;
475 
476  /*
477  * Consume 3 bytes of input for 4 bytes of b64 (5 iterations)
478  */
479  while (u_p < u_end) {
480  *out_p++ = fr_base64_str[(u_p[0] & 0xfc) >> 2]; /* 6 high bits of p[0] */
481  *out_p++ = fr_base64_str[((u_p[0] & 0x03) << 4) | ((u_p[1] & 0xf0) >> 4)]; /* 2 low bits of p[0] + 4 high bits of p[1] */
482  *out_p++ = fr_base64_str[((u_p[1] & 0x0f) << 2) | ((u_p[2] & 0xc0) >> 6)]; /* 4 low bits of p[1] + 2 high bits of p[2] */
483  *out_p++ = fr_base64_str[u_p[2] & 0x3f]; /* 6 low bits of p[2] */
484  u_p += 3;
485  }
486  if ((out_p - out) != AKA_SIM_3GPP_PSEUDONYM_LEN) {
487  fr_strerror_printf("Base64 output length invalid, expected %i bytes, got %zu bytes",
488  AKA_SIM_3GPP_PSEUDONYM_LEN, out_p - out);
489  return -1;
490  }
491 
492  out[AKA_SIM_3GPP_PSEUDONYM_LEN] = '\0';
493 
494  return 0;
495 }

519 /** Decrypt the 3GPP pseudonym
520  *
521  * @param[out] out Where to write the decypted, uncompressed IMSI.
522  * @param[in] encr_id to decypt. Will read exactly 23 bytes from the buffer.
523  * @param[in] key to use to decrypt the encrypted and compressed IMSI.
524  * Must be 128 bits (16 bytes).
525  * @return
526  * - 0 on success.
527  * - -1 if any of the parameters were invalid.
528  */
529 int fr_aka_sim_id_3gpp_pseudonym_decrypt(char out[AKA_SIM_IMSI_MAX_LEN + 1],
530  char const encr_id[AKA_SIM_3GPP_PSEUDONYM_LEN], uint8_t const key[16])
531 {
532  EVP_CIPHER_CTX *evp_ctx;
533 
534  char *out_p = out;
535 
536  uint8_t dec[16];
537  uint8_t *dec_p = dec;
538 
539  uint8_t decr[16];
540  uint8_t *compressed = decr + 8; /* Pointer into plaintext after the random component */
541  size_t decr_len;
542 
543  char const *p = encr_id, *end = p + AKA_SIM_3GPP_PSEUDONYM_LEN;
544 
545  size_t len = 0;
546  int i;
547 
548  for (i = 0; i < AKA_SIM_3GPP_PSEUDONYM_LEN; i++) {
549  if (!fr_is_base64(encr_id[i])) {
550  fr_strerror_printf("Encrypted IMSI contains non-base64 char '%pV'",
551  fr_box_strvalue_len(&encr_id[i], 1));
552  return -1;
553  }
554  }
555 
556  *dec_p++ = (((fr_base64_sextet[us(p[1])] & 0x03) << 6) | fr_base64_sextet[us(p[2])]);
557  p += 3;
558 
559  while (p < end) {
560  *dec_p++ = ((fr_base64_sextet[us(p[0])] << 2) | (fr_base64_sextet[us(p[1])] >> 4));
561  *dec_p++ = ((fr_base64_sextet[us(p[1])] << 4) & 0xf0) | (fr_base64_sextet[us(p[2])] >> 2);
562  *dec_p++ = ((fr_base64_sextet[us(p[2])] << 6) & 0xc0) | fr_base64_sextet[us(p[3])];
563 
564  p += 4; /* 32bit input -> 24bit output */
565  }
566 
567  evp_ctx = EVP_CIPHER_CTX_new();
568  if (!evp_ctx) {
569  tls_strerror_printf("Failed allocating EVP context");
570  return -1;
571  }
572 
573  if (unlikely(EVP_DecryptInit_ex(evp_ctx, EVP_aes_128_ecb(), NULL, key, NULL) != 1)) {
574  tls_strerror_printf("Failed initialising AES-128-ECB context");
575  error:
576  EVP_CIPHER_CTX_free(evp_ctx);
577  return -1;
578  }
579 
580  /*
581  * By default OpenSSL expects 16 bytes of plaintext
582  * to produce 32 bytes of ciphertext, due to padding
583  * being added if the plaintext is a multiple of 16.
584  *
585  * There's no way for OpenSSL to determine if a
586  * 16 byte ciphertext was padded or not, so we need to
587  * inform OpenSSL explicitly that there's no padding.
588  */
589  EVP_CIPHER_CTX_set_padding(evp_ctx, 0);
590  if (unlikely(EVP_DecryptUpdate(evp_ctx, decr, (int *)&len, dec, sizeof(dec)) != 1)) {
591  tls_strerror_printf("Failed decypting IMSI");
592  goto error;
593  }
594  decr_len = len;
595 
596  if (unlikely(EVP_DecryptFinal_ex(evp_ctx, decr + len, (int *)&len) != 1)) {
597  tls_strerror_printf("Failed finalising decypted IMSI");
598  goto error;
599  }
600  decr_len += len;
601 
602  /*
603  * This should never happen, and probably means that
604  * some sort of memory corruption has occured.
605  */
606  if (unlikely(decr_len > (AKA_SIM_IMSI_MAX_LEN + 1))) {
607  fr_strerror_printf("Decrypted data len invalid. Expected %i bytes, got %zu bytes",
608  (AKA_SIM_IMSI_MAX_LEN + 1), decr_len);
609  goto error;
610  }
611 
612  /*
613  * Decompress the IMSI
614  *
615  * The first 8 octets are a junk random value which
616  * we ignore.
617  */
618  *out_p++ = (compressed[0] & 0x0f) + '0';
619  for (i = 1; i < 8; i++) {
620  *out_p++ = ((compressed[i] & 0xf0) >> 4) + '0';
621  *out_p++ = (compressed[i] & 0x0f) + '0';
622  }
623 
624  EVP_CIPHER_CTX_free(evp_ctx);
625 
626  out[AKA_SIM_IMSI_MAX_LEN] = '\0';
627 
628  return 0;
629 }
