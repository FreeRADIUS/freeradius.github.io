271  * A child thread reads the packet from a pipe, and processes it.

273 static void bfd_pipe_recv(UNUSED fr_event_list_t *xel, int fd, UNUSED int flags, void *ctx)
274 {
275  ssize_t num;
276  bfd_state_t *session = ctx;
277  bfd_packet_t bfd;
278 
279  if (session->blocked) return;
280 
281  /*
282  * Read the header
283  */
284  num = read(fd, &bfd, 4);
285  if ((num < 4) || (bfd.length < 4)) {
286  fail:
287  ERROR("BFD Failed reading from pipe!");
288  session->blocked = true;
289  return;
290  }
291 
292  /*
293  * This is already checked in the caller, but what the heck...
294  */
295  if (bfd.length > sizeof(bfd)) goto fail;
296 
297  /*
298  * Read the rest of the packet.
299  */
300  num = read(fd, ((uint8_t *) &bfd) + 4, bfd.length - 4);
301  if ((num < 0) || ((num + 4) != bfd.length)) goto fail;
302 
303  bfd_process(session, &bfd);
304 }

325 static int bfd_pthread_create(bfd_state_t *session)
326 {
327  pthread_attr_t attr;
328 
329  if (pipe(session->pipefd) < 0) {
330  ERROR("Failed opening pipe: %s", fr_syserror(errno));
331  return 0;
332  }
333 
334  session->el = fr_event_list_alloc(session, NULL, NULL);
335  if (!session->el) {
336  ERROR("Failed creating event list");
337  close_pipes:
338  close(session->pipefd[0]);
339  close(session->pipefd[1]);
340  session->pipefd[0] = session->pipefd[1] = -1;
341  return 0;
342  }
343 
344 #ifdef O_NONBLOCK
345  fcntl(session->pipefd[0], F_SETFL, O_NONBLOCK | FD_CLOEXEC);
346  fcntl(session->pipefd[1], F_SETFL, O_NONBLOCK | FD_CLOEXEC);
347 #endif
348 
349  if (fr_event_fd_insert(session, session->el, session->pipefd[0],
350  bfd_pipe_recv,
351  NULL,
352  NULL,
353  session) < 0) {
354  PERROR("Failed inserting file descriptor into event list");
355  goto close_pipes;
356  }
357 
358  pthread_attr_init(&attr);
359  pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_DETACHED);
360 
361  /*
362  * Create the thread detached, so that it cleans up it's
363  * own memory when it exits.
364  *
365  * Note that the function returns non-zero on error, NOT
366  * -1. The return code is the error, and errno isn't set.
367  */
368  if (fr_schedule_pthread_create(&session->pthread_id, bfd_child_thread, session) < 0) {
369  talloc_free(session->el);
370  session->el = NULL;
371  PERROR("Thread create failed");
372  goto close_pipes;
373  }
374  pthread_attr_destroy(&attr);
375 
376  return 1;
377 }

429 static ssize_t bfd_parse_secret(CONF_SECTION *cs, uint8_t secret[BFD_MAX_SECRET_LENGTH])
430 {
431  int rcode;
432  size_t len;
433  char const *value = NULL;
434 
435  rcode = cf_pair_parse(NULL, cs, "secret", FR_ITEM_POINTER(FR_TYPE_STRING, &value), NULL, T_INVALID);
436  if (rcode != 0) return 0;
437 
438  len = strlen(value);
439 
440  if ((value[0] == '0') && (value[1] == 'x')) {
441  if (len > 42) {
442  cf_log_err(cf_section_to_item(cs), "Secret is too long");
443  return -1;
444  }
445 
446  if ((len & 0x01) != 0) {
447  cf_log_err(cf_section_to_item(cs), "Invalid hex length");
448  return -1;
449  }
450 
451  return fr_hex2bin(NULL, &FR_DBUFF_TMP(secret, BFD_MAX_SECRET_LENGTH),
452  &FR_SBUFF_IN(value + 2, (len - 2)), false);
453  }
454 
455  if (len >= 20) {
456  cf_log_err(cf_section_to_item(cs), "Secret is too long");
457  return -1;
458  }
459 
460  memset(secret, 0, BFD_MAX_SECRET_LENGTH);
461  memcpy(secret, value, len);
462  return len;
463 }

471  CONF_SECTION *cs,
472  const fr_ipaddr_t *ipaddr, uint16_t port)

476  bfd_state_t *session;
477 
478  session = talloc_zero(sock, bfd_state_t);
479 
480  /*
481  * Initialize according to RFC.
482  */
483  session->number = sock->number++;
484  session->socket.fd = sockfd;
485  session->session_state = BFD_STATE_DOWN;
486  session->server_cs = sock->server_cs;
487  session->unlang = sock->unlang;
488  session->local_disc = fr_rand();
489  session->remote_disc = 0;
490  session->local_diag = BFD_DIAG_NONE;
491  session->desired_min_tx_interval = sock->min_tx_interval * 1000;
492  session->required_min_rx_interval = sock->min_rx_interval * 1000;
493  session->remote_min_rx_interval = 1;
494  session->demand_mode = sock->demand;
495  session->remote_demand_mode = false;
496  session->detect_multi = sock->max_timeouts;
497  session->auth_type = BFD_AUTH_RESERVED;
498  session->recv_auth_seq = 0;
499  session->xmit_auth_seq = fr_rand();
500  session->auth_seq_known = 0;
501 
502  /*
503  * Allow over-riding of variables per session.
504  */
505  rcode = cf_pair_parse(NULL, cs, "demand", FR_ITEM_POINTER(FR_TYPE_BOOL, &flag), NULL, T_INVALID);
506  if (rcode == 0) {
507  session->demand_mode = flag;
508  }
509 
510  /*
511  * Number is moved out of scope to shut up lgtm
512  * static analysis, but really these should be
513  * moved into conf_parser callback functions.
514  */
515  {
516  uint32_t number;
517 
518  rcode = cf_pair_parse(NULL, cs, "min_transmit_interval", FR_ITEM_POINTER(FR_TYPE_UINT32, &number), NULL, T_INVALID);
519  if (rcode == 0) {
520  if (number < 100) number = 100;
521  if (number > 10000) number = 10000;
522 
523  session->desired_min_tx_interval = number * 1000;
524  }
525  }
526 
527  {
528  uint32_t number;
529 
530  rcode = cf_pair_parse(NULL, cs, "min_receive_interval", FR_ITEM_POINTER(FR_TYPE_UINT32, &number), NULL, T_INVALID);
531  if (rcode == 0) {
532  if (number < 100) number = 100;
533  if (number > 10000) number = 10000;
534 
535  session->required_min_rx_interval = number * 1000;
536  }
537  }
538 
539  {
540  uint32_t number;
541 
542  rcode = cf_pair_parse(NULL, cs, "max_timeouts", FR_ITEM_POINTER(FR_TYPE_UINT32, &number), NULL, T_INVALID);
543  if (rcode == 0) {
544  if (number == 0) number = 1;
545  if (number > 10) number = 10;
546 
547  session->detect_multi = number;
548  }
549  }
550  session->auth_type = sock->auth_type;
551 
552  /*
553  * Parse / over-ride the secrets.
554  */
555  session->secret_len = bfd_parse_secret(cs, session->secret);
556  if ((session->secret_len == 0) &&
557  (session->auth_type != BFD_AUTH_RESERVED)) {
558  if (sock->secret_len == 0) {
559  cf_log_err(cf_section_to_item(cs), "auth_type requires a secret");
560  talloc_free(session);
561  return NULL;
562  }
563 
564  session->secret_len = sock->secret_len;
565  memcpy(session->secret, sock->secret, sizeof(session->secret));
566  }
567 
568  /*
569  * Initialize the detection time.
570  */
571  if (!session->demand_mode) {
572  session->detection_time = session->required_min_rx_interval;
573  } else {
574  session->detection_time = session->desired_min_tx_interval;
575  }
576  session->detection_time *= session->detect_multi;
577 
578  /*
579  * And finally remember the session.
580  */
581  session->socket.inet.dst_ipaddr = *ipaddr;
582  session->socket.inet.dst_port = port;
583 
584  session->socket.inet.src_ipaddr = sock->my_ipaddr;
585  session->socket.inet.src_port = sock->my_port;
586 
587  fr_ipaddr_to_sockaddr(&session->remote_sockaddr, &session->salen,
588  ipaddr, port);
589 
590  if (!rbtree_insert(sock->session_tree, session)) {
591  ERROR("FAILED creating new session!");
592  talloc_free(session);
593  return NULL;
594  }
595 
596  bfd_trigger(session);
597 
598  /*
599  * Check for threaded / non-threaded operation.
600  */
601  if (event_list) {
602  session->el = event_list;
603 
604  bfd_start_control(session);
605 
606  session->pipefd[0] = session->pipefd[1] = -1;
607  session->pthread_id = pthread_self();
608  } else {
609  if (!bfd_pthread_create(session)) {
610  rbtree_deletebydata(sock->session_tree, session);
611  talloc_free(session);
612  return NULL;
613  }
614  }
615 
616  return session;
617 }

620 static int bfd_verify_sequence(bfd_state_t *session, uint32_t sequence_no,
621  int keyed)
622 {
623  uint32_t start, stop;
624 
625  start = session->recv_auth_seq;
626  if (keyed) {
627  start++;
628  }
629  stop = start + 3 * session->detect_multi;
630 
631  if (start < stop) {
632  if ((sequence_no < start) ||
633  (sequence_no > stop)) {
634  return 0;
635  }
636 
637  } else { /* start is ~2^32, stop is ~10 */
638  if ((sequence_no > start) &&
639  (sequence_no < stop)) {
640  return 0;
641  }
642  }
643 
644  return 1;
645 }

674 static int bfd_verify_md5(bfd_state_t *session, bfd_packet_t *bfd)
675 {
676  int rcode;
677  bfd_auth_md5_t *md5 = &bfd->auth.md5;
678  uint8_t digest[sizeof(md5->digest)];
679 
680  if (md5->auth_len != sizeof(*md5)) return 0;
681 
682  if (md5->key_id != 0) return 0;
683 
684  memcpy(digest, md5->digest, sizeof(digest));
685 
686  bfd_calc_md5(session, bfd);
687  rcode = fr_digest_cmp(digest, md5->digest, sizeof(digest));
688 
689  memcpy(md5->digest, digest, sizeof(md5->digest)); /* pedantic */
690 
691  if (rcode != 0) {
692  DEBUG("BFD %d MD5 Digest failed: **** RE-ENTER THE SECRET ON BOTH ENDS ****", session->number);
693  return 0;
694  }
695 
696  /*
697  * Do this AFTER the authentication instead of before!
698  */
699  if (!session->auth_seq_known) {
700  session->auth_seq_known = 1;
701 
702  } else if (!bfd_verify_sequence(session, md5->sequence_no,
703  (md5->auth_type == BFD_AUTH_MET_KEYED_MD5))) {
704  DEBUG("MD5 sequence out of window");
705  return 0;
706  }
707 
708  session->recv_auth_seq = md5->sequence_no;
709 
710  return 1;
711 }

743 static int bfd_verify_sha1(bfd_state_t *session, bfd_packet_t *bfd)
744 {
745  int rcode;
746  bfd_auth_sha1_t *sha1 = &bfd->auth.sha1;
747  uint8_t digest[sizeof(sha1->digest)];
748 
749  if (sha1->auth_len != sizeof(*sha1)) return 0;
750 
751  if (sha1->key_id != 0) return 0;
752 
753  memcpy(digest, sha1->digest, sizeof(digest));
754 
755  bfd_calc_sha1(session, bfd);
756  rcode = fr_digest_cmp(digest, sha1->digest, sizeof(digest));
757 
758  memcpy(sha1->digest, digest, sizeof(sha1->digest)); /* pedantic */
759 
760  if (rcode != 0) {
761  DEBUG("BFD %d SHA1 Digest failed: **** RE-ENTER THE SECRET ON BOTH ENDS ****", session->number);
762  return 0;
763  }
764 
765  /*
766  * Do this AFTER the authentication instead of before!
767  */
768  if (!session->auth_seq_known) {
769  session->auth_seq_known = 1;
770 
771  } else if (!bfd_verify_sequence(session, sha1->sequence_no,
772  (sha1->auth_type == BFD_AUTH_MET_KEYED_SHA1))) {
773  DEBUG("SHA1 sequence out of window");
774  return 0;
775  }
776 
777  session->recv_auth_seq = sha1->sequence_no;
778 
779  return 1;
780 }

905 static int bfd_start_packets(bfd_state_t *session)
906 {

934  jitter >>= 32;
935  interval = base;
936  interval += jitter;

939  fr_time_delta_from_usec(interval),
