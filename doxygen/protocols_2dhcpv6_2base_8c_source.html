146 /** Return the on-the-wire length of an attribute value
147  *
148  * @param[in] vp to return the length of.
149  * @return the length of the attribute.
150  */
151 size_t fr_dhcpv6_option_len(fr_pair_t const *vp)
152 {
153  switch (vp->vp_type) {
154  case FR_TYPE_VARIABLE_SIZE:
155  if (vp->da->flags.length) return vp->da->flags.length; /* Variable type with fixed length */
156  return vp->vp_length;
157 
158  case FR_TYPE_DATE:
159  case FR_TYPE_TIME_DELTA:
160  if (vp->data.enumv->flags.length) return vp->data.enumv->flags.length;
161  return sizeof(uint32_t);
162 
163  case FR_TYPE_STRUCTURAL:
164  fr_assert_fail(NULL);
165  return 0;
166 
167  default:
168  return fr_dhcpv6_attr_sizes[vp->vp_type][0];
169  }
170 }

175 static ssize_t fr_dhcpv6_options_ok(uint8_t const *packet, uint8_t const *end, size_t max_attributes,
176  bool allow_relay, int depth, char const **error)
177 {
178  size_t attributes;
179  uint8_t const *p;
180 
181  attributes = 0;
182  p = packet;
183 
184  while (p < end) {
185  uint16_t len;
186 
187  if ((size_t)(end - p) < DHCPV6_OPT_HDR_LEN) {
188  *error = "Not enough room for option header";
189  return -(p - packet);
190  }
191 
192  len = DHCPV6_GET_OPTION_LEN(p);
193  if ((size_t)(end - p) < (DHCPV6_OPT_HDR_LEN + len)) {
194  *error = "Option length overflows the packet";
195  return -(p - packet);
196  }
197 
198  attributes++;
199  if (attributes > (size_t) max_attributes) {
200  *error = "Too many attributes";
201  return -(p - packet);
202  }
203 
204  /*
205  * Recurse into the Relay-Message attribute, but
206  * only if the outer packet was a relayed message.
207  */
208  if (allow_relay && (p[0] == 0) && (p[1] == attr_relay_message->attr)) {
209  ssize_t child;
210 
211  /*
212  * Recurse to check the encapsulated packet.
213  */
214  child = fr_dhcpv6_ok_internal(p + 4, p + 4 + len, max_attributes - attributes, depth + 1, error);
215  if (child <= 0) return -((p + 4) - packet) + child;
216 
217  attributes += child;
218  }
219 
220  p += DHCPV6_OPT_HDR_LEN + len;
221  }
222 
223  return attributes;
224 }

226 static ssize_t fr_dhcpv6_ok_internal(uint8_t const *packet, uint8_t const *end, size_t max_attributes, int depth,
227  char const **error)
228 {
229  uint8_t const *p;
230  ssize_t attributes;
231  bool allow_relay;
232  size_t packet_len = end - packet;
233 
234  if (depth > DHCPV6_MAX_RELAY_NESTING) {
235  *error = "Too many layers forwarded packets";
236  return 0;
237  }
238 
239  switch (packet[0]) {
240  case FR_DHCPV6_RELAY_FORWARD:
241  case FR_DHCPV6_RELAY_REPLY:
242  if (packet_len < DHCPV6_RELAY_HDR_LEN) {
243  *error = "Packet is too small for relay header";
244  return 0;
245  }
246 
247  p = packet + DHCPV6_RELAY_HDR_LEN;
248  allow_relay = true;
249  break;
250 
251  default:
252  /*
253  * 8 bit code + 24 bits of transaction ID
254  */
255  if (packet_len < DHCPV6_HDR_LEN) {
256  *error = "Packet is too small for DHCPv6 header";
257  return 0;
258  }
259 
260  p = packet + DHCPV6_HDR_LEN;
261  allow_relay = false;
262  break;
263  }
264 
265  attributes = fr_dhcpv6_options_ok(p, end, max_attributes, allow_relay, depth, error);
266  if (attributes < 0) return -(p - packet) + attributes;
267 
268  return attributes;
269 }

272 /** See if the data pointed to by PTR is a valid DHCPv6 packet.
273  *
274  * @param[in] packet to check.
275  * @param[in] packet_len The size of the packet data.
276  * @param[in] max_attributes to allow in the packet.
277  * @return
278  * - True on success.
279  * - False on failure.
280  */
281 bool fr_dhcpv6_ok(uint8_t const *packet, size_t packet_len,
282  uint32_t max_attributes)
283 {
284  ssize_t slen;
285  char const *error = NULL;
286 
287  slen = fr_dhcpv6_ok_internal(packet, packet + packet_len, max_attributes, 0, &error);
288  if (slen <= 0) {
289  fr_strerror_printf("Invalid DHCPv6 packet starting at offset %zd - %s", -slen, error);
290  return false;
291  }
292 
293  return true;
294 }

296 /*
297  * Return pointer to a particular option.
298  */
299 uint8_t const *fr_dhcpv6_option_find(uint8_t const *start, uint8_t const *end, unsigned int option)
300 {
301  uint8_t const *p = start;
302 
303  while (p < end) {
304  uint16_t found;
305  uint16_t len;
306 
307  if ((size_t)(end - p) < DHCPV6_OPT_HDR_LEN) return NULL;
308 
309  found = DHCPV6_GET_OPTION_NUM(p);
310  len = DHCPV6_GET_OPTION_LEN(p);
311 
312  if ((p + DHCPV6_OPT_HDR_LEN + len) > end) return NULL;
313 
314  if (found == option) return p;
315 
316  p += DHCPV6_OPT_HDR_LEN + len;
317  }
318 
319  return NULL;
320 }

322 static bool duid_match(uint8_t const *option, fr_dhcpv6_decode_ctx_t const *packet_ctx)
323 {
324  uint16_t len;
325 
326  len = DHCPV6_GET_OPTION_LEN(option);
327  if (len != packet_ctx->duid_len) return false;
328  if (memcmp(option + 4, packet_ctx->duid, packet_ctx->duid_len) != 0) return false;
329 
330  return true;
331 }

333 /** Verify a reply packet from a server to a client
334  *
335  */
336 static bool verify_to_client(uint8_t const *packet, size_t packet_len, fr_dhcpv6_decode_ctx_t const *packet_ctx)
337 {
338  uint32_t transaction_id;
339  uint8_t const *option;
340  uint8_t const *options = packet + 4;
341  uint8_t const *end = packet + packet_len;
342 
343  switch (packet[0]) {
344  case FR_PACKET_TYPE_VALUE_ADVERTISE:
345  transaction_id = fr_net_to_uint24(&packet[1]);
346  if (transaction_id != packet_ctx->transaction_id) {
347  fail_tid:
348  fr_strerror_printf("Transaction ID does not match");
349  return false;
350  }
351 
352  if (!fr_dhcpv6_option_find(options, end, FR_SERVER_ID)) {
353  fail_sid:
354  fr_strerror_printf("Packet does not contain a Server-Id option");
355  return false;
356  }
357 
358  option = fr_dhcpv6_option_find(options, end, FR_CLIENT_ID);
359  if (!option) {
360  fail_cid:
361  fr_strerror_printf("Packet does not contain a Client-Id option");
362  return false;
363  }
364 
365  /*
366  * The DUID MUST exist.
367  */
368  if (!packet_ctx->duid) {
369  fail_duid:
370  fr_strerror_printf("Packet context does not contain a DUID");
371  return false;
372  }
373 
374  check_duid:
375  if (!duid_match(option, packet_ctx)) {
376  fail_match:
377  fr_strerror_printf("DUID in packet does not match our DUID");
378  return false;
379  }
380  return true;
381 
382  case FR_PACKET_TYPE_VALUE_REPLY:
383  transaction_id = fr_net_to_uint24(&packet[1]);
384  if (transaction_id != packet_ctx->transaction_id) goto fail_tid;
385 
386  if (!fr_dhcpv6_option_find(options, end, FR_SERVER_ID)) goto fail_sid;
387 
388  /*
389  * It's OK to not have a client ID in the reply if we didn't send one.
390  */
391  option = fr_dhcpv6_option_find(options, end, FR_CLIENT_ID);
392  if (!option) {
393  if (!packet_ctx->duid) return true;
394  goto fail_cid;
395  }
396  goto check_duid;
397 
398  case FR_PACKET_TYPE_VALUE_RECONFIGURE:
399  if (!fr_dhcpv6_option_find(options, end, FR_SERVER_ID)) goto fail_sid;
400 
401  option = fr_dhcpv6_option_find(options, end, FR_CLIENT_ID);
402  if (!option) goto fail_cid;
403 
404  /*
405  * The DUID MUST exist.
406  */
407  if (!packet_ctx->duid) goto fail_duid;
408  if (!duid_match(option, packet_ctx)) goto fail_match;
409 
410  option = fr_dhcpv6_option_find(options, end, FR_RECONF_MSG);
411  if (!option) {
412  fr_strerror_printf("Packet does not contain a Reconf-Msg option");
413  return false;
414  }
415 
416  /*
417  * @todo - check reconfigure message type, and
418  * reject if it doesn't match.
419  */
420 
421  /*
422  * @todo - check for authentication option and
423  * verify it.
424  */
425  break;
426 
427  case FR_DHCPV6_RELAY_REPLY:
428  if (packet_len < DHCPV6_RELAY_HDR_LEN) {
429  fr_strerror_printf("Relay-Reply message is too small");
430  return false;
431  }
432 
433  options += (DHCPV6_RELAY_HDR_LEN - 4); /* we assumed it was a normal packet above */
434  option = fr_dhcpv6_option_find(options, end, FR_RELAY_MESSAGE);
435  if (!option) {
436  fr_strerror_printf("Packet does not contain a Relay-Message option");
437  return false;
438  }
439  return verify_to_client(option + 4, DHCPV6_GET_OPTION_LEN(option), packet_ctx);
440 
441  case FR_DHCPV6_LEASE_QUERY_REPLY:
442  transaction_id = fr_net_to_uint24(&packet[1]);
443  if (transaction_id != packet_ctx->transaction_id) goto fail_tid;
444 
445  if (!fr_dhcpv6_option_find(options, end, FR_SERVER_ID)) goto fail_sid;
446 
447  option = fr_dhcpv6_option_find(options, end, FR_CLIENT_ID);
448  if (!option) goto fail_cid;
449 
450  /*
451  * The DUID MUST exist.
452  */
453  if (!packet_ctx->duid) goto fail_duid;
454  if (!duid_match(option, packet_ctx)) goto fail_match;
455  break;
456 
457  case FR_PACKET_TYPE_VALUE_REQUEST:
458  case FR_PACKET_TYPE_VALUE_CONFIRM:
459  case FR_PACKET_TYPE_VALUE_RENEW:
460  case FR_PACKET_TYPE_VALUE_REBIND:
461  case FR_PACKET_TYPE_VALUE_RELEASE:
462  case FR_PACKET_TYPE_VALUE_DECLINE:
463  case FR_PACKET_TYPE_VALUE_INFORMATION_REQUEST:
464  default:
465  fr_strerror_printf("Invalid message type sent to client");
466  return false;
467  }
468 
469  return true;
470 }

473 /** Verify a packet from a client to a server
474  *
475  */
476 static bool verify_from_client(uint8_t const *packet, size_t packet_len, fr_dhcpv6_decode_ctx_t const *packet_ctx)
477 {
478  uint8_t const *option;
479  uint8_t const *options = packet + 4;
480  uint8_t const *end = packet + packet_len;
481 
482  /*
483  * Servers MUST have a DUID
484  */
485  if (!packet_ctx->duid) {
486  fr_strerror_printf("Packet context does not contain a DUID");
487  return false;
488  }
489 
490  switch (packet[0]) {
491  case FR_PACKET_TYPE_VALUE_SOLICIT:
492  case FR_PACKET_TYPE_VALUE_CONFIRM:
493  case FR_PACKET_TYPE_VALUE_REBIND:
494  if (!fr_dhcpv6_option_find(options, end, FR_CLIENT_ID)) {
495  fail_cid:
496  fr_strerror_printf("Packet does not contain a Client-Id option");
497  return false;
498  }
499 
500  if (!fr_dhcpv6_option_find(options, end, FR_SERVER_ID)) {
501  fail_sid:
502  fr_strerror_printf("Packet does not contain a Server-Id option");
503  return false;
504  }
505  break;
506 
507  case FR_PACKET_TYPE_VALUE_REQUEST:
508  case FR_PACKET_TYPE_VALUE_RENEW:
509  case FR_PACKET_TYPE_VALUE_DECLINE:
510  case FR_PACKET_TYPE_VALUE_RELEASE:
511  if (!fr_dhcpv6_option_find(options, end, FR_CLIENT_ID)) goto fail_cid;
512 
513  option = fr_dhcpv6_option_find(options, end, FR_SERVER_ID);
514  if (!option) goto fail_sid;
515 
516  if (!duid_match(option, packet_ctx)) {
517  fail_match:
518  fr_strerror_printf("DUID in packet does not match our DUID");
519  return false;
520  }
521  break;
522 
523  case FR_PACKET_TYPE_VALUE_INFORMATION_REQUEST:
524  option = fr_dhcpv6_option_find(options, end, FR_SERVER_ID);
525  if (!option) goto fail_sid;
526 
527  if (!duid_match(option, packet_ctx)) goto fail_match;
528 
529  /*
530  * IA options are forbidden.
531  */
532  if (fr_dhcpv6_option_find(options, end, FR_IA_NA)) {
533  fr_strerror_printf("Packet contains an IA-NA option");
534  return false;
535  }
536  if (fr_dhcpv6_option_find(options, end, FR_IA_TA)) {
537  fr_strerror_printf("Packet contains an IA-TA option");
538  return false;
539  }
540  if (fr_dhcpv6_option_find(options, end, FR_IA_ADDR)) {
541  fr_strerror_printf("Packet contains an IA-Addr option");
542  return false;
543  }
544  break;
545 
546  case FR_DHCPV6_RELAY_FORWARD:
547  if (packet_len < DHCPV6_RELAY_HDR_LEN) {
548  fr_strerror_printf("Relay-Forward message is too small");
549  return false;
550  }
551 
552  options += (DHCPV6_RELAY_HDR_LEN - 4); /* we assumed it was a normal packet above */
553  option = fr_dhcpv6_option_find(options, end, FR_RELAY_MESSAGE);
554  if (!option) {
555  fr_strerror_printf("Packet does not contain a Relay-Message option");
556  return false;
557  }
558 
559  return verify_from_client(option + 4, DHCPV6_GET_OPTION_LEN(option), packet_ctx);
560 
561  case FR_PACKET_TYPE_VALUE_LEASE_QUERY:
562  if (!fr_dhcpv6_option_find(options, end, FR_CLIENT_ID)) goto fail_cid;
563 
564  /*
565  * Server-ID is a SHOULD, but if it exists, it
566  * MUST match.
567  */
568  option = fr_dhcpv6_option_find(options, end, FR_SERVER_ID);
569  if (option && !duid_match(option, packet_ctx)) goto fail_match;
570 
571  option = fr_dhcpv6_option_find(options, end, FR_LEASE_QUERY);
572  if (!option) {
573  fr_strerror_printf("Packet does not contain a Lease-Query option");
574  return false;
575  }
576  break;
577 
578  case FR_PACKET_TYPE_VALUE_ADVERTISE:
579  case FR_PACKET_TYPE_VALUE_REPLY:
580  case FR_PACKET_TYPE_VALUE_RECONFIGURE:
581  default:
582  fr_strerror_printf("Invalid message type sent to server");
583  return false;
584  }

629 /** Decode a DHCPv6 packet
630  *
631  */
632 ssize_t fr_dhcpv6_decode(TALLOC_CTX *ctx, uint8_t const *packet, size_t packet_len, fr_cursor_t *cursor)
633 {
634  ssize_t slen;
635  uint8_t const *p, *end;
636  fr_dhcpv6_decode_ctx_t packet_ctx;
637  fr_pair_t *vp;
638 
639  /*
640  * Get the packet type.
641  */
642  vp = fr_pair_afrom_da(ctx, attr_packet_type);
643  if (!vp) return -1;
644 
645  vp->vp_uint32 = packet[0];
646  vp->type = VT_DATA;
647  fr_cursor_append(cursor, vp);
648 
649  switch (packet[0]) {
650  case FR_DHCPV6_RELAY_FORWARD:
651  case FR_DHCPV6_RELAY_REPLY:
652  /*
653  * Just for sanity check.
654  */
655  if (packet_len < DHCPV6_RELAY_HDR_LEN) return -1;
656 
657  /*
658  * Decode the header fields.
659  */
660  vp = fr_pair_afrom_da(ctx, attr_hop_count);
661  if (!vp) goto fail;
662  if (fr_value_box_from_network(vp, &vp->data, vp->da->type, NULL, packet + 1, 1, true) < 0) {
663  goto fail;
664  }
665  fr_cursor_append(cursor, vp);
666 
667  vp = fr_pair_afrom_da(ctx, attr_relay_link_address);
668  if (!vp) goto fail;
669  if (fr_value_box_from_network(vp, &vp->data, vp->da->type, NULL, packet + 2, 16, true) < 0) {
670  goto fail;
671  }
672  fr_cursor_append(cursor, vp);
673 
674  vp = fr_pair_afrom_da(ctx, attr_relay_peer_address);
675  if (!vp) goto fail;
676  if (fr_value_box_from_network(vp, &vp->data, vp->da->type, NULL, packet + 2 + 16, 16, true) < 0) {
677  goto fail;
678  }
679 
680  fr_cursor_append(cursor, vp);
681 
682  p = packet + DHCPV6_RELAY_HDR_LEN;
683  goto decode_options;
684 
685  default:
686  break;
687  }
688 
689  /*
690  * And the transaction ID.
691  */
692  vp = fr_pair_afrom_da(ctx, attr_transaction_id);
693  if (!vp) {
694  fail:
695  fr_cursor_head(cursor);
696  fr_cursor_free_list(cursor);
697  return -1;
698  }
699 
700  /*
701  * Copy 3 octets over.
702  */
703  (void) fr_pair_value_memdup(vp, packet + 1, 3, false);
704 
705  vp->type = VT_DATA;
706  fr_cursor_append(cursor, vp);
707 
708  p = packet + 4;
709 
710 decode_options:
711  end = packet + packet_len;
712  packet_ctx.tmp_ctx = talloc_init_const("tmp");
713 
714  /*
715  * The caller MUST have called fr_dhcpv6_ok() first. If
716  * he doesn't, all hell breaks loose.
717  */
718  while (p < end) {
719  slen = fr_dhcpv6_decode_option(ctx, cursor, dict_dhcpv6, p, (end - p), &packet_ctx);
720  if (slen < 0) {
721  fr_cursor_head(cursor);
722  fr_cursor_free_list(cursor);
723  talloc_free(packet_ctx.tmp_ctx);
724  return slen;
725  }
726 
727  /*
728  * If slen is larger than the room in the packet,
729  * all kinds of bad things happen.
730  */
731  if (!fr_cond_assert(slen <= (end - p))) {
732  fr_cursor_head(cursor);
733  fr_cursor_free_list(cursor);
734  talloc_free(packet_ctx.tmp_ctx);
735  return -1;
736  }
737 
738  p += slen;
739  talloc_free_children(packet_ctx.tmp_ctx);
740  }
741 
742  /*
743  * We've parsed the whole packet, return that.
744  */
745  talloc_free(packet_ctx.tmp_ctx);
746  return packet_len;
747 }

772 /** Encode a DHCPv6 packet
773  *
774  */
775 ssize_t fr_dhcpv6_encode(fr_dbuff_t *dbuff, uint8_t const *original, size_t length, int msg_type, fr_pair_t *vps)
776 {
777  fr_pair_t *vp;
778  fr_dict_attr_t const *root;
779  ssize_t slen;
780  fr_cursor_t cursor;
781  fr_dhcpv6_encode_ctx_t packet_ctx;
782 
783  root = fr_dict_root(dict_dhcpv6);
784 
785  if (!msg_type) {
786  vp = fr_pair_find_by_da(&vps, attr_packet_type);
787  if (vp) msg_type = vp->vp_uint32;
788  }
789 
790  if ((msg_type <= 0) || (msg_type > 255)) {
791  fr_strerror_printf("Invalid message type %d", msg_type);
792  return -1;
793  }
794 
795  FR_DBUFF_IN_RETURN(dbuff, (uint8_t)msg_type);
796 
797  switch (msg_type) {
798  case FR_DHCPV6_RELAY_REPLY:
799  if (!original) return -1;
800 
801  FR_DBUFF_IN_MEMCPY_RETURN(dbuff, original + 1, 1 + 32);
802  break;
803 
804  case FR_DHCPV6_RELAY_FORWARD:
805  FR_DBUFF_EXTEND_LOWAT_OR_RETURN(dbuff, 1 + 32);
806 
807  vp = fr_pair_find_by_da(&vps, attr_hop_count);
808  if (vp) (void) fr_value_box_to_network_dbuff(NULL, dbuff, &vp->data);
809 
810  vp = fr_pair_find_by_da(&vps, attr_relay_link_address);
811  if (vp) (void) fr_value_box_to_network_dbuff(NULL, dbuff, &vp->data);
812 
813  vp = fr_pair_find_by_da(&vps, attr_relay_peer_address);
814  if (vp) (void) fr_value_box_to_network_dbuff(NULL, dbuff, &vp->data);
815 
816  break;
817 
818  default:
819  /*
820  * Copy over original transaction ID if we have it.
821  */
822  if (original) {
823  FR_DBUFF_IN_MEMCPY_RETURN(dbuff, original + 1, 3);
824  } else {
825  /*
826  * We can set an XID, or we can pick a random one.
827  */
828  vp = fr_pair_find_by_da(&vps, attr_transaction_id);
829  if (vp && (vp->vp_length >= 3)) {
830  FR_DBUFF_IN_MEMCPY_RETURN(dbuff, vp->vp_octets, 3);
831  } else {
832  uint8_t id[sizeof(uint32_t)];
833 
834  fr_net_from_uint32(id, fr_rand());
835  FR_DBUFF_IN_MEMCPY_RETURN(dbuff, &id[1], 3);
836  }
837  }
838  break;
839  }
840 
841  /*
842  * Encode options.
843  */
844  packet_ctx.root = root;
845  packet_ctx.original = original;
846  packet_ctx.original_length = length;
847 
848  fr_cursor_talloc_iter_init(&cursor, &vps, fr_dhcpv6_next_encodable, dict_dhcpv6, fr_pair_t);
849  while ((fr_dbuff_extend(dbuff) > 0) && (fr_cursor_current(&cursor) != NULL)) {
850  slen = fr_dhcpv6_encode_option(dbuff, &cursor, &packet_ctx);
851  switch (slen) {
852  case PAIR_ENCODE_SKIPPED:
853  continue;
854 
855  case PAIR_ENCODE_FATAL_ERROR:
856  return slen;
857 
858  default:
859  break;
860 
861  }
862 
863  if (slen < 0) return slen - fr_dbuff_used(dbuff);
864  }
865 
866  return fr_dbuff_used(dbuff);
867 }

869 /** Bootstrap a reply from the request
870  *
871  * We should arguably operate on VPs instead of raw packets.
872  * However, that would prevent us from properly copying structures.
873  *
874  * @param[in] ctx the context to use for allocations
875  * @param[out] reply_list list of the reply attributes to create
876  * @param[in] packet the input packet to check
877  * @param[in] packet_len the length of the input packet.
878  */
879 int fr_dhcpv6_reply_initialize(TALLOC_CTX *ctx, fr_pair_list_t *reply_list, uint8_t const *packet, size_t packet_len)
880 {
881  uint8_t const *option, *options, *end;
882  ssize_t slen;
883  fr_cursor_t cursor;
884  fr_pair_t *vp;
885  fr_dhcpv6_decode_ctx_t packet_ctx;
886 
887  end = packet + packet_len;
888 
889  fr_pair_list_init(reply_list);
890  fr_cursor_init(&cursor, reply_list);
891  packet_ctx.tmp_ctx = talloc_init_const("tmp");
892 
893  /*
894  * For normal packets, echo the Client-Id back in the
895  * reply. Note that the Client-Id attribute doesn't
896  * always need to exist
897  */
898  if (packet[0] != FR_DHCPV6_RELAY_FORWARD) {
899  options = packet + 4;
900 
901  option = fr_dhcpv6_option_find(options, end, FR_CLIENT_ID);
902  if (!option) return 0;
903 
904  slen = fr_dhcpv6_decode_option(ctx, &cursor, dict_dhcpv6, option, end - option, &packet_ctx);
905  talloc_free(packet_ctx.tmp_ctx);
906  return slen;
907  }
908 
909  /*
910  * Relay-Forward packets MAY include an Interface-ID. In
911  * which case it MUST be echoed in the reply/
912  */
913  options = packet + DHCPV6_RELAY_HDR_LEN;
914  option = fr_dhcpv6_option_find(options, end, FR_INTERFACE_ID);
915  if (option) {
916  slen = fr_dhcpv6_decode_option(ctx, &cursor, dict_dhcpv6, option, end - option, &packet_ctx);
917  if (slen <= 0) {
918  talloc_free(packet_ctx.tmp_ctx);
919  return slen;
920  }
921  }
922 
923  /*
924  * The reply to a Relay-Forward is a Relay-Reply.
925  *
926  * Since we call ourselves recursively, we have to add
927  * this to the reply now. We can't rely on
928  * proto_dhcpv6_process() to do it for us.
929  */
930  vp = fr_pair_afrom_da(ctx, attr_packet_type);
931  if (!vp) goto fail;
932 
933  vp->vp_uint32 = FR_DHCPV6_RELAY_REPLY;
934  fr_cursor_append(&cursor, vp);
935 
936  /*
937  * A Relay-Forward message MUST contain a Relay-Message
938  */
939  option = fr_dhcpv6_option_find(options, end, FR_RELAY_MESSAGE);
940  if (!option) goto fail;
941 
942  vp = fr_pair_afrom_da(ctx, attr_relay_message);
943  if (!vp) goto fail;
944 
945  fr_cursor_append(&cursor, vp);
946 
947  /*
948  * Recurse to create the appropriate nested VPs.
949  */
950  if (fr_dhcpv6_reply_initialize(vp, &vp->vp_group, option + 4, DHCPV6_GET_OPTION_LEN(option)) < 0) {
951  fail:
952  talloc_free(packet_ctx.tmp_ctx);
953  return -1;
954  }
955 
956  talloc_free(packet_ctx.tmp_ctx);
957  return 0;
958 }
