76 /** Add an additional destructor to a talloc chunk
77  *
78  * @param[in] fire_ctx When this ctx is freed the destructor function
79  * will be called.
80  * @param[in] disarm_ctx When this ctx is freed the destructor will be
81  * disarmed. May be NULL. #talloc_destructor_disarm
82  * may be used to disarm the destructor too.
83  * @param[in] func to call when the fire_ctx is freed.
84  * @param[in] uctx data to pass to the above function.
85  * @return
86  * - A handle to access the destructor on success.
87  * - NULL on failure.
88  */
89 fr_talloc_destructor_t *talloc_destructor_add(TALLOC_CTX *fire_ctx, TALLOC_CTX *disarm_ctx,
90  fr_talloc_free_func_t func, void const *uctx)
91 {
92  fr_talloc_destructor_t *d;
93 
94  if (!fire_ctx) {
95  fr_strerror_printf("No firing ctx provided when setting destructor");
96  return NULL;
97  }
98 
99  d = talloc(fire_ctx, fr_talloc_destructor_t);
100  if (!d) {
101  oom:
102  fr_strerror_printf("Out of Memory");
103  return NULL;
104  }
105 
106  d->fire = fire_ctx;
107  d->func = func;
108  memcpy(&d->uctx, &uctx, sizeof(d->uctx));
109 
110  if (disarm_ctx) {
111  fr_talloc_destructor_disarm_t *ds;
112 
113  ds = talloc(disarm_ctx, fr_talloc_destructor_disarm_t);
114  if (!ds) {
115  talloc_free(d);
116  goto oom;
117  }
118  ds->d = d;
119  d->ds = ds;
120  talloc_set_destructor(ds, _talloc_destructor_disarm);
121  }
122 
123  talloc_set_destructor(d, _talloc_destructor_fire);
124 
125  return d;
126 }

167 /** Return a page aligned talloc memory array
168  *
169  * Because we can't intercept talloc's malloc() calls, we need to do some tricks
170  * in order to get the first allocation in the array page aligned, and to limit
171  * the size of the array to a multiple of the page size.
172  *
173  * The reason for wanting a page aligned talloc array, is it allows us to
174  * mprotect() the pages that belong to the array.
175  *
176  * Talloc chunks appear to be allocated within the protected region, so this should
177  * catch frees too.
178  *
179  * @param[in] ctx to allocate array memory in.
180  * @param[out] start The first aligned address in the array.
181  * @param[in] alignment What alignment the memory chunk should have.
182  * @param[in] size How big to make the array. Will be corrected to a multiple
183  * of the page size. The actual array size will be size
184  * rounded to a multiple of the (page_size), + page_size
185  * @return
186  * - A talloc chunk on success.
187  * - NULL on failure.
188  */
189 TALLOC_CTX *talloc_aligned_array(TALLOC_CTX *ctx, void **start, size_t alignment, size_t size)
190 {
191  size_t rounded;
192  size_t array_size;
193  void *next;
194  TALLOC_CTX *array;
195 
196  rounded = ROUND_UP(size, alignment); /* Round up to a multiple of the page size */
197  if (rounded == 0) rounded = alignment;
198 
199  array_size = rounded + alignment;
200  array = talloc_array(ctx, uint8_t, array_size); /* Over allocate */
201  if (!array) {
202  fr_strerror_printf("Out of memory");
203  return NULL;
204  }
205 
206  next = (void *)ROUND_UP((uintptr_t)array, alignment); /* Round up address to the next multiple */
207  *start = next;
208 
209  return array;
210 }

212 /** Return a page aligned talloc memory pool
213  *
214  * Because we can't intercept talloc's malloc() calls, we need to do some tricks
215  * in order to get the first allocation in the pool page aligned, and to limit
216  * the size of the pool to a multiple of the page size.
217  *
218  * The reason for wanting a page aligned talloc pool, is it allows us to
219  * mprotect() the pages that belong to the pool.
220  *
221  * Talloc chunks appear to be allocated within the protected region, so this should
222  * catch frees too.
223  *
224  * @param[in] ctx to allocate pool memory in.
225  * @param[out] start A page aligned address within the pool. This can be passed
226  * to mprotect().
227  * @param[out] end of the pages that should be protected.
228  * @param[in] size How big to make the pool. Will be corrected to a multiple
229  * of the page size. The actual pool size will be size
230  * rounded to a multiple of the (page_size), + page_size
231  */
232 TALLOC_CTX *talloc_page_aligned_pool(TALLOC_CTX *ctx, void **start, void **end, size_t size)
233 {
234  size_t rounded, page_size = (size_t)getpagesize();
235  size_t hdr_size, pool_size;
236  void *next, *chunk;
237  TALLOC_CTX *pool;
238 
239  rounded = ROUND_UP(size, page_size); /* Round up to a multiple of the page size */
240  if (rounded == 0) rounded = page_size;
241 
242  pool_size = rounded + page_size;
243  pool = talloc_pool(ctx, pool_size); /* Over allocate */
244  if (!pool) {
245  fr_strerror_printf("Out of memory");
246  return NULL;
247  }
248 
249  chunk = talloc_size(pool, 1); /* Get the starting address */
250  if (!fr_cond_assert((chunk > pool) && ((uintptr_t)chunk < ((uintptr_t)pool + rounded)))) {
251  fr_strerror_printf("Initial allocation outside of pool memory");
252  error:
253  talloc_free(pool);
254  return NULL;
255  }
256  hdr_size = (uintptr_t)chunk - (uintptr_t)pool;
257 
258  next = (void *)ROUND_UP((uintptr_t)chunk, page_size); /* Round up address to the next page */
259 
260  /*
261  * Depending on how talloc allocates the chunk headers,
262  * the memory allocated here might not align to a page
263  * boundary, but that's ok, we just need future allocations
264  * to occur on or after 'next'.
265  */
266  if (((uintptr_t)next - (uintptr_t)chunk) > 0) {
267  size_t pad_size;
268  void *padding;
269 
270  pad_size = ((uintptr_t)next - (uintptr_t)chunk);
271  if (pad_size > hdr_size) {
272  pad_size -= hdr_size; /* Save ~111 bytes by not over-padding */
273  } else {
274  pad_size = 1;
275  }
276 
277  padding = talloc_size(pool, pad_size);
278  if (!fr_cond_assert(((uintptr_t)padding + (uintptr_t)pad_size) >= (uintptr_t)next)) {
279  fr_strerror_printf("Failed padding pool memory");
280  goto error;
281  }
282  }
283 
284  *start = next; /* This is the address we feed into mprotect */
285  *end = (void *)((uintptr_t)next + (uintptr_t)rounded);
286 
287  /*
288  * Disabled until samba fixes upstream issue
289  * with memlimits on pools.
290  */
291 // if (talloc_set_memlimit(pool, pool_size) < 0) goto error; /* Don't allow allocations outside of the pool */
292 
293  return pool;
294 }

373  * @return duped string.
374  */
375 char *talloc_bstrdup(TALLOC_CTX *ctx, char const *in)
376 {
377  char *p;

486 /** Concatenate to + from
487  *
488  * @param[in] ctx to allocate realloced buffer in.
489  * @param[in] to talloc string buffer to append to.
490  * @param[in] from talloc string buffer to append.
491  * @return

519 /** Concatenate to + ...
520  *
521  * @param[in] ctx to allocate realloced buffer in.
522  * @param[in] to talloc string buffer to append to.
523  * @param[in] argc how many variadic arguments were passed.
524  * @param[in] ... talloc string buffer(s) to append.
525  * Arguments can be NULL to simplify
526  * calling logic.
527  * @return
528  * - NULL if to or from are NULL or if the realloc fails.
529  * Note: You'll still need to free to if this function
530  * returns NULL.
531  * - The concatenation of to + from. After this function
532  * returns to may point to invalid memory and should
533  * not be used.
534  */
535 char *talloc_buffer_append_variadic_buffer(TALLOC_CTX *ctx, char *to, int argc, ...)
536 {
537  va_list ap_val, ap_len;
538  int i;
539 
540  size_t to_len, total_len = 0;
541  char *out, *p;
542 
543  if (!to) return NULL;
544 
545  va_start(ap_val, argc);
546  va_copy(ap_len, ap_val);
547 
548  total_len += to_len = talloc_array_length(to) - 1;
549 
550  /*
551  * Figure out how much we need to realloc
552  */
553  for (i = 0; i < argc; i++) {
554  char *arg;
555 
556  arg = va_arg(ap_len, char *);
557  if (!arg) continue;
558 
559  total_len += (talloc_array_length(arg) - 1);
560  }
561 
562  /*
563  * It's a noop...
564  */
565  if (total_len == to_len) {
566  va_end(ap_val);
567  va_end(ap_len);
568  return to;
569  }
570 
571  out = talloc_realloc(ctx, to, char, total_len + 1);
572  if (!out) goto finish;
573 
574  p = out + to_len;
575 
576  /*
577  * Copy the args in
578  */
579  for (i = 0; i < argc; i++) {
580  char *arg;
581  size_t len;
582 
583  arg = va_arg(ap_val, char *);
584  if (!arg) continue;
585 
586  len = talloc_array_length(arg) - 1;
587 
588  memcpy(p, arg, len);
589  p += len;
590  }
591  *p = '\0';
592 
593 finish:
594  va_end(ap_val);
595  va_end(ap_len);
596 
597  return out;
598 }

678 /** Add a NULL pointer to an array of pointers
679  *
680  * This is needed by some 3rd party libraries which take NULL terminated
681  * arrays for arguments.
682  *
683  * If allocation fails, NULL will be returned and the original array
684  * will not be touched.
685  *
686  * @param[in] array to null terminate. Will be invalidated (realloced).
687  * @return
688  * - NULL if array is NULL, or if reallocation fails.
689  * - A realloced version of array with an additional NULL element.
690  */
691 void **talloc_array_null_terminate(void **array)
692 {
693  size_t len;
694  TALLOC_CTX *ctx;
695  void **new;

772 /** Allocate and initialize a TALLOC_CHILD_CTX
773  *
774  * The TALLOC_CHILD_CTX ensures ordering for allocators and
775  * destructors. When a TALLOC_CHILD_CTX is created, it is added to
776  * parent, in FILO order. In contrast, the basic talloc operations
777  * do not guarantee any kind of order.
778  *
779  * When the TALLOC_CHILD_CTX is freed, the children are freed in FILO
780  * order. That process ensures that the children are freed before
781  * the parent, and that the younger siblings are freed before the
782  * older siblings.
783  *
784  * The idea is that if we have an initializer for A, which in turn
785  * initializes B and C. When the memory is freed, we should do the
786  * operations in the reverse order.
787  */
788 TALLOC_CHILD_CTX *talloc_child_ctx_init(TALLOC_CTX *ctx)
789 {
790  TALLOC_CHILD_CTX *child;
791 
792  child = talloc_zero(ctx, TALLOC_CHILD_CTX);
793  if (!child) return NULL;
794 
795  talloc_set_destructor(child, _child_ctx_free);
796  return child;
797 }

800  *
801  */
802 TALLOC_CHILD_CTX *talloc_child_ctx_alloc(TALLOC_CHILD_CTX *parent)
803 {
804  TALLOC_CHILD_CTX *child;
805 
806  child = talloc(parent, TALLOC_CHILD_CTX);
807  if (!child) return NULL;
808 
809  child->next = parent->next;
810  parent->next = child;
811  return child;
