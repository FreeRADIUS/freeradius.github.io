78 static unlang_action_t unlang_foreach_next(rlm_rcode_t *p_result, request_t *request)
79 {
80  fr_pair_t *vp;
81  unlang_stack_t *stack = request->stack;
82  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
83  unlang_t *instruction = frame->instruction;
84  unlang_frame_state_foreach_t *foreach = NULL;
85  unlang_group_t *g;
86 
87  g = unlang_generic_to_group(instruction);
88 
89  foreach = talloc_get_type_abort(frame->state, unlang_frame_state_foreach_t);
90 
91  vp = fr_cursor_current(&foreach->cursor);
92  if (!vp) {
93  *p_result = frame->result;
94 #ifndef NDEBUG
95  fr_assert(foreach->indent == request->log.unlang_indent);
96 #endif
97  return UNLANG_ACTION_CALCULATE_RESULT;
98  }
99  (void) fr_cursor_next(&foreach->cursor);
100 
101 #ifndef NDEBUG
102  RDEBUG2("# looping with: Foreach-Variable-%d = %pV", foreach->depth, &vp->data);
103 #endif
104 
105  fr_assert(vp);
106 
107  /*
108  * Add the vp to the request, so that
109  * xlat.c, xlat_foreach() can find it.
110  */
111  foreach->variable = vp;
112  request_data_add(request, FOREACH_REQUEST_DATA, foreach->depth, &foreach->variable,
113  false, false, false);
114 
115  /*
116  * Push the child, and yield for a later return.
117  */
118  if (unlang_interpret_push(request, g->children, frame->result, UNLANG_NEXT_SIBLING, UNLANG_SUB_FRAME) < 0) {
119  *p_result = RLM_MODULE_FAIL;
120  return UNLANG_ACTION_STOP_PROCESSING;
121  }
122 
123  repeatable_set(frame);
124 
125  return UNLANG_ACTION_PUSHED_CHILD;
126 }

129 static unlang_action_t unlang_foreach(rlm_rcode_t *p_result, request_t *request)
130 {
131  unlang_stack_t *stack = request->stack;
132  unlang_stack_frame_t *frame;
133  unlang_t *instruction;
134  unlang_frame_state_foreach_t *foreach = NULL;
135 
136  unlang_group_t *g;
137  unlang_foreach_t *gext;
138 
139  int i, foreach_depth = 0;
140  fr_pair_t *vps;
141 
142  frame = &stack->frame[stack->depth];
143  instruction = frame->instruction;
144  g = unlang_generic_to_group(instruction);
145  gext = unlang_group_to_foreach(g);
146 
147  /*
148  * Ensure any breaks terminate here...
149  */
150  break_point_set(frame);
151 
152  /*
153  * Figure out foreach depth by walking back up the stack
154  */
155  if (stack->depth > 0) for (i = (stack->depth - 1); i >= 0; i--) {
156  unlang_t *our_instruction;
157  our_instruction = stack->frame[i].instruction;
158  if (!our_instruction || (our_instruction->type != UNLANG_TYPE_FOREACH)) continue;
159  foreach_depth++;
160  }
161 
162  if (foreach_depth >= (int)NUM_ELEMENTS(xlat_foreach_names)) {
163  REDEBUG("foreach Nesting too deep!");
164  *p_result = RLM_MODULE_FAIL;
165  return UNLANG_ACTION_CALCULATE_RESULT;
166  }
167 
168  MEM(frame->state = foreach = talloc_zero(stack, unlang_frame_state_foreach_t));
169 
170  /*
171  * Copy the VPs from the original request, this ensures deterministic
172  * behaviour if someone decides to add or remove VPs in the set we're
173  * iterating over.
174  */
175  if (tmpl_copy_pairs(frame->state, &vps, request, gext->vpt) < 0) { /* nothing to loop over */
176  *p_result = RLM_MODULE_NOOP;
177  return UNLANG_ACTION_CALCULATE_RESULT;
178  }
179 
180  fr_assert(vps != NULL);
181 
182  foreach->request = request;
183  foreach->depth = foreach_depth;
184  foreach->vps = vps;
185  fr_cursor_talloc_init(&foreach->cursor, &foreach->vps, fr_pair_t);
186 #ifndef NDEBUG
187  foreach->indent = request->log.unlang_indent;
188 #endif
189  talloc_set_destructor(foreach, _free_unlang_frame_state_foreach);
190 
191  frame->process = unlang_foreach_next;
192  return unlang_foreach_next(p_result, request);
193 }

195 static unlang_action_t unlang_break(rlm_rcode_t *p_result, request_t *request)
196 {
197  unlang_stack_t *stack = request->stack;
198  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
199  unlang_t *instruction = frame->instruction;
200 
201  RDEBUG2("%s", unlang_ops[instruction->type].name);
202 
203  *p_result = frame->result;
204 
205  /*
206  * Stop at the next break point, or if we hit
207  * the a top frame.
208  */
209  return unwind_to_break(stack);
210 }
