145 static ssize_t mod_read_command(fr_listen_t *li, UNUSED void **packet_ctx, UNUSED fr_time_t *recv_time_p, uint8_t *buffer, UNUSED size_t buffer_len, UNUSED size_t *leftover, UNUSED uint32_t *priority, UNUSED bool *is_dup
146 )
147 {
148  proto_control_unix_t const *inst = talloc_get_type_abort_const(li->app_io_instance, proto_control_unix_t);
149  proto_control_unix_thread_t *thread = talloc_get_type_abort(li->thread_instance, proto_control_unix_thread_t);
150  fr_conduit_hdr_t *hdr = (fr_conduit_hdr_t *) buffer;
151  uint32_t status;
152  uint8_t *cmd = buffer + sizeof(*hdr);
153  int rcode;
154  char string[1024];
155 
156  hdr->length = ntohl(hdr->length);
157  if (hdr->length >= sizeof(string)) goto fail;
158 
159  /*
160  * If the write gives us nothing, send an empty SUCCESS back.
161  */
162  if (!hdr->length) {
163  status = FR_CONDUIT_SUCCESS;
164  goto done;
165  }
166 
167  /*
168  * fr_command_run() expects a zero-terminated string...
169  */
170  memcpy(string, cmd, hdr->length);
171  string[hdr->length] = '\0';
172 
173  /*
174  * Content is the string we need help for.
175  */
176  if (htons(hdr->conduit) == FR_CONDUIT_HELP) {
177  fr_radmin_help(thread->stdout, string);
178  // @todo - have in-band signalling saying that the help is done?
179  // we want to be able to say that *this* help is done.
180  // the best way to do that is to have a token, and every command
181  // from the other end sends a token, and we echo it back here...
182  // Or, since we currently can't do streaming commands, it's OK?
183  // Or, we assume that origin 0 is for interactive commands,
184  // and that the other origins are for streaming output...
185  status = FR_CONDUIT_SUCCESS;
186  goto done;
187  }
188 
189  if (htons(hdr->conduit) == FR_CONDUIT_COMPLETE) {
190  uint16_t start;
191 
192  if (hdr->length < 2) goto fail;
193 
194  start = (string[0] << 8) | string[1];
195 
196  thread->misc_conduit = FR_CONDUIT_COMPLETE;
197 
198  fr_radmin_complete(thread->misc, string + 2, start);
199  thread->misc_conduit = FR_CONDUIT_STDOUT;
200  status = FR_CONDUIT_SUCCESS;
201  goto done;
202  }
203 
204  if (htons(hdr->conduit) != FR_CONDUIT_STDIN) {
205  DEBUG("ERROR: Ignoring data which is from wrong input");
206  return 0;
207  }
208 
209  DEBUG("radmin-remote> %.*s", (int) hdr->length, cmd);
210 
211  rcode = fr_radmin_run(thread->info, thread->stdout, thread->stderr, string, inst->read_only);
212  if (rcode < 0) {
213 fail:
214  status = FR_CONDUIT_FAIL;
215 
216  } else if (rcode == 0) {
217  /*
218  * The other end should keep track of it's
219  * context, and send us full lines.
220  */
221  (void) fr_command_clear(0, thread->info);
222  status = FR_CONDUIT_PARTIAL;
223  } else {
224  status = FR_CONDUIT_SUCCESS;
225  }
226 
227 done:
228  status = htonl(status);
229  (void) fr_conduit_write(thread->sockfd, FR_CONDUIT_CMD_STATUS, &status, sizeof(status));
230 
231  return 0;
232 }

237 static ssize_t mod_read_init(fr_listen_t *li, UNUSED void **packet_ctx, UNUSED fr_time_t *recv_time_p, uint8_t *buffer, size_t buffer_len, UNUSED size_t *leftover, UNUSED uint32_t *priority, UNUSED bool *is_dup
238 )
239 {
240  proto_control_unix_thread_t *thread = talloc_get_type_abort(li->thread_instance, proto_control_unix_thread_t);
241  fr_conduit_hdr_t *hdr = (fr_conduit_hdr_t *) buffer;
242  uint32_t magic;
243 
244  if (htons(hdr->conduit) != FR_CONDUIT_INIT_ACK) {
245  DEBUG("ERROR: Connection is missing initial ACK packet.");
246  return -1;
247  }
248 
249  if (buffer_len < sizeof(*hdr)) {
250  DEBUG("ERROR: Initial ACK is malformed");
251  return -1;
252  }
253 
254  if (htonl(hdr->length) != 8) {
255  DEBUG("ERROR: Initial ACK has wrong length (%lu).", (size_t) htonl(hdr->length));
256  return -1;
257  }
258 
259  memcpy(&magic, buffer + sizeof(*hdr), sizeof(magic));
260  magic = htonl(magic);
261  if (magic != FR_CONDUIT_MAGIC) {
262  DEBUG("ERROR: Connection from incompatible version of radmin.");
263  return -1;
264  }
265 
266  /*
267  * Next 4 bytes are zero, we ignore them.
268  */
269  if (write(thread->sockfd, buffer, buffer_len) < (ssize_t) buffer_len) {
270  DEBUG("ERROR: Blocking write to socket... oops");
271  return -1;
272  }
273 
274  thread->read = mod_read_command;
275 
276  return 0;
277 }

279 static ssize_t mod_read(fr_listen_t *li, void **packet_ctx, fr_time_t *recv_time_p, uint8_t *buffer, size_t buffer_len, size_t *leftover, uint32_t *priority, bool *is_dup)
280 {
281  proto_control_unix_thread_t *thread = talloc_get_type_abort(li->thread_instance, proto_control_unix_thread_t);
282  ssize_t data_size;
283 
284  fr_conduit_type_t conduit;
285  bool want_more;
286 
287  /*
288  * Read data into the buffer.
289  */
290  data_size = fr_conduit_read_async(thread->sockfd, &conduit, buffer, buffer_len, leftover, &want_more);
291  if (data_size < 0) {
292  DEBUG2("proto_control_unix got read error %zd: %s", data_size, fr_strerror());
293  return data_size;
294  }
295 
296  /*
297  * Note that we return ERROR for all bad packets, as
298  * there's no point in reading packets from a TCP
299  * connection which isn't sending us properly formatted
300  * packets.
301  */
302 
303  /*
304  * Not enough for a full packet, ask the caller to read more.
305  */
306  if (want_more) {
307  return 0;
308  }
309 
310  // @todo - maybe convert timestamp?
311  *recv_time_p = fr_time();
312  *leftover = 0;
313 
314  /*
315  * proto_control sets the priority
316  */
317 
318  /*
319  * Print out what we received.
320  */
321  DEBUG3("proto_control_unix - Received command packet length %d on %s",
322  (int) data_size, thread->name);
323 
324  /*
325  * Run the state machine to process the rest of the packet.
326  */
327  return thread->read(li, packet_ctx, recv_time_p, buffer, (size_t) data_size, leftover, priority, is_dup);
328 }

377 /** Initialise a socket for use with peercred authentication
378  *
379  * This function initialises a socket and path in a way suitable for use with
380  * peercred.
381  *
382  * @param path to socket.
383  * @param uid that should own the socket (linux only).
384  * @param gid that should own the socket (linux only).
385  * @return 0 on success -1 on failure.
386  */
387 #ifdef __linux__
388 static int fr_server_domain_socket_peercred(char const *path, uid_t uid, gid_t gid)
389 #else
390 static int fr_server_domain_socket_peercred(char const *path, uid_t UNUSED uid, UNUSED gid_t gid)
391 #endif
392 {
393  int sockfd;
394  size_t len;
395  socklen_t socklen;
396  struct sockaddr_un salocal;
397  struct stat buf;
398 
399  if (!path) {
400  fr_strerror_printf("No path provided, was NULL");
401  return -1;
402  }
403 
404  len = strlen(path);
405  if (len >= sizeof(salocal.sun_path)) {
406  fr_strerror_printf("Path too long in socket filename");
407  return -1;
408  }
409 
410  if ((sockfd = socket(AF_UNIX, SOCK_STREAM, 0)) < 0) {
411  fr_strerror_printf("Failed creating socket: %s", fr_syserror(errno));
412  return -1;
413  }
414 
415  memset(&salocal, 0, sizeof(salocal));
416  salocal.sun_family = AF_UNIX;
417  memcpy(salocal.sun_path, path, len + 1); /* SUN_LEN does strlen */
418 
419  socklen = SUN_LEN(&salocal);
420 
421  /*
422  * Check the path.
423  */
424  if (stat(path, &buf) < 0) {
425  if (errno != ENOENT) {
426  fr_strerror_printf("Failed to stat %s: %s", path, fr_syserror(errno));
427  close(sockfd);
428  return -1;
429  }
430 
431  /*
432  * FIXME: Check the enclosing directory?
433  */
434  } else { /* it exists */
435  int client_fd;
436 
437  if (!S_ISREG(buf.st_mode)
438 #ifdef S_ISSOCK
439  && !S_ISSOCK(buf.st_mode)
440 #endif
441  ) {
442  fr_strerror_printf("Cannot turn %s into socket", path);
443  close(sockfd);
444  return -1;
445  }
446 
447  /*
448  * Refuse to open sockets not owned by us.
449  */
450  if (buf.st_uid != geteuid()) {
451  fr_strerror_printf("We do not own %s", path);
452  close(sockfd);
453  return -1;
454  }
455 
456  /*
457  * Check if a server is already listening on the
458  * socket?
459  */
460  client_fd = fr_socket_client_unix(path, false);
461  if (client_fd >= 0) {
462  fr_strerror_printf("Control socket '%s' is already in use", path);
463  close(client_fd);
464  close(sockfd);
465  return -1;
466  }
467 
468  if (unlink(path) < 0) {
469  fr_strerror_printf("Failed to delete %s: %s", path, fr_syserror(errno));
470  close(sockfd);
471  return -1;
472  }
473  }
474 
475  if (bind(sockfd, (struct sockaddr *)&salocal, socklen) < 0) {
476  fr_strerror_printf("Failed binding to %s: %s", path, fr_syserror(errno));
477  close(sockfd);
478  return -1;
479  }
480 
481  /*
482  * FIXME: There's a race condition here. But Linux
483  * doesn't seem to permit fchmod on domain sockets.
484  */
485  if (chmod(path, S_IRUSR | S_IWUSR | S_IRGRP | S_IWGRP) < 0) {
486  fr_strerror_printf("Failed setting permissions on %s: %s", path, fr_syserror(errno));
487  close(sockfd);
488  return -1;
489  }
490 
491  if (listen(sockfd, 8) < 0) {
492  fr_strerror_printf("Failed listening to %s: %s", path, fr_syserror(errno));
493  close(sockfd);
494  return -1;
495  }
496 
497 #ifdef O_NONBLOCK
498  {
499  int flags;
500 
501  if ((flags = fcntl(sockfd, F_GETFL, NULL)) < 0) {
502  fr_strerror_printf("Failure getting socket flags: %s", fr_syserror(errno));
503  close(sockfd);
504  return -1;
505  }
506 
507  flags |= O_NONBLOCK;
508  if( fcntl(sockfd, F_SETFL, flags) < 0) {
509  fr_strerror_printf("Failure setting socket flags: %s", fr_syserror(errno));
510  close(sockfd);
511  return -1;
512  }
513  }
514 #endif
515 
516  /*
517  * Changing socket permissions only works on linux.
518  * BSDs ignore socket permissions.
519  */
520 #ifdef __linux__
521  /*
522  * Don't chown it from (possibly) non-root to root.
523  * Do chown it from (possibly) root to non-root.
524  */
525  if ((uid != (uid_t) -1) || (gid != (gid_t) -1)) {
526  /*
527  * Don't do chown if it's already owned by us.
528  */
529  if (fstat(sockfd, &buf) < 0) {
530  fr_strerror_printf("Failed reading %s: %s", path, fr_syserror(errno));
531  close(sockfd);
532  return -1;
533  }
534 
535  if ((buf.st_uid != uid) || (buf.st_gid != gid)) {
536  rad_suid_up();
537  if (fchown(sockfd, uid, gid) < 0) {
538  fr_strerror_printf("Failed setting ownership of %s to (%d, %d): %s",
539  path, uid, gid, fr_syserror(errno));
540  rad_suid_down();
541  close(sockfd);
542  return -1;
543  }
544  rad_suid_down();
545  }
546  }
547 #endif
548 
549  return sockfd;
550 }

560 /** Alternative function for creating Unix domain sockets and enforcing permissions
561  *
562  * Unlike fr_server_unix_socket which is intended to be used with peercred auth
563  * this function relies on the file system to enforce access.
564  *
565  * The way it does this depends on the operating system. On Linux systems permissions
566  * can be set on the socket directly and the system will enforce them.
567  *
568  * On most other systems fchown and fchmod fail when called with socket descriptors,
569  * and although permissions can be changed in other ways, they're not enforced.
570  *
571  * For these systems we use the permissions on the parent directory to enforce
572  * permissions on the socket. It's not safe to modify these permissions ourselves
573  * due to TOCTOU attacks, so if they don't match what we require, we error out and
574  * get the user to change them (which arguably isn't any safer, but releases us of
575  * the responsibility).
576  *
577  * @note must be called without effective root permissions (fr_suid_down).
578  *
579  * @param[in] path where domain socket should be created.
580  * @param[in] uid Owner of the socket.
581  * @param[in] gid Group of the socket.
582  * @return a file descriptor for the bound socket on success, -1 on failure.
583  */
584 static int fr_server_domain_socket_perm(char const *path, uid_t uid, gid_t gid)
585 {
586  int dir_fd = -1, sock_fd = -1, parent_fd = -1;
587  char const *name;
588  char *buff = NULL, *dir = NULL, *p;
589 
590  uid_t euid;
591  gid_t egid;
592 
593  mode_t perm = 0;
594  struct stat st;
595 
596 
597  size_t len;
598 
599  socklen_t socklen;
600  struct sockaddr_un salocal;
601 
602  fr_assert(path);
603 
604  euid = geteuid();
605  egid = getegid();
606 
607  /*
608  * Determine the correct permissions for the socket, or its
609  * containing directory.
610  */
611  perm |= S_IREAD | S_IWRITE | S_IEXEC;
612  if (gid != (gid_t) -1) perm |= S_IRGRP | S_IWGRP | S_IXGRP;
613 
614  buff = talloc_strdup(NULL, path);
615  if (!buff) return -1;
616 
617  /*
618  * Some implementations modify it in place others use internal
619  * storage *sigh*. dirname also formats the path else we wouldn't
620  * be using it.
621  */
622  dir = dirname(buff);
623  if (dir != buff) {
624  dir = talloc_strdup(NULL, dir);
625  if (!dir) return -1;
626  talloc_free(buff);
627  }
628 
629  p = strrchr(dir, FR_DIR_SEP);
630  if (!p) {
631  fr_strerror_printf("Failed determining parent directory");
632  error:
633  talloc_free(dir);
634  if (sock_fd >= 0) close(sock_fd);
635  if (dir_fd >= 0) close(dir_fd);
636  if (parent_fd >= 0) close(parent_fd);
637  return -1;
638  }
639 
640  *p = '\0';
641 
642  /*
643  * Ensure the parent of the control socket directory exists,
644  * and the euid we're running under has access to it.
645  */
646  parent_fd = open(dir, O_DIRECTORY);
647  if (parent_fd < 0) {
648  struct passwd *user;
649  struct group *group;
650 
651  if (rad_getpwuid(NULL, &user, euid) < 0) goto error;
652  if (rad_getgrgid(NULL, &group, egid) < 0) {
653  talloc_free(user);
654  goto error;
