156 static void eap_peap_soh_verify(request_t *request, fr_radius_packet_t *packet,
157  uint8_t const *data, unsigned int data_len) {
158 
159  fr_pair_t *vp;
160  uint8_t eap_method_base;
161  uint32_t eap_vendor;
162  uint32_t eap_method;
163  int rv;
164 
165  MEM(vp = fr_pair_afrom_da(packet, attr_soh_supported));
166  vp->vp_bool = false;
167  fr_pair_add(&request->request_pairs, vp);
168 
169  if (data && data[0] == FR_EAP_METHOD_NAK) {
170  REDEBUG("SoH - client NAKed");
171  return;
172  }
173 
174  if (!data || data_len < 8) {
175  REDEBUG("SoH - eap payload too short");
176  return;
177  }
178 
179  eap_method_base = *data++;
180  if (eap_method_base != 254) {
181  REDEBUG("SoH - response is not extended EAP: %i", eap_method_base);
182  return;
183  }
184 
185  eap_vendor = soh_pull_be_24(data); data += 3;
186  if (eap_vendor != 0x137) {
187  REDEBUG("SoH - extended eap vendor %08x is not Microsoft", eap_vendor);
188  return;
189  }
190 
191  eap_method = soh_pull_be_32(data); data += 4;
192  if (eap_method != 0x21) {
193  REDEBUG("SoH - response eap type %08x is not EAP-SoH", eap_method);
194  return;
195  }
196 
197  rv = soh_verify(request, data, data_len - 8);
198  if (rv < 0) {
199  RPEDEBUG("SoH - error decoding payload");
200  } else {
201  vp->vp_uint32 = 1;
202  }
203 }

206 /*
207  * Verify the tunneled EAP message.
208  */
209 static int eap_peap_verify(request_t *request, peap_tunnel_t *peap_tunnel,
210  uint8_t const *data, size_t data_len)
211 {
212  eap_packet_raw_t const *eap_packet = (eap_packet_raw_t const *) data;
213  eap_type_t eap_method;
214 
215  /*
216  * No data, OR only 1 byte of EAP type.
217  */
218  if (!data || (data_len == 0) || ((data_len <= 1) && (data[0] != FR_EAP_METHOD_IDENTITY))) return 0;
219 
220  /*
221  * Since the full EAP header is sent for the EAP Extensions type (Type 33),
222  * but not for other Types, it is difficult for the implementation to distinguish
223  * an Extensions Request (Code 1) from an EAP Type 1 (Identity) Request packet.
224  *
225  * i.e. The only way to validate PEAP inner method packets properly is to know
226  * we just send a protected success/failure.
227  */
228  switch (peap_tunnel->status) {
229  case PEAP_STATUS_SENT_TLV_SUCCESS:
230  case PEAP_STATUS_SENT_TLV_FAILURE:
231  if (eap_packet->data[0] != FR_PEAP_EXTENSIONS_TYPE) {
232  REDEBUG("Invalid inner tunnel data, expected method (%u), got (%u)",
233  FR_PEAP_EXTENSIONS_TYPE, eap_packet->data[0]);
234  return -1;
235  }
236  return 0;
237 
238  default:
239  break;
240  }
241 
242  eap_method = data[0]; /* Inner EAP header misses off code and identifier */
243  switch (eap_method) {
244  case FR_EAP_METHOD_IDENTITY:
245  RDEBUG2("Received EAP-Identity-Response");
246  return 0;
247 
248  /*
249  * We normally do Microsoft MS-CHAPv2 (26), versus
250  * Cisco MS-CHAPv2 (29).
251  */
252  case FR_EAP_METHOD_MSCHAPV2:
253  default:
254  RDEBUG2("EAP method %s (%d)", eap_type2name(eap_method), eap_method);
255  return 0;
256  }
257

260 /*
261  * Convert a pseudo-EAP packet to a list of fr_pair_t's.
262  */
263 static fr_pair_t *eap_peap_inner_to_pairs(UNUSED request_t *request, fr_radius_packet_t *packet,
264  eap_round_t *eap_round,
265  uint8_t const *data, size_t data_len)
266 {
267  size_t total;
268  uint8_t *p;
269  fr_pair_t *vp = NULL, *head = NULL;
270  fr_cursor_t cursor;
271 
272  if (data_len > 65535) return NULL; /* paranoia */
273 
274  MEM(vp = fr_pair_afrom_da(packet, attr_eap_message));
275  total = data_len;
276  if (total > 249) total = 249;
277 
278  /*
279  * Hand-build an EAP packet from the crap in PEAP version 0.
280  */
281  MEM(fr_pair_value_mem_alloc(vp, &p, EAP_HEADER_LEN + total, false) == 0);
282  p[0] = FR_EAP_CODE_RESPONSE;
283  p[1] = eap_round->response->id;
284  p[2] = (data_len + EAP_HEADER_LEN) >> 8;
285  p[3] = (data_len + EAP_HEADER_LEN) & 0xff;
286  memcpy(p + EAP_HEADER_LEN, data, total);
287 
288  fr_cursor_init(&cursor, &head);
289  fr_cursor_append(&cursor, vp);
290  while (total < data_len) {
291  MEM(vp = fr_pair_afrom_da(packet, attr_eap_message));
292  fr_pair_value_memdup(vp, data + total, (data_len - total), false);
293 
294  total += vp->vp_length;
295 
296  fr_cursor_append(&cursor, vp);
297  }
298 
299  return head;
300 }

303 /*
304  * Convert a list of fr_pair_t's to an EAP packet, through the
305  * simple expedient of dumping the EAP message
306  */
307 static int eap_peap_inner_from_pairs(request_t *request, fr_tls_session_t *tls_session, fr_pair_t *vp)
308 {
309  fr_assert(vp != NULL);
310  fr_pair_t *this;
311  fr_cursor_t cursor;
312 
313  /*
314  * Send the EAP data in the first attribute, WITHOUT the
315  * header.
316  */
317  (tls_session->record_from_buff)(&tls_session->clean_in, vp->vp_octets + EAP_HEADER_LEN,
318  vp->vp_length - EAP_HEADER_LEN);
319 
320  /*
321  * Send the rest of the EAP data, but skipping the first VP.
322  */
323  fr_cursor_init(&cursor, &vp);
324  for (this = fr_cursor_next(&cursor);
325  this;
326  this = fr_cursor_next(&cursor)) {
327  (tls_session->record_from_buff)(&tls_session->clean_in, this->vp_octets, this->vp_length);
328  }
329 
330  fr_tls_session_send(request, tls_session);
331 
332  return 1;
333 }

336 /*
337  * See if there's a TLV in the response.
338  */
339 static int eap_peap_check_tlv(request_t *request, uint8_t const *data, size_t data_len)
340 {
341  eap_packet_raw_t const *eap_packet = (eap_packet_raw_t const *) data;
342 
343  if (data_len < 11) return 0;
344 
345  /*
346  * Look for success or failure.
347  */
348  if ((eap_packet->code == FR_EAP_CODE_RESPONSE) &&
349  (eap_packet->data[0] == FR_PEAP_EXTENSIONS_TYPE)) {
350  if (data[10] == EAP_TLV_SUCCESS) {
351  return 1;
352  }
353 
354  if (data[10] == EAP_TLV_FAILURE) {
355  RDEBUG2("Client rejected our response. The password is probably incorrect");
356  return 0;
357  }
358  }
359 
360  RDEBUG2("Unknown TLV %02x", data[10]);
361 
362  return 0;
363 }

366 /*
367  * Use a reply packet to determine what to do.
368  */
369 static rlm_rcode_t CC_HINT(nonnull) process_reply(eap_session_t *eap_session, fr_tls_session_t *tls_session,
370  request_t *request, fr_radius_packet_t *reply)
371 {
372  rlm_rcode_t rcode = RLM_MODULE_REJECT;
373  fr_pair_t *vp;
374  peap_tunnel_t *t = tls_session->opaque;
375 
376  if (RDEBUG_ENABLED2) {
377 
378  /*
379  * Note that we don't do *anything* with the reply
380  * attributes.
381  */
382  if (is_radius_code(reply->code)) {
383  RDEBUG2("Got tunneled reply %s", fr_packet_codes[reply->code]);
384  } else {
385  RDEBUG2("Got tunneled reply code %i", reply->code);
386  }
387  log_request_pair_list(L_DBG_LVL_2, request, reply->vps, NULL);
388  }
389 
390  switch (reply->code) {
391  case FR_CODE_ACCESS_ACCEPT:
392  RDEBUG2("Tunneled authentication was successful");
393  t->status = PEAP_STATUS_SENT_TLV_SUCCESS;
394  eap_peap_success(request, eap_session, tls_session);
395  rcode = RLM_MODULE_HANDLED;
396  break;
397 
398  case FR_CODE_ACCESS_REJECT:
399  RDEBUG2("Tunneled authentication was rejected");
400  t->status = PEAP_STATUS_SENT_TLV_FAILURE;
401  eap_peap_failure(request, eap_session, tls_session);
402  rcode = RLM_MODULE_HANDLED;
403  break;
404 
405  case FR_CODE_ACCESS_CHALLENGE:
406  RDEBUG2("Got tunneled Access-Challenge");
407 
408  /*
409  * PEAP takes only EAP-Message attributes inside
410  * of the tunnel. Any Reply-Message in the
411  * Access-Challenge is ignored.
412  */
413  vp = NULL;
414  fr_pair_list_copy_by_da(t, &vp, &reply->vps, attr_eap_message);
415 
416  /*
417  * Handle the ACK, by tunneling any necessary reply
418  * VP's back to the client.
419  */
420  if (vp) {
421  eap_peap_inner_from_pairs(request, tls_session, vp);
422  fr_pair_list_free(&vp);
423  }
424 
425  rcode = RLM_MODULE_HANDLED;
426  break;
427 
428  default:
429  RDEBUG2("Unknown RADIUS packet type %d: rejecting tunneled user", reply->code);
430  rcode = RLM_MODULE_REJECT;
431  break;
432  }
433 
434  return rcode;
435 }

468 /*
469  * Process the pseudo-EAP contents of the tunneled data.
470  */
471 unlang_action_t eap_peap_process(rlm_rcode_t *p_result, request_t *request,
472  eap_session_t *eap_session, fr_tls_session_t *tls_session)
473 {
474  peap_tunnel_t *t = tls_session->opaque;
475  request_t *fake = NULL;
476  fr_pair_t *vp;
477  rlm_rcode_t rcode = RLM_MODULE_REJECT;
478  uint8_t const *data;
479  size_t data_len;
480  eap_round_t *eap_round = eap_session->this_round;
481 
482  /*
483  * Just look at the buffer directly, without doing
484  * record_to_buff. This lets us avoid another data copy.
485  */
486  data_len = tls_session->clean_out.used;
487  tls_session->clean_out.used = 0;
488  data = tls_session->clean_out.data;
489 
490  RDEBUG2("PEAP state %s", peap_state(t));
491 
492  if ((t->status != PEAP_STATUS_TUNNEL_ESTABLISHED) && (eap_peap_verify(request, t, data, data_len) < 0)) {
493  REDEBUG("Tunneled data is invalid");
494  RETURN_MODULE_REJECT;
495  }
496 
497  switch (t->status) {
498  case PEAP_STATUS_TUNNEL_ESTABLISHED:
499  /* FIXME: should be no data in the buffer here, check & assert? */
500 
501  if (SSL_session_reused(tls_session->ssl)) {
502  RDEBUG2("Skipping Phase2 because of session resumption");
503  t->session_resumption_state = PEAP_RESUMPTION_YES;
504  if (t->soh) {
505  t->status = PEAP_STATUS_WAIT_FOR_SOH_RESPONSE;
506  RDEBUG2("Requesting SoH from client");
507  eap_peap_soh(request, tls_session);
508 
509  rcode = RLM_MODULE_HANDLED;
510  goto finish;
511  }
512  /* we're good, send success TLV */
513  t->status = PEAP_STATUS_SENT_TLV_SUCCESS;
514  eap_peap_success(request, eap_session, tls_session);
515 
516  } else {
517  /* send an identity request */
518  t->session_resumption_state = PEAP_RESUMPTION_NO;
519  t->status = PEAP_STATUS_INNER_IDENTITY_REQ_SENT;
520  eap_peap_identity(request, eap_session, tls_session);
521  }
522  rcode = RLM_MODULE_HANDLED;
523  goto finish;
524 
525  case PEAP_STATUS_INNER_IDENTITY_REQ_SENT:
526  /* we're expecting an identity response */
527  if (data[0] != FR_EAP_METHOD_IDENTITY) {
528  REDEBUG("Expected EAP-Identity, got something else");
529  rcode = RLM_MODULE_REJECT;
530  goto finish;
531  }
532 
533  /*
534  * Save it for later.
535  */
536  MEM(t->username = fr_pair_afrom_da(t, attr_user_name));
537  t->username->vp_tainted = true;
538 
539  fr_pair_value_bstrndup(t->username, (char const *)data + 1, data_len - 1, true);
540 
541  RDEBUG2("Got inner identity \"%pV\"", &t->username->data);
542  if (t->soh) {
543  t->status = PEAP_STATUS_WAIT_FOR_SOH_RESPONSE;
544  RDEBUG2("Requesting SoH from client");
545  eap_peap_soh(request, tls_session);
546  rcode = RLM_MODULE_HANDLED;
547  goto finish;
548  }
549  t->status = PEAP_STATUS_PHASE2_INIT;
550  break;
551 
552  case PEAP_STATUS_WAIT_FOR_SOH_RESPONSE:
553  fake = request_alloc_fake(request, NULL);
554  fr_assert(!fake->request_pairs);
555  eap_peap_soh_verify(fake, fake->packet, data, data_len);
556  setup_fake_request(request, fake, t);
557 
558  if (t->soh_virtual_server) fake->server_cs = virtual_server_find(t->soh_virtual_server);
559 
560  RDEBUG2("Sending SoH request to server %s",
561  fake->server_cs ? cf_section_name2(fake->server_cs) : "NULL");
562  rad_virtual_server(&rcode, fake);
563 
564  if (fake->reply->code != FR_CODE_ACCESS_ACCEPT) {
565  RDEBUG2("SoH was rejected");
566  TALLOC_FREE(fake);
567  t->status = PEAP_STATUS_SENT_TLV_FAILURE;
568  eap_peap_failure(request, eap_session, tls_session);
569  rcode = RLM_MODULE_HANDLED;
570  goto finish;
571  }
572 
573  /* save the SoH VPs */
574  fr_assert(!t->soh_reply_vps);
575  MEM(fr_pair_list_copy(t, &t->soh_reply_vps, &fake->reply->vps) >= 0);
576  fr_assert(!fake->reply->vps);
577  TALLOC_FREE(fake);
578 
579  if (t->session_resumption_state == PEAP_RESUMPTION_YES) {
580  /* we're good, send success TLV */
581  t->status = PEAP_STATUS_SENT_TLV_SUCCESS;
582  eap_peap_success(request, eap_session, tls_session);
583  rcode = RLM_MODULE_HANDLED;
584  goto finish;
585  }
586 
587  t->status = PEAP_STATUS_PHASE2_INIT;
588  break;
589 
590 
591  /*
592  * If we authenticated the user, then it's OK.
593  */
594  case PEAP_STATUS_SENT_TLV_SUCCESS:
595  if (eap_peap_check_tlv(request, data, data_len)) {
596  RDEBUG2("Success");
597  rcode = RLM_MODULE_OK;
598  goto finish;
599  }
600 
601  /*
602  * Otherwise, the client rejected the session
603  * resumption. If the session is being re-used,
604  * we need to do a full authentication.
605  *
606  * We do this by sending an EAP-Identity request
607  * inside of the PEAP tunnel.
608  */
609  if (t->session_resumption_state == PEAP_RESUMPTION_YES) {
610  RDEBUG2("Client rejected session resumption. Re-starting full authentication");
611 
612  /*
613  * Mark session resumption status.
614  */
615  t->status = PEAP_STATUS_INNER_IDENTITY_REQ_SENT;
616  t->session_resumption_state = PEAP_RESUMPTION_NO;
617 
618  eap_peap_identity(request, eap_session, tls_session);
619  rcode = RLM_MODULE_HANDLED;
620  goto finish;
621  }
622 
623  REDEBUG("Sent a success, but received something weird in return");
624  rcode = RLM_MODULE_REJECT;
625  goto finish;
626 
627  /*
628  * Supplicant ACKs our failure.
629  */
630  case PEAP_STATUS_SENT_TLV_FAILURE:
631  RINDENT();
632  REDEBUG("The users session was previously rejected: returning reject (again.)");
633  RIDEBUG("This means you need to read the PREVIOUS messages in the debug output");
634  RIDEBUG("to find out the reason why the user was rejected");
635  RIDEBUG("Look for \"reject\" or \"fail\". Those earlier messages will tell you");
636  RIDEBUG("what went wrong, and how to fix the problem");
637  REXDENT();
638

640 
641  case PEAP_STATUS_PHASE2_INIT:
642  RDEBUG2("In state machine in phase2 init?");
643  break;
644 
645  case PEAP_STATUS_PHASE2:
646  break;
647 
648  default:
649  REDEBUG("Unhandled state in peap");
650  rcode = RLM_MODULE_REJECT;
651  goto finish;
652  }
653 
654  fake = request_alloc_fake(request, NULL);
655  fr_assert(!fake->request_pairs);
656 
657  switch (t->status) {
658  /*
659  * If we're in PHASE2_INIT, the phase2 method hasn't been
660  * sent an Identity packet yet; do so from the stored
661  * username and this will kick off the phase2 eap method
662  */
663  case PEAP_STATUS_PHASE2_INIT:
664  {
665  size_t len;
666  uint8_t *q;
667 
668  fr_assert(t->username);
669 
670  len = t->username->vp_length + EAP_HEADER_LEN + 1;
671  t->status = PEAP_STATUS_PHASE2;
672 
673  MEM(vp = fr_pair_afrom_da(fake->packet, attr_eap_message));
674  MEM(fr_pair_value_mem_alloc(vp, &q, len, false) == 0);
675  q[0] = FR_EAP_CODE_RESPONSE;
676  q[1] = eap_round->response->id;
677  q[2] = (len >> 8) & 0xff;
678  q[3] = len & 0xff;
679  q[4] = FR_EAP_METHOD_IDENTITY;
680  memcpy(q + EAP_HEADER_LEN + 1,
681  t->username->vp_strvalue, t->username->vp_length);
682  fr_pair_add(&fake->request_pairs, vp);
683  }
684  break;
685 
686  case PEAP_STATUS_PHASE2:
687  fake->request_pairs = eap_peap_inner_to_pairs(request, fake->packet,
688  eap_round, data, data_len);
689  if (!fake->request_pairs) {
690  talloc_free(fake);
691  RDEBUG2("Unable to convert tunneled EAP packet to internal server data structures");
692  rcode = RLM_MODULE_REJECT;
693  goto finish;
694  }
695  break;
696 
697  default:
698  REDEBUG("Invalid state change in PEAP");
699  rcode = RLM_MODULE_REJECT;
700  goto finish;
701  }
702 
703  RDEBUG2("Got tunneled request");
704  log_request_pair_list(L_DBG_LVL_2, request, fake->request_pairs, NULL);
705 
706  /*
707  * Update other items in the request_t data structure.
708  */
709  if (!t->username) {
710  /*
711  * There's no User-Name in the tunneled session,
712  * so we add one here, by pulling it out of the
713  * EAP-Identity packet.
714  */
715  if ((data[0] == FR_EAP_METHOD_IDENTITY) && (data_len > 1)) {
716  MEM(t->username = fr_pair_afrom_da(t, attr_user_name));
717  fr_assert(t->username != NULL);
718  t->username->vp_tainted = true;
719 
720  fr_pair_value_bstrndup(t->username, (char const *)data + 1, data_len - 1, true);
721 
722  RDEBUG2("Got tunneled identity of %pV", &t->username->data);
723  }
724  } /* else there WAS a t->username */
725 
726  setup_fake_request(request, fake, t);
727 
728  /*
729  * Call authentication recursively, which will
730  * do PAP, CHAP, MS-CHAP, etc.
731  */
732  eap_virtual_server(request, eap_session, t->virtual_server);
733 
734  /*
735  * Decide what to do with the reply.
736  */
737  if (!fake->reply->code) {
738  REDEBUG("Unknown RADIUS packet type %d: rejecting tunneled user", fake->reply->code);
739  rcode = RLM_MODULE_REJECT;
740  } else {
741  rcode = process_reply(eap_session, tls_session, request, fake->reply);
742  }
743 
744 finish:
745  talloc_free(fake);
746 
747  RETURN_MODULE_RCODE(rcode);
748 }
