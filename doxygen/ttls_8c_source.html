47 /*
48  * Verify that the diameter packet is valid.
49  */
50 static int diameter_verify(request_t *request, uint8_t const *data, unsigned int data_len)
51 {
52  uint32_t attr;
53  uint32_t length;
54  unsigned int hdr_len;
55  unsigned int remaining = data_len;
56 
57  while (remaining > 0) {
58  hdr_len = 12;
59 
60  if (remaining < hdr_len) {
61  RDEBUG2("Diameter attribute is too small (%u) to contain a Diameter header", remaining);
62  return 0;
63  }
64 
65  memcpy(&attr, data, sizeof(attr));
66  attr = ntohl(attr);
67  memcpy(&length, data + 4, sizeof(length));
68  length = ntohl(length);
69 
70  if ((data[4] & 0x80) != 0) {
71  if (remaining < 16) {
72  RDEBUG2("Diameter attribute is too small to contain a Diameter header with Vendor-Id");
73  return 0;
74  }
75 
76  hdr_len = 16;
77  }
78 
79  /*
80  * Get the length. If it's too big, die.
81  */
82  length &= 0x00ffffff;
83 
84  /*
85  * Too short or too long is bad.
86  */
87  if (length <= (hdr_len - 4)) {
88  RDEBUG2("Tunneled attribute %u is too short (%u < %u) to contain anything useful.", attr,
89  length, hdr_len);
90  return 0;
91  }
92 
93  if (length > remaining) {
94  RDEBUG2("Tunneled attribute %u is longer than room remaining in the packet (%u > %u).", attr,
95  length, remaining);
96  return 0;
97  }
98 
99  /*
100  * Check for broken implementations, which don't
101  * pad the AVP to a 4-octet boundary.
102  */
103  if (remaining == length) break;
104 
105  /*
106  * The length does NOT include the padding, so
107  * we've got to account for it here by rounding up
108  * to the nearest 4-byte boundary.
109  */
110  length += 0x03;
111  length &= ~0x03;
112 
113  /*
114  * If the rest of the diameter packet is larger than
115  * this attribute, continue.
116  *
117  * Otherwise, if the attribute over-flows the end
118  * of the packet, die.
119  */
120  if (remaining < length) {
121  REDEBUG2("Diameter attribute overflows packet!");
122  return 0;
123  }
124 
125  /*
126  * remaining > length, continue.
127  */
128  remaining -= length;
129  data += length;
130  }
131 
132  /*
133  * We got this far. It looks OK.
134  */
135  return 1;
136 }

139 /*
140  * Convert diameter attributes to our fr_pair_t's
141  */
142 static ssize_t eap_ttls_decode_pair(TALLOC_CTX *ctx, fr_cursor_t *cursor, fr_dict_attr_t const *parent,
143  uint8_t const *data, size_t data_len,
144  void *decoder_ctx)
145 {
146  uint8_t const *p = data, *end = p + data_len;
147 
148  fr_pair_t *vp = NULL;
149  SSL *ssl = decoder_ctx;
150  fr_dict_t const *dict_radius;
151  fr_dict_attr_t const *attr_radius;
152 
153  dict_radius = fr_dict_by_protocol_name("radius");
154  fr_assert(dict_radius != NULL);
155  attr_radius = fr_dict_root(dict_radius);
156 
157  while (p < end) {
158  ssize_t ret;
159  uint32_t attr, vendor;
160  uint64_t value_len;
161  uint8_t flags;
162  fr_dict_attr_t const *our_parent = parent;
163 
164  if ((end - p) < 8) {
165  fr_strerror_printf("Malformed diameter VPs. Needed at least 8 bytes, got %zu bytes", end - p);
166  error:
167  fr_cursor_free_list(cursor);
168  return -1;
169  }
170 
171  attr = fr_net_to_uint32(p);
172  p += 4;
173 
174  flags = p[0];
175  p++;
176 
177  value_len = fr_net_to_uint64v(p, 3); /* Yes, that is a 24 bit length field */
178  p += 3;
179 
180  if (value_len < 8) {
181  fr_strerror_printf("Malformed diameter VPs. Needed at least length of 8, got %u",
182  (unsigned int) value_len);
183  goto error;
184  }
185 
186  if ((p + ((value_len + 0x03) & ~0x03)) > end) {
187  fr_strerror_printf("Malformed diameter VPs. Value length %u overflows input",
188  (unsigned int) value_len);
189  goto error;
190  }
191 
192  value_len -= 8; /* -= 8 for AVP code (4), flags (1), AVP length (3) */
193 
194  MEM(vp = fr_pair_alloc(ctx));
195 
196  /*
197  * Do we have a vendor field?
198  */
199  if (flags & FR_DIAMETER_AVP_FLAG_VENDOR) {
200  vendor = fr_net_to_uint32(p);
201  p += 4;
202  value_len -= 4; /* -= 4 for the vendor ID field */
203 
204  our_parent = fr_dict_vendor_da_by_num(attr_vendor_specific, vendor);
205  if (!our_parent) {
206  if (flags & FR_DIAMETER_AVP_FLAG_MANDATORY) {
207  fr_strerror_printf("Mandatory bit set and no vendor %u found", vendor);
208  talloc_free(vp);
209  goto error;
210  }
211 
212  MEM(vp->da = fr_dict_unknown_afrom_fields(vp, attr_vendor_specific, vendor, attr));
213  goto do_value;
214  }
215  } else {
216  our_parent = attr_radius;
217  }
218 
219  /*
220  * Is the attribute known?
221  */
222  vp->da = fr_dict_attr_child_by_num(our_parent, attr);
223  if (!vp->da) {
224  if (flags & FR_DIAMETER_AVP_FLAG_MANDATORY) {
225  fr_strerror_printf("Mandatory bit set and no attribute %u defined for parent %s", attr, parent->name);
226  talloc_free(vp);
227  goto error;
228  }
229  MEM(vp->da = fr_dict_unknown_afrom_fields(vp, parent, 0, attr));
230  }
231 
232 do_value:
233  ret = fr_value_box_from_network(vp, &vp->data, vp->da->type, vp->da, p, value_len, true);
234  if (ret < 0) {
235  /*
236  * Mandatory bit is set, and the attribute
237  * is malformed. Fail.
238  */
239  if (flags & FR_DIAMETER_AVP_FLAG_MANDATORY) {
240  fr_strerror_printf("Mandatory bit is set and attribute is malformed");
241  talloc_free(vp);
242  goto error;
243  }
244 
245  fr_pair_to_unknown(vp);
246  fr_pair_value_memdup(vp, p, value_len, true);
247  }
248 
249  /*
250  * The length does NOT include the padding, so
251  * we've got to account for it here by rounding up
252  * to the nearest 4-byte boundary.
253  */
254  p += (value_len + 0x03) & ~0x03;
255  fr_cursor_append(cursor, vp);
256 
257  if (vp->da->flags.is_unknown) continue;
258 
259  /*
260  * Ensure that the client is using the correct challenge.
261  *
262  * This weirdness is to protect against against replay
263  * attacks, where anyone observing the CHAP exchange could
264  * pose as that user, by simply choosing to use the same
265  * challenge.
266  * By using a challenge based on information from the
267  * current session, we can guarantee that the client is
268  * not *choosing* a challenge. We're a little forgiving in
269  * that we have loose checks on the length, and we do NOT
270  * check the Id (first octet of the response to the
271  * challenge) But if the client gets the challenge correct,
272  * we're not too worried about the Id.
273  */
274  if ((vp->da == attr_chap_challenge) || (vp->da == attr_ms_chap_challenge)) {
275  uint8_t challenge[16];
276  char label[] = "ttls challenge";
277 
278  if ((vp->vp_length < 8) || (vp->vp_length > 16)) {
279  fr_strerror_printf("Tunneled challenge has invalid length");
280  goto error;
281  }
282 
283  if (SSL_export_keying_material(ssl, challenge, sizeof(challenge),
284  label, sizeof(label) - 1, NULL, 0, 0) != 1) {
285  tls_strerror_printf("Failed generating phase2 challenge");
286  goto error;
287  }
288 
289  if (memcmp(challenge, vp->vp_octets, vp->vp_length) != 0) {
290  fr_strerror_printf("Tunneled challenge is incorrect");
291  goto error;
292  }
293  }
294 
295  /*
296  * Diameter pads strings (i.e. User-Password) with trailing zeros.
297  */
298  if (vp->vp_type == FR_TYPE_STRING) fr_pair_value_strdup(vp, vp->vp_strvalue);
299  }
300 
301  /*
302  * We got this far. It looks OK.
303  */
304  return p - data;
305 }

307 /*
308  * Convert fr_pair_t's to diameter attributes, and write them
309  * to an SSL session.
310  *
311  * The ONLY fr_pair_t's which may be passed to this function
312  * are ones which can go inside of a RADIUS (i.e. diameter)
313  * packet. So no server-configuration attributes, or the like.
314  */
315 static int vp2diameter(request_t *request, fr_tls_session_t *tls_session, fr_pair_t *first)
316 {
317  /*
318  * RADIUS packets are no more than 4k in size, so if
319  * we've got more than 4k of data to write, it's very
320  * bad.
321  */
322  uint8_t buffer[4096];
323  uint8_t *p;
324  uint32_t attr;
325  uint32_t length;
326  uint32_t vendor;
327  size_t total;
328  uint64_t attr64;
329  fr_pair_t *vp;
330  fr_cursor_t cursor;
331 
332  p = buffer;
333  total = 0;
334 
335  for (vp = fr_cursor_init(&cursor, &first);
336  vp;
337  vp = fr_cursor_next(&cursor)) {
338  /*
339  * Too much data: die.
340  */
341  if ((total + vp->vp_length + 12) >= sizeof(buffer)) {
342  RDEBUG2("output buffer is full!");
343  return 0;
344  }
345 
346  /*
347  * Hmm... we don't group multiple EAP-Messages
348  * together. Maybe we should...
349  */
350 
351  length = vp->vp_length;
352  vendor = fr_dict_vendor_num_by_da(vp->da);
353  if (vendor != 0) {
354  attr = vp->da->attr & 0xffff;
355  length |= ((uint32_t)1 << 31);
356  } else {
357  attr = vp->da->attr;
358  }
359 
360  /*
361  * Hmm... set the M bit for all attributes?
362  */
363  length |= (1 << 30);
364 
365  attr = ntohl(attr);
366 
367  memcpy(p, &attr, sizeof(attr));
368  p += 4;
369  total += 4;
370 
371  length += 8; /* includes 8 bytes of attr & length */
372 
373  if (vendor != 0) {
374  length += 4; /* include 4 bytes of vendor */
375 
376  length = ntohl(length);
377  memcpy(p, &length, sizeof(length));
378  p += 4;
379  total += 4;
380 
381  vendor = ntohl(vendor);
382  memcpy(p, &vendor, sizeof(vendor));
383  p += 4;
384  total += 4;
385  } else {
386  length = ntohl(length);
387  memcpy(p, &length, sizeof(length));
388  p += 4;
389  total += 4;
390  }
391 
392  switch (vp->vp_type) {
393  case FR_TYPE_DATE:
394  attr = htonl(fr_unix_time_to_sec(vp->vp_date)); /* stored in host order */
395  memcpy(p, &attr, sizeof(attr));
396  length = 4;
397  break;
398 
399  case FR_TYPE_UINT32:
400  attr = htonl(vp->vp_uint32); /* stored in host order */
401  memcpy(p, &attr, sizeof(attr));
402  length = 4;
403  break;
404 
405  case FR_TYPE_UINT64:
406  attr64 = htonll(vp->vp_uint64); /* stored in host order */
407  memcpy(p, &attr64, sizeof(attr64));
408  length = 8;
409  break;
410 
411  case FR_TYPE_IPV4_ADDR:
412  memcpy(p, &vp->vp_ipv4addr, 4); /* network order */
413  length = 4;
414  break;
415 
416  case FR_TYPE_STRING:
417  case FR_TYPE_OCTETS:
418  default:
419  memcpy(p, vp->vp_strvalue, vp->vp_length);
420  length = vp->vp_length;
421  break;
422  }
423 
424  /*
425  * Skip to the end of the data.
426  */
427  p += length;
428  total += length;
429 
430  /*
431  * Align the data to a multiple of 4 bytes.
432  */
433  if ((total & 0x03) != 0) {
434  size_t i;
435 
436  length = 4 - (total & 0x03);
437  for (i = 0; i < length; i++) {
438  *p = '\0';
439  p++;
440  total++;
441  }
442  }
443  } /* loop over the VP's to write. */
444 
445  /*
446  * Write the data in the buffer to the SSL session.
447  */
448  if (total > 0) {
449  (tls_session->record_from_buff)(&tls_session->clean_in, buffer, total);
450 
451  /*
452  * FIXME: Check the return code.
453  */
454  fr_tls_session_send(request, tls_session);
455  }
456 
457  /*
458  * Everything's OK.
459  */
460  return 1;
461 }

463 /*
464  * Use a reply packet to determine what to do.
465  */
466 static rlm_rcode_t CC_HINT(nonnull) process_reply(NDEBUG_UNUSED eap_session_t *eap_session, fr_tls_session_t *tls_session,
467  request_t *request, fr_radius_packet_t *reply)
468 {
469  rlm_rcode_t rcode = RLM_MODULE_REJECT;
470  fr_pair_t *vp, *tunnel_vps = NULL;
471  fr_cursor_t cursor;
472  fr_cursor_t to_tunnel;
473 
474  ttls_tunnel_t *t = tls_session->opaque;
475 
476  fr_assert(eap_session->request == request);
477 
478  /*
479  * If the response packet was Access-Accept, then
480  * we're OK. If not, die horribly.
481  *
482  * FIXME: Take MS-CHAP2-Success attribute, and
483  * tunnel it back to the client, to authenticate
484  * ourselves to the client.
485  *
486  * FIXME: If we have an Access-Challenge, then
487  * the Reply-Message is tunneled back to the client.
488  *
489  * FIXME: If we have an EAP-Message, then that message
490  * must be tunneled back to the client.
491  *
492  * FIXME: If we have an Access-Challenge with a State
493  * attribute, then do we tunnel that to the client, or
494  * keep track of it ourselves?
495  *
496  * FIXME: EAP-Messages can only start with 'identity',
497  * NOT 'eap start', so we should check for that....
498  */
499  switch (reply->code) {
500  case FR_CODE_ACCESS_ACCEPT:
501  {
502  RDEBUG2("Got tunneled Access-Accept");
503 
504  fr_cursor_init(&to_tunnel, &tunnel_vps);
505  rcode = RLM_MODULE_OK;
506 
507  /*
508  * Copy what we need into the TTLS tunnel and leave
509  * the rest to be cleaned up.
510  */
511  for (vp = fr_cursor_init(&cursor, &reply->vps);
512  vp;
513  vp = fr_cursor_next(&cursor)) {
514  if (vp->da == attr_ms_chap2_success) {
515  RDEBUG2("Got MS-CHAP2-Success, tunneling it to the client in a challenge");
516 
517  rcode = RLM_MODULE_HANDLED;
518  t->authenticated = true;
519  fr_cursor_prepend(&to_tunnel, fr_pair_copy(tls_session, vp));
520  } else if (vp->da == attr_eap_channel_binding_message) {
521  rcode = RLM_MODULE_HANDLED;
522  t->authenticated = true;
523  fr_cursor_prepend(&to_tunnel, fr_pair_copy(tls_session, vp));
524  }
525  }
526  }
527  break;
528 
529  case FR_CODE_ACCESS_REJECT:
530  REDEBUG("Got tunneled Access-Reject");
531  rcode = RLM_MODULE_REJECT;
532  break;
533 
534  /*
535  * Handle Access-Challenge, but only if we
536  * send tunneled reply data. This is because
537  * an Access-Challenge means that we MUST tunnel
538  * a Reply-Message to the client.
539  */
540  case FR_CODE_ACCESS_CHALLENGE:
541  RDEBUG2("Got tunneled Access-Challenge");
542 
543  fr_cursor_init(&to_tunnel, &tunnel_vps);
544 
545  /*
546  * Copy what we need into the TTLS tunnel and leave
547  * the rest to be cleaned up.
548  */
549  for (vp = fr_cursor_init(&cursor, &reply->vps);
550  vp;
551  vp = fr_cursor_next(&cursor)) {
552  if ((vp->da == attr_eap_message) || (vp->da == attr_reply_message)) {
553  fr_cursor_prepend(&to_tunnel, fr_pair_copy(tls_session, vp));
554  } else if (vp->da == attr_eap_channel_binding_message) {
555  fr_cursor_prepend(&to_tunnel, fr_pair_copy(tls_session, vp));
556  }
557  }
558  rcode = RLM_MODULE_HANDLED;
559  break;
560 
561  default:
562  REDEBUG("Unknown RADIUS packet type %d: rejecting tunneled user", reply->code);
563  rcode = RLM_MODULE_INVALID;
564  break;
565  }
566 
567 
568  /*
569  * Pack any tunneled VPs and send them back
570  * to the supplicant.
571  */
572  if (tunnel_vps) {
573  RDEBUG2("Sending tunneled reply attributes");
574  log_request_pair_list(L_DBG_LVL_2, request, tunnel_vps, NULL);
575 
576  vp2diameter(request, tls_session, tunnel_vps);
577  fr_pair_list_free(&tunnel_vps);
578  }
579 
580  return rcode;
581 }

584 /*
585  * Process the "diameter" contents of the tunneled data.
586  */
587 FR_CODE eap_ttls_process(request_t *request, eap_session_t *eap_session, fr_tls_session_t *tls_session)
588 {
589  FR_CODE code = FR_CODE_ACCESS_REJECT;
590  rlm_rcode_t rcode;
591  fr_pair_t *vp = NULL;
592  fr_cursor_t cursor;
593  ttls_tunnel_t *t;
594  uint8_t const *data;
595  size_t data_len;
596  chbind_packet_t *chbind;
597  fr_pair_t *username;
598 
599  /*
600  * Just look at the buffer directly, without doing
601  * record_to_buff.
602  */
603  data_len = tls_session->clean_out.used;
604  tls_session->clean_out.used = 0;
605  data = tls_session->clean_out.data;
606 
607  t = (ttls_tunnel_t *) tls_session->opaque;
608 
609  /*
610  * If there's no data, maybe this is an ACK to an
611  * MS-CHAP2-Success.
612  */
613  if (data_len == 0) {
614  if (t->authenticated) {
615  RDEBUG2("Got ACK, and the user was already authenticated");
616  code = FR_CODE_ACCESS_ACCEPT;
617  goto finish;
618  } /* else no session, no data, die. */
619 
620  /*
621  * FIXME: Call SSL_get_error() to see what went
622  * wrong.
623  */
624  RDEBUG2("SSL_read Error");
625  code = FR_CODE_ACCESS_REJECT;
626  goto finish;
627  }
628 
629  if (!diameter_verify(request, data, data_len)) {
630  code = FR_CODE_ACCESS_REJECT;
631  goto finish;
632  }
633 
634  /*
635  * Add the tunneled attributes to the request request.
636  */
637  fr_cursor_init(&cursor, &request->request_pairs);
638  if (eap_ttls_decode_pair(request->packet, &cursor, fr_dict_root(fr_dict_internal()),
639  data, data_len, tls_session->ssl) < 0) {
640  RPEDEBUG("Decoding TTLS TLVs failed");
641  code = FR_CODE_ACCESS_REJECT;
642  goto finish;
643  }
644 
645  /*
646  * Update other items in the request_t data structure.
647  */
648 
649  /*
650  * No User-Name, try to create one from stored data.
651  */
652  username = fr_pair_find_by_da(&request->request_pairs, attr_user_name);
653  if (!username) {
654  /*
655  * No User-Name in the stored data, look for
656  * an EAP-Identity, and pull it out of there.
657  */
658  if (!t->username) {
659  vp = fr_pair_find_by_da(&request->request_pairs, attr_eap_message);
660  if (vp &&
661  (vp->vp_length >= EAP_HEADER_LEN + 2) &&
662  (vp->vp_strvalue[0] == FR_EAP_CODE_RESPONSE) &&
663  (vp->vp_strvalue[EAP_HEADER_LEN] == FR_EAP_METHOD_IDENTITY) &&
664  (vp->vp_strvalue[EAP_HEADER_LEN + 1] != 0)) {
665  /*
666  * Create & remember a User-Name
667  */
668  MEM(t->username = fr_pair_afrom_da(t, attr_user_name));
669  t->username->vp_tainted = true;
670 
671  fr_pair_value_bstrndup(t->username,
672  (char const *)vp->vp_octets + 5, vp->vp_length - 5, true);
673 
674  RDEBUG2("Got tunneled identity of %pV", &t->username->data);
675  } else {
676  /*
677  * Don't reject the request outright,
678  * as it's permitted to do EAP without
679  * user-name.
680  */
681  RWDEBUG2("No EAP-Identity found to start EAP conversation");
682  }
683  } /* else there WAS a t->username */
684 
685  if (t->username) {
686  vp = fr_pair_copy(request->packet, t->username);
687  fr_pair_add(&request->request_pairs, vp);
688  }
689  } /* else the request ALREADY had a User-Name */
690 
691  /*
692  * Process channel binding.
693  */
694  chbind = eap_chbind_vp2packet(request, request->request_pairs);
695  if (chbind) {
696  FR_CODE chbind_code;
697  CHBIND_REQ *req = talloc_zero(request, CHBIND_REQ);
698 
699  RDEBUG2("received chbind request");
