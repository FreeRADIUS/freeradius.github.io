49 {
50  DEBUG2("Socket destructor called, closing socket");

52  if (conn->dbc_handle) {

54  SQLFreeHandle(SQL_HANDLE_DBC, conn->dbc_handle);
55  }
56

58 
59  return RLM_SQL_OK;

62 static sql_rcode_t sql_socket_init(rlm_sql_handle_t *handle, rlm_sql_config_t *config,
63  UNUSED fr_time_delta_t timeout)
64 {
65  SQLRETURN row;
66 #if 0
67  uint32_t timeout_ms = FR_TIMEVAL_TO_MS(timeout);
68 #endif
69  rlm_sql_db2_conn_t *conn;
70 
71  MEM(conn = handle->conn = talloc_zero(handle, rlm_sql_db2_conn_t));
72  talloc_set_destructor(conn, _sql_socket_destructor);
73 
74  /* Allocate handles */
75  SQLAllocHandle(SQL_HANDLE_ENV, SQL_NULL_HANDLE, &(conn->env_handle));
76  SQLAllocHandle(SQL_HANDLE_DBC, conn->env_handle, &(conn->dbc_handle));
77 
78  /* Set the connection timeout */
79 #if 0
80  /* Not suported ? */
81  SQLSetConnectAttr(conn->dbc_handle, SQL_ATTR_LOGIN_TIMEOUT, &timeout_ms, SQL_IS_UINTEGER);
82 #endif
83  /*
84  * The db2 API doesn't qualify arguments as const even when they should be.
85  */
86  {
87  SQLCHAR *server, *login, *password;
88 
89  memcpy(&server, &config->sql_server, sizeof(server));
90  memcpy(&login, &config->sql_login, sizeof(login));
91  memcpy(&password, &config->sql_password, sizeof(password));
92 
93  /*
94  * We probably want to use SQLDriverConnect, which connects
95  * to a remote server.
96  *
97  * http://www.ibm.com/support/knowledgecenter/SSEPGG_10.5.0/com.ibm.db2.luw.apdv.cli.doc/doc/r0000584.html
98  * http://stackoverflow.com/questions/27167070/connection-string-to-a-remote-db2-db-in-another-server
99  *
100  * And probably synthesis the retarded connection string ourselves,
101  * probably via config file expansions:
102  *
103  * Driver={IBM DB2 ODBC Driver};Database=testDb;Hostname=remoteHostName.com;UID=username;PWD=mypasswd;PORT=50000
104  */
105  row = SQLConnect(conn->dbc_handle,
106  server, SQL_NTS,
107  login, SQL_NTS,
108  password, SQL_NTS);
109  }
110 
111  if (row != SQL_SUCCESS) {
112  ERROR("could not connect to DB2 server %s", config->sql_server);
113 
114  return RLM_SQL_ERROR;
115  }
116 
117  return RLM_SQL_OK;
118 }

120 static sql_rcode_t sql_query(rlm_sql_handle_t *handle, UNUSED rlm_sql_config_t *config, char const *query)
121 {
122  SQLRETURN row;
123  rlm_sql_db2_conn_t *conn;
124 
125  conn = handle->conn;
126 
127  /* allocate handle for statement */
128  SQLAllocHandle(SQL_HANDLE_STMT, conn->dbc_handle, &(conn->stmt));
129 
130  /* execute query */
131  {
132  SQLCHAR *db2_query;
133  memcpy(&db2_query, &query, sizeof(query));
134 
135  row = SQLExecDirect(conn->stmt, db2_query, SQL_NTS);
136  if(row != SQL_SUCCESS) {
137  /* XXX Check if row means we should return RLM_SQL_RECONNECT */

139  return RLM_SQL_ERROR;
140  }
141  }
142 
143  return RLM_SQL_OK;

161 static sql_rcode_t sql_fields(char const **out[], rlm_sql_handle_t *handle, UNUSED rlm_sql_config_t *config)
162 {
163  rlm_sql_db2_conn_t *conn = handle->conn;
164 
165  SQLSMALLINT fields, len, i;
166 
167  char const **names;
168  char field[128];
169 
170  SQLNumResultCols(conn->stmt, &fields);
171  if (fields == 0) return RLM_SQL_ERROR;
172 
173  MEM(names = talloc_array(handle, char const *, fields));
174 
175  for (i = 0; i < fields; i++) {
176  char *p;
177 
178  switch (SQLColAttribute(conn->stmt, i, SQL_DESC_BASE_COLUMN_NAME,
179  field, sizeof(field), &len, NULL)) {
180  case SQL_INVALID_HANDLE:
181  case SQL_ERROR:
182  ERROR("Failed retrieving field name at index %i", i);
183  talloc_free(names);
184  return RLM_SQL_ERROR;
185 
186  default:
187  break;
188  }
189 
190  MEM(p = talloc_array(names, char, (size_t)len + 1));
191  strlcpy(p, field, (size_t)len + 1);
192  names[i] = p;
193  }
194  *out = names;
195 
196  return RLM_SQL_OK;
197 }

199 static sql_rcode_t sql_fetch_row(rlm_sql_row_t *out, rlm_sql_handle_t *handle, rlm_sql_config_t *config)
200 {
201  int c, i;
202  SQLINTEGER len, slen;
203  rlm_sql_row_t row;
204  rlm_sql_db2_conn_t *conn;
205 
206  *out = NULL;
207 
208  TALLOC_FREE(handle->row);
209 
210  conn = handle->conn;
211  c = sql_num_fields(handle, config);
212 
213  /* advance cursor */
214  if (SQLFetch(conn->stmt) == SQL_NO_DATA_FOUND) return RLM_SQL_NO_MORE_ROWS;
215 
216  MEM(row = (rlm_sql_row_t)talloc_zero_array(handle, char *, c + 1));
217  for (i = 0; i < c; i++) {
218  /* get column length */
219  SQLColAttribute(conn->stmt, i + 1, SQL_DESC_DISPLAY_SIZE, NULL, 0, NULL, &len);
220 
221  MEM(row[i] = talloc_array(row, char, len + 1));
222 
223  /* get the actual column */
224  SQLGetData(conn->stmt, i + 1, SQL_C_CHAR, row[i], len + 1, &slen);
225  if (slen == SQL_NULL_DATA) row[i][0] = '\0';
226  }
227 
228  *out = handle->row = row;
229 
230  return RLM_SQL_OK;
231 }

244 /** Retrieves any errors associated with the connection handle
245  *
246  * @note Caller will free any memory allocated in ctx.
247  *
248  * @param ctx to allocate temporary error buffers in.
249  * @param out Array of sql_log_entrys to fill.
250  * @param outlen Length of out array.

252  * @param config rlm_sql config.
253  * @return number of errors written to the #sql_log_entry_t array.
254  */
255 static size_t sql_error(TALLOC_CTX *ctx, sql_log_entry_t out[], NDEBUG_UNUSED size_t outlen,
256  rlm_sql_handle_t *handle, UNUSED rlm_sql_config_t *config)
257 {
258  char state[6];
259  char errbuff[1024];
260  SQLINTEGER err;
261  SQLSMALLINT rl;
262  rlm_sql_db2_conn_t *conn = handle->conn;
263 
264  fr_assert(conn);
265  fr_assert(outlen > 0);
266 
267  errbuff[0] = '\0';
268  SQLGetDiagRec(SQL_HANDLE_STMT, conn->stmt, 1, (SQLCHAR *) state, &err,
269  (SQLCHAR *) errbuff, sizeof(errbuff), &rl);
270  if (errbuff[0] == '\0') return 0;
271 
272  out[0].type = L_ERR;
273  out[0].msg = talloc_typed_asprintf(ctx, "%s: %s", state, errbuff);
274 
275  return 1;
276 }
