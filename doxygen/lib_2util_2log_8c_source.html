52 /** Canonicalize error strings, removing tabs, and generate spaces for error marker
53  *
54  * @note talloc_free must be called on the buffer returned in spaces and text
55  *
56  * Used to produce error messages such as this:
57  @verbatim
58  I'm a string with a parser # error
59  ^ Unexpected character in string
60  @endverbatim
61  *
62  * With code resembling this:
63  @code{.c}
64  ERROR("%s", parsed_str);
65  ERROR("%s^ %s", space, text);
66  @endcode
67  *
68  * @todo merge with above function (log_request_marker)
69  *
70  * @param sp Where to write a dynamically allocated buffer of spaces used to indent the error text.
71  * @param text Where to write the canonicalized version of fmt (the error text).
72  * @param ctx to allocate the spaces and text buffers in.
73  * @param slen of error marker. Expects negative integer value, as returned by parse functions.
74  * @param fmt to canonicalize.
75  */
76 void fr_canonicalize_error(TALLOC_CTX *ctx, char **sp, char **text, ssize_t slen, char const *fmt)
77 {
78  size_t offset, prefix, suffix;
79  char *spaces, *p;
80  char const *start;
81  char *value;
82  size_t inlen;
83 
84  offset = -slen;
85 
86  inlen = strlen(fmt);
87  start = fmt;
88  prefix = suffix = 0;
89 
90  /*
91  * Catch bad callers.
92  */
93  if (offset > inlen) {
94  *sp = NULL;
95  *text = NULL;
96  return;
97  }
98 
99  /*
100  * Too many characters before the inflection point. Skip
101  * leading text until we have only 45 characters before it.
102  */
103  if (offset > 30) {
104  size_t skip = offset - 30;
105 
106  start += skip;
107  inlen -= skip;
108  offset -= skip;
109  prefix = 4;
110  }
111 
112  /*
113  * Too many characters after the inflection point,
114  * truncate it to 30 characters after the inflection
115  * point.
116  */
117  if (inlen > (offset + 30)) {
118  inlen = offset + 30;
119  suffix = 4;
120  }
121 
122  /*
123  * Allocate an array to hold just the text we need.
124  */
125  value = talloc_array(ctx, char, prefix + inlen + 1 + suffix);
126  if (prefix) {
127  memcpy(value, "... ", 4);
128  }
129  memcpy(value + prefix, start, inlen);
130  if (suffix) {
131  memcpy(value + prefix + inlen, "...", 3);
132  value[prefix + inlen + 3] = '\0';
133  }
134  value[prefix + inlen + suffix] = '\0';
135 
136  /*
137  * Smash tabs to spaces for the input string.
138  */
139  for (p = value; *p != '\0'; p++) {
140  if (*p == '\t') *p = ' ';
141  }
142 
143  /*
144  * Allocate the spaces array
145  */
146  spaces = talloc_array(ctx, char, prefix + offset + 1);
147  memset(spaces, ' ', prefix + offset);
148  spaces[prefix + offset] = '\0';
149 
150  *sp = spaces;
151  *text = value;
152 }

242 /** Send a server log message to its destination
243  *
244  * @param[in] log destination.
245  * @param[in] type of log message.
246  * @param[in] file src file the log message was generated in.
247  * @param[in] line number the log message was generated on.
248  * @param[in] fmt with printf style substitution tokens.
249  * @param[in] ap Substitution arguments.
250  */
251 int fr_vlog(fr_log_t const *log, fr_log_type_t type, char const *file, int line, char const *fmt, va_list ap)
252 {
253  int colourise = log->colourise;
254  char *buffer;
255  TALLOC_CTX *pool, *thread_log_pool;
256  int ret = 0;
257  char const *fmt_colour = "";
258  char const *fmt_location = "";
259  char fmt_time[50];
260  char const *fmt_facility = "";
261  char const *fmt_type = "";
262  char *fmt_msg;
263 
264  static char const *spaces = " "; /* 40 */
265 
266  fmt_time[0] = '\0';
267 
268  /*
269  * If we don't want any messages, then
270  * throw them away.
271  */
272  if (log->dst == L_DST_NULL) return 0;
273 
274  thread_log_pool = fr_log_pool_init();
275  pool = talloc_new(thread_log_pool); /* Track our local allocations */
276 
277  /*
278  * Set colourisation
279  */
280  if (colourise) {
281  fmt_colour = fr_table_str_by_value(colours, type, NULL);
282  if (!fmt_colour) colourise = false;
283  }
284 
285  /*
286  * Print src file/line
287  */
288  if (log->line_number) {
289  size_t len;
290  int pad = 0;
291  char *str;
292 
293  str = talloc_asprintf(pool, "%s:%i", file, line);
294  len = talloc_array_length(str) - 1;
295 
296  /*
297  * Only increase the indent
298  */
299  if (len > location_indent) {
300  location_indent = len;
301  } else {
302  pad = location_indent - len;
303  }
304 
305  fmt_location = talloc_asprintf_append_buffer(str, "%.*s : ", pad, spaces);
306  }
307  /*
308  * Determine if we need to add a timestamp to the start of the message
309  */
310  switch (log->timestamp) {
311  case L_TIMESTAMP_OFF:
312  break;
313 
314  /*
315  * If we're not logging to syslog, and the debug level is -xxx
316  * then log timestamps by default.
317  */
318  case L_TIMESTAMP_AUTO:
319  if (log->dst == L_DST_SYSLOG) break;
320  if ((log->dst != L_DST_FILES) && (fr_debug_lvl <= L_DBG_LVL_2)) break;
321  FALL_THROUGH;
322 
323  case L_TIMESTAMP_ON:
324  {
325  time_t timeval;
326  size_t len;
327 
328  timeval = time(NULL);
329 #ifdef HAVE_GMTIME_R
330  if (log->dates_utc) {
331  struct tm utc;
332  gmtime_r(&timeval, &utc);
333  ASCTIME_R(&utc, fmt_time, sizeof(fmt_time));
334  } else
335 #endif
336  {
337  CTIME_R(&timeval, fmt_time, sizeof(fmt_time));
338  }
339 
340  /*
341  * ctime adds '\n'
342  */
343  len = strlen(fmt_time);
344  if ((len > 0) && (fmt_time[len - 1] == '\n')) fmt_time[len - 1] = '\0';
345  }
346  break;
347  }
348 
349  /*
350  * Add ERROR or WARNING prefixes to messages not going to
351  * syslog. It's redundant for syslog because of syslog
352  * facilities.
353  */
354  if (log->dst != L_DST_SYSLOG) {
355  /*
356  * Only print the 'facility' if we're not colourising the log messages
357  * and this isn't syslog.
358  */
359  if (!log->colourise && log->print_level) fmt_facility = fr_table_str_by_value(fr_log_levels, type, ": ");
360 
361  /*
362  * Add an additional prefix to highlight that this is a bad message
363  * the user should pay attention to.
364  */
365  switch (type) {
366  case L_DBG_WARN:
367  case L_DBG_ERR:
368  fmt_type = fr_table_str_by_value(fr_log_levels, type, NULL);
369  break;
370 
371  default:
372  break;
373  }
374  }
375 
376  /*
377  * Sanitize output.
378  *
379  * Most strings should be escaped before they get here.
380  */
381  {
382  char *p, *end;
383 
384  p = fmt_msg = fr_vasprintf(pool, fmt, ap);
385  end = p + talloc_array_length(fmt_msg) - 1;
386 
387  /*
388  * Filter out control chars and non UTF8 chars
389  */
390  for (p = fmt_msg; p < end; p++) {
391  int clen;
392 
393  switch (*p) {
394  case '\r':
395  case '\n':
396  *p = ' ';
397  break;
398 
399  case '\t':
400  continue;
401 
402  default:
403  clen = fr_utf8_char((uint8_t *)p, -1);
404  if (!clen) {
405  *p = '?';
406  continue;
407  }
408  p += (clen - 1);
409  break;
410  }
411  }
412  }
413 
414  switch (log->dst) {
415 
416 #ifdef HAVE_SYSLOG_H
417  case L_DST_SYSLOG:
418  {
419  int syslog_priority;
420 
421  switch (type) {
422  case L_DBG:
423  case L_DBG_INFO:
424  case L_DBG_WARN:
425  case L_DBG_ERR:
426  case L_DBG_ERR_REQ:
427  case L_DBG_WARN_REQ:
428  syslog_priority= LOG_DEBUG;
429  break;
430 
431  case L_INFO:
432  syslog_priority = LOG_INFO;
433  break;
434 
435  case L_WARN:
436  syslog_priority = LOG_WARNING;
437  break;
438 
439  case L_ERR:
440  syslog_priority = LOG_ERR;
441  break;
442 
443  case L_AUTH:
444  syslog_priority = LOG_AUTH | LOG_INFO;
445  break;
446  }
447  syslog(syslog_priority,
448  "%s" /* time */
449  "%s" /* time sep */
450  "%s", /* message */
451  fmt_time,
452  fmt_time[0] ? ": " : "",
453  fmt_msg);
454  }
455  break;
456 #endif
457 
458  case L_DST_FILES:
459  case L_DST_STDOUT:
460  case L_DST_STDERR:
461  {
462  size_t len, wrote;
463 
464  buffer = talloc_asprintf(pool,
465  "%s" /* colourise */
466  "%s" /* location */
467  "%s" /* time */
468  "%s" /* time sep */
469  "%s" /* facility */
470  "%s" /* message type */
471  "%s" /* message */
472  "%s" /* colourise reset */
473  "\n",
474  colourise ? fmt_colour : "",
475  fmt_location,
476  fmt_time,
477  fmt_time[0] ? ": " : "",
478  fmt_facility,
479  fmt_type,
480  fmt_msg,
481  colourise ? VTC_RESET : "");
482 
483  len = talloc_array_length(buffer) - 1;
484  wrote = write(log->fd, buffer, len);
485  if (wrote < len) ret = -1;
486  }
487  break;
488 
489  default:
490  case L_DST_NULL: /* should have been caught above */
491  break;
492  }
493 
494  talloc_free(pool); /* clears all temporary allocations */
495 
496  return ret;
497 }

525 /** Drain any outstanding messages from the fr_strerror buffers
526  *
527  * This function drains any messages from fr_strerror buffer prefixing
528  * the first message with fmt + args.
529  *
530  * If a prefix is specified in rules, this is prepended to all lines
531  * logged. The prefix is useful for adding context, i.e. configuration
532  * file and line number information.
533  *
534  * @param[in] log destination.
535  * @param[in] type of log message.
536  * @param[in] file src file the log message was generated in.
537  * @param[in] line number the log message was generated on.
538  * @param[in] f_rules for printing multiline errors.
539  * @param[in] fmt with printf style substitution tokens.
540  * @param[in] ap Substitution arguments.
541  */
542 int fr_vlog_perror(fr_log_t const *log, fr_log_type_t type, char const *file, int line,
543  fr_log_perror_format_t const *f_rules, char const *fmt, va_list ap)
544 {
545  char const *error;
546  int ret;
547  static fr_log_perror_format_t default_f_rules;
548 
549  TALLOC_CTX *thread_log_pool;
550  fr_sbuff_marker_t prefix_m;
551 
552  fr_sbuff_t sbuff;
553  fr_sbuff_uctx_talloc_t tctx;
554 
555  /*
556  * Non-debug message, or debugging is enabled. Log it.
557  */
558  if (!(((type & L_DBG) == 0) || (fr_debug_lvl > 0))) return 0;
559 
560  if (!f_rules) f_rules = &default_f_rules;
561 
562  thread_log_pool = fr_log_pool_init();
563 
564  /*
565  * Setup the aggregation buffer
566  */
567  fr_sbuff_init_talloc(thread_log_pool, &sbuff, &tctx, 1024, 16384);
568 
569  /*
570  * Add the prefix for the first line
571  */
572  if (f_rules->first_prefix) fr_sbuff_in_strcpy(&sbuff, f_rules->first_prefix);
573 
574  /*
575  * Add the (optional) message, and/or (optional) error
576  * with the error_sep.
577  * i.e. <msg>: <error>
578  */
579  error = fr_strerror_pop();
580  if (error) {
581  if (fmt) {
582  va_list aq;
583 
584  va_copy(aq, ap);
585  fr_sbuff_in_vsprintf(&sbuff, fmt, aq);
586  va_end(aq);
587 
588  fr_sbuff_in_strcpy(&sbuff, ": ");
589  fr_sbuff_in_bstrcpy_buffer(&sbuff, error);
590  error = fr_sbuff_start(&sbuff);
591  }
592  /*
593  * No error, just print the fmt string
594  */
595  } else {
596  va_list aq;
597 
598  if (!fmt) return 0; /* NOOP */
599 
600  va_copy(aq, ap);
601  fr_sbuff_in_vsprintf(&sbuff, fmt, ap);
602  va_end(aq);
603 
604  error = fr_sbuff_start(&sbuff);
605  }
606 
607  /*
608  * Log the first line
609  */
610  ret = fr_log(log, type, file, line, "%s", error);
611  if (ret < 0) {
612  error:
613  talloc_free(sbuff.buff);
614  return ret;
615  }
616 
617  fr_sbuff_set_to_start(&sbuff);
618  if (f_rules->subsq_prefix) {
619  fr_sbuff_in_strcpy(&sbuff, f_rules->subsq_prefix);
620  fr_sbuff_marker(&prefix_m, &sbuff);
621  }
622 
623  /*
624  * Print out additional error lines
625  */
626  while ((error = fr_strerror_pop())) {
627  if (f_rules->subsq_prefix) {
628  fr_sbuff_set(&sbuff, &prefix_m);
629  fr_sbuff_in_bstrcpy_buffer(&sbuff, error);
630  error = fr_sbuff_start(&sbuff);
631  }
632 
633  ret = fr_log(log, type, file, line, "%s", error);
634  if (ret < 0) goto error;
635  }
636 
637  talloc_free(sbuff.buff);
638 
639  return 0;
640 }

669 void fr_log_hex(fr_log_t const *log, fr_log_type_t type, char const *file, int line,
670  uint8_t const *data, size_t data_len, char const *fmt, ...)
671 {
672  size_t i, j, len;
673  char *p;
674  char buffer[(0x10 * 3) + 1];
675  TALLOC_CTX *thread_log_pool = fr_log_pool_init();

678  if (fmt) {
679  va_list ap;

698  }

703 void fr_log_hex_marker(fr_log_t const *log, fr_log_type_t type, char const *file, int line,
704  uint8_t const *data, size_t data_len, ssize_t slen,
705  char const *error, char const *fmt, ...)
706 {
707  size_t i, j, len;
708  char *p;
709  char buffer[(0x10 * 3) + 1];
710  TALLOC_CTX *thread_log_pool = fr_log_pool_init();
711 
712  char *prefix = NULL;
713  static char spaces[3 * 0x10]; /* Bytes per line */
714 
715  if (!*spaces) memset(spaces, ' ', sizeof(spaces) - 1); /* Leave a \0 */
716 
717  if (slen < 0) slen = +(slen);
718 
719  if (fmt) {
720  va_list ap;
721 
722  va_start(ap, fmt);
723  prefix = talloc_asprintf(thread_log_pool, fmt, ap);
724  va_end(ap);
725  }
726

782 /** Initialise file descriptors based on logging destination
783  *
784  * @param log Logger to manipulate.
785  * @param daemonize Whether the server is starting as a daemon.
786  * @return
787  * - 0 on success.
788  * - -1 on failure.
789  */
790 int fr_log_init(fr_log_t *log, bool daemonize)
791 {
792  int devnull;
793 
794  fr_log_rate_limit = daemonize;
795 
796  /*
797  * If we're running in foreground mode, save STDIN /
798  * STDERR as higher FDs, which won't get used by anyone
799  * else. When we fork/exec a program, its STD FDs will
800  * get set to pipes. We later set STDOUT / STDERR to
801  * /dev/null, so that any library trying to write to them
802  * doesn't screw anything up.
803  *
804  * Then, when something goes wrong, restore them so that
805  * any debugger called from the panic action has access
806  * to STDOUT / STDERR.
807  */
808  if (!daemonize) {
809  fr_fault_set_cb(_restore_std);
810 
811  stdout_fd = dup(STDOUT_FILENO);
812  stderr_fd = dup(STDERR_FILENO);
813  }
814 
815  devnull = open("/dev/null", O_RDWR);
816  if (devnull < 0) {
817  fr_strerror_printf("Error opening /dev/null: %s", fr_syserror(errno));
818  return -1;
819  }
820 
821  /*
822  * STDOUT & STDERR go to /dev/null, unless we have "-x",
823  * then STDOUT & STDERR go to the "-l log" destination.
824  *
825  * The complexity here is because "-l log" can go to
826  * STDOUT or STDERR, too.
827  */
828  if (log->dst == L_DST_STDOUT) {
829  setlinebuf(stdout);
830  log->fd = STDOUT_FILENO;
831 
832  /*
833  * If we're debugging, allow STDERR to go to
834  * STDOUT too, for executed programs.
835  *
836  * Allow stdout when running in foreground mode
837  * as it's useful for some profiling tools,
838  * like mutrace.
839  */
840  if (fr_debug_lvl || !daemonize) {
841  dup2(STDOUT_FILENO, STDERR_FILENO);
842  } else {
843  dup2(devnull, STDERR_FILENO);
844  }
845 
846  } else if (log->dst == L_DST_STDERR) {
847  setlinebuf(stderr);
848  log->fd = STDERR_FILENO;
849 
850  /*
851  * If we're debugging, allow STDOUT to go to
852  * STDERR too, for executed programs.
853  *
854  * Allow stdout when running in foreground mode
855  * as it's useful for some profiling tools,
856  * like mutrace.
857  */
858  if (fr_debug_lvl || !daemonize) {
859  dup2(STDERR_FILENO, STDOUT_FILENO);
860  } else {
861  dup2(devnull, STDOUT_FILENO);
862  }
863 
864  } else if (log->dst == L_DST_SYSLOG) {
865  /*
866  * Discard STDOUT and STDERR no matter what the
867  * status of debugging. Syslog isn't a file
868  * descriptor, so we can't use it.
869  */
870  dup2(devnull, STDOUT_FILENO);
871  dup2(devnull, STDERR_FILENO);
872  log->print_level = false;
873 
874  } else if (fr_debug_lvl) {
875  /*
876  * If we're debugging, allow STDOUT and STDERR to
877  * go to the log file.
878  */
879  dup2(log->fd, STDOUT_FILENO);
880  dup2(log->fd, STDERR_FILENO);
881 
882  } else {
883  /*
884  * Not debugging, and the log isn't STDOUT or
885  * STDERR. Ensure that we move both of them to
886  * /dev/null, so that the calling terminal can
887  * exit, and the output from executed programs
888  * doesn't pollute STDOUT / STDERR.
889  */
890  dup2(devnull, STDOUT_FILENO);
891  dup2(devnull, STDERR_FILENO);
892  }
893 
894  close(devnull);
895 
896  fr_fault_set_log_fd(log->fd);
897 
898  return 0;
899 }
