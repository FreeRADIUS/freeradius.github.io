17 /**
18  * $Id: 8e181826539e9102478318708a0a133bd8afe008 $
19  * @file rlm_redis_ippool.c
20  * @brief IP Allocation module with a redis backend.
21  *
22  * @author Arran Cudbard-Bell
23  *
24  * Performs lease management using a Redis backed.
25  *
26  *
27  * Creates three types of objects:
28  * - @verbatim {<pool name>:<pool type>}:pool @endverbatim (zset) contains IP addresses
29  * with priority set by expiry time.
30  * - @verbatim {<pool name>:<pool type>}:ip:<address> @endverbatim (hash) contains four keys
31  * * range - Range identifier, used to lookup attributes associated with a range within a pool.
32  * * device - Lease owner identifier for the device which last bound this address.
33  * * gateway - Gateway of device which last bound this address.
34  * * counter - How many times this IP address has been bound.
35  * - @verbatim {<pool name>:<pool type>}:device:<client id> @endverbatim (string) contains last
36  * IP address bound by this client.
37  *
38  * @copyright 2015 Arran Cudbard-Bell (a.cudbardb@freeradius.org)
39  * @copyright 2015 The FreeRADIUS server project
40  */

358 /** Check the requisite number of slaves replicated the lease info
359  *
360  * @param request The current request.
361  * @param wait_num Number of slaves required.
362  * @param reply we got from the server.
363  * @return
364  * - 0 if enough slaves replicated the data.
365  * - -1 if too few slaves replicated the data, or another error.
366  */
367 static inline int ippool_wait_check(request_t *request, uint32_t wait_num, redisReply *reply)
368 {
369  if (!wait_num) return 0;
370 
371  if (reply->type != REDIS_REPLY_INTEGER) {
372  REDEBUG("WAIT result is wrong type, expected integer got %s",
373  fr_table_str_by_value(redis_reply_types, reply->type, "<UNKNOWN>"));
374  return -1;
375  }
376  if (reply->integer < wait_num) {
377  REDEBUG("Too few slaves acknowledged allocation, needed %i, got %lli",
378  wait_num, reply->integer);
379  return -1;
380  }
381  return 0;
382 }

384 static void ippool_action_print(request_t *request, ippool_action_t action,
385  fr_log_lvl_t lvl,
386  uint8_t const *key_prefix, size_t key_prefix_len,
387  char const *ip_str,
388  uint8_t const *owner, size_t owner_len,
389  uint8_t const *gateway_id, size_t gateway_id_len,
390  uint32_t expires)
391 {
392  char *key_prefix_str, *device_str = NULL, *gateway_str = NULL;
393 
394  key_prefix_str = fr_asprint(request, (char const *)key_prefix, key_prefix_len, '"');
395  if (gateway_id) gateway_str = fr_asprint(request, (char const *)gateway_id, gateway_id_len, '"');
396  if (owner) device_str = fr_asprint(request, (char const *)owner, owner_len, '"');
397 
398  switch (action) {
399  case POOL_ACTION_ALLOCATE:
400  RDEBUGX(lvl, "Allocating lease from pool \"%s\"%s%s%s%s%s%s, expires in %us",
401  key_prefix_str,
402  device_str ? ", to \"" : "", device_str ? device_str : "",
403  device_str ? "\"" : "",
404  gateway_str ? ", on \"" : "", gateway_str ? gateway_str : "",
405  gateway_str ? "\"" : "",
406  expires);
407  break;
408 
409  case POOL_ACTION_UPDATE:
410  RDEBUGX(lvl, "Updating %s in pool \"%s\"%s%s%s%s%s%s, expires in %us",
411  ip_str, key_prefix_str,
412  device_str ? ", device \"" : "", device_str ? device_str : "",
413  device_str ? "\"" : "",
414  gateway_str ? ", gateway \"" : "", gateway_str ? gateway_str : "",
415  gateway_str ? "\"" : "",
416  expires);
417  break;
418 
419  case POOL_ACTION_RELEASE:
420  RDEBUGX(lvl, "Releasing %s%s%s%s to pool \"%s\"",
421  ip_str,
422  device_str ? " leased by \"" : "", device_str ? device_str : "",
423  device_str ? "\"" : "",
424  key_prefix_str);
425  break;
426 
427  default:
428  break;
429  }
430 
431  /*
432  * Ordering is important, needs to be LIFO
433  * for proper talloc pool re-use.
434  */
435  talloc_free(key_prefix_str);
436  talloc_free(device_str);
437  talloc_free(gateway_str);
438 }

440 /** Execute a script against Redis cluster
441  *
442  * Handles uploading the script to the server if required.
443  *
444  * @note All replies will be freed on error.
445  *
446  * @param[out] out Where to write Redis reply object resulting from the command.
447  * @param[in] request The current request.
448  * @param[in] cluster configuration.
449  * @param[in] key to use to determine the cluster node.
450  * @param[in] key_len length of the key.
451  * @param[in] wait_num If > 0 wait until this many slaves have replicated the data
452  * from the last command.
453  * @param[in] wait_timeout How long to wait for slaves.
454  * @param[in] digest of script.
455  * @param[in] script to upload.
456  * @param[in] cmd EVALSHA command to execute.
457  * @param[in] ... Arguments for the eval command.
458  * @return status of the command.
459  */
460 static fr_redis_rcode_t ippool_script(redisReply **out, request_t *request, fr_redis_cluster_t *cluster,
461  uint8_t const *key, size_t key_len,
462  uint32_t wait_num, fr_time_delta_t wait_timeout,
463  char const digest[], char const *script,
464  char const *cmd, ...)
465 {
466  fr_redis_conn_t *conn;
467  redisReply *replies[5]; /* Must be equal to the maximum number of pipelined commands */
468  size_t reply_cnt = 0, i;
469 
470  fr_redis_cluster_state_t state;
471  fr_redis_rcode_t s_ret, status;
472  unsigned int pipelined = 0;
473 
474  va_list ap;
475 
476  *out = NULL;
477 
478 #ifndef NDEBUG
479  memset(replies, 0, sizeof(replies));
480 #endif
481 
482  va_start(ap, cmd);
483 
484  for (s_ret = fr_redis_cluster_state_init(&state, &conn, cluster, request, key, key_len, false);
485  s_ret == REDIS_RCODE_TRY_AGAIN; /* Continue */
486  s_ret = fr_redis_cluster_state_next(&state, &conn, cluster, request, status, &replies[0])) {
487  va_list copy;
488 
489  RDEBUG3("Calling script 0x%s", digest);
490  va_copy(copy, ap); /* copy or segv */
491  redisvAppendCommand(conn->handle, cmd, copy);
492  va_end(copy);
493  pipelined = 1;
494  if (wait_num) {
495  redisAppendCommand(conn->handle, "WAIT %i %i", wait_num, fr_time_delta_to_msec(wait_timeout));
496  pipelined++;
497  }
498  reply_cnt = fr_redis_pipeline_result(&pipelined, &status,
499  replies, NUM_ELEMENTS(replies),
500  conn);
501  if (status != REDIS_RCODE_NO_SCRIPT) continue;
502 
503  /*
504  * Clear out the existing reply
505  */
506  fr_redis_pipeline_free(replies, reply_cnt);
507 
508  /*
509  * Last command failed with NOSCRIPT, this means
510  * we have to send the Lua script up to the node
511  * so it can be cached.
512  */
513  RDEBUG3("Loading script 0x%s", digest);
514  redisAppendCommand(conn->handle, "MULTI");
515  redisAppendCommand(conn->handle, "SCRIPT LOAD %s", script);
516  va_copy(copy, ap); /* copy or segv */
517  redisvAppendCommand(conn->handle, cmd, copy);
518  va_end(copy);
519  redisAppendCommand(conn->handle, "EXEC");
520  pipelined = 4;
521  if (wait_num) {
522  redisAppendCommand(conn->handle, "WAIT %i %i", wait_num, wait_timeout);
523  pipelined++;
524  }
525 
526  reply_cnt = fr_redis_pipeline_result(&pipelined, &status,
527  replies, NUM_ELEMENTS(replies),
528  conn);
529  if (status == REDIS_RCODE_SUCCESS) {
530  if (RDEBUG_ENABLED3) for (i = 0; i < reply_cnt; i++) {
531  fr_redis_reply_print(L_DBG_LVL_3, replies[i], request, i);
532  }
533 
534  if (replies[3]->type != REDIS_REPLY_ARRAY) {
535  REDEBUG("Bad response to EXEC, expected array got %s",
536  fr_table_str_by_value(redis_reply_types, replies[3]->type, "<UNKNOWN>"));
537  error:
538  fr_redis_pipeline_free(replies, reply_cnt);
539  status = REDIS_RCODE_ERROR;
540  goto finish;
541  }
542  if (replies[3]->elements != 2) {
543  REDEBUG("Bad response to EXEC, expected 2 result elements, got %zu",
544  replies[3]->elements);
545  goto error;
546  }
547  if (replies[3]->element[0]->type != REDIS_REPLY_STRING) {
548  REDEBUG("Bad response to SCRIPT LOAD, expected string got %s",
549  fr_table_str_by_value(redis_reply_types, replies[3]->element[0]->type, "<UNKNOWN>"));
550  goto error;
551  }
552  if (strcmp(replies[3]->element[0]->str, digest) != 0) {
553  RWDEBUG("Incorrect SHA1 from SCRIPT LOAD, expected %s, got %s",
554  digest, replies[3]->element[0]->str);
555  goto error;
556  }
557  }
558  }
559  if (s_ret != REDIS_RCODE_SUCCESS) goto error;
560 
561  switch (reply_cnt) {
562  case 2: /* EVALSHA with wait */
563  if (ippool_wait_check(request, wait_num, replies[1]) < 0) goto error;
564  fr_redis_reply_free(&replies[1]); /* Free the wait response */
565  break;
566 
567  case 1: /* EVALSHA */
568  *out = replies[0];
569  break;
570 
571  case 5: /* LOADSCRIPT + EVALSHA + WAIT */
572  if (ippool_wait_check(request, wait_num, replies[4]) < 0) goto error;
573  fr_redis_reply_free(&replies[4]); /* Free the wait response */
574  FALL_THROUGH;
575 
576  case 4: /* LOADSCRIPT + EVALSHA */
577  fr_redis_reply_free(&replies[2]); /* Free the queued cmd response*/
578  fr_redis_reply_free(&replies[1]); /* Free the queued script load response */
579  fr_redis_reply_free(&replies[0]); /* Free the queued multi response */
580  *out = replies[3]->element[1];
581  replies[3]->element[1] = NULL; /* Prevent double free */
582  fr_redis_reply_free(&replies[3]); /* This works because hiredis checks for NULL elements */
583  break;
584 
585  case 0:
586  break;
587  }
588 
589 finish:
590  va_end(ap);
591  return s_ret;
