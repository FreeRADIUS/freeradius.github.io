167 int8_t fr_dhcpv4_attr_cmp(void const *a, void const *b)
168 {
169  fr_pair_t const *my_a = a, *my_b = b;
170  fr_dict_attr_t const *a_82, *b_82;
171 
172  VP_VERIFY(my_a);
173  VP_VERIFY(my_b);
174 
175  /*
176  * We can only use attribute numbers if we know they're
177  * not nested attributes.
178  *
179  * @fixme We should be able to use my_a->da->parent->flags.is_root,
180  * but the DHCP attributes are hacked into the server under a vendor
181  * dictionary, so we can't.
182  */
183 
184  /*
185  * DHCP-Message-Type is first, for simplicity.
186  */
187  if (((my_a->da->parent->type != FR_TYPE_TLV) && (my_a->da == attr_dhcp_message_type)) &&
188  ((my_b->da->parent->type == FR_TYPE_TLV) || (my_b->da != attr_dhcp_message_type))) return -1;
189  if (((my_a->da->parent->type == FR_TYPE_TLV) || (my_a->da != attr_dhcp_message_type)) &&
190  ((my_b->da->parent->type != FR_TYPE_TLV) && (my_b->da == attr_dhcp_message_type))) return +1;
191 
192  /*
193  * Relay-Agent is last.
194  *
195  * Check if either of the options are descended from option 82.
196  */
197  a_82 = fr_dict_attr_common_parent(dhcp_option_82, my_a->da, true);
198  b_82 = fr_dict_attr_common_parent(dhcp_option_82, my_b->da, true);
199  if (a_82 && !b_82) return +1;
200  if (!a_82 && !b_82) return -1;
201 
202  return fr_pair_cmp_by_parent_num(my_a, my_b);
203 }

205 /** Check received DHCP request is valid and build fr_radius_packet_t structure if it is
206  *
207  * @param data pointer to received packet.
208  * @param data_len length of received data, and then length of the actual DHCP data.
209  * @param[out] message_type where the message type will be stored (if used)
210  * @param[out] xid where the xid will be stored (if used)
211  *
212  * @return
213  * - true if the packet is well-formed
214  * - false if it's a bad packet
215  */
216 bool fr_dhcpv4_ok(uint8_t const *data, ssize_t data_len, uint8_t *message_type, uint32_t *xid)
217 {
218  uint32_t magic;
219  uint8_t const *code;
220  size_t hlen;
221 
222  if (data_len < MIN_PACKET_SIZE) {
223  fr_strerror_printf("DHCP packet is too small (%zu < %d)", data_len, MIN_PACKET_SIZE);
224  return false;
225  }
226 
227  if (data_len > MAX_PACKET_SIZE) {
228  fr_strerror_printf("DHCP packet is too large (%zx > %d)", data_len, MAX_PACKET_SIZE);
229  return false;
230  }
231 
232  if (data[1] > 1) {
233  fr_strerror_printf("DHCP can only process ethernet requests, not type %02x", data[1]);
234  return false;
235  }
236 
237  hlen = data[2];
238  if ((hlen != 0) && (hlen != 6)) {
239  fr_strerror_printf("Ethernet HW length incorrect. Expected 6 got %zu", hlen);
240  return false;
241  }
242 
243  memcpy(&magic, data + 236, 4);
244  magic = ntohl(magic);
245  if (magic != DHCP_OPTION_MAGIC_NUMBER) {
246  fr_strerror_printf("BOOTP not supported");
247  return false;
248  }
249 
250  code = fr_dhcpv4_packet_get_option((dhcp_packet_t const *) data, data_len, attr_dhcp_message_type);
251  if (!code || (code[1] == 0)) {
252  fr_strerror_printf("No message-type option was found in the packet");
253  return false;
254  }
255 
256  if ((code[2] == 0) || (code[2] >= DHCP_MAX_MESSAGE_TYPE)) {
257  fr_strerror_printf("Unknown value %d for message-type option", code[2]);
258  return false;
259  }
260 
261  /*
262  * @todo - data_len MAY be larger than the data in the
263  * packet. In which case, we should update data_len with
264  * the true size of the packet.
265  */
266 
267  if (message_type) *message_type = code[2];
268 
269  if (xid) {
270  memcpy(&magic, data + 4, 4);
271  *xid = ntohl(magic);
272  }
273 
274  return true;
275 }

480  /*
481  * Pre-sort attributes into contiguous blocks so that fr_dhcpv4_encode_option
482  * operates correctly. This changes the order of the list, but never mind...
483  */
484  fr_pair_list_sort(&vps, fr_dhcpv4_attr_cmp);
485  fr_cursor_talloc_iter_init(&cursor, &vps, fr_proto_next_encodable, dict_dhcpv4, fr_pair_t);
486 
487  /*
488  * Each call to fr_dhcpv4_encode_option will encode one complete DHCP option,
489  * and sub options.
490  */
491  while ((vp = fr_cursor_current(&cursor))) {
492  /*
493  * The encoder skips message type, and returns
494  * "len==0" for it. We want to allow that, BUT
495  * stop when the encoder returns "len==0" for
496  * other attributes. So we need to skip it
497  * manually, too.
498  */
499  if (vp->da == attr_dhcp_message_type) {
500  (void) fr_cursor_next(&cursor);
501  continue;
502  }
503 
504  len = fr_dhcpv4_encode_option(&work_dbuff,
505  &cursor, &(fr_dhcpv4_ctx_t){ .root = fr_dict_root(dict_dhcpv4) });
506  if (len <= 0) break;
507  };
508 
509  FR_DBUFF_IN_BYTES_RETURN(&work_dbuff, FR_DHCP_END_OF_OPTIONS, 0x00);
510 
511  /*
512  * FIXME: if (fr_dbuff_used(&work_dbuff) > mms),
513  * then we put the extra options into the "sname" and "file"
514  * fields, AND set the "end option option" in the "options"
515  * field. We also set the "overload option",
516  * and put options into the "file" field, followed by
517  * the "sname" field. Where each option is completely
518  * enclosed in the "file" and/or "sname" field, AND
519  * followed by the "end of option", and MUST be followed
520  * by padding option.
521  *
522  * Yuck. That sucks...
523  */
524  if (fr_dbuff_used(&work_dbuff) < DEFAULT_PACKET_SIZE) {
525  FR_DBUFF_MEMSET_RETURN(&work_dbuff, 0, DEFAULT_PACKET_SIZE - fr_dbuff_used(&work_dbuff));
526  }
527 
528  return fr_dbuff_set(dbuff, fr_dbuff_used(&work_dbuff));
529 }
