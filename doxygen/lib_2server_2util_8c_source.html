41 /*
42  * The signal() function in Solaris 2.5.1 sets SA_NODEFER in
43  * sa_flags, which causes grief if signal() is called in the
44  * handler before the cause of the signal has been cleared.
45  * (Infinite recursion).
46  *
47  * The same problem appears on HPUX, so we avoid it, if we can.
48  *
49  * Using sigaction() to reset the signal handler fixes the problem,
50  * so where available, we prefer that solution.
51  */
52 
53 void (*reset_signal(int signo, void (*func)(int)))(int)
54 {
55 #ifdef HAVE_SIGACTION
56  struct sigaction act, oact;
57 
58  memset(&act, 0, sizeof(act));
59  act.sa_handler = func;
60  sigemptyset(&act.sa_mask);
61  act.sa_flags = 0;
62 #ifdef SA_INTERRUPT /* SunOS */
63  act.sa_flags |= SA_INTERRUPT;
64 #endif
65  if (sigaction(signo, &act, &oact) < 0)
66  return SIG_ERR;
67  return oact.sa_handler;
68 #else
69 
70  /*
71  * re-set by calling the 'signal' function, which
72  * may cause infinite recursion and core dumps due to
73  * stack growth.
74  *
75  * However, the system is too dumb to implement sigaction(),
76  * so we don't have a choice.
77  */
78  signal(signo, func);
79 
80  return NULL;
81 #endif
82 }

84 /** Ensures that a filename cannot walk up the directory structure
85  *
86  * Also sanitizes control chars.
87  *
88  * @param request Current request (may be NULL).
89  * @param out Output buffer.
90  * @param outlen Size of the output buffer.
91  * @param in string to escape.
92  * @param arg Context arguments (unused, should be NULL).
93  */
94 size_t rad_filename_make_safe(UNUSED request_t *request, char *out, size_t outlen, char const *in, UNUSED void *arg)
95 {
96  char const *q = in;
97  char *p = out;
98  size_t left = outlen;
99 
100  while (*q) {
101  if (*q != '/') {
102  if (left < 2) break;
103 
104  /*
105  * Smash control characters and spaces to
106  * something simpler.
107  */
108  if (*q < ' ') {
109  *(p++) = '_';
110  q++;
111  continue;
112  }
113 
114  *(p++) = *(q++);
115  left--;
116  continue;
117  }
118 
119  /*
120  * For now, allow slashes in the expanded
121  * filename. This allows the admin to set
122  * attributes which create sub-directories.
123  * Unfortunately, it also allows users to send
124  * attributes which *may* end up creating
125  * sub-directories.
126  */
127  if (left < 2) break;
128  *(p++) = *(q++);
129 
130  /*
131  * Get rid of ////../.././///.///..//
132  */
133  redo:
134  /*
135  * Get rid of ////
136  */
137  if (*q == '/') {
138  q++;
139  goto redo;
140  }
141 
142  /*
143  * Get rid of /./././
144  */
145  if ((q[0] == '.') &&
146  (q[1] == '/')) {
147  q += 2;
148  goto redo;
149  }
150 
151  /*
152  * Get rid of /../../../
153  */
154  if ((q[0] == '.') && (q[1] == '.') &&
155  (q[2] == '/')) {
156  q += 3;
157  goto redo;
158  }
159  }
160  *p = '\0';
161 
162  return (p - out);
163 }

165 /** Escapes the raw string such that it should be safe to use as part of a file path
166  *
167  * This function is designed to produce a string that's still readable but portable
168  * across the majority of file systems.
169  *
170  * For security reasons it cannot remove characters from the name, and must not allow
171  * collisions to occur between different strings.
172  *
173  * With that in mind '-' has been chosen as the escape character, and will be double
174  * escaped '-' -> '--' to avoid collisions.
175  *
176  * Escaping should be reversible if the original string needs to be extracted.
177  *
178  * @note function takes additional arguments so that it may be used as an xlat escape
179  * function but it's fine to call it directly.
180  *
181  * @note OSX/Unix/NTFS/VFAT have a max filename size of 255 bytes.
182  *
183  * @param request Current request (may be NULL).
184  * @param out Output buffer.
185  * @param outlen Size of the output buffer.
186  * @param in string to escape.
187  * @param arg Context arguments (unused, should be NULL).
188  */
189 size_t rad_filename_escape(UNUSED request_t *request, char *out, size_t outlen, char const *in, UNUSED void *arg)
190 {
191  size_t freespace = outlen;
192 
193  while (*in != '\0') {
194  size_t utf8_len;
195 
196  /*
197  * Encode multibyte UTF8 chars
198  */
199  utf8_len = fr_utf8_char((uint8_t const *) in, -1);
200  if (utf8_len > 1) {
201  if (freespace <= (utf8_len * 3)) break;
202 
203  switch (utf8_len) {
204  case 2:
205  snprintf(out, freespace, "-%x-%x", in[0], in[1]);
206  break;
207 
208  case 3:
209  snprintf(out, freespace, "-%x-%x-%x", in[0], in[1], in[2]);
210  break;
211 
212  case 4:
213  snprintf(out, freespace, "-%x-%x-%x-%x", in[0], in[1], in[2], in[3]);
214  break;
215  }
216 
217  freespace -= (utf8_len * 3);
218  out += (utf8_len * 3);
219  in += utf8_len;
220 
221  continue;
222  }
223 
224  /*
225  * Safe chars
226  */
227  if (((*in >= 'A') && (*in <= 'Z')) ||
228  ((*in >= 'a') && (*in <= 'z')) ||
229  ((*in >= '0') && (*in <= '9')) ||
230  (*in == '_')) {
231  if (freespace <= 1) break;
232 
233  *out++ = *in++;
234  freespace--;
235  continue;
236  }
237  if (freespace <= 2) break;
238 
239  /*
240  * Double escape '-' (like \\)
241  */
242  if (*in == '-') {
243  *out++ = '-';
244  *out++ = '-';
245 
246  freespace -= 2;
247  in++;
248  continue;
249  }
250 
251  /*
252  * Unsafe chars
253  */
254  *out++ = '-';
255  in++;
256  fr_bin2hex(&FR_SBUFF_OUT(out, freespace), &FR_DBUFF_TMP((uint8_t const *)in, 1), SIZE_MAX);
257  out += 2;
258  freespace -= 3;
259  }
260  *out = '\0';
261 
262  return outlen - freespace;
263 }

265 /** Converts data stored in a file name back to its original form
266  *
267  * @param out Where to write the unescaped string (may be the same as in).
268  * @param outlen Length of the output buffer.
269  * @param in Input filename.
270  * @param inlen Length of input.
271  * @return
272  * - Number of bytes written to output buffer
273  * - offset where parse error occurred on failure.
274  */
275 ssize_t rad_filename_unescape(char *out, size_t outlen, char const *in, size_t inlen)
276 {
277  char const *p, *end = in + inlen;
278  size_t freespace = outlen;
279 
280  for (p = in; p < end; p++) {
281  if (freespace <= 1) break;
282 
283  if (((*p >= 'A') && (*p <= 'Z')) ||
284  ((*p >= 'a') && (*p <= 'z')) ||
285  ((*p >= '0') && (*p <= '9')) ||
286  (*p == '_')) {
287  *out++ = *p;
288  freespace--;
289  continue;
290  }
291 
292  if (p[0] == '-') {
293  /*
294  * End of input, '-' needs at least one extra char after
295  * it to be valid.
296  */
297  if ((end - p) < 2) return in - p;
298  if (p[1] == '-') {
299  p++;
300  *out++ = '-';
301  freespace--;
302  continue;
303  }
304 
305  /*
306  * End of input, '-' must be followed by <hex><hex>
307  * but there aren't enough chars left
308  */
309  if ((end - p) < 3) return in - p;
310 
311  /*
312  * If hex2bin returns 0 the next two chars weren't hexits.
313  */
314  if (fr_hex2bin(NULL,
315  &FR_DBUFF_TMP((uint8_t *) out, 1),
316  &FR_SBUFF_IN(in, 1), false) == 0) return in - (p + 1);
317  in += 2;
318  out++;
319  freespace--;
320  }
321 
322  return in - p; /* offset we found the bad char at */
323  }
324  *out = '\0';
325 
326  return outlen - freespace; /* how many bytes were written */
327 }

426 /*
427  * Copy a %{} string.
428  */
429 static int rad_copy_variable(char *to, char const *from)
430 {
431  int length = 0;
432  int sublen;
433 
434  *(to++) = *(from++);
435  length++;
436 
437  while (*from) {
438  switch (*from) {
439  case '"':
440  case '\'':
441  sublen = rad_copy_string(to, from);
442  if (sublen < 0) return sublen;
443  from += sublen;
444  to += sublen;
445  length += sublen;
446  break;
447 
448  case '}': /* end of variable expansion */
449  *(to++) = *(from++);
450  *to = '\0';
451  length++;
452  return length; /* proper end of variable */
453 
454  case '\\':
455  *(to++) = *(from++);
456  *(to++) = *(from++);
457  length += 2;
458  break;
459 
460  case '%': /* start of variable expansion */
461  if (from[1] == '{') {
462  *(to++) = *(from++);
463  length++;
464 
465  sublen = rad_copy_variable(to, from);
466  if (sublen < 0) return sublen;
467  from += sublen;
468  to += sublen;
469  length += sublen;

471  } /* else FIXME: catch %%{ ?*/

473  FALL_THROUGH;
474  default:
475  *(to++) = *(from++);
476  length++;
477  break;
478  }
479  } /* loop over the input string */
480 
481  /*
482  * We ended the string before a trailing '}'
483  */
484 
485  return -1;

488 uint32_t rad_pps(uint32_t *past, uint32_t *present, time_t *then, struct timeval *now)
489 {
490  uint32_t pps;
491 
492  if (*then != now->tv_sec) {
493  *then = now->tv_sec;
494  *past = *present;
495  *present = 0;
496  }
497 
498  /*
499  * Bootstrap PPS by looking at a percentage of
500  * the previous PPS. This lets us take a moving
501  * count, without doing a moving average. If
502  * we're a fraction "f" (0..1) into the current
503  * second, we can get a good guess for PPS by
504  * doing:
505  *
506  * PPS = pps_now + pps_old * (1 - f)
507  *
508  * It's an instantaneous measurement, rather than
509  * a moving average. This will hopefully let it
510  * respond better to sudden spikes.
511  *
512  * Doing the calculations by thousands allows us
513  * to not overflow 2^32, AND to not underflow
514  * when we divide by USEC.
515  */
516  pps = USEC - now->tv_usec; /* useconds left in previous second */
517  pps /= 1000; /* scale to milliseconds */
518  pps *= *past; /* multiply by past count to get fraction */
519  pps /= 1000; /* scale to usec again */
520  pps += *present; /* add in current count */
521 
522  return pps;
523 }

526  *
527  * @param request Current request.
528  * @param cmd string to split.
529  * @param max_argc the maximum number of arguments to split into.
530  * @param argv Where to write the pointers into argv_buf.
531  * @param can_fail If false, stop processing if any of the xlat expansions fail.
532  * @param argv_buflen size of argv_buf.
533  * @param argv_buf temporary buffer we used to mangle/expand cmd.
534  * Pointers to offsets of this buffer will be written to argv.
535  * @return argc or -1 on failure.
536  */
537 
538 int rad_expand_xlat(request_t *request, char const *cmd,
539  int max_argc, char const *argv[], bool can_fail,
540  size_t argv_buflen, char *argv_buf)
541 {
542  char const *from;
543  char *to;
544  int argc = -1;
545  int i;
546  int left;
547 
548  if (strlen(cmd) > (argv_buflen - 1)) {
549  fr_strerror_printf("Expansion string is too long for output buffer");
550  return -1;
551  }
552 
553  /*
554  * Check for bad escapes.
555  */
556  if (cmd[strlen(cmd) - 1] == '\\') {
557  fr_strerror_printf("Expansion string ends with a trailing backslash - invalid escape sequence");
558  return -1;
559  }
560 
561  strlcpy(argv_buf, cmd, argv_buflen);
562 
563  /*
564  * Split the string into argv's BEFORE doing xlat_eval...
565  */
566  from = cmd;
567  to = argv_buf;
568  argc = 0;
569  while (*from) {
570  int length;
571 
572  fr_skip_whitespace(from);
573 
574  argv[argc] = to;
575  argc++;
576 
577  if (argc >= (max_argc - 1)) break;
578 
579  /*
580  * Copy the argv over to our buffer.
581  */
582  while (*from && (*from != ' ') && (*from != '\t')) {
583  if (to >= argv_buf + argv_buflen - 1) {
584  fr_strerror_printf("Expansion string is too long for output buffer");
585  return -1;
586  }
587 
588  switch (*from) {
589  case '"':
590  case '\'':
591  length = rad_copy_string_bare(to, from);
592  if (length < 0) {
593  fr_strerror_printf("Invalid quoted string in expansion");
594  return -1;
595  }
596  from += length+2;
597  to += length;
598  break;
599 
600  case '%':
601  if (from[1] == '{') {
602  *(to++) = *(from++);
603 
604  length = rad_copy_variable(to, from);
605  if (length < 0) {
606  fr_strerror_printf("Invalid variable in expansion");
607  return -1;
608  }
609  from += length;
610  to += length;
611  } else { /* FIXME: catch %%{ ? */
612  *(to++) = *(from++);
613  }
614  break;
615 
616  case '\\':
617  if (from[1] == ' ') from++;
618  FALL_THROUGH;
619 
620  default:
621  *(to++) = *(from++);
622  }
623  } /* end of string, or found a space */
624 
625  *(to++) = '\0'; /* terminate the string */
626  }
627 
628  /*
629  * We have to have SOMETHING, at least.
630  */
631  if (argc <= 0) {
632  fr_strerror_printf("Expansion string is empty");
633  return -1;
634  }
635 
636  /*
637  * Expand each string, as appropriate.
638  */
639  left = argv_buf + argv_buflen - to;
640  for (i = 0; i < argc; i++) {
641  int sublen;
642 
643  /*
644  * Don't touch argv's which won't be translated.
645  */
646  if (strchr(argv[i], '%') == NULL) continue;
647 
648  if (!request) continue;
649 
650  sublen = xlat_eval(to, left - 1, request, argv[i], NULL, NULL);
651  if (sublen <= 0) {
652  if (can_fail) {
653  /*
654  * Fail to be backwards compatible.
655  *
656  * It's yucky, but it won't break anything,
657  * and it won't cause security problems.
658  */
659  sublen = 0;
660  } else {
661  fr_strerror_printf("Failed expanding substring");
662  return -1;
663  }
664  }
665 
666  argv[i] = to;
667  to += sublen;
668  *(to++) = '\0';
669  left -= sublen;
670  left--;
671 
672  if (left <= 0) {
673  fr_strerror_printf("Ran out of space while expanding arguments");
674  return -1;
675  }
676  }
677  argv[argc] = NULL;
678 
679  return argc;
680 }
