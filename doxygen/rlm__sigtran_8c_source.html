209 /** Convert our sccp address config structure into sockaddr_sccp
210  *
211  * @param ctx to allocated address in.
212  * @param inst of rlm_sigtran.
213  * @param out Where to write the parsed data.
214  * @param conf to parse.
215  * @param cs specifying sccp address.
216  * @return
217  * - 0 on success.
218  * - -1 on failure.
219  */
220 static int sigtran_sccp_sockaddr_from_conf(TALLOC_CTX *ctx, rlm_sigtran_t *inst,
221  struct sockaddr_sccp *out,
222  sigtran_sccp_address_t *conf, CONF_SECTION *cs)
223 {
224  /*
225  * Fixme should be conf->gt_is_set
226  */
227  if (!conf->ssn_is_set && !conf->pc_is_set && !conf->gt.address) {
228  cf_log_err(cs, "At least one of 'pc', 'ssn', or 'gt', must be set");
229  return -1;
230  }
231 
232  if (conf->ssn_is_set) out->ssn = conf->ssn;
233  if (conf->pc_is_set) {
234  if (conf->pc > 16777215) {
235  cf_log_err(cs, "Invalid value \"%d\" for 'pc', must be between 0-"
236  STRINGIFY(16777215), conf->pc);
237  return -1;
238  }
239  out->use_poi = 1;
240 
241  memcpy(&out->poi, &conf->pc, sizeof(out->poi));
242  }
243 
244  /*
245  * Fixme should be conf->gt_is_set && conf->gt.address
246  * But we don't have subsection presence checks yet.
247  */
248  if (conf->gt_is_set || conf->gt.address) {
249  int gti_ind = SCCP_TITLE_IND_NONE;
250  size_t i;
251  size_t len = talloc_array_length(conf->gt.address) - 1;
252 
253  if (conf->gt.nai_is_set && (conf->gt.nai & 0x80)) {
254  cf_log_err(cs, "Global title 'nai' must be between 0-127");
255  return -1;
256  }
257 
258  if (conf->gt.tt_is_set) {
259  if ((conf->gt.np_is_set && !conf->gt.es_is_set) ||
260  (!conf->gt.np_is_set && conf->gt.np_is_set)) {
261  cf_log_err(cs, "Global title 'np' and 'es' must be "
262  "specified together");
263  return -1;
264  }
265 
266  if (conf->gt.np) {
267  cf_log_err(cs, "Global title 'np' must be between 0-15");
268  return -1;
269  }
270 
271  if (conf->gt.es > 0x0f) {
272  cf_log_err(cs, "Global title 'es' must be between 0-15");
273  return -1;
274  }
275 
276  if (conf->gt.np_is_set) {
277  gti_ind = conf->gt.nai_is_set ? SCCP_TITLE_IND_TRANS_NUM_ENC_NATURE :
278  SCCP_TITLE_IND_TRANS_NUM_ENC;
279  } else {
280  gti_ind = SCCP_TITLE_IND_TRANSLATION_ONLY;
281  }
282  } else if (conf->gt.nai_is_set) {
283  gti_ind = SCCP_TITLE_IND_NATURE_ONLY;
284  }
285 
286  for (i = 0; i < len; i++) {
287  if (!is_char_tbcd[(uint8_t)conf->gt.address[i]]) {
288  cf_log_err(cs, "Global title address contains invalid digit \"%c\". "
289  "Valid digits are [0-9#*a-c]", conf->gt.address[i]);
290  return -1;
291  }
292  }
293 
294  if (sigtran_sccp_global_title(ctx, &out->gti_data, gti_ind, conf->gt.address,
295  conf->gt.tt, conf->gt.np, conf->gt.es, conf->gt.nai) < 0) return -1;
296  out->gti_len = talloc_array_length(out->gti_data);
297  out->gti_ind = gti_ind;
298  out->national = 1;
299 
300  /*
301  * Print out the constructed global title blob.
302  */
303  DEBUG4("gt_ind: 0x%x", out->gti_ind);
304  DEBUG4("digits: 0x%pH (%i)", fr_box_octets(out->gti_data, out->gti_len), out->gti_len);
305  }
306  return 0;
307 }

338 static int mod_instantiate(void *instance, CONF_SECTION *conf)
339 {
340  rlm_sigtran_t *inst = instance;
341 
342  inst->name = cf_section_name2(conf);
343  if (!inst->name) inst->name = cf_section_name1(conf);
344 
345  /*
346  * Translate traffic mode string to integer
347  */
348  inst->conn_conf.m3ua_traffic_mode = fr_table_value_by_str(m3ua_traffic_mode_table,
349  inst->conn_conf.m3ua_traffic_mode_str, -1);
350  if (inst->conn_conf.m3ua_traffic_mode < 0) {
351  cf_log_err(conf, "Invalid 'm3ua_traffic_mode' value \"%s\", expected 'override', "
352  "'loadshare' or 'broadcast'", inst->conn_conf.m3ua_traffic_mode_str);
353  return -1;
354  }
355 
356 #define MTP3_PC_CHECK(_x) \
357  do { \
358  if (inst->conn_conf.mtp3_##_x > 16777215) { \
359  cf_log_err(conf, "Invalid value \"%d\" for '#_x', must be between 0-16777215", \
360  inst->conn_conf.mtp3_##_x); \
361  return -1; \
362  } \
363  __hack_##_x = inst->conn_conf.mtp3_##_x; \
364  } while (0)
365 
366  MTP3_PC_CHECK(dpc);
367  MTP3_PC_CHECK(opc);
368 
369  if (sigtran_sccp_sockaddr_from_conf(inst, inst, &inst->conn_conf.sccp_called_sockaddr,
370  &inst->conn_conf.sccp_called, conf) < 0) return -1;
371  if (sigtran_sccp_sockaddr_from_conf(inst, inst, &inst->conn_conf.sccp_calling_sockaddr,
372  &inst->conn_conf.sccp_calling, conf) < 0) return -1;
373 
374  /*
375  * Don't bother starting the sigtran thread if we're
376  * just checking the config.
377  */
378  if (check_config) return 0;
379 
380  /*
381  * If this is the first instance of rlm_sigtran
382  * We spawn a new thread to run all the libosmo-* I/O
383  * and events.
384  *
385  * We talk to the thread using the ctrl_pipe, with
386  * each thread registering its own pipe via the ctrl_pipe.
387  *
388  * This makes it really easy to collect and distribute
389  * requests/responses, whilst using libosmo in a
390  * threadsafe way.
391  */
392  if (sigtran_instances == 0) sigtran_event_start();
393  sigtran_instances++;
394 
395  /*
396  * Should bring the SCTP/M3UA/MTP3/SCCP link up.
397  */
398  if (sigtran_client_link_up(&inst->conn, &inst->conn_conf) < 0) return -1;
399 
400  return 0;
401 }

403 /**
404  * Cleanup internal state.
405  */
406 static int mod_detach(UNUSED void *instance)
407 {
408  rlm_sigtran_t *inst = instance;
409 
410  /*
411  * If we're just checking the config we didn't start the
412  * thread.
413  */
414  if (check_config) return 0;
415 
416  sigtran_client_link_down(&inst->conn);
417 
418  if ((--sigtran_instances) == 0) sigtran_event_exit();
419 
420  return 0;
421 }
