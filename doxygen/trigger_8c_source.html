74  if (!request_data_reference(request, &trigger_exec_main, REQUEST_INDEX_TRIGGER_NAME)) {
75  ERROR("trigger xlat may only be used in a trigger command");
76  return -1;
77  }
78 
79  head = request_data_reference(request, &trigger_exec_main, REQUEST_INDEX_TRIGGER_ARGS);
80 
81  /*
82  * No arguments available.
83  */
84  if (!head) return -1;
85

87  if (!da) {
88  ERROR("Unknown attribute \"%s\"", fmt);
89  return -1;
90  }
91 
92  vp = fr_pair_find_by_da(&head, da);
93  if (!vp) {
94  ERROR("Attribute \"%s\" is not valid for this trigger", fmt);
95  return -1;
96  }
97 
98  return fr_value_box_aprint(request, out, &vp->data, NULL);

112 /** Compares two last fired structures
113  *
114  * @param a first pointer to compare.
115  * @param b second pointer to compare.
116  * @return
117  * - -1 if a < b.
118  * - +1 if b > a.
119  * - 0 if both equal.
120  */
121 static int _trigger_last_fired_cmp(void const *a, void const *b)
122 {
123  trigger_last_fired_t const *lf_a = a, *lf_b = b;
124 
125  return (lf_a->ci < lf_b->ci) - (lf_a->ci > lf_b->ci);
126 }

128 /** Set the global trigger section trigger_exec will search in, and register xlats
129  *
130  * This function exists because triggers are used by the connection pool, which
131  * is used in the server library which may not have the mainconfig available.
132  * Additionally, utilities may want to set their own root config sections.
133  *
134  * We don't register the trigger xlat here, as we may inadvertently initialise
135  * the xlat code, which is annoying when this is called from a utility.
136  *
137  * @param cs to use as global trigger section.
138  * @return
139  * - 0 on success.
140  * - -1 on failure.
141  */
142 int trigger_exec_init(CONF_SECTION const *cs)
143 {
144  if (!cs) {
145  ERROR("%s - Pointer to main_config was NULL", __FUNCTION__);
146  return -1;
147  }
148 
149  trigger_exec_main = cs;
150  trigger_exec_subcs = cf_section_find(cs, "trigger", NULL);
151 
152  if (!trigger_exec_subcs) {
153  WARN("trigger { ... } subsection not found, triggers will be disabled");
154  return 0;
155  }
156 
157  MEM(trigger_last_fired_tree = rbtree_talloc_alloc(talloc_null_ctx(),
158  _trigger_last_fired_cmp, trigger_last_fired_t,
159  _trigger_last_fired_free, 0));
160 
161  trigger_mutex = talloc(talloc_null_ctx(), pthread_mutex_t);
162  pthread_mutex_init(trigger_mutex, 0);
163  talloc_set_destructor(trigger_mutex, _mutex_free);
164 
165  triggers_init = true;
166 
167  return 0;
168 }

195 static unlang_action_t trigger_process(rlm_rcode_t *p_result, module_ctx_t const *mctx, request_t *request)
196 {
197  fr_trigger_t *ctx = talloc_get_type_abort(mctx->instance, fr_trigger_t);
198  rlm_rcode_t rcode;
199 
200  if (!ctx->expanded) {
201  RDEBUG("Running trigger %s", ctx->name);
202 
203  /*
204  * Bootstrap these for simpliciy.
205  */
206  (void) fr_pair_list_copy(request->packet, &request->request_pairs, &ctx->vps);
207 
208  if (unlang_interpret_push_instruction(request, NULL,
209  RLM_MODULE_REJECT, UNLANG_TOP_FRAME) < 0) {
210  RETURN_MODULE_FAIL;
211  }
212  if (unlang_xlat_push(request, &ctx->box, request, ctx->xlat, true) < 0) {
213  RETURN_MODULE_FAIL;
214  }
215  ctx->expanded = true;
216 
217  /*
218  * Run the interpreter.
219  */
220  rcode = unlang_interpret(request);
221 
222  if (request->master_state == REQUEST_STOP_PROCESSING) {
223  RETURN_MODULE_HANDLED;
224  }
225 
226  if (rcode == RLM_MODULE_YIELD) {
227  *p_result = RLM_MODULE_YIELD;
228  return UNLANG_ACTION_YIELD;
229  }
230 
231  /*
232  * Always fall through, no matter what the return code is.
233  */
234  }
235 
236  if (!ctx->box) {
237  RERROR("Failed trigger %s - did not expand to anything", ctx->name);
238  RETURN_MODULE_FAIL;
239  }
240 
241  /*
242  * Execute the program without waiting for results.
243  */
244  if (fr_exec_nowait(request, ctx->box, NULL) < 0) {
245  RPERROR("Failed trigger %s", ctx->name);
246  RETURN_MODULE_FAIL;
247  }
248 
249  RETURN_MODULE_OK;
250 }

253 /** Execute a trigger - call an executable to process an event
254  *
255  * @note Calls to this function will be ignored if #trigger_exec_init has not been called.
256  *
257  * @param request The current request.
258  * @param cs to search for triggers in.
259  * If cs is not NULL, the portion after the last '.' in name is used for the trigger.
260  * If cs is NULL, the entire name is used to find the trigger in the global trigger
261  * section.
262  * @param name the path relative to the global trigger section ending in the trigger name
263  * e.g. module.ldap.pool.start.
264  * @param rate_limit whether to rate limit triggers.
265  * @param args to make available via the @verbatim %{trigger:<arg>} @endverbatim xlat.
266  * @return - 0 on success.
267  * - -1 on failure.
268  */
269 int trigger_exec(request_t *request, CONF_SECTION const *cs, char const *name, bool rate_limit, fr_pair_t *args)
270 {
271  CONF_SECTION const *subcs;
272 
273  CONF_ITEM *ci;
274  CONF_PAIR *cp;
275 
276  char const *attr;
277  char const *value;
278 
279  request_t *fake;
280  fr_trigger_t *ctx;
281  ssize_t slen;
282 
283  /*
284  * noop if trigger_exec_init was never called
285  */
286  if (!triggers_init || !trigger_worker_request_add) return 0;
287 
288  /*
289  * Use global "trigger" section if no local config is given.
290  */
291  if (!cs) {
292  cs = trigger_exec_main;
293  attr = name;
294  } else {
295  /*
296  * Try to use pair name, rather than reference.
297  */
298  attr = strrchr(name, '.');
299  if (attr) {
300  attr++;
301  } else {
302  attr = name;
303  }
304  }
305 
306  /*
307  * Find local "trigger" subsection. If it isn't found,
308  * try using the global "trigger" section, and reset the
309  * reference to the full path, rather than the sub-path.
310  */
311  subcs = cf_section_find(cs, "trigger", NULL);
312  if (!subcs && trigger_exec_main && (cs != trigger_exec_main)) {
313  subcs = trigger_exec_subcs;
314  attr = name;
315  }
316  if (!subcs) return -1;
317 
318  ci = cf_reference_item(subcs, trigger_exec_main, attr);
319  if (!ci) {
320  ROPTIONAL(RDEBUG2, DEBUG2, "No trigger configured for: %s", attr);
321  return -1;
322  }
323 
324  if (!cf_item_is_pair(ci)) {
325  ROPTIONAL(RERROR, ERROR, "Trigger is not a configuration variable: %s", attr);
326  return -1;
327  }
328 
329  cp = cf_item_to_pair(ci);
330  if (!cp) return -1;
331 
332  value = cf_pair_value(cp);
333  if (!value) {
334  ROPTIONAL(RERROR, ERROR, "Trigger has no value: %s", name);
335  return -1;
336  }
337 
338  /*
339  * Don't do any real work if we're checking the
340  * configuration. i.e. don't run "start" or "stop"
341  * triggers on "radiusd -XC".
342  */
343  if (check_config) return 0;
344 
345  /*
346  * Perform periodic rate_limiting.
347  */
348  if (rate_limit) {
349  trigger_last_fired_t find, *found;
350  time_t now = time(NULL);
351 
352  find.ci = ci;
353 
354  pthread_mutex_lock(trigger_mutex);
355 
356  found = rbtree_finddata(trigger_last_fired_tree, &find);
357  if (!found) {
358  MEM(found = talloc(NULL, trigger_last_fired_t));
359  found->ci = ci;
360  found->last_fired = 0;
361 
362  rbtree_insert(trigger_last_fired_tree, found);
363  }
364 
365  pthread_mutex_unlock(trigger_mutex);
366 
367  /*
368  * Send the rate_limited traps at most once per second.
369  */
370  if (found->last_fired == now) return -1;
371  found->last_fired = now;
372  }
373 
374  /*
375  * radius_exec_program always needs a request.
376  */
377  fake = request_alloc(NULL);
378  memcpy(&fake->server_cs, &subcs, sizeof(subcs)); /* completely wrong, but we need to use _something_ */
379 
380  /*
381  * Add the args to the request data, so they can be picked up by the
382  * trigger_xlat function.
383  */
384  if (args && (request_data_add(fake, &trigger_exec_main, REQUEST_INDEX_TRIGGER_ARGS, args,
385  false, false, false) < 0)) {
386  talloc_free(fake);
387  return -1;
388  }
389 
390  {
391  void *name_tmp;
392 
393  memcpy(&name_tmp, &name, sizeof(name_tmp));
394 
395  if (request_data_add(fake, &trigger_exec_main, REQUEST_INDEX_TRIGGER_NAME,
396  name_tmp, false, false, false) < 0) {
397  talloc_free(fake);
398  return -1;
399  }
400  }
401 
402  MEM(ctx = talloc_zero(fake, fr_trigger_t));
403  ctx->name = talloc_strdup(ctx, value);
404 
405  if (request) {
406  if (request->request_pairs) {
407  (void) fr_pair_list_copy(ctx, &ctx->vps, &request->request_pairs);
408  }
409 
410  fake->log = request->log;
411  } else {
412  fake->log.dst = talloc_zero(fake, log_dst_t);
413  fake->log.dst->func = vlog_request;
414  fake->log.dst->uctx = &default_log;
415  fake->log.lvl = fr_debug_lvl;
416  }
417 
418  slen = xlat_tokenize_argv(ctx, &ctx->xlat, NULL,
419  &FR_SBUFF_IN(ctx->name, talloc_array_length(ctx->name) - 1), NULL, NULL);
420  if (slen <= 0) {
421  char *spaces, *text;
422 
423  fr_canonicalize_error(ctx, &spaces, &text, slen, fr_strerror());
424 
425  cf_log_err(cp, "Syntax error");
426  cf_log_err(cp, "%s", ctx->name);
427  cf_log_err(cp, "%s^ %s", spaces, text);
428 
429  talloc_free(fake);
430  talloc_free(spaces);
431  talloc_free(text);
432  return -1;
433  }
434 
435  /*
436  * Run the trigger asynchronously.
437  */
438  if (trigger_worker_request_add(fake, trigger_process, ctx) < 0) {
439  talloc_free(fake);
440  return -1;
441  }
442 
443  /*
444  * Otherwise the worker cleans up the fake request.
445  */
446  return 0;
447 }
