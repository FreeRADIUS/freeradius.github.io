31 static unlang_action_t unlang_if(rlm_rcode_t *p_result, request_t *request)
32 {
33  int condition;
34  unlang_stack_t *stack = request->stack;
35  unlang_stack_frame_t *frame = &stack->frame[stack->depth];
36  unlang_t *instruction = frame->instruction;
37 
38  unlang_group_t *g;
39  unlang_cond_t *gext;
40 
41  g = unlang_generic_to_group(instruction);
42  gext = unlang_group_to_cond(g);
43  fr_assert(gext->cond != NULL);
44 
45  condition = cond_eval(request, *p_result, 0, gext->cond);
46  if (condition < 0) {
47  switch (condition) {
48  case -2:
49  REDEBUG("Condition evaluation failed because a referenced attribute "
50  "was not found in the request");
51  break;
52  default:
53  case -1:
54  REDEBUG("Condition evaluation failed because the value of an operand "
55  "could not be determined");
56  break;
57  }
58  condition = 0;
59  }
60 
61  /*
62  * Didn't pass. Remember that.
63  */
64  if (!condition) {
65  RDEBUG2("...");
66  return UNLANG_ACTION_EXECUTE_NEXT;
67  }
68 
69  /*
70  * Tell the main interpreter to skip over the else /
71  * elsif blocks, as this "if" condition was taken.
72  */
73  while (frame->next &&
74  ((frame->next->type == UNLANG_TYPE_ELSE) ||
75  (frame->next->type == UNLANG_TYPE_ELSIF))) {
76  frame->next = frame->next->next;
77  }
78 
79  /*
80  * We took the "if". Go recurse into its' children.
81  */
82  return unlang_group(p_result, request);
83 }
