244 /** Compare module instances by parent and name
245  *
246  * The reason why we need parent, is because we could have submodules with names
247  * that conflict with their parent.
248  */
249 static int module_instance_name_cmp(void const *one, void const *two)
250 {
251  module_instance_t const *a = one;
252  module_instance_t const *b = two;
253  dl_module_inst_t const *dl_inst;
254  int a_depth = 0, b_depth = 0;
255  int ret;
256 
257  /*
258  * Sort by depth, so for tree walking we start
259  * at the shallowest node, and finish with
260  * the deepest child.
261  */
262  for (dl_inst = a->dl_inst; dl_inst; dl_inst = dl_inst->parent) a_depth++;
263  for (dl_inst = b->dl_inst; dl_inst; dl_inst = dl_inst->parent) b_depth++;
264 
265  ret = (a_depth > b_depth) - (a_depth < b_depth);
266  if (ret != 0) return ret;
267 
268  /*
269  * This happens, as dl_inst is is used in
270  * as the loop condition above.
271  */
272 #ifdef __clang_analyzer__
273  if (!fr_cond_assert(a->dl_inst)) return +1;
274  if (!fr_cond_assert(b->dl_inst)) return -1;
275 #endif
276 
277  ret = (a->dl_inst->parent > b->dl_inst->parent) - (a->dl_inst->parent < b->dl_inst->parent);
278  if (ret != 0) return ret;
279 
280  return strcmp(a->name, b->name);
281 }

332 /** Resolve polymorphic item's from a module's #CONF_SECTION to a subsection in another module
333  *
334  * This allows certain module sections to reference module sections in other instances
335  * of the same module and share #CONF_DATA associated with them.
336  *
337  * @verbatim
338  example {
339  data {
340  ...
341  }
342  }
343 
344  example inst {
345  data = example
346  }
347  * @endverbatim
348  *
349  * @param[out] out where to write the pointer to a module's config section. May be NULL on success,
350  * indicating the config item was not found within the module #CONF_SECTION
351  * or the chain of module references was followed and the module at the end of the chain
352  * did not a subsection.
353  * @param[in] module #CONF_SECTION.
354  * @param[in] name of the polymorphic sub-section.
355  * @return
356  * - 0 on success with referenced section.
357  * - 1 on success with local section.
358  * - -1 on failure.
359  */
360 int module_sibling_section_find(CONF_SECTION **out, CONF_SECTION *module, char const *name)
361 {
362  CONF_PAIR *cp;
363  CONF_SECTION *cs;
364  CONF_DATA const *cd;
365 
366 
367  module_instance_t *mi;
368  char const *inst_name;
369 
370 #define FIND_SIBLING_CF_KEY "find_sibling"
371 
372  *out = NULL;
373 
374  /*
375  * Is a real section (not referencing sibling module).
376  */
377  cs = cf_section_find(module, name, NULL);
378  if (cs) {
379  *out = cs;
380 
381  return 0;
382  }
383 
384  /*
385  * Item omitted completely from module config.
386  */
387  cp = cf_pair_find(module, name);
388  if (!cp) return 0;
389 
390  if (cf_data_find(module, CONF_SECTION, FIND_SIBLING_CF_KEY)) {
391  cf_log_err(cp, "Module reference loop found");
392 
393  return -1;
394  }
395  cd = cf_data_add(module, module, FIND_SIBLING_CF_KEY, false);
396 
397  /*
398  * Item found, resolve it to a module instance.
399  * This triggers module loading, so we don't have
400  * instantiation order issues.
401  */
402  inst_name = cf_pair_value(cp);
403  mi = module_by_name(NULL, inst_name);
404  if (!mi) {
405  cf_log_err(cp, "Unknown module instance \"%s\"", inst_name);
406 
407  return -1;
408  }
409 
410  if (!mi->instantiated) {
411  CONF_SECTION *parent = module;
412 
413  /*
414  * Find the root of the config...
415  */
416  do {
417  CONF_SECTION *tmp;
418 
419  tmp = cf_item_to_section(cf_parent(parent));
420  if (!tmp) break;
421 
422  parent = tmp;
423  } while (true);
424 
425  _module_instantiate(module_by_name(NULL, inst_name), NULL);
426  }
427 
428  /*
429  * Remove the config data we added for loop
430  * detection.
431  */
432  cf_data_remove(module, cd);
433 
434  /*
435  * Check the module instances are of the same type.
436  */
437  if (strcmp(cf_section_name1(mi->dl_inst->conf), cf_section_name1(module)) != 0) {
438  cf_log_err(cp, "Referenced module is a rlm_%s instance, must be a rlm_%s instance",
439  cf_section_name1(mi->dl_inst->conf), cf_section_name1(module));
440 
441  return -1;
442  }
443 
444  *out = cf_section_find(mi->dl_inst->conf, name, NULL);
445 
446  return 1;
447 }

449 /** Initialise a module specific connection pool
450  *
451  * @see fr_pool_init
452  *
453  * @param[in] module section.
454  * @param[in] opaque data pointer to pass to callbacks.
455  * @param[in] c Callback to create new connections.
456  * @param[in] a Callback to check the status of connections.
457  * @param[in] log_prefix override, if NULL will be set automatically from the module CONF_SECTION.
458  * @param[in] trigger_prefix if NULL will be set automatically from the module CONF_SECTION.
459  * @param[in] trigger_args to make available in any triggers executed by the connection pool.
460  * @return
461  * - New connection pool.
462  * - NULL on error.
463  */
464 fr_pool_t *module_connection_pool_init(CONF_SECTION *module,
465  void *opaque,
466  fr_pool_connection_create_t c,
467  fr_pool_connection_alive_t a,
468  char const *log_prefix,
469  char const *trigger_prefix,
470  fr_pair_t *trigger_args)
471 {
472  CONF_SECTION *cs, *mycs;
473  char log_prefix_buff[128];
474  char trigger_prefix_buff[128];
475 
476  fr_pool_t *pool;
477  char const *cs_name1, *cs_name2;
478 
479  int ret;
480 
481 #define parent_name(_x) cf_section_name(cf_item_to_section(cf_parent(_x)))
482 
483  cs_name1 = cf_section_name1(module);
484  cs_name2 = cf_section_name2(module);
485  if (!cs_name2) cs_name2 = cs_name1;
486 
487  if (!trigger_prefix) {
488  snprintf(trigger_prefix_buff, sizeof(trigger_prefix_buff), "modules.%s.pool", cs_name1);
489  trigger_prefix = trigger_prefix_buff;
490  }
491 
492  if (!log_prefix) {
493  snprintf(log_prefix_buff, sizeof(log_prefix_buff), "rlm_%s (%s)", cs_name1, cs_name2);
494  log_prefix = log_prefix_buff;
495  }
496 
497  /*
498  * Get sibling's pool config section
499  */
500  ret = module_sibling_section_find(&cs, module, "pool");
501  switch (ret) {
502  case -1:
503  return NULL;
504 
505  case 1:
506  DEBUG4("%s: Using pool section from \"%s\"", log_prefix, parent_name(cs));
507  break;
508 
509  case 0:
510  DEBUG4("%s: Using local pool section", log_prefix);
511  break;
512  }
513 
514  /*
515  * Get our pool config section
516  */
517  mycs = cf_section_find(module, "pool", NULL);
518  if (!mycs) {
519  DEBUG4("%s: Adding pool section to config item \"%s\" to store pool references", log_prefix,
520  cf_section_name(module));
521 
522  mycs = cf_section_alloc(module, module, "pool", NULL);
523  }
524 
525  /*
526  * Sibling didn't have a pool config section
527  * Use our own local pool.
528  */
529  if (!cs) {
530  DEBUG4("%s: \"%s.pool\" section not found, using \"%s.pool\"", log_prefix,
531  parent_name(cs), parent_name(mycs));
532  cs = mycs;
533  }
534 
535  /*
536  * If fr_pool_init has already been called
537  * for this config section, reuse the previous instance.
538  *
539  * This allows modules to pass in the config sections
540  * they would like to use the connection pool from.
541  */
542  pool = cf_data_value(cf_data_find(cs, fr_pool_t, NULL));
543  if (!pool) {
544  DEBUG4("%s: No pool reference found for config item \"%s.pool\"", log_prefix, parent_name(cs));
545  pool = fr_pool_init(cs, cs, opaque, c, a, log_prefix);
546  if (!pool) return NULL;
547 
548  fr_pool_enable_triggers(pool, trigger_prefix, trigger_args);
549 
550  if (fr_pool_start(pool) < 0) {
551  ERROR("%s: Starting initial connections failed", log_prefix);
552  return NULL;
553  }
554 
555  DEBUG4("%s: Adding pool reference %p to config item \"%s.pool\"", log_prefix, pool, parent_name(cs));
556  cf_data_add(cs, pool, NULL, false);
557  return pool;
558  }
559  fr_pool_ref(pool);
560 
561  DEBUG4("%s: Found pool reference %p in config item \"%s.pool\"", log_prefix, pool, parent_name(cs));
562 
563  /*
564  * We're reusing pool data add it to our local config
565  * section. This allows other modules to transitively
566  * re-use a pool through this module.
567  */
568  if (mycs != cs) {
569  DEBUG4("%s: Copying pool reference %p from config item \"%s.pool\" to config item \"%s.pool\"",
570  log_prefix, pool, parent_name(cs), parent_name(mycs));
571  cf_data_add(mycs, pool, NULL, false);
572  }
573 
574  return pool;
575 }

698  * @param[in] name The name of the module we're attempting to find, possibly concatenated with the method
699  * @return
700  * - The module instance on success.
701  * - NULL on not found
702  *
703  * If the module exists but the method doesn't exist, then `method` is set to NULL.
704  */
705 module_instance_t *module_by_name_and_method(module_method_t *method, rlm_components_t *component,
706  char const **name1, char const **name2,
707  char const *name)
708 {
709  char *p, *q, *inst_name;
710  size_t len;
711  int j;
712  rlm_components_t i;
713  module_instance_t *mi;
714  module_method_names_t const *methods;
715  char const *method_name1, *method_name2;
716 
717  if (method) *method = NULL;
718 
719  method_name1 = method_name2 = NULL;
720  if (name1) {
721  method_name1 = *name1;
722  *name1 = NULL;
723  }
724  if (name2) {
725  method_name2 = *name2;
726  *name2 = NULL;
727  }
728 
729  /*
730  * Module names are allowed to contain '.'
731  * so we search for the bare module name first.
732  */
733  mi = module_by_name(NULL, name);
734  if (mi) {
735  virtual_server_method_t const *allowed_list;
736 
737  if (!method) return mi;
738 
739  /*
740  * We're not searching for a named method, OR the
741  * module has no named methods. Try to return a
742  * method based on the component.
743  */
744  if (!method_name1 || !mi->module->method_names) goto return_component;
745 
746  /*
747  * Walk through the module, finding a matching
748  * method.
749  */
750  for (j = 0; mi->module->method_names[j].name1 != NULL; j++) {
751  methods = &mi->module->method_names[j];
752 
753  /*
754  * Wildcard match name1, we're
755  * done.
756  */
757  if (methods->name1 == CF_IDENT_ANY) {
758  found:
759  *method = methods->method;
760  if (name1) *name1 = method_name1;
761  if (name2) *name2 = method_name2;
762  return mi;
763  }
764 
765  /*
766  * If name1 doesn't match, skip it.
767  */
768  if (strcmp(methods->name1, method_name1) != 0) continue;
769 
770  /*
771  * The module can declare a
772  * wildcard for name2, in which
773  * case it's a match.
774  */
775  if (methods->name2 == CF_IDENT_ANY) goto found;
776 
777  /*
778  * No name2 is also a match to no name2.
779  */
780  if (!methods->name2 && !method_name2) goto found;
781 
782  /*
783  * Don't do strcmp on NULLs
784  */
785  if (!methods->name2 || !method_name2) continue;
786 
787  if (strcmp(methods->name2, method_name2) == 0) goto found;
788  }
789 
790  /*
791  * No match for "recv Access-Request", or
792  * whatever else the section is. Let's see if
793  * the section has a list of allowed methods.
794  */
795  allowed_list = virtual_server_section_methods(method_name1, method_name2);
796  if (!allowed_list) goto return_component;
797 
798  /*
799  * Walk over allowed methods for this section,
800  * (implicitly ordered by priority), and see if
801  * the allowed method matches any of the module
802  * methods. This process lets us reference a
803  * module as "foo" in the configuration. If the
804  * module exports a "recv bar" method, and the
805  * virtual server has a "recv bar" processing
806  * section, then they shoul match.
807  *
808  * Unfortunately, this process is O(N*M).
809  * Luckily, we only do it if all else fails, so
810  * it's mostly OK.
811  *
812  * Note that the "allowed" list CANNOT include
813  * CF_IDENT_ANY. Only the module can do that.
814  * If the "allowed" list exported CF_IDENT_ANY,
815  * then any module method would match, which is
816  * bad.
817  */
818  for (j = 0; allowed_list[j].name != NULL; j++) {
819  int k;
820  virtual_server_method_t const *allowed = &allowed_list[j];
821 
822  for (k = 0; mi->module->method_names[k].name1 != NULL; k++) {
823  methods = &mi->module->method_names[k];
824 
825  fr_assert(methods->name1 != CF_IDENT_ANY); /* should have been caught above */
826 
827  if (strcmp(methods->name1, allowed->name) != 0) continue;
828 
829  /*
830  * The module matches "recv *",
831  * call this method.
832  */
833  if (methods->name2 == CF_IDENT_ANY) {
834  found_allowed:
835  *method = methods->method;
836  return mi;
837  }
838 
839  /*
840  * No name2 is also a match to no name2.
841  */
842  if (!methods->name2 && !allowed->name2) goto found_allowed;
843 
844  /*
845  * Don't do strcmp on NULLs
846  */
847  if (!methods->name2 || !allowed->name2) continue;
848 
849  if (strcmp(methods->name2, allowed->name2) == 0) goto found_allowed;
850  }
851  }
852 
853  return_component:
854  /*
855  * No matching method. Just return a method
856  * based on the component.
857  */
858  if (component && mi->module->methods[*component]) {
859  *method = mi->module->methods[*component];
860  }
861 
862  /*
863  * Didn't find a matching method. Just return
864  * the module.
865  */
866  return mi;
867  }
868 
869  /*
870  * Find out if the instance name contains
871  * a method, if it doesn't, then the module
872  * doesn't exist.
873  */
874  p = strchr(name, '.');
875  if (!p) return NULL;
876 
877  /*
878  * The module name may have a '.' in it, AND it may have
879  * a method <sigh> So we try to find out which is which.
880  */
881  inst_name = talloc_strdup(NULL, name);
882  p = inst_name + (p - name);
883 
884  /*
885  * Loop over the '.' portions, gradually looking up a
886  * longer string, in order to find the full module name.
887  */
888  do {
889  *p = '\0';
890 
891  mi = module_by_name(NULL, inst_name);
892  if (mi) break;
893 
894  /*
895  * Find the next '.'
896  */
897  *p = '.';
898  p = strchr(p + 1, '.');
899  } while (p);
900 
901  /*
902  * No such module, we're done.
903  */
904  if (!mi) {
905  talloc_free(inst_name);
906  return NULL;
907  }
908 
909  /*
910  * We have a module, but the caller doesn't care about
911  * method or names, so just return the module.
912  */
913  if (!method || !method_name1 || !method_name2) {
914  talloc_free(inst_name);
915  return mi;
916  }
917 
918  /*
919  * We MAY have two names.
920  */
921  p++;
922  q = strchr(p, '.');
923 
924  /*
925  * If there's only one component, look for it in the
926  * "authorize", etc. list first.
927  */
928  if (!q) {
929  for (i = MOD_AUTHENTICATE; i < MOD_COUNT; i++) {
930  if (strcmp(section_type_value[i], p) != 0) continue;
931 
932  /*
933  * Tell the caller which component was
934  * referenced, and set the method to the found
935  * function.
936  */
937  if (component) {
938  *component = i;
939  if (method) *method = mi->module->methods[*component];
940  }
941 
942  /*
943  * The string matched. Return it. Also set the
944  * names so that the caller gets told the method
945  * name being used.
946  */
947  *name1 = name + (p - inst_name);
948  *name2 = NULL;
949  talloc_free(inst_name);
950  return mi;
951  }
952  }
953 
954  /*
955  * We've found the module, but it has no named methods.
956  */
957  if (!mi->module->method_names) {
958  *name1 = name + (p - inst_name);
959  *name2 = NULL;
960  talloc_free(inst_name);
961  return mi;
962  }
963 
964  /*
965  * We have "module.METHOD", but METHOD doesn't match
966  * "authorize", "authenticate", etc. Let's see if it
967  * matches anything else.
968  */
969  if (!q) {
970  for (j = 0; mi->module->method_names[j].name1 != NULL; j++) {
971  methods = &mi->module->method_names[j];
972  /*
973  * If we do not have the second $method, then ignore it!
974  */
975  if (methods->name2 != CF_IDENT_ANY) continue;
976 
977  /*
978  * Wildcard match name1, we're
979  * done.
980  */
981  if (!methods->name1 || (methods->name1 == CF_IDENT_ANY)) goto found_name1;
982 
983  /*
984  * If name1 doesn't match, skip it.
985  */
986  if (strcmp(methods->name1, p) != 0) continue;
987 
988  found_name1:
989  /*
990  * We've matched "*", or "name1" or
991  * "name1 *". Return that.
992  */
993  *name1 = p;
994  *name2 = NULL;
995  *method = methods->method;
996  break;
997  }
998 
999  /*
1000  * Return the found module.
1001  */
1002  talloc_free(inst_name);
1003  return mi;
1004  }
1005 
1006  /*
1007  * We CANNOT have '.' in method names.
1008  */
1009  if (strchr(q + 1, '.') != 0) {
1010  talloc_free(inst_name);
1011  return mi;
1012  }
1013 
1014  len = q - p;
1015 
1016  /*
1017  * Trim the '.'.
1018  */
1019  if (*q == '.' && *(q + 1)) q++;
1020 
1021  /*
1022  * We have "module.METHOD1.METHOD2".
1023  *
1024  * Loop over the method names, seeing if we have a match.
1025  */
1026  for (j = 0; mi->module->method_names[j].name1 != NULL; j++) {
1027  methods = &mi->module->method_names[j];
1028 
1029  /*
1030  * If name1 doesn't match, skip it.
1031  */
1032  if (strncmp(methods->name1, p, len) != 0) continue;
1033 
1034  /*
1035  * It may have been a partial match, like "rec",
1036  * instead of "recv". In which case check if it
1037  * was a FULL match.
1038  */
1039  if (strlen(methods->name1) != len) continue;
1040 
1041  /*
1042  * The module can declare a
1043  * wildcard for name2, in which
1044  * case it's a match.
1045  */
1046  if (!methods->name2 || (methods->name2 == CF_IDENT_ANY)) goto found_name2;
1047 
1048  /*
1049  * Don't do strcmp on NULLs
1050  */
1051  if (!methods->name2) continue;
1052 
1053  if (strcmp(methods->name2, q) != 0) continue;
1054 
1055  found_name2:
1056  /*
1057  * Update name1/name2 with the methods
1058  * that were found.
1059  */
1060  *name1 = methods->name1;
1061  *name2 = name + (q - inst_name);
1062  *method = methods->method;
1063  break;
1064  }
1065 
1066  *name1 = name + (p - inst_name);
1067  *name2 = NULL;
1068 
1069  talloc_free(inst_name);
1070  return mi;
1071 }
